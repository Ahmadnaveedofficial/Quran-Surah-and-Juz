17 پارہ


سورة الأنبياء
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
ٱقْتَرَبَ لِلنَّاسِ حِسَابُهُمْ وَهُمْ فِي غَفْلَةٍۢ مُّعْرِضُونَ ﴿1﴾
لوگوں کے حساب کا وقت قریب آ گیا ہے، اور وہ غفلت میں منہ موڑے ہوئے ہیں۔
مَا يَأْتِيهِم مِّن ذِكْرٖ مِّن رَّبِّهِم مُّحْدَثٍ إِلَّا ٱسْتَمَعُوهُ وَهُمْ يَلْعَبُونَ ﴿2﴾
ان کے پاس ان کے رب کی طرف سے جو بھی نئی نصیحت آتی ہے، وہ اسے کھیلتے ہوئے ہی سنتے ہیں۔
لَاهِيَةً قُلُوبُهُمْۗ وَأَسَرُّوا۟ ٱلنَّجْوَى ٱلَّذِينَ ظَلَمُوا۟ هَلْ هَـٰذَآ إِلَّا بَشَرٞ مِّثْلُكُمۖ أَفَتَأْتُونَ ٱلسِّحْرَ وَأَنتُمْ تُبْصِرُونَ ﴿3﴾
ان کے دل کھیل میں لگے ہوئے ہیں، اور ظالم لوگ آپس میں چپکے چپکے باتیں کرتے ہیں کہ یہ تو تمہاری طرح کا ایک انسان ہے، کیا تم کھلی آنکھوں سے جادو کی طرف جا رہے ہو؟
قَالَ رَبِّي يَعْلَمُ ٱلْقَوْلَ فِي ٱلسَّمَآءِ وَٱلْأَرْضِۖ وَهُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ ﴿4﴾
(نبی نے) کہا: میرا رب آسمان اور زمین کی بات کو جانتا ہے، اور وہ سننے والا، جاننے والا ہے۔
بَلْ قَالُوٓا۟ أَضْغَـٰثُ أَحْلَـٰمِۭ بَلِ ٱفْتَرَىٰهُ بَلْ هُوَ شَاعِرٞ فَلْيَأْتِنَا بِـَٔايَةٖ كَمَآ أُرْسِلَ ٱلْأَوَّلُونَ ﴿5﴾
بلکہ وہ کہتے ہیں: یہ تو پریشان خواب ہیں، بلکہ اس نے خود گھڑ لیا ہے، بلکہ وہ شاعر ہے، تو وہ ہمارے پاس کوئی نشانی لائے جیسے پہلے لوگ بھیجے گئے تھے۔
مَآ ءَامَنَتْ قَبْلَهُم مِّن قَرْيَةٍ أَهْلَكْنَـٰهَآۖ أَفَهُمْ يُؤْمِنُونَ ﴿6﴾
ان سے پہلے ہم نے جن بستیوں کو ہلاک کیا، ان میں سے کوئی ایمان نہ لائی، تو کیا یہ ایمان لائیں گے؟
وَمَآ أَرْسَلْنَا قَبْلَكَ إِلَّا رِجَالٗا نُّوحِيٓ إِلَيْهِمْۖ فَسْـَٔلُوٓا۟ أَهْلَ ٱلذِّكْرِ إِن كُنتُمْ لَا تَعْلَمُونَ ﴿7﴾
اور ہم نے آپ سے پہلے صرف مردوں کو ہی بھیجا جن کی طرف ہم وحی کرتے تھے، پس اہلِ ذکر سے پوچھ لو اگر تم نہیں جانتے۔
وَمَا جَعَلْنَـٰهُمْ جَسَدٗا لَّا يَأْكُلُونَ ٱلطَّعَامَ وَمَا كَانُوا۟ خَالِدِينَ ﴿8﴾
اور ہم نے انہیں ایسے جسم نہیں بنایا جو کھانا نہ کھاتے ہوں، اور نہ وہ ہمیشہ زندہ رہنے والے تھے۔
ثُمَّ صَدَقْنَـٰهُمُ ٱلْوَعْدَ فَأَنجَيْنَـٰهُمْ وَمَن نَّشَآءُ وَأَهْلَكْنَا ٱلْمُسْرِفِينَ ﴿9﴾
پھر ہم نے ان سے وعدہ سچ کر دکھایا، اور انہیں اور جسے ہم چاہیں بچا لیا، اور حد سے بڑھنے والوں کو ہلاک کر دیا۔
لَقَدْ أَنزَلْنَآ إِلَيْكُمْ كِتَـٰبٗا فِيهِ ذِكْرُكُمۖ أَفَلَا تَعْقِلُونَ ﴿10﴾
بے شک ہم نے تمہاری طرف ایک کتاب نازل کی ہے جس میں تمہارا ذکر ہے، کیا تم عقل نہیں رکھتے؟
وَكَمْ قَصَمْنَا مِن قَرْيَةٍ كَانَتْ ظَالِمَةً وَأَنشَأْنَا بَعْدَهَا قَوْمًا آخَرِينَ ﴿11﴾
اور ہم نے کتنی ہی بستیاں جو ظالم تھیں تباہ کر دیں، اور ان کے بعد دوسرے لوگوں کو پیدا کیا۔
فَلَمَّآ أَحَسُّوا۟ بَأْسَنَآ إِذَا هُم مِّنْهَا يَرْكُضُونَ ﴿12﴾
پھر جب انہوں نے ہمارا عذاب محسوس کیا تو وہ وہاں سے بھاگنے لگے۔
لَا تَرْكُضُوا۟ وَٱرْجِعُوٓا۟ إِلَىٰ مَآ أُتْرِفْتُمْ فِيهِ وَمَسَـٰكِنِكُمْ لَعَلَّكُمْ تُسْـَٔلُونَ ﴿13﴾
(کہا گیا) مت بھاگو، اور لوٹ آؤ اس عیش و عشرت کی طرف جس میں تم تھے، اور اپنے گھروں کی طرف، شاید تم سے سوال کیا جائے۔
قَالُوا۟ يَـٰوَيْلَنَآ إِنَّا كُنَّا ظَـٰلِمِينَ ﴿14﴾
انہوں نے کہا: ہائے ہماری بدبختی! بے شک ہم ہی ظالم تھے۔
فَمَا زَالَتْ تِلْكَ دَعْوَىٰهُمْ حَتَّىٰ جَعَلْنَـٰهُمْ حَصِيدًا خَـٰمِدِينَ ﴿15﴾
اور ان کی یہی فریاد رہی یہاں تک کہ ہم نے انہیں کٹی ہوئی کھیتی کی طرح بے جان کر دیا۔
وَمَا خَلَقْنَا ٱلسَّمَآءَ وَٱلْأَرْضَ وَمَا بَيْنَهُمَا لَـٰعِبِينَ ﴿16﴾
اور ہم نے آسمان اور زمین اور جو کچھ ان کے درمیان ہے کھیل کے طور پر نہیں بنایا۔
لَّوْ أَرَدْنَآ أَن نَّتَّخِذَ لَهْوٗا لَّٱتَّخَذْنَـٰهُ مِن لَّدُنَّآ إِن كُنَّا فَـٰعِلِينَ ﴿17﴾
اگر ہم کھیل ہی بنانا چاہتے تو اسے اپنے پاس سے ہی بنا لیتے، اگر ہم ایسا کرنے والے ہوتے۔
بَلْ نَقْذِفُ بِٱلْحَقِّ عَلَى ٱلْبَـٰطِلِ فَيَدْمَغُهُ فَإِذَا هُوَ زَاهِقٞۚ وَلَكُمُ ٱلْوَيْلُ مِمَّا تَصِفُونَ ﴿18﴾
بلکہ ہم حق کو باطل پر پھینکتے ہیں تو وہ اسے چیر دیتا ہے، پھر وہ مٹ جاتا ہے، اور تمہارے لیے افسوس ہے ان باتوں پر جو تم بیان کرتے ہو۔
وَلَهُۥ مَن فِي ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۖ وَمَنْ عِندَهُۥ لَا يَسْتَكْبِرُونَ عَنْ عِبَادَتِهِۦ وَلَا يَسْتَحْسِرُونَ ﴿19﴾
اور اسی کے لیے ہے جو آسمانوں اور زمین میں ہیں، اور جو اس کے پاس ہیں وہ اس کی عبادت سے تکبر نہیں کرتے اور نہ تھکتے ہیں۔
يُسَبِّحُونَ ٱلَّيْلَ وَٱلنَّهَارَ لَا يَفْتُرُونَ ﴿20﴾
وہ رات دن اس کی تسبیح کرتے ہیں اور تھمتے نہیں۔
أَمِ ٱتَّخَذُوا۟ ءَالِهَةٗ مِّنَ ٱلْأَرْضِ هُمْ يُنشِرُونَ ﴿21﴾
کیا انہوں نے زمین سے ایسے معبود بنا لیے ہیں جو مردوں کو زندہ کرتے ہیں؟
لَوْ كَانَ فِيهِمَآ ءَالِهَةٌ إِلَّا ٱللَّهُ لَفَسَدَتَاۚ فَسُبْحَـٰنَ ٱللَّهِ رَبِّ ٱلْعَرْشِ عَمَّا يَصِفُونَ ﴿22﴾
اگر آسمان و زمین میں اللہ کے سوا اور معبود ہوتے تو دونوں بگڑ جاتے، پس اللہ عرش کے رب کی پاکی ہے ان باتوں سے جو وہ بیان کرتے ہیں۔
لَا يُسْـَٔلُ عَمَّا يَفْعَلُ وَهُمْ يُسْـَٔلُونَ ﴿23﴾
وہ نہیں پوچھا جاتا جو وہ کرتا ہے، اور وہ پوچھے جائیں گے۔
أَمِ ٱتَّخَذُوا۟ مِن دُونِهِۦٓ ءَالِهَةٗۚ قُلْ هَاتُوا۟ بُرْهَـٰنَكُمْۖ هَـٰذَا ذِكْرُ مَن مَّعِىَ وَذِكْرُ مَن قَبْلِيۚ بَلْ أَكْثَرُهُمْ لَا يَعْلَمُونَ ٱلْحَقَّ فَهُم مُّعْرِضُونَ ﴿24﴾
کیا انہوں نے اس کے سوا معبود بنا لیے ہیں؟ کہہ دو: اپنی دلیل لاؤ، یہ میرے ساتھ والوں کا ذکر ہے اور مجھ سے پہلے والوں کا بھی، بلکہ ان میں سے اکثر حق کو نہیں جانتے، پس وہ منہ موڑتے ہیں۔
وَمَآ أَرْسَلْنَا مِن قَبْلِكَ مِن رَّسُولٍ إِلَّا نُوحِيٓ إِلَيْهِ أَنَّهُۥ لَآ إِلَـٰهَ إِلَّآ أَنَا۠ فَٱعْبُدُونِ ﴿25﴾
اور ہم نے آپ سے پہلے کوئی رسول نہیں بھیجا مگر ہم نے اس کی طرف وحی کی کہ میرے سوا کوئی معبود نہیں، پس میری عبادت کرو۔
وَقَالُوا۟ ٱتَّخَذَ ٱلرَّحْمَـٰنُ وَلَدٗاۗ سُبْحَـٰنَهُۥۚ بَلْ عِبَادٞ مُّكْرَمُونَ ﴿26﴾
اور انہوں نے کہا: رحمٰن نے بیٹا بنایا ہے، وہ پاک ہے، بلکہ وہ عزت والے بندے ہیں۔
لَا يَسْبِقُونَهُۥ بِٱلْقَوْلِ وَهُم بِأَمْرِهِۦ يَعْمَلُونَ ﴿27﴾
وہ اس سے بات میں سبقت نہیں کرتے، اور وہ اسی کے حکم پر عمل کرتے ہیں۔
يَعْلَمُ مَا بَيْنَ أَيْدِيهِمْ وَمَا خَلْفَهُمْۖ وَلَا يَشْفَعُونَ إِلَّا لِمَنِ ٱرْتَضَىٰ وَهُم مِّنْ خَشْيَتِهِۦ مُشْفِقُونَ ﴿28﴾
وہ جانتا ہے جو ان کے آگے ہے اور جو ان کے پیچھے ہے، اور وہ سفارش نہیں کرتے مگر اس کے لیے جس سے وہ راضی ہو، اور وہ اس کی خشیت سے ڈرتے ہیں۔
وَمَن يَقُلْ مِنْهُمْ إِنِّيٓ إِلَـٰهٞ مِّن دُونِهِۦ فَذَٰلِكَ نَجْزِيهِ جَهَنَّمَۚ كَذَٰلِكَ نَجْزِي ٱلظَّـٰلِمِينَ ﴿29﴾
اور ان میں سے جو کہے کہ میں اس کے سوا معبود ہوں، تو ہم اسے جہنم کی سزا دیں گے، ہم ظالموں کو اسی طرح سزا دیتے ہیں۔
أَوَلَمْ يَرَ ٱلَّذِينَ كَفَرُوٓا۟ أَنَّ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ كَانَتَا رَتْقٗا فَفَتَقْنَـٰهُمَاۖ وَجَعَلْنَا مِنَ ٱلْمَآءِ كُلَّ شَيْءٍ حَيٍّۚ أَفَلَا يُؤْمِنُونَ ﴿30﴾
کیا کافروں نے نہیں دیکھا کہ آسمان اور زمین دونوں بند تھے، پھر ہم نے انہیں کھولا، اور ہم نے پانی سے ہر زندہ چیز پیدا کی، کیا وہ ایمان نہیں لاتے؟
وَجَعَلْنَا فِي ٱلْأَرْضِ رَوَٰسِيَ أَن تَمِيدَ بِهِمْ وَجَعَلْنَا فِيهَا فِجَاجٗا سُبُلٗا لَّعَلَّهُمْ يَهْتَدُونَ ﴿31﴾
اور ہم نے زمین میں مضبوط پہاڑ رکھے تاکہ وہ انہیں لے کر نہ ہلے، اور ہم نے اس میں راستے بنائے تاکہ وہ راہ پا سکیں۔
وَجَعَلْنَا ٱلسَّمَآءَ سَقْفٗا مَّحْفُوظٗاۖ وَهُمْ عَنْ ءَايَـٰتِهَا مُعْرِضُونَ ﴿32﴾
اور ہم نے آسمان کو محفوظ چھت بنایا، اور وہ اس کی نشانیوں سے منہ موڑتے ہیں۔
وَهُوَ ٱلَّذِي خَلَقَ ٱلَّيْلَ وَٱلنَّهَارَ وَٱلشَّمْسَ وَٱلْقَمَرَۖ كُلّٞ فِي فَلَكٖ يَسْبَحُونَ ﴿33﴾
اور وہی ہے جس نے رات اور دن، سورج اور چاند کو پیدا کیا، سب ایک مدار میں تیر رہے ہیں۔
وَمَا جَعَلْنَا لِبَشَرٖ مِّن قَبْلِكَ ٱلْخُلْدَۖ أَفَإِيْن مِّتَّ فَهُمُ ٱلْخَـٰلِدُونَ ﴿34﴾
اور ہم نے آپ سے پہلے کسی انسان کو ہمیشہ کی زندگی نہیں دی، تو کیا اگر آپ فوت ہو جائیں تو وہ ہمیشہ رہیں گے؟
كُلُّ نَفْسٖ ذَآئِقَةُ ٱلْمَوْتِۖ وَنَبْلُوكُم بِٱلشَّرِّ وَٱلْخَيْرِ فِتْنَةٗۖ وَإِلَيْنَا تُرْجَعُونَ ﴿35﴾
ہر جان کو موت کا مزہ چکھنا ہے، اور ہم تمہیں برائی اور بھلائی سے آزماتے ہیں، اور ہماری ہی طرف تم لوٹائے جاؤ گے۔
وَإِذَا رَءَاكَ ٱلَّذِينَ كَفَرُوٓا۟ إِن يَتَّخِذُونَكَ إِلَّا هُزُوًا أَهَـٰذَا ٱلَّذِي يَذْكُرُ ءَالِهَتَكُمْۚ وَهُم بِذِكْرِ ٱلرَّحْمَـٰنِ هُمْ كَـٰفِرُونَ ﴿36﴾
اور جب کافر آپ کو دیکھتے ہیں تو آپ کا مذاق ہی اڑاتے ہیں، (کہتے ہیں) کیا یہی ہے جو تمہارے معبودوں کا ذکر کرتا ہے؟ حالانکہ وہ خود رحمٰن کے ذکر سے انکار کرتے ہیں۔
خُلِقَ ٱلْإِنسَـٰنُ مِنْ عَجَلٖۚ سَأُرِيكُمْ ءَايَـٰتِي فَلَا تَسْتَعْجِلُونِ ﴿37﴾
انسان جلد بازی سے پیدا کیا گیا ہے، میں تمہیں اپنی نشانیاں دکھاؤں گا، پس تم جلدی نہ کرو۔
وَيَقُولُونَ مَتَىٰ هَـٰذَا ٱلْوَعْدُ إِن كُنتُمْ صَـٰدِقِينَ ﴿38﴾
اور وہ کہتے ہیں: یہ وعدہ کب پورا ہوگا اگر تم سچے ہو؟
لَوْ يَعْلَمُ ٱلَّذِينَ كَفَرُوا۟ حِينَ لَا يَكُفُّونَ عَن وُجُوهِهِمُ ٱلنَّارَ وَلَا عَن ظُهُورِهِمْ وَلَا هُمْ يُنصَرُونَ ﴿39﴾
کاش کافر جانتے جب وہ آگ کو اپنے چہروں اور اپنی پیٹھوں سے روک نہ سکیں گے، اور نہ ہی ان کی مدد کی جائے گی۔
بَلْ تَأْتِيهِم بَغْتَةٗ فَتَبْهَتُهُمْ فَلَا يَسْتَطِيعُونَ رَدَّهَا وَلَا هُمْ يُنظَرُونَ ﴿40﴾
بلکہ وہ ان پر اچانک آ جائے گی، پھر وہ انہیں حیران کر دے گی، اور وہ اسے روک نہیں سکیں گے، اور نہ ہی انہیں مہلت دی جائے گی۔
وَلَقَدِ ٱسْتُهْزِئَ بِرُسُلٖ مِّن قَبْلِكَ فَحَاقَ بِٱلَّذِينَ سَخِرُوا۟ مِنْهُم مَّا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ ﴿41﴾
اور آپ سے پہلے بھی رسولوں کا مذاق اڑایا گیا، تو ان مذاق اڑانے والوں کو وہی چیز گھیر لی جس کا وہ مذاق اڑاتے تھے۔
قُلْ مَن يَكْلَؤُكُم بِٱلَّيْلِ وَٱلنَّهَارِ مِنَ ٱلرَّحْمَـٰنِۚ بَلْ هُمْ عَن ذِكْرِ رَبِّهِم مُّعْرِضُونَ ﴿42﴾
کہہ دو: رات اور دن میں رحمٰن سے تمہیں کون بچاتا ہے؟ بلکہ وہ اپنے رب کے ذکر سے منہ موڑتے ہیں۔
أَمْ لَهُمْ ءَالِهَةٞ تَمْنَعُهُم مِّن دُونِنَاۚ لَا يَسْتَطِيعُونَ نَصْرَ أَنفُسِهِمْ وَلَا هُم مِّنَّا يُصْحَبُونَ ﴿43﴾
کیا ان کے ایسے معبود ہیں جو ہمیں چھوڑ کر انہیں بچا سکیں؟ وہ تو خود اپنی مدد نہیں کر سکتے، اور نہ ہی ہمارے مقابلے میں ان کی مدد کی جائے گی۔
بَلْ مَتَّعْنَا هَـٰٓؤُلَآءِ وَءَابَآءَهُمْ حَتَّىٰ طَالَ عَلَيْهِمُ ٱلْعُمُرُۚ أَفَلَا يَرَوْنَ أَنَّا نَأْتِي ٱلْأَرْضَ نَنقُصُهَا مِنْ أَطْرَافِهَآۚ أَفَهُمُ ٱلْغَـٰلِبُونَ ﴿44﴾
بلکہ ہم نے ان کو اور ان کے باپ دادا کو فائدہ دیا یہاں تک کہ ان پر عمر دراز ہو گئی، کیا وہ نہیں دیکھتے کہ ہم زمین کو اس کے کناروں سے کم کرتے جا رہے ہیں؟ تو کیا وہ غالب آنے والے ہیں؟
قُلْ إِنَّمَآ أُنذِرُكُم بِٱلْوَحْيِۚ وَلَا يَسْمَعُ ٱلصُّمُّ ٱلدُّعَآءَ إِذَا مَا يُنذَرُونَ ﴿45﴾
کہہ دو: میں تو تمہیں وحی کے ذریعے خبردار کرتا ہوں، اور جب انہیں خبردار کیا جاتا ہے تو بہرے پکار کو نہیں سنتے۔
وَلَئِن مَّسَّتْهُمْ نَفْحَةٞ مِّنْ عَذَابِ رَبِّكَ لَيَقُولُنَّ يَـٰوَيْلَنَآ إِنَّا كُنَّا ظَـٰلِمِينَ ﴿46﴾
اور اگر انہیں آپ کے رب کے عذاب کی ایک جھونک بھی چھو جائے تو وہ ضرور کہیں گے: ہائے ہماری بدبختی! بے شک ہم ہی ظالم تھے۔
وَنَضَعُ ٱلْمَوَٰزِينَ ٱلْقِسْطَ لِيَوْمِ ٱلْقِيَـٰمَةِ فَلَا تُظْلَمُ نَفْسٞ شَيْـٔٗاۖ وَإِن كَانَ مِثْقَالَ حَبَّةٖ مِّنْ خَرْدَلٍ أَتَيْنَا بِهَاۗ وَكَفَىٰ بِنَا حَـٰسِبِينَ ﴿47﴾
اور ہم قیامت کے دن انصاف کے ترازو رکھیں گے، تو کسی جان پر کچھ بھی ظلم نہیں کیا جائے گا، اور اگر وہ رائی کے دانے کے برابر بھی ہو تو ہم اسے لے آئیں گے، اور ہم حساب لینے کے لیے کافی ہیں۔
وَلَقَدْ ءَاتَيْنَا مُوسَىٰ وَهَـٰرُونَ ٱلْفُرْقَانَ وَضِيَآءٗ وَذِكْرٗا لِّلْمُتَّقِينَ ﴿48﴾
اور ہم نے موسیٰ اور ہارون کو فرقان، روشنی اور متقیوں کے لیے نصیحت دی۔
ٱلَّذِينَ يَخْشَوْنَ رَبَّهُم بِٱلْغَيْبِ وَهُم مِّنَ ٱلسَّاعَةِ مُشْفِقُونَ ﴿49﴾
جو اپنے رب سے غیب میں ڈرتے ہیں، اور وہ قیامت سے لرزاں ہیں۔
وَهَـٰذَا ذِكْرٞ مُّبَارَكٌ أَنزَلْنَـٰهُۚ أَفَأَنتُمْ لَهُۥ مُنكِرُونَ ﴿50﴾
اور یہ ایک بابرکت نصیحت ہے جسے ہم نے نازل کیا ہے، تو کیا تم اس کا انکار کرتے ہو؟
وَلَقَدْ ءَاتَيْنَآ إِبْرَٰهِيمَ رُشْدَهُۥ مِن قَبْلُ وَكُنَّا بِهِۦ عَـٰلِمِينَ ﴿51﴾
اور ہم نے ابراہیم کو پہلے ہی اس کی ہدایت دی تھی، اور ہم اس کے حال سے خوب واقف تھے۔
إِذْ قَالَ لِأَبِيهِ وَقَوْمِهِۦ مَا هَـٰذِهِ ٱلتَّمَاثِيلُ ٱلَّتِيٓ أَنتُمْ لَهَا عَـٰكِفُونَ ﴿52﴾
جب اس نے اپنے باپ اور قوم سے کہا: یہ مورتیاں کیا ہیں جن پر تم جمے بیٹھے ہو؟
قَالُوا۟ وَجَدْنَآ ءَابَآءَنَا لَهَا عَـٰبِدِينَ ﴿53﴾
انہوں نے کہا: ہم نے اپنے باپ دادا کو ان کی عبادت کرتے پایا ہے۔
قَالَ لَقَدْ كُنتُمْ أَنتُمْ وَءَابَآؤُكُمْ فِي ضَلَـٰلٖ مُّبِينٖ ﴿54﴾
اس نے کہا: بے شک تم اور تمہارے باپ دادا کھلی گمراہی میں تھے۔
قَالُوا۟ أَجِئْتَنَا بِٱلْحَقِّ أَمْ أَنتَ مِنَ ٱللَّـٰعِبِينَ ﴿55﴾
انہوں نے کہا: کیا تم ہمارے پاس حق لے کر آئے ہو یا تم مذاق کرنے والوں میں سے ہو؟
قَالَ بَل رَّبُّكُمْ رَبُّ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ ٱلَّذِي فَطَرَهُنَّ وَأَنَا۠ عَلَىٰ ذَٰلِكُم مِّنَ ٱلشَّـٰهِدِينَ ﴿56﴾
اس نے کہا: بلکہ تمہارا رب آسمانوں اور زمین کا رب ہے جس نے انہیں پیدا کیا، اور میں اس پر گواہی دینے والوں میں سے ہوں۔
وَتَٱللَّهِ لَأَكِيدَنَّ أَصْنَـٰمَكُم بَعْدَ أَن تُوَلُّوا۟ مُدْبِرِينَ ﴿57﴾
اور اللہ کی قسم! میں تمہارے بتوں کے ساتھ ضرور چال چلوں گا جب تم پیٹھ پھیر کر چلے جاؤ گے۔
فَجَعَلَهُمْ جُذَٰذٗا إِلَّا كَبِيرٗا لَّهُمْ لَعَلَّهُمْ إِلَيْهِ يَرْجِعُونَ ﴿58﴾
پھر اس نے انہیں ٹکڑے ٹکڑے کر دیا، سوائے ان کے بڑے کے، تاکہ وہ اس کی طرف رجوع کریں۔
قَالُوا۟ مَن فَعَلَ هَـٰذَا بِـَٔالِهَتِنَآ إِنَّهُۥ لَمِنَ ٱلظَّـٰلِمِينَ ﴿59﴾
انہوں نے کہا: ہمارے معبودوں کے ساتھ یہ کس نے کیا؟ بے شک وہ ضرور ظالموں میں سے ہے۔
قَالُوا۟ سَمِعْنَا فَتٗى يَذْكُرُهُمْ يُقَالُ لَهُۥٓ إِبْرَٰهِيمُ ﴿60﴾
انہوں نے کہا: ہم نے ایک نوجوان کو سنا ہے جو ان کا ذکر کرتا ہے، اسے ابراہیم کہا جاتا ہے۔
قَالُوا۟ فَأْتُوا۟ بِهِۦ عَلَىٰٓ أَعْيُنِ ٱلنَّاسِ لَعَلَّهُمْ يَشْهَدُونَ ﴿61﴾
انہوں نے کہا: اسے لوگوں کے سامنے لاؤ تاکہ وہ گواہی دیں۔
قَالُوٓا۟ أَأَنتَ فَعَلْتَ هَـٰذَا بِـَٔالِهَتِنَا يَـٰٓإِبْرَٰهِيمُ ﴿62﴾
انہوں نے کہا: اے ابراہیم! کیا تم نے ہمارے معبودوں کے ساتھ یہ کیا ہے؟
قَالَ بَلْ فَعَلَهُۥ كَبِيرُهُمْ هَـٰذَا فَسْـَٔلُوهُمْ إِن كَانُوا۟ يَنطِقُونَ ﴿63﴾
اس نے کہا: بلکہ ان کے بڑے نے یہ کیا ہے، پس ان سے پوچھو اگر وہ بولتے ہیں۔
فَرَجَعُوٓا۟ إِلَىٰٓ أَنفُسِهِمْ فَقَالُوٓا۟ إِنَّكُمْ أَنتُمُ ٱلظَّـٰلِمُونَ ﴿64﴾
پس وہ اپنے دلوں کی طرف لوٹے اور کہنے لگے: بے شک تم ہی ظالم ہو۔
ثُمَّ نُكِسُوا۟ عَلَىٰ رُءُوسِهِمْ لَقَدْ عَلِمْتَ مَا هَـٰٓؤُلَآءِ يَنطِقُونَ ﴿65﴾
پھر وہ الٹے ہو گئے اور کہنے لگے: تم جانتے ہو کہ یہ بولتے نہیں۔
قَالَ أَفَتَعْبُدُونَ مِن دُونِ ٱللَّهِ مَا لَا يَنفَعُكُمْ شَيْـٔٗا وَلَا يَضُرُّكُمْ ﴿66﴾
اس نے کہا: کیا تم اللہ کے سوا ایسی چیزوں کی عبادت کرتے ہو جو نہ تمہیں نفع دے سکتی ہیں اور نہ نقصان؟
أُفّٖ لَّكُمْ وَلِمَا تَعْبُدُونَ مِن دُونِ ٱللَّهِۚ أَفَلَا تَعْقِلُونَ ﴿67﴾
تم پر افسوس ہو اور ان پر بھی جن کی تم اللہ کے سوا عبادت کرتے ہو، کیا تم عقل نہیں رکھتے؟
قَالُوا۟ حَرِّقُوهُ وَٱنصُرُوٓا۟ ءَالِهَتَكُمْ إِن كُنتُمْ فَـٰعِلِينَ ﴿68﴾
انہوں نے کہا: اسے جلا دو اور اپنے معبودوں کی مدد کرو اگر تم کچھ کرنے والے ہو۔
قُلْنَا يَـٰنَارُ كُونِي بَرْدٗا وَسَلَـٰمًا عَلَىٰٓ إِبْرَٰهِيمَ ﴿69﴾
ہم نے کہا: اے آگ! ابراہیم پر ٹھنڈی اور سلامتی والی ہو جا۔
وَأَرَادُوا۟ بِهِۦ كَيْدٗا فَجَعَلْنَـٰهُمُ ٱلْأَخْسَرِينَ ﴿70﴾
اور انہوں نے اس کے ساتھ چال چلنے کا ارادہ کیا، تو ہم نے انہیں سب سے زیادہ نقصان اٹھانے والوں میں کر دیا۔
وَنَجَّيْنَـٰهُ وَلُوطًا إِلَى ٱلْأَرْضِ ٱلَّتِي بَـٰرَكْنَا فِيهَا لِلْعَـٰلَمِينَ ﴿71﴾
اور ہم نے اسے اور لوط کو نجات دی اس زمین کی طرف جس میں ہم نے جہان والوں کے لیے برکت رکھی تھی۔
وَوَهَبْنَا لَهُۥٓ إِسْحَـٰقَ وَيَعْقُوبَ نَافِلَةٗۖ وَكُلّٗا جَعَلْنَا صَـٰلِحِينَ ﴿72﴾
اور ہم نے اسے اسحاق عطا کیا اور یعقوب بطور مزید عطیہ، اور ہم نے سب کو نیک بنایا۔
وَجَعَلْنَـٰهُمْ أَئِمَّةٗ يَهْدُونَ بِأَمْرِنَا وَأَوْحَيْنَآ إِلَيْهِمْ فِعْلَ ٱلْخَيْرَٰتِ وَإِقَامَ ٱلصَّلَوٰةِ وَإِيتَآءَ ٱلزَّكَوٰةِۖ وَكَانُوا۟ لَنَا عَـٰبِدِينَ ﴿73﴾
اور ہم نے انہیں پیشوا بنایا، وہ ہمارے حکم سے ہدایت دیتے تھے، اور ہم نے ان کی طرف نیکیوں کے کام، نماز قائم کرنے اور زکوٰۃ دینے کی وحی کی، اور وہ ہمارے عبادت گزار تھے۔
وَلُوطٗا ءَاتَيْنَـٰهُ حُكْمٗا وَعِلْمٗا وَنَجَّيْنَـٰهُ مِنَ ٱلْقَرْيَةِ ٱلَّتِي كَانَت تَّعْمَلُ ٱلْخَبَـٰٓئِثَۚ إِنَّهُمْ كَانُوا۟ قَوْمَ سَوْءٖ فَـٰسِقِينَ ﴿74﴾
اور لوط کو ہم نے حکمت اور علم دیا، اور ہم نے اسے اس بستی سے نجات دی جو گندے کام کرتی تھی، بے شک وہ بدکار فاسق لوگ تھے۔
وَأَدْخَلْنَـٰهُ فِي رَحْمَتِنَآۖ إِنَّهُۥ مِنَ ٱلصَّـٰلِحِينَ ﴿75﴾
اور ہم نے اسے اپنی رحمت میں داخل کیا، بے شک وہ نیک لوگوں میں سے تھا۔
وَنُوحًا إِذْ نَادَىٰ مِن قَبْلُ فَٱسْتَجَبْنَا لَهُۥ فَنَجَّيْنَـٰهُ وَأَهْلَهُۥ مِنَ ٱلْكَرْبِ ٱلْعَظِيمِ ﴿76﴾
اور نوح کو، جب اس نے پہلے پکارا، تو ہم نے اس کی دعا قبول کی، اور اسے اور اس کے گھر والوں کو بڑی مصیبت سے نجات دی۔
وَنَصَرْنَـٰهُ مِنَ ٱلْقَوْمِ ٱلَّذِينَ كَذَّبُوا۟ بِـَٔايَـٰتِنَآۚ إِنَّهُمْ كَانُوا۟ قَوْمَ سَوْءٖ فَأَغْرَقْنَـٰهُمْ أَجْمَعِينَ ﴿77﴾
اور ہم نے اس کی مدد کی ان لوگوں کے مقابلے میں جنہوں نے ہماری آیات کو جھٹلایا، بے شک وہ بدکار قوم تھی، تو ہم نے ان سب کو غرق کر دیا۔
وَدَاوُۥدَ وَسُلَيْمَـٰنَ إِذْ يَحْكُمَانِ فِي ٱلْحَرْثِ إِذْ نَفَشَتْ فِيهِ غَنَمُ ٱلْقَوْمِ وَكُنَّا لِحُكْمِهِمْ شَـٰهِدِينَ ﴿78﴾
اور داؤد اور سلیمان کو، جب وہ کھیتی کے بارے میں فیصلہ کر رہے تھے، جب قوم کی بکریاں اس میں چر گئیں، اور ہم ان کے فیصلے کے گواہ تھے۔
فَفَهَّمْنَـٰهَا سُلَيْمَـٰنَۚ وَكُلّٗا ءَاتَيْنَا حُكْمٗا وَعِلْمٗاۚ وَسَخَّرْنَا مَعَ دَاوُۥدَ ٱلْجِبَالَ يُسَبِّحْنَ وَٱلطَّيْرَۚ وَكُنَّا فَـٰعِلِينَ ﴿79﴾
تو ہم نے سلیمان کو اس کا فہم دیا، اور ہم نے دونوں کو حکمت اور علم دیا، اور ہم نے داؤد کے ساتھ پہاڑوں اور پرندوں کو مسخر کر دیا کہ وہ تسبیح کرتے تھے، اور ہم ہی کرنے والے تھے۔
وَعَلَّمْنَـٰهُ صَنْعَةَ لَبُوسٖ لَّكُمْ لِتُحْصِنَكُم مِّنۢ بَأْسِكُمْۖ فَهَلْ أَنتُمْ شَـٰكِرُونَ ﴿80﴾
اور ہم نے اسے تمہارے لیے لباس بنانے کی صنعت سکھائی تاکہ وہ تمہیں تمہاری جنگ سے بچائے، تو کیا تم شکر گزار ہو؟
وَلِسُلَيْمَـٰنَ ٱلرِّيحَ عَاصِفَةًۭ تَجْرِى بِأَمْرِهِۦٓ إِلَى ٱلْأَرْضِ ٱلَّتِى بَـٰرَكْنَا فِيهَا ۚ وَكُنَّا بِكُلِّ شَىْءٍ عَـٰلِمِينَ ﴿81﴾
اور ہم نے سلیمان کے لیے تیز ہوا کو مسخر کر دیا، جو اس کے حکم سے اس زمین کی طرف چلتی تھی جس میں ہم نے برکت رکھی ہے، اور ہم ہر چیز کو جاننے والے تھے۔
وَمِنَ ٱلشَّيَـٰطِينِ مَن يَغُوصُونَ لَهُۥ وَيَعْمَلُونَ عَمَلًۭا دُونَ ذَٰلِكَ ۖ وَكُنَّا لَهُمْ حَـٰفِظِينَ ﴿82﴾
اور شیطانوں میں سے کچھ اس کے لیے غوطہ لگاتے تھے اور اس کے علاوہ بھی کام کرتے تھے، اور ہم ان کی حفاظت کرنے والے تھے۔
وَأَيُّوبَ إِذْ نَادَىٰ رَبَّهُۥٓ أَنِّى مَسَّنِىَ ٱلضُّرُّ وَأَنتَ أَرْحَمُ ٱلرَّٰحِمِينَ ﴿83﴾
اور ایوب کو (یاد کرو) جب اس نے اپنے رب کو پکارا: مجھے تکلیف پہنچی ہے، اور تو سب رحم کرنے والوں سے زیادہ رحم کرنے والا ہے۔
فَٱسْتَجَبْنَا لَهُۥ فَكَشَفْنَا مَا بِهِۦ مِن ضُرٍّۢ ۖ وَءَاتَيْنَـٰهُ أَهْلَهُۥ وَمِثْلَهُم مَّعَهُمْ رَحْمَةًۭ مِّنْ عِندِنَا وَذِكْرَىٰ لِلْعَـٰبِدِينَ ﴿84﴾
تو ہم نے اس کی دعا قبول کی اور اس کی تکلیف دور کر دی، اور اسے اس کے اہل و عیال دیے اور ان کے ساتھ اتنے ہی اور بھی، اپنی طرف سے رحمت کے طور پر اور عبادت گزاروں کے لیے نصیحت کے طور پر۔
وَإِسْمَـٰعِيلَ وَإِدْرِيسَ وَذَا ٱلْكِفْلِ ۖ كُلٌّۭ مِّنَ ٱلصَّـٰبِرِينَ ﴿85﴾
اور اسماعیل، ادریس اور ذوالکفل کو (یاد کرو)، یہ سب صبر کرنے والوں میں سے تھے۔
وَأَدْخَلْنَـٰهُمْ فِى رَحْمَتِنَآ ۖ إِنَّهُم مِّنَ ٱلصَّـٰلِحِينَ ﴿86﴾
اور ہم نے انہیں اپنی رحمت میں داخل کیا، بے شک وہ نیک لوگوں میں سے تھے۔
وَذَا ٱلنُّونِ إِذ ذَّهَبَ مُغَـٰضِبًۭا فَظَنَّ أَن لَّن نَّقْدِرَ عَلَيْهِ فَنَادَىٰ فِى ٱلظُّلُمَـٰتِ أَن لَّآ إِلَـٰهَ إِلَّآ أَنتَ سُبْحَـٰنَكَ إِنِّى كُنتُ مِنَ ٱلظَّـٰلِمِينَ ﴿87﴾
اور ذوالنون (یونس) کو (یاد کرو)، جب وہ غصے میں چلا گیا اور سمجھا کہ ہم اس پر قابو نہیں پائیں گے، تو اس نے اندھیروں میں پکارا: تیرے سوا کوئی معبود نہیں، تو پاک ہے، بے شک میں ہی ظالموں میں سے تھا۔
فَٱسْتَجَبْنَا لَهُۥ وَنَجَّيْنَـٰهُ مِنَ ٱلْغَمِّ ۚ وَكَذَٰلِكَ نُـۨجِى ٱلْمُؤْمِنِينَ ﴿88﴾
تو ہم نے اس کی دعا قبول کی اور اسے غم سے نجات دی، اور ہم اسی طرح ایمان والوں کو نجات دیتے ہیں۔
وَزَكَرِيَّآ إِذْ نَادَىٰ رَبَّهُۥ رَبِّ لَا تَذَرْنِى فَرْدًۭا وَأَنتَ خَيْرُ ٱلْوَٰرِثِينَ ﴿89﴾
اور زکریا کو (یاد کرو)، جب اس نے اپنے رب کو پکارا: اے میرے رب! مجھے اکیلا نہ چھوڑ، اور تو سب وارثوں سے بہتر ہے۔
فَٱسْتَجَبْنَا لَهُۥ وَوَهَبْنَا لَهُۥ يَحْيَىٰ وَأَصْلَحْنَا لَهُۥ زَوْجَهُۥٓ ۚ إِنَّهُمْ كَانُوا۟ يُسَـٰرِعُونَ فِى ٱلْخَيْرَٰتِ وَيَدْعُونَنَا رَغَبًۭا وَرَهَبًۭا ۖ وَكَانُوا۟ لَنَا خَـٰشِعِينَ ﴿90﴾
تو ہم نے اس کی دعا قبول کی اور اسے یحییٰ عطا کیا، اور اس کی بیوی کو اس کے لیے درست کر دیا، بے شک وہ نیکیوں میں جلدی کرتے تھے، اور ہمیں امید اور خوف سے پکارتے تھے، اور ہمارے لیے عاجزی کرنے والے تھے۔
وَٱلَّتِيٓ أَحْصَنَتْ فَرْجَهَا فَنَفَخْنَا فِيهَا مِن رُّوحِنَا وَجَعَلْنَـٰهَا وَٱبْنَهَآ ءَايَةٗ لِّلْعَـٰلَمِينَ ﴿91﴾
اور وہ عورت جس نے اپنی شرمگاہ کی حفاظت کی، تو ہم نے اس میں اپنی روح پھونکی، اور اسے اور اس کے بیٹے کو جہان والوں کے لیے نشانی بنا دیا۔
إِنَّ هَـٰذِهِۦٓ أُمَّتُكُمْ أُمَّةٗ وَٰحِدَةٗ وَأَنَا۠ رَبُّكُمْ فَٱعْبُدُونِ ﴿92﴾
بے شک یہ تمہاری امت ایک ہی امت ہے، اور میں تمہارا رب ہوں، پس میری عبادت کرو۔
وَتَقَطَّعُوٓا۟ أَمْرَهُم بَيْنَهُمْۖ كُلٌّ إِلَيْنَا رَٰجِعُونَ ﴿93﴾
اور انہوں نے اپنے معاملے کو آپس میں ٹکڑے ٹکڑے کر لیا، ہر ایک ہماری طرف لوٹنے والا ہے۔
فَمَن يَعْمَلْ مِنَ ٱلصَّـٰلِحَـٰتِ وَهُوَ مُؤْمِنٞ فَلَا كُفْرَانَ لِسَعْيِهِۦۚ وَإِنَّا لَهُۥ كَـٰتِبُونَ ﴿94﴾
پس جو کوئی نیک عمل کرے اور وہ مؤمن ہو، تو اس کی کوشش کی ناقدری نہیں کی جائے گی، اور بے شک ہم اس کے اعمال لکھنے والے ہیں۔
وَحَرَامٌ عَلَىٰ قَرْيَةٍ أَهْلَكْنَـٰهَآ أَنَّهُمْ لَا يَرْجِعُونَ ﴿95﴾
اور جس بستی کو ہم نے ہلاک کر دیا، اس پر واپسی حرام ہے۔
حَتَّىٰٓ إِذَا فُتِحَتْ يَأْجُوجُ وَمَأْجُوجُ وَهُم مِّن كُلِّ حَدَبٖ يَنسِلُونَ ﴿96﴾
یہاں تک کہ جب یأجوج و مأجوج کھول دیے جائیں گے اور وہ ہر بلندی سے دوڑتے آئیں گے۔
وَٱقْتَرَبَ ٱلْوَعْدُ ٱلْحَقُّ فَإِذَا هِيَ شَـٰخِصَةٌ أَبْصَـٰرُ ٱلَّذِينَ كَفَرُوا۟ ۚ يَـٰوَيْلَنَا قَدْ كُنَّا فِى غَفْلَةٖ مِّنْ هَـٰذَا بَلْ كُنَّا ظَـٰلِمِينَ ﴿97﴾
اور جب سچا وعدہ قریب آ جائے گا تو کافروں کی آنکھیں پھٹی کی پھٹی رہ جائیں گی، (کہیں گے:) ہائے ہماری بدبختی! ہم تو اس سے غفلت میں تھے، بلکہ ہم ہی ظالم تھے۔
إِنَّكُمْ وَمَا تَعْبُدُونَ مِن دُونِ ٱللَّهِ حَصَبُ جَهَنَّمَ ۖ أَنتُمْ لَهَا وَارِدُونَ ﴿98﴾
بے شک تم اور جنہیں تم اللہ کے سوا پوجتے ہو، جہنم کا ایندھن ہیں، تم سب اس میں داخل ہونے والے ہو۔
لَوْ كَانَ هَـٰٓؤُلَآءِ ءَالِهَةٗ مَّا وَرَدُوهَا ۖ وَكُلّٞ فِيهَا خَـٰلِدُونَ ﴿99﴾
اگر یہ (تمہارے معبود) واقعی معبود ہوتے تو جہنم میں داخل نہ ہوتے، اور سب اس میں ہمیشہ رہیں گے۔
لَهُمْ فِيهَا زَفِيرٞ وَهُمْ فِيهَا لَا يَسْمَعُونَ ﴿100﴾
ان کے لیے اس میں آہیں ہوں گی، اور وہ اس میں کچھ نہیں سن سکیں گے۔
إِنَّ ٱلَّذِينَ سَبَقَتْ لَهُم مِّنَّا ٱلْحُسْنَىٰٓ أُو۟لَـٰٓئِكَ عَنْهَا مُبْعَدُونَ ﴿101﴾
بے شک جن کے لیے ہماری طرف سے بھلائی پہلے ہی طے ہو چکی ہے، وہ اس سے دور رکھے جائیں گے۔
لَا يَسْمَعُونَ حَسِيسَهَا وَهُمْ فِى مَا ٱشْتَهَتْ أَنفُسُهُمْ خَـٰلِدُونَ ﴿102﴾
وہ اس کی آہٹ بھی نہیں سنیں گے، اور وہ اپنی خواہش کے مطابق نعمتوں میں ہمیشہ رہیں گے۔
لَا يَحْزُنُهُمُ ٱلْفَزَعُ ٱلْأَكْبَرُ وَتَتَلَقَّىٰهُمُ ٱلْمَلَـٰٓئِكَةُ هَـٰذَا يَوْمُكُمُ ٱلَّذِى كُنتُمْ تُوعَدُونَ ﴿103﴾
انہیں بڑی گھبراہٹ غمگین نہیں کرے گی، اور فرشتے ان کا استقبال کریں گے (کہیں گے:) یہ ہے تمہارا وہ دن جس کا تم سے وعدہ کیا جاتا تھا۔
يَوْمَ نَطْوِى ٱلسَّمَآءَ كَطَىِّ ٱلسِّجِلِّ لِلْكُتُبِ ۚ كَمَا بَدَأْنَآ أَوَّلَ خَلْقٖ نُّعِيدُهُۥ ۚ وَعْدًا عَلَيْنَآ ۚ إِنَّا كُنَّا فَـٰعِلِينَ ﴿104﴾
جس دن ہم آسمان کو لپیٹ دیں گے جیسے لکھنے والا رجسٹر لپیٹتا ہے، جس طرح ہم نے پہلی بار پیدا کیا تھا، اسی طرح ہم دوبارہ لوٹائیں گے، یہ ہم پر وعدہ ہے، بے شک ہم کرنے والے ہیں۔
وَلَقَدْ كَتَبْنَا فِى ٱلزَّبُورِ مِنۢ بَعْدِ ٱلذِّكْرِ أَنَّ ٱلْأَرْضَ يَرِثُهَا عِبَادِيَ ٱلصَّـٰلِحُونَ ﴿105﴾
اور ہم نے زبور میں، ذکر کے بعد لکھ دیا تھا کہ زمین کے وارث میرے نیک بندے ہوں گے۔
إِنَّ فِي هَـٰذَا لَبَلَـٰغٗا لِّقَوْمٍ عَـٰبِدِينَ ﴿106﴾
بے شک اس میں عبادت گزار لوگوں کے لیے کافی پیغام ہے۔
وَمَآ أَرْسَلْنَـٰكَ إِلَّا رَحْمَةٗ لِّلْعَـٰلَمِينَ ﴿107﴾
اور ہم نے آپ کو تمام جہانوں کے لیے رحمت بنا کر ہی بھیجا ہے۔
قُلْ إِنَّمَا يُوحَىٰٓ إِلَيَّ أَنَّمَآ إِلَـٰهُكُمْ إِلَـٰهٞ وَٰحِدٞۖ فَهَلْ أَنتُم مُّسْلِمُونَ ﴿108﴾
کہہ دو: مجھ پر تو یہی وحی کی جاتی ہے کہ تمہارا معبود ایک ہی معبود ہے، تو کیا تم فرمانبردار بنتے ہو؟
فَإِن تَوَلَّوْا۟ فَقُلْ ءَاذَنتُكُمْ عَلَىٰ سَوَآءٖۖ وَإِنْ أَدْرِيٓ أَقَرِيبٌ أَم بَعِيدٞ مَّا تُوعَدُونَ ﴿109﴾
پھر اگر وہ منہ موڑیں تو کہہ دو: میں نے تم سب کو برابر خبردار کر دیا ہے، اور میں نہیں جانتا کہ جس کا تم سے وعدہ کیا جا رہا ہے وہ قریب ہے یا دور۔
إِنَّهُۥ يَعْلَمُ ٱلْجَهْرَ مِنَ ٱلْقَوْلِ وَيَعْلَمُ مَا تَكْتُمُونَ ﴿110﴾
بے شک وہ کھلی بات کو بھی جانتا ہے اور جو تم چھپاتے ہو اسے بھی جانتا ہے۔
وَإِنْ أَدْرِي لَعَلَّهُۥ فِتْنَةٞ لَّكُمْ وَمَتَـٰعٌ إِلَىٰ حِينٖ ﴿111﴾
اور میں نہیں جانتا، شاید وہ تمہارے لیے آزمائش ہو اور ایک وقت تک فائدہ۔
قَـٰلَ رَبِّ ٱحْكُم بِٱلْحَقِّۗ وَرَبُّنَا ٱلرَّحْمَـٰنُ ٱلْمُسْتَعَانُ عَلَىٰ مَا تَصِفُونَ ﴿112﴾
کہہ دو: اے میرے رب! حق کے ساتھ فیصلہ فرما، اور ہمارا رب رحمٰن ہی ہے جس سے تمہاری باتوں پر مدد مانگی جاتی ہے۔
سورة الحج
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
يَـٰٓأَيُّهَا ٱلنَّاسُ ٱتَّقُوا۟ رَبَّكُمْۚ إِنَّ زَلْزَلَةَ ٱلسَّاعَةِ شَىْءٌ عَظِيمٌ ﴿1﴾
اے لوگو! اپنے رب سے ڈرو، بے شک قیامت کا زلزلہ بڑی چیز ہے۔
يَوْمَ تَرَوْنَهَا تَذْهَلُ كُلُّ مُرْضِعٍ عَمَّآ أَرْضَعَتْ وَتَضَعُ كُلُّ ذَاتِ حَمْلٍ حَمْلَهَا وَتَرَى ٱلنَّاسَ سُكَـٰرَىٰ وَمَا هُم بِسُكَـٰرَىٰ وَلَـٰكِنَّ عَذَابَ ٱللَّهِ شَدِيدٞ ﴿2﴾
جس دن تم اسے دیکھو گے، ہر دودھ پلانے والی اپنے دودھ پیتے بچے کو بھول جائے گی، اور ہر حاملہ اپنا حمل گرا دے گی، اور تم لوگوں کو مدہوش دیکھو گے، حالانکہ وہ مدہوش نہ ہوں گے، بلکہ اللہ کا عذاب سخت ہے۔
وَمِنَ ٱلنَّاسِ مَن يُجَـٰدِلُ فِي ٱللَّهِ بِغَيْرِ عِلْمٖ وَيَتَّبِعُ كُلَّ شَيْطَـٰنٖ مَّرِيدٍ ﴿3﴾
اور لوگوں میں سے کچھ ایسے ہیں جو اللہ کے بارے میں بغیر علم کے جھگڑتے ہیں اور ہر سرکش شیطان کی پیروی کرتے ہیں۔
كُتِبَ عَلَيْهِۦٓ أَنَّهُۥ مَن تَوَلَّاهُ فَأَنَّهُۥ يُضِلُّهُۥ وَيَهْدِيهِ إِلَىٰ عَذَابِ ٱلسَّعِيرِ ﴿4﴾
اس پر لکھ دیا گیا ہے کہ جو اسے دوست بنائے گا، وہ اسے گمراہ کرے گا اور اسے بھڑکتی آگ کے عذاب کی طرف لے جائے گا۔
يَـٰٓأَيُّهَا ٱلنَّاسُ إِن كُنتُمْ فِي رَيْبٖ مِّنَ ٱلْبَعْثِ فَإِنَّا خَلَقْنَـٰكُم مِّن تُرَابٖ ثُمَّ مِن نُّطْفَةٖ ثُمَّ مِنْ عَلَقَةٖ ثُمَّ مِن مُّضْغَةٖ مُّخَلَّقَةٖ وَغَيْرِ مُخَلَّقَةٖ لِّنُبَيِّنَ لَكُمۚ وَنُقِرُّ فِي ٱلْأَرْحَامِ مَا نَشَآءُ إِلَىٰٓ أَجَلٖ مُّسَمّٗى ثُمَّ نُخْرِجُكُمْ طِفْلٗا ثُمَّ لِتَبْلُغُوٓا۟ أَشُدَّكُمْۖ وَمِنكُم مَّن يُتَوَفَّىٰ وَمِنكُم مَّن يُرَدُّ إِلَىٰٓ أَرْذَلِ ٱلْعُمُرِ لِكَيْ لَا يَعْلَمَ مِنۢ بَعْدِ عِلْمٖ شَيْـٔٗاۚ وَتَرَى ٱلْأَرْضَ هَامِدَةٗ فَإِذَآ أَنزَلْنَا عَلَيْهَا ٱلْمَآءَ ٱهْتَزَّتْ وَرَبَتْ وَأَنبَتَتْ مِن كُلِّ زَوْجِۭ بَهِيجٖ ﴿5﴾
اے لوگو! اگر تم قیامت کے بعد دوبارہ زندہ ہونے میں شک رکھتے ہو تو ہم نے تمہیں مٹی سے، پھر نطفہ سے، پھر خون کے لوتھڑے سے، پھر گوشت کے ٹکڑے سے پیدا کیا — جو مکمل بھی ہوتا ہے اور نامکمل بھی — تاکہ ہم تمہیں واضح کریں، اور ہم رحموں میں جسے چاہتے ہیں ایک مقرر وقت تک ٹھہراتے ہیں، پھر ہم تمہیں بچے کی صورت میں نکالتے ہیں، پھر تاکہ تم اپنی جوانی کو پہنچو، اور تم میں سے کچھ وفات پا جاتے ہیں اور کچھ بدترین عمر کی طرف لوٹائے جاتے ہیں تاکہ وہ علم کے بعد کچھ نہ جان سکیں، اور تم زمین کو خشک دیکھتے ہو، پھر جب ہم اس پر پانی برساتے ہیں تو وہ جنبش کرتی ہے اور پھولتی ہے اور ہر قسم کی خوشنما چیز اگاتی ہے۔
ذَٰلِكَ بِأَنَّ ٱللَّهَ هُوَ ٱلْحَقُّ وَأَنَّهُۥ يُحْيِ ٱلْمَوْتَىٰ وَأَنَّهُۥ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٞ ﴿6﴾
یہ اس لیے ہے کہ اللہ ہی حق ہے، اور وہی مردوں کو زندہ کرتا ہے، اور وہ ہر چیز پر قدرت رکھتا ہے۔
وَأَنَّ ٱلسَّاعَةَ ءَاتِيَةٞ لَّا رَيْبَ فِيهَا وَأَنَّ ٱللَّهَ يَبْعَثُ مَن فِي ٱلْقُبُورِ ﴿7﴾
اور بے شک قیامت آنے والی ہے، اس میں کوئی شک نہیں، اور اللہ ان کو قبروں سے اٹھائے گا۔
وَمِنَ ٱلنَّاسِ مَن يُجَـٰدِلُ فِي ٱللَّهِ بِغَيْرِ عِلْمٖ وَلَا هُدٗى وَلَا كِتَـٰبٖ مُّنِيرٖ ﴿8﴾
اور لوگوں میں سے کچھ ایسے ہیں جو اللہ کے بارے میں بغیر علم، بغیر ہدایت اور بغیر روشن کتاب کے جھگڑتے ہیں۔
ثَانِيَ عِطْفِهِۦ لِيُضِلَّ عَن سَبِيلِ ٱللَّهِۚ لَهُۥ فِي ٱلدُّنْيَا خِزْيٞۖ وَنُذِيقُهُۥ يَوْمَ ٱلْقِيَـٰمَةِ عَذَابَ ٱلْحَرِيقِ ﴿9﴾
وہ اپنے پہلو موڑ کر اللہ کے راستے سے گمراہ کرنے کی کوشش کرتا ہے، اس کے لیے دنیا میں رسوائی ہے، اور قیامت کے دن ہم اسے جلنے والے عذاب کا مزہ چکھائیں گے۔
ذَٰلِكَ بِمَا قَدَّمَتْ يَدَاكَ وَأَنَّ ٱللَّهَ لَيْسَ بِظَلَّـٰمٖ لِّلْعَبِيدِ ﴿10﴾
یہ اس وجہ سے ہے جو تیرے ہاتھوں نے آگے بھیجا، اور بے شک اللہ بندوں پر ظلم کرنے والا نہیں ہے۔
وَمِنَ ٱلنَّاسِ مَن يَعْبُدُ ٱللَّهَ عَلَىٰ حَرْفٖۖ فَإِنْ أَصَابَهُۥ خَيْرٌ ٱطْمَأَنَّ بِهِۦۖ وَإِنْ أَصَابَتْهُۥ فِتْنَةٌ ٱنقَلَبَ عَلَىٰ وَجْهِهِۦۚ خَسِرَ ٱلدُّنْيَا وَٱلْـَٔاخِرَةَۚ ذَٰلِكَ هُوَ ٱلْخُسْرَانُ ٱلْمُبِينُ ﴿11﴾
اور لوگوں میں کچھ ایسے ہیں جو کنارے پر کھڑے ہو کر اللہ کی عبادت کرتے ہیں، اگر انہیں کوئی بھلائی پہنچے تو مطمئن ہو جاتے ہیں، اور اگر کوئی آزمائش آئے تو منہ کے بل پلٹ جاتے ہیں، انہوں نے دنیا اور آخرت دونوں کا نقصان کیا، یہی کھلا نقصان ہے۔
يَدْعُوا۟ مِن دُونِ ٱللَّهِ مَا لَا يَضُرُّهُۥ وَمَا لَا يَنفَعُهُۥۚ ذَٰلِكَ هُوَ ٱلضَّلَـٰلُ ٱلْبَعِيدُ ﴿12﴾
وہ اللہ کے سوا ایسی چیز کو پکارتا ہے جو نہ اسے نقصان پہنچا سکتی ہے اور نہ فائدہ دے سکتی ہے، یہی دور کی گمراہی ہے۔
يَدْعُوا۟ لَمَن ضَرُّهُۥٓ أَقْرَبُ مِن نَّفْعِهِۦۚ لَبِئْسَ ٱلْمَوْلَىٰ وَلَبِئْسَ ٱلْعَشِيرُ ﴿13﴾
وہ ایسے کو پکارتا ہے جس کا نقصان اس کے نفع سے زیادہ قریب ہے، برا ہے وہ دوست اور برا ہے وہ ساتھی۔
إِنَّ ٱللَّهَ يُدْخِلُ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ جَنَّـٰتٖ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُۖ إِنَّ ٱللَّهَ يَفْعَلُ مَا يُرِيدُ ﴿14﴾
بے شک اللہ ایمان والوں اور نیک عمل کرنے والوں کو باغوں میں داخل کرے گا جن کے نیچے نہریں بہتی ہیں، بے شک اللہ جو چاہتا ہے کرتا ہے۔
مَن كَانَ يَظُنُّ أَن لَّن يَنصُرَهُ ٱللَّهُ فِي ٱلدُّنْيَا وَٱلْـَٔاخِرَةِ فَلْيَمْدُدْ بِسَبَبٍ إِلَى ٱلسَّمَآءِ ثُمَّ لْيَقْطَعْ فَلْيَنظُرْ هَلْ يُذْهِبَنَّ كَيْدُهُۥ مَا يَغِيظُ ﴿15﴾
جو شخص یہ گمان کرتا ہے کہ اللہ اس کی مدد نہیں کرے گا دنیا اور آخرت میں، تو اسے چاہیے کہ آسمان کی طرف کوئی ذریعہ بنائے، پھر اسے کاٹ دے، پھر دیکھے کہ کیا اس کی چال وہ چیز دور کر سکتی ہے جو اسے غصہ دلاتی ہے؟
وَكَذَٰلِكَ أَنزَلْنَـٰهُ ءَايَـٰتِۭ بَيِّنَـٰتٖۚ وَأَنَّ ٱللَّهَ يَهْدِي مَن يُرِيدُ ﴿16﴾
اور اسی طرح ہم نے اسے واضح آیات کے طور پر نازل کیا، اور بے شک اللہ جسے چاہتا ہے ہدایت دیتا ہے۔
إِنَّ ٱلَّذِينَ ءَامَنُوا۟ وَٱلَّذِينَ هَادُوا۟ وَٱلصَّـٰبِئِينَ وَٱلنَّصَـٰرَىٰ وَٱلْمَجُوسَ وَٱلَّذِينَ أَشْرَكُوا۟ إِنَّ ٱللَّهَ يَفْصِلُ بَيْنَهُمْ يَوْمَ ٱلْقِيَـٰمَةِۚ إِنَّ ٱللَّهَ عَلَىٰ كُلِّ شَيْءٖ شَهِيدٌ ﴿17﴾
بے شک جو لوگ ایمان لائے، اور جو یہودی ہوئے، اور صابی، اور نصرانی، اور مجوسی، اور جنہوں نے شرک کیا — اللہ قیامت کے دن ان کے درمیان فیصلہ کرے گا، بے شک اللہ ہر چیز پر گواہ ہے۔
أَلَمْ تَرَ أَنَّ ٱللَّهَ يُسْجُدُ لَهُۥ مَن فِي ٱلسَّمَـٰوَٰتِ وَمَن فِي ٱلْأَرْضِ وَٱلشَّمْسُ وَٱلْقَمَرُ وَٱلنُّجُومُ وَٱلْجِبَالُ وَٱلشَّجَرُ وَٱلدَّوَآبُّ وَكَثِيرٞ مِّنَ ٱلنَّاسِۖ وَكَثِيرٌ حَقَّ عَلَيْهِ ٱلْعَذَابُۗ وَمَن يُهِنِ ٱللَّهُ فَمَا لَهُۥ مِن مُّكْرِمٍۚ إِنَّ ٱللَّهَ يَفْعَلُ مَا يَشَآءُ ﴿18﴾
کیا تم نے نہیں دیکھا کہ اللہ کے لیے سجدہ کرتے ہیں وہ جو آسمانوں میں ہیں اور جو زمین میں ہیں، اور سورج، اور چاند، اور ستارے، اور پہاڑ، اور درخت، اور جانور، اور بہت سے انسان، اور بہت سے وہ ہیں جن پر عذاب واجب ہو چکا ہے، اور جسے اللہ ذلیل کرے تو اس کے لیے کوئی عزت دینے والا نہیں، بے شک اللہ جو چاہتا ہے کرتا ہے۔
هَـٰذَانِ خَصْمَانِ ٱخْتَصَمُوا۟ فِي رَبِّهِمْۖ فَٱلَّذِينَ كَفَرُوا۟ قُطِّعَتْ لَهُمْ ثِيَابٞ مِّن نَّارٖ يُصَبُّ مِن فَوْقِ رُءُوسِهِمُ ٱلْحَمِيمُ ﴿19﴾
یہ دو گروہ ہیں جو اپنے رب کے بارے میں جھگڑتے ہیں، پس جنہوں نے کفر کیا ان کے لیے آگ کے لباس کاٹے جائیں گے، ان کے سروں پر کھولتا ہوا پانی ڈالا جائے گا۔
يُصْهَرُ بِهِۦ مَا فِي بُطُونِهِمْ وَٱلْجُلُودُ ﴿20﴾
جس سے ان کے پیٹوں کی چیزیں اور کھالیں پگھلا دی جائیں گی۔
وَلَهُم مَّقَـٰمِعُ مِنْ حَدِيدٍ ﴿21﴾
اور ان کے لیے لوہے کے گرز ہوں گے۔
كُلَّمَآ أَرَادُوا۟ أَن يَخْرُجُوا۟ مِنْهَا مِنْ غَمٍّ أُعِيدُوا۟ فِيهَاۖ وَذُوقُوا۟ عَذَابَ ٱلْحَرِيقِ ﴿22﴾
جب بھی وہ اس سے نکلنے کی کوشش کریں گے غم کی وجہ سے، انہیں اس میں لوٹا دیا جائے گا، اور کہا جائے گا: جلنے والے عذاب کا مزہ چکھو۔
إِنَّ ٱللَّهَ يُدْخِلُ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ جَنَّـٰتٖ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُ يُحَلَّوْنَ فِيهَا مِنْ أَسَاوِرَ مِن ذَهَبٖ وَلُؤْلُؤٗاۖ وَلِبَاسُهُمْ فِيهَا حَرِيرٌ ﴿23﴾
بے شک اللہ ایمان والوں اور نیک عمل کرنے والوں کو باغوں میں داخل کرے گا جن کے نیچے نہریں بہتی ہیں، وہاں انہیں سونے اور موتیوں کے کنگن پہنائے جائیں گے، اور ان کا لباس ریشم ہوگا۔
وَهُدُوٓا۟ إِلَى ٱلطَّيِّبِ مِنَ ٱلْقَوْلِ وَهُدُوٓا۟ إِلَىٰ صِرَٰطِ ٱلْحَمِيدِ ﴿24﴾
اور انہیں پاکیزہ بات کی طرف ہدایت دی گئی، اور انہیں قابلِ تعریف راستے کی طرف ہدایت دی گئی۔
إِنَّ ٱلَّذِينَ كَفَرُوا۟ وَيَصُدُّونَ عَن سَبِيلِ ٱللَّهِ وَٱلْمَسْجِدِ ٱلْحَرَامِ ٱلَّذِي جَعَلْنَـٰهُ لِلنَّاسِ سَوَآءً ٱلْعَاكِفُ فِيهِ وَٱلْبَادِۚ وَمَن يُرِدْ فِيهِ بِإِلْحَادِۢ بِظُلْمٖ نُّذِقْهُ مِنْ عَذَابٍ أَلِيمٖ ﴿25﴾
بے شک جنہوں نے کفر کیا اور اللہ کے راستے سے اور مسجدِ حرام سے روکتے ہیں، جسے ہم نے سب لوگوں کے لیے بنایا ہے — وہاں رہنے والا اور باہر سے آنے والا برابر ہے — اور جو اس میں ظلم کے ساتھ کج روی کا ارادہ کرے، ہم اسے دردناک عذاب چکھائیں گے۔
وَإِذْ بَوَّأْنَا لِإِبْرَٰهِيمَ مَكَانَ ٱلْبَيْتِ أَن لَّا تُشْرِكْ بِي شَيْـٔٗا وَطَهِّرْ بَيْتِيَ لِلطَّآئِفِينَ وَٱلْقَآئِمِينَ وَٱلرُّكَّعِ ٱلسُّجُودِ ﴿26﴾
اور جب ہم نے ابراہیم کے لیے گھر کی جگہ مقرر کی کہ میرے ساتھ کسی کو شریک نہ کرو، اور میرے گھر کو طواف کرنے والوں، قیام کرنے والوں، رکوع کرنے والوں اور سجدہ کرنے والوں کے لیے پاک رکھو۔
وَأَذِّنْ فِي ٱلنَّاسِ بِٱلْحَجِّ يَأْتُوكَ رِجَالٗا وَعَلَىٰ كُلِّ ضَامِرٖ يَأْتِينَ مِن كُلِّ فَجٍّ عَمِيقٖ ﴿27﴾
اور لوگوں میں حج کا اعلان کرو، وہ تمہارے پاس پیدل آئیں گے اور ہر دبلی سواری پر آئیں گے، جو ہر دور دراز راستے سے آئیں گی۔
لِّيَشْهَدُوا۟ مَنَـٰفِعَ لَهُمْ وَيَذْكُرُوا۟ ٱسْمَ ٱللَّهِ فِيٓ أَيَّامٖ مَّعْلُومَـٰتٍ عَلَىٰ مَا رَزَقَهُم مِّنۢ بَهِيمَةِ ٱلْأَنْعَـٰمِۖ فَكُلُوا۟ مِنْهَا وَأَطْعِمُوا۟ ٱلْبَآئِسَ ٱلْفَقِيرَ ﴿28﴾
تاکہ وہ اپنے فائدے دیکھیں اور مقرر دنوں میں اللہ کا نام یاد کریں اس پر جو اس نے انہیں چوپایوں میں سے رزق دیا، پس اس میں سے کھاؤ اور محتاج فقیر کو کھلاؤ۔
ثُمَّ لْيَقْضُوا۟ تَفَثَهُمْ وَلْيُوفُوا۟ نُذُورَهُمْ وَلْيَطَّوَّفُوا۟ بِٱلْبَيْتِ ٱلْعَتِيقِ ﴿29﴾
پھر وہ اپنی میل کچیل دور کریں، اور اپنی نذریں پوری کریں، اور قدیم گھر کا طواف کریں۔
ذَٰلِكَ وَمَن يُعَظِّمْ حُرُمَـٰتِ ٱللَّهِ فَهُوَ خَيْرٞ لَّهُۥ عِندَ رَبِّهِۦۗ وَأُحِلَّتْ لَكُمُ ٱلْأَنْعَـٰمُ إِلَّا مَا يُتْلَىٰ عَلَيْكُمْۖ فَٱجْتَنِبُوا۟ ٱلرِّجْسَ مِنَ ٱلْأَوْثَـٰنِ وَٱجْتَنِبُوا۟ قَوْلَ ٱلزُّورِ ﴿30﴾
یہ ہے (اللہ کا حکم)، اور جو اللہ کی حرمتوں کی تعظیم کرے تو یہ اس کے رب کے نزدیک اس کے لیے بہتر ہے، اور تمہارے لیے چوپائے حلال کیے گئے ہیں، سوائے ان کے جو تم پر پڑھ کر سنائے گئے، پس بتوں کی گندگی سے بچو، اور جھوٹی بات سے بچو۔
حُنَفَآءَ لِلَّهِ غَيْرَ مُشْرِكِينَ بِهِۦۚ وَمَن يُشْرِكْ بِٱللَّهِ فَكَأَنَّمَا خَرَّ مِنَ ٱلسَّمَآءِ فَتَخْطَفُهُ ٱلطَّيْرُ أَوْ تَهْوِي بِهِ ٱلرِّيحُ فِي مَكَانٖ سَحِيقٖ ﴿31﴾
اللہ کے لیے یکسو ہو کر، اس کے ساتھ کسی کو شریک نہ کرتے ہوئے، اور جو اللہ کے ساتھ شرک کرے تو گویا وہ آسمان سے گر پڑا، پھر پرندے اسے اچک لیں یا ہوا اسے کسی دور جگہ میں پھینک دے۔
ذَٰلِكَ وَمَن يُعَظِّمْ شَعَـٰٓئِرَ ٱللَّهِ فَإِنَّهَا مِن تَقْوَى ٱلْقُلُوبِ ﴿32﴾
یہ (حکم) ہے، اور جو اللہ کی نشانیوں کی تعظیم کرے تو یہ دلوں کے تقویٰ سے ہے۔
لَكُمْ فِيهَا مَنَـٰفِعُ إِلَىٰٓ أَجَلٖ مُّسَمّٗى ثُمَّ مَحِلُّهَآ إِلَى ٱلْبَيْتِ ٱلْعَتِيقِ ﴿33﴾
تمہارے لیے ان (قربانی کے جانوروں) میں ایک مقرر وقت تک فائدے ہیں، پھر ان کی قربانی کا مقام قدیم گھر تک ہے۔
وَلِكُلِّ أُمَّةٖ جَعَلْنَا مَنسَكٗا لِّيَذْكُرُوا۟ ٱسْمَ ٱللَّهِ عَلَىٰ مَا رَزَقَهُم مِّنۢ بَهِيمَةِ ٱلْأَنْعَـٰمِۗ فَإِلَـٰهُكُمْ إِلَـٰهٞ وَٰحِدٞ فَلَهُۥٓ أَسْلِمُوا۟ۚ وَبَشِّرِ ٱلْمُخْبِتِينَ ﴿34﴾
اور ہر امت کے لیے ہم نے قربانی کا طریقہ مقرر کیا تاکہ وہ اللہ کا نام یاد کریں اس پر جو اس نے انہیں چوپایوں میں سے رزق دیا، پس تمہارا معبود ایک ہی معبود ہے، تو اسی کے فرمانبردار بنو، اور عاجزی کرنے والوں کو خوشخبری دے دو۔
ٱلَّذِينَ إِذَا ذُكِرَ ٱللَّهُ وَجِلَتْ قُلُوبُهُمْ وَٱلصَّـٰبِرِينَ عَلَىٰ مَآ أَصَابَهُمْ وَٱلْمُقِيمِي ٱلصَّلَوٰةِ وَمِمَّا رَزَقْنَـٰهُمْ يُنفِقُونَ ﴿35﴾
وہ لوگ کہ جب اللہ کا ذکر کیا جائے تو ان کے دل کانپ جاتے ہیں، اور جو مصیبت انہیں پہنچے اس پر صبر کرتے ہیں، اور نماز قائم کرتے ہیں، اور جو ہم نے انہیں دیا اس میں سے خرچ کرتے ہیں۔
وَٱلْبُدْنَ جَعَلْنَـٰهَا لَكُم مِّن شَعَـٰٓئِرِ ٱللَّهِ لَكُمْ فِيهَا خَيْرٞۖ فَٱذْكُرُوا۟ ٱسْمَ ٱللَّهِ عَلَيْهَا صَوَآفَّۖ فَإِذَا وَجَبَتْ جُنُوبُهَا فَكُلُوا۟ مِنْهَا وَأَطْعِمُوا۟ ٱلْقَانِعَ وَٱلْمُعْتَرَّۚ كَذَٰلِكَ سَخَّرْنَـٰهَا لَكُمْ لَعَلَّكُمْ تَشْكُرُونَ ﴿36﴾
اور قربانی کے اونٹوں کو ہم نے تمہارے لیے اللہ کی نشانیوں میں سے بنایا، تمہارے لیے ان میں بھلائی ہے، پس جب وہ قطار میں کھڑے ہوں تو ان پر اللہ کا نام لو، پھر جب وہ پہلو کے بل گر جائیں تو ان میں سے کھاؤ اور قناعت کرنے والے اور مانگنے والے کو کھلاؤ، اسی طرح ہم نے انہیں تمہارے لیے مسخر کیا تاکہ تم شکر ادا کرو۔
لَن يَنَالَ ٱللَّهَ لُحُومُهَا وَلَا دِمَآؤُهَا وَلَـٰكِن يَنَالُهُ ٱلتَّقْوَىٰ مِنكُمْۚ كَذَٰلِكَ سَخَّرَهَا لَكُمْ لِتُكَبِّرُوا۟ ٱللَّهَ عَلَىٰ مَا هَدَىٰكُمْۚ وَبَشِّرِ ٱلْمُحْسِنِينَ ﴿37﴾
اللہ کو نہ ان کا گوشت پہنچتا ہے اور نہ خون، بلکہ اسے تمہارا تقویٰ پہنچتا ہے، اسی طرح ہم نے انہیں تمہارے لیے مسخر کیا تاکہ تم اللہ کی بڑائی بیان کرو اس پر کہ اس نے تمہیں ہدایت دی، اور نیکی کرنے والوں کو خوشخبری دے دو۔
إِنَّ ٱللَّهَ يُدَٰفِعُ عَنِ ٱلَّذِينَ ءَامَنُوٓا۟ۗ إِنَّ ٱللَّهَ لَا يُحِبُّ كُلَّ خَوَّانٖ كَفُورٍ ﴿38﴾
بے شک اللہ ایمان والوں کی طرف سے دفاع کرتا ہے، بے شک اللہ ہر خیانت کرنے والے ناشکرے کو پسند نہیں کرتا۔
أُذِنَ لِلَّذِينَ يُقَـٰتَلُونَ بِأَنَّهُمْ ظُلِمُوا۟ۚ وَإِنَّ ٱللَّهَ عَلَىٰ نَصْرِهِمْ لَقَدِيرٌ ﴿39﴾
اجازت دی گئی ان لوگوں کو جن سے جنگ کی جاتی ہے، کیونکہ وہ مظلوم ہیں، اور بے شک اللہ ان کی مدد پر قادر ہے۔
ٱلَّذِينَ أُخْرِجُوا۟ مِن دِيَـٰرِهِم بِغَيْرِ حَقٍّ إِلَّآ أَن يَقُولُوا۟ رَبُّنَا ٱللَّهُۗ وَلَوْلَا دَفْعُ ٱللَّهِ ٱلنَّاسَ بَعْضَهُم بِبَعْضٖ لَّهُدِّمَتْ صَوَٰمِعُ وَبِيَعٞ وَصَلَوَٰتٞ وَمَسَـٰجِدُ يُذْكَرُ فِيهَا ٱسْمُ ٱللَّهِ كَثِيرٗاۗ وَلَيَنصُرَنَّ ٱللَّهُ مَن يَنصُرُهُۥٓۚ إِنَّ ٱللَّهَ لَقَوِيٌّ عَزِيزٌ ﴿40﴾
وہ لوگ جنہیں ناحق ان کے گھروں سے نکالا گیا، صرف اس لیے کہ وہ کہتے تھے: ہمارا رب اللہ ہے، اور اگر اللہ لوگوں کو ایک دوسرے کے ذریعے دفع نہ کرتا تو خانقاہیں، گرجے، عبادت گاہیں اور مسجدیں جن میں اللہ کا نام بہت لیا جاتا ہے، ضرور گرائی جاتیں، اور اللہ ضرور اس کی مدد کرے گا جو اس کی مدد کرے گا، بے شک اللہ قوت والا، غالب ہے۔
ٱلَّذِينَ إِن مَّكَّنَّـٰهُمْ فِي ٱلْأَرْضِ أَقَامُوا۟ ٱلصَّلَوٰةَ وَءَاتَوُا۟ ٱلزَّكَوٰةَ وَأَمَرُوا۟ بِٱلْمَعْرُوفِ وَنَهَوْا۟ عَنِ ٱلْمُنكَرِۗ وَلِلَّهِ عَـٰقِبَةُ ٱلْأُمُورِ ﴿41﴾
وہ لوگ کہ اگر ہم انہیں زمین میں اقتدار دیں تو وہ نماز قائم کریں گے، زکوٰۃ دیں گے، نیکی کا حکم دیں گے اور برائی سے روکیں گے، اور تمام امور کا انجام اللہ ہی کے ہاتھ میں ہے۔
وَإِن يُكَذِّبُوكَ فَقَدْ كَذَّبَتْ قَبْلَهُمْ قَوْمُ نُوحٖ وَعَادٞ وَثَمُودُ ﴿42﴾
اور اگر وہ آپ کو جھٹلاتے ہیں تو ان سے پہلے نوح کی قوم، عاد اور ثمود بھی جھٹلا چکے ہیں۔
وَقَوْمُ إِبْرَٰهِيمَ وَقَوْمُ لُوطٖ ﴿43﴾
اور ابراہیم کی قوم اور لوط کی قوم بھی۔
وَأَصْحَـٰبُ مَدْيَنَۖ وَكُذِّبَ مُوسَىٰ فَأَمْلَيْتُ لِلْكَـٰفِرِينَ ثُمَّ أَخَذْتُهُمْۖ فَكَيْفَ كَانَ نَكِيرِ ﴿44﴾
اور مدین والے، اور موسیٰ کو بھی جھٹلایا گیا، تو میں نے کافروں کو مہلت دی، پھر انہیں پکڑ لیا، تو کیسا تھا میرا عذاب؟
فَكَأَيِّن مِّن قَرْيَةٍ أَهْلَكْنَـٰهَا وَهِىَ ظَـٰلِمَةٞ فَهِيَ خَاوِيَةٌ عَلَىٰ عُرُوشِهَا وَبِئْرٖ مُّعَطَّلَةٖ وَقَصْرٖ مَّشِيدٍ ﴿45﴾
تو کتنی ہی بستیاں ہم نے ہلاک کر دیں جو ظالم تھیں، پس وہ اپنی چھتوں پر گری پڑی تھیں، اور کتنے ہی بے کار کنویں اور بلند و بالا محل (ویران ہو گئے)۔
أَفَلَمْ يَسِيرُوا۟ فِي ٱلْأَرْضِ فَتَكُونَ لَهُمْ قُلُوبٞ يَعْقِلُونَ بِهَآ أَوْ ءَاذَانٞ يَسْمَعُونَ بِهَاۖ فَإِنَّهَا لَا تَعْمَى ٱلْأَبْصَـٰرُ وَلَـٰكِن تَعْمَى ٱلْقُلُوبُ ٱلَّتِي فِي ٱلصُّدُورِ ﴿46﴾
کیا وہ زمین میں چلے پھرے نہیں تاکہ ان کے دل ہوتے جن سے وہ سمجھتے، یا کان ہوتے جن سے وہ سنتے؟ پس حقیقت یہ ہے کہ آنکھیں اندھی نہیں ہوتیں بلکہ وہ دل اندھے ہو جاتے ہیں جو سینوں میں ہیں۔
وَيَسْتَعْجِلُونَكَ بِٱلْعَذَابِ وَلَن يُخْلِفَ ٱللَّهُ وَعْدَهُۥۚ وَإِنَّ يَوْمٗا عِندَ رَبِّكَ كَأَلْفِ سَنَةٖ مِّمَّا تَعُدُّونَ ﴿47﴾
اور وہ آپ سے عذاب کو جلدی مانگتے ہیں، حالانکہ اللہ اپنے وعدے کے خلاف نہیں کرتا، اور بے شک آپ کے رب کے نزدیک ایک دن تمہارے شمار کے ہزار سال کے برابر ہے۔
وَكَأَيِّن مِّن قَرْيَةٍ أَمْلَيْتُ لَهَا وَهِىَ ظَـٰلِمَةٞ ثُمَّ أَخَذْتُهَاۖ وَإِلَيَّ ٱلْمَصِيرُ ﴿48﴾
اور کتنی ہی بستیاں ہیں جنہیں میں نے مہلت دی حالانکہ وہ ظالم تھیں، پھر میں نے انہیں پکڑ لیا، اور میری ہی طرف لوٹ کر آنا ہے۔
قُلْ يَـٰٓأَيُّهَا ٱلنَّاسُ إِنَّمَآ أَنَا۠ لَكُمْ نَذِيرٞ مُّبِينٞ ﴿49﴾
کہہ دو: اے لوگو! میں تو تمہارے لیے صرف ایک کھلا ہوا ڈرانے والا ہوں۔
فَٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ لَهُم مَّغْفِرَةٞ وَرِزْقٞ كَرِيمٞ ﴿50﴾
پھر جو لوگ ایمان لائے اور نیک عمل کیے، ان کے لیے بخشش اور عزت والا رزق ہے۔
وَٱلَّذِينَ سَعَوْا۟ فِىٓ ءَايَـٰتِنَا مُعَـٰجِزِينَ أُو۟لَـٰٓئِكَ أَصْحَـٰبُ ٱلْجَحِيمِ ﴿51﴾
اور جو لوگ ہماری آیات کو ناکام بنانے کی کوشش کرتے ہیں، وہی جہنم والے ہیں۔
وَمَآ أَرْسَلْنَا مِن قَبْلِكَ مِن رَّسُولٍۢ وَلَا نَبِىٍّ إِلَّآ إِذَا تَمَنَّىٰٓ أَلْقَى ٱلشَّيْطَـٰنُ فِىٓ أُمْنِيَّتِهِۦ فَيَنسَخُ ٱللَّهُ مَا يُلْقِى ٱلشَّيْطَـٰنُ ثُمَّ يُحْكِمُ ٱللَّهُ ءَايَـٰتِهِۦ ۗ وَٱللَّهُ عَلِيمٌ حَكِيمٌۭ ﴿52﴾
اور ہم نے آپ سے پہلے کوئی رسول یا نبی نہیں بھیجا مگر جب وہ کچھ چاہتا تو شیطان اس کی چاہت میں کچھ ڈال دیتا، پس اللہ شیطان کی ڈالی ہوئی چیز کو مٹا دیتا ہے، پھر اللہ اپنی آیات کو مضبوط کرتا ہے، اور اللہ علم والا، حکمت والا ہے۔
لِّيَجْعَلَ مَا يُلْقِى ٱلشَّيْطَـٰنُ فِتْنَةًۭ لِّلَّذِينَ فِى قُلُوبِهِم مَّرَضٌۭ وَٱلْقَاسِيَةِ قُلُوبُهُمْ ۗ وَإِنَّ ٱلظَّـٰلِمِينَ لَفِى شِقَاقٍۭ بَعِيدٍۢ ﴿53﴾
تاکہ شیطان کی ڈالی ہوئی چیز ان لوگوں کے لیے آزمائش بن جائے جن کے دلوں میں بیماری ہے اور جن کے دل سخت ہیں، اور بے شک ظالم لوگ دور کی مخالفت میں ہیں۔
وَلِيَعْلَمَ ٱلَّذِينَ أُوتُوا۟ ٱلْعِلْمَ أَنَّهُ ٱلْحَقُّ مِن رَّبِّكَ فَيُؤْمِنُوا۟ بِهِۦ فَتُخْبِتَ لَهُۥ قُلُوبُهُمْۗ وَإِنَّ ٱللَّهَ لَهَادِ ٱلَّذِينَ ءَامَنُوٓا۟ إِلَىٰ صِرَٰطٖ مُّسْتَقِيمٖ ﴿54﴾
اور تاکہ وہ لوگ جنہیں علم دیا گیا ہے جان لیں کہ یہ آپ کے رب کی طرف سے حق ہے، پس وہ اس پر ایمان لائیں اور ان کے دل اس کے لیے عاجزی کریں، اور بے شک اللہ ایمان والوں کو سیدھے راستے کی طرف ہدایت دیتا ہے۔
وَلَا يَزَالُ ٱلَّذِينَ كَفَرُوا۟ فِي شَكّٖ مِّنْهُ حَتَّىٰ تَأْتِيَهُمُ ٱلسَّاعَةُ بَغْتَةً أَوْ يَأْتِيَهُمْ عَذَابُ يَوْمٍ عَقِيمٍ ﴿55﴾
اور جو لوگ کفر کرتے ہیں وہ اس کے بارے میں شک میں ہی رہیں گے یہاں تک کہ قیامت ان پر اچانک آ جائے یا انہیں کسی بے فائدہ دن کا عذاب آ جائے۔
ٱلْمُلْكُ يَوْمَئِذٖ لِّلَّهِۚ يَحْكُمُ بَيْنَهُمْۚ فَٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ فِي جَنَّـٰتِ ٱلنَّعِيمِ ﴿56﴾
اس دن بادشاہی صرف اللہ کی ہوگی، وہ ان کے درمیان فیصلہ کرے گا، پس جو لوگ ایمان لائے اور نیک عمل کیے وہ نعمتوں والے باغوں میں ہوں گے۔
وَٱلَّذِينَ كَفَرُوا۟ وَكَذَّبُوا۟ بِـَٔايَـٰتِنَا فَأُو۟لَـٰٓئِكَ لَهُمْ عَذَابٞ مُّهِينٞ ﴿57﴾
اور جنہوں نے کفر کیا اور ہماری آیات کو جھٹلایا، وہی لوگ ہیں جن کے لیے ذلت دینے والا عذاب ہے۔
وَٱلَّذِينَ هَاجَرُوا۟ فِي سَبِيلِ ٱللَّهِ ثُمَّ قُتِلُوٓا۟ أَوْ مَاتُوا۟ لَيَرْزُقَنَّهُمُ ٱللَّهُ رِزْقٗا حَسَنٗاۚ وَإِنَّ ٱللَّهَ لَهُوَ خَيْرُ ٱلرَّٰزِقِينَ ﴿58﴾
اور جو لوگ اللہ کی راہ میں ہجرت کرتے ہیں، پھر قتل کیے جاتے ہیں یا مر جاتے ہیں، اللہ انہیں بہترین رزق دے گا، اور بے شک اللہ سب سے بہتر رزق دینے والا ہے۔
لَيُدْخِلَنَّهُم مُّدْخَلٗا يَرْضَوْنَهُۥۚ وَإِنَّ ٱللَّهَ لَعَلِيمٌ حَلِيمٞ ﴿59﴾
اللہ انہیں ایسی جگہ داخل کرے گا جس سے وہ خوش ہوں گے، اور بے شک اللہ خوب جاننے والا، بردبار ہے۔
ذَٰلِكَ وَمَنْ عَاقَبَ بِمِثْلِ مَا عُوقِبَ بِهِۦ ثُمَّ بُغِيَ عَلَيْهِ لَيَنصُرَنَّهُ ٱللَّهُۚ إِنَّ ٱللَّهَ لَعَفُوٌّ غَفُورٌ ﴿60﴾
یہ (اللہ کا وعدہ) ہے، اور جو شخص بدلہ لے اس کے برابر جو اس پر ظلم کیا گیا، پھر اس پر زیادتی کی جائے، تو اللہ ضرور اس کی مدد کرے گا، بے شک اللہ معاف کرنے والا، بخشنے والا ہے۔
ذَٰلِكَ بِأَنَّ ٱللَّهَ يُولِجُ ٱلَّيْلَ فِي ٱلنَّهَارِ وَيُولِجُ ٱلنَّهَارَ فِي ٱلَّيْلِ وَأَنَّ ٱللَّهَ سَمِيعُۢ بَصِيرٞ ﴿61﴾
یہ اس لیے ہے کہ اللہ رات کو دن میں داخل کرتا ہے اور دن کو رات میں داخل کرتا ہے، اور بے شک اللہ سننے والا، دیکھنے والا ہے۔
ذَٰلِكَ بِأَنَّ ٱللَّهَ هُوَ ٱلْحَقُّ وَأَنَّ مَا يَدْعُونَ مِن دُونِهِ ٱلْبَـٰطِلُ وَأَنَّ ٱللَّهَ هُوَ ٱلْعَلِيُّ ٱلْكَبِيرُ ﴿62﴾
یہ اس لیے ہے کہ اللہ ہی حق ہے، اور جو کچھ وہ اس کے سوا پکارتے ہیں وہ باطل ہے، اور بے شک اللہ ہی بلند، عظمت والا ہے۔
أَلَمْ تَرَ أَنَّ ٱللَّهَ أَنزَلَ مِنَ ٱلسَّمَآءِ مَآءٗ فَتُصْبِحُ ٱلْأَرْضُ مُخْضَرَّةًۚ إِنَّ ٱللَّهَ لَطِيفٌ خَبِيرٞ ﴿63﴾
کیا تم نے نہیں دیکھا کہ اللہ نے آسمان سے پانی نازل کیا، تو زمین سبز ہو گئی؟ بے شک اللہ لطیف، خوب جاننے والا ہے۔
لَّهُۥ مَا فِي ٱلسَّمَـٰوَٰتِ وَمَا فِي ٱلْأَرْضِۗ وَإِنَّ ٱللَّهَ لَهُوَ ٱلْغَنِيُّ ٱلْحَمِيدُ ﴿64﴾
اسی کے لیے ہے جو کچھ آسمانوں میں ہے اور جو کچھ زمین میں ہے، اور بے شک اللہ ہی بے نیاز، قابلِ تعریف ہے۔
أَلَمْ تَرَ أَنَّ ٱللَّهَ سَخَّرَ لَكُم مَّا فِي ٱلْأَرْضِ وَٱلْفُلْكَ تَجْرِي فِي ٱلْبَحْرِ بِأَمْرِهِۦۖ وَيُمْسِكُ ٱلسَّمَآءَ أَن تَقَعَ عَلَى ٱلْأَرْضِ إِلَّا بِإِذْنِهِۦٓۚ إِنَّ ٱللَّهَ بِٱلنَّاسِ لَرَءُوفٞ رَّحِيمٞ ﴿65﴾
کیا تم نے نہیں دیکھا کہ اللہ نے تمہارے لیے زمین کی چیزیں مسخر کر دیں، اور کشتی کو جو سمندر میں اس کے حکم سے چلتی ہے؟ اور وہ آسمان کو تھامے ہوئے ہے کہ وہ زمین پر نہ گرے مگر اس کے حکم سے، بے شک اللہ لوگوں پر بہت مہربان، رحم کرنے والا ہے۔
وَهُوَ ٱلَّذِيٓ أَحْيَاكُمْ ثُمَّ يُمِيتُكُمْ ثُمَّ يُحْيِيكُمْۚ إِنَّ ٱلْإِنسَـٰنَ لَكَفُورٞ ﴿66﴾
اور وہی ہے جس نے تمہیں زندگی دی، پھر وہ تمہیں موت دے گا، پھر وہ تمہیں زندہ کرے گا، بے شک انسان بڑا ناشکرا ہے۔
لِّكُلِّ أُمَّةٖ جَعَلْنَا مَنسَكًا هُمْ نَـٰسِكُوهُۖ فَلَا يُنَـٰزِعُنَّكَ فِي ٱلْأَمْرِۚ وَٱدْعُ إِلَىٰ رَبِّكَۖ إِنَّكَ لَعَلَىٰ هُدٗى مُّسْتَقِيمٖ ﴿67﴾
ہر امت کے لیے ہم نے عبادت کا ایک طریقہ مقرر کیا ہے جس پر وہ عمل کرتے ہیں، پس وہ اس معاملے میں آپ سے جھگڑا نہ کریں، اور آپ اپنے رب کی طرف دعوت دیں، بے شک آپ سیدھی ہدایت پر ہیں۔
وَإِن جَـٰدَلُوكَ فَقُلِ ٱللَّهُ أَعْلَمُ بِمَا تَعْمَلُونَ ﴿68﴾
اور اگر وہ آپ سے جھگڑا کریں تو کہہ دیں: اللہ خوب جانتا ہے جو تم کرتے ہو۔
ٱللَّهُ يَحْكُمُ بَيْنَكُمْ يَوْمَ ٱلْقِيَـٰمَةِ فِيمَا كُنتُمْ فِيهِ تَخْتَلِفُونَ ﴿69﴾
اللہ قیامت کے دن تمہارے درمیان اس چیز میں فیصلہ کرے گا جس میں تم اختلاف کرتے تھے۔
أَلَمْ تَعْلَمْ أَنَّ ٱللَّهَ يَعْلَمُ مَا فِي ٱلسَّمَآءِ وَٱلْأَرْضِۗ إِنَّ ذَٰلِكَ فِي كِتَـٰبٍۢۚ إِنَّ ذَٰلِكَ عَلَى ٱللَّهِ يَسِيرٌ ﴿70﴾
کیا تم نہیں جانتے کہ اللہ جانتا ہے جو کچھ آسمان اور زمین میں ہے؟ بے شک یہ سب کچھ کتاب میں ہے، اور یہ اللہ کے لیے آسان ہے۔
وَيَعْبُدُونَ مِن دُونِ ٱللَّهِ مَا لَمْ يُنَزِّلْ بِهِۦ سُلْطَـٰنٗا وَمَا لَيْسَ لَهُم بِهِۦ عِلْمٞۗ وَمَا لِلظَّـٰلِمِينَ مِن نَّصِيرٖ ﴿71﴾
اور وہ اللہ کے سوا ایسی چیزوں کی عبادت کرتے ہیں جن کے لیے اس نے کوئی دلیل نازل نہیں کی، اور نہ ہی انہیں اس کا علم ہے، اور ظالموں کا کوئی مددگار نہیں۔
وَإِذَا تُتْلَىٰ عَلَيْهِمْ ءَايَـٰتُنَا بَيِّنَـٰتٖ تَعْرِفُ فِي وُجُوهِ ٱلَّذِينَ كَفَرُوا۟ ٱلْمُنكَرَۖ يَكَادُونَ يَسْطُونَ بِٱلَّذِينَ يَتْلُونَ عَلَيْهِمْ ءَايَـٰتِنَاۗ قُلْ أَفَأُنَبِّئُكُم بِشَرّٖ مِّن ذَٰلِكُمُ ٱلنَّارُۖ وَعَدَهَا ٱللَّهُ ٱلَّذِينَ كَفَرُوا۟ۖ وَبِئْسَ ٱلْمَصِيرُ ﴿72﴾
اور جب ان پر ہماری واضح آیات پڑھی جاتی ہیں تو تم کافروں کے چہروں پر ناگواری دیکھتے ہو، وہ قریب ہوتے ہیں کہ ان پر حملہ کر دیں جو ان پر ہماری آیات پڑھتے ہیں۔ کہہ دو: کیا میں تمہیں اس سے بھی بدتر چیز بتاؤں؟ وہ آگ ہے، جس کا اللہ نے کافروں سے وعدہ کیا ہے، اور وہ بہت ہی بری جگہ ہے۔
يَـٰٓأَيُّهَا ٱلنَّاسُ ضُرِبَ مَثَلٞ فَٱسْتَمِعُوا۟ لَهُۥٓۚ إِنَّ ٱلَّذِينَ تَدْعُونَ مِن دُونِ ٱللَّهِ لَن يَخْلُقُوا۟ ذُبَابٗا وَلَوِ ٱجْتَمَعُوا۟ لَهُۥۖ وَإِن يَسْلُبْهُمُ ٱلذُّبَابُ شَيْـٔٗا لَّا يَسْتَنقِذُوهُ مِنْهُۚ ضَعُفَ ٱلطَّالِبُ وَٱلْمَطْلُوبُ ﴿73﴾
اے لوگو! ایک مثال بیان کی جاتی ہے، اسے غور سے سنو: بے شک وہ جنہیں تم اللہ کے سوا پکارتے ہو، وہ ایک مکھی بھی پیدا نہیں کر سکتے، اگرچہ وہ سب اس کے لیے جمع ہو جائیں، اور اگر مکھی ان سے کوئی چیز چھین لے تو وہ اسے واپس نہیں لے سکتے، طالب اور مطلوب دونوں کمزور ہیں۔
مَا قَدَرُوا۟ ٱللَّهَ حَقَّ قَدْرِهِۦٓۚ إِنَّ ٱللَّهَ لَقَوِيٌّ عَزِيزٌ ﴿74﴾
انہوں نے اللہ کی قدر نہیں کی جیسا کہ اس کی قدر کرنے کا حق تھا، بے شک اللہ قوت والا، غالب ہے۔
ٱللَّهُ يَصْطَفِي مِنَ ٱلْمَلَـٰٓئِكَةِ رُسُلٗا وَمِنَ ٱلنَّاسِۚ إِنَّ ٱللَّهَ سَمِيعُۢ بَصِيرٞ ﴿75﴾
اللہ فرشتوں میں سے بھی پیغام پہنچانے والے چنتا ہے اور انسانوں میں سے بھی، بے شک اللہ سننے والا، دیکھنے والا ہے۔
يَعْلَمُ مَا بَيْنَ أَيْدِيهِمْ وَمَا خَلْفَهُمْۚ وَإِلَى ٱللَّهِ تُرْجَعُ ٱلْأُمُورُ ﴿76﴾
وہ جانتا ہے جو کچھ ان کے آگے ہے اور جو کچھ ان کے پیچھے ہے، اور تمام امور اللہ ہی کی طرف لوٹائے جاتے ہیں۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ ٱرْكَعُوا۟ وَٱسْجُدُوا۟ وَٱعْبُدُوا۟ رَبَّكُمْ وَٱفْعَلُوا۟ ٱلْخَيْرَ لَعَلَّكُمْ تُفْلِحُونَ ﴿77﴾
اے ایمان والو! رکوع کرو، سجدہ کرو، اپنے رب کی عبادت کرو، اور نیک کام کرو تاکہ تم کامیاب ہو جاؤ۔
وَجَـٰهِدُوا۟ فِي ٱللَّهِ حَقَّ جِهَادِهِۦۚ هُوَ ٱجْتَبَىٰكُمْ وَمَا جَعَلَ عَلَيْكُمْ فِي ٱلدِّينِ مِنْ حَرَجٖۚ مِّلَّةَ أَبِيكُمْ إِبْرَٰهِيمَۚ هُوَ سَمَّىٰكُمُ ٱلْمُسْلِمِينَ مِن قَبْلُ وَفِي هَـٰذَا لِيَكُونَ ٱلرَّسُولُ شَهِيدٗا عَلَيْكُمْ وَتَكُونُوا۟ شُهَدَآءَ عَلَى ٱلنَّاسِۚ فَأَقِيمُوا۟ ٱلصَّلَوٰةَ وَءَاتُوا۟ ٱلزَّكَوٰةَ وَٱعْتَصِمُوا۟ بِٱللَّهِۚ هُوَ مَوْلَىٰكُمۖ فَنِعْمَ ٱلْمَوْلَىٰ وَنِعْمَ ٱلنَّصِيرُ ﴿78﴾
اور اللہ کی راہ میں جہاد کرو جیسا کہ اس کے حق میں جہاد کرنا چاہیے، اس نے تمہیں چن لیا ہے، اور دین میں تم پر کوئی تنگی نہیں رکھی، تمہارے باپ ابراہیم کی ملت ہے، اس نے تمہیں پہلے بھی مسلمان کہا اور اس (قرآن) میں بھی، تاکہ رسول تم پر گواہ ہو اور تم لوگوں پر گواہ بنو، پس نماز قائم کرو، زکوٰۃ دو، اور اللہ کو مضبوطی سے تھام لو، وہی تمہارا مددگار ہے، بہترین مددگار اور بہترین کارساز۔
