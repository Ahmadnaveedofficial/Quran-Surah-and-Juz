سورة السجدة

بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
الم ﴿1﴾
الف، لام، میم
تَنزِيلُ ٱلْكِتَـٰبِ لَا رَيْبَ فِيهِ مِن رَّبِّ ٱلْعَـٰلَمِينَ ﴿2﴾
اس کتاب کا نازل کیا جانا، جس میں کوئی شک نہیں، رب العالمین کی طرف سے ہے۔
أَمْ يَقُولُونَ ٱفْتَرَىٰهُۚ بَلْ هُوَ ٱلْحَقُّ مِن رَّبِّكَ لِتُنذِرَ قَوْمٗا مَّآ أَتَىٰهُم مِّن نَّذِيرٖ مِّن قَبْلِكَ لَعَلَّهُمْ يَهْتَدُونَ ﴿3﴾
کیا وہ کہتے ہیں کہ اس نے اسے خود گھڑ لیا ہے؟ بلکہ یہ حق ہے تیرے رب کی طرف سے، تاکہ تو ایسے لوگوں کو ڈرائے جن کے پاس تجھ سے پہلے کوئی ڈرانے والا نہیں آیا، شاید وہ ہدایت پا جائیں۔
ٱللَّهُ ٱلَّذِي خَلَقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ وَمَا بَيْنَهُمَا فِي سِتَّةِ أَيَّامٖ ثُمَّ ٱسْتَوَىٰ عَلَى ٱلْعَرْشِۖ مَا لَكُم مِّن دُونِهِۦ مِن وَلِيّٖ وَلَا شَفِيعٍۚ أَفَلَا تَتَذَكَّرُونَ ﴿4﴾
اللہ وہ ہے جس نے آسمانوں اور زمین کو اور جو کچھ ان کے درمیان ہے چھ دنوں میں پیدا کیا، پھر عرش پر قائم ہوا۔ اس کے سوا نہ تمہارا کوئی دوست ہے اور نہ سفارش کرنے والا۔ کیا تم نصیحت نہیں پکڑتے؟
يُدَبِّرُ ٱلْأَمْرَ مِنَ ٱلسَّمَآءِ إِلَى ٱلْأَرْضِ ثُمَّ يَعْرُجُ إِلَيْهِ فِي يَوْمٖ كَانَ مِقْدَارُهُۥٓ أَلْفَ سَنَةٖ مِّمَّا تَعُدُّونَ ﴿5﴾
وہ آسمان سے زمین تک تمام امور کی تدبیر کرتا ہے، پھر وہ اس کی طرف چڑھتے ہیں ایک دن میں جس کی مقدار تمہارے شمار کے مطابق ہزار سال ہے۔
ذَٰلِكَ عَـٰلِمُ ٱلْغَيْبِ وَٱلشَّهَـٰدَةِ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿6﴾
یہی ہے غیب اور ظاہر کا جاننے والا، زبردست، رحم کرنے والا۔
ٱلَّذِيٓ أَحْسَنَ كُلَّ شَيْءٍ خَلَقَهُۥ وَبَدَأَ خَلْقَ ٱلْإِنسَـٰنِ مِن طِينٖ ﴿7﴾
جس نے ہر چیز کو خوبصورتی سے پیدا کیا، اور انسان کی پیدائش مٹی سے شروع کی۔
ثُمَّ جَعَلَ نَسْلَهُۥ مِن سُلَـٰلَةٖ مِّن مَّآءٖ مَّهِينٖ ﴿8﴾
پھر اس کی نسل کو ایک حقیر پانی کے نچوڑ سے بنایا۔
ثُمَّ سَوَّىٰهُ وَنَفَخَ فِيهِ مِن رُّوحِهِۦۖ وَجَعَلَ لَكُمُ ٱلسَّمْعَ وَٱلْأَبْصَـٰرَ وَٱلْأَفْـِٔدَةَۚ قَلِيلٗا مَّا تَشْكُرُونَ ﴿9﴾
پھر اسے درست کیا اور اس میں اپنی روح پھونکی، اور تمہارے لیے کان، آنکھیں اور دل بنائے، تم بہت کم شکر کرتے ہو۔
وَقَالُوا۟ أَءِذَا ضَلَلْنَا فِي ٱلْأَرْضِ أَءِنَّا لَفِي خَلْقٖ جَدِيدِۭۚ بَلْ هُم بِلِقَآءِ رَبِّهِمْ كَـٰفِرُونَ ﴿10﴾
اور وہ کہتے ہیں: کیا جب ہم زمین میں گم ہو جائیں گے، کیا ہم پھر نئی پیدائش میں آئیں گے؟ بلکہ وہ اپنے رب کی ملاقات کے منکر ہیں۔
قُلْ يَتَوَفَّىٰكُم مَّلَكُ ٱلْمَوْتِ ٱلَّذِي وُكِّلَ بِكُمْ ثُمَّ إِلَىٰ رَبِّكُمْ تُرْجَعُونَ ﴿11﴾
کہہ دیجیے: تمہیں موت کا فرشتہ قبض کرے گا جو تم پر مقرر کیا گیا ہے، پھر تم اپنے رب کی طرف لوٹائے جاؤ گے۔
وَلَوْ تَرَىٰٓ إِذِ ٱلْمُجْرِمُونَ نَاكِسُوا۟ رُءُوسِهِمْ عِندَ رَبِّهِمْ رَبَّنَآ أَبْصَرْنَا وَسَمِعْنَا فَٱرْجِعْنَا نَعْمَلْ صَـٰلِحًا إِنَّا مُوقِنُونَ ﴿12﴾
اور اگر آپ دیکھیں جب مجرم اپنے رب کے سامنے سر جھکائے ہوں گے، کہیں گے: اے ہمارے رب! ہم نے دیکھ لیا اور سن لیا، پس ہمیں واپس بھیج دے تاکہ ہم نیک عمل کریں، بے شک ہم یقین لانے والے ہیں۔
وَلَوْ شِئْنَا لَـَٔاتَيْنَا كُلَّ نَفْسٍ هُدَىٰهَا وَلَـٰكِنْ حَقَّ ٱلْقَوْلُ مِنِّي لَأَمْلَأَنَّ جَهَنَّمَ مِنَ ٱلْجِنَّةِ وَٱلنَّاسِ أَجْمَعِينَ ﴿13﴾
اور اگر ہم چاہتے تو ہر جان کو اس کی ہدایت دے دیتے، لیکن میری بات سچ ہو چکی ہے کہ میں ضرور جہنم کو جنوں اور انسانوں سب سے بھر دوں گا۔
فَذُوقُوا۟ بِمَا نَسِيتُمْ لِقَآءَ يَوْمِكُمْ هَـٰذَآ إِنَّا نَسِينَـٰكُمْۖ وَذُوقُوا۟ عَذَابَ ٱلْخُلْدِ بِمَا كُنتُمْ تَعْمَلُونَ ﴿14﴾
پس چکھو، کیونکہ تم نے اس دن کی ملاقات کو بھلا دیا تھا، ہم نے بھی تمہیں بھلا دیا، اور اپنے اعمال کے بدلے ہمیشہ کا عذاب چکھو۔
إِنَّمَا يُؤْمِنُ بِـَٔايَـٰتِنَا ٱلَّذِينَ إِذَا ذُكِّرُوا۟ بِهَا خَرُّوا۟ سُجَّدٗا وَسَبَّحُوا۟ بِحَمْدِ رَبِّهِمْ وَهُمْ لَا يَسْتَكْبِرُونَ ﴿15﴾
ہماری آیات پر وہی لوگ ایمان لاتے ہیں، جب انہیں ان کے ذریعے نصیحت کی جاتی ہے تو وہ سجدے میں گر پڑتے ہیں اور اپنے رب کی حمد کے ساتھ تسبیح کرتے ہیں، اور وہ تکبر نہیں کرتے۔
تَتَجَافَىٰ جُنُوبُهُمْ عَنِ ٱلْمَضَاجِعِ يَدْعُونَ رَبَّهُمْ خَوْفٗا وَطَمَعٗا وَمِمَّا رَزَقْنَـٰهُمْ يُنفِقُونَ ﴿16﴾
ان کے پہلو بستروں سے الگ رہتے ہیں، وہ اپنے رب کو خوف اور امید کے ساتھ پکارتے ہیں، اور جو کچھ ہم نے انہیں دیا ہے اس میں سے خرچ کرتے ہیں۔
فَلَا تَعْلَمُ نَفْسٞ مَّآ أُخْفِيَ لَهُم مِّن قُرَّةِ أَعْيُنٖ جَزَآءًۢ بِمَا كَانُوا۟ يَعْمَلُونَ ﴿17﴾
کوئی جان نہیں جانتی کہ ان کے لیے آنکھوں کی ٹھنڈک سے کیا کچھ چھپا کر رکھا گیا ہے، ان کے اعمال کے بدلے۔
أَفَمَن كَانَ مُؤْمِنٗا كَمَن كَانَ فَاسِقٗاۚ لَا يَسْتَوُۥنَ ﴿18﴾
تو کیا جو ایمان والا ہے وہ اس جیسا ہو سکتا ہے جو فاسق ہے؟ وہ برابر نہیں ہو سکتے۔
أَمَّا ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ فَلَهُمْ جَنَّـٰتُ ٱلْمَأْوَىٰ نُزُلٗا بِمَا كَانُوا۟ يَعْمَلُونَ ﴿19﴾
رہے وہ لوگ جو ایمان لائے اور نیک عمل کیے، تو ان کے لیے مہمانی کے طور پر رہنے کی جنتیں ہیں، ان کے اعمال کے بدلے۔
وَأَمَّا ٱلَّذِينَ فَسَقُوا۟ فَمَأْوَىٰهُمُ ٱلنَّارُۖ كُلَّمَآ أَرَادُوٓا۟ أَن يَخْرُجُوا۟ مِنْهَآ أُعِيدُوا۟ فِيهَا وَقِيلَ لَهُمْ ذُوقُوا۟ عَذَابَ ٱلنَّارِ ٱلَّذِي كُنتُم بِهِۦ تُكَذِّبُونَ ﴿20﴾
اور جو لوگ فاسق ہیں، ان کا ٹھکانہ آگ ہے، جب بھی وہ اس سے نکلنا چاہیں گے، انہیں اس میں لوٹا دیا جائے گا، اور ان سے کہا جائے گا: چکھو آگ کا عذاب جسے تم جھٹلاتے تھے۔
وَلَنُذِيقَنَّهُم مِّنَ ٱلْعَذَابِ ٱلْأَدْنَىٰ دُونَ ٱلْعَذَابِ ٱلْأَكْبَرِ لَعَلَّهُمْ يَرْجِعُونَ ﴿21﴾
اور ہم انہیں بڑے عذاب سے پہلے چھوٹے عذاب کا مزہ ضرور چکھائیں گے، شاید وہ رجوع کر لیں۔
وَمَنْ أَظْلَمُ مِمَّن ذُكِّرَ بِـَٔايَـٰتِ رَبِّهِۦ ثُمَّ أَعْرَضَ عَنْهَآۚ إِنَّا مِنَ ٱلْمُجْرِمِينَ مُنتَقِمُونَ ﴿22﴾
اور اس سے بڑھ کر ظالم کون ہے جسے اس کے رب کی آیات یاد دلائی گئیں، پھر اس نے ان سے منہ موڑ لیا؟ بے شک ہم مجرموں سے انتقام لینے والے ہیں۔
وَلَقَدْ ءَاتَيْنَا مُوسَى ٱلْكِتَـٰبَ فَلَا تَكُن فِي مِرْيَةٖ مِّن لِّقَآئِهِۦۖ وَجَعَلْنَـٰهُ هُدٗى لِّبَنِيٓ إِسْرَٰٓءِيلَ ﴿23﴾
اور بے شک ہم نے موسیٰ کو کتاب دی، پس تم اس کی ملاقات میں شک نہ کرو، اور ہم نے اسے بنی اسرائیل کے لیے ہدایت بنایا۔
وَجَعَلْنَا مِنْهُمْ أَئِمَّةٗ يَهْدُونَ بِأَمْرِنَا لَمَّا صَبَرُوا۟ۖ وَكَانُوا۟ بِـَٔايَـٰتِنَا يُوقِنُونَ ﴿24﴾
اور ہم نے ان میں سے کچھ پیشوا بنائے جو ہمارے حکم سے ہدایت دیتے تھے، جب انہوں نے صبر کیا، اور وہ ہماری آیات پر یقین رکھتے تھے۔
إِنَّ رَبَّكَ هُوَ يَفْصِلُ بَيْنَهُمْ يَوْمَ ٱلْقِيَـٰمَةِ فِيمَا كَانُوا۟ فِيهِ يَخْتَلِفُونَ ﴿25﴾
بے شک تیرا رب ہی قیامت کے دن ان کے درمیان ان باتوں کا فیصلہ کرے گا جن میں وہ اختلاف کرتے تھے۔
أَوَلَمْ يَهْدِ لَهُمْ كَمْ أَهْلَكْنَا مِن قَبْلِهِم مِّنَ ٱلْقُرُونِ يَمْشُونَ فِي مَسَـٰكِنِهِمْۚ إِنَّ فِي ذَٰلِكَ لَـَٔايَـٰتٍۢۖ أَفَلَا يَسْمَعُونَ ﴿26﴾
کیا انہیں یہ بات ہدایت نہیں دیتی کہ ہم نے ان سے پہلے کتنی ہی قومیں ہلاک کر دیں، جن کے رہنے کی جگہوں میں یہ چلتے پھرتے ہیں؟ بے شک اس میں نشانیاں ہیں، کیا وہ سنتے نہیں؟
أَوَلَمْ يَرَوْا۟ أَنَّا نَسُوقُ ٱلْمَآءَ إِلَى ٱلْأَرْضِ ٱلْجُرُزِ فَنُخْرِجُ بِهِۦ زَرْعٗا تَأْكُلُ مِنْهُ أَنْعَـٰمُهُمْ وَأَنفُسُهُمْۚ أَفَلَا يُبْصِرُونَ ﴿27﴾
کیا انہوں نے نہیں دیکھا کہ ہم پانی کو بنجر زمین کی طرف لے جاتے ہیں، پھر اس سے کھیتی نکالتے ہیں جسے ان کے جانور اور وہ خود کھاتے ہیں؟ کیا وہ دیکھتے نہیں؟
وَيَقُولُونَ مَتَىٰ هَـٰذَا ٱلْفَتْحُ إِن كُنتُمْ صَـٰدِقِينَ ﴿28﴾
اور وہ کہتے ہیں: یہ فیصلہ کب ہوگا اگر تم سچے ہو؟
قُلْ يَوْمَ ٱلْفَتْحِ لَا يَنفَعُ ٱلَّذِينَ كَفَرُوٓا۟ إِيمَـٰنُهُمْ وَلَا هُمْ يُنظَرُونَ ﴿29﴾
کہہ دیجیے: فیصلے کے دن کافروں کو ان کا ایمان فائدہ نہ دے گا، اور نہ ہی انہیں مہلت دی جائے گی۔
فَأَعْرِضْ عَنْهُمْ وَٱنتَظِرْ إِنَّهُم مُّنتَظِرُونَ ﴿30﴾
پس ان سے منہ موڑ لیجیے اور انتظار کیجیے، بے شک وہ بھی انتظار کر رہے ہیں۔
