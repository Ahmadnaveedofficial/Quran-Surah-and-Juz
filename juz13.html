13 پارہ


بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
وَمَآ أُبَرِّئُ نَفْسِيٓۚ إِنَّ ٱلنَّفْسَ لَأَمَّارَةُۢ بِٱلسُّوٓءِ إِلَّا مَا رَحِمَ رَبِّيٓۚ إِنَّ رَبِّي غَفُورٞ رَّحِيمٞ ﴿53﴾
اور میں اپنے نفس کو پاک نہیں سمجھتی، بے شک نفس تو برائی کا بہت حکم دینے والا ہے، مگر جس پر میرا رب رحم کرے، بے شک میرا رب بخشنے والا، مہربان ہے۔
وَقَالَ ٱلْمَلِكُ ٱئْتُونِي بِهِۦٓ أَسْتَخْلِصْهُ لِنَفْسِيۖ فَلَمَّا كَلَّمَهُۥ قَالَ إِنَّكَ ٱلْيَوْمَ لَدَيْنَا مَكِينٌ أَمِينٞ ﴿54﴾
اور بادشاہ نے کہا: اسے میرے پاس لاؤ تاکہ میں اسے اپنے لیے مخصوص کر لوں، پھر جب یوسف نے اس سے بات کی تو بادشاہ نے کہا: آج سے تم ہمارے ہاں صاحبِ مقام اور امانت دار ہو۔
قَالَ ٱجْعَلْنِي عَلَىٰ خَزَآئِنِ ٱلْأَرْضِ إِنِّي حَفِيظٌ عَلِيمٞ ﴿55﴾
یوسف نے کہا: مجھے زمین کے خزانے پر مقرر کر دو، بے شک میں حفاظت کرنے والا اور علم والا ہوں۔
وَكَذَٰلِكَ مَكَّنَّا لِيُوسُفَ فِي ٱلْأَرْضِ يَتَبَوَّأُ مِنْهَا حَيْثُ يَشَآءُۚ نُصِيبُ بِرَحْمَتِنَا مَن نَّشَآءُ وَلَا نُضِيعُ أَجْرَ ٱلْمُحْسِنِينَ ﴿56﴾
اور اسی طرح ہم نے یوسف کو زمین میں اقتدار دیا، وہ اس میں جہاں چاہے رہتا تھا، ہم اپنی رحمت جسے چاہیں دیتے ہیں، اور ہم نیکوکاروں کا اجر ضائع نہیں کرتے۔
وَلَأَجْرُ ٱلْـَٔاخِرَةِ خَيْرٞ لِّلَّذِينَ ءَامَنُوا۟ وَكَانُوا۟ يَتَّقُونَ ﴿57﴾
اور آخرت کا اجر ان لوگوں کے لیے بہتر ہے جو ایمان لائے اور تقویٰ اختیار کرتے رہے۔
وَجَآءَ إِخْوَةُ يُوسُفَ فَدَخَلُوا۟ عَلَيْهِ فَعَرَفَهُمْ وَهُم لَهُۥ مُنكِرُونَ ﴿58﴾
اور یوسف کے بھائی آئے اور اس کے پاس داخل ہوئے، تو یوسف نے انہیں پہچان لیا، اور وہ اسے نہ پہچان سکے۔
وَلَمَّا جَهَّزَهُم بِجَهَازِهِمْ قَالَ ٱئْتُونِي بِأَخٖ لَّكُم مِّنْ أَبِيكُمْۚ أَلَا تَرَوْنَ أَنِّيٓ أُوفِي ٱلْكَيْلَ وَأَنَا۠ خَيْرُ ٱلْمُنزِلِينَ ﴿59﴾
اور جب اس نے ان کا سامان تیار کر دیا تو کہا: اپنے والد کی طرف سے اپنے بھائی کو میرے پاس لاؤ، کیا تم نہیں دیکھتے کہ میں پورا ناپ تول دیتا ہوں اور بہترین مہمان نواز ہوں؟
فَإِن لَّمْ تَأْتُونِي بِهِۦ فَلَا كَيْلَ لَكُمْ عِندِي وَلَا تَقْرَبُونِ ﴿60﴾
پھر اگر تم اسے میرے پاس نہ لاؤ گے تو میرے پاس تمہارے لیے کوئی ناپ تول نہیں ہوگا، اور میرے قریب بھی نہ آنا۔
قَالُوا۟ سَنُرَٰوِدُ عَنْهُ أَبَاهُ وَإِنَّا لَفَـٰعِلُونَ ﴿61﴾
انہوں نے کہا: ہم اس کے بارے میں اس کے باپ کو راضی کرنے کی کوشش کریں گے، اور ہم ضرور ایسا کریں گے۔
وَقَالَ لِفِتْيَانِهِ ٱجْعَلُوا۟ بِضَـٰعَتَهُمْ فِي رِحَالِهِمْ لَعَلَّهُمْ يَعْرِفُونَهَآ إِذَا ٱنقَلَبُوٓا۟ إِلَىٰٓ أَهْلِهِمْ لَعَلَّهُمْ يَرْجِعُونَ ﴿62﴾
اور یوسف نے اپنے خادموں سے کہا: ان کا مال ان کے سامان میں رکھ دو، شاید جب وہ اپنے گھر والوں کے پاس جائیں تو اسے پہچان لیں اور واپس آ جائیں۔
فَلَمَّا رَجَعُوٓا۟ إِلَىٰٓ أَبِيهِمْ قَالُوا۟ يَـٰٓأَبَانَا مُنِعَ مِنَّا ٱلْكَيْلُ فَأَرْسِلْ مَعَنَآ أَخَانَا نَكْتَلْ وَإِنَّا لَهُۥ لَحَـٰفِظُونَ ﴿63﴾
پھر جب وہ اپنے باپ کے پاس واپس آئے تو کہا: اے ہمارے باپ! ہمیں ناپ تول سے روک دیا گیا ہے، پس ہمارے ساتھ ہمارے بھائی کو بھیج دیں تاکہ ہم ناپ تول حاصل کریں، اور ہم اس کے محافظ ہیں۔
قَالَ هَلْ ءَامَنُكُمْ عَلَيْهِ إِلَّا كَمَآ أَمِنتُكُمْ عَلَىٰٓ أَخِيهِ مِن قَبْلُۚ فَٱللَّهُ خَيْرٌ حَـٰفِظٗاۖ وَهُوَ أَرْحَمُ ٱلرَّٰحِمِينَ ﴿64﴾
اس نے کہا: کیا میں اس پر تمہارا اعتماد کروں جیسے اس کے بھائی پر پہلے کیا تھا؟ پس اللہ ہی بہتر محافظ ہے، اور وہ سب سے زیادہ رحم کرنے والا ہے۔
وَلَمَّا فَتَحُوا۟ مَتَـٰعَهُمْ وَجَدُوا۟ بِضَـٰعَتَهُمْ رُدَّتْ إِلَيْهِمْۖ قَالُوا۟ يَـٰٓأَبَانَا مَا نَبْغِيۖ هَـٰذِهِۦ بِضَـٰعَتُنَا رُدَّتْ إِلَيْنَاۖ وَنَمِيرُ أَهْلَنَا وَنَحْفَظُ أَخَانَا وَنَزْدَادُ كَيْلَ بَعِيرٖۖ ذَٰلِكَ كَيْلٞ يَسِيرٞ ﴿65﴾
اور جب انہوں نے اپنا سامان کھولا تو اپنا مال واپس پایا، انہوں نے کہا: اے ہمارے باپ! ہمیں اور کیا چاہیے؟ یہ ہمارا مال ہمیں واپس دے دیا گیا ہے، ہم اپنے گھر والوں کے لیے غلہ لائیں گے، اور اپنے بھائی کی حفاظت کریں گے، اور ایک اونٹ کا بوجھ زیادہ لائیں گے، یہ آسان ناپ تول ہے۔
قَالَ لَنْ أُرْسِلَهُۥ مَعَكُمْ حَتَّىٰ تُؤْتُونِ مَوْثِقٗا مِّنَ ٱللَّهِ لَتَأْتُنَّنِي بِهِۦٓ إِلَّآ أَن يُحَاطَ بِكُمْۖ فَلَمَّآ ءَاتَوْهُ مَوْثِقَهُمْ قَالَ ٱللَّهُ عَلَىٰ مَا نَقُولُ وَكِيلٞ ﴿66﴾
اس نے کہا: میں اسے تمہارے ساتھ نہیں بھیجوں گا جب تک کہ تم اللہ کی قسم نہ کھاؤ کہ تم اسے ضرور میرے پاس لاؤ گے، مگر یہ کہ تم گھیر لیے جاؤ، پھر جب انہوں نے قسم کھائی تو کہا: اللہ اس بات پر نگہبان ہے جو ہم کہہ رہے ہیں۔
وَقَالَ يَـٰبَنِيَّ لَا تَدْخُلُوا۟ مِنۢ بَابٖ وَٰحِدٖ وَٱدْخُلُوا۟ مِنْ أَبْوَٰبٖ مُّتَفَرِّقَةٖۖ وَمَآ أُغْنِي عَنكُم مِّنَ ٱللَّهِ مِن شَيْءٍۖ إِنِ ٱلْحُكْمُ إِلَّا لِلَّهِۖ عَلَيْهِ تَوَكَّلْتُۖ وَعَلَيْهِ فَلْيَتَوَكَّلِ ٱلْمُتَوَكِّلُونَ ﴿67﴾
اور اس نے کہا: اے میرے بیٹو! ایک دروازے سے داخل نہ ہونا، بلکہ مختلف دروازوں سے داخل ہونا، اور میں اللہ کے مقابلے میں تمہارے لیے کچھ نہیں کر سکتا، حکم تو صرف اللہ ہی کا ہے، اسی پر میں نے بھروسہ کیا، اور اسی پر بھروسہ کرنے والوں کو بھروسہ کرنا چاہیے۔
وَلَمَّا دَخَلُوا۟ مِنْ حَيْثُ أَمَرَهُمْ أَبُوهُم مَّا كَانَ يُغْنِي عَنْهُم مِّنَ ٱللَّهِ مِن شَيْءٍ إِلَّا حَاجَةٗ فِي نَفْسِ يَعْقُوبَ قَضَىٰهَاۚ وَإِنَّهُۥ لَذُو عِلْمٖ لِّمَا عَلَّمْنَـٰهُ وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَعْلَمُونَ ﴿68﴾
اور جب وہ اس راستے سے داخل ہوئے جس کا ان کے باپ نے حکم دیا تھا، وہ اللہ کے مقابلے میں ان کے لیے کچھ نہیں کر سکتا تھا، مگر یعقوب کے دل کی ایک خواہش تھی جسے اس نے پورا کیا، اور بے شک وہ علم والا تھا اس چیز کا جو ہم نے اسے سکھائی تھی، لیکن اکثر لوگ نہیں جانتے۔
وَلَمَّا دَخَلُوا۟ عَلَىٰ يُوسُفَ ءَاوَىٰٓ إِلَيْهِ أَخَاهُ قَالَ إِنِّيٓ أَنَا۠ أَخُوكَ فَلَا تَبْتَئِسْ بِمَا كَانُوا۟ يَعْمَلُونَ ﴿69﴾
اور جب وہ یوسف کے پاس داخل ہوئے تو اس نے اپنے بھائی کو اپنے پاس جگہ دی، اور کہا: بے شک میں ہی تیرا بھائی ہوں، پس ان کے کیے پر غم نہ کر۔
فَلَمَّا جَهَّزَهُم بِجَهَازِهِمْ جَعَلَ ٱلسِّقَايَةَ فِي رَحْلِ أَخِيهِ ثُمَّ أَذَّنَ مُؤَذِّنٌ أَيَّتُهَا ٱلْعِيرُ إِنَّكُمْ لَسَـٰرِقُونَ ﴿70﴾
پھر جب اس نے ان کا سامان تیار کر دیا تو اپنے بھائی کے سامان میں پیالہ رکھ دیا، پھر ایک منادی نے آواز دی: اے قافلہ والو! تم چور ہو۔
قَالُوا۟ وَأَقْبَلُوا۟ عَلَيْهِم مَّاذَا تَفْقِدُونَ ﴿71﴾
انہوں نے ان کی طرف رخ کر کے کہا: تم کیا چیز گم پاتے ہو؟
قَالُوا۟ نَفْقِدُ صُوَاعَ ٱلْمَلِكِ وَلِمَن جَآءَ بِهِۦ حِمْلُ بَعِيرٖ وَأَنَا۠ بِهِۦ زَعِيمٞ ﴿72﴾
انہوں نے کہا: ہم بادشاہ کا پیالہ گم پاتے ہیں، اور جو اسے لائے گا اسے ایک اونٹ کا بوجھ انعام دیا جائے گا، اور میں اس کا ضامن ہوں۔
قَالُوا۟ تَٱللَّهِ لَقَدْ عَلِمْتُم مَّا جِئْنَا لِنُفْسِدَ فِي ٱلْأَرْضِ وَمَا كُنَّا سَـٰرِقِينَ ﴿73﴾
انہوں نے کہا: اللہ کی قسم! تم جانتے ہو کہ ہم زمین میں فساد کرنے نہیں آئے اور نہ ہم چور ہیں۔
قَالُوا۟ فَمَا جَزَٰٓؤُهُۥٓ إِن كُنتُمْ كَـٰذِبِينَ ﴿74﴾
انہوں نے کہا: اگر تم جھوٹے ہو تو اس کی سزا کیا ہے؟
قَالُوا۟ جَزَٰٓؤُهُۥ مَن وُجِدَ فِي رَحْلِهِۦ فَهُوَ جَزَٰٓؤُهُۚ كَذَٰلِكَ نَجْزِي ٱلظَّـٰلِمِينَ ﴿75﴾
انہوں نے کہا: جس کے سامان میں وہ چیز ملے، وہی اس کی سزا ہے، ہم ظالموں کو اسی طرح سزا دیتے ہیں۔
فَبَدَأَ بِأَوْعِيَتِهِمْ قَبْلَ وِعَآءِ أَخِيهِ ثُمَّ ٱسْتَخْرَجَهَا مِن وِعَآءِ أَخِيهِۚ كَذَٰلِكَ كِدْنَا لِيُوسُفَۚ مَا كَانَ لِيَأْخُذَ أَخَاهُ فِي دِينِ ٱلْمَلِكِ إِلَّآ أَن يَشَآءَ ٱللَّهُۚ نَرْفَعُ دَرَجَـٰتٖ مَّن نَّشَآءُۗ وَفَوْقَ كُلِّ ذِي عِلْمٍ عَلِيمٞ ﴿76﴾
پھر یوسف نے ان کے سامان کی تلاشی اپنے بھائی کے سامان سے پہلے لی، پھر اسے اپنے بھائی کے سامان سے نکال لیا، ہم نے یوسف کے لیے اسی طرح تدبیر کی، وہ بادشاہ کے قانون کے مطابق اپنے بھائی کو نہیں رکھ سکتا تھا مگر یہ کہ اللہ چاہے، ہم جسے چاہیں درجے بلند کرتے ہیں، اور ہر علم والے سے اوپر ایک علم والا ہے۔
قَالُوٓا۟ إِن يَسْرِقْ فَقَدْ سَرَقَ أَخٗ لَّهُۥ مِن قَبْلُۖ فَأَسَرَّهَا يُوسُفُ فِي نَفْسِهِۦ وَلَمْ يُبْدِهَا لَهُمْۚ قَالَ أَنتُمْ شَرّٞ مَّكَانٗاۖ وَٱللَّهُ أَعْلَمُ بِمَا تَصِفُونَ ﴿77﴾
انہوں نے کہا: اگر اس نے چوری کی ہے تو اس کے بھائی نے بھی پہلے چوری کی تھی، یوسف نے یہ بات اپنے دل میں چھپا لی اور انہیں ظاہر نہ کی، کہا: تم بدتر ہو، اور اللہ خوب جانتا ہے جو تم بیان کرتے ہو۔
قَالُوا۟ يَـٰٓأَيُّهَا ٱلْعَزِيزُ إِنَّ لَهُۥٓ أَبٗا شَيْخٗا كَبِيرٗا فَخُذْ أَحَدَنَا مَكَانَهُۥٓ إِنَّا نَرَىٰكَ مِنَ ٱلْمُحْسِنِينَ ﴿78﴾
انہوں نے کہا: اے عزیز! اس کا ایک بوڑھا باپ ہے، پس اس کی جگہ ہم میں سے کسی کو لے لو، بے شک ہم تمہیں نیکوکاروں میں سے دیکھتے ہیں۔
قَالَ مَعَاذَ ٱللَّهِ أَن نَّأْخُذَ إِلَّا مَن وَجَدْنَا مَتَـٰعَنَا عِندَهُۥٓ إِنَّآ إِذٗا لَّظَـٰلِمُونَ ﴿79﴾
یوسف نے کہا: اللہ کی پناہ! ہم اس کے سوا کسی کو نہیں پکڑ سکتے جس کے پاس ہم نے اپنا سامان پایا، اگر ہم ایسا کریں تو ہم ظالم ہوں گے۔
فَلَمَّا ٱسْتَيْـَٔسُوا۟ مِنْهُ خَلَصُوا۟ نَجِيّٗاۖ قَالَ كَبِيرُهُمْ أَلَمْ تَعْلَمُوٓا۟ أَنَّ أَبَاكُمْ قَدْ أَخَذَ عَلَيْكُم مَّوْثِقٗا مِّنَ ٱللَّهِ وَمِن قَبْلُ مَا فَرَّطتُّم فِي يُوسُفَۖ فَلَنْ أَبْرَحَ ٱلْأَرْضَ حَتَّىٰ يَأْذَنَ لِيٓ أَبِيٓ أَوْ يَحْكُمَ ٱللَّهُ لِيۖ وَهُوَ خَيْرُ ٱلْحَـٰكِمِينَ ﴿80﴾
پھر جب وہ اس سے مایوس ہو گئے تو الگ ہو کر مشورہ کرنے لگے، ان میں سے بڑے نے کہا: کیا تمہیں معلوم نہیں کہ تمہارے باپ نے تم سے اللہ کا عہد لیا تھا، اور اس سے پہلے یوسف کے معاملے میں تم کوتاہی کر چکے ہو؟ پس میں تو اس زمین سے ہرگز نہ جاؤں گا جب تک کہ میرا باپ مجھے اجازت نہ دے یا اللہ میرے لیے کوئی فیصلہ کرے، اور وہ سب سے بہتر فیصلہ کرنے والا ہے۔
ٱرْجِعُوٓا۟ إِلَىٰٓ أَبِيكُمْ فَقُولُوا۟ يَـٰٓأَبَانَآ إِنَّ ٱبْنَكَ سَرَقَ وَمَا شَهِدْنَآ إِلَّا بِمَا عَلِمْنَا وَمَا كُنَّا لِلْغَيْبِ حَـٰفِظِينَ ﴿81﴾
تم اپنے باپ کے پاس واپس جاؤ اور کہو: اے ہمارے باپ! آپ کے بیٹے نے چوری کی ہے، اور ہم نے وہی گواہی دی جو ہمیں معلوم تھی، اور ہم غیب کے نگہبان نہ تھے۔
وَسْـَٔلِ ٱلْقَرْيَةَ ٱلَّتِي كُنَّا فِيهَا وَٱلْعِيرَ ٱلَّتِيٓ أَقْبَلْنَا فِيهَاۖ وَإِنَّا لَصَـٰدِقُونَ ﴿82﴾
اور اس بستی سے پوچھ لیجیے جہاں ہم تھے، اور اس قافلے سے بھی جس میں ہم آئے ہیں، اور بے شک ہم سچے ہیں۔
قَالَ بَلْ سَوَّلَتْ لَكُمْ أَنفُسُكُمْ أَمْرٗاۖ فَصَبْرٞ جَمِيلٌۖ عَسَى ٱللَّهُ أَن يَأْتِيَنِي بِهِمْ جَمِيعًاۚ إِنَّهُۥ هُوَ ٱلْعَلِيمُ ٱلْحَكِيمُ ﴿83﴾
اس نے کہا: بلکہ تمہارے نفس نے تمہیں کوئی بات سجھائی ہے، پس صبر ہی بہتر ہے، شاید اللہ ان سب کو میرے پاس لے آئے، بے شک وہی علم والا، حکمت والا ہے۔
وَتَوَلَّىٰ عَنْهُمْ وَقَالَ يَـٰٓأَسَفَىٰ عَلَىٰ يُوسُفَ وَٱبْيَضَّتْ عَيْنَاهُ مِنَ ٱلْحُزْنِ فَهُوَ كَظِيمٞ ﴿84﴾
اور وہ ان سے منہ موڑ کر کہنے لگا: ہائے افسوس یوسف پر! اور اس کی آنکھیں غم سے سفید ہو گئیں، اور وہ غم کو چھپائے ہوئے تھا۔
قَالُوا۟ تَٱللَّهِ تَفْتَأُ تَذْكُرُ يُوسُفَ حَتَّىٰ تَكُونَ حَرَضًا أَوْ تَكُونَ مِنَ ٱلْهَـٰلِكِينَ ﴿85﴾
انہوں نے کہا: اللہ کی قسم! آپ یوسف کو یاد کرتے ہی رہیں گے یہاں تک کہ آپ بیمار ہو جائیں یا ہلاک ہو جائیں۔
قَالَ إِنَّمَآ أَشْكُوا۟ بَثِّي وَحُزْنِيٓ إِلَى ٱللَّهِ وَأَعْلَمُ مِنَ ٱللَّهِ مَا لَا تَعْلَمُونَ ﴿86﴾
اس نے کہا: میں تو اپنا غم اور دکھ اللہ ہی سے بیان کرتا ہوں، اور میں اللہ کی طرف سے وہ جانتا ہوں جو تم نہیں جانتے۔
يَـٰبَنِيَّ ٱذْهَبُوا۟ فَتَحَسَّسُوا۟ مِن يُوسُفَ وَأَخِيهِ وَلَا تَيْـَٔسُوا۟ مِن رَّوْحِ ٱللَّهِۖ إِنَّهُۥ لَا يَيْـَٔسُ مِن رَّوْحِ ٱللَّهِ إِلَّا ٱلْقَوْمُ ٱلْكَـٰفِرُونَ ﴿87﴾
اے میرے بیٹو! جاؤ یوسف اور اس کے بھائی کی تلاش کرو، اور اللہ کی رحمت سے ناامید نہ ہو، بے شک اللہ کی رحمت سے صرف کافر ہی ناامید ہوتے ہیں۔
فَلَمَّا دَخَلُوا۟ عَلَيْهِ قَالُوا۟ يَـٰٓأَيُّهَا ٱلْعَزِيزُ مَسَّنَا وَأَهْلَنَا ٱلضُّرُّ وَجِئْنَا بِبِضَـٰعَةٖ مُّزْجَىٰةٖ فَأَوْفِ لَنَا ٱلْكَيْلَ وَتَصَدَّقْ عَلَيْنَآۖ إِنَّ ٱللَّهَ يَجْزِي ٱلْمُتَصَدِّقِينَ ﴿88﴾
پھر جب وہ یوسف کے پاس داخل ہوئے تو کہا: اے عزیز! ہمیں اور ہمارے گھر والوں کو سختی پہنچی ہے، اور ہم تھوڑا سا مال لے کر آئے ہیں، پس ہمیں پورا ناپ تول دو اور ہم پر صدقہ کرو، بے شک اللہ صدقہ کرنے والوں کو جزا دیتا ہے۔
قَالَ هَلْ عَلِمْتُم مَّا فَعَلْتُم بِيُوسُفَ وَأَخِيهِ إِذْ أَنتُمْ جَـٰهِلُونَ ﴿89﴾
یوسف نے کہا: کیا تم جانتے ہو کہ تم نے یوسف اور اس کے بھائی کے ساتھ کیا کیا تھا جب تم جاہل تھے؟
قَالُوٓا۟ أَإِنَّكَ لَأَنتَ يُوسُفُۖ قَالَ أَنَا۠ يُوسُفُ وَهَـٰذَآ أَخِيۖ قَدْ مَنَّ ٱللَّهُ عَلَيْنَآۖ إِنَّهُۥ مَن يَتَّقِ وَيَصْبِرْ فَإِنَّ ٱللَّهَ لَا يُضِيعُ أَجْرَ ٱلْمُحْسِنِينَ ﴿90﴾
انہوں نے کہا: کیا واقعی تم یوسف ہو؟ اس نے کہا: میں یوسف ہوں اور یہ میرا بھائی ہے، اللہ نے ہم پر فضل فرمایا، بے شک جو تقویٰ اختیار کرے اور صبر کرے تو اللہ نیکوکاروں کا اجر ضائع نہیں کرتا۔
قَالُوا۟ تَٱللَّهِ لَقَدْ ءَاثَرَكَ ٱللَّهُ عَلَيْنَا وَإِن كُنَّا لَخَـٰطِـِٔينَ ﴿91﴾
انہوں نے کہا: اللہ کی قسم! اللہ نے تمہیں ہم پر فضیلت دی، اور بے شک ہم خطا کار تھے۔
قَالَ لَا تَثْرِيبَ عَلَيْكُمُ ٱلْيَوْمَۖ يَغْفِرُ ٱللَّهُ لَكُمْۖ وَهُوَ أَرْحَمُ ٱلرَّٰحِمِينَ ﴿92﴾
یوسف نے کہا: آج تم پر کوئی ملامت نہیں، اللہ تمہیں معاف کرے، اور وہ سب سے زیادہ رحم کرنے والا ہے۔
ٱذْهَبُوا۟ بِقَمِيصِي هَـٰذَا فَأَلْقُوهُ عَلَىٰ وَجْهِ أَبِي يَأْتِ بَصِيرٗاۖ وَأْتُونِي بِأَهْلِكُمْ أَجْمَعِينَ ﴿93﴾
میرا یہ قمیص لے جاؤ اور میرے باپ کے چہرے پر ڈال دو، وہ بینا ہو جائے گا، اور اپنے سب گھر والوں کو میرے پاس لے آؤ۔
وَلَمَّا فَصَلَتِ ٱلْعِيرُ قَالَ أَبُوهُمْ إِنِّي لَأَجِدُ رِيحَ يُوسُفَ لَوْلَآ أَن تُفَنِّدُونِ ﴿94﴾
اور جب قافلہ روانہ ہوا تو ان کے باپ نے کہا: مجھے یوسف کی خوشبو آ رہی ہے، اگر تم مجھے بے عقل نہ سمجھو۔
قَالُوا۟ تَٱللَّهِ إِنَّكَ لَفِي ضَلَـٰلِكَ ٱلْقَدِيمِ ﴿95﴾
انہوں نے کہا: اللہ کی قسم! آپ اپنی پرانی گمراہی میں ہیں۔
فَلَمَّآ أَن جَآءَ ٱلْبَشِيرُ أَلْقَىٰهُ عَلَىٰ وَجْهِهِۦ فَٱرْتَدَّ بَصِيرٗاۖ قَالَ أَلَمْ أَقُلْ لَكُمْ إِنِّيٓ أَعْلَمُ مِنَ ٱللَّهِ مَا لَا تَعْلَمُونَ ﴿96﴾
پھر جب خوشخبری دینے والا آیا اور قمیص یعقوب کے چہرے پر ڈال دیا تو وہ بینا ہو گیا، کہا: کیا میں نے تم سے نہیں کہا تھا کہ میں اللہ کی طرف سے وہ جانتا ہوں جو تم نہیں جانتے؟
قَالُوا۟ يَـٰٓأَبَانَا ٱسْتَغْفِرْ لَنَا ذُنُوبَنَآ إِنَّا كُنَّا خَـٰطِـِٔينَ ﴿97﴾
انہوں نے کہا: اے ہمارے باپ! ہمارے گناہوں کی معافی مانگیں، بے شک ہم خطا کار تھے۔
قَالَ سَوْفَ أَسْتَغْفِرُ لَكُمْ رَبِّيٓۖ إِنَّهُۥ هُوَ ٱلْغَفُورُ ٱلرَّحِيمُ ﴿98﴾
اس نے کہا: میں جلد ہی اپنے رب سے تمہارے لیے معافی مانگوں گا، بے شک وہی بخشنے والا، رحم کرنے والا ہے۔
فَلَمَّا دَخَلُوا۟ عَلَىٰ يُوسُفَ ءَاوَىٰٓ إِلَيْهِ أَبَوَيْهِ وَقَالَ ٱدْخُلُوا۟ مِصْرَ إِن شَآءَ ٱللَّهُ ءَامِنِينَ ﴿99﴾
پھر جب وہ یوسف کے پاس داخل ہوئے تو اس نے اپنے والدین کو اپنے پاس جگہ دی، اور کہا: مصر میں داخل ہو جاؤ، اگر اللہ نے چاہا تو امن سے۔
وَرَفَعَ أَبَوَيْهِ عَلَى ٱلْعَرْشِ وَخَرُّوا۟ لَهُۥ سُجَّدٗاۖ وَقَالَ يَـٰٓأَبَتِ هَـٰذَا تَأْوِيلُ رُءْيَـٰيَ مِن قَبْلُۖ قَدْ جَعَلَهَا رَبِّي حَقّٗاۖ وَقَدْ أَحْسَنَ بِيٓ إِذْ أَخْرَجَنِي مِنَ ٱلسِّجْنِ وَجَآءَ بِكُم مِّنَ ٱلْبَدْوِ مِنۢ بَعْدِ أَن نَّزَغَ ٱلشَّيْطَـٰنُ بَيْنِي وَبَيْنَ إِخْوَتِيٓۚ إِنَّ رَبِّي لَطِيفٞ لِّمَا يَشَآءُۚ إِنَّهُۥ هُوَ ٱلْعَلِيمُ ٱلْحَكِيمُ ﴿100﴾
اور اس نے اپنے والدین کو تخت پر بٹھایا، اور سب اس کے لیے سجدہ میں گر پڑے، اور یوسف نے کہا: اے میرے باپ! یہ میرے پہلے خواب کی تعبیر ہے، میرے رب نے اسے سچ کر دکھایا، اور اس نے مجھے قید سے نکالا، اور آپ سب کو دیہات سے لے آیا، اس کے بعد کہ شیطان نے میرے اور میرے بھائیوں کے درمیان فساد ڈالا، بے شک میرا رب جسے چاہے تدبیر سے نوازتا ہے، بے شک وہی علم والا، حکمت والا ہے۔
رَبِّ قَدْ ءَاتَيْتَنِي مِنَ ٱلْمُلْكِ وَعَلَّمْتَنِي مِن تَأْوِيلِ ٱلْأَحَادِيثِۚ فَاطِرَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ أَنتَ وَلِيِّي فِي ٱلدُّنْيَا وَٱلْـَٔاخِرَةِۖ تَوَفَّنِي مُسْلِمٗا وَأَلْحِقْنِي بِٱلصَّـٰلِحِينَ ﴿101﴾
اے میرے رب! تو نے مجھے بادشاہی عطا کی، اور مجھے خوابوں کی تعبیر سکھائی، اے آسمانوں اور زمین کے پیدا کرنے والے! تو ہی دنیا اور آخرت میں میرا ولی ہے، مجھے مسلمان ہونے کی حالت میں وفات دے اور مجھے نیک لوگوں کے ساتھ ملا دے۔
ذَٰلِكَ مِنْ أَنۢبَآءِ ٱلْغَيْبِ نُوحِيهِ إِلَيْكَۚ وَمَا كُنتَ لَدَيْهِمْ إِذْ أَجْمَعُوٓا۟ أَمْرَهُمْ وَهُمْ يَمْكُرُونَ ﴿102﴾
یہ غیب کی خبریں ہیں جو ہم تمہیں وحی کے ذریعے بتاتے ہیں، اور تم ان کے پاس نہ تھے جب انہوں نے اپنا معاملہ طے کیا اور وہ چالیں چل رہے تھے۔
وَمَآ أَكْثَرُ ٱلنَّاسِ وَلَوْ حَرَصْتَ بِمُؤْمِنِينَ ﴿103﴾
اور اکثر لوگ ایمان نہیں لائیں گے، چاہے تم کتنی ہی کوشش کر لو۔
وَمَا تَسْـَٔلُهُمْ عَلَيْهِ مِنْ أَجْرٍۚ إِنْ هُوَ إِلَّا ذِكْرٞ لِّلْعَـٰلَمِينَ ﴿104﴾
اور تم ان سے اس (قرآن) پر کوئی اجر نہیں مانگتے، یہ تو صرف تمام جہانوں کے لیے نصیحت ہے۔
وَكَأَيِّن مِّنْ ءَايَةٖ فِي ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ يَمُرُّونَ عَلَيْهَا وَهُمْ عَنْهَا مُعْرِضُونَ ﴿105﴾
اور کتنی ہی نشانیاں آسمانوں اور زمین میں ہیں جن پر وہ گزرتے ہیں اور ان سے منہ موڑ لیتے ہیں۔
وَمَا يُؤْمِنُ أَكْثَرُهُم بِٱللَّهِ إِلَّا وَهُم مُّشْرِكُونَ ﴿106﴾
اور ان میں سے اکثر اللہ پر ایمان نہیں رکھتے مگر اس حال میں کہ وہ شرک کرتے ہیں۔
أَفَأَمِنُوٓا۟ أَن تَأْتِيَهُمْ غَٰشِيَةٞ مِّنْ عَذَابِ ٱللَّهِ أَوْ تَأْتِيَهُمُ ٱلسَّاعَةُ بَغْتَةٗ وَهُمْ لَا يَشْعُرُونَ ﴿107﴾
کیا وہ اس بات سے بے خوف ہو گئے ہیں کہ ان پر اللہ کا عذاب چھا جائے یا قیامت اچانک آ جائے اور انہیں خبر بھی نہ ہو؟
قُلْ هَـٰذِهِۦ سَبِيلِيٓ أَدْعُوٓا۟ إِلَى ٱللَّهِۚ عَلَىٰ بَصِيرَةٍ أَنَا۠ وَمَنِ ٱتَّبَعَنِيۖ وَسُبْحَـٰنَ ٱللَّهِ وَمَآ أَنَا۠ مِنَ ٱلْمُشْرِكِينَ ﴿108﴾
کہہ دو: یہ میرا راستہ ہے، میں اللہ کی طرف بلاتا ہوں بصیرت کے ساتھ، میں اور وہ لوگ جو میری پیروی کرتے ہیں، اور اللہ پاک ہے، اور میں مشرکوں میں سے نہیں ہوں۔
وَمَآ أَرْسَلْنَا مِن قَبْلِكَ إِلَّا رِجَالٗا نُّوحِيٓ إِلَيْهِم مِّنْ أَهْلِ ٱلْقُرَىٰٓۚ أَفَلَمْ يَسِيرُوا۟ فِي ٱلْأَرْضِ فَيَنظُرُوا۟ كَيْفَ كَانَ عَـٰقِبَةُ ٱلَّذِينَ مِن قَبْلِهِمۖ وَلَدَارُ ٱلْـَٔاخِرَةِ خَيْرٞ لِّلَّذِينَ ٱتَّقَوْا۟ۚ أَفَلَا تَعْقِلُونَ ﴿109﴾
اور ہم نے تم سے پہلے صرف مردوں کو ہی بھیجا جن پر ہم نے وحی کی، جو بستیوں کے رہنے والے تھے، کیا انہوں نے زمین میں سفر نہیں کیا تاکہ دیکھیں کہ ان سے پہلے لوگوں کا انجام کیا ہوا؟ اور آخرت کا گھر ان لوگوں کے لیے بہتر ہے جو تقویٰ اختیار کرتے ہیں، کیا تم عقل نہیں رکھتے؟
حَتَّىٰٓ إِذَا ٱسْتَيْـَٔسَ ٱلرُّسُلُ وَظَنُّوٓا۟ أَنَّهُمْ قَدْ كُذِبُوا۟ جَآءَهُمْ نَصْرُنَا فَنُجِّيَ مَن نَّشَآءُۖ وَلَا يُرَدُّ بَأْسُنَا عَنِ ٱلْقَوْمِ ٱلْمُجْرِمِينَ ﴿110﴾
یہاں تک کہ جب رسول مایوس ہو گئے اور انہوں نے گمان کیا کہ ان سے جھوٹ بولا گیا، تو ہماری مدد آ گئی، پھر ہم جسے چاہیں بچا لیتے ہیں، اور ہمارا عذاب مجرم قوم سے ٹالا نہیں جاتا۔
لَقَدْ كَانَ فِي قَصَصِهِمْ عِبْرَةٞ لِّأُوْلِي ٱلْأَلْبَـٰبِۗ مَا كَانَ حَدِيثٗا يُفْتَرَىٰ وَلَـٰكِن تَصْدِيقَ ٱلَّذِي بَيْنَ يَدَيْهِ وَتَفْصِيلَ كُلِّ شَيْءٖ وَهُدٗى وَرَحْمَةٗ لِّقَوْمٖ يُؤْمِنُونَ ﴿111﴾
بے شک ان کے قصوں میں عقل والوں کے لیے عبرت ہے، یہ کوئی گھڑی ہوئی بات نہیں بلکہ اس سے پہلے کی کتابوں کی تصدیق ہے، اور ہر چیز کی تفصیل، اور ایمان والوں کے لیے ہدایت اور رحمت ہے۔
سورة الرعد
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
الٓمۚ تِلْكَ ءَايَاتُ ٱلْكِتَـٰبِۗ وَٱلَّذِيٓ أُنزِلَ إِلَيْكَ مِن رَّبِّكَ ٱلْحَقُّ وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يُؤْمِنُونَ ﴿1﴾
الٓم۔ یہ کتاب کی آیات ہیں، اور جو کچھ تمہارے رب کی طرف سے تم پر نازل کیا گیا ہے وہ حق ہے، لیکن اکثر لوگ ایمان نہیں لاتے۔
ٱللَّهُ ٱلَّذِي رَفَعَ ٱلسَّمَـٰوَٰتِ بِغَيْرِ عَمَدٖ تَرَوْنَهَا ثُمَّ ٱسْتَوَىٰ عَلَى ٱلْعَرْشِ وَسَخَّرَ ٱلشَّمْسَ وَٱلْقَمَرَۖ كُلّٞ يَجْرِي لِأَجَلٖ مُّسَمّٗىۚ يُدَبِّرُ ٱلْأَمْرَ يُفَصِّلُ ٱلْـَٔايَـٰتِ لَعَلَّكُم بِلِقَآءِ رَبِّكُمْ تُوقِنُونَ ﴿2﴾
اللہ وہ ہے جس نے آسمانوں کو بغیر ستونوں کے بلند کیا جنہیں تم دیکھ رہے ہو، پھر وہ عرش پر قائم ہوا، اور سورج اور چاند کو مسخر کیا، ہر ایک مقررہ وقت تک چل رہا ہے، وہ ہر کام کی تدبیر کرتا ہے، نشانیاں کھول کر بیان کرتا ہے تاکہ تم اپنے رب سے ملاقات کا یقین کرو۔
وَهُوَ ٱلَّذِي مَدَّ ٱلْأَرْضَ وَجَعَلَ فِيهَا رَوَٰسِيَ وَأَنْهَـٰرٗاۖ وَمِن كُلِّ ٱلثَّمَرَٰتِ جَعَلَ فِيهَا زَوْجَيْنِ ٱثْنَيْنِۖ يُغْشِي ٱلَّيْلَ ٱلنَّهَارَۚ إِنَّ فِي ذَٰلِكَ لَـَٔايَـٰتٖ لِّقَوْمٖ يَتَفَكَّرُونَ ﴿3﴾
اور وہی ہے جس نے زمین کو پھیلایا، اور اس میں پہاڑ اور نہریں بنائیں، اور اس میں ہر قسم کے پھلوں کے جوڑے پیدا کیے، وہ رات کو دن سے ڈھانپ دیتا ہے، بے شک اس میں غور و فکر کرنے والوں کے لیے نشانیاں ہیں۔
وَفِي ٱلْأَرْضِ قِطَعٞ مُّتَجَـٰوِرَٰتٞ وَجَنَّـٰتٞ مِّنْ أَعْنَـٰبٖ وَزَرْعٞ وَنَخِيلٞ صِنْوَانٞ وَغَيْرُ صِنْوَانٖ يُسْقَىٰ بِمَآءٖ وَٰحِدٖ وَنُفَضِّلُ بَعْضَهَا عَلَىٰ بَعْضٖ فِي ٱلْأُكُلِۚ إِنَّ فِي ذَٰلِكَ لَـَٔايَـٰتٖ لِّقَوْمٖ يَعْقِلُونَ ﴿4﴾
اور زمین میں ایک دوسرے سے ملے ہوئے ٹکڑے ہیں، اور انگوروں کے باغات، اور کھیتیاں، اور کھجور کے درخت، کچھ جڑ سے جڑے ہوئے اور کچھ الگ، سب کو ایک ہی پانی سے سیراب کیا جاتا ہے، اور ہم بعض کو بعض پر پھل میں فضیلت دیتے ہیں، بے شک اس میں عقل والوں کے لیے نشانیاں ہیں۔
وَإِن تَعْجَبْ فَعَجَبٌ قَوْلُهُمْ أَءِذَا كُنَّا تُرَٰبًا أَءِنَّا لَفِي خَلْقٖ جَدِيدٍۗ أُوْلَـٰٓئِكَ ٱلَّذِينَ كَفَرُوا۟ بِرَبِّهِمْۖ وَأُوْلَـٰٓئِكَ ٱلْأَغْلَـٰلُ فِيٓ أَعْنَاقِهِمْۖ وَأُوْلَـٰٓئِكَ أَصْحَـٰبُ ٱلنَّارِۖ هُمْ فِيهَا خَـٰلِدُونَ ﴿5﴾
اور اگر تم تعجب کرو تو تعجب انگیز ان کا یہ کہنا ہے: کیا جب ہم مٹی ہو جائیں گے تو کیا ہم نئے سرے سے پیدا کیے جائیں گے؟ یہی وہ لوگ ہیں جنہوں نے اپنے رب سے کفر کیا، اور یہی وہ ہیں جن کی گردنوں میں طوق ہوں گے، اور یہی دوزخ والے ہیں، وہ اس میں ہمیشہ رہیں گے۔
وَيَسْتَعْجِلُونَكَ بِٱلسَّيِّئَةِ قَبْلَ ٱلْحَسَنَةِ وَقَدْ خَلَتْ مِن قَبْلِهِمُ ٱلْمَثُلَـٰتُۗ وَإِنَّ رَبَّكَ لَذُو مَغْفِرَةٖ لِّلنَّاسِ عَلَىٰ ظُلْمِهِمْۖ وَإِنَّ رَبَّكَ لَشَدِيدُ ٱلْعِقَابِ ﴿6﴾
اور وہ تم سے برائی کو نیکی سے پہلے جلدی مانگتے ہیں، حالانکہ ان سے پہلے کئی مثالیں گزر چکی ہیں، اور بے شک تمہارا رب لوگوں کے ظلم کے باوجود بخشنے والا ہے، اور بے شک تمہارا رب سخت سزا دینے والا بھی ہے۔
وَيَقُولُ ٱلَّذِينَ كَفَرُوا۟ لَوْلَآ أُنزِلَ عَلَيْهِ ءَايَةٞ مِّن رَّبِّهِۦٓۚ إِنَّمَآ أَنتَ مُنذِرٞ وَلِكُلِّ قَوْمٍ هَادٍ ﴿7﴾
اور کافر کہتے ہیں: اس پر اس کے رب کی طرف سے کوئی نشانی کیوں نہیں اتاری گئی؟ تم تو صرف ڈرانے والے ہو، اور ہر قوم کے لیے ایک رہنما ہے۔
ٱللَّهُ يَعْلَمُ مَا تَحْمِلُ كُلُّ أُنثَىٰ وَمَا تَغِيضُ ٱلْأَرْحَامُ وَمَا تَزْدَادُۚ وَكُلُّ شَيْءٍ عِندَهُۥ بِمِقْدَارٍ ﴿8﴾
اللہ جانتا ہے جو کچھ ہر مادہ اٹھاتی ہے، اور رحم میں جو کمی یا زیادتی ہوتی ہے، اور ہر چیز اس کے ہاں ایک اندازے سے ہے۔
عَـٰلِمُ ٱلْغَيْبِ وَٱلشَّهَـٰدَةِ ٱلْكَبِيرُ ٱلْمُتَعَالِ ﴿9﴾
وہ غیب اور ظاہر کا جاننے والا ہے، بڑا ہے، بلند مرتبہ ہے۔
سَوَآءٞ مِّنكُم مَّنْ أَسَرَّ ٱلْقَوْلَ وَمَن جَهَرَ بِهِۦ وَمَنْ هُوَ مُسْتَخْفٖۢ بِٱلَّيْلِ وَسَارِبُۢ بِٱلنَّهَارِ ﴿10﴾
تم میں سے جو بات چھپاتا ہے اور جو ظاہر کرتا ہے، اور جو رات کو چھپ کر چلتا ہے اور جو دن میں ظاہر ہو کر چلتا ہے — سب برابر ہیں۔
لَهُۥ مُعَقِّبَـٰتٞ مِّنۢ بَيْنِ يَدَيْهِ وَمِنْ خَلْفِهِۦ يَحْفَظُونَهُۥ مِنْ أَمْرِ ٱللَّهِۗ إِنَّ ٱللَّهَ لَا يُغَيِّرُ مَا بِقَوْمٍ حَتَّىٰ يُغَيِّرُوا۟ مَا بِأَنفُسِهِمْۗ وَإِذَآ أَرَادَ ٱللَّهُ بِقَوْمٖ سُوٓءٗا فَلَا مَرَدَّ لَهُۥۚ وَمَا لَهُم مِّن دُونِهِۦ مِن وَالٍ ﴿11﴾
اس کے لیے فرشتے مقرر ہیں، جو اس کے آگے اور پیچھے سے اللہ کے حکم سے اس کی حفاظت کرتے ہیں۔ بے شک اللہ کسی قوم کی حالت نہیں بدلتا جب تک وہ خود اپنی حالت نہ بدلیں۔ اور جب اللہ کسی قوم کے ساتھ برائی کا ارادہ کرتا ہے تو اسے کوئی ٹالنے والا نہیں، اور اس کے سوا ان کا کوئی مددگار نہیں۔
هُوَ ٱلَّذِي يُرِيكُمُ ٱلْبَرْقَ خَوْفٗا وَطَمَعٗا وَيُنشِئُ ٱلسَّحَابَ ٱلثِّقَالَ ﴿12﴾
وہی ہے جو تمہیں بجلی دکھاتا ہے، خوف اور امید کے لیے، اور بھاری بادل پیدا کرتا ہے۔
وَيُسَبِّحُ ٱلرَّعْدُ بِحَمْدِهِۦ وَٱلْمَلَـٰٓئِكَةُ مِنْ خِيفَتِهِۦ وَيُرْسِلُ ٱلصَّوَٰعِقَ فَيُصِيبُ بِهَا مَن يَشَآءُ وَهُمْ يُجَـٰدِلُونَ فِي ٱللَّهِ وَهُوَ شَدِيدُ ٱلْمِحَالِ ﴿13﴾
اور گرج اس کی حمد کے ساتھ تسبیح کرتی ہے، اور فرشتے اس کے خوف سے، اور وہ بجلیاں بھیجتا ہے، پھر جسے چاہتا ہے ان سے مار دیتا ہے، اور وہ اللہ کے بارے میں جھگڑتے ہیں، حالانکہ وہ سخت پکڑ والا ہے۔
لَهُۥ دَعْوَةُ ٱلْحَقِّۖ وَٱلَّذِينَ يَدْعُونَ مِن دُونِهِۦ لَا يَسْتَجِيبُونَ لَهُم بِشَيْءٍ إِلَّا كَبَاسِطِ كَفَّيْهِ إِلَى ٱلْمَآءِ لِيَبْلُغَ فَاهُۥ وَمَا هُوَ بِبَـٰلِغِهِۦۚ وَمَا دُعَآءُ ٱلْكَـٰفِرِينَ إِلَّا فِي ضَلَـٰلٖ ﴿14﴾
اسی کے لیے حق کی دعا ہے، اور جو لوگ اس کے سوا دوسروں کو پکارتے ہیں وہ ان کی دعا قبول نہیں کرتے، مگر جیسے کوئی پانی کی طرف ہاتھ پھیلائے کہ وہ اس کے منہ تک پہنچ جائے، حالانکہ وہ اسے نہیں پہنچتا، اور کافروں کی دعا صرف گمراہی میں ہے۔
وَلِلَّهِ يَسْجُدُ مَن فِي ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ طَوْعٗا وَكَرْهٗا وَظِلَـٰلُهُم بِٱلْغُدُوِّ وَٱلْأٓصَالِ ﴿15﴾
اور اللہ ہی کے لیے سجدہ کرتے ہیں جو آسمانوں اور زمین میں ہیں، خوشی سے اور مجبوری سے، اور ان کے سائے بھی صبح و شام جھکتے ہیں۔
قُلْ مَن رَّبُّ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۖ قُلِ ٱللَّهُۚ قُلْ أَفَٱتَّخَذْتُم مِّن دُونِهِۦٓ أَوْلِيَآءَ لَا يَمْلِكُونَ لِأَنفُسِهِمْ نَفْعٗا وَلَا ضَرّٗاۚ قُلْ هَلْ يَسْتَوِي ٱلْأَعْمَىٰ وَٱلْبَصِيرُ أَمْ هَلْ تَسْتَوِي ٱلظُّلُمَـٰتُ وَٱلنُّورُۗ أَمْ جَعَلُوا۟ لِلَّهِ شُرَكَآءَ خَلَقُوا۟ كَخَلْقِهِۦ فَتَشَـٰبَهَ ٱلْخَلْقُ عَلَيْهِمۚ قُلِ ٱللَّهُ خَـٰلِقُ كُلِّ شَيْءٖ وَهُوَ ٱلْوَٰحِدُ ٱلْقَهَّارُ ﴿16﴾
کہہ دو: آسمانوں اور زمین کا رب کون ہے؟ کہہ دو: اللہ۔ کہہ دو: کیا تم نے اس کے سوا ایسے مددگار بنا لیے ہیں جو اپنے لیے نہ نفع کے مالک ہیں نہ نقصان کے؟ کہہ دو: کیا اندھا اور دیکھنے والا برابر ہو سکتے ہیں؟ یا کیا اندھیرے اور روشنی برابر ہو سکتے ہیں؟ یا کیا انہوں نے اللہ کے لیے شریک بنائے ہیں جنہوں نے اس کی طرح پیدا کیا ہو، تو ان پر پیدا کرنا مشتبہ ہو گیا ہو؟ کہہ دو: اللہ ہر چیز کا خالق ہے، اور وہی واحد، زبردست ہے۔
أَنزَلَ مِنَ ٱلسَّمَآءِ مَآءٗ فَسَالَتْ أَوْدِيَةُۢ بِقَدَرِهَا فَٱحْتَمَلَ ٱلسَّيْلُ زَبَدٗا رَّابِيٗاۖ وَمِمَّا يُوقِدُونَ عَلَيْهِ فِي ٱلنَّارِ ٱبْتِغَآءَ حِلْيَةٍ أَوْ مَتَـٰعٖ زَبَدٞ مِّثْلُهُۥۚ كَذَٰلِكَ يَضْرِبُ ٱللَّهُ ٱلْحَقَّ وَٱلْبَـٰطِلَۚ فَأَمَّا ٱلزَّبَدُ فَيَذْهَبُ جُفَآءٗۖ وَأَمَّا مَا يَنفَعُ ٱلنَّاسَ فَيَمْكُثُ فِي ٱلْأَرْضِۚ كَذَٰلِكَ يَضْرِبُ ٱللَّهُ ٱلْأَمْثَالَ ﴿17﴾
اس نے آسمان سے پانی نازل کیا، تو وادیاں اپنی گنجائش کے مطابق بہہ نکلیں، پھر سیلاب نے ابھرا ہوا جھاگ اٹھایا، اور جس چیز کو وہ آگ پر پگھلاتے ہیں زیور یا سامان بنانے کے لیے، اس پر بھی ایسا ہی جھاگ ہوتا ہے۔ اسی طرح اللہ حق اور باطل کی مثال بیان کرتا ہے۔ پس جو جھاگ ہے وہ تو بہہ کر ختم ہو جاتا ہے، اور جو لوگوں کو نفع دیتا ہے وہ زمین میں باقی رہتا ہے۔ اسی طرح اللہ مثالیں بیان کرتا ہے۔
لِلَّذِينَ ٱسْتَجَابُوا۟ لِرَبِّهِمُ ٱلْحُسْنَىٰۚ وَٱلَّذِينَ لَمْ يَسْتَجِيبُوا۟ لَهُۥ لَوْ أَنَّ لَهُم مَّا فِي ٱلْأَرْضِ جَمِيعٗا وَمِثْلَهُۥ مَعَهُۥ لَٱفْتَدَوْا۟ بِهِۦٓۚ أُوْلَـٰٓئِكَ لَهُمْ سُوٓءُ ٱلْحِسَابِۖ وَمَأْوَىٰهُمْ جَهَنَّمُۖ وَبِئْسَ ٱلْمِهَادُ ﴿18﴾
جنہوں نے اپنے رب کی بات مان لی، ان کے لیے بھلائی ہے۔ اور جنہوں نے اس کی بات نہ مانی، اگر ان کے پاس زمین کی ساری چیزیں ہوں اور ان کے ساتھ اتنی ہی اور بھی ہو، تو وہ ضرور فدیہ میں دے دیں گے۔ یہی لوگ ہیں جن کے لیے برا حساب ہے، اور ان کا ٹھکانا جہنم ہے، اور وہ بہت ہی برا ٹھکانا ہے۔
أَفَمَن يَعْلَمُ أَنَّمَآ أُنزِلَ إِلَيْكَ مِن رَّبِّكَ ٱلْحَقُّ كَمَنْ هُوَ أَعْمَىٰٓۚ إِنَّمَا يَتَذَكَّرُ أُوْلُوا۟ ٱلْأَلْبَـٰبِ ﴿19﴾
تو کیا وہ شخص جو جانتا ہے کہ جو کچھ تمہارے رب کی طرف سے نازل ہوا ہے وہ حق ہے، اس شخص کی طرح ہو سکتا ہے جو اندھا ہے؟ نصیحت تو عقل والے ہی حاصل کرتے ہیں۔
ٱلَّذِينَ يُوفُونَ بِعَهْدِ ٱللَّهِ وَلَا يَنقُضُونَ ٱلْمِيثَـٰقَ ﴿20﴾
جو اللہ کے عہد کو پورا کرتے ہیں اور وعدہ کو نہیں توڑتے۔
وَٱلَّذِينَ يَصِلُونَ مَآ أَمَرَ ٱللَّهُ بِهِۦٓ أَن يُوصَلَ وَيَخْشَوْنَ رَبَّهُمْ وَيَخَافُونَ سُوٓءَ ٱلْحِسَابِ ﴿21﴾
اور جو ان چیزوں کو جوڑتے ہیں جن کے جوڑنے کا اللہ نے حکم دیا ہے، اور اپنے رب سے ڈرتے ہیں، اور برے حساب سے خوف کھاتے ہیں۔
وَٱلَّذِينَ صَبَرُوا۟ ٱبْتِغَآءَ وَجْهِ رَبِّهِمْ وَأَقَامُوا۟ ٱلصَّلَوٰةَ وَأَنفَقُوا۟ مِمَّا رَزَقْنَـٰهُمْ سِرّٗا وَعَلَانِيَةٗ وَيَدْرَءُونَ بِٱلْحَسَنَةِ ٱلسَّيِّئَةَ أُوْلَـٰٓئِكَ لَهُمْ عُقْبَى ٱلدَّارِ ﴿22﴾
اور جو اپنے رب کی رضا کے لیے صبر کرتے ہیں، اور نماز قائم کرتے ہیں، اور ہم نے انہیں جو رزق دیا ہے اس میں سے چھپ کر اور ظاہر طور پر خرچ کرتے ہیں، اور برائی کو بھلائی سے دور کرتے ہیں — یہی لوگ ہیں جن کے لیے آخرت کا اچھا انجام ہے۔
جَنَّـٰتُ عَدْنٖ يَدْخُلُونَهَا وَمَن صَلَحَ مِنْ ءَابَآئِهِمْ وَأَزْوَٰجِهِمْ وَذُرِّيَّـٰتِهِمْۖ وَٱلْمَلَـٰٓئِكَةُ يَدْخُلُونَ عَلَيْهِم مِّن كُلِّ بَابٖ ﴿23﴾
ہمیشہ رہنے والے باغات جن میں وہ داخل ہوں گے، اور ان کے نیک باپ دادا، بیویاں اور اولاد بھی، اور فرشتے ان پر ہر دروازے سے داخل ہوں گے۔
سَلَـٰمٌ عَلَيْكُم بِمَا صَبَرْتُمْۚ فَنِعْمَ عُقْبَى ٱلدَّارِ ﴿24﴾
سلام ہو تم پر اس صبر کے بدلے جو تم نے کیا، پس کیا ہی خوب ہے آخرت کا انجام۔
وَٱلَّذِينَ يَنقُضُونَ عَهْدَ ٱللَّهِ مِنۢ بَعْدِ مِيثَـٰقِهِۦ وَيَقْطَعُونَ مَآ أَمَرَ ٱللَّهُ بِهِۦٓ أَن يُوصَلَ وَيُفْسِدُونَ فِي ٱلْأَرْضِ أُوْلَـٰٓئِكَ لَهُمُ ٱللَّعْنَةُ وَلَهُمْ سُوٓءُ ٱلدَّارِ ﴿25﴾
اور جو اللہ کے عہد کو اس کے پختہ کرنے کے بعد توڑتے ہیں، اور ان چیزوں کو کاٹتے ہیں جن کے جوڑنے کا اللہ نے حکم دیا ہے، اور زمین میں فساد کرتے ہیں — یہی لوگ ہیں جن پر لعنت ہے، اور ان کے لیے برا ٹھکانا ہے۔
ٱللَّهُ يَبْسُطُ ٱلرِّزْقَ لِمَن يَشَآءُ وَيَقْدِرُۚ وَفَرِحُوا۟ بِٱلْحَيَوٰةِ ٱلدُّنْيَاۖ وَمَا ٱلْحَيَوٰةُ ٱلدُّنْيَا فِي ٱلْـَٔاخِرَةِ إِلَّا مَتَـٰعٞ ﴿26﴾
اللہ جس کے لیے چاہتا ہے رزق کشادہ کرتا ہے اور تنگ بھی کرتا ہے، اور وہ دنیا کی زندگی پر خوش ہو گئے، حالانکہ دنیا کی زندگی آخرت کے مقابلے میں صرف ایک سامان ہے۔
وَيَقُولُ ٱلَّذِينَ كَفَرُوا۟ لَوْلَآ أُنزِلَ عَلَيْهِ ءَايَةٞ مِّن رَّبِّهِۦۚ قُلْ إِنَّ ٱللَّهَ يُضِلُّ مَن يَشَآءُ وَيَهْدِيٓ إِلَيْهِ مَنْ أَنَابَ ﴿27﴾
اور کافر کہتے ہیں: اس پر اس کے رب کی طرف سے کوئی نشانی کیوں نہیں اتاری گئی؟ کہہ دو: بے شک اللہ جسے چاہتا ہے گمراہ کرتا ہے، اور اپنی طرف اس کو ہدایت دیتا ہے جو رجوع کرتا ہے۔
ٱلَّذِينَ ءَامَنُوا۟ وَتَطْمَئِنُّ قُلُوبُهُم بِذِكْرِ ٱللَّهِۗ أَلَا بِذِكْرِ ٱللَّهِ تَطْمَئِنُّ ٱلْقُلُوبُ ﴿28﴾
وہ لوگ جو ایمان لائے اور ان کے دل اللہ کے ذکر سے مطمئن ہوتے ہیں، خبردار! اللہ کے ذکر سے ہی دلوں کو اطمینان حاصل ہوتا ہے۔
ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ طُوبَىٰ لَهُمْ وَحُسْنُ مَـَٔابٖ ﴿29﴾
جو لوگ ایمان لائے اور نیک عمل کیے، ان کے لیے خوشخبری ہے اور بہترین ٹھکانا ہے۔
كَذَٰلِكَ أَرْسَلْنَـٰكَ فِيٓ أُمَّةٖ قَدْ خَلَتْ مِن قَبْلِهَآ أُمَمٞ لِّتَتْلُوَا۟ عَلَيْهِمُ ٱلَّذِيٓ أَوْحَيْنَآ إِلَيْكَ وَهُمْ يَكْفُرُونَ بِٱلرَّحْمَـٰنِۚ قُلْ هُوَ رَبِّي لَآ إِلَـٰهَ إِلَّا هُوَۖ عَلَيْهِ تَوَكَّلْتُ وَإِلَيْهِ مَتَابِ ﴿30﴾
اسی طرح ہم نے تمہیں ایک ایسی امت میں بھیجا ہے جس سے پہلے کئی امتیں گزر چکی ہیں، تاکہ تم ان پر وہ چیز پڑھ کر سناؤ جو ہم نے تمہاری طرف وحی کی ہے، اور وہ رحمٰن کا انکار کرتے ہیں۔ کہہ دو: وہی میرا رب ہے، اس کے سوا کوئی معبود نہیں، اسی پر میں نے بھروسہ کیا، اور اسی کی طرف میری رجوع ہے۔
وَلَوْ أَنَّ قُرْءَانٗا سُيِّرَتْ بِهِ ٱلْجِبَالُ أَوْ قُطِّعَتْ بِهِ ٱلْأَرْضُ أَوْ كُلِّمَ بِهِ ٱلْمَوْتَىٰۗ بَل لِّلَّهِ ٱلْأَمْرُ جَمِيعًاۗ أَفَلَمْ يَاْيْـَٔسِ ٱلَّذِينَ ءَامَنُوٓا۟ أَن لَّوْ يَشَآءُ ٱللَّهُ لَهَدَى ٱلنَّاسَ جَمِيعٗاۗ وَلَا يَزَالُ ٱلَّذِينَ كَفَرُوا۟ تُصِيبُهُم بِمَا صَنَعُوا۟ قَارِعَةٌ أَوْ تَحُلُّ قَرِيبٗا مِّن دَارِهِمْ حَتَّىٰ يَأْتِيَ وَعْدُ ٱللَّهِۚ إِنَّ ٱللَّهَ لَا يُخْلِفُ ٱلْمِيعَادَ ﴿31﴾
اور اگر کوئی قرآن ایسا ہوتا کہ اس کے ذریعے پہاڑ چلائے جاتے یا زمین کے ٹکڑے کیے جاتے یا مردوں سے بات کی جاتی (تو یہی ہوتا)، بلکہ سب کام اللہ کے اختیار میں ہیں۔ کیا ایمان والوں کو یقین نہیں ہوا کہ اگر اللہ چاہتا تو سب لوگوں کو ہدایت دے دیتا؟ اور کافروں کو ان کے اعمال کے بدلے کوئی نہ کوئی آفت پہنچتی رہے گی یا ان کے گھروں کے قریب اترے گی، یہاں تک کہ اللہ کا وعدہ آ پہنچے۔ بے شک اللہ وعدہ خلافی نہیں کرتا۔
وَلَقَدِ ٱسْتُهْزِئَ بِرُسُلٖ مِّن قَبْلِكَ فَأَمْلَيْتُ لِلَّذِينَ كَفَرُوا۟ ثُمَّ أَخَذْتُهُمْۖ فَكَيْفَ كَانَ عِقَابِ ﴿32﴾
اور تم سے پہلے بھی رسولوں کا مذاق اڑایا گیا، تو میں نے کافروں کو مہلت دی، پھر انہیں پکڑ لیا، تو کیسا تھا میرا عذاب؟
أَفَمَنْ هُوَ قَآئِمٌ عَلَىٰ كُلِّ نَفْسِۢ بِمَا كَسَبَتْۗ وَجَعَلُوا۟ لِلَّهِ شُرَكَآءَۚ قُلْ سَمُّوهُمْۚ أَمْ تُنَبِّـُٔونَهُۥ بِمَا لَا يَعْلَمُ فِي ٱلْأَرْضِ أَم بِظَـٰهِرٖ مِّنَ ٱلْقَوْلِۗ بَلْ زُيِّنَ لِلَّذِينَ كَفَرُوا۟ مَكْرُهُمْ وَصُدُّوا۟ عَنِ ٱلسَّبِيلِۗ وَمَن يُضْلِلِ ٱللَّهُ فَمَا لَهُۥ مِنْ هَادٍ ﴿33﴾
تو کیا وہ جو ہر جان کے اعمال پر نگران ہے (اللہ)، اس جیسا ہو سکتا ہے؟ اور انہوں نے اللہ کے لیے شریک بنا لیے، کہہ دو: ان کے نام لو! یا کیا تم اللہ کو ایسی بات کی خبر دیتے ہو جو وہ زمین میں نہیں جانتا؟ یا تم صرف ظاہری باتیں کرتے ہو؟ بلکہ کافروں کے لیے ان کی چالیں خوشنما بنا دی گئیں، اور وہ راہِ راست سے روک دیے گئے، اور جسے اللہ گمراہ کرے، اس کا کوئی ہادی نہیں۔
لَّهُمْ عَذَابٞ فِي ٱلْحَيَوٰةِ ٱلدُّنْيَاۖ وَلَعَذَابُ ٱلْـَٔاخِرَةِ أَشَقُّۖ وَمَا لَهُم مِّنَ ٱللَّهِ مِن وَاقٍ ﴿34﴾
ان کے لیے دنیا کی زندگی میں عذاب ہے، اور آخرت کا عذاب تو بہت ہی سخت ہے، اور ان کے لیے اللہ سے بچانے والا کوئی نہیں۔
مَّثَلُ ٱلْجَنَّةِ ٱلَّتِي وُعِدَ ٱلْمُتَّقُونَۖ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُۖ أُكُلُهَا دَآئِمٞ وَظِلُّهَاۚ تِلْكَ عُقْبَى ٱلَّذِينَ ٱتَّقَوا۟ۖ وَعُقْبَى ٱلْكَـٰفِرِينَ ٱلنَّارُ ﴿35﴾
اس جنت کی مثال جس کا وعدہ پرہیزگاروں سے کیا گیا ہے: اس کے نیچے نہریں بہتی ہیں، اس کا پھل ہمیشہ باقی ہے اور اس کا سایہ بھی۔ یہی انجام ہے ان لوگوں کا جو تقویٰ اختیار کرتے ہیں، اور کافروں کا انجام آگ ہے۔
وَٱلَّذِينَ ءَاتَيْنَـٰهُمُ ٱلْكِتَـٰبَ يَفْرَحُونَ بِمَآ أُنزِلَ إِلَيْكَۖ وَمِنَ ٱلْأَحْزَابِ مَن يُنكِرُ بَعْضَهُۥۚ قُلْ إِنَّمَآ أُمِرْتُ أَنْ أَعْبُدَ ٱللَّهَ وَلَآ أُشْرِكَ بِهِۦٓۚ إِلَيْهِ أَدْعُوا۟ وَإِلَيْهِ مَـَٔابِ ﴿36﴾
اور جنہیں ہم نے کتاب دی ہے وہ اس چیز پر خوش ہوتے ہیں جو تم پر نازل کی گئی، اور گروہوں میں سے کچھ ایسے ہیں جو اس کا کچھ انکار کرتے ہیں۔ کہہ دو: مجھے تو یہی حکم دیا گیا ہے کہ میں اللہ کی عبادت کروں اور اس کے ساتھ کسی کو شریک نہ بناؤں، اسی کی طرف میں بلاتا ہوں اور اسی کی طرف میرا لوٹنا ہے۔
وَكَذَٰلِكَ أَنزَلْنَـٰهُ حُكْمًا عَرَبِيّٗاۚ وَلَئِنِ ٱتَّبَعْتَ أَهْوَآءَهُم بَعْدَ مَا جَآءَكَ مِنَ ٱلْعِلْمِ مَا لَكَ مِنَ ٱللَّهِ مِن وَلِيّٖ وَلَا وَاقٖ ﴿37﴾
اور اسی طرح ہم نے اسے عربی فیصلہ کے طور پر نازل کیا، اور اگر تم ان کی خواہشات کی پیروی کرو گے اس علم کے بعد جو تمہارے پاس آ چکا ہے، تو تمہارے لیے اللہ کے مقابلے میں نہ کوئی دوست ہوگا اور نہ کوئی بچانے والا۔
وَلَقَدْ أَرْسَلْنَا رُسُلٗا مِّن قَبْلِكَ وَجَعَلْنَا لَهُمْ أَزْوَٰجٗا وَذُرِّيَّةٗۚ وَمَا كَانَ لِرَسُولٍ أَن يَأْتِيَ بِـَٔايَةٍ إِلَّا بِإِذْنِ ٱللَّهِۗ لِكُلِّ أَجَلٍ كِتَابٞ ﴿38﴾
اور ہم نے تم سے پہلے بھی رسول بھیجے اور ان کے لیے بیویاں اور اولاد بنائی، اور کسی رسول کے لیے ممکن نہیں کہ وہ کوئی نشانی لائے مگر اللہ کے حکم سے، ہر وقت کے لیے ایک نوشتہ ہے۔
يَمْحُوا۟ ٱللَّهُ مَا يَشَآءُ وَيُثْبِتُۖ وَعِندَهُۥٓ أُمُّ ٱلْكِتَـٰبِ ﴿39﴾
اللہ جو چاہے مٹا دیتا ہے اور باقی رکھتا ہے، اور اس کے پاس اصل کتاب ہے۔
وَإِن مَّا نُرِيَنَّكَ بَعْضَ ٱلَّذِي نَعِدُهُمْ أَوْ نَتَوَفَّيَنَّكَ فَإِنَّمَا عَلَيْكَ ٱلْبَلَـٰغُ وَعَلَيْنَا ٱلْحِسَابُ ﴿40﴾
اور اگر ہم تمہیں وہ کچھ دکھا دیں جس کا ہم ان سے وعدہ کر رہے ہیں یا تمہیں وفات دے دیں، تو تمہارا کام صرف پہنچا دینا ہے، اور حساب لینا ہمارا کام ہے۔
أَوَلَمْ يَرَوْا۟ أَنَّا نَأْتِي ٱلْأَرْضَ نَنقُصُهَا مِنْ أَطْرَافِهَاۚ وَٱللَّهُ يَحْكُمُ لَا مُعَقِّبَ لِحُكْمِهِۦۚ وَهُوَ سَرِيعُ ٱلْحِسَابِ ﴿41﴾
کیا انہوں نے نہیں دیکھا کہ ہم زمین کو اس کے کناروں سے کم کرتے جا رہے ہیں؟ اور اللہ فیصلہ کرتا ہے، اس کے فیصلے کو کوئی پلٹانے والا نہیں، اور وہ جلد حساب لینے والا ہے۔
وَقَدْ مَكَرَ ٱلَّذِينَ مِن قَبْلِهِمْ فَلِلَّهِ ٱلْمَكْرُ جَمِيعٞۖ يَعْلَمُ مَا تَكْسِبُ كُلُّ نَفْسٖۚ وَسَيَعْلَمُ ٱلْكُفَّـٰرُ لِمَنْ عُقْبَى ٱلدَّارِ ﴿42﴾
اور ان سے پہلے بھی لوگوں نے چالیں چلیں، تو تمام چالیں اللہ ہی کے اختیار میں ہیں، وہ جانتا ہے جو ہر جان کماتی ہے، اور کافر جلد جان لیں گے کہ انجام کس کا بہتر ہے۔
وَيَقُولُ ٱلَّذِينَ كَفَرُوا۟ لَسْتَ مُرْسَلٗاۚ قُلْ كَفَىٰ بِٱللَّهِ شَهِيدٗا بَيْنِي وَبَيْنَكُمْ وَمَنْ عِندَهُۥ عِلْمُ ٱلْكِتَـٰبِ ﴿43﴾
اور کافر کہتے ہیں: تم رسول نہیں ہو۔ کہہ دو: میرے اور تمہارے درمیان اللہ اور وہ شخص گواہ کافی ہے جس کے پاس کتاب کا علم ہے۔
سورة إبراهيم
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
الٓرۚ كِتَـٰبٌ أَنزَلْنَـٰهُ إِلَيْكَ لِتُخْرِجَ ٱلنَّاسَ مِنَ ٱلظُّلُمَـٰتِ إِلَى ٱلنُّورِ بِإِذْنِ رَبِّهِمْ إِلَىٰ صِرَٰطِ ٱلْعَزِيزِ ٱلْحَمِيدِ ﴿1﴾
الٓر۔ یہ ایک کتاب ہے جسے ہم نے تمہاری طرف نازل کیا تاکہ تم لوگوں کو ان کے رب کے حکم سے اندھیروں سے نکال کر روشنی کی طرف لے آؤ، اس زبردست، قابلِ تعریف ذات کے راستے کی طرف۔
ٱللَّهِ ٱلَّذِي لَهُۥ مَا فِي ٱلسَّمَـٰوَٰتِ وَمَا فِي ٱلْأَرْضِۗ وَوَيْلٞ لِّلْكَـٰفِرِينَ مِنْ عَذَابٖ شَدِيدٍ ﴿2﴾
اللہ ہی کا ہے جو کچھ آسمانوں اور زمین میں ہے، اور کافروں کے لیے سخت عذاب ہے۔
ٱلَّذِينَ يَسْتَحِبُّونَ ٱلْحَيَوٰةَ ٱلدُّنْيَا عَلَى ٱلْـَٔاخِرَةِ وَيَصُدُّونَ عَن سَبِيلِ ٱللَّهِ وَيَبْغُونَهَا عِوَجًاۚ أُوْلَـٰٓئِكَ فِي ضَلَـٰلِۢ بَعِيدٖ ﴿3﴾
جو دنیا کی زندگی کو آخرت پر ترجیح دیتے ہیں، اور اللہ کے راستے سے روکتے ہیں اور اس میں کجی چاہتے ہیں، یہی لوگ دور کی گمراہی میں ہیں۔
وَمَآ أَرْسَلْنَا مِن رَّسُولٍ إِلَّا بِلِسَانِ قَوْمِهِۦ لِيُبَيِّنَ لَهُمْۖ فَيُضِلُّ ٱللَّهُ مَن يَشَآءُ وَيَهْدِي مَن يَشَآءُۚ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿4﴾
اور ہم نے کوئی رسول نہیں بھیجا مگر اپنی قوم کی زبان میں تاکہ وہ ان کے لیے واضح کرے، پھر اللہ جسے چاہتا ہے گمراہ کرتا ہے اور جسے چاہتا ہے ہدایت دیتا ہے، اور وہ زبردست، حکمت والا ہے۔
وَلَقَدْ أَرْسَلْنَا مُوسَىٰ بِـَٔايَـٰتِنَآ أَنْ أَخْرِجْ قَوْمَكَ مِنَ ٱلظُّلُمَـٰتِ إِلَى ٱلنُّورِ وَذَكِّرْهُم بِأَيَّامِ ٱللَّهِۚ إِنَّ فِي ذَٰلِكَ لَـَٔايَـٰتٖ لِّكُلِّ صَبَّارٖ شَكُورٖ ﴿5﴾
اور بے شک ہم نے موسیٰ کو اپنی نشانیاں دے کر بھیجا کہ اپنی قوم کو اندھیروں سے نکال کر روشنی کی طرف لے آؤ، اور انہیں اللہ کے دن یاد دلاؤ، بے شک اس میں ہر صبر کرنے والے، شکر گزار کے لیے نشانیاں ہیں۔
وَإِذْ قَالَ مُوسَىٰ لِقَوْمِهِ ٱذْكُرُوا۟ نِعْمَتَ ٱللَّهِ عَلَيْكُمْ إِذْ أَنجَىٰكُم مِّنْ ءَالِ فِرْعَوْنَ يَسُومُونَكُمْ سُوٓءَ ٱلْعَذَابِ وَيُذَبِّحُونَ أَبْنَآءَكُمْ وَيَسْتَحْيُونَ نِسَآءَكُمْۚ وَفِي ذَٰلِكُم بَلَآءٞ مِّن رَّبِّكُمْ عَظِيمٞ ﴿6﴾
اور جب موسیٰ نے اپنی قوم سے کہا: اللہ کی نعمت کو یاد کرو جو اس نے تم پر کی، جب اس نے تمہیں فرعون کی قوم سے نجات دی، وہ تمہیں سخت عذاب دیتے تھے، تمہارے بیٹوں کو ذبح کرتے تھے اور تمہاری عورتوں کو زندہ رکھتے تھے، اور اس میں تمہارے رب کی طرف سے بڑی آزمائش تھی۔
وَإِذْ تَأَذَّنَ رَبُّكُمْ لَئِن شَكَرْتُمْ لَأَزِيدَنَّكُمْ وَلَئِن كَفَرْتُمْ إِنَّ عَذَابِي لَشَدِيدٞ ﴿7﴾
اور جب تمہارے رب نے اعلان کیا: اگر تم شکر کرو گے تو میں تمہیں ضرور زیادہ دوں گا، اور اگر تم ناشکری کرو گے تو بے شک میرا عذاب سخت ہے۔
وَقَالَ مُوسَىٰٓ إِن تَكْفُرُوا۟ أَنتُمْ وَمَن فِي ٱلْأَرْضِ جَمِيعٗا فَإِنَّ ٱللَّهَ لَغَنِيٌّ حَمِيدٌ ﴿8﴾
اور موسیٰ نے کہا: اگر تم اور زمین میں جو بھی ہیں سب کے سب کفر کرو تو بے شک اللہ بے نیاز، قابلِ تعریف ہے۔
أَلَمْ يَأْتِكُمْ نَبَؤُا۟ ٱلَّذِينَ مِن قَبْلِكُمْ قَوْمِ نُوحٖ وَعَادٖ وَثَمُودَۚ وَٱلَّذِينَ مِنۢ بَعْدِهِمْۛ لَا يَعْلَمُهُمْ إِلَّا ٱللَّهُۚ جَآءَتْهُمْ رُسُلُهُم بِٱلْبَيِّنَـٰتِ فَرَدُّوٓا۟ أَيْدِيَهُمْ فِيٓ أَفْوَٰهِهِمْ وَقَالُوٓا۟ إِنَّا كَفَرْنَا بِمَآ أُرْسِلْتُم بِهِۦ وَإِنَّا لَفِي شَكّٖ مِّمَّا تَدْعُونَنَآ إِلَيْهِ مُرِيبٖ ﴿9﴾
کیا تمہارے پاس ان لوگوں کی خبر نہیں آئی جو تم سے پہلے تھے — نوح کی قوم، عاد، ثمود، اور ان کے بعد کے لوگ جنہیں اللہ کے سوا کوئی نہیں جانتا؟ ان کے پاس ان کے رسول واضح دلائل لے کر آئے، تو انہوں نے اپنے ہاتھ اپنے منہ پر رکھ لیے اور کہا: ہم اس چیز کے منکر ہیں جس کے ساتھ تم بھیجے گئے ہو، اور جس چیز کی طرف تم ہمیں بلا رہے ہو، ہم اس میں سخت شک میں ہیں۔
قَالَتْ رُسُلُهُمْ أَفِي ٱللَّهِ شَكّٞ فَاطِرِ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۖ يَدْعُوكُمْ لِيَغْفِرَ لَكُم مِّن ذُنُوبِكُمْ وَيُؤَخِّرَكُمْ إِلَىٰٓ أَجَلٖ مُّسَمًّىۚ قَالُوٓا۟ إِنْ أَنتُمْ إِلَّا بَشَرٞ مِّثْلُنَا تُرِيدُونَ أَن تَصُدُّونَا عَمَّا كَانَ يَعْبُدُ ءَابَآؤُنَا فَأْتُونَا بِسُلْطَـٰنٖ مُّبِينٖ ﴿10﴾
ان کے رسولوں نے کہا: کیا اللہ میں شک ہے؟ جو آسمانوں اور زمین کا پیدا کرنے والا ہے؟ وہ تمہیں بلاتا ہے تاکہ تمہارے گناہ بخش دے اور تمہیں ایک مقرر وقت تک مہلت دے۔ انہوں نے کہا: تم تو ہمارے جیسے انسان ہو، تم چاہتے ہو کہ ہمیں اس سے روک دو جس کی ہمارے باپ دادا عبادت کرتے تھے، پس ہمارے پاس کوئی واضح دلیل لے آؤ۔
قَالَتْ لَهُمْ رُسُلُهُمْ إِن نَّحْنُ إِلَّا بَشَرٞ مِّثْلُكُمْ وَلَـٰكِنَّ ٱللَّهَ يَمُنُّ عَلَىٰ مَن يَشَآءُ مِنْ عِبَادِهِۦۖ وَمَا كَانَ لَنَآ أَن نَّأْتِيَكُم بِسُلْطَـٰنٍ إِلَّا بِإِذْنِ ٱللَّهِۚ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُؤْمِنُونَ ﴿11﴾
ان کے رسولوں نے کہا: بے شک ہم تمہارے جیسے انسان ہی ہیں، لیکن اللہ اپنے بندوں میں سے جس پر چاہتا ہے فضل کرتا ہے، اور ہمارے لیے ممکن نہیں کہ ہم تمہارے پاس کوئی دلیل لائیں مگر اللہ کے حکم سے، اور ایمان والوں کو اللہ ہی پر بھروسہ کرنا چاہیے۔
وَمَا لَنَآ أَلَّا نَتَوَكَّلَ عَلَى ٱللَّهِ وَقَدْ هَدَىٰنَا سُبُلَنَاۚ وَلَنَصْبِرَنَّ عَلَىٰ مَآ ءَاذَيْتُمُونَاۚ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُتَوَكِّلُونَ ﴿12﴾
اور ہمیں کیا ہوا ہے کہ ہم اللہ پر بھروسہ نہ کریں جبکہ اس نے ہمیں ہمارے راستے دکھا دیے ہیں؟ اور ہم ضرور صبر کریں گے اس اذیت پر جو تم ہمیں دیتے ہو، اور بھروسہ کرنے والوں کو اللہ ہی پر بھروسہ کرنا چاہیے۔
وَقَالَ ٱلَّذِينَ كَفَرُوا۟ لِرُسُلِهِمْ لَنُخْرِجَنَّكُم مِّنْ أَرْضِنَآ أَوْ لَتَعُودُنَّ فِي مِلَّتِنَاۖ فَأَوْحَىٰٓ إِلَيْهِمْ رَبُّهُمْ لَنُهْلِكَنَّ ٱلظَّـٰلِمِينَ ﴿13﴾
اور کافروں نے اپنے رسولوں سے کہا: ہم تمہیں اپنی زمین سے ضرور نکال دیں گے یا تم ضرور ہمارے دین میں واپس آ جاؤ گے، تو ان کے رب نے ان کی طرف وحی کی: ہم ضرور ظالموں کو ہلاک کریں گے۔
وَلَنُسْكِنَنَّكُمُ ٱلْأَرْضَ مِنۢ بَعْدِهِمۡۚ ذَٰلِكَ لِمَنْ خَافَ مَقَامِي وَخَافَ وَعِيدِ ﴿14﴾
اور ہم تمہیں ان کے بعد زمین میں ضرور بسائیں گے، یہ اس کے لیے ہے جو میرے مقام سے ڈرے اور میری وعید سے ڈرے۔
وَٱسْتَفْتَحُوا۟ وَخَابَ كُلُّ جَبَّارٍ عَنِيدٍ ﴿15﴾
اور انہوں نے فیصلہ چاہا، تو ہر سرکش ضدی ناکام ہوا۔
مِّن وَرَآئِهِۦ جَهَنَّمُ وَيُسْقَىٰ مِن مَّآءٍ صَدِيدٖ ﴿16﴾
اس کے پیچھے جہنم ہے، اور اسے پیپ کا پانی پلایا جائے گا۔
يَتَجَرَّعُهُۥ وَلَا يَكَادُ يُسِيغُهُۥ وَيَأْتِيهِ ٱلْمَوْتُ مِن كُلِّ مَكَانٖ وَمَا هُوَ بِمَيِّتٖۖ وَمِن وَرَآئِهِۦ عَذَابٌ غَلِيظٞ ﴿17﴾
وہ اسے گھونٹ گھونٹ کر پئے گا اور اسے آسانی سے نگل نہ سکے گا، اور اسے ہر طرف سے موت آئے گی مگر وہ مرے گا نہیں، اور اس کے پیچھے سخت عذاب ہے۔
مَّثَلُ ٱلَّذِينَ كَفَرُوا۟ بِرَبِّهِمْ أَعْمَـٰلُهُمْ كَرَمَادٍ ٱشْتَدَّتْ بِهِ ٱلرِّيحُ فِي يَوْمٍ عَاصِفٖۖ لَّا يَقْدِرُونَ مِمَّا كَسَبُوا۟ عَلَىٰ شَيْءٖۚ ذَٰلِكَ هُوَ ٱلضَّلَـٰلُ ٱلْبَعِيدُ ﴿18﴾
جن لوگوں نے اپنے رب سے کفر کیا، ان کے اعمال کی مثال اس راکھ کی طرح ہے جسے آندھی والے دن تیز ہوا اڑا لے جائے، وہ اپنی کمائی میں سے کچھ بھی حاصل نہ کر سکیں گے، یہی دور کی گمراہی ہے۔
أَلَمْ تَرَ أَنَّ ٱللَّهَ خَلَقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ بِٱلْحَقِّۚ إِن يَشَأْ يُذْهِبْكُمْ وَيَأْتِ بِخَلْقٍ جَدِيدٖ ﴿19﴾
کیا تم نے نہیں دیکھا کہ اللہ نے آسمانوں اور زمین کو حق کے ساتھ پیدا کیا؟ اگر وہ چاہے تو تمہیں لے جائے اور نئی مخلوق لے آئے۔
وَمَا ذَٰلِكَ عَلَى ٱللَّهِ بِعَزِيزٖ ﴿20﴾
اور یہ اللہ کے لیے کچھ بھی مشکل نہیں۔
وَبَرَزُوا۟ لِلَّهِ جَمِيعٗا فَقَالَ ٱلضُّعَفَـٰٓؤُا۟ لِلَّذِينَ ٱسْتَكْبَرُوا۟ إِنَّا كُنَّا لَكُمْ تَبَعٗا فَهَلْ أَنتُم مُّغْنُونَ عَنَّا مِنْ عَذَابِ ٱللَّهِ مِن شَيْءٖۚ قَالُوا۟ لَوْ هَدَىٰنَا ٱللَّهُ لَهَدَيْنَـٰكُمۚ سَوَآءٌ عَلَيْنَآ أَجَزِعْنَآ أَمْ صَبَرْنَا مَا لَنَا مِن مَّحِيصٖ ﴿21﴾
اور سب اللہ کے سامنے ظاہر ہوں گے، تو کمزور لوگ ان سے کہیں گے جو تکبر کرتے تھے: ہم تمہارے تابع تھے، تو کیا تم اللہ کے عذاب سے کچھ بچا سکتے ہو؟ وہ کہیں گے: اگر اللہ ہمیں ہدایت دیتا تو ہم تمہیں بھی ہدایت دیتے، اب ہمارے لیے برابر ہے کہ ہم جزع کریں یا صبر کریں، ہمارے لیے کوئی بچاؤ نہیں۔
وَقَالَ ٱلشَّيْطَـٰنُ لَمَّا قُضِيَ ٱلْأَمْرُ إِنَّ ٱللَّهَ وَعَدَكُمْ وَعْدَ ٱلْحَقِّ وَوَعَدتُّكُمْ فَأَخْلَفْتُكُمْۖ وَمَا كَانَ لِيَ عَلَيْكُم مِّن سُلْطَـٰنٍ إِلَّآ أَن دَعَوْتُكُمْ فَٱسْتَجَبْتُمْ لِيۖ فَلَا تَلُومُونِي وَلُومُوٓا۟ أَنفُسَكُمۖ مَّآ أَنَا۠ بِمُصْرِخِكُمْ وَمَآ أَنتُم بِمُصْرِخِيَّۖ إِنِّي كَفَرْتُ بِمَآ أَشْرَكْتُمُونِ مِن قَبْلُۗ إِنَّ ٱلظَّـٰلِمِينَ لَهُمْ عَذَابٌ أَلِيمٞ ﴿22﴾
اور جب فیصلہ ہو چکے گا تو شیطان کہے گا: بے شک اللہ نے تم سے سچا وعدہ کیا تھا، اور میں نے بھی وعدہ کیا تھا، تو میں نے تم سے وعدہ خلافی کی، اور مجھے تم پر کوئی زور حاصل نہ تھا، سوائے اس کے کہ میں نے تمہیں بلایا اور تم نے میری بات مان لی، پس مجھے ملامت نہ کرو بلکہ اپنے آپ کو ملامت کرو، نہ میں تمہاری فریاد کو پہنچ سکتا ہوں اور نہ تم میری فریاد کو، میں اس سے بیزار ہوں جو تم نے پہلے مجھے شریک بنایا تھا، بے شک ظالموں کے لیے دردناک عذاب ہے۔
وَأُدْخِلَ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ جَنَّـٰتٖ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُ خَـٰلِدِينَ فِيهَا بِإِذْنِ رَبِّهِمۖ تَحِيَّتُهُمْ فِيهَا سَلَـٰمٌ ﴿23﴾
اور جو لوگ ایمان لائے اور نیک عمل کیے، وہ باغوں میں داخل کیے جائیں گے جن کے نیچے نہریں بہتی ہیں، وہ اس میں ہمیشہ رہیں گے اپنے رب کے حکم سے، ان کی ملاقات کا کلمہ اس میں "سلام" ہوگا۔
أَلَمْ تَرَ كَيْفَ ضَرَبَ ٱللَّهُ مَثَلٗا كَلِمَةٗ طَيِّبَةٗ كَشَجَرَةٖ طَيِّبَةٍ أَصْلُهَا ثَابِتٞ وَفَرْعُهَا فِي ٱلسَّمَآءِ ﴿24﴾
کیا تم نے نہیں دیکھا کہ اللہ نے پاکیزہ بات کی مثال ایک پاکیزہ درخت کی دی ہے، جس کی جڑ مضبوط ہے اور اس کی شاخ آسمان میں ہے۔
تُؤْتِيٓ أُكُلَهَا كُلَّ حِينِۢ بِإِذْنِ رَبِّهَاۗ وَيَضْرِبُ ٱللَّهُ ٱلْأَمْثَالَ لِلنَّاسِ لَعَلَّهُمْ يَتَذَكَّرُونَ ﴿25﴾
وہ اپنے رب کے حکم سے ہر وقت پھل دیتی ہے، اور اللہ لوگوں کے لیے مثالیں بیان کرتا ہے تاکہ وہ نصیحت حاصل کریں۔
وَمَثَلُ كَلِمَةٍ خَبِيثَةٖ كَشَجَرَةٍ خَبِيثَةٍ ٱجْتُثَّتْ مِن فَوْقِ ٱلْأَرْضِ مَا لَهَا مِن قَرَارٍ ﴿26﴾
اور ناپاک بات کی مثال ایک ناپاک درخت کی ہے جو زمین کے اوپر سے اکھاڑ دیا گیا ہو، اس کے لیے کوئی ٹھہراؤ نہیں۔
يُثَبِّتُ ٱللَّهُ ٱلَّذِينَ ءَامَنُوا۟ بِٱلْقَوْلِ ٱلثَّابِتِ فِي ٱلْحَيَوٰةِ ٱلدُّنْيَا وَفِي ٱلْـَٔاخِرَةِۖ وَيُضِلُّ ٱللَّهُ ٱلظَّـٰلِمِينَۚ وَيَفْعَلُ ٱللَّهُ مَا يَشَآءُ ﴿27﴾
اللہ ایمان والوں کو دنیا کی زندگی میں اور آخرت میں ثابت بات پر قائم رکھتا ہے، اور اللہ ظالموں کو گمراہ کرتا ہے، اور اللہ جو چاہتا ہے کرتا ہے۔
أَلَمْ تَرَ إِلَى ٱلَّذِينَ بَدَّلُوا۟ نِعْمَتَ ٱللَّهِ كُفْرٗا وَأَحَلُّوا۟ قَوْمَهُمْ دَارَ ٱلْبَوَارِ ﴿28﴾
کیا تم نے ان لوگوں کو نہیں دیکھا جنہوں نے اللہ کی نعمت کو ناشکری سے بدل دیا اور اپنی قوم کو ہلاکت کے گھر میں لا اتارا؟
جَهَنَّمَ يَصْلَوْنَهَاۖ وَبِئْسَ ٱلْقَرَارُ ﴿29﴾
جہنم، جس میں وہ داخل ہوں گے، اور وہ بہت ہی برا ٹھکانا ہے۔
وَجَعَلُوا۟ لِلَّهِ أَندَادٗا لِّيُضِلُّوا۟ عَن سَبِيلِهِۦۗ قُلْ تَمَتَّعُوا۟ فَإِنَّ مَصِيرَكُمْ إِلَى ٱلنَّارِ ﴿30﴾
اور انہوں نے اللہ کے لیے شریک بنائے تاکہ اس کے راستے سے گمراہ کریں، کہہ دو: فائدہ اٹھا لو، بے شک تمہارا انجام آگ کی طرف ہے۔
قُل لِّعِبَادِيَ ٱلَّذِينَ ءَامَنُوا۟ يُقِيمُوا۟ ٱلصَّلَوٰةَ وَيُنفِقُوا۟ مِمَّا رَزَقْنَـٰهُمْ سِرّٗا وَعَلَانِيَةٗ مِّن قَبْلِ أَن يَأْتِيَ يَوْمٞ لَّا بَيْعٞ فِيهِ وَلَا خِلَـٰلٌ ﴿31﴾
کہہ دو: میرے بندوں سے جو ایمان لائے ہیں کہ نماز قائم کریں اور ہم نے انہیں جو رزق دیا ہے اس میں سے چھپ کر اور ظاہر طور پر خرچ کریں، اس دن کے آنے سے پہلے جس میں نہ خرید و فروخت ہوگی اور نہ دوستی۔
ٱللَّهُ ٱلَّذِي خَلَقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ وَأَنزَلَ مِنَ ٱلسَّمَآءِ مَآءٗ فَأَخْرَجَ بِهِۦ مِنَ ٱلثَّمَرَٰتِ رِزْقٗا لَّكُمْۖ وَسَخَّرَ لَكُمُ ٱلْفُلْكَ لِتَجْرِيَ فِي ٱلْبَحْرِ بِأَمْرِهِۦۖ وَسَخَّرَ لَكُمُ ٱلْأَنْهَـٰرَ ﴿32﴾
اللہ وہ ہے جس نے آسمانوں اور زمین کو پیدا کیا، اور آسمان سے پانی نازل کیا، پھر اس سے تمہارے لیے پھلوں کو رزق بنایا، اور تمہارے لیے کشتی کو مسخر کیا تاکہ وہ اس کے حکم سے سمندر میں چلے، اور تمہارے لیے نہریں مسخر کیں۔
وَسَخَّرَ لَكُمُ ٱلشَّمْسَ وَٱلْقَمَرَ دَآئِبَيْنِۖ وَسَخَّرَ لَكُمُ ٱلَّيْلَ وَٱلنَّهَارَ ﴿33﴾
اور تمہارے لیے سورج اور چاند کو مسخر کیا جو مسلسل چل رہے ہیں، اور تمہارے لیے رات اور دن کو مسخر کیا۔
وَءَاتَىٰكُم مِّن كُلِّ مَا سَأَلْتُمُوهُۚ وَإِن تَعُدُّوا۟ نِعْمَتَ ٱللَّهِ لَا تُحْصُوهَآۗ إِنَّ ٱلْإِنسَـٰنَ لَظَلُومٞ كَفَّارٞ ﴿34﴾
اور اس نے تمہیں ہر وہ چیز دی جو تم نے مانگی، اور اگر تم اللہ کی نعمتوں کو شمار کرنا چاہو تو انہیں گن نہیں سکتے، بے شک انسان بہت ہی ظالم، ناشکرا ہے۔
وَإِذْ قَالَ إِبْرَٰهِيمُ رَبِّ ٱجْعَلْ هَـٰذَا ٱلْبَلَدَ ءَامِنٗا وَٱجْنُبْنِي وَبَنِيَّ أَن نَّعْبُدَ ٱلْأَصْنَامَ ﴿35﴾
اور جب ابراہیم نے کہا: اے میرے رب! اس شہر کو امن والا بنا دے، اور مجھے اور میری اولاد کو بتوں کی عبادت سے بچا۔
رَبِّ إِنَّهُنَّ أَضْلَلْنَ كَثِيرٗا مِّنَ ٱلنَّاسِۖ فَمَن تَبِعَنِي فَإِنَّهُۥ مِنِّيۖ وَمَنْ عَصَانِي فَإِنَّكَ غَفُورٞ رَّحِيمٞ ﴿36﴾
اے میرے رب! بے شک انہوں نے بہت سے لوگوں کو گمراہ کیا، پس جو میرے پیچھے چلے وہ مجھ سے ہے، اور جس نے میری نافرمانی کی تو بے شک تو بخشنے والا، رحم کرنے والا ہے۔
رَبَّنَآ إِنِّيٓ أَسْكَنْتُ مِن ذُرِّيَّتِي بِوَادٍ غَيْرِ ذِي زَرْعٍ عِندَ بَيْتِكَ ٱلْمُحَرَّمِ رَبَّنَا لِيُقِيمُوا۟ ٱلصَّلَوٰةَ فَٱجْعَلْ أَفْـِٔدَةٗ مِّنَ ٱلنَّاسِ تَهْوِيٓ إِلَيْهِمْ وَٱرْزُقْهُم مِّنَ ٱلثَّمَرَٰتِ لَعَلَّهُمْ يَشْكُرُونَ ﴿37﴾
اے ہمارے رب! میں نے اپنی اولاد کو تیرے محترم گھر کے پاس ایک بے زراعت وادی میں بسایا ہے، اے ہمارے رب! تاکہ وہ نماز قائم کریں، پس لوگوں کے دلوں کو ان کی طرف مائل کر دے، اور انہیں پھلوں سے رزق دے تاکہ وہ شکر گزار بنیں۔
رَبَّنَآ إِنَّكَ تَعْلَمُ مَا نُخْفِي وَمَا نُعْلِنُۖ وَمَا يَخْفَىٰ عَلَى ٱللَّهِ مِن شَيْءٖ فِي ٱلْأَرْضِ وَلَا فِي ٱلسَّمَآءِ ﴿38﴾
اے ہمارے رب! بے شک تو جانتا ہے جو ہم چھپاتے ہیں اور جو ظاہر کرتے ہیں، اور اللہ سے کوئی چیز چھپی نہیں ہے، نہ زمین میں اور نہ آسمان میں۔
ٱلْحَمْدُ لِلَّهِ ٱلَّذِي وَهَبَ لِي عَلَى ٱلْكِبَرِ إِسْمَـٰعِيلَ وَإِسْحَـٰقَۚ إِنَّ رَبِّي لَسَمِيعُ ٱلدُّعَآءِ ﴿39﴾
سب تعریف اللہ کے لیے ہے جس نے مجھے بڑھاپے میں اسماعیل اور اسحاق عطا کیے، بے شک میرا رب دعا سننے والا ہے۔
رَبِّ ٱجْعَلْنِي مُقِيمَ ٱلصَّلَوٰةِ وَمِن ذُرِّيَّتِيۚ رَبَّنَا وَتَقَبَّلْ دُعَآءِ ﴿40﴾
اے میرے رب! مجھے نماز قائم کرنے والا بنا اور میری اولاد میں سے بھی، اے ہمارے رب! اور میری دعا قبول فرما۔
رَبَّنَا ٱغْفِرْ لِي وَلِوَٰلِدَيَّ وَلِلْمُؤْمِنِينَ يَوْمَ يَقُومُ ٱلْحِسَابُ ﴿41﴾
اے ہمارے رب! مجھے، میرے والدین کو اور سب ایمان والوں کو بخش دے جس دن حساب قائم ہوگا۔
وَلَا تَحْسَبَنَّ ٱللَّهَ غَـٰفِلًا عَمَّا يَعْمَلُ ٱلظَّـٰلِمُونَۚ إِنَّمَا يُؤَخِّرُهُمْ لِيَوْمٖ تَشْخَصُ فِيهِ ٱلْأَبْصَـٰرُ ﴿42﴾
اور تم اللہ کو ان ظالموں کے اعمال سے غافل نہ سمجھو، وہ تو انہیں اس دن تک مہلت دے رہا ہے جس دن آنکھیں پھٹی کی پھٹی رہ جائیں گی۔
مُهْطِعِينَ مُقْنِعِي رُءُوسِهِمْ لَا يَرْتَدُّ إِلَيْهِمْ طَرْفُهُمْ وَأَفْـِٔدَتُهُمْ هَوَآءٞ ﴿43﴾
وہ سر اٹھائے دوڑتے ہوں گے، ان کی نگاہیں ان کی طرف لوٹ کر نہ آئیں گی، اور ان کے دل خالی ہوں گے۔
وَأَنذِرِ ٱلنَّاسَ يَوْمَ يَأْتِيهِمُ ٱلْعَذَابُ فَيَقُولُ ٱلَّذِينَ ظَلَمُوا۟ رَبَّنَآ أَخِّرْنَآ إِلَىٰٓ أَجَلٖ قَرِيبٖ نُّجِبْ دَعْوَتَكَ وَنَتَّبِعِ ٱلرُّسُلَۚ أَوَ لَمْ تَكُونُوا۟ أَقْسَمْتُم مِّن قَبْلُ مَا لَكُم مِّن زَوَالٖ ﴿44﴾
اور لوگوں کو اس دن سے ڈراؤ جب ان کے پاس عذاب آئے گا، تو ظالم کہیں گے: اے ہمارے رب! ہمیں تھوڑی مہلت دے دے، ہم تیری دعوت قبول کریں گے اور رسولوں کی پیروی کریں گے۔ کیا تم نے پہلے قسم نہیں کھائی تھی کہ تمہیں زوال نہیں ہوگا؟
وَسَكَنتُمْ فِي مَسَـٰكِنِ ٱلَّذِينَ ظَلَمُوٓا۟ أَنفُسَهُمْ وَتَبَيَّنَ لَكُمْ كَيْفَ فَعَلْنَا بِهِمْ وَضَرَبْنَا لَكُمُ ٱلْأَمْثَالَ ﴿45﴾
اور تم ان لوگوں کے گھروں میں بسے جنہوں نے اپنی جانوں پر ظلم کیا، اور تمہیں واضح ہو گیا کہ ہم نے ان کے ساتھ کیا کیا، اور ہم نے تمہارے لیے مثالیں بیان کیں۔
وَقَدْ مَكَرُوا۟ مَكْرَهُمْ وَعِندَ ٱللَّهِ مَكْرُهُمْۖ وَإِن كَانَ مَكْرُهُمْ لِتَزُولَ مِنْهُ ٱلْجِبَالُ ﴿46﴾
اور انہوں نے اپنی چالیں چلیں، اور اللہ کے پاس ان کی چالیں ہیں، اور اگر ان کی چالیں ایسی ہوں کہ پہاڑ بھی ان سے ہٹ جائیں۔
فَلَا تَحْسَبَنَّ ٱللَّهَ مُخْلِفَ وَعْدِهِۦ رُسُلَهُۥٓۚ إِنَّ ٱللَّهَ عَزِيزٞ ذُو ٱنتِقَامٖ ﴿47﴾
پس تم اللہ کو اس کے رسولوں سے کیے گئے وعدے کا خلاف کرنے والا نہ سمجھو، بے شک اللہ زبردست، انتقام لینے والا ہے۔
يَوْمَ تُبَدَّلُ ٱلْأَرْضُ غَيْرَ ٱلْأَرْضِ وَٱلسَّمَـٰوَٰتُ وَبَرَزُوا۟ لِلَّهِ ٱلْوَٰحِدِ ٱلْقَهَّارِ ﴿48﴾
جس دن زمین دوسری زمین سے بدل دی جائے گی اور آسمان بھی، اور سب اللہ کے سامنے حاضر ہوں گے، جو واحد، زبردست ہے۔
وَتَرَى ٱلْمُجْرِمِينَ يَوْمَئِذٖ مُّقَرَّنِينَ فِي ٱلْأَصْفَادِ ﴿49﴾
اور تم اس دن مجرموں کو زنجیروں میں جکڑے ہوئے دیکھو گے۔
سَرَابِيلُهُم مِّن قَطِرَانٖ وَتَغْشَىٰ وُجُوهَهُمُ ٱلنَّارُ ﴿50﴾
ان کے لباس تارکول کے ہوں گے، اور ان کے چہروں کو آگ ڈھانپ لے گی۔
لِيَجْزِيَ ٱللَّهُ كُلَّ نَفْسٖ مَّا كَسَبَتۚ إِنَّ ٱللَّهَ سَرِيعُ ٱلْحِسَابِ ﴿51﴾
تاکہ اللہ ہر جان کو اس کے کیے کا بدلہ دے، بے شک اللہ حساب لینے میں بہت تیز ہے۔
هَـٰذَا بَلَـٰغٞ لِّلنَّاسِ وَلِيُنذَرُوا۟ بِهِۦ وَلِيَعْلَمُوٓا۟ أَنَّمَا هُوَ إِلَـٰهٞ وَٰحِدٞ وَلِيَذَّكَّرَ أُوْلُوا۟ ٱلْأَلْبَـٰبِ ﴿52﴾
یہ لوگوں کے لیے ایک پیغام ہے تاکہ وہ اس کے ذریعے ڈرائے جائیں، اور جان لیں کہ وہی ایک معبود ہے، اور عقل والے نصیحت حاصل کریں۔
