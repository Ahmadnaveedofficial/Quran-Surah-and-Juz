سورة الرحمن

بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
ٱلرَّحْمَـٰنُ ﴿1﴾
رحمٰن
عَلَّمَ ٱلْقُرْءَانَ ﴿2﴾
اس نے قرآن سکھایا۔
خَلَقَ ٱلْإِنسَـٰنَ ﴿3﴾
اس نے انسان کو پیدا کیا۔
عَلَّمَهُ ٱلْبَيَانَ ﴿4﴾
اسے بولنا سکھایا۔
ٱلشَّمْسُ وَٱلْقَمَرُ بِحُسْبَانٍۢ ﴿5﴾
سورج اور چاند حساب کے پابند ہیں۔
وَٱلنَّجْمُ وَٱلشَّجَرُ يَسْجُدَانِ ﴿6﴾
اور ستارے اور درخت سجدہ کرتے ہیں۔
وَٱلسَّمَآءَ رَفَعَهَا وَوَضَعَ ٱلْمِيزَانَ ﴿7﴾
اور آسمان کو بلند کیا اور میزان قائم کی۔
أَلَّا تَطْغَوْا۟ فِى ٱلْمِيزَانِ ﴿8﴾
تاکہ تم میزان میں زیادتی نہ کرو۔
وَأَقِيمُوا۟ ٱلْوَزْنَ بِٱلْقِسْطِ وَلَا تُخْسِرُوا۟ ٱلْمِيزَانَ ﴿9﴾
اور انصاف کے ساتھ وزن قائم کرو اور میزان میں کمی نہ کرو۔
وَٱلْأَرْضَ وَضَعَهَا لِلْأَنَامِ ﴿10﴾
اور زمین کو مخلوق کے لیے بچھا دیا۔
فِيهَا فَـٰكِهَةٌۭ وَٱلنَّخْلُ ذَاتُ ٱلْأَكْمَامِ ﴿11﴾
اس میں پھل ہیں اور خوشوں والے کھجور کے درخت۔
وَٱلْحَبُّ ذُو ٱلْعَصْفِ وَٱلرَّيْحَانُ ﴿12﴾
اور دانے، بھوسے والے اور خوشبو دار پودے۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿13﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
خَلَقَ ٱلْإِنسَـٰنَ مِن صَلْصَـٰلٍۢ كَٱلْفَخَّارِ ﴿14﴾
اس نے انسان کو بجنے والی مٹی سے پیدا کیا جیسے کمھار کی مٹی۔
وَخَلَقَ ٱلْجَانَّ مِن مَّارِجٍۢ مِّن نَّارٍۢ ﴿15﴾
اور جن کو آگ کی لپٹ سے پیدا کیا۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿16﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
رَبُّ ٱلْمَشْرِقَيْنِ وَرَبُّ ٱلْمَغْرِبَيْنِ ﴿17﴾
وہی مشرقوں اور مغربوں کا رب ہے۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿18﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
مَرَجَ ٱلْبَحْرَيْنِ يَلْتَقِيَانِ ﴿19﴾
اس نے دو سمندروں کو چھوڑا کہ وہ آپس میں ملتے ہیں۔
بَيْنَهُمَا بَرْزَخٌۭ لَّا يَبْغِيَانِ ﴿20﴾
ان کے درمیان ایک پردہ ہے، وہ تجاوز نہیں کرتے۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿21﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
يَخْرُجُ مِنْهُمَا ٱللُّؤْلُؤُ وَٱلْمَرْجَانُ ﴿22﴾
ان دونوں سے موتی اور مونگے نکلتے ہیں۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿23﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
وَلَهُ ٱلْجَوَارِ ٱلْمُنشَـَٔاتُ فِى ٱلْبَحْرِ كَٱلْأَعْلَـٰمِ ﴿24﴾
اور اسی کے لیے سمندر میں بلند پہاڑوں کی مانند چلنے والے جہاز ہیں۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿25﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
كُلُّ مَنْ عَلَيْهَا فَانٍۢ ﴿26﴾
جو کوئی زمین پر ہے وہ فنا ہونے والا ہے۔
وَيَبْقَىٰ وَجْهُ رَبِّكَ ذُو ٱلْجَلَـٰلِ وَٱلْإِكْرَامِ ﴿27﴾
اور تیرے رب کا چہرہ باقی رہے گا جو عظمت اور عزت والا ہے۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿28﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
يَسْـَٔلُهُۥ مَن فِى ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ ۚ كُلَّ يَوْمٍ هُوَ فِى شَأْنٍۢ ﴿29﴾
جو کوئی آسمانوں اور زمین میں ہے وہ اسی سے مانگتا ہے، ہر دن وہ ایک نئی شان میں ہوتا ہے۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿30﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
سَنَفْرُغُ لَكُمْ أَيُّهَ ٱلثَّقَلَانِ ﴿31﴾
اے جن و انس کی جماعت! ہم تمہاری طرف متوجہ ہونے والے ہیں۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿32﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
يَـٰمَعْشَرَ ٱلْجِنِّ وَٱلْإِنسِ إِنِ ٱسْتَطَعْتُمْ أَن تَنفُذُوا۟ مِنْ أَقْطَارِ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ فَٱنفُذُوا۟ ۚ لَا تَنفُذُونَ إِلَّا بِسُلْطَـٰنٍۢ ﴿33﴾
اے جن و انس کی جماعت! اگر تم آسمانوں اور زمین کے کناروں سے نکل سکتے ہو تو نکل جاؤ، تم نہیں نکل سکتے مگر طاقت کے ساتھ۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿34﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
يُرْسَلُ عَلَيْكُمَا شُوَاظٌۭ مِّن نَّارٍۢ وَنُحَاسٌۭ فَلَا تَنتَصِرَانِ ﴿35﴾
تم پر آگ کا شعلہ اور دھواں چھوڑا جائے گا، پھر تم بچ نہ سکو گے۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿36﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
فَإِذَا ٱنشَقَّتِ ٱلسَّمَآءُ فَكَانَتْ وَرْدَةًۭ كَٱلدِّهَانِ ﴿37﴾
پھر جب آسمان پھٹ جائے گا تو وہ گلابی رنگ کا ہو جائے گا جیسے روغن۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿38﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
فَيَوْمَئِذٍۢ لَّا يُسْـَٔلُ عَن ذَنۢبِهِۦٓ إِنسٌۭ وَلَا جَآنٌّۭ ﴿39﴾
پس اس دن نہ انسان سے اس کے گناہ کے بارے میں پوچھا جائے گا اور نہ جن سے۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿40﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
يُعْرَفُ ٱلْمُجْرِمُونَ بِسِيمَـٰهُمْ فَيُؤْخَذُ بِٱلنَّوَاصِى وَٱلْأَقْدَامِ ﴿41﴾
مجرموں کو ان کی علامتوں سے پہچان لیا جائے گا، پھر انہیں پیشانیوں اور پاؤں سے پکڑا جائے گا۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿42﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
هَـٰذِهِۦ جَهَنَّمُ ٱلَّتِى يُكَذِّبُ بِهَا ٱلْمُجْرِمُونَ ﴿43﴾
یہی جہنم ہے جسے مجرم جھٹلایا کرتے تھے۔
يَطُوفُونَ بَيْنَهَا وَبَيْنَ حَمِيمٍۢ آنٍۢ ﴿44﴾
وہ اس کے درمیان اور کھولتے پانی کے درمیان گھومتے پھریں گے۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿45﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
وَلِمَنْ خَافَ مَقَامَ رَبِّهِۦ جَنَّتَانِ ﴿46﴾
اور جو اپنے رب کے سامنے کھڑے ہونے سے ڈرا، اس کے لیے دو جنتیں ہیں۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿47﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
ذَوَاتَآ أَفْنَانٍۢ ﴿48﴾
وہ دونوں شاخوں والی ہوں گی۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿49﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
فِيهِمَا عَيْنَانِ تَجْرِيَانِ ﴿50﴾
ان دونوں میں دو چشمے بہہ رہے ہوں گے۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿51﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
فِيهِمَا مِن كُلِّ فَـٰكِهَةٍۢ زَوْجَانِ ﴿52﴾
ان دونوں میں ہر قسم کے پھلوں کی دو قسمیں ہوں گی۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿53﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
مُتَّكِـِٔينَ عَلَىٰ فُرُشٍۢ بَطَآئِنُهَا مِنْ إِسْتَبْرَقٍۢ ۚ وَجَنَى ٱلْجَنَّتَيْنِ دَانٍۢ ﴿54﴾
وہ تکیہ لگائے ہوں گے مخمل کی اندرونی استر والے بچھونوں پر، اور دونوں جنتوں کے پھل قریب ہوں گے۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿55﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
فِيهِنَّ قَـٰصِرَٰتُ ٱلطَّرْفِ لَمْ يَطْمِثْهُنَّ إِنسٌۭ قَبْلَهُمْ وَلَا جَآنٌّۭ ﴿56﴾
ان میں نظریں جھکائے رکھنے والی عورتیں ہوں گی، جنہیں ان سے پہلے کسی انسان یا جن نے نہیں چھوا۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿57﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
كَأَنَّهُنَّ ٱلْيَاقُوتُ وَٱلْمَرْجَانُ ﴿58﴾
گویا وہ یاقوت اور مونگے ہیں۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿59﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
هَلْ جَزَآءُ ٱلْإِحْسَـٰنِ إِلَّا ٱلْإِحْسَـٰنُ ﴿60﴾
نیکی کا بدلہ نیکی کے سوا کیا ہو سکتا ہے؟
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿61﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
وَمِن دُونِهِمَا جَنَّتَانِ ﴿62﴾
اور ان کے علاوہ دو اور جنتیں ہوں گی۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿63﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
مُدْهَآمَّتَانِ ﴿64﴾
وہ گہری سبز رنگ کی ہوں گی۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿65﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
فِيهِمَا عَيْنَانِ نَضَّاخَتَانِ ﴿66﴾
ان دونوں میں دو چشمے ہوں گے جو خوب بہتے ہوں گے۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿67﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
فِيهِمَا فَـٰكِهَةٌۭ وَنَخْلٌۭ وَرُمَّانٌۭ ﴿68﴾
ان دونوں میں میوے، کھجوریں اور انار ہوں گے۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿69﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
فِيهِنَّ خَيْرَٰتٌ حِسَانٌۭ ﴿70﴾
ان میں نیک سیرت اور خوبصورت عورتیں ہوں گی۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿71﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
حُورٌۭ مَّقْصُورَٰتٌۭ فِى ٱلْخِيَامِ ﴿72﴾
خیموں میں رہنے والی حوریں ہوں گی۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿73﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
لَمْ يَطْمِثْهُنَّ إِنسٌۭ قَبْلَهُمْ وَلَا جَآنٌّۭ ﴿74﴾
انہیں نہ کسی انسان نے چھوا ہوگا اور نہ کسی جن نے۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿75﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
مُتَّكِـِٔينَ عَلَىٰ رِفْرَفٍۢ خُضْرٍۢ وَعَبْقَرِىٍّ حِسَانٍۢ ﴿76﴾
وہ سبز قالینوں اور خوبصورت مسندوں پر تکیہ لگائے ہوں گے۔
فَبِأَىِّ ءَالَآءِ رَبِّكُمَا تُكَذِّبَانِ ﴿77﴾
تو تم دونوں اپنے رب کی کون کون سی نعمتوں کو جھٹلاؤ گے؟
تَبَـٰرَكَ ٱسْمُ رَبِّكَ ذِى ٱلْجَلَـٰلِ وَٱلْإِكْرَامِ ﴿78﴾
برکت والا ہے تیرے رب کا نام، جو عظمت اور عزت والا ہے۔
