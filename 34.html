سورة سبأ

بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
ٱلْحَمْدُ لِلَّهِ ٱلَّذِي لَهُۥ مَا فِي ٱلسَّمَـٰوَٰتِ وَمَا فِي ٱلْأَرْضِۖ وَلَهُ ٱلْحَمْدُ فِي ٱلْـَٔاخِرَةِۚ وَهُوَ ٱلْحَكِيمُ ٱلْخَبِيرُ ﴿1﴾
سب تعریف اللہ کے لیے ہے جس کے لیے ہے جو کچھ آسمانوں میں ہے اور جو کچھ زمین میں ہے، اور آخرت میں بھی اسی کے لیے تعریف ہے، اور وہی حکمت والا، خبردار ہے۔
يَعْلَمُ مَا يَلِجُ فِي ٱلْأَرْضِ وَمَا يَخْرُجُ مِنْهَا وَمَا يَنزِلُ مِنَ ٱلسَّمَآءِ وَمَا يَعْرُجُ فِيهَاۚ وَهُوَ ٱلرَّحِيمُ ٱلْغَفُورُ ﴿2﴾
وہ جانتا ہے جو زمین میں داخل ہوتا ہے اور جو اس سے نکلتا ہے، اور جو آسمان سے نازل ہوتا ہے اور جو اس میں چڑھتا ہے، اور وہی رحم کرنے والا، بخشنے والا ہے۔
وَقَالَ ٱلَّذِينَ كَفَرُوا۟ لَا تَأْتِينَا ٱلسَّاعَةُۖ قُلْ بَلَىٰ وَرَبِّي لَتَأْتِيَنَّكُمْۖ عَـٰلِمِ ٱلْغَيْبِۚ لَا يَعْزُبُ عَنْهُ مِثْقَالُ ذَرَّةٖ فِي ٱلسَّمَـٰوَٰتِ وَلَا فِي ٱلْأَرْضِ وَلَآ أَصْغَرُ مِن ذَٰلِكَ وَلَآ أَكْبَرُ إِلَّا فِي كِتَـٰبٖ مُّبِينٖ ﴿3﴾
اور کافروں نے کہا: قیامت ہم پر نہیں آئے گی۔ کہہ دیجیے: کیوں نہیں! میرے رب کی قسم! وہ تم پر ضرور آئے گی، وہ غیب کا جاننے والا ہے، اس سے آسمانوں اور زمین میں ذرہ برابر بھی چیز چھپی نہیں، نہ اس سے چھوٹی اور نہ بڑی، مگر وہ سب واضح کتاب میں ہے۔
لِّيَجْزِيَ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِۚ أُوْلَـٰٓئِكَ لَهُم مَّغْفِرَةٞ وَرِزْقٞ كَرِيمٞ ﴿4﴾
تاکہ وہ ان لوگوں کو جزا دے جو ایمان لائے اور نیک عمل کیے، ان کے لیے بخشش اور عزت والا رزق ہے۔
وَٱلَّذِينَ سَعَوْا۟ فِيٓ ءَايَـٰتِنَا مُعَـٰجِزِينَ أُوْلَـٰٓئِكَ لَهُمْ عَذَابٞ مِّن رِّجْزٍ أَلِيمٞ ﴿5﴾
اور جنہوں نے ہماری آیات کو ناکام بنانے کی کوشش کی، ان کے لیے دردناک عذاب ہے، سخت سزا سے۔
وَيَرَى ٱلَّذِينَ أُوتُوا۟ ٱلْعِلْمَ ٱلَّذِيٓ أُنزِلَ إِلَيْكَ مِن رَّبِّكَ هُوَ ٱلْحَقَّ وَيَهْدِيٓ إِلَىٰ صِرَٰطِ ٱلْعَزِيزِ ٱلْحَمِيدِ ﴿6﴾
اور جنہیں علم دیا گیا وہ دیکھتے ہیں کہ جو کچھ آپ کے رب کی طرف سے آپ پر نازل کیا گیا ہے وہی حق ہے، اور وہ غالب، تعریف کے لائق کے راستے کی طرف رہنمائی کرتا ہے۔
وَقَالَ ٱلَّذِينَ كَفَرُوا۟ هَلْ نَدُلُّكُمْ عَلَىٰ رَجُلٖ يُنَبِّئُكُمْ إِذَا مُزِّقْتُمْ كُلَّ مُمَزَّقٍ إِنَّكُمْ لَفِي خَلْقٖ جَدِيدٍ ﴿7﴾
اور کافروں نے کہا: کیا ہم تمہیں ایسے شخص کی طرف رہنمائی کریں جو تمہیں خبر دیتا ہے کہ جب تم مکمل طور پر ریزہ ریزہ ہو جاؤ گے تو تم ضرور نئی پیدائش میں آ جاؤ گے؟
أَفْتَرَىٰ عَلَى ٱللَّهِ كَذِبًا أَم بِهِۦ جِنَّةُۭۚ بَلِ ٱلَّذِينَ لَا يُؤْمِنُونَ بِٱلْـَٔاخِرَةِ فِي ٱلْعَذَابِ وَٱلضَّلَـٰلِ ٱلْبَعِيدِ ﴿8﴾
کیا اس نے اللہ پر جھوٹ باندھا ہے یا اسے جنون ہے؟ بلکہ جو لوگ آخرت پر ایمان نہیں رکھتے وہی عذاب میں اور دور کی گمراہی میں ہیں۔
أَفَلَمْ يَرَوْا۟ إِلَىٰ مَا بَيْنَ أَيْدِيهِمْ وَمَا خَلْفَهُم مِّنَ ٱلسَّمَآءِ وَٱلْأَرْضِۚ إِن نَّشَأْ نَخْسِفْ بِهِمُ ٱلْأَرْضَ أَوْ نُسْقِطْ عَلَيْهِمْ كِسَفٗا مِّنَ ٱلسَّمَآءِۚ إِنَّ فِي ذَٰلِكَ لَءَايَةٗ لِّكُلِّ عَبْدٖ مُّنِيبٖ ﴿9﴾
کیا انہوں نے نہیں دیکھا جو ان کے آگے اور پیچھے آسمان اور زمین میں ہے؟ اگر ہم چاہیں تو انہیں زمین میں دھسا دیں یا ان پر آسمان سے ٹکڑے گرا دیں۔ بے شک اس میں ہر رجوع کرنے والے بندے کے لیے نشانی ہے۔
وَلَقَدْ ءَاتَيْنَا دَاوُۥدَ مِنَّا فَضْلٗاۖ يَـٰجِبَالُ أَوِّبِي مَعَهُۥ وَٱلطَّيْرَۖ وَأَلَنَّا لَهُ ٱلْحَدِيدَ ﴿10﴾
اور بے شک ہم نے داؤد کو اپنی طرف سے فضل دیا، اے پہاڑو! اس کے ساتھ تسبیح کرو، اور پرندے بھی۔ اور ہم نے اس کے لیے لوہا نرم کر دیا۔
أَنِ ٱعْمَلْ سَـٰبِغَـٰتٖ وَقَدِّرْ فِي ٱلسَّرْدِۖ وَٱعْمَلُوا۟ صَـٰلِحًاۖ إِنِّي بِمَا تَعْمَلُونَ بَصِيرٞ ﴿11﴾
کہ تو مکمل زرہیں بنا اور جوڑ میں اندازہ رکھ، اور نیک عمل کرو، بے شک میں تمہارے اعمال کو دیکھنے والا ہوں۔
وَلِسُلَيْمَـٰنَ ٱلرِّيحَ غُدُوُّهَا شَهْرٞ وَرَوَاحُهَا شَهْرٞۖ وَأَسَلْنَا لَهُۥ عَيْنَ ٱلْقِطْرِۖ وَمِنَ ٱلْجِنِّ مَن يَعْمَلُ بَيْنَ يَدَيْهِ بِإِذْنِ رَبِّهِۦۖ وَمَن يَزِغْ مِنْهُمْ عَنْ أَمْرِنَا نُذِقْهُ مِنْ عَذَابِ ٱلسَّعِيرِ ﴿12﴾
اور ہم نے سلیمان کے لیے ہوا کو مسخر کر دیا، اس کی صبح کی رفتار ایک مہینے کی اور شام کی رفتار بھی ایک مہینے کی تھی۔ اور ہم نے اس کے لیے پگھلے ہوئے تانبے کا چشمہ بہا دیا۔ اور جنوں میں سے کچھ اس کے رب کے حکم سے اس کے سامنے کام کرتے تھے، اور جو ان میں سے ہمارے حکم سے ہٹتا ہم اسے بھڑکتی آگ کا عذاب چکھاتے۔
يَعْمَلُونَ لَهُۥ مَا يَشَآءُ مِن مَّحَـٰرِيبَ وَتَمَاثِيلَ وَجِفَانٖ كَٱلْجَوَابِ وَقُدُورٖ رَاسِيَـٰتٍۚ ٱعْمَلُوٓا۟ ءَالَ دَاوُۥدَ شُكْرٗاۚ وَقَلِيلٞ مِّنْ عِبَادِيَ ٱلشَّكُورُ ﴿13﴾
وہ اس کے لیے جو چاہتا بناتے تھے: بلند عمارتیں، مجسمے، حوضوں جیسے بڑے برتن، اور بھاری دیگیں۔ اے آلِ داؤد! شکر ادا کرو، اور میرے بندوں میں کم ہی شکر گزار ہیں۔
فَلَمَّا قَضَيْنَا عَلَيْهِ ٱلْمَوْتَ مَا دَلَّهُمْ عَلَىٰ مَوْتِهِۦٓ إِلَّا دَابَّةُ ٱلْأَرْضِ تَأْكُلُ مِنسَأَتَهُۥۖ فَلَمَّا خَرَّ تَبَيَّنَتِ ٱلْجِنُّ أَن لَّوْ كَانُوا۟ يَعْلَمُونَ ٱلْغَيْبَ مَا لَبِثُوا۟ فِي ٱلْعَذَابِ ٱلْمُهِينِ ﴿14﴾
پھر جب ہم نے اس پر موت کا حکم نافذ کیا، تو ان کو اس کی موت کا علم نہ ہوا مگر زمین کی ایک کیڑے نے اس کا عصا کھا لیا۔ پھر جب وہ گر پڑا، جنوں کو معلوم ہو گیا کہ اگر وہ غیب جانتے ہوتے تو ذلت والے عذاب میں نہ پڑے رہتے۔
لَقَدْ كَانَ لِسَبَإٖ فِي مَسْكَنِهِمْ ءَايَةٞۖ جَنَّتَانِ عَن يَمِينٖ وَشِمَالٖۖ كُلُوا۟ مِن رِّزْقِ رَبِّكُمْ وَٱشْكُرُوا۟ لَهُۥۚ بَلْدَةٞ طَيِّبَةٞ وَرَبٌّ غَفُورٞ ﴿15﴾
سبأ کے لیے ان کے رہنے کی جگہ میں ایک نشانی تھی: دائیں اور بائیں دو باغ۔ اپنے رب کے رزق سے کھاؤ اور اس کا شکر ادا کرو۔ یہ پاکیزہ شہر ہے اور رب بخشنے والا ہے۔
فَأَعْرَضُوا۟ فَأَرْسَلْنَا عَلَيْهِمْ سَيْلَ ٱلْعَرِمِ وَبَدَّلْنَـٰهُم بِجَنَّتَيْهِمْ جَنَّتَيْنِ ذَوَاتَيْ أُكُلٍ خَمْطٖ وَأَثْلٖ وَشَيْءٖ مِّن سِدْرٖ قَلِيلٖ ﴿16﴾
پھر انہوں نے منہ موڑا، تو ہم نے ان پر زور کا سیلاب بھیج دیا، اور ان کے دونوں باغوں کو ایسے باغوں سے بدل دیا جن کے پھل کڑوے، جھاڑیاں اور تھوڑا سا بیری تھا۔
ذَٰلِكَ جَزَيْنَـٰهُم بِمَا كَفَرُوا۟ۖ وَهَلْ نُجَـٰزِيٓ إِلَّا ٱلْكَفُورَ ﴿17﴾
یہ ہم نے ان کو ان کے کفر کے بدلے دیا، اور ہم ناشکرے کے سوا کسی کو سزا نہیں دیتے۔
وَجَعَلْنَا بَيْنَهُمْ وَبَيْنَ ٱلْقُرَى ٱلَّتِي بَـٰرَكْنَا فِيهَا قُرًى ظَـٰهِرَةٗ وَقَدَّرْنَا فِيهَا ٱلسَّيْرَۖ سِيرُوا۟ فِيهَا لَيَالِيَ وَأَيَّامًا ءَامِنِينَ ﴿18﴾
اور ہم نے ان کے درمیان اور ان بستیوں کے درمیان جن میں ہم نے برکت دی تھی، نمایاں بستیاں بنائیں، اور ان میں سفر کا اندازہ مقرر کیا۔ ان میں راتوں اور دنوں کو امن سے سفر کرو۔
فَقَالُوا۟ رَبَّنَا بَـٰعِدْ بَيْنَ أَسْفَارِنَا وَظَلَمُوٓا۟ أَنفُسَهُمْ فَجَعَلْنَـٰهُمْ أَحَادِيثَ وَمَزَّقْنَـٰهُمْ كُلَّ مُمَزَّقٍۚ إِنَّ فِي ذَٰلِكَ لَءَايَـٰتٖ لِّكُلِّ صَبَّارٖ شَكُورٖ ﴿19﴾
تو انہوں نے کہا: اے ہمارے رب! ہمارے سفر کے درمیان فاصلے کر دے، اور انہوں نے اپنے آپ پر ظلم کیا، تو ہم نے انہیں کہانیاں بنا دیا اور انہیں پوری طرح بکھیر دیا۔ بے شک اس میں ہر صبر کرنے والے، شکر گزار کے لیے نشانیاں ہیں۔
وَلَقَدْ صَدَّقَ عَلَيْهِمْ إِبْلِيسُ ظَنَّهُۥ فَٱتَّبَعُوهُ إِلَّا فَرِيقٗا مِّنَ ٱلْمُؤْمِنِينَ ﴿20﴾
اور ابلیس نے ان پر اپنا گمان سچ کر دکھایا، تو انہوں نے اس کی پیروی کی، سوائے ایک گروہ کے جو ایمان والے تھے۔
وَمَا كَانَ لَهُۥ عَلَيْهِم مِّن سُلْطَـٰنٍ إِلَّا لِنَعْلَمَ مَن يُؤْمِنُ بِٱلْـَٔاخِرَةِ مِمَّنْ هُوَ مِنْهَا فِي شَكّٖۚ وَرَبُّكَ عَلَىٰ كُلِّ شَيْءٍ حَفِيظٞ ﴿21﴾
اور ابلیس کو ان پر کوئی زور حاصل نہ تھا، مگر تاکہ ہم جان لیں کون آخرت پر ایمان رکھتا ہے اور کون اس میں شک میں ہے۔ اور آپ کا رب ہر چیز کا نگہبان ہے۔
قُلِ ٱدْعُوا۟ ٱلَّذِينَ زَعَمْتُم مِّن دُونِ ٱللَّهِ لَا يَمْلِكُونَ مِثْقَالَ ذَرَّةٖ فِي ٱلسَّمَـٰوَٰتِ وَلَا فِي ٱلْأَرْضِ وَمَا لَهُمْ فِيهِمَا مِن شِرْكٖ وَمَا لَهُۥ مِنْهُم مِّن ظَهِيرٖ ﴿22﴾
کہہ دیجیے: جنہیں تم اللہ کے سوا معبود سمجھتے ہو انہیں پکارو، وہ آسمانوں اور زمین میں ذرہ برابر بھی مالک نہیں ہیں، اور نہ ہی ان کا ان میں کوئی شریک ہے، اور نہ ہی اس کا ان میں سے کوئی مددگار ہے۔
وَلَا تَنفَعُ ٱلشَّفَـٰعَةُ عِندَهُۥٓ إِلَّا لِمَنْ أَذِنَ لَهُۥۚ حَتَّىٰٓ إِذَا فُزِّعَ عَن قُلُوبِهِمْ قَالُوا۟ مَاذَا قَالَ رَبُّكُمْۖ قَالُوا۟ ٱلْحَقَّۖ وَهُوَ ٱلْعَلِيُّ ٱلْكَبِيرُ ﴿23﴾
اور اس کے پاس سفارش فائدہ نہیں دیتی مگر اس کے لیے جسے وہ اجازت دے۔ یہاں تک کہ جب ان کے دلوں سے گھبراہٹ دور کی جاتی ہے، تو وہ کہتے ہیں: تمہارے رب نے کیا فرمایا؟ وہ کہتے ہیں: حق فرمایا، اور وہی بلند، بڑا ہے۔
قُلْ مَن يَرْزُقُكُم مِّنَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۖ قُلِ ٱللَّهُۚ وَإِنَّآ أَوْ إِيَّاكُمْ لَعَلَىٰ هُدٗى أَوْ فِي ضَلَـٰلٖ مُّبِينٖ ﴿24﴾
کہہ دیجیے: تمہیں آسمانوں اور زمین سے کون رزق دیتا ہے؟ کہہ دیجیے: اللہ۔ اور ہم یا تم ضرور ہدایت پر ہیں یا کھلی گمراہی میں۔
قُل لَّا تُسْـَٔلُونَ عَمَّآ أَجْرَمْنَا وَلَا نُسْـَٔلُ عَمَّا تَعْمَلُونَ ﴿25﴾
کہہ دیجیے: تم سے ہمارے جرائم کے بارے میں نہیں پوچھا جائے گا، اور ہم سے تمہارے اعمال کے بارے میں نہیں پوچھا جائے گا۔
قُلْ يَجْمَعُ بَيْنَنَا رَبُّنَا ثُمَّ يَفْتَحُ بَيْنَنَا بِٱلْحَقِّۚ وَهُوَ ٱلْفَتَّاحُ ٱلْعَلِيمُ ﴿26﴾
کہہ دیجیے: ہمارا رب ہمیں جمع کرے گا، پھر ہمارے درمیان حق کے ساتھ فیصلہ کرے گا، اور وہی فیصلہ کرنے والا، جاننے والا ہے۔
قُلْ أَرُونِي ٱلَّذِينَ أَلْحَقْتُم بِهِۦ شُرَكَآءَۚ كَلَّاۖ بَلْ هُوَ ٱللَّهُ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿27﴾
کہہ دیجیے: مجھے وہ دکھاؤ جنہیں تم نے اس کے ساتھ شریک ٹھہرایا ہے۔ ہرگز نہیں! بلکہ وہی اللہ زبردست، حکمت والا ہے۔
وَمَآ أَرْسَلْنَـٰكَ إِلَّا كَآفَّةٗ لِّلنَّاسِ بَشِيرٗا وَنَذِيرٗا وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَعْلَمُونَ ﴿28﴾
اور ہم نے آپ کو تمام لوگوں کے لیے خوشخبری دینے والا اور ڈرانے والا بنا کر ہی بھیجا ہے، لیکن اکثر لوگ نہیں جانتے۔
وَيَقُولُونَ مَتَىٰ هَـٰذَا ٱلْوَعْدُ إِن كُنتُمْ صَـٰدِقِينَ ﴿29﴾
اور وہ کہتے ہیں: یہ وعدہ کب پورا ہوگا اگر تم سچے ہو؟
قُلْ لَكُم مِّيعَادُ يَوْمٖ لَّا تَسْتَـْٔخِرُونَ عَنْهُ سَاعَةٗ وَلَا تَسْتَقْدِمُونَ ﴿30﴾
کہہ دیجیے: تمہارے لیے ایک دن مقرر ہے جس سے نہ تم ایک گھڑی پیچھے رہ سکتے ہو اور نہ آگے بڑھ سکتے ہو۔
وَقَالَ ٱلَّذِينَ كَفَرُوا۟ لَن نُّؤْمِنَ بِهَـٰذَا ٱلْقُرْءَانِ وَلَا بِٱلَّذِي بَيْنَ يَدَيْهِۚ وَلَوْ تَرَىٰٓ إِذِ ٱلظَّـٰلِمُونَ مَوْقُوفُونَ عِندَ رَبِّهِمْ يَرْجِعُ بَعْضُهُمْ إِلَىٰ بَعْضٍ ٱلْقَوْلَۖ يَقُولُ ٱلَّذِينَ ٱسْتُضْعِفُوا۟ لِلَّذِينَ ٱسْتَكْبَرُوا۟ لَوْلَآ أَنتُمْ لَكُنَّا مُؤْمِنِينَ ﴿31﴾
اور کافروں نے کہا: ہم نہ اس قرآن پر ایمان لائیں گے اور نہ اس سے پہلے کی کسی چیز پر۔ اور اگر آپ دیکھیں جب ظالم اپنے رب کے سامنے کھڑے کیے جائیں گے، تو ایک دوسرے سے باتیں کریں گے۔ کمزور لوگ بڑائی والوں سے کہیں گے: اگر تم نہ ہوتے تو ہم ضرور ایمان لے آتے۔
قَالَ ٱلَّذِينَ ٱسْتَكْبَرُوا۟ لِلَّذِينَ ٱسْتُضْعِفُوٓا۟ أَنَحْنُ صَدَدْنَـٰكُمْ عَنِ ٱلْهُدَىٰ بَعْدَ إِذْ جَآءَكُمۖ بَلْ كُنتُم مُّجْرِمِينَ ﴿32﴾
بڑائی والوں نے کمزوروں سے کہا: کیا ہم نے تمہیں ہدایت سے روکا تھا بعد اس کے کہ وہ تمہارے پاس آئی؟ بلکہ تم خود مجرم تھے۔
وَقَالَ ٱلَّذِينَ ٱسْتُضْعِفُوا۟ لِلَّذِينَ ٱسْتَكْبَرُوا۟ بَلْ مَكْرُ ٱلَّيْلِ وَٱلنَّهَارِ إِذْ تَأْمُرُونَنَآ أَن نَّكْفُرَ بِٱللَّهِ وَنَجْعَلَ لَهُۥٓ أَندَادٗاۚ وَأَسَرُّوا۟ ٱلنَّدَٰمَةَ لَمَّا رَأَوُا۟ ٱلْعَذَابَۖ وَجَعَلْنَا ٱلْأَغْلَـٰلَ فِيٓ أَعْنَاقِ ٱلَّذِينَ كَفَرُوا۟ۚ هَلْ يُجْزَوْنَ إِلَّا مَا كَانُوا۟ يَعْمَلُونَ ﴿33﴾
اور کمزور لوگ بڑائی والوں سے کہیں گے: بلکہ تمہاری رات دن کی چالاکیاں تھیں جب تم ہمیں کہتے تھے کہ ہم اللہ سے کفر کریں اور اس کے لیے شریک بنائیں۔ اور جب انہوں نے عذاب دیکھا تو وہ ندامت چھپائیں گے، اور ہم نے کافروں کی گردنوں میں طوق ڈال دیے، کیا وہ صرف وہی بدلہ پائیں گے جو وہ کرتے رہے؟
وَمَآ أَرْسَلْنَا فِي قَرْيَةٖ مِّن نَّذِيرٍ إِلَّا قَالَ مُتْرَفُوهَآ إِنَّا بِمَآ أُرْسِلْتُم بِهِۦ كَـٰفِرُونَ ﴿34﴾
اور ہم نے کسی بستی میں کوئی خبردار کرنے والا نہیں بھیجا مگر وہاں کے خوشحال لوگوں نے کہا: ہم اس چیز کے ساتھ کفر کرتے ہیں جس کے ساتھ تم بھیجے گئے ہو۔
وَقَالُوا۟ نَحْنُ أَكْثَرُ أَمْوَٰلٗا وَأَوْلَـٰدٗا وَمَا نَحْنُ بِمُعَذَّبِينَ ﴿35﴾
اور انہوں نے کہا: ہم مال اور اولاد میں زیادہ ہیں، اور ہم پر کبھی عذاب نہیں آئے گا۔
قُلْ إِنَّ رَبِّي يَبْسُطُ ٱلرِّزْقَ لِمَن يَشَآءُ وَيَقْدِرُ وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَعْلَمُونَ ﴿36﴾
کہہ دیجیے: بے شک میرا رب جس کے لیے چاہے رزق کشادہ کرتا ہے اور تنگ بھی کرتا ہے، لیکن اکثر لوگ نہیں جانتے۔
وَمَآ أَمْوَٰلُكُمْ وَلَآ أَوْلَـٰدُكُم بِٱلَّتِي تُقَرِّبُكُمْ عِندَنَا زُلْفَىٰٓ إِلَّا مَنْ ءَامَنَ وَعَمِلَ صَـٰلِحٗا فَأُوْلَـٰٓئِكَ لَهُمْ جَزَآءُ ٱلضِّعْفِ بِمَا عَمِلُوا۟ وَهُمْ فِي ٱلْغُرُفَـٰتِ ءَامِنُونَ ﴿37﴾
اور نہ تمہارے مال اور نہ تمہاری اولاد ایسی چیزیں ہیں جو تمہیں ہمارے قریب کر دیں، مگر جو ایمان لائے اور نیک عمل کیے، تو وہی لوگ ہیں جن کے لیے ان کے اعمال کے بدلے دوہرا اجر ہے، اور وہ بالا خانوں میں امن سے ہوں گے۔
وَٱلَّذِينَ يَسْعَوْنَ فِيٓ ءَايَـٰتِنَا مُعَـٰجِزِينَ أُوْلَـٰٓئِكَ فِي ٱلْعَذَابِ مُحْضَرُونَ ﴿38﴾
اور جو لوگ ہماری آیات کو ناکام بنانے کی کوشش کرتے ہیں، وہی لوگ عذاب میں حاضر کیے جائیں گے۔
قُلْ إِنَّ رَبِّي يَبْسُطُ ٱلرِّزْقَ لِمَن يَشَآءُ مِنْ عِبَادِهِۦ وَيَقْدِرُ لَهُۥۚ وَمَآ أَنفَقْتُم مِّن شَيْءٖ فَهُوَ يُخْلِفُهُۥۖ وَهُوَ خَيْرُ ٱلرَّٰزِقِينَ ﴿39﴾
کہہ دیجیے: بے شک میرا رب اپنے بندوں میں سے جس کے لیے چاہے رزق کشادہ کرتا ہے اور اس کے لیے تنگ بھی کرتا ہے، اور جو کچھ تم خرچ کرتے ہو وہ اس کا بدل دیتا ہے، اور وہ سب سے بہتر رزق دینے والا ہے۔
وَيَوْمَ يَحْشُرُهُمْ جَمِيعٗا ثُمَّ يَقُولُ لِلْمَلَـٰٓئِكَةِ أَهَـٰٓؤُلَآءِ إِيَّاكُمْ كَانُوا۟ يَعْبُدُونَ ﴿40﴾
اور جس دن وہ سب کو جمع کرے گا، پھر فرشتوں سے کہے گا: کیا یہ تمہیں پوجتے تھے؟
قَالُوا۟ سُبْحَـٰنَكَ أَنتَ وَلِيُّنَا مِن دُونِهِمۖ بَلْ كَانُوا۟ يَعْبُدُونَ ٱلْجِنَّۖ أَكْثَرُهُم بِهِم مُّؤْمِنُونَ ﴿41﴾
وہ کہیں گے: تو پاک ہے! تو ہی ہمارا ولی ہے، ان کے سوا۔ بلکہ وہ جنوں کی عبادت کرتے تھے، ان میں سے اکثر ان پر ایمان رکھتے تھے۔
فَٱلْيَوْمَ لَا يَمْلِكُ بَعْضُكُمْ لِبَعْضٖ نَفْعٗا وَلَا ضَرّٗاۚ وَنَقُولُ لِلَّذِينَ ظَلَمُوا۟ ذُوقُوا۟ عَذَابَ ٱلنَّارِ ٱلَّذِي كُنتُم بِهِۦ تُكَذِّبُونَ ﴿42﴾
پس آج تم میں سے کوئی دوسرے کے لیے نہ نفع کا مالک ہے اور نہ نقصان کا، اور ہم ظالموں سے کہیں گے: چکھو آگ کا عذاب جسے تم جھٹلاتے تھے۔
وَإِذَا تُتْلَىٰ عَلَيْهِمْ ءَايَـٰتُنَا بَيِّنَـٰتٖ قَالُوا۟ مَا هَـٰذَآ إِلَّا رَجُلٞ يُرِيدُ أَن يَصُدَّكُمْ عَمَّا كَانَ يَعْبُدُ ءَابَآؤُكُمْۚ وَقَالُوا۟ مَا هَـٰذَآ إِلَّآ إِفْكٞ مُّفْتَرٗىۚ وَقَالَ ٱلَّذِينَ كَفَرُوا۟ لِلْحَقِّ لَمَّا جَآءَهُمْ إِنْ هَـٰذَآ إِلَّا سِحْرٞ مُّبِينٞ ﴿43﴾
اور جب ان پر ہماری واضح آیات پڑھی جاتی ہیں تو وہ کہتے ہیں: یہ تو صرف ایک آدمی ہے جو چاہتا ہے کہ تمہیں اس سے روک دے جس کی تمہارے باپ دادا عبادت کرتے تھے۔ اور وہ کہتے ہیں: یہ تو محض جھوٹ ہے جو گھڑ لیا گیا ہے۔ اور کافروں نے حق کے بارے میں جب وہ ان کے پاس آیا کہا: یہ تو کھلا جادو ہے۔
وَمَآ ءَاتَيْنَـٰهُم مِّن كُتُبٖ يَدْرُسُونَهَا وَمَآ أَرْسَلْنَآ إِلَيْهِمْ قَبْلَكَ مِن نَّذِيرٖ ﴿44﴾
اور ہم نے انہیں کوئی کتاب نہیں دی جسے وہ پڑھتے ہوں، اور نہ ہی ہم نے آپ سے پہلے ان کی طرف کوئی خبردار کرنے والا بھیجا۔
وَكَذَّبَ ٱلَّذِينَ مِن قَبْلِهِمْۖ وَمَا بَلَغُوا۟ مِعْشَارَ مَآ ءَاتَيْنَـٰهُمْ فَكَذَّبُوا۟ رُسُلِيۖ فَكَيْفَ كَانَ نَكِيرِ ﴿45﴾
اور ان سے پہلے والوں نے بھی جھٹلایا، حالانکہ وہ ان کو دی گئی چیزوں کے دسویں حصے کو بھی نہیں پہنچے، پھر انہوں نے میرے رسولوں کو جھٹلایا، تو میرا انکار کیسا تھا؟
قُلْ إِنَّمَآ أَعِظُكُم بِوَٰحِدَةٍ أَن تَقُومُوا۟ لِلَّهِ مَثْنَىٰ وَفُرَٰدَىٰ ثُمَّ تَتَفَكَّرُواۚ مَا بِصَاحِبِكُم مِّن جِنَّةٍۚ إِنْ هُوَ إِلَّا نَذِيرٞ لَّكُم بَيْنَ يَدَيْ عَذَابٖ شَدِيدٖ ﴿46﴾
کہہ دیجیے: میں تمہیں صرف ایک بات کی نصیحت کرتا ہوں، کہ تم اللہ کے لیے دو دو اور اکیلے اکیلے کھڑے ہو کر سوچو، تمہارے ساتھی کو کوئی جنون نہیں ہے، وہ تو تمہیں ایک سخت عذاب سے پہلے خبردار کرنے والا ہے۔
قُلْ مَا سَأَلْتُكُم مِّنْ أَجْرٖ فَهُوَ لَكُمْۖ إِنْ أَجْرِيَ إِلَّا عَلَى ٱللَّهِۖ وَهُوَ عَلَىٰ كُلِّ شَيْءٖ شَهِيدٞ ﴿47﴾
کہہ دیجیے: میں تم سے کوئی اجر نہیں مانگتا، وہ تمہارے لیے ہے، میرا اجر تو صرف اللہ پر ہے، اور وہ ہر چیز پر گواہ ہے۔
قُلْ إِنَّ رَبِّي يَقْذِفُ بِٱلْحَقِّۖ عَلَّـٰمُ ٱلْغُيُوبِ ﴿48﴾
کہہ دیجیے: بے شک میرا رب حق کو نازل کرتا ہے، وہ غیبوں کا جاننے والا ہے۔
قُلْ جَآءَ ٱلْحَقُّ وَمَا يُبْدِئُ ٱلْبَـٰطِلُ وَمَا يُعِيدُ ﴿49﴾
کہہ دیجیے: حق آ چکا ہے، اور باطل نہ کچھ نیا شروع کر سکتا ہے اور نہ لوٹا سکتا ہے۔
قُلْ إِن ضَلَلْتُ فَإِنَّمَآ أَضِلُّ عَلَىٰ نَفْسِيۖ وَإِنِ ٱهْتَدَيْتُ فَبِمَا يُوحِيٓ إِلَيَّ رَبِّيٓۚ إِنَّهُۥ سَمِيعٞ قَرِيبٞ ﴿50﴾
کہہ دیجیے: اگر میں گمراہ ہو جاؤں تو میں اپنی ہی ذات کے لیے گمراہ ہوں، اور اگر میں ہدایت پا گیا ہوں تو یہ اس وحی کی وجہ سے ہے جو میرا رب مجھے کرتا ہے۔ بے شک وہ سننے والا، قریب ہے۔
وَلَوْ تَرَىٰٓ إِذْ فَزِعُوا۟ فَلَا فَوْتَ وَأُخِذُوا۟ مِن مَّكَانٖ قَرِيبٖ ﴿51﴾
اور اگر آپ دیکھیں جب وہ گھبرا جائیں گے، تو کوئی بچاؤ نہ ہوگا، اور وہ ایک قریب جگہ سے پکڑ لیے جائیں گے۔
وَقَالُوٓا۟ ءَامَنَّا بِهِۦۚ وَأَنَّىٰ لَهُمُ ٱلتَّنَاوُشُ مِن مَّكَانِۭ بَعِيدٖ ﴿52﴾
اور وہ کہیں گے: ہم اس پر ایمان لائے، حالانکہ ان کے لیے اسے حاصل کرنا دور کی جگہ سے ممکن نہ ہوگا۔
وَقَدۡ كَفَرُوا۟ بِهِۦ مِن قَبۡلُۖ وَيَقۡذِفُونَ بِٱلۡغَيۡبِ مِن مَّكَانِۭ بَعِيدٖ ﴿53﴾
اور وہ اس سے پہلے اس کا انکار کر چکے تھے، اور وہ دور کی جگہ سے غیب کی باتیں پھینکتے ہیں۔
وَحِيلَ بَيْنَهُمۡ وَبَيْنَ مَا يَشۡتَهُونَ كَمَا فُعِلَ بِأَشْيَاعِهِم مِّن قَبۡلُۚ إِنَّهُمۡ كَانُوا۟ فِي شَكّٖ مُّرِيبِۢ ﴿54﴾
اور ان کے درمیان اور ان کی خواہشوں کے درمیان روک دیا گیا، جیسے ان سے پہلے ان کے ہم جنسوں کے ساتھ کیا گیا۔ بے شک وہ شک میں پڑے ہوئے تھے۔
