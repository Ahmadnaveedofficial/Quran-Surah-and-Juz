10 پارہ


بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
وَٱعْلَمُوٓا۟ أَنَّمَا غَنِمْتُم مِّن شَيْءٖ فَأَنَّ لِلَّهِ خُمُسَهُۥ وَلِلرَّسُولِ وَلِذِي ٱلْقُرْبَىٰ وَٱلْيَتَـٰمَىٰ وَٱلْمَسَـٰكِينِ وَٱبْنِ ٱلسَّبِيلِۖ إِن كُنتُمْ ءَامَنتُم بِٱللَّهِ وَمَآ أَنزَلْنَا عَلَىٰ عَبْدِنَا يَوْمَ ٱلْفُرْقَانِ يَوْمَ ٱلْتَقَى ٱلْجَمْعَانِۗ وَٱللَّهُ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٌ ﴿41﴾
اور جان لو کہ جو کچھ بھی تم نے غنیمت حاصل کی ہو، اس کا پانچواں حصہ اللہ، رسول، قرابت داروں، یتیموں، مسکینوں اور مسافروں کے لیے ہے، اگر تم اللہ پر ایمان لائے ہو اور اس چیز پر جو ہم نے اپنے بندے پر فیصلے کے دن نازل کی، جس دن دونوں جماعتیں آپس میں ٹکرائیں، اور اللہ ہر چیز پر قادر ہے۔
إِذْ أَنتُم بِٱلْعُدْوَةِ ٱلدُّنْيَا وَهُم بِٱلْعُدْوَةِ ٱلْقُصْوَىٰ وَٱلرَّكْبُ أَسْفَلَ مِنكُمْۚ وَلَوْ تَوَاعَدتُّمْ لَٱخْتَلَفْتُمْ فِي ٱلْمِيعَـٰدِ وَلَـٰكِن لِّيَقْضِيَ ٱللَّهُ أَمْرٗا كَانَ مَفْعُولٗا لِّيَهْلِكَ مَنْ هَلَكَ عَنۢ بَيِّنَةٖ وَيَحْيَىٰ مَنْ حَيَّ عَنۢ بَيِّنَةٖۗ وَإِنَّ ٱللَّهَ لَسَمِيعٌ عَلِيمٌ ﴿42﴾
جب تم وادی کے قریبی کنارے پر تھے اور وہ دور والے کنارے پر، اور قافلہ تم سے نیچے کی طرف تھا، اور اگر تم آپس میں وعدہ کرتے تو وقت میں اختلاف کرتے، لیکن (یہ سب کچھ اس لیے ہوا) تاکہ اللہ ایک فیصلہ کر دے جو ہونا ہی تھا، تاکہ جسے ہلاک ہونا ہے وہ دلیل کے ساتھ ہلاک ہو، اور جسے زندہ رہنا ہے وہ دلیل کے ساتھ زندہ رہے، اور بے شک اللہ خوب سننے والا، جاننے والا ہے۔
إِذْ يُرِيكَهُمُ ٱللَّهُ فِي مَنَامِكَ قَلِيلٗاۖ وَلَوْ أَرَىٰكَهُمْ كَثِيرٗا لَّفَشِلْتُمْ وَلَتَنَـٰزَعْتُمْ فِي ٱلْأَمْرِ وَلَـٰكِنَّ ٱللَّهَ سَلَّمَۚ إِنَّهُۥ عَلِيمُۢ بِذَاتِ ٱلصُّدُورِ ﴿43﴾
جب اللہ نے تمہیں خواب میں انہیں تھوڑا دکھایا، اور اگر وہ تمہیں زیادہ دکھاتا تو تم ضرور بزدل ہو جاتے اور معاملے میں جھگڑنے لگتے، لیکن اللہ نے بچا لیا، بے شک وہ دلوں کے بھید خوب جاننے والا ہے۔
وَإِذْ يُرِيكُمُوهُمْ إِذِ ٱلْتَقَيْتُمْ فِيٓ أَعْيُنِكُمْ قَلِيلٗا وَيُقَلِّلُكُمْ فِيٓ أَعْيُنِهِمْ لِيَقْضِيَ ٱللَّهُ أَمْرٗا كَانَ مَفْعُولٗاۗ وَإِلَى ٱللَّهِ تُرْجَعُ ٱلْأُمُورُ ﴿44﴾
اور جب تم آمنے سامنے ہوئے تو اللہ نے انہیں تمہاری آنکھوں میں تھوڑا دکھایا، اور تمہیں ان کی آنکھوں میں کم کر کے دکھایا، تاکہ اللہ وہ فیصلہ پورا کر دے جو ہونا ہی تھا، اور تمام معاملات اللہ ہی کی طرف لوٹتے ہیں۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ إِذَا لَقِيتُمْ فِئَةٗ فَٱثْبُتُوا۟ وَٱذْكُرُوا۟ ٱللَّهَ كَثِيرٗا لَّعَلَّكُمْ تُفْلِحُونَ ﴿45﴾
اے ایمان والو! جب تم کسی جماعت سے مقابلہ کرو تو ثابت قدم رہو، اور اللہ کو بہت یاد کرو تاکہ تم کامیاب ہو جاؤ۔
وَأَطِيعُوا۟ ٱللَّهَ وَرَسُولَهُۥ وَلَا تَنَـٰزَعُوا۟ فَتَفْشَلُوا۟ وَتَذْهَبَ رِيحُكُمْۖ وَٱصْبِرُوٓا۟ۚ إِنَّ ٱللَّهَ مَعَ ٱلصَّـٰبِرِينَ ﴿46﴾
اور اللہ اور اس کے رسول کی اطاعت کرو، اور آپس میں جھگڑا نہ کرو ورنہ تم کمزور ہو جاؤ گے اور تمہاری ہوا اکھڑ جائے گی، اور صبر کرو، بے شک اللہ صبر کرنے والوں کے ساتھ ہے۔
وَلَا تَكُونُوا۟ كَٱلَّذِينَ خَرَجُوا۟ مِن دِيَـٰرِهِم بَطَرٗا وَرِئَآءَ ٱلنَّاسِ وَيَصُدُّونَ عَن سَبِيلِ ٱللَّهِۚ وَٱللَّهُ بِمَا يَعْمَلُونَ مُحِيطٞ ﴿47﴾
اور ان لوگوں کی طرح نہ ہو جانا جو اپنے گھروں سے اتراتے ہوئے اور لوگوں کو دکھانے کے لیے نکلے، اور اللہ کے راستے سے روکتے ہیں، اور اللہ ان کے اعمال کو گھیرے ہوئے ہے۔
وَإِذْ زَيَّنَ لَهُمُ ٱلشَّيْطَـٰنُ أَعْمَـٰلَهُمْ وَقَالَ لَا غَالِبَ لَكُمُ ٱلْيَوْمَ مِنَ ٱلنَّاسِ وَإِنِّي جَارٞ لَّكُمْۖ فَلَمَّا تَرَآءَتِ ٱلْفِئَتَانِ نَكَصَ عَلَىٰ عَقِبَيْهِ وَقَالَ إِنِّي بَرِيٓءٞ مِّنكُمْ إِنِّيٓ أَرَىٰ مَا لَا تَرَوْنَ إِنِّيٓ أَخَافُ ٱللَّهَۚ وَٱللَّهُ شَدِيدُ ٱلْعِقَابِ ﴿48﴾
اور جب شیطان نے ان کے اعمال کو ان کے لیے خوشنما بنا دیا اور کہا: آج کوئی تم پر غالب نہیں آئے گا اور میں تمہارا مددگار ہوں، پھر جب دونوں جماعتیں آمنے سامنے ہوئیں تو وہ الٹے پاؤں پھر گیا اور کہا: میں تم سے بری ہوں، میں وہ دیکھ رہا ہوں جو تم نہیں دیکھ رہے، میں اللہ سے ڈرتا ہوں، اور اللہ سخت سزا دینے والا ہے۔
إِذْ يَقُولُ ٱلْمُنَـٰفِقُونَ وَٱلَّذِينَ فِي قُلُوبِهِم مَّرَضٌ غَرَّ هَـٰٓؤُلَآءِ دِينُهُمْۚ وَمَن يَتَوَكَّلْ عَلَى ٱللَّهِ فَإِنَّ ٱللَّهَ عَزِيزٌ حَكِيمٌ ﴿49﴾
جب منافق اور وہ لوگ جن کے دلوں میں بیماری ہے کہتے تھے: ان لوگوں کو ان کے دین نے دھوکے میں ڈال رکھا ہے، اور جو اللہ پر بھروسہ کرے تو بے شک اللہ زبردست، حکمت والا ہے۔
وَلَوْ تَرَىٰٓ إِذْ يَتَوَفَّى ٱلْمَلَـٰٓئِكَةُ ٱلَّذِينَ كَفَرُوا۟ يَضْرِبُونَ وُجُوهَهُمْ وَأَدْبَـٰرَهُمْ وَذُوقُوا۟ عَذَابَ ٱلْحَرِيقِ ﴿50﴾
اور اگر تم دیکھو جب فرشتے ان لوگوں کی جان نکالتے ہیں جنہوں نے کفر کیا، وہ ان کے چہروں اور پیٹھوں پر مارتے ہیں (اور کہتے ہیں): جلتی آگ کا عذاب چکھو۔
ذَٰلِكَ بِمَا قَدَّمَتْ أَيْدِيكُمْ وَأَنَّ ٱللَّهَ لَيْسَ بِظَلَّـٰمٖ لِّلْعَبِيدِ ﴿51﴾
یہ اس وجہ سے ہے جو تمہارے ہاتھوں نے آگے بھیجا، اور بے شک اللہ بندوں پر ظلم کرنے والا نہیں ہے۔
كَدَأْبِ ءَالِ فِرْعَوْنَ وَٱلَّذِينَ مِن قَبْلِهِمۚ كَفَرُوا۟ بِـَٔايَـٰتِ ٱللَّهِ فَأَخَذَهُمُ ٱللَّهُ بِذُنُوبِهِمۚ إِنَّ ٱللَّهَ قَوِيّٞ شَدِيدُ ٱلْعِقَابِ ﴿52﴾
جیسے فرعون کی قوم اور ان سے پہلے والوں کا حال تھا، انہوں نے اللہ کی آیات کا انکار کیا، تو اللہ نے انہیں ان کے گناہوں کی وجہ سے پکڑ لیا، بے شک اللہ طاقتور، سخت سزا دینے والا ہے۔
ذَٰلِكَ بِأَنَّ ٱللَّهَ لَمْ يَكُ مُغَيِّرٗا نِّعْمَةً أَنْعَمَهَا عَلَىٰ قَوْمٍ حَتَّىٰ يُغَيِّرُوا۟ مَا بِأَنفُسِهِمْۙ وَأَنَّ ٱللَّهَ سَمِيعٌ عَلِيمٞ ﴿53﴾
یہ اس لیے کہ اللہ کسی قوم کو دی ہوئی نعمت کو نہیں بدلتا جب تک وہ خود اپنی حالت کو نہ بدلیں، اور بے شک اللہ سننے والا، جاننے والا ہے۔
كَدَأْبِ ءَالِ فِرْعَوْنَ وَٱلَّذِينَ مِن قَبْلِهِمۚ كَذَّبُوا۟ بِـَٔايَـٰتِ رَبِّهِمْ فَأَهْلَكْنَـٰهُم بِذُنُوبِهِمۡ وَأَغْرَقْنَآ ءَالَ فِرْعَوْنَۚ وَكُلٌّ كَانُوا۟ ظَـٰلِمِينَ ﴿54﴾
جیسے فرعون کی قوم اور ان سے پہلے والوں کا حال تھا، انہوں نے اپنے رب کی آیات کو جھٹلایا، تو ہم نے انہیں ان کے گناہوں کی وجہ سے ہلاک کر دیا، اور فرعون کی قوم کو غرق کر دیا، اور سب ظالم تھے۔
إِنَّ شَرَّ ٱلدَّوَآبِّ عِندَ ٱللَّهِ ٱلَّذِينَ كَفَرُوا۟ فَهُمْ لَا يُؤْمِنُونَ ﴿55﴾
بے شک اللہ کے نزدیک بدترین جانور وہ لوگ ہیں جنہوں نے کفر کیا، پس وہ ایمان نہیں لاتے۔
ٱلَّذِينَ عَـٰهَدتَّ مِنْهُمْ ثُمَّ يَنقُضُونَ عَهْدَهُمْ فِي كُلِّ مَرَّةٖ وَهُمْ لَا يَتَّقُونَ ﴿56﴾
وہ لوگ جن سے تم نے معاہدہ کیا، پھر وہ ہر بار اپنا عہد توڑ دیتے ہیں اور وہ نہیں ڈرتے۔
فَإِمَّا تَثْقَفَنَّهُمْ فِي ٱلْحَرْبِ فَشَرِّدْ بِهِم مَّنْ خَلْفَهُمْ لَعَلَّهُمْ يَذَّكَّرُونَ ﴿57﴾
پھر اگر تم انہیں جنگ میں پاؤ تو ان کے ذریعے ان کے پیچھے والوں کو عبرت دلاؤ، شاید وہ نصیحت پکڑیں۔
وَإِمَّا تَخَافَنَّ مِن قَوْمٍ خِيَانَةٗ فَٱنۢبِذْ إِلَيْهِمْ عَلَىٰ سَوَآءٍۚ إِنَّ ٱللَّهَ لَا يُحِبُّ ٱلْخَآئِنِينَ ﴿58﴾
اور اگر تم کسی قوم سے خیانت کا اندیشہ رکھو تو ان کا معاہدہ برابری کے ساتھ ان پر پھینک دو، بے شک اللہ خیانت کرنے والوں کو پسند نہیں کرتا۔
وَلَا يَحْسَبَنَّ ٱلَّذِينَ كَفَرُوا۟ سَبَقُوٓا۟ۚ إِنَّهُمْ لَا يُعْجِزُونَ ﴿59﴾
اور کافر یہ گمان نہ کریں کہ وہ بچ نکلے ہیں، بے شک وہ عاجز نہیں کر سکتے۔
وَأَعِدُّوا۟ لَهُم مَّا ٱسْتَطَعْتُم مِّن قُوَّةٖ وَمِن رِّبَـٰطِ ٱلْخَيْلِ تُرْهِبُونَ بِهِۦ عَدُوَّ ٱللَّهِ وَعَدُوَّكُمْ وَءَاخَرِينَ مِن دُونِهِمْ لَا تَعْلَمُونَهُمُ ٱللَّهُ يَعْلَمُهُمْۚ وَمَا تُنفِقُوا۟ مِن شَيْءٖ فِي سَبِيلِ ٱللَّهِ يُوَفَّ إِلَيْكُمْ وَأَنتُمْ لَا تُظْلَمُونَ ﴿60﴾
اور ان کے مقابلے کے لیے جو کچھ تم سے ہو سکے قوت تیار رکھو اور گھوڑوں کی تیاری بھی، تاکہ تم اس سے اللہ کے دشمن اور اپنے دشمن کو ڈراؤ، اور ان کے علاوہ کچھ اوروں کو بھی جنہیں تم نہیں جانتے، اللہ انہیں جانتا ہے، اور جو کچھ تم اللہ کی راہ میں خرچ کرو گے وہ تمہیں پورا دیا جائے گا اور تم پر ظلم نہیں کیا جائے گا۔
وَإِن جَنَحُوا۟ لِلسَّلْمِ فَٱجْنَحْ لَهَا وَتَوَكَّلْ عَلَى ٱللَّهِۚ إِنَّهُۥ هُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ ﴿61﴾
اور اگر وہ صلح کی طرف مائل ہوں تو تم بھی اس کی طرف مائل ہو جاؤ، اور اللہ پر بھروسہ رکھو، بے شک وہی سننے والا، جاننے والا ہے۔
وَإِن يُرِيدُوٓا۟ أَن يَخْدَعُوكَ فَإِنَّ حَسْبَكَ ٱللَّهُۚ هُوَ ٱلَّذِىٓ أَيَّدَكَ بِنَصْرِهِۦ وَبِٱلْمُؤْمِنِينَ ﴿62﴾
اور اگر وہ تمہیں دھوکہ دینا چاہیں تو بے شک تمہارے لیے اللہ کافی ہے، وہی ہے جس نے اپنی مدد اور مؤمنوں کے ذریعے تمہاری تائید کی۔
وَأَلَّفَ بَيْنَ قُلُوبِهِمۚ لَوْ أَنفَقْتَ مَا فِى ٱلْأَرْضِ جَمِيعٗا مَّآ أَلَّفْتَ بَيْنَ قُلُوبِهِمْ وَلَـٰكِنَّ ٱللَّهَ أَلَّفَ بَيْنَهُمْۚ إِنَّهُۥ عَزِيزٌ حَكِيمٞ ﴿63﴾
اور اس نے ان کے دلوں میں الفت ڈال دی، اگر تم زمین کی ساری چیزیں خرچ کر ڈالتے تو ان کے دلوں میں الفت نہ ڈال سکتے، لیکن اللہ نے ان میں الفت ڈال دی، بے شک وہ زبردست، حکمت والا ہے۔
يَـٰٓأَيُّهَا ٱلنَّبِىُّ حَسْبُكَ ٱللَّهُ وَمَنِ ٱتَّبَعَكَ مِنَ ٱلْمُؤْمِنِينَ ﴿64﴾
اے نبی! تمہارے لیے اللہ کافی ہے، اور وہ مؤمن بھی جو تمہاری پیروی کرتے ہیں۔
يَـٰٓأَيُّهَا ٱلنَّبِىُّ حَرِّضِ ٱلْمُؤْمِنِينَ عَلَى ٱلْقِتَالِۚ إِن يَكُن مِّنكُمْ عِشْرُونَ صَـٰبِرُونَ يَغْلِبُوا۟ مِاْئَتَيْنِۚ وَإِن يَكُن مِّنكُم مِّاْئَةٞ يَغْلِبُوٓا۟ أَلْفٗا مِّنَ ٱلَّذِينَ كَفَرُوا۟ بِأَنَّهُمْ قَوْمٞ لَّا يَفْقَهُونَ ﴿65﴾
اے نبی! مؤمنوں کو قتال پر ابھارو، اگر تم میں سے بیس صبر کرنے والے ہوں تو وہ دو سو پر غالب آ جائیں گے، اور اگر تم میں سے سو ہوں تو وہ ہزار کافروں پر غالب آ جائیں گے، اس لیے کہ وہ سمجھ نہیں رکھتے۔
ٱلْـَٔـٰنَ خَفَّفَ ٱللَّهُ عَنكُمْ وَعَلِمَ أَنَّ فِيكُمْ ضَعْفٗاۚ فَإِن يَكُن مِّنكُم مِّاْئَةٞ صَابِرَةٞ يَغْلِبُوا۟ مِاْئَتَيْنِۚ وَإِن يَكُن مِّنكُم أَلْفٞ يَغْلِبُوٓا۟ أَلْفَيْنِ بِإِذْنِ ٱللَّهِۗ وَٱللَّهُ مَعَ ٱلصَّـٰبِرِينَ ﴿66﴾
اب اللہ نے تم سے بوجھ ہلکا کر دیا اور جان لیا کہ تم میں کمزوری ہے، پس اگر تم میں سے سو صبر کرنے والے ہوں تو وہ دو سو پر غالب آ جائیں گے، اور اگر تم میں سے ہزار ہوں تو وہ اللہ کے حکم سے دو ہزار پر غالب آ جائیں گے، اور اللہ صبر کرنے والوں کے ساتھ ہے۔
مَا كَانَ لِنَبِيٍّ أَن يَكُونَ لَهُۥٓ أَسْرَىٰ حَتَّىٰ يُثْخِنَ فِي ٱلْأَرْضِۚ تُرِيدُونَ عَرَضَ ٱلدُّنْيَا وَٱللَّهُ يُرِيدُ ٱلْـَٔاخِرَةَۗ وَٱللَّهُ عَزِيزٌ حَكِيمٞ ﴿67﴾
کسی نبی کے لیے جائز نہیں کہ اس کے پاس قیدی ہوں جب تک وہ زمین میں خوب خونریزی نہ کرے، تم دنیا کا فائدہ چاہتے ہو اور اللہ آخرت چاہتا ہے، اور اللہ زبردست، حکمت والا ہے۔
لَّوْلَا كِتَـٰبٞ مِّنَ ٱللَّهِ سَبَقَ لَمَسَّكُمۡ فِيمَآ أَخَذْتُمۡ عَذَابٌ عَظِيمٞ ﴿68﴾
اگر اللہ کی طرف سے پہلے لکھا ہوا نہ ہوتا تو جو کچھ تم نے لیا اس میں تمہیں بڑا عذاب پہنچتا۔
فَكُلُوا۟ مِمَّا غَنِمْتُمۡ حَلَـٰلٗا طَيِّبٗاۚ وَٱتَّقُوا۟ ٱللَّهَۚ إِنَّ ٱللَّهَ غَفُورٞ رَّحِيمٞ ﴿69﴾
پس جو کچھ تم نے غنیمت میں لیا ہے اسے حلال اور پاکیزہ سمجھ کر کھاؤ، اور اللہ سے ڈرو، بے شک اللہ بخشنے والا، رحم کرنے والا ہے۔
يَـٰٓأَيُّهَا ٱلنَّبِيُّ قُل لِّمَن فِيٓ أَيْدِيكُم مِّنَ ٱلْأَسْرَىٰٓ إِن يَعْلَمِ ٱللَّهُ فِي قُلُوبِكُمْ خَيْرٗا يُؤْتِكُمْ خَيْرٗا مِّمَّآ أُخِذَ مِنكُمْ وَيَغْفِرْ لَكُمْۚ وَٱللَّهُ غَفُورٞ رَّحِيمٞ ﴿70﴾
اے نبی! ان قیدیوں سے کہہ دو جو تمہارے قبضے میں ہیں: اگر اللہ تمہارے دلوں میں کچھ بھلائی دیکھے گا تو تمہیں اس سے بہتر دے گا جو تم سے لیا گیا ہے، اور تمہیں بخش دے گا، اور اللہ بخشنے والا، رحم کرنے والا ہے۔
وَإِن يُرِيدُوا۟ خِيَانَتَكَ فَقَدْ خَانُوا۟ ٱللَّهَ مِن قَبْلُ فَأَمْكَنَ مِنْهُمْۗ وَٱللَّهُ عَلِيمٌ حَكِيمٞ ﴿71﴾
اور اگر وہ تم سے خیانت کرنا چاہیں تو وہ اس سے پہلے اللہ سے بھی خیانت کر چکے ہیں، تو اس نے انہیں قابو میں دے دیا، اور اللہ سب کچھ جاننے والا، حکمت والا ہے۔
إِنَّ ٱلَّذِينَ ءَامَنُوا۟ وَهَاجَرُوا۟ وَجَـٰهَدُوا۟ بِأَمْوَٰلِهِمْ وَأَنفُسِهِمْ فِي سَبِيلِ ٱللَّهِ وَٱلَّذِينَ ءَاوَوا۟ وَّنَصَرُوٓا۟ أُوْلَـٰٓئِكَ بَعْضُهُمْ أَوْلِيَآءُ بَعْضٖۚ وَٱلَّذِينَ ءَامَنُوا۟ وَلَمْ يُهَاجِرُوا۟ مَا لَكُم مِّن وَلَـٰيَتِهِم مِّن شَيْءٍ حَتَّىٰ يُهَاجِرُوا۟ۚ وَإِنِ ٱسْتَنصَرُوكُمْ فِي ٱلدِّينِ فَعَلَيْكُمُ ٱلنَّصْرُ إِلَّا عَلَىٰ قَوْمِۭ بَيْنَكُمْ وَبَيْنَهُم مِّيثَـٰقٞۗ وَٱللَّهُ بِمَا تَعْمَلُونَ بَصِيرٞ ﴿72﴾
بے شک جو لوگ ایمان لائے اور ہجرت کی اور اللہ کی راہ میں اپنے مال اور جان سے جہاد کیا، اور جنہوں نے ان کو جگہ دی اور مدد کی، وہی ایک دوسرے کے ولی ہیں۔ اور جو لوگ ایمان لائے لیکن ہجرت نہیں کی، تمہارے لیے ان کی ولایت میں کچھ نہیں جب تک وہ ہجرت نہ کریں، اور اگر وہ دین کے معاملے میں تم سے مدد مانگیں تو تم پر مدد کرنا لازم ہے، مگر اس قوم کے خلاف نہیں جس کے ساتھ تمہارا معاہدہ ہے، اور اللہ تمہارے اعمال کو خوب دیکھنے والا ہے۔
وَٱلَّذِينَ كَفَرُوا۟ بَعْضُهُمْ أَوْلِيَآءُ بَعْضٍۚ إِلَّا تَفْعَلُوهُ تَكُن فِتْنَةٞ فِي ٱلْأَرْضِ وَفَسَادٞ كَبِيرٞ ﴿73﴾
اور جو لوگ کافر ہیں وہ ایک دوسرے کے ولی ہیں، اگر تم ایسا نہ کرو گے (یعنی مؤمنوں کی ولایت قائم نہ رکھو گے) تو زمین میں فتنہ اور بڑا فساد ہو جائے گا۔
وَٱلَّذِينَ ءَامَنُوا۟ وَهَاجَرُوا۟ وَجَـٰهَدُوا۟ فِي سَبِيلِ ٱللَّهِ وَٱلَّذِينَ ءَاوَوا۟ وَّنَصَرُوا۟ أُوْلَـٰٓئِكَ هُمُ ٱلْمُؤْمِنُونَ حَقّٗاۚ لَّهُم مَّغْفِرَةٞ وَرِزْقٞ كَرِيمٞ ﴿74﴾
اور جو لوگ ایمان لائے اور ہجرت کی اور اللہ کی راہ میں جہاد کیا، اور جنہوں نے ان کو جگہ دی اور مدد کی، وہی سچے مؤمن ہیں، ان کے لیے مغفرت اور عزت کی روزی ہے۔
وَٱلَّذِينَ ءَامَنُوا۟ مِنۢ بَعْدُ وَهَاجَرُوا۟ وَجَـٰهَدُوا۟ مَعَكُمْ فَأُوْلَـٰٓئِكَ مِنكُمْۚ وَأُوْلُوا۟ ٱلْأَرْحَامِ بَعْضُهُمْ أَوْلَىٰ بِبَعْضٖ فِي كِتَـٰبِ ٱللَّهِۗ إِنَّ ٱللَّهَ بِكُلِّ شَيْءٍ عَلِيمُۢ ﴿75﴾
اور جو لوگ بعد میں ایمان لائے اور ہجرت کی اور تمہارے ساتھ مل کر جہاد کیا، وہ بھی تم میں سے ہیں، اور رشتہ دار اللہ کی کتاب میں ایک دوسرے کے زیادہ حق دار ہیں، بے شک اللہ ہر چیز کو جاننے والا ہے۔
سورة التوبة
بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
بَرَآءَةٞ مِّنَ ٱللَّهِ وَرَسُولِهِۦٓ إِلَى ٱلَّذِينَ عَـٰهَدتُّم مِّنَ ٱلْمُشْرِكِينَ ﴿1﴾
اللہ اور اس کے رسول کی طرف سے ان مشرکوں سے اعلانِ براءت ہے جن سے تم نے معاہدہ کیا تھا۔
فَسِيحُوا۟ فِي ٱلْأَرْضِ أَرْبَعَةَ أَشْهُرٖ وَٱعْلَمُوٓا۟ أَنَّكُمْ غَيْرُ مُعْجِزِي ٱللَّهِ وَأَنَّ ٱللَّهَ مُخْزِي ٱلْكَـٰفِرِينَ ﴿2﴾
پس تم زمین میں چار مہینے تک چل پھر لو، اور جان لو کہ تم اللہ کو عاجز نہیں کر سکتے، اور بے شک اللہ کافروں کو رسوا کرنے والا ہے۔
وَأَذَانٞ مِّنَ ٱللَّهِ وَرَسُولِهِۦٓ إِلَى ٱلنَّاسِ يَوْمَ ٱلْحَجِّ ٱلْأَكْبَرِ أَنَّ ٱللَّهَ بَرِيٓءٞ مِّنَ ٱلْمُشْرِكِينَ وَرَسُولُهُۥۚ فَإِن تُبْتُمْ فَهُوَ خَيْرٞ لَّكُمْۖ وَإِن تَوَلَّيْتُمْ فَٱعْلَمُوٓا۟ أَنَّكُمْ غَيْرُ مُعْجِزِي ٱللَّهِۗ وَبَشِّرِ ٱلَّذِينَ كَفَرُوا۟ بِعَذَابٍ أَلِيمٍ ﴿3﴾
اور اللہ اور اس کے رسول کی طرف سے لوگوں کو حجِ اکبر کے دن اعلان ہے کہ اللہ اور اس کا رسول مشرکوں سے بری ہیں، پس اگر تم توبہ کرو تو وہ تمہارے لیے بہتر ہے، اور اگر تم منہ موڑو تو جان لو کہ تم اللہ کو عاجز نہیں کر سکتے، اور کافروں کو دردناک عذاب کی خوشخبری دے دو۔
إِلَّا ٱلَّذِينَ عَـٰهَدتُّم مِّنَ ٱلْمُشْرِكِينَ ثُمَّ لَمْ يَنقُصُوكُمْ شَيْـٔٗا وَلَمْ يُظَـٰهِرُوا۟ عَلَيْكُمْ أَحَدٗا فَأَتِمُّوٓا۟ إِلَيْهِمْ عَهْدَهُمْ إِلَىٰ مُدَّتِهِمْۚ إِنَّ ٱللَّهَ يُحِبُّ ٱلْمُتَّقِينَ ﴿4﴾
سوائے ان مشرکوں کے جن سے تم نے معاہدہ کیا، پھر انہوں نے تم سے کچھ کمی نہیں کی اور نہ تمہارے خلاف کسی کی مدد کی، تو ان کے ساتھ ان کا معاہدہ ان کی مدت تک پورا کرو، بے شک اللہ متقیوں سے محبت کرتا ہے۔
فَإِذَا ٱنسَلَخَ ٱلْأَشْهُرُ ٱلْحُرُمُ فَٱقْتُلُوا۟ ٱلْمُشْرِكِينَ حَيْثُ وَجَدتُّمُوهُمْ وَخُذُوهُمْ وَٱحْصُرُوهُمْ وَٱقْعُدُوا۟ لَهُمْ كُلَّ مَرْصَدٖۚ فَإِن تَابُوا۟ وَأَقَامُوا۟ ٱلصَّلَوٰةَ وَءَاتَوُا۟ ٱلزَّكَوٰةَ فَخَلُّوا۟ سَبِيلَهُمْۚ إِنَّ ٱللَّهَ غَفُورٞ رَّحِيمٞ ﴿5﴾
پھر جب حرمت والے مہینے گزر جائیں تو مشرکوں کو جہاں پاؤ قتل کرو، اور انہیں پکڑو، اور انہیں گھیر لو، اور ہر گھات کی جگہ ان کی تاک میں بیٹھو، پھر اگر وہ توبہ کر لیں اور نماز قائم کریں اور زکوٰۃ دیں تو ان کا راستہ چھوڑ دو، بے شک اللہ بخشنے والا، رحم کرنے والا ہے۔
وَإِنْ أَحَدٞ مِّنَ ٱلْمُشْرِكِينَ ٱسْتَجَارَكَ فَأَجِرْهُ حَتَّىٰ يَسْمَعَ كَلَـٰمَ ٱللَّهِ ثُمَّ أَبْلِغْهُ مَأْمَنَهُۥۚ ذَٰلِكَ بِأَنَّهُمْ قَوْمٞ لَّا يَعْلَمُونَ ﴿6﴾
اور اگر مشرکوں میں سے کوئی تم سے پناہ مانگے تو اسے پناہ دے دو یہاں تک کہ وہ اللہ کا کلام سن لے، پھر اسے اس کی امن کی جگہ پہنچا دو، یہ اس لیے کہ وہ لوگ علم نہیں رکھتے۔
كَيْفَ يَكُونُ لِلْمُشْرِكِينَ عَهْدٌ عِندَ ٱللَّهِ وَعِندَ رَسُولِهِۦٓ إِلَّا ٱلَّذِينَ عَـٰهَدتُّمْ عِندَ ٱلْمَسْجِدِ ٱلْحَرَامِۖ فَمَا ٱسْتَقَـٰمُوا۟ لَكُمْ فَٱسْتَقِيمُوا۟ لَهُمْۚ إِنَّ ٱللَّهَ يُحِبُّ ٱلْمُتَّقِينَ ﴿7﴾
مشرکوں کے لیے اللہ اور اس کے رسول کے نزدیک کوئی عہد کیسے ہو سکتا ہے؟ سوائے ان کے جن سے تم نے مسجدِ حرام کے پاس معاہدہ کیا، پس جب تک وہ تمہارے ساتھ سیدھے رہیں تم بھی ان کے ساتھ سیدھے رہو، بے شک اللہ متقیوں سے محبت کرتا ہے۔
كَيْفَ وَإِن يَظْهَرُوا۟ عَلَيْكُمْ لَا يَرْقُبُوا۟ فِيكُمْ إِلّٗا وَلَا ذِمَّةٗۚ يُرْضُونَكُم بِأَفْوَٰهِهِمْ وَتَأْبَىٰ قُلُوبُهُمۡ وَأَكْثَرُهُمْ فَـٰسِقُونَ ﴿8﴾
کیسے (عہد ہو سکتا ہے)، حالانکہ اگر وہ تم پر غالب آ جائیں تو تمہارے بارے میں نہ قرابت کا لحاظ کریں گے اور نہ عہد کا، وہ تمہیں زبان سے خوش کرتے ہیں مگر ان کے دل انکار کرتے ہیں، اور ان میں سے اکثر نافرمان ہیں۔
ٱشْتَرَوْا۟ بِـَٔايَـٰتِ ٱللَّهِ ثَمَنٗا قَلِيلٗا فَصَدُّوا۟ عَن سَبِيلِهِۦٓۚ إِنَّهُمْ سَآءَ مَا كَانُوا۟ يَعْمَلُونَ ﴿9﴾
انہوں نے اللہ کی آیات کے بدلے تھوڑی قیمت لے لی، پھر اللہ کے راستے سے روکا، بے شک وہ جو کچھ کرتے ہیں وہ بہت برا ہے۔
لَا يَرْقُبُونَ فِي مُؤْمِنٍ إِلّٗا وَلَا ذِمَّةٗۚ وَأُوْلَـٰٓئِكَ هُمُ ٱلْمُعْتَدُونَ ﴿10﴾
وہ کسی مؤمن کے بارے میں نہ قرابت کا لحاظ کرتے ہیں اور نہ عہد کا، یہی لوگ حد سے بڑھنے والے ہیں۔
فَإِن تَابُوا۟ وَأَقَامُوا۟ ٱلصَّلَوٰةَ وَءَاتَوُا۟ ٱلزَّكَوٰةَ فَإِخْوَٰنُكُمْ فِي ٱلدِّينِۗ وَنُفَصِّلُ ٱلْـَٔايَـٰتِ لِقَوْمٖ يَعْلَمُونَ ﴿11﴾
پھر اگر وہ توبہ کر لیں اور نماز قائم کریں اور زکوٰۃ دیں تو وہ تمہارے دین میں بھائی ہیں، اور ہم آیات کو ان لوگوں کے لیے کھول کر بیان کرتے ہیں جو علم رکھتے ہیں۔
وَإِن نَّكَثُوٓا۟ أَيْمَـٰنَهُم مِّنۢ بَعْدِ عَهْدِهِمْ وَطَعَنُوا۟ فِي دِينِكُمْ فَقَـٰتِلُوٓا۟ أَئِمَّةَ ٱلْكُفْرِ إِنَّهُمْ لَآ أَيْمَـٰنَ لَهُمْ لَعَلَّهُمْ يَنتَهُونَ ﴿12﴾
اور اگر وہ اپنے عہد کے بعد اپنی قسمیں توڑ دیں اور تمہارے دین میں طعن کریں تو کفر کے سرداروں سے جنگ کرو، بے شک ان کی کوئی قسم معتبر نہیں، شاید وہ باز آ جائیں۔
أَلَا تُقَـٰتِلُونَ قَوْمٗا نَّكَثُوٓا۟ أَيْمَـٰنَهُمْ وَهَمُّوا۟ بِإِخْرَاجِ ٱلرَّسُولِ وَهُم بَدَءُوكُمْ أَوَّلَ مَرَّةٍۚ أَتَخْشَوْنَهُمْۚ فَٱللَّهُ أَحَقُّ أَن تَخْشَوْهُ إِن كُنتُم مُّؤْمِنِينَ ﴿13﴾
کیا تم ایسے لوگوں سے جنگ نہیں کرو گے جنہوں نے اپنی قسمیں توڑ دیں اور رسول کو نکالنے کا ارادہ کیا، اور وہی تم سے پہلے حملہ آور ہوئے؟ کیا تم ان سے ڈرتے ہو؟ اللہ زیادہ حق دار ہے کہ تم اس سے ڈرو اگر تم مؤمن ہو۔
قَـٰتِلُوهُمْ يُعَذِّبْهُمُ ٱللَّهُ بِأَيْدِيكُمْ وَيُخْزِهِمْ وَيَنصُرْكُمْ عَلَيْهِمْ وَيَشْفِ صُدُورَ قَوْمٖ مُّؤْمِنِينَ ﴿14﴾
ان سے جنگ کرو، اللہ انہیں تمہارے ہاتھوں سے عذاب دے گا، اور انہیں رسوا کرے گا، اور تمہیں ان پر فتح دے گا، اور مؤمنوں کے دلوں کو تسکین دے گا۔
وَيُذْهِبْ غَيْظَ قُلُوبِهِمۡۗ وَيَتُوبُ ٱللَّهُ عَلَىٰ مَن يَشَآءُۗ وَٱللَّهُ عَلِيمٌ حَكِيمٌ ﴿15﴾
اور ان کے دلوں کے غصے کو دور کرے گا، اور اللہ جس پر چاہے توبہ قبول کرے گا، اور اللہ سب کچھ جاننے والا، حکمت والا ہے۔
أَمْ حَسِبْتُمْ أَن تُتْرَكُوا۟ وَلَمَّا يَعْلَمِ ٱللَّهُ ٱلَّذِينَ جَـٰهَدُوا۟ مِنكُمْ وَلَمْ يَتَّخِذُوا۟ مِن دُونِ ٱللَّهِ وَلَا رَسُولِهِۦ وَلَا ٱلْمُؤْمِنِينَ وَلِيجَةٗۚ وَٱللَّهُ خَبِيرُۢ بِمَا تَعْمَلُونَ ﴿16﴾
کیا تم نے گمان کیا کہ تم چھوڑ دیے جاؤ گے، حالانکہ ابھی اللہ نے ان لوگوں کو نہیں جانچا جو تم میں سے جہاد کرتے ہیں اور اللہ، اس کے رسول اور مؤمنوں کے سوا کسی کو رازدار نہیں بناتے؟ اور اللہ تمہارے اعمال سے خوب باخبر ہے۔
مَا كَانَ لِلْمُشْرِكِينَ أَن يَعْمُرُوا۟ مَسَـٰجِدَ ٱللَّهِ شَـٰهِدِينَ عَلَىٰٓ أَنفُسِهِم بِٱلْكُفْرِۚ أُوْلَـٰٓئِكَ حَبِطَتْ أَعْمَـٰلُهُمْ وَفِي ٱلنَّارِ هُمْ خَـٰلِدُونَ ﴿17﴾
مشرکوں کے لیے یہ مناسب نہیں کہ وہ اللہ کی مسجدوں کو آباد کریں، درآنحالیکہ وہ اپنے کفر کے خود گواہ ہیں، یہی لوگ ہیں جن کے اعمال ضائع ہو گئے، اور وہ ہمیشہ جہنم میں رہیں گے۔
إِنَّمَا يَعْمُرُ مَسَـٰجِدَ ٱللَّهِ مَنْ ءَامَنَ بِٱللَّهِ وَٱلْيَوْمِ ٱلْـَٔاخِرِ وَأَقَامَ ٱلصَّلَوٰةَ وَءَاتَى ٱلزَّكَوٰةَ وَلَمْ يَخْشَ إِلَّا ٱللَّهَ فَعَسَىٰٓ أُوْلَـٰٓئِكَ أَن يَكُونُوا۟ مِنَ ٱلْمُهْتَدِينَ ﴿18﴾
اللہ کی مسجدوں کو وہی لوگ آباد کرتے ہیں جو اللہ اور آخرت کے دن پر ایمان رکھتے ہیں، اور نماز قائم کرتے ہیں، اور زکوٰۃ دیتے ہیں، اور اللہ کے سوا کسی سے نہیں ڈرتے، پس امید ہے کہ یہی لوگ ہدایت یافتہ ہوں گے۔
أَجَعَلْتُمْ سِقَايَةَ ٱلْحَاجِّ وَعِمَارَةَ ٱلْمَسْجِدِ ٱلْحَرَامِ كَمَنْ ءَامَنَ بِٱللَّهِ وَٱلْيَوْمِ ٱلْـَٔاخِرِ وَجَـٰهَدَ فِي سَبِيلِ ٱللَّهِۚ لَا يَسْتَوُۥنَ عِندَ ٱللَّهِۗ وَٱللَّهُ لَا يَهْدِي ٱلْقَوْمَ ٱلظَّـٰلِمِينَ ﴿19﴾
کیا تم نے حاجیوں کو پانی پلانے اور مسجدِ حرام کو آباد کرنے کو اس شخص کے برابر کر دیا جو اللہ اور آخرت کے دن پر ایمان لایا اور اللہ کی راہ میں جہاد کیا؟ وہ اللہ کے نزدیک برابر نہیں ہیں، اور اللہ ظالم قوم کو ہدایت نہیں دیتا۔
ٱلَّذِينَ ءَامَنُوا۟ وَهَاجَرُوا۟ وَجَـٰهَدُوا۟ فِي سَبِيلِ ٱللَّهِ بِأَمْوَٰلِهِمْ وَأَنفُسِهِمْ أَعْظَمُ دَرَجَةً عِندَ ٱللَّهِۚ وَأُوْلَـٰٓئِكَ هُمُ ٱلْفَآئِزُونَ ﴿20﴾
وہ لوگ جو ایمان لائے، اور ہجرت کی، اور اللہ کی راہ میں اپنے مالوں اور جانوں سے جہاد کیا، وہ اللہ کے نزدیک بلند درجے والے ہیں، اور یہی لوگ کامیاب ہیں۔
يُبَشِّرُهُمْ رَبُّهُم بِرَحْمَةٖ مِّنْهُ وَرِضْوَٰنٖ وَجَنَّـٰتٖ لَّهُمْ فِيهَا نَعِيمٞ مُّقِيمٌ ﴿21﴾
ان کا رب انہیں اپنی رحمت، خوشنودی اور ایسی جنتوں کی خوشخبری دیتا ہے جن میں ان کے لیے دائمی نعمت ہے۔
خَـٰلِدِينَ فِيهَآ أَبَدٗآۚ إِنَّ ٱللَّهَ عِندَهُۥٓ أَجْرٌ عَظِيمٌ ﴿22﴾
وہ اس میں ہمیشہ ہمیشہ رہیں گے، بے شک اللہ کے پاس بہت بڑا اجر ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تَتَّخِذُوٓا۟ ءَابَآءَكُمْ وَإِخْوَٰنَكُمْ أَوْلِيَآءَ إِنِ ٱسْتَحَبُّوا۟ ٱلْكُفْرَ عَلَى ٱلْإِيمَـٰنِۚ وَمَن يَتَوَلَّهُم مِّنكُمْ فَأُوْلَـٰٓئِكَ هُمُ ٱلظَّـٰلِمُونَ ﴿23﴾
اے ایمان والو! اگر تمہارے باپ اور بھائی ایمان کے مقابلے میں کفر کو پسند کریں تو انہیں دوست نہ بناؤ، اور جو ان سے دوستی کرے گا وہی ظالم ہے۔
قُلْ إِن كَانَ ءَابَآؤُكُمْ وَأَبْنَآؤُكُمْ وَإِخْوَٰنُكُمْ وَأَزْوَٰجُكُمْ وَعَشِيرَتُكُمْ وَأَمْوَٰلٌ ٱقْتَرَفْتُمُوهَا وَتِجَـٰرَةٞ تَخْشَوْنَ كَسَادَهَا وَمَسَـٰكِنُ تَرْضَوْنَهَآ أَحَبَّ إِلَيْكُم مِّنَ ٱللَّهِ وَرَسُولِهِۦ وَجِهَادٖ فِي سَبِيلِهِۦ فَتَرَبَّصُوا۟ حَتَّىٰ يَأْتِيَ ٱللَّهُ بِأَمْرِهِۦۗ وَٱللَّهُ لَا يَهْدِي ٱلْقَوْمَ ٱلْفَـٰسِقِينَ ﴿24﴾
کہہ دو: اگر تمہارے باپ، بیٹے، بھائی، بیویاں، رشتہ دار، وہ مال جو تم نے کمائے ہیں، وہ تجارت جس کے نقصان سے تم ڈرتے ہو، اور وہ مکانات جنہیں تم پسند کرتے ہو — اگر یہ سب تمہیں اللہ، اس کے رسول اور اس کی راہ میں جہاد سے زیادہ محبوب ہیں، تو انتظار کرو یہاں تک کہ اللہ اپنا حکم لے آئے، اور اللہ فاسق قوم کو ہدایت نہیں دیتا۔
لَقَدْ نَصَرَكُمُ ٱللَّهُ فِي مَوَٰطِنَ كَثِيرَةٖۙ وَيَوْمَ حُنَيْنٍ إِذْ أَعْجَبَتْكُمْ كَثْرَتُكُمْ فَلَمْ تُغْنِ عَنكُمْ شَيْـٔٗا وَضَاقَتْ عَلَيْكُمُ ٱلْأَرْضُ بِمَا رَحُبَتْ ثُمَّ وَلَّيْتُم مُّدْبِرِينَ ﴿25﴾
بے شک اللہ نے تمہیں بہت سے مواقع پر مدد دی، اور حنین کے دن بھی، جب تمہیں اپنی کثرت نے خوش کر دیا، مگر وہ تمہارے کچھ کام نہ آئی، اور زمین باوجود کشادگی کے تم پر تنگ ہو گئی، پھر تم پیٹھ پھیر کر بھاگ گئے۔
ثُمَّ أَنزَلَ ٱللَّهُ سَكِينَتَهُۥ عَلَىٰ رَسُولِهِۦ وَعَلَى ٱلْمُؤْمِنِينَ وَأَنزَلَ جُنُودٗا لَّمْ تَرَوْهَا وَعَذَّبَ ٱلَّذِينَ كَفَرُوا۟ۚ وَذَٰلِكَ جَزَآءُ ٱلْكَـٰفِرِينَ ﴿26﴾
پھر اللہ نے اپنے رسول پر اور مؤمنوں پر اپنی سکینت نازل فرمائی، اور ایسے لشکر اتارے جنہیں تم نے نہیں دیکھا، اور کافروں کو عذاب دیا، اور یہی کافروں کی سزا ہے۔
ثُمَّ يَتُوبُ ٱللَّهُ مِنۢ بَعْدِ ذَٰلِكَ عَلَىٰ مَن يَشَآءُۗ وَٱللَّهُ غَفُورٞ رَّحِيمٞ ﴿27﴾
پھر اس کے بعد اللہ جس پر چاہے توبہ قبول فرماتا ہے، اور اللہ بخشنے والا، رحم کرنے والا ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ إِنَّمَا ٱلْمُشْرِكُونَ نَجَسٞ فَلَا يَقْرَبُوا۟ ٱلْمَسْجِدَ ٱلْحَرَامَ بَعْدَ عَامِهِمْ هَـٰذَاۚ وَإِنْ خِفْتُمْ عَيْلَةٗ فَسَوْفَ يُغْنِيكُمُ ٱللَّهُ مِن فَضْلِهِۦٓ إِن شَآءَۚ إِنَّ ٱللَّهَ عَلِيمٌ حَكِيمٌ ﴿28﴾
اے ایمان والو! بے شک مشرک ناپاک ہیں، پس وہ اس سال کے بعد مسجدِ حرام کے قریب نہ آئیں، اور اگر تم تنگ دستی سے ڈرو تو عنقریب اللہ تمہیں اپنے فضل سے غنی کر دے گا اگر وہ چاہے، بے شک اللہ جاننے والا، حکمت والا ہے۔
قَـٰتِلُوا۟ ٱلَّذِينَ لَا يُؤْمِنُونَ بِٱللَّهِ وَلَا بِٱلْيَوْمِ ٱلْـَٔاخِرِ وَلَا يُحَرِّمُونَ مَا حَرَّمَ ٱللَّهُ وَرَسُولُهُۥ وَلَا يَدِينُونَ دِينَ ٱلْحَقِّ مِنَ ٱلَّذِينَ أُوتُوا۟ ٱلْكِتَـٰبَ حَتَّىٰ يُعْطُوا۟ ٱلْجِزْيَةَ عَن يَدٖ وَهُمْ صَـٰغِرُونَ ﴿29﴾
ان لوگوں سے جنگ کرو جو نہ اللہ پر ایمان رکھتے ہیں، نہ آخرت کے دن پر، اور نہ ان چیزوں کو حرام سمجھتے ہیں جنہیں اللہ اور اس کے رسول نے حرام کیا ہے، اور نہ دینِ حق کو قبول کرتے ہیں — ان لوگوں میں سے جنہیں کتاب دی گئی — یہاں تک کہ وہ ذلت کے ساتھ جزیہ دیں۔
وَقَالَتِ ٱلْيَهُودُ عُزَيْرٌ ٱبْنُ ٱللَّهِ وَقَالَتِ ٱلنَّصَـٰرَى ٱلْمَسِيحُ ٱبْنُ ٱللَّهِۖ ذَٰلِكَ قَوْلُهُم بِأَفْوَٰهِهِمْۖ يُضَـٰهِـُٔونَ قَوْلَ ٱلَّذِينَ كَفَرُوا۟ مِن قَبْلُۚ قَـٰتَلَهُمُ ٱللَّهُۚ أَنَّىٰ يُؤْفَكُونَ ﴿30﴾
اور یہود کہتے ہیں: عزیر اللہ کا بیٹا ہے، اور نصاریٰ کہتے ہیں: مسیح اللہ کا بیٹا ہے، یہ ان کے منہ کی باتیں ہیں، وہ ان لوگوں کی باتوں کی نقل کرتے ہیں جو ان سے پہلے کافر ہو چکے، اللہ انہیں ہلاک کرے، کہاں بہکائے جا رہے ہیں۔
ٱتَّخَذُوٓا۟ أَحْبَارَهُمْ وَرُهْبَـٰنَهُمْ أَرْبَابٗا مِّن دُونِ ٱللَّهِ وَٱلْمَسِيحَ ٱبْنَ مَرْيَمَۖ وَمَآ أُمِرُوٓا۟ إِلَّا لِيَعْبُدُوا۟ إِلَـٰهٗا وَٰحِدٗاۖ لَّآ إِلَـٰهَ إِلَّا هُوَۚ سُبْحَـٰنَهُۥ عَمَّا يُشْرِكُونَ ﴿31﴾
انہوں نے اپنے علماء اور درویشوں کو اللہ کے سوا رب بنا لیا، اور مسیح ابن مریم کو بھی، حالانکہ انہیں حکم دیا گیا تھا کہ صرف ایک معبود کی عبادت کریں، اس کے سوا کوئی معبود نہیں، وہ پاک ہے اس شرک سے جو وہ کرتے ہیں۔
يُرِيدُونَ أَن يُطْفِـُٔوا۟ نُورَ ٱللَّهِ بِأَفْوَٰهِهِمْ وَيَأْبَى ٱللَّهُ إِلَّآ أَن يُتِمَّ نُورَهُۥ وَلَوْ كَرِهَ ٱلْكَـٰفِرُونَ ﴿32﴾
وہ چاہتے ہیں کہ اللہ کے نور کو اپنے منہ سے بجھا دیں، اور اللہ انکار کرتا ہے مگر یہ کہ اپنا نور مکمل کرے، اگرچہ کافر اسے ناپسند کریں۔
هُوَ ٱلَّذِيٓ أَرْسَلَ رَسُولَهُۥ بِٱلْهُدَىٰ وَدِينِ ٱلْحَقِّ لِيُظْهِرَهُۥ عَلَى ٱلدِّينِ كُلِّهِۦ وَلَوْ كَرِهَ ٱلْمُشْرِكُونَ ﴿33﴾
وہی ہے جس نے اپنے رسول کو ہدایت اور دینِ حق کے ساتھ بھیجا تاکہ اسے تمام دینوں پر غالب کرے، اگرچہ مشرک اسے ناپسند کریں۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ إِنَّ كَثِيرٗا مِّنَ ٱلْأَحْبَارِ وَٱلرُّهْبَانِ لَيَأْكُلُونَ أَمْوَٰلَ ٱلنَّاسِ بِٱلْبَـٰطِلِ وَيَصُدُّونَ عَن سَبِيلِ ٱللَّهِۗ وَٱلَّذِينَ يَكْنِزُونَ ٱلذَّهَبَ وَٱلْفِضَّةَ وَلَا يُنفِقُونَهَا فِي سَبِيلِ ٱللَّهِ فَبَشِّرْهُم بِعَذَابٍ أَلِيمٖ ﴿34﴾
اے ایمان والو! بے شک بہت سے علماء اور درویش لوگوں کے مال ناحق کھاتے ہیں اور اللہ کے راستے سے روکتے ہیں، اور جو لوگ سونا اور چاندی جمع کرتے ہیں اور اسے اللہ کی راہ میں خرچ نہیں کرتے، انہیں دردناک عذاب کی خوشخبری دے دو۔
يَوْمَ يُحْمَىٰ عَلَيْهَا فِي نَارِ جَهَنَّمَ فَتُكْوَىٰ بِهَا جِبَاهُهُمْ وَجُنُوبُهُمْ وَظُهُورُهُمْۖ هَـٰذَا مَا كَنَزْتُمْ لِأَنفُسِكُمْ فَذُوقُوا۟ مَا كُنتُمْ تَكْنِزُونَ ﴿35﴾
جس دن وہ (سونا چاندی) جہنم کی آگ میں گرم کیے جائیں گے، پھر ان سے ان کی پیشانیاں، پہلو اور پیٹھیں داغی جائیں گی، (کہا جائے گا:) یہ ہے جو تم نے اپنے لیے جمع کیا تھا، سو چکھو جو تم جمع کرتے تھے۔
إِنَّ عِدَّةَ ٱلشُّهُورِ عِندَ ٱللَّهِ ٱثْنَا عَشَرَ شَهْرٗا فِي كِتَـٰبِ ٱللَّهِ يَوْمَ خَلَقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ مِنْهَآ أَرْبَعَةٌ حُرُمٞۚ ذَٰلِكَ ٱلدِّينُ ٱلْقَيِّمُۚ فَلَا تَظْلِمُوا۟ فِيهِنَّ أَنفُسَكُمْۚ وَقَـٰتِلُوا۟ ٱلْمُشْرِكِينَ كَآفَّةً كَمَا يُقَـٰتِلُونَكُمْ كَآفَّةٗۚ وَٱعْلَمُوٓا۟ أَنَّ ٱللَّهَ مَعَ ٱلْمُتَّقِينَ ﴿36﴾
بے شک اللہ کے نزدیک مہینوں کی گنتی بارہ ہے، اللہ کی کتاب میں جس دن اس نے آسمانوں اور زمین کو پیدا کیا، ان میں سے چار حرمت والے ہیں، یہی سیدھا دین ہے، پس ان مہینوں میں اپنے نفسوں پر ظلم نہ کرو، اور مشرکوں سے سب مل کر جنگ کرو جیسے وہ تم سے سب مل کر جنگ کرتے ہیں، اور جان لو کہ اللہ متقیوں کے ساتھ ہے۔
إِنَّمَا ٱلنَّسِيٓءُ زِيَادَةٞ فِي ٱلْكُفْرِ يُضَلُّ بِهِ ٱلَّذِينَ كَفَرُوا۟ يُحِلُّونَهُۥ عَامٗا وَيُحَرِّمُونَهُۥ عَامٗا لِّيُوَاطِـُٔوا۟ عِدَّةَ مَا حَرَّمَ ٱللَّهُ فَيُحِلُّوا۟ مَا حَرَّمَ ٱللَّهُۚ زُيِّنَ لَهُمْ سُوٓءُ أَعْمَـٰلِهِمۡۗ وَٱللَّهُ لَا يَهْدِي ٱلْقَوْمَ ٱلْكَـٰفِرِينَ ﴿37﴾
بے شک نسیء (مہینوں کی تقدیم و تاخیر) کفر میں اضافہ ہے، جس سے کافر گمراہ ہوتے ہیں، وہ اسے ایک سال حلال کرتے ہیں اور ایک سال حرام کرتے ہیں تاکہ اللہ کے حرام کیے ہوئے مہینوں کی گنتی پوری کر سکیں، پس وہ اللہ کے حرام کیے ہوئے کو حلال کر لیتے ہیں، ان کے برے اعمال ان کے لیے خوشنما بنا دیے گئے ہیں، اور اللہ کافر قوم کو ہدایت نہیں دیتا۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ مَا لَكُمْ إِذَا قِيلَ لَكُمُ ٱنفِرُوا۟ فِي سَبِيلِ ٱللَّهِ ٱثَّاقَلْتُمْ إِلَى ٱلْأَرْضِۚ أَرَضِيتُم بِٱلْحَيَوٰةِ ٱلدُّنْيَا مِنَ ٱلْـَٔاخِرَةِۚ فَمَا مَتَـٰعُ ٱلْحَيَوٰةِ ٱلدُّنْيَا فِي ٱلْـَٔاخِرَةِ إِلَّا قَلِيلٌ ﴿38﴾
اے ایمان والو! تمہیں کیا ہو گیا ہے کہ جب تم سے کہا جاتا ہے کہ اللہ کی راہ میں نکلو تو تم زمین سے چمٹ جاتے ہو؟ کیا تم آخرت کے مقابلے میں دنیا کی زندگی پر راضی ہو گئے ہو؟ تو دنیا کی زندگی کا فائدہ آخرت کے مقابلے میں کچھ بھی نہیں مگر تھوڑا۔
إِلَّا تَنفِرُوا۟ يُعَذِّبْكُمْ عَذَابًا أَلِيمٗا وَيَسْتَبْدِلْ قَوْمًا غَيْرَكُمْ وَلَا تَضُرُّوهُ شَيْـٔٗاۚ وَٱللَّهُ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٌ ﴿39﴾
اگر تم (اللہ کی راہ میں) نہ نکلو گے تو وہ تمہیں دردناک عذاب دے گا، اور تمہاری جگہ کسی اور قوم کو لے آئے گا، اور تم اسے کچھ نقصان نہیں پہنچا سکو گے، اور اللہ ہر چیز پر قدرت رکھتا ہے۔
إِلَّا تَنصُرُوهُ فَقَدْ نَصَرَهُ ٱللَّهُ إِذْ أَخْرَجَهُ ٱلَّذِينَ كَفَرُوا۟ ثَانِيَ ٱثْنَيْنِ إِذْ هُمَا فِي ٱلْغَارِ إِذْ يَقُولُ لِصَاحِبِهِۦ لَا تَحْزَنْ إِنَّ ٱللَّهَ مَعَنَاۖ فَأَنزَلَ ٱللَّهُ سَكِينَتَهُۥ عَلَيْهِ وَأَيَّدَهُۥ بِجُنُودٖ لَّمْ تَرَوْهَا وَجَعَلَ كَلِمَةَ ٱلَّذِينَ كَفَرُوا۟ ٱلسُّفْلَىٰۗ وَكَلِمَةُ ٱللَّهِ هِيَ ٱلْعُلْيَاۗ وَٱللَّهُ عَزِيزٌ حَكِيمٌ ﴿40﴾
اگر تم اس کی مدد نہ کرو تو اللہ نے اس کی مدد کی جب کافروں نے اسے نکالا، جب وہ دو میں سے دوسرا تھا، جب وہ دونوں غار میں تھے، جب وہ اپنے ساتھی سے کہہ رہا تھا: غم نہ کرو، بے شک اللہ ہمارے ساتھ ہے۔ پھر اللہ نے اس پر اپنی سکینت نازل فرمائی، اور اسے ایسے لشکروں سے مدد دی جنہیں تم نے نہیں دیکھا، اور کافروں کی بات کو نیچا کر دیا، اور اللہ کا کلمہ ہی بلند ہے، اور اللہ غالب، حکمت والا ہے۔
ٱنفِرُوا۟ خِفَافٗا وَثِقَالٗا وَجَـٰهِدُوا۟ بِأَمْوَٰلِكُمْ وَأَنفُسِكُمْ فِي سَبِيلِ ٱللَّهِۚ ذَٰلِكُمْ خَيْرٞ لَّكُمْ إِن كُنتُمْ تَعْلَمُونَ ﴿41﴾
نکلو، خواہ ہلکے ہو یا بھاری، اور اللہ کی راہ میں اپنے مالوں اور جانوں سے جہاد کرو، یہ تمہارے لیے بہتر ہے اگر تم جانتے ہو۔
لَوْ كَانَ عَرَضٗا قَرِيبٗا وَسَفَرٗا قَاصِدٗا لَّٱتَّبَعُوكَ وَلَـٰكِنۢ بَعُدَتْ عَلَيْهِمُ ٱلشُّقَّةُۚ وَسَيَحْلِفُونَ بِٱللَّهِ لَوِ ٱسْتَطَعْنَا لَخَرَجْنَا مَعَكُمْۚ يُهْلِكُونَ أَنفُسَهُمْۖ وَٱللَّهُ يَعْلَمُ إِنَّهُمْ لَكَـٰذِبُونَ ﴿42﴾
اگر فائدہ قریب ہوتا اور سفر آسان ہوتا تو وہ ضرور تمہارا ساتھ دیتے، لیکن ان پر مشقت دور معلوم ہوئی، اور وہ اللہ کی قسمیں کھائیں گے: اگر ہم استطاعت رکھتے تو ضرور تمہارے ساتھ نکلتے، وہ اپنے آپ کو ہلاک کر رہے ہیں، اور اللہ جانتا ہے کہ وہ یقیناً جھوٹے ہیں۔
عَفَا ٱللَّهُ عَنكَ لِمَ أَذِنتَ لَهُمْ حَتَّىٰ يَتَبَيَّنَ لَكَ ٱلَّذِينَ صَدَقُوا۟ وَتَعْلَمَ ٱلْكَـٰذِبِينَ ﴿43﴾
اللہ نے تمہیں معاف کر دیا، تم نے انہیں کیوں اجازت دی؟ یہاں تک کہ تمہیں معلوم ہو جاتا کہ کون سچے ہیں اور تم جھوٹوں کو جان لیتے۔
لَا يَسْتَـْٔذِنُكَ ٱلَّذِينَ يُؤْمِنُونَ بِٱللَّهِ وَٱلْيَوْمِ ٱلْـَٔاخِرِ أَن يُجَـٰهِدُوا۟ بِأَمْوَٰلِهِمْ وَأَنفُسِهِمۚ وَٱللَّهُ عَلِيمُۢ بِٱلْمُتَّقِينَ ﴿44﴾
وہ لوگ جو اللہ اور آخرت کے دن پر ایمان رکھتے ہیں، وہ تم سے اجازت نہیں مانگتے کہ وہ اپنے مالوں اور جانوں سے جہاد نہ کریں، اور اللہ متقیوں کو خوب جانتا ہے۔
إِنَّمَا يَسْتَـْٔذِنُكَ ٱلَّذِينَ لَا يُؤْمِنُونَ بِٱللَّهِ وَٱلْيَوْمِ ٱلْـَٔاخِرِ وَٱرْتَابَتْ قُلُوبُهُمْ فَهُمْ فِي رَيْبِهِمْ يَتَرَدَّدُونَ ﴿45﴾
تم سے وہی لوگ اجازت مانگتے ہیں جو اللہ اور آخرت کے دن پر ایمان نہیں رکھتے، اور ان کے دل شک میں پڑے ہوئے ہیں، پس وہ اپنے شک میں ڈگمگاتے رہتے ہیں۔
وَلَوْ أَرَادُوا۟ ٱلْخُرُوجَ لَأَعَدُّوا۟ لَهُۥ عُدَّةٗ وَلَـٰكِن كَرِهَ ٱللَّهُ ٱنۢبِعَاثَهُمْ فَثَبَّطَهُمْ وَقِيلَ ٱقْعُدُوا۟ مَعَ ٱلْقَـٰعِدِينَ ﴿46﴾
اور اگر وہ نکلنے کا ارادہ رکھتے تو اس کے لیے تیاری کرتے، لیکن اللہ نے ان کے نکلنے کو ناپسند کیا، پس انہیں روک دیا، اور کہا گیا: بیٹھے رہو بیٹھنے والوں کے ساتھ۔
لَوْ خَرَجُوا۟ فِيكُم مَّا زَادُوكُمْ إِلَّا خَبَالٗا وَلَأَوْضَعُوا۟ خِلَـٰلَكُمْ يَبْغُونَكُمُ ٱلْفِتْنَةَۚ وَفِيكُمْ سَمَّـٰعُونَ لَهُمْۗ وَٱللَّهُ عَلِيمُۢ بِٱلظَّـٰلِمِينَ ﴿47﴾
اگر وہ تمہارے ساتھ نکلتے تو تمہارے لیے فساد کے سوا کچھ نہ بڑھاتے، اور تمہارے درمیان فتنے کی تلاش میں دوڑ دھوپ کرتے، اور تم میں ان کے لیے سننے والے بھی ہیں، اور اللہ ظالموں کو خوب جانتا ہے۔
لَقَدِ ٱبْتَغَوُا۟ ٱلْفِتْنَةَ مِن قَبْلُ وَقَلَّبُوا۟ لَكَ ٱلْأُمُورَ حَتَّىٰ جَآءَ ٱلْحَقُّ وَظَهَرَ أَمْرُ ٱللَّهِ وَهُمْ كَـٰرِهُونَ ﴿48﴾
بے شک انہوں نے اس سے پہلے بھی فتنے کی کوشش کی، اور تمہارے لیے معاملات کو الٹ پلٹ کیا، یہاں تک کہ حق آ گیا اور اللہ کا حکم ظاہر ہو گیا، اور وہ اسے ناپسند کرتے تھے۔
وَمِنْهُم مَّن يَقُولُ ٱئْذَن لِّي وَلَا تَفْتِنِّيٓۚ أَلَا فِي ٱلْفِتْنَةِ سَقَطُوا۟ۗ وَإِنَّ جَهَنَّمَ لَمُحِيطَةُۢ بِٱلْكَـٰفِرِينَ ﴿49﴾
اور ان میں سے کچھ کہتے ہیں: مجھے اجازت دے دو اور مجھے فتنے میں نہ ڈالو، خبردار! وہ فتنے میں پڑ چکے ہیں، اور بے شک جہنم کافروں کو گھیرے ہوئے ہے۔
إِن تُصِبْكَ حَسَنَةٞ تَسُؤْهُمْ وَإِن تُصِبْكَ مُصِيبَةٞ يَقُولُوا۟ قَدْ أَخَذْنَآ أَمْرَنَا مِن قَبْلُ وَيَتَوَلَّوا۟ وَّهُمْ فَرِحُونَ ﴿50﴾
اگر تمہیں کوئی بھلائی پہنچے تو انہیں برا لگتا ہے، اور اگر تمہیں کوئی مصیبت پہنچے تو کہتے ہیں: ہم نے تو پہلے ہی اپنا معاملہ سنبھال لیا تھا، اور وہ پیٹھ پھیر کر خوشی سے چلے جاتے ہیں۔
قُل لَّن يُصِيبَنَآ إِلَّا مَا كَتَبَ ٱللَّهُ لَنَاۚ هُوَ مَوْلَىٰنَاۚ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُؤْمِنُونَ ﴿51﴾
کہہ دو: ہمیں وہی کچھ پہنچتا ہے جو اللہ نے ہمارے لیے لکھ دیا ہے، وہی ہمارا کارساز ہے، اور مؤمنوں کو اللہ ہی پر بھروسہ کرنا چاہیے۔
قُلْ هَلْ تَرَبَّصُونَ بِنَآ إِلَّآ إِحْدَى ٱلْحُسْنَيَيْنِۖ وَنَحْنُ نَتَرَبَّصُ بِكُمْ أَن يُصِيبَكُمُ ٱللَّهُ بِعَذَابٖ مِّنْ عِندِهِۦٓ أَوْ بِأَيْدِينَاۖ فَتَرَبَّصُوٓا۟ إِنَّا مَعَكُم مُّتَرَبِّصُونَ ﴿52﴾
کہہ دو: کیا تم ہمارے بارے میں صرف دو بھلائیوں میں سے ایک کا انتظار کر رہے ہو؟ اور ہم تمہارے بارے میں انتظار کر رہے ہیں کہ اللہ تمہیں اپنے پاس سے یا ہمارے ہاتھوں سے عذاب دے، پس انتظار کرو، ہم بھی تمہارے ساتھ انتظار کرنے والے ہیں۔
قُلْ أَنفِقُوا۟ طَوْعًا أَوْ كَرْهٗا لَّن يُتَقَبَّلَ مِنكُمْۖ إِنَّكُمْ كُنتُمْ قَوْمٗا فَـٰسِقِينَ ﴿53﴾
کہہ دو: خرچ کرو، خواہ خوشی سے یا ناخوشی سے، تم سے قبول نہیں کیا جائے گا، بے شک تم فاسق قوم ہو۔
وَمَا مَنَعَهُمْ أَن تُقْبَلَ مِنْهُمْ نَفَقَـٰتُهُمْ إِلَّآ أَنَّهُمْ كَفَرُوا۟ بِٱللَّهِ وَبِرَسُولِهِۦ وَلَا يَأْتُونَ ٱلصَّلَوٰةَ إِلَّا وَهُمْ كُسَالَىٰ وَلَا يُنفِقُونَ إِلَّا وَهُمْ كَـٰرِهُونَ ﴿54﴾
اور ان کی خرچ کی ہوئی چیزوں کے قبول نہ ہونے کی وجہ یہی ہے کہ انہوں نے اللہ اور اس کے رسول سے کفر کیا، اور نماز کو نہیں آتے مگر سستی کے ساتھ، اور خرچ نہیں کرتے مگر ناخوشی سے۔
فَلَا تُعْجِبْكَ أَمْوَٰلُهُمْ وَلَآ أَوْلَـٰدُهُمْۚ إِنَّمَا يُرِيدُ ٱللَّهُ لِيُعَذِّبَهُم بِهَا فِي ٱلْحَيَوٰةِ ٱلدُّنْيَا وَتَزْهَقَ أَنفُسُهُمْ وَهُمْ كَـٰفِرُونَ ﴿55﴾
پس ان کے مال اور اولاد تمہیں تعجب میں نہ ڈالیں، اللہ تو یہی چاہتا ہے کہ انہیں دنیا کی زندگی میں ان چیزوں کے ذریعے عذاب دے، اور ان کی جانیں نکلیں اس حال میں کہ وہ کافر ہوں۔
وَيَحْلِفُونَ بِٱللَّهِ إِنَّهُمْ لَمِنكُمْ وَمَا هُم مِّنكُمْ وَلَـٰكِنَّهُمْ قَوْمٞ يَفْرَقُونَ ﴿56﴾
اور وہ اللہ کی قسمیں کھاتے ہیں کہ وہ تم میں سے ہیں، حالانکہ وہ تم میں سے نہیں، بلکہ وہ ایسے لوگ ہیں جو ڈرتے ہیں۔
لَوْ يَجِدُونَ مَلْجَـٔٗا أَوْ مَغَـٰرَٰتٍ أَوْ مُدَّخَلٗا لَّوَلَّوْا۟ إِلَيْهِ وَهُمْ يَجْمَحُونَ ﴿57﴾
اگر وہ کوئی پناہ کی جگہ، یا غاریں، یا چھپنے کی جگہ پا لیں تو وہ اس کی طرف دوڑتے ہوئے چلے جائیں گے۔
وَمِنْهُم مَّن يَلْمِزُكَ فِي ٱلصَّدَقَـٰتِ فَإِنْ أُعْطُوا۟ مِنْهَا رَضُوا۟ وَإِن لَّمْ يُعْطَوْا۟ مِنْهَآ إِذَا هُمْ يَسْخَطُونَ ﴿58﴾
اور ان میں سے کچھ صدقات کے معاملے میں تم پر طعن کرتے ہیں، اگر انہیں اس میں سے دیا جائے تو راضی ہو جاتے ہیں، اور اگر نہ دیا جائے تو فوراً ناراض ہو جاتے ہیں۔
وَلَوْ أَنَّهُمْ رَضُوا۟ مَآ ءَاتَىٰهُمُ ٱللَّهُ وَرَسُولُهُۥ وَقَالُوا۟ حَسْبُنَا ٱللَّهُ سَيُؤْتِينَا ٱللَّهُ مِن فَضْلِهِۦ وَرَسُولُهُۥٓ إِنَّآ إِلَى ٱللَّهِ رَاغِبُونَ ﴿59﴾
اور اگر وہ اس پر راضی ہوتے جو اللہ اور اس کے رسول نے انہیں دیا، اور کہتے: ہمارے لیے اللہ کافی ہے، عنقریب اللہ ہمیں اپنے فضل سے دے گا اور اس کا رسول بھی، ہم تو اللہ ہی کی طرف رغبت رکھتے ہیں۔
إِنَّمَا ٱلصَّدَقَـٰتُ لِلْفُقَرَآءِ وَٱلْمَسَـٰكِينِ وَٱلْعَـٰمِلِينَ عَلَيْهَا وَٱلْمُؤَلَّفَةِ قُلُوبُهُمْ وَفِي ٱلرِّقَابِ وَٱلْغَـٰرِمِينَ وَفِي سَبِيلِ ٱللَّهِ وَٱبْنِ ٱلسَّبِيلِۖ فَرِيضَةٗ مِّنَ ٱللَّهِۗ وَٱللَّهُ عَلِيمٌ حَكِيمٌ ﴿60﴾
بے شک صدقات فقیروں، مسکینوں، ان پر کام کرنے والوں، جن کی دلجوئی مطلوب ہو، غلاموں کی آزادی، قرض داروں، اللہ کی راہ میں اور مسافروں کے لیے ہیں — یہ اللہ کی طرف سے مقرر کردہ فریضہ ہے، اور اللہ جاننے والا، حکمت والا ہے۔
وَمِنْهُمُ ٱلَّذِينَ يُؤْذُونَ ٱلنَّبِيَّ وَيَقُولُونَ هُوَ أُذُنٞۚ قُلْ أُذُنُ خَيْرٖ لَّكُمْ يُؤْمِنُ بِٱللَّهِ وَيُؤْمِنُ لِلْمُؤْمِنِينَ وَرَحْمَةٞ لِّلَّذِينَ ءَامَنُوا۟ مِنكُمْۚ وَٱلَّذِينَ يُؤْذُونَ رَسُولَ ٱللَّهِ لَهُمْ عَذَابٌ أَلِيمٌ ﴿61﴾
اور ان میں کچھ ایسے ہیں جو نبی کو اذیت دیتے ہیں اور کہتے ہیں: وہ تو کان ہے (جو ہر بات سن لیتا ہے)، کہہ دو: وہ تمہارے لیے بھلائی کا کان ہے، وہ اللہ پر ایمان رکھتا ہے اور مؤمنوں کی بات پر یقین کرتا ہے، اور تم میں سے ایمان والوں کے لیے رحمت ہے، اور جو لوگ رسول اللہ کو اذیت دیتے ہیں ان کے لیے دردناک عذاب ہے۔
يَحْلِفُونَ بِٱللَّهِ لَكُمْ لِيُرْضُوكُمْۚ وَٱللَّهُ وَرَسُولُهُۥٓ أَحَقُّ أَن يُرْضُوهُ إِن كَانُوا۟ مُؤْمِنِينَ ﴿62﴾
وہ تمہیں راضی کرنے کے لیے اللہ کی قسمیں کھاتے ہیں، حالانکہ اللہ اور اس کا رسول زیادہ حق دار ہیں کہ انہیں راضی کیا جائے اگر وہ مؤمن ہیں۔
أَلَمْ يَعْلَمُوٓا۟ أَنَّهُۥ مَن يُحَادِدِ ٱللَّهَ وَرَسُولَهُۥ فَأَنَّ لَهُۥ نَارَ جَهَنَّمَ خَـٰلِدٗا فِيهَاۚ ذَٰلِكَ ٱلْخِزْيُ ٱلْعَظِيمُ ﴿63﴾
کیا وہ نہیں جانتے کہ جو اللہ اور اس کے رسول کی مخالفت کرتا ہے، اس کے لیے جہنم کی آگ ہے، جس میں وہ ہمیشہ رہے گا؟ یہی بڑی رسوائی ہے۔
يَحْذَرُ ٱلْمُنَـٰفِقُونَ أَن تُنَزَّلَ عَلَيْهِمْ سُورَةٞ تُنَبِّئُهُم بِمَا فِي قُلُوبِهِمۚ قُلِ ٱسْتَهْزِءُوا۟ إِنَّ ٱللَّهَ مُخْرِجٞ مَّا تَحْذَرُونَ ﴿64﴾
منافق ڈرتے ہیں کہ ان پر کوئی ایسی سورت نازل ہو جائے جو انہیں ان کے دلوں کی باتیں بتا دے، کہہ دو: تم مذاق کرتے رہو، بے شک اللہ وہ چیز ظاہر کرنے والا ہے جس سے تم ڈرتے ہو۔
وَلَئِن سَأَلْتَهُمْ لَيَقُولُنَّ إِنَّمَا كُنَّا نَخُوضُ وَنَلْعَبُۚ قُلْ أَبِٱللَّهِ وَءَايَـٰتِهِۦ وَرَسُولِهِۦ كُنتُمْ تَسْتَهْزِءُونَ ﴿65﴾
اور اگر تم ان سے پوچھو تو ضرور کہیں گے: ہم تو صرف باتیں کر رہے تھے اور کھیل رہے تھے، کہہ دو: کیا تم اللہ، اس کی آیات اور اس کے رسول سے مذاق کرتے تھے؟
لَا تَعْتَذِرُوا۟ قَدْ كَفَرْتُم بَعْدَ إِيمَـٰنِكُمْۚ إِن نَّعْفُ عَن طَآئِفَةٖ مِّنكُمْ نُعَذِّبْ طَآئِفَةَۢ بِأَنَّهُمْ كَانُوا۟ مُجْرِمِينَ ﴿66﴾
عذر مت پیش کرو، تم ایمان کے بعد کفر کر چکے ہو، اگر ہم تم میں سے ایک گروہ کو معاف بھی کر دیں تو دوسرے گروہ کو ضرور عذاب دیں گے کیونکہ وہ مجرم تھے۔
ٱلْمُنَـٰفِقُونَ وَٱلْمُنَـٰفِقَـٰتُ بَعْضُهُم مِّنۢ بَعْضٖۚ يَأْمُرُونَ بِٱلْمُنكَرِ وَيَنْهَوْنَ عَنِ ٱلْمَعْرُوفِ وَيَقْبِضُونَ أَيْدِيَهُمْۚ نَسُوا۟ ٱللَّهَ فَنَسِيَهُمْۚ إِنَّ ٱلْمُنَـٰفِقِينَ هُمُ ٱلْفَـٰسِقُونَ ﴿67﴾
منافق مرد اور منافق عورتیں ایک دوسرے کے ہم جنس ہیں، برائی کا حکم دیتے ہیں اور بھلائی سے روکتے ہیں، اور اپنے ہاتھ بند رکھتے ہیں، انہوں نے اللہ کو بھلا دیا تو اللہ نے بھی انہیں بھلا دیا، بے شک منافق ہی فاسق ہیں۔
وَعَدَ ٱللَّهُ ٱلْمُنَـٰفِقِينَ وَٱلْمُنَـٰفِقَـٰتِ وَٱلْكُفَّارَ نَارَ جَهَنَّمَ خَـٰلِدِينَ فِيهَاۚ هِيَ حَسْبُهُمْۚ وَلَعَنَهُمُ ٱللَّهُۖ وَلَهُمْ عَذَابٞ مُّقِيمٞ ﴿68﴾
اللہ نے منافق مردوں، منافق عورتوں اور کافروں سے جہنم کی آگ کا وعدہ کیا ہے، جس میں وہ ہمیشہ رہیں گے، وہی ان کے لیے کافی ہے، اور اللہ نے ان پر لعنت کی ہے، اور ان کے لیے دائمی عذاب ہے۔
كَٱلَّذِينَ مِن قَبْلِكُمْ كَانُوٓا۟ أَشَدَّ مِنكُمْ قُوَّةٗ وَأَكْثَرَ أَمْوَٰلٗا وَأَوْلَـٰدٗا فَٱسْتَمْتَعُوا۟ بِخَلَـٰقِهِمْ فَٱسْتَمْتَعْتُم بِخَلَـٰقِكُمْ كَمَا ٱسْتَمْتَعَ ٱلَّذِينَ مِن قَبْلِكُم بِخَلَـٰقِهِمْ وَخُضْتُمْ كَٱلَّذِي خَاضُوٓا۟ۚ أُوْلَـٰٓئِكَ حَبِطَتْ أَعْمَـٰلُهُمْ فِي ٱلدُّنْيَا وَٱلْـَٔاخِرَةِۖ وَأُوْلَـٰٓئِكَ هُمُ ٱلْخَـٰسِرُونَ ﴿69﴾
تم ان لوگوں کی طرح ہو جو تم سے پہلے تھے، وہ تم سے زیادہ قوت والے، مال اور اولاد میں زیادہ تھے، پس انہوں نے اپنے حصے سے فائدہ اٹھایا، اور تم نے بھی اپنے حصے سے ویسے ہی فائدہ اٹھایا جیسے انہوں نے اٹھایا، اور تم نے بھی ویسی ہی باتیں کیں جیسی انہوں نے کیں، ان کے اعمال دنیا اور آخرت میں ضائع ہو گئے، اور وہی لوگ خسارے میں ہیں۔
أَلَمْ يَأْتِهِمْ نَبَأُ ٱلَّذِينَ مِن قَبْلِهِمْ قَوْمِ نُوحٖ وَعَادٖ وَثَمُودَ وَقَوْمِ إِبْرَٰهِيمَ وَأَصْحَـٰبِ مَدْيَنَ وَٱلْمُؤْتَفِكَـٰتِۚ أَتَتْهُمْ رُسُلُهُم بِٱلْبَيِّنَـٰتِۖ فَمَا كَانَ ٱللَّهُ لِيَظْلِمَهُمْ وَلَـٰكِن كَانُوٓا۟ أَنفُسَهُمْ يَظْلِمُونَ ﴿70﴾
کیا ان کے پاس ان لوگوں کی خبر نہیں آئی جو ان سے پہلے گزر چکے، نوح کی قوم، عاد، ثمود، ابراہیم کی قوم، مدین والے اور الٹی ہوئی بستیوں والے؟ ان کے پاس ان کے رسول واضح دلائل لے کر آئے، تو اللہ ایسا نہ تھا کہ ان پر ظلم کرتا، بلکہ وہ خود اپنی جانوں پر ظلم کرتے تھے۔
وَٱلْمُؤْمِنُونَ وَٱلْمُؤْمِنَـٰتُ بَعْضُهُمْ أَوْلِيَآءُ بَعْضٖۚ يَأْمُرُونَ بِٱلْمَعْرُوفِ وَيَنْهَوْنَ عَنِ ٱلْمُنكَرِ وَيُقِيمُونَ ٱلصَّلَوٰةَ وَيُؤْتُونَ ٱلزَّكَوٰةَ وَيُطِيعُونَ ٱللَّهَ وَرَسُولَهُۥٓۚ أُوْلَـٰٓئِكَ سَيَرْحَمُهُمُ ٱللَّهُۗ إِنَّ ٱللَّهَ عَزِيزٌ حَكِيمٌ ﴿71﴾
اور مؤمن مرد اور مؤمن عورتیں ایک دوسرے کے مددگار ہیں، وہ نیکی کا حکم دیتے ہیں اور برائی سے روکتے ہیں، اور نماز قائم کرتے ہیں، اور زکوٰۃ دیتے ہیں، اور اللہ اور اس کے رسول کی اطاعت کرتے ہیں، یہی لوگ ہیں جن پر اللہ رحم فرمائے گا، بے شک اللہ زبردست، حکمت والا ہے۔
وَعَدَ ٱللَّهُ ٱلْمُؤْمِنِينَ وَٱلْمُؤْمِنَـٰتِ جَنَّـٰتٖ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُ خَـٰلِدِينَ فِيهَا وَمَسَـٰكِنَ طَيِّبَةٗ فِي جَنَّـٰتِ عَدْنٖۚ وَرِضْوَٰنٗ مِّنَ ٱللَّهِ أَكْبَرُۚ ذَٰلِكَ هُوَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿72﴾
اللہ نے مؤمن مردوں اور مؤمن عورتوں سے وعدہ کیا ہے ایسی جنتوں کا جن کے نیچے نہریں بہتی ہیں، جن میں وہ ہمیشہ رہیں گے، اور پاکیزہ رہائش گاہوں کا عدن کی جنتوں میں، اور اللہ کی رضا سب سے بڑی چیز ہے، یہی ہے بڑی کامیابی۔
يَـٰٓأَيُّهَا ٱلنَّبِيُّ جَـٰهِدِ ٱلْكُفَّارَ وَٱلْمُنَـٰفِقِينَ وَٱغْلُظْ عَلَيْهِمْۚ وَمَأْوَىٰهُمْ جَهَنَّمُۖ وَبِئْسَ ٱلْمَصِيرُ ﴿73﴾
اے نبی! کافروں اور منافقوں سے جہاد کرو اور ان پر سختی کرو، اور ان کا ٹھکانہ جہنم ہے، اور وہ بہت ہی برا ٹھکانہ ہے۔
يَحْلِفُونَ بِٱللَّهِ مَا قَالُوا۟ وَلَقَدْ قَالُوا۟ كَلِمَةَ ٱلْكُفْرِ وَكَفَرُوا۟ بَعْدَ إِسْلَـٰمِهِمْ وَهَمُّوا۟ بِمَا لَمْ يَنَالُوا۟ۚ وَمَا نَقَمُوٓا۟ إِلَّآ أَنْ أَغْنَىٰهُمُ ٱللَّهُ وَرَسُولُهُۥ مِن فَضْلِهِۦۚ فَإِن يَتُوبُوا۟ يَكُ خَيْرٗا لَّهُمْۖ وَإِن يَتَوَلَّوْا۟ يُعَذِّبْهُمُ ٱللَّهُ عَذَابًا أَلِيمٗا فِي ٱلدُّنْيَا وَٱلْـَٔاخِرَةِۚ وَمَا لَهُمْ فِي ٱلْأَرْضِ مِن وَلِيّٖ وَلَا نَصِيرٖ ﴿74﴾
وہ اللہ کی قسم کھاتے ہیں کہ انہوں نے کچھ نہیں کہا، حالانکہ انہوں نے کفر کا کلمہ کہا، اور اسلام لانے کے بعد کفر کیا، اور ایسی بات کا ارادہ کیا جو وہ حاصل نہ کر سکے، اور انہوں نے صرف اس بات کا بدلہ لیا کہ اللہ اور اس کے رسول نے انہیں اپنے فضل سے غنی کر دیا، پس اگر وہ توبہ کر لیں تو ان کے لیے بہتر ہے، اور اگر منہ موڑیں تو اللہ انہیں دنیا اور آخرت میں دردناک عذاب دے گا، اور ان کے لیے زمین میں نہ کوئی دوست ہو گا اور نہ مددگار۔
وَمِنْهُم مَّنْ عَـٰهَدَ ٱللَّهَ لَئِنْ ءَاتَىٰنَا مِن فَضْلِهِۦ لَنَصَّدَّقَنَّ وَلَنَكُونَنَّ مِنَ ٱلصَّـٰلِحِينَ ﴿75﴾
اور ان میں سے کچھ ایسے ہیں جنہوں نے اللہ سے عہد کیا کہ اگر وہ ہمیں اپنے فضل سے دے گا تو ہم ضرور صدقہ دیں گے اور ضرور نیک لوگوں میں سے ہوں گے۔
فَلَمَّآ ءَاتَىٰهُم مِّن فَضْلِهِۦ بَخِلُوا۟ بِهِۦ وَتَوَلَّوا۟ وَّهُم مُّعْرِضُونَ ﴿76﴾
پھر جب اللہ نے انہیں اپنے فضل سے دیا تو وہ اس میں بخل کرنے لگے اور منہ موڑ گئے اور وہ اعراض کرنے والے تھے۔
فَأَعْقَبَهُمْ نِفَاقٗا فِي قُلُوبِهِمْ إِلَىٰ يَوْمِ يَلْقَوْنَهُۥ بِمَآ أَخْلَفُوا۟ ٱللَّهَ مَا وَعَدُوهُ وَبِمَا كَانُوا۟ يَكْذِبُونَ ﴿77﴾
تو اس نے ان کے دلوں میں نفاق ڈال دیا اس دن تک کہ وہ اس سے ملاقات کریں گے، اس وجہ سے کہ انہوں نے اللہ سے کیے ہوئے وعدے کو توڑا اور اس وجہ سے کہ وہ جھوٹ بولتے تھے۔
أَلَمْ يَعْلَمُوٓا۟ أَنَّ ٱللَّهَ يَعْلَمُ سِرَّهُمْ وَنَجْوَىٰهُمْ وَأَنَّ ٱللَّهَ عَلَّـٰمُ ٱلْغُيُوبِ ﴿78﴾
کیا وہ نہیں جانتے کہ اللہ ان کے راز اور ان کی سرگوشیاں جانتا ہے؟ اور بے شک اللہ غیبوں کا خوب جاننے والا ہے۔
ٱلَّذِينَ يَلْمِزُونَ ٱلْمُطَّوِّعِينَ مِنَ ٱلْمُؤْمِنِينَ فِي ٱلصَّدَقَـٰتِ وَٱلَّذِينَ لَا يَجِدُونَ إِلَّا جُهْدَهُمْ فَيَسْخَرُونَ مِنْهُمْۖ سَخِرَ ٱللَّهُ مِنْهُمْۖ وَلَهُمْ عَذَابٌ أَلِيمٌ ﴿79﴾
جو لوگ صدقات دینے والے مؤمنوں پر طعن کرتے ہیں، اور ان پر بھی جو اپنی استطاعت کے مطابق دیتے ہیں، اور ان کا مذاق اڑاتے ہیں — اللہ ان کا مذاق اڑاتا ہے، اور ان کے لیے دردناک عذاب ہے۔
ٱسْتَغْفِرْ لَهُمْ أَوْ لَا تَسْتَغْفِرْ لَهُمْۖ إِن تَسْتَغْفِرْ لَهُمْ سَبْعِينَ مَرَّةٗ فَلَن يَغْفِرَ ٱللَّهُ لَهُمْۚ ذَٰلِكَ بِأَنَّهُمْ كَفَرُوا۟ بِٱللَّهِ وَرَسُولِهِۦۚ وَٱللَّهُ لَا يَهْدِي ٱلْقَوْمَ ٱلْفَـٰسِقِينَ ﴿80﴾
تم ان کے لیے بخشش مانگو یا نہ مانگو، اگر تم ان کے لیے ستر بار بھی بخشش مانگو تو اللہ انہیں ہرگز نہیں بخشے گا، یہ اس لیے کہ انہوں نے اللہ اور اس کے رسول سے کفر کیا، اور اللہ فاسق قوم کو ہدایت نہیں دیتا۔
فَرِحَ ٱلْمُخَلَّفُونَ بِمَقْعَدِهِمْ خِلَـٰفَ رَسُولِ ٱللَّهِ وَكَرِهُوٓا۟ أَن يُجَـٰهِدُوا۟ بِأَمْوَٰلِهِمْ وَأَنفُسِهِمْ فِي سَبِيلِ ٱللَّهِ وَقَالُوا۟ لَا تَنفِرُوا۟ فِي ٱلْحَرِّۗ قُلْ نَارُ جَهَنَّمَ أَشَدُّ حَرّٗاۚ لَّوْ كَانُوا۟ يَفْقَهُونَ ﴿81﴾
پیچھے رہ جانے والے لوگ رسول اللہ کے پیچھے نہ جانے پر خوش ہوئے، اور اللہ کی راہ میں اپنے مالوں اور جانوں سے جہاد کو ناپسند کیا، اور کہنے لگے: گرمی میں نہ نکلو، کہہ دو: جہنم کی آگ اس سے زیادہ گرم ہے، کاش وہ سمجھتے۔
فَلْيَضْحَكُوا۟ قَلِيلٗا وَلْيَبْكُوا۟ كَثِيرٗا جَزَآءٗۢ بِمَا كَانُوا۟ يَكْسِبُونَ ﴿82﴾
پس وہ تھوڑا ہنسیں اور زیادہ روئیں، اس بدلے میں جو وہ کماتے رہے۔
فَإِن رَّجَعَكَ ٱللَّهُ إِلَىٰ طَآئِفَةٖ مِّنْهُمْ فَٱسْتَـْٔذَنُوكَ لِلْخُرُوجِ فَقُلْ لَّن تَخْرُجُوا۟ مَعِيَ أَبَدٗا وَلَن تُقَـٰتِلُوا۟ مَعِيَ عَدُوًّاۖ إِنَّكُمْ رَضِيتُم بِٱلْقُعُودِ أَوَّلَ مَرَّةٖ فَٱقْعُدُوا۟ مَعَ ٱلْخَـٰلِفِينَ ﴿83﴾
پھر اگر اللہ تمہیں ان میں سے کسی گروہ کی طرف واپس لے جائے اور وہ تم سے نکلنے کی اجازت مانگیں تو کہہ دو: تم میرے ساتھ کبھی نہیں نکلو گے اور میرے ساتھ کسی دشمن سے جنگ نہیں کرو گے، تم پہلی بار بیٹھنے پر راضی ہوئے تھے، پس اب پیچھے رہنے والوں کے ساتھ بیٹھے رہو۔
وَلَا تُصَلِّ عَلَىٰٓ أَحَدٖ مِّنْهُم مَّاتَ أَبَدٗا وَلَا تَقُمْ عَلَىٰ قَبْرِهِۦٓۖ إِنَّهُمْ كَفَرُوا۟ بِٱللَّهِ وَرَسُولِهِۦ وَمَاتُوا۟ وَهُمْ فَـٰسِقُونَ ﴿84﴾
اور ان میں سے جو مر جائے اس پر کبھی نماز نہ پڑھو اور نہ اس کی قبر پر کھڑے ہو، بے شک انہوں نے اللہ اور اس کے رسول سے کفر کیا، اور وہ فاسق ہونے کی حالت میں مرے۔
وَلَا تُعْجِبْكَ أَمْوَٰلُهُمْ وَأَوْلَـٰدُهُمْۚ إِنَّمَا يُرِيدُ ٱللَّهُ أَن يُعَذِّبَهُم بِهَا فِي ٱلدُّنْيَا وَتَزْهَقَ أَنفُسُهُمْ وَهُمْ كَـٰفِرُونَ ﴿85﴾
اور ان کے مال اور اولاد تمہیں تعجب میں نہ ڈالیں، اللہ تو یہی چاہتا ہے کہ انہیں ان چیزوں کے ذریعے دنیا میں عذاب دے، اور ان کی جانیں نکلیں اس حال میں کہ وہ کافر ہوں۔
وَإِذَآ أُنزِلَتْ سُورَةٌ أَنْ ءَامِنُوا۟ بِٱللَّهِ وَجَـٰهِدُوا۟ مَعَ رَسُولِهِ ٱسْتَـْٔذَنَكَ أُوْلُوا۟ ٱلطَّوْلِ مِنْهُمْ وَقَالُوا۟ ذَرْنَا نَكُن مَّعَ ٱلْقَـٰعِدِينَ ﴿86﴾
اور جب کوئی سورت نازل کی جاتی ہے کہ اللہ پر ایمان لاؤ اور اس کے رسول کے ساتھ جہاد کرو، تو ان میں سے صاحبِ حیثیت لوگ تم سے اجازت مانگتے ہیں اور کہتے ہیں: ہمیں چھوڑ دو کہ ہم بیٹھنے والوں کے ساتھ رہیں۔
رَضُوا۟ بِأَن يَكُونُوا۟ مَعَ ٱلْخَوَالِفِ وَطُبِعَ عَلَىٰ قُلُوبِهِمْ فَهُمْ لَا يَفْقَهُونَ ﴿87﴾
وہ اس پر راضی ہو گئے کہ پیچھے رہنے والیوں کے ساتھ رہیں، اور ان کے دلوں پر مہر لگا دی گئی، پس وہ سمجھتے نہیں۔
لَـٰكِنِ ٱلرَّسُولُ وَٱلَّذِينَ ءَامَنُوا۟ مَعَهُۥ جَـٰهَدُوا۟ بِأَمْوَٰلِهِمْ وَأَنفُسِهِمْۚ وَأُوْلَـٰٓئِكَ لَهُمُ ٱلْخَيْرَٰتُۖ وَأُوْلَـٰٓئِكَ هُمُ ٱلْمُفْلِحُونَ ﴿88﴾
لیکن رسول اور وہ لوگ جو اس کے ساتھ ایمان لائے، انہوں نے اپنے مالوں اور جانوں سے جہاد کیا، یہی لوگ ہیں جن کے لیے بھلائیاں ہیں، اور یہی کامیاب ہونے والے ہیں۔
أَعَدَّ ٱللَّهُ لَهُمْ جَنَّـٰتٗ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُ خَـٰلِدِينَ فِيهَاۚ ذَٰلِكَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿89﴾
اللہ نے ان کے لیے ایسی جنتیں تیار کی ہیں جن کے نیچے نہریں بہتی ہیں، جن میں وہ ہمیشہ رہیں گے، یہی ہے بڑی کامیابی۔
وَجَآءَ ٱلْمُعَذِّرُونَ مِنَ ٱلْأَعْرَابِ لِيُؤْذَنَ لَهُمْ وَقَعَدَ ٱلَّذِينَ كَذَبُوا۟ ٱللَّهَ وَرَسُولَهُۥۚ سَيُصِيبُ ٱلَّذِينَ كَفَرُوا۟ مِنْهُمْ عَذَابٌ أَلِيمٌ ﴿90﴾
اور دیہاتیوں میں سے عذر پیش کرنے والے آئے تاکہ انہیں اجازت دی جائے، اور وہ لوگ بیٹھ رہے جنہوں نے اللہ اور اس کے رسول سے جھوٹ بولا، ان میں سے کافروں کو عنقریب دردناک عذاب پہنچے گا۔
لَّيْسَ عَلَى ٱلضُّعَفَآءِ وَلَا عَلَى ٱلْمَرْضَىٰ وَلَا عَلَى ٱلَّذِينَ لَا يَجِدُونَ مَا يُنفِقُونَ حَرَجٌ إِذَا نَصَحُوا۟ لِلَّهِ وَرَسُولِهِۦۚ مَا عَلَى ٱلْمُحْسِنِينَ مِن سَبِيلٖۚ وَٱللَّهُ غَفُورٞ رَّحِيمٌ ﴿91﴾
کمزوروں، بیماروں اور ان لوگوں پر کوئی گناہ نہیں جو خرچ کرنے کے لیے کچھ نہیں پاتے، جب کہ وہ اللہ اور اس کے رسول کے لیے خلوص رکھیں، نیک لوگوں پر کوئی الزام نہیں، اور اللہ بخشنے والا، رحم کرنے والا ہے۔
وَلَا عَلَى ٱلَّذِينَ إِذَا مَآ أَتَوْكَ لِتَحْمِلَهُمْ قُلْتَ لَآ أَجِدُ مَآ أَحْمِلُكُمْ عَلَيْهِ تَوَلَّوا۟ وَّأَعْيُنُهُمْ تَفِيضُ مِنَ ٱلدَّمْعِ حَزَنًا أَلَّا يَجِدُوا۟ مَا يُنفِقُونَ ﴿92﴾
اور ان پر بھی کوئی الزام نہیں جو تمہارے پاس آئے کہ تم انہیں سواری دو، اور تم کہو: میرے پاس وہ چیز نہیں جس پر تمہیں سوار کروں، تو وہ واپس لوٹتے ہیں اور ان کی آنکھیں آنسو بہاتی ہیں اس غم میں کہ وہ خرچ کرنے کے لیے کچھ نہیں پاتے۔
إِنَّمَا ٱلسَّبِيلُ عَلَى ٱلَّذِينَ يَسْتَـْٔذِنُونَكَ وَهُمْ أَغْنِيَآءُۚ رَضُوا۟ بِأَن يَكُونُوا۟ مَعَ ٱلْخَوَالِفِۖ وَطَبَعَ ٱللَّهُ عَلَىٰ قُلُوبِهِمْ فَهُمْ لَا يَعْلَمُونَ ﴿93﴾
الزام تو ان پر ہے جو تم سے اجازت مانگتے ہیں حالانکہ وہ مالدار ہیں، وہ اس پر راضی ہو گئے کہ پیچھے رہنے والوں کے ساتھ رہیں، اور اللہ نے ان کے دلوں پر مہر لگا دی ہے، پس وہ نہیں جانتے۔
