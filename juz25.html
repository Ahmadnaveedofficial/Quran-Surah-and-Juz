25 پارہ


بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
إِلَيْهِ يُرَدُّ عِلْمُ ٱلسَّاعَةِۚ وَمَا تَخْرُجُ مِن ثَمَرَٰتٖ مِّنْ أَكْمَامِهَا وَلَا تَحْمِلُ مِنْ أُنثَىٰ وَلَا تَضَعُ إِلَّا بِعِلْمِهِۦۚ وَيَوْمَ يُنَادِيهِمْ أَيْنَ شُرَكَآءِي قَالُوٓا۟ ءَاذَنَّـٰكَ مَا مِنَّا مِن شَهِيدٖ ﴿47﴾
اسی کی طرف قیامت کا علم لوٹایا جاتا ہے، اور کوئی پھل اپنے غلاف سے نہیں نکلتا، اور کوئی مادہ حاملہ نہیں ہوتی اور نہ وضع حمل کرتی ہے مگر اس کے علم سے۔ اور جس دن وہ انہیں پکارے گا: کہاں ہیں میرے شریک؟ وہ کہیں گے: ہم نے آپ کو بتا دیا، ہم میں سے کوئی گواہ نہیں۔
وَضَلَّ عَنْهُم مَّا كَانُوا۟ يَدْعُونَ مِن قَبْلُ وَظَنُّوا۟ مَا لَهُم مِّن مَّحِيصٖ ﴿48﴾
اور وہ سب کچھ ان سے گم ہو جائے گا جسے وہ پہلے پکارا کرتے تھے، اور وہ سمجھ جائیں گے کہ ان کے لیے کوئی بچاؤ نہیں۔
لَا يَسْـَٔمُ ٱلْإِنسَـٰنُ مِن دُعَآءِ ٱلْخَيْرِ وَإِن مَّسَّهُ ٱلشَّرُّ فَيَـُٔوسٞ قَنُوطٞ ﴿49﴾
انسان بھلائی کی دعا مانگنے سے نہیں تھکتا، اور اگر اسے برائی پہنچے تو وہ مایوس اور ناامید ہو جاتا ہے۔
وَلَئِنْ أَذَقْنَـٰهُ رَحْمَةٗ مِّنَّا بَعْدَ ضَرَّآءَ مَسَّتْهُ لَيَقُولَنَّ هَـٰذَا لِي وَمَآ أَظُنُّ ٱلسَّاعَةَ قَآئِمَةٗۖ وَلَئِن رُّجِعْتُ إِلَىٰ رَبِّيٓ إِنَّ لِي عِندَهُۥ لَلْحُسْنَىٰۚ فَلَنُنَبِّئَنَّ ٱلَّذِينَ كَفَرُوا۟ بِمَا عَمِلُوا۟ وَلَنُذِيقَنَّهُم مِّنْ عَذَابٍ غَلِيظٖ ﴿50﴾
اور اگر ہم اسے اپنی طرف سے رحمت چکھا دیں اس مصیبت کے بعد جو اسے پہنچی تھی، تو وہ ضرور کہے گا: یہ تو میرے لیے ہے، اور میں نہیں سمجھتا کہ قیامت قائم ہوگی، اور اگر میں اپنے رب کی طرف لوٹایا بھی گیا تو میرے لیے وہاں بھی بھلائی ہوگی۔ پس ہم ضرور کافروں کو ان کے اعمال سے آگاہ کریں گے اور انہیں سخت عذاب چکھائیں گے۔
وَإِذَآ أَنْعَمْنَا عَلَى ٱلْإِنسَـٰنِ أَعْرَضَ وَنَـَٔا بِجَانِبِهِۦۖ وَإِذَا مَسَّهُ ٱلشَّرُّ فَذُو دُعَآءٍ عَرِيضٖ ﴿51﴾
اور جب ہم انسان پر انعام کرتے ہیں تو وہ منہ موڑ لیتا ہے اور پہلو پھیر لیتا ہے، اور جب اسے تکلیف پہنچتی ہے تو لمبی چوڑی دعائیں کرنے لگتا ہے۔
قُلْ أَرَءَيْتُمْ إِن كَانَ مِنْ عِندِ ٱللَّهِ ثُمَّ كَفَرْتُم بِهِۦ مَنْ أَضَلُّ مِمَّنْ هُوَ فِي شِقَاقِۭ بَعِيدٖ ﴿52﴾
کہہ دو: بتاؤ اگر یہ اللہ کی طرف سے ہو اور تم اس کا انکار کرو، تو اس شخص سے بڑھ کر کون گمراہ ہوگا جو سخت مخالفت میں پڑا ہو؟
سَنُرِيهِمْ ءَايَـٰتِنَا فِي ٱلْأٓفَاقِ وَفِيٓ أَنفُسِهِمْ حَتَّىٰ يَتَبَيَّنَ لَهُمْ أَنَّهُ ٱلْحَقُّۗ أَوَلَمْ يَكْفِ بِرَبِّكَ أَنَّهُۥ عَلَىٰ كُلِّ شَيْءٖ شَهِيدٌ ﴿53﴾
ہم انہیں اپنی نشانیاں آفاق میں اور ان کے نفسوں میں دکھائیں گے یہاں تک کہ ان پر واضح ہو جائے گا کہ وہی حق ہے۔ کیا آپ کے رب کی یہ بات کافی نہیں کہ وہ ہر چیز پر گواہ ہے؟
أَلَآ إِنَّهُمْ فِي مِرْيَةٖ مِّن لِّقَآءِ رَبِّهِمۡۗ أَلَآ إِنَّهُۥ بِكُلِّ شَيْءٖ مُّحِيطُۢ ﴿54﴾
خبردار! وہ اپنے رب سے ملاقات کے بارے میں شک میں ہیں، خبردار! بے شک وہ ہر چیز کو گھیرے ہوئے ہے۔

عٓسٓقٓ
كَذَٰلِكَ يُوحِيٓ إِلَيْكَ وَإِلَى ٱلَّذِينَ مِن قَبْلِكَ ٱللَّهُ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿3﴾
اسی طرح اللہ، جو زبردست، حکمت والا ہے، آپ کی طرف اور آپ سے پہلے والوں کی طرف وحی کرتا ہے۔
لَهُۥ مَا فِي ٱلسَّمَـٰوَٰتِ وَمَا فِي ٱلْأَرْضِ وَهُوَ ٱلْعَلِيُّ ٱلْعَظِيمُ ﴿4﴾
اسی کا ہے جو کچھ آسمانوں میں ہے اور جو کچھ زمین میں ہے، اور وہی بلند مرتبہ، عظمت والا ہے۔
تَكَادُ ٱلسَّمَـٰوَٰتُ يَتَفَطَّرْنَ مِن فَوْقِهِنَّۚ وَٱلْمَلَـٰٓئِكَةُ يُسَبِّحُونَ بِحَمْدِ رَبِّهِمْ وَيَسْتَغْفِرُونَ لِمَنْ فِي ٱلْأَرْضِۚ أَلَآ إِنَّ ٱللَّهَ هُوَ ٱلْغَفُورُ ٱلرَّحِيمُ ﴿5﴾
قریب ہے کہ آسمان اوپر سے پھٹ پڑیں، اور فرشتے اپنے رب کی حمد کے ساتھ تسبیح کرتے ہیں اور زمین والوں کے لیے بخشش مانگتے ہیں، خبردار! بے شک اللہ ہی بخشنے والا، رحم کرنے والا ہے۔
وَٱلَّذِينَ ٱتَّخَذُوا۟ مِن دُونِهِۦٓ أَوْلِيَآءَ ٱللَّهُ حَفِيظٌ عَلَيْهِمْ وَمَآ أَنتَ عَلَيْهِم بِوَكِيلٍ ﴿6﴾
اور جنہوں نے اس کے سوا مددگار بنا رکھے ہیں، اللہ ان پر نگہبان ہے، اور آپ ان پر نگہبان نہیں ہیں۔
وَكَذَٰلِكَ أَوْحَيْنَآ إِلَيْكَ قُرْءَانًا عَرَبِيّٗا لِّتُنذِرَ أُمَّ ٱلْقُرَىٰ وَمَنْ حَوْلَهَا وَتُنذِرَ يَوْمَ ٱلْجَمْعِ لَا رَيْبَ فِيهِۚ فَرِيقٞ فِي ٱلْجَنَّةِ وَفَرِيقٞ فِي ٱلسَّعِيرِ ﴿7﴾
اور اسی طرح ہم نے آپ کی طرف عربی قرآن وحی کیا تاکہ آپ مکہ والوں اور اس کے اردگرد والوں کو ڈرائیں، اور جمع ہونے والے دن سے ڈرائیں جس میں کوئی شک نہیں، ایک گروہ جنت میں ہوگا اور ایک گروہ جہنم میں۔
وَلَوْ شَآءَ ٱللَّهُ لَجَعَلَهُمْ أُمَّةٗ وَٰحِدَةٗ وَلَـٰكِن يُدْخِلُ مَن يَشَآءُ فِي رَحْمَتِهِۦۚ وَٱلظَّـٰلِمُونَ مَا لَهُم مِّن وَلِيّٖ وَلَا نَصِيرٍ ﴿8﴾
اور اگر اللہ چاہتا تو انہیں ایک ہی امت بنا دیتا، لیکن وہ جسے چاہتا ہے اپنی رحمت میں داخل کرتا ہے، اور ظالموں کے لیے نہ کوئی دوست ہے اور نہ مددگار۔
أَمِ ٱتَّخَذُوا۟ مِن دُونِهِۦٓ أَوْلِيَآءَ فَٱللَّهُ هُوَ ٱلْوَلِيُّ وَهُوَ يُحْيِ ٱلْمَوْتَىٰ وَهُوَ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٞ ﴿9﴾
کیا انہوں نے اس کے سوا مددگار بنا لیے ہیں؟ پس اللہ ہی مددگار ہے، اور وہی مردوں کو زندہ کرتا ہے، اور وہ ہر چیز پر قدرت رکھتا ہے۔
وَمَا ٱخْتَلَفْتُمْ فِيهِ مِن شَيْءٖ فَحُكْمُهُۥٓ إِلَى ٱللَّهِۚ ذَٰلِكُمُ ٱللَّهُ رَبِّي عَلَيْهِ تَوَكَّلْتُ وَإِلَيْهِ أُنِيبُ ﴿10﴾
اور جس چیز میں تم اختلاف کرتے ہو، اس کا فیصلہ اللہ کے سپرد ہے، یہی اللہ میرا رب ہے، اسی پر میں نے بھروسہ کیا اور اسی کی طرف رجوع کرتا ہوں۔
فَاطِرُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ جَعَلَ لَكُم مِّنْ أَنفُسِكُمْ أَزْوَٰجٗا وَمِنَ ٱلْأَنْعَـٰمِ أَزْوَٰجٗا يَذْرَؤُكُمْ فِيهِۦۚ لَيْسَ كَمِثْلِهِۦ شَيْءٞۖ وَهُوَ ٱلسَّمِيعُ ٱلْبَصِيرُ ﴿11﴾
آسمانوں اور زمین کو پیدا کرنے والا، اس نے تمہارے لیے تمہاری جنس سے جوڑے بنائے، اور چوپایوں میں بھی جوڑے بنائے، وہ تمہیں اسی طرح پھیلاتا ہے۔ اس جیسا کوئی نہیں، اور وہی سننے والا، دیکھنے والا ہے۔
لَهُۥ مَقَالِيدُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۖ يَبْسُطُ ٱلرِّزْقَ لِمَن يَشَآءُ وَيَقْدِرُۚ إِنَّهُۥ بِكُلِّ شَيْءٍ عَلِيمٞ ﴿12﴾
اسی کے ہاتھ میں آسمانوں اور زمین کی کنجیاں ہیں، وہ جس کے لیے چاہے رزق کشادہ کرتا ہے اور تنگ کرتا ہے، بے شک وہ ہر چیز کو جاننے والا ہے۔
شَرَعَ لَكُم مِّنَ ٱلدِّينِ مَا وَصَّىٰ بِهِۦ نُوحٗا وَٱلَّذِيٓ أَوْحَيْنَآ إِلَيْكَ وَمَا وَصَّيْنَا بِهِۦٓ إِبْرَٰهِيمَ وَمُوسَىٰ وَعِيسَىٰٓ أَنْ أَقِيمُوا۟ ٱلدِّينَ وَلَا تَتَفَرَّقُوا۟ فِيهِۚ كَبُرَ عَلَى ٱلْمُشْرِكِينَ مَا تَدْعُوهُمْ إِلَيْهِۚ ٱللَّهُ يَجْتَبِيٓ إِلَيْهِ مَن يَشَآءُ وَيَهْدِيٓ إِلَيْهِ مَن يُنِيبُ ﴿13﴾
اس نے تمہارے لیے دین کا وہی راستہ مقرر کیا جس کا حکم اس نے نوح کو دیا، اور جس کی وحی ہم نے آپ کی طرف کی، اور جس کا حکم ہم نے ابراہیم، موسیٰ اور عیسیٰ کو دیا، کہ دین کو قائم رکھو اور اس میں فرق نہ ڈالو۔ مشرکوں پر وہ بات بھاری ہے جس کی طرف آپ انہیں بلا رہے ہیں۔ اللہ جسے چاہتا ہے اپنے لیے چن لیتا ہے، اور جو اس کی طرف رجوع کرتا ہے اسے ہدایت دیتا ہے۔
وَمَا تَفَرَّقُوٓا۟ إِلَّا مِنۢ بَعْدِ مَا جَآءَهُمُ ٱلْعِلْمُ بَغْيَۢا بَيْنَهُمْۚ وَلَوْلَا كَلِمَةٞ سَبَقَتْ مِن رَّبِّكَ إِلَىٰٓ أَجَلٖ مُّسَمّٗى لَّقُضِيَ بَيْنَهُمْۚ وَإِنَّ ٱلَّذِينَ أُورِثُوا۟ ٱلْكِتَـٰبَ مِنۢ بَعْدِهِمْ لَفِي شَكّٖ مِّنْهُ مُرِيبٖ ﴿14﴾
اور وہ اس وقت تک اختلاف نہ کرتے جب تک ان کے پاس علم نہ آ گیا، آپس کی ضد کی وجہ سے۔ اور اگر آپ کے رب کی طرف سے ایک مقررہ وقت تک بات نہ ہو چکی ہوتی تو ان کے درمیان فیصلہ کر دیا جاتا۔ اور جو لوگ ان کے بعد کتاب کے وارث بنائے گئے وہ اس کے بارے میں شک میں پڑے ہوئے ہیں۔
فَلِذَٰلِكَ فَٱدْعُ وَٱسْتَقِمْ كَمَآ أُمِرْتَۖ وَلَا تَتَّبِعْ أَهْوَآءَهُمْۖ وَقُلْ ءَامَنتُ بِمَآ أَنزَلَ ٱللَّهُ مِن كِتَـٰبٖۖ وَأُمِرْتُ لِأَعْدِلَ بَيْنَكُمُۖ ٱللَّهُ رَبُّنَا وَرَبُّكُمۡۖ لَنَآ أَعْمَـٰلُنَا وَلَكُمْ أَعْمَـٰلُكُمۡۖ لَا حُجَّةَ بَيْنَنَا وَبَيْنَكُمُ ٱللَّهُ يَجْمَعُ بَيْنَنَا وَإِلَيْهِ ٱلْمَصِيرُ ﴿15﴾
پس اسی کی طرف دعوت دو اور جیسے آپ کو حکم دیا گیا ہے، ثابت قدم رہو، اور ان کی خواہشات کی پیروی نہ کرو، اور کہہ دو: میں اس کتاب پر ایمان لایا ہوں جو اللہ نے نازل کی ہے، اور مجھے حکم دیا گیا ہے کہ تمہارے درمیان انصاف کروں۔ اللہ ہمارا رب ہے اور تمہارا رب بھی، ہمارے لیے ہمارے اعمال ہیں اور تمہارے لیے تمہارے اعمال، ہمارے اور تمہارے درمیان کوئی جھگڑا نہیں، اللہ ہم سب کو جمع کرے گا اور اسی کی طرف لوٹنا ہے۔
وَٱلَّذِينَ يُحَآجُّونَ فِي ٱللَّهِ مِنۢ بَعْدِ مَا ٱسْتُجِيبَ لَهُۥ حُجَّتُهُمْ دَاحِضَةٌ عِندَ رَبِّهِمۡۖ وَعَلَيْهِمْ غَضَبٞ وَلَهُمْ عَذَابٞ شَدِيدٌ ﴿16﴾
اور جو لوگ اللہ کے بارے میں جھگڑتے ہیں اس کے بعد کہ اس کی دعوت قبول کی جا چکی ہو، ان کی دلیل ان کے رب کے نزدیک باطل ہے، اور ان پر غضب ہے اور ان کے لیے سخت عذاب ہے۔
ٱللَّهُ ٱلَّذِيٓ أَنزَلَ ٱلْكِتَـٰبَ بِٱلْحَقِّ وَٱلْمِيزَانَۚ وَمَا يُدْرِيكَ لَعَلَّ ٱلسَّاعَةَ قَرِيبٞ ﴿17﴾
اللہ وہ ہے جس نے کتاب اور میزان حق کے ساتھ نازل کی، اور آپ کو کیا معلوم شاید قیامت قریب ہی ہو۔
يَسْتَعْجِلُ بِهَا ٱلَّذِينَ لَا يُؤْمِنُونَ بِهَاۖ وَٱلَّذِينَ ءَامَنُوا۟ مُشْفِقُونَ مِنْهَا وَيَعْلَمُونَ أَنَّهَا ٱلْحَقُّۗ أَلَآ إِنَّ ٱلَّذِينَ يُمَارُونَ فِي ٱلسَّاعَةِ لَفِي ضَلَـٰلِۭ بَعِيدٖ ﴿18﴾
جو لوگ اس پر ایمان نہیں رکھتے وہ اس کے لیے جلدی مچاتے ہیں، اور جو ایمان رکھتے ہیں وہ اس سے ڈرتے ہیں اور جانتے ہیں کہ وہ حق ہے۔ خبردار! جو لوگ قیامت کے بارے میں جھگڑتے ہیں وہ دور کی گمراہی میں ہیں۔
ٱللَّهُ لَطِيفُۢ بِعِبَادِهِۦ يَرْزُقُ مَن يَشَآءُۖ وَهُوَ ٱلْقَوِيُّ ٱلْعَزِيزُ ﴿19﴾
اللہ اپنے بندوں پر مہربان ہے، جسے چاہتا ہے رزق دیتا ہے، اور وہی طاقتور، زبردست ہے۔
مَن كَانَ يُرِيدُ حَرْثَ ٱلْـَٔاخِرَةِ نَزِدْ لَهُۥ فِي حَرْثِهِۦۖ وَمَن كَانَ يُرِيدُ حَرْثَ ٱلدُّنْيَا نُؤْتِهِۦ مِنْهَا وَمَا لَهُۥ فِي ٱلْـَٔاخِرَةِ مِن نَّصِيبٍ ﴿20﴾
جو شخص آخرت کی کھیتی چاہتا ہے ہم اس کے لیے اس کی کھیتی بڑھا دیتے ہیں، اور جو دنیا کی کھیتی چاہتا ہے ہم اسے اس میں سے دیتے ہیں، مگر اس کے لیے آخرت میں کوئی حصہ نہیں۔
أَمْ لَهُمْ شُرَكَـٰٓؤُا۟ شَرَعُوا۟ لَهُم مِّنَ ٱلدِّينِ مَا لَمْ يَأْذَنۢ بِهِ ٱللَّهُۚ وَلَوْلَا كَلِمَةُ ٱلْفَصْلِ لَقُضِيَ بَيْنَهُمْۗ وَإِنَّ ٱلظَّـٰلِمِينَ لَهُمْ عَذَابٌ أَلِيمٞ ﴿21﴾
کیا ان کے کچھ شریک ہیں جنہوں نے ان کے لیے دین میں وہ چیز مقرر کی جس کی اللہ نے اجازت نہیں دی؟ اور اگر فیصلہ کی بات نہ ہوتی تو ان کے درمیان فیصلہ کر دیا جاتا، اور بے شک ظالموں کے لیے دردناک عذاب ہے۔
تَرَى ٱلظَّـٰلِمِينَ مُشْفِقِينَ مِمَّا كَسَبُوا۟ وَهُوَ وَاقِعُۢ بِهِمۚ وَٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ فِي رَوْضَـٰتِ ٱلْجَنَّـٰتِۖ لَهُم مَّا يَشَآءُونَ عِندَ رَبِّهِمۡۚ ذَٰلِكَ هُوَ ٱلْفَضْلُ ٱلْكَبِيرُ ﴿22﴾
آپ ظالموں کو دیکھیں گے کہ وہ اپنے کیے سے ڈر رہے ہوں گے، اور وہ ان پر واقع ہو کر رہے گا۔ اور جو لوگ ایمان لائے اور نیک عمل کیے وہ جنت کے باغوں میں ہوں گے، ان کے لیے ان کے رب کے پاس وہ سب کچھ ہوگا جو وہ چاہیں گے، یہی بڑی کامیابی ہے۔
ذَٰلِكَ ٱلَّذِي يُبَشِّرُ ٱللَّهُ عِبَادَهُ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِۗ قُل لَآ أَسْـَٔلُكُمْ عَلَيْهِ أَجْرًا إِلَّا ٱلْمَوَدَّةَ فِي ٱلْقُرْبَىٰۗ وَمَن يَقْتَرِفْ حَسَنَةٗ نَّزِدْ لَهُۥ فِيهَا حُسْنًاۚ إِنَّ ٱللَّهَ غَفُورٞ شَكُورٌ ﴿23﴾
یہی وہ چیز ہے جس کی اللہ اپنے بندوں کو خوشخبری دیتا ہے جو ایمان لائے اور نیک عمل کیے۔ کہہ دو: میں تم سے اس پر کوئی اجر نہیں مانگتا، سوائے قرابت کی محبت کے۔ اور جو کوئی نیکی کرے ہم اس میں اس کے لیے خوبی بڑھا دیتے ہیں، بے شک اللہ بخشنے والا، قدر دان ہے۔
أَمْ يَقُولُونَ ٱفْتَرَىٰ عَلَى ٱللَّهِ كَذِبٗاۖ فَإِن يَشَإِ ٱللَّهُ يَخْتِمْ عَلَىٰ قَلْبِكَۚ وَيَمْحُ ٱللَّهُ ٱلْبَـٰطِلَ وَيُحِقُّ ٱلْحَقَّ بِكَلِمَـٰتِهِۦٓۚ إِنَّهُۥ عَلِيمُۢ بِذَاتِ ٱلصُّدُورِ ﴿24﴾
کیا وہ کہتے ہیں کہ اس نے اللہ پر جھوٹ باندھا ہے؟ اگر اللہ چاہے تو آپ کے دل پر مہر لگا دے، اور اللہ باطل کو مٹا دیتا ہے اور حق کو اپنے کلمات سے ثابت کرتا ہے، بے شک وہ سینوں کی بات جاننے والا ہے۔
وَهُوَ ٱلَّذِي يَقْبَلُ ٱلتَّوْبَةَ عَنْ عِبَادِهِۦ وَيَعْفُوا۟ عَنِ ٱلسَّيِّـَٔاتِ وَيَعْلَمُ مَا تَفْعَلُونَ ﴿25﴾
اور وہی ہے جو اپنے بندوں سے توبہ قبول کرتا ہے اور برائیوں کو معاف کرتا ہے، اور وہ جانتا ہے جو تم کرتے ہو۔
وَيَسْتَجِيبُ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ وَيَزِيدُهُم مِّن فَضْلِهِۦۚ وَٱلْكَـٰفِرُونَ لَهُمْ عَذَابٞ شَدِيدٞ ﴿26﴾
اور جو لوگ ایمان لائے اور نیک عمل کیے، اللہ ان کی دعا قبول کرتا ہے اور انہیں اپنے فضل سے بڑھاتا ہے، اور کافروں کے لیے سخت عذاب ہے۔
وَلَوْ بَسَطَ ٱللَّهُ ٱلرِّزْقَ لِعِبَادِهِۦ لَبَغَوْا۟ فِي ٱلْأَرْضِ وَلَـٰكِن يُنَزِّلُ بِقَدَرٖ مَّا يَشَآءُۚ إِنَّهُۥ بِعِبَادِهِۦ خَبِيرُۢ بَصِيرٞ ﴿27﴾
اور اگر اللہ اپنے بندوں کے لیے رزق کشادہ کر دیتا تو وہ زمین میں سرکشی کرتے، لیکن وہ جتنا چاہتا ہے اندازے سے نازل کرتا ہے، بے شک وہ اپنے بندوں سے خوب باخبر، دیکھنے والا ہے۔
وَهُوَ ٱلَّذِي يُنَزِّلُ ٱلْغَيْثَ مِنۢ بَعْدِ مَا قَنَطُوا۟ وَيَنشُرُ رَحْمَتَهُۥۚ وَهُوَ ٱلْوَلِيُّ ٱلْحَمِيدُ ﴿28﴾
اور وہی ہے جو بارش نازل کرتا ہے اس کے بعد کہ لوگ ناامید ہو چکے ہوتے ہیں، اور اپنی رحمت کو پھیلاتا ہے، اور وہی کارساز، قابلِ تعریف ہے۔
وَمِنْ ءَايَـٰتِهِۦ خَلْقُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَمَا بَثَّ فِيهِمَا مِن دَآبَّةٖۚ وَهُوَ عَلَىٰ جَمْعِهِمْ إِذَا يَشَآءُ قَدِيرٞ ﴿29﴾
اور اس کی نشانیوں میں سے آسمانوں اور زمین کی پیدائش ہے، اور جو جاندار اس میں پھیلا رکھے ہیں، اور وہ جب چاہے ان سب کو جمع کرنے پر قادر ہے۔
وَمَآ أَصَـٰبَكُم مِّن مُّصِيبَةٖ فَبِمَا كَسَبَتْ أَيْدِيكُمْ وَيَعْفُوا۟ عَن كَثِيرٖ ﴿30﴾
اور تمہیں جو مصیبت پہنچتی ہے وہ تمہارے اپنے ہاتھوں کے کیے کی وجہ سے ہے، اور وہ بہت کچھ معاف کر دیتا ہے۔
وَمَآ أَنتُم بِمُعْجِزِينَ فِي ٱلْأَرْضِۖ وَمَا لَكُم مِّن دُونِ ٱللَّهِ مِن وَلِيّٖ وَلَا نَصِيرٖ ﴿31﴾
اور تم زمین میں اللہ کو عاجز نہیں کر سکتے، اور اللہ کے سوا تمہارا کوئی مددگار اور حمایتی نہیں۔
وَمِنْ ءَايَـٰتِهِ ٱلْجَوَارِ فِي ٱلْبَحْرِ كَٱلْأَعْلَـٰمِ ﴿32﴾
اور اس کی نشانیوں میں سے ہیں سمندر میں پہاڑوں کی مانند چلنے والی کشتیاں۔
إِن يَشَأْ يُسْكِنِ ٱلرِّيحَ فَيَظْلَلْنَ رَوَاكِدَ عَلَىٰ ظَهْرِهِۦٓۚ إِنَّ فِي ذَٰلِكَ لَـَٔايَـٰتٖ لِّكُلِّ صَبَّارٖ شَكُورٍ ﴿33﴾
اگر وہ چاہے تو ہوا کو روک دے، تو وہ (کشتیاں) سمندر کی سطح پر ٹھہری رہ جائیں، بے شک اس میں ہر صبر کرنے والے اور شکر گزار کے لیے نشانیاں ہیں۔
أَوْ يُوبِقْهُنَّ بِمَا كَسَبُوٓا۟ وَيَعْفُ عَن كَثِيرٖ ﴿34﴾
یا وہ ان کو ان کے اعمال کے سبب غرق کر دے، اور بہت سے لوگوں کو معاف بھی کر دے۔
وَيَعْلَمَ ٱلَّذِينَ يُجَـٰدِلُونَ فِيٓ ءَايَـٰتِنَا مَا لَهُم مِّن مَّحِيصٖ ﴿35﴾
اور تاکہ وہ لوگ جان لیں جو ہماری آیات میں جھگڑتے ہیں کہ ان کے لیے کوئی بچاؤ نہیں۔
فَمَآ أُوتِيتُم مِّن شَيْءٖ فَمَتَـٰعُ ٱلْحَيَوٰةِ ٱلدُّنْيَاۚ وَمَا عِندَ ٱللَّهِ خَيْرٞ وَأَبْقَىٰ لِلَّذِينَ ءَامَنُوا۟ وَعَلَىٰ رَبِّهِمْ يَتَوَكَّلُونَ ﴿36﴾
پس جو کچھ تمہیں دیا گیا ہے وہ دنیا کی زندگی کا سامان ہے، اور جو اللہ کے پاس ہے وہ بہتر اور باقی رہنے والا ہے ان لوگوں کے لیے جو ایمان لائے اور اپنے رب پر بھروسہ کرتے ہیں۔
وَٱلَّذِينَ يَجْتَنِبُونَ كَبَـٰٓئِرَ ٱلْإِثْمِ وَٱلْفَوَٰحِشَ وَإِذَا مَا غَضِبُوا۟ هُمْ يَغْفِرُونَ ﴿37﴾
اور وہ لوگ جو بڑے گناہوں اور بے حیائی سے بچتے ہیں، اور جب غصہ آتا ہے تو معاف کر دیتے ہیں۔
وَٱلَّذِينَ ٱسْتَجَابُوا۟ لِرَبِّهِمْ وَأَقَامُوا۟ ٱلصَّلَوٰةَ وَأَمْرُهُمْ شُورَىٰ بَيْنَهُمْ وَمِمَّا رَزَقْنَـٰهُمْ يُنفِقُونَ ﴿38﴾
اور وہ لوگ جو اپنے رب کا حکم مانتے ہیں، اور نماز قائم کرتے ہیں، اور ان کے معاملات آپس کے مشورے سے طے ہوتے ہیں، اور جو کچھ ہم نے انہیں دیا ہے اس میں سے خرچ کرتے ہیں۔
وَٱلَّذِينَ إِذَآ أَصَابَهُمُ ٱلْبَغْيُ هُمْ يَنتَصِرُونَ ﴿39﴾
اور وہ لوگ کہ جب ان پر ظلم کیا جاتا ہے تو وہ بدلہ لیتے ہیں۔
وَجَزَٰٓؤُا۟ سَيِّئَةٖ سَيِّئَةٞ مِّثْلُهَاۖ فَمَنْ عَفَا وَأَصْلَحَ فَأَجْرُهُۥ عَلَى ٱللَّهِۚ إِنَّهُۥ لَا يُحِبُّ ٱلظَّـٰلِمِينَ ﴿40﴾
اور برائی کا بدلہ اسی جیسی برائی ہے، پھر جو معاف کرے اور اصلاح کرے تو اس کا اجر اللہ کے ذمہ ہے، بے شک وہ ظالموں کو پسند نہیں کرتا۔
وَلَمَنِ ٱنتَصَرَ بَعْدَ ظُلْمِهِۦ فَأُو۟لَـٰٓئِكَ مَا عَلَيْهِم مِّن سَبِيلٍ ﴿41﴾
اور جس نے ظلم کے بعد بدلہ لیا تو ایسے لوگوں پر کوئی الزام نہیں۔
إِنَّمَا ٱلسَّبِيلُ عَلَى ٱلَّذِينَ يَظْلِمُونَ ٱلنَّاسَ وَيَبْغُونَ فِي ٱلْأَرْضِ بِغَيْرِ ٱلْحَقِّۚ أُو۟لَـٰٓئِكَ لَهُمْ عَذَابٌ أَلِيمٞ ﴿42﴾
الزام تو ان لوگوں پر ہے جو لوگوں پر ظلم کرتے ہیں اور زمین میں ناحق سرکشی کرتے ہیں، یہی لوگ ہیں جن کے لیے دردناک عذاب ہے۔
وَلَمَن صَبَرَ وَغَفَرَ إِنَّ ذَٰلِكَ لَمِنْ عَزْمِ ٱلْأُمُورِ ﴿43﴾
اور جو صبر کرے اور معاف کر دے، بے شک یہ بڑے ہمت کے کاموں میں سے ہے۔
وَمَن يُضْلِلِ ٱللَّهُ فَمَا لَهُۥ مِن وَلِيّٖ مِّنۢ بَعْدِهِۦۚ وَتَرَى ٱلظَّـٰلِمِينَ لَمَّا رَأَوُا۟ ٱلْعَذَابَ يَقُولُونَ هَلْ إِلَىٰ مَرَدّٖ مِّن سَبِيلٍ ﴿44﴾
اور جسے اللہ گمراہ کر دے تو اس کے بعد اس کا کوئی مددگار نہیں، اور آپ ظالموں کو دیکھیں گے کہ جب وہ عذاب کو دیکھیں گے تو کہیں گے: کیا واپسی کا کوئی راستہ ہے؟
وَتَرَىٰهُمْ يُعْرَضُونَ عَلَيْهَا خَـٰشِعِينَ مِنَ ٱلذُّلِّ يَنظُرُونَ مِن طَرْفٍ خَفِيّٖۗ وَقَالَ ٱلَّذِينَ ءَامَنُوٓا۟ إِنَّ ٱلْخَاسِرِينَ ٱلَّذِينَ خَسِرُوٓا۟ أَنفُسَهُمْ وَأَهْلِيهِمْ يَوْمَ ٱلْقِيَـٰمَةِۗ أَلَآ إِنَّ ٱلظَّـٰلِمِينَ فِي عَذَابٖ مُّقِيمٖ ﴿45﴾
اور آپ انہیں دیکھیں گے کہ وہ جہنم پر پیش کیے جائیں گے، ذلت سے جھکے ہوئے، چھپی نظر سے دیکھتے ہوں گے، اور ایمان والے کہیں گے: بے شک نقصان اٹھانے والے وہ ہیں جنہوں نے قیامت کے دن اپنے آپ کو اور اپنے گھر والوں کو نقصان پہنچایا، خبردار! بے شک ظالم ہمیشہ کے عذاب میں ہوں گے۔
وَمَا كَانَ لَهُم مِّنْ أَوْلِيَآءَ يَنصُرُونَهُم مِّن دُونِ ٱللَّهِۗ وَمَن يُضْلِلِ ٱللَّهُ فَمَا لَهُۥ مِن سَبِيلٍ ﴿46﴾
اور ان کے لیے اللہ کے سوا کوئی مددگار نہ ہوگا جو ان کی مدد کرے، اور جسے اللہ گمراہ کر دے تو اس کے لیے کوئی راستہ نہیں۔
ٱسْتَجِيبُوا۟ لِرَبِّكُم مِّن قَبْلِ أَن يَأْتِيَ يَوْمٞ لَّا مَرَدَّ لَهُۥ مِنَ ٱللَّهِۚ مَا لَكُم مِّن مَّلْجَإٖ يَوْمَئِذٖ وَمَا لَكُم مِّن نَّكِيرٖ ﴿47﴾
اپنے رب کی بات مان لو اس سے پہلے کہ وہ دن آ جائے جسے اللہ کی طرف سے ٹالا نہیں جا سکتا، اس دن تمہارے لیے کوئی پناہ نہ ہوگی اور نہ ہی تم انکار کر سکو گے۔
فَإِنْ أَعْرَضُوا۟ فَمَآ أَرْسَلْنَـٰكَ عَلَيْهِمْ حَفِيظًاۖ إِنْ عَلَيْكَ إِلَّا ٱلْبَلَـٰغُۗ وَإِنَّآ إِذَآ أَذَقْنَا ٱلْإِنسَـٰنَ مِنَّا رَحْمَةٗ فَرِحَ بِهَاۖ وَإِن تُصِبْهُمْ سَيِّئَةُۢ بِمَا قَدَّمَتْ أَيْدِيهِمْ فَإِنَّ ٱلْإِنسَـٰنَ كَفُورٞ ﴿48﴾
پھر اگر وہ منہ موڑیں تو ہم نے آپ کو ان پر نگہبان بنا کر نہیں بھیجا، آپ پر صرف پہنچا دینا ہے، اور جب ہم انسان کو اپنی طرف سے رحمت چکھاتے ہیں تو وہ خوش ہو جاتا ہے، اور اگر انہیں ان کے اعمال کے سبب کوئی مصیبت پہنچے تو بے شک انسان بڑا ناشکرا ہے۔
لِلَّهِ مُلْكُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۚ يَخْلُقُ مَا يَشَآءُۚ يَهَبُ لِمَن يَشَآءُ إِنَـٰثٗا وَيَهَبُ لِمَن يَشَآءُ ٱلذُّكُورَ ﴿49﴾
آسمانوں اور زمین کی بادشاہی اللہ ہی کے لیے ہے، وہ جو چاہے پیدا کرتا ہے، جسے چاہے بیٹیاں عطا کرتا ہے اور جسے چاہے بیٹے عطا کرتا ہے۔
أَوْ يُزَوِّجُهُمْ ذُكْرَانٗا وَإِنَـٰثٗا وَيَجْعَلُ مَن يَشَآءُ عَقِيمًاۚ إِنَّهُۥ عَلِيمٞ قَدِيرٞ ﴿50﴾
یا انہیں بیٹے اور بیٹیاں دونوں عطا کرتا ہے، اور جسے چاہے بانجھ بنا دیتا ہے، بے شک وہ جاننے والا، قدرت والا ہے۔
وَمَا كَانَ لِبَشَرٍ أَن يُكَلِّمَهُ ٱللَّهُ إِلَّا وَحۡيًا أَوۡ مِن وَرَآءِ حِجَابٍ أَوۡ يُرۡسِلَ رَسُولٗا فَيُوحِيَ بِإِذۡنِهِۦ مَا يَشَآءُۚ إِنَّهُۥ عَلِيٌّ حَكِيمٞ ﴿51﴾
کسی بشر کے لیے ممکن نہیں کہ اللہ اس سے بات کرے مگر وحی کے ذریعے، یا پردے کے پیچھے سے، یا وہ کوئی رسول بھیجے، پھر وہ اللہ کے حکم سے وحی کرے جو وہ چاہے، بے شک وہ بلند مرتبہ، حکمت والا ہے۔
وَكَذَٰلِكَ أَوْحَيْنَآ إِلَيْكَ رُوحٗا مِّنْ أَمْرِنَاۚ مَا كُنتَ تَدْرِي مَا ٱلْكِتَـٰبُ وَلَا ٱلْإِيمَـٰنُ وَلَـٰكِن جَعَلْنَـٰهُ نُورٗا نَّهْدِي بِهِۦ مَن نَّشَآءُ مِنْ عِبَادِنَاۚ وَإِنَّكَ لَتَهْدِيٓ إِلَىٰ صِرَٰطٖ مُّسْتَقِيمٖ ﴿52﴾
اور اسی طرح ہم نے آپ کی طرف اپنے حکم سے ایک روح (وحی) نازل کی، آپ نہیں جانتے تھے کہ کتاب کیا ہے اور ایمان کیا ہے، لیکن ہم نے اسے نور بنایا، جس کے ذریعے ہم اپنے بندوں میں سے جسے چاہیں ہدایت دیتے ہیں، اور بے شک آپ سیدھے راستے کی طرف رہنمائی کرتے ہیں۔
صِرَٰطِ ٱللَّهِ ٱلَّذِي لَهُۥ مَا فِي ٱلسَّمَـٰوَٰتِ وَمَا فِي ٱلْأَرْضِۗ أَلَآ إِلَى ٱللَّهِ تَصِيرُ ٱلْأُمُورُ ﴿53﴾
اللہ کے راستے کی طرف، جس کے لیے ہے جو کچھ آسمانوں اور زمین میں ہے، خبردار! تمام امور اللہ ہی کی طرف لوٹتے ہیں۔
سورة الزخرف
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
حمٓ ﴿1﴾
حٰمٓ
وَٱلْكِتَـٰبِ ٱلْمُبِينِ ﴿2﴾
قسم ہے واضح کتاب کی۔
إِنَّا جَعَلْنَـٰهُ قُرْءَانًا عَرَبِيّٗا لَّعَلَّكُمْ تَعْقِلُونَ ﴿3﴾
بے شک ہم نے اسے عربی قرآن بنایا تاکہ تم سمجھو۔
وَإِنَّهُۥ فِيٓ أُمِّ ٱلْكِتَـٰبِ لَدَيْنَا لَعَلِيٌّ حَكِيمٌ ﴿4﴾
اور بے شک وہ ہمارے پاس اصل کتاب میں بلند مرتبہ اور حکمت والا ہے۔
أَفَنَضْرِبُ عَنكُمُ ٱلذِّكْرَ صَفْحًا أَن كُنتُمْ قَوْمٗا مُّسْرِفِينَ ﴿5﴾
تو کیا ہم تم سے نصیحت کو یوں ہی ہٹا دیں اس لیے کہ تم حد سے بڑھنے والے لوگ ہو؟
وَكَمْ أَرْسَلْنَا مِن نَّبِيّٖ فِي ٱلْأَوَّلِينَ ﴿6﴾
اور ہم نے پہلے لوگوں میں کتنے ہی نبی بھیجے۔
وَمَا يَأْتِيهِم مِّن نَّبِيٍّ إِلَّا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ ﴿7﴾
اور ان کے پاس کوئی نبی نہیں آتا مگر وہ اس کا مذاق اڑاتے۔
فَأَهْلَكْنَآ أَشَدَّ مِنْهُم بَطْشٗا وَمَضَىٰ مَثَلُ ٱلْأَوَّلِينَ ﴿8﴾
تو ہم نے ان سے زیادہ سخت پکڑ والے کو ہلاک کر دیا، اور پہلے لوگوں کی مثال گزر چکی۔
وَلَئِن سَأَلْتَهُم مَّنْ خَلَقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ لَيَقُولُنَّ خَلَقَهُنَّ ٱلْعَزِيزُ ٱلْعَلِيمُ ﴿9﴾
اور اگر آپ ان سے پوچھیں کہ آسمانوں اور زمین کو کس نے پیدا کیا؟ تو ضرور کہیں گے: انہیں زبردست، جاننے والے نے پیدا کیا۔
ٱلَّذِي جَعَلَ لَكُمُ ٱلْأَرْضَ مَهْدٗا وَجَعَلَ لَكُمْ فِيهَا سُبُلٗا لَّعَلَّكُمْ تَهْتَدُونَ ﴿10﴾
جس نے تمہارے لیے زمین کو بچھونا بنایا اور اس میں تمہارے لیے راستے بنائے تاکہ تم راہ پا سکو۔
وَٱلَّذِي نَزَّلَ مِنَ ٱلسَّمَآءِ مَآءَۭ بِقَدَرٖ فَأَنشَرْنَا بِهِۦ بَلْدَةٗ مَّيْتٗاۚ كَذَٰلِكَ تُخْرَجُونَ ﴿11﴾
اور وہی ہے جس نے آسمان سے اندازے کے ساتھ پانی نازل کیا، پھر ہم نے اس سے مردہ زمین کو زندہ کیا، اسی طرح تم نکالے جاؤ گے۔
وَٱلَّذِي خَلَقَ ٱلْأَزْوَٰجَ كُلَّهَا وَجَعَلَ لَكُم مِّنَ ٱلْفُلْكِ وَٱلْأَنْعَـٰمِ مَا تَرْكَبُونَ ﴿12﴾
اور وہی ہے جس نے تمام جوڑے پیدا کیے، اور تمہارے لیے کشتیوں اور چوپایوں میں سے وہ چیزیں بنائیں جن پر تم سوار ہوتے ہو۔
لِتَسْتَوُۥا۟ عَلَىٰ ظُهُورِهِۦ ثُمَّ تَذْكُرُوا۟ نِعْمَةَ رَبِّكُمْ إِذَا ٱسْتَوَيْتُمْ عَلَيْهِۦ وَتَقُولُوا۟ سُبْحَـٰنَ ٱلَّذِي سَخَّرَ لَنَا هَـٰذَا وَمَا كُنَّا لَهُۥ مُقْرِنِينَ ﴿13﴾
تاکہ تم ان کی پشت پر بیٹھو، پھر جب تم ان پر بیٹھ جاؤ تو اپنے رب کی نعمت کو یاد کرو اور کہو: پاک ہے وہ جس نے ہمیں یہ مسخر کیا، ورنہ ہم اس پر قابو پانے والے نہ تھے۔
وَإِنَّآ إِلَىٰ رَبِّنَا لَمُنقَلِبُونَ ﴿14﴾
اور بے شک ہم اپنے رب کی طرف لوٹنے والے ہیں۔
وَجَعَلُوا۟ لَهُۥ مِنْ عِبَادِهِۦ جُزْءًاۚ إِنَّ ٱلْإِنسَـٰنَ لَكَفُورٞ مُّبِينٌ ﴿15﴾
اور انہوں نے اس کے بندوں میں سے اس کے لیے اولاد بنائی، بے شک انسان کھلا ناشکرا ہے۔
أَمِ ٱتَّخَذَ مِمَّا يَخْلُقُ بَنَاتٖ وَأَصْفَىٰكُم بِٱلْبَنِينَ ﴿16﴾
کیا اس نے اپنی مخلوق میں سے بیٹیاں لے لیں اور تمہیں بیٹوں کے ساتھ خاص کر دیا؟
وَإِذَا بُشِّرَ أَحَدُهُم بِمَا ضَرَبَ لِلرَّحْمَـٰنِ مَثَلٗا ظَلَّ وَجْهُهُۥ مُسْوَدّٗا وَهُوَ كَظِيمٌ ﴿17﴾
اور جب ان میں سے کسی کو اس چیز کی خوشخبری دی جائے جو اس نے رحمٰن کے لیے مثال بنائی، تو اس کا چہرہ سیاہ ہو جاتا ہے اور وہ غم سے بھر جاتا ہے۔
أَوَ مَن يُنَشَّؤُا۟ فِي ٱلْحِلْيَةِ وَهُوَ فِي ٱلْخِصَامِ غَيْرُ مُبِينٖ ﴿18﴾
کیا وہ جو زیور میں پرورش پاتی ہے اور بحث میں واضح نہیں ہوتی؟
وَجَعَلُوا۟ ٱلْمَلَـٰٓئِكَةَ ٱلَّذِينَ هُمْ عِبَادُ ٱلرَّحْمَـٰنِ إِنَـٰثًاۚ أَشَهِدُوا۟ خَلْقَهُمْۚ سَتُكْتَبُ شَهَـٰدَتُهُمْ وَيُسْـَٔلُونَ ﴿19﴾
اور انہوں نے فرشتوں کو، جو رحمٰن کے بندے ہیں، عورتیں بنا دیا، کیا وہ ان کی پیدائش کے وقت موجود تھے؟ ان کی گواہی لکھی جائے گی اور ان سے پوچھا جائے گا۔
وَقَالُوا۟ لَوْ شَآءَ ٱلرَّحْمَـٰنُ مَا عَبَدْنَـٰهُمۚ مَّا لَهُم بِذَٰلِكَ مِنْ عِلْمٍۖ إِنْ هُمْ إِلَّا يَخْرُصُونَ ﴿20﴾
اور وہ کہتے ہیں: اگر رحمٰن چاہتا تو ہم انہیں نہ پوجتے، انہیں اس کا کوئی علم نہیں، وہ صرف اندازے لگاتے ہیں۔
أَمْ ءَاتَيْنَـٰهُمْ كِتَـٰبٗا مِّن قَبْلِهِۦ فَهُم بِهِۦ مُسْتَمْسِكُونَ ﴿21﴾
کیا ہم نے انہیں اس سے پہلے کوئی کتاب دی ہے جس کو وہ مضبوطی سے تھامے ہوئے ہیں؟
بَلْ قَالُوٓا۟ إِنَّا وَجَدْنَآ ءَابَآءَنَا عَلَىٰٓ أُمَّةٖ وَإِنَّا عَلَىٰٓ ءَاثَـٰرِهِم مُّهْتَدُونَ ﴿22﴾
بلکہ وہ کہتے ہیں: ہم نے اپنے باپ دادا کو ایک طریقے پر پایا اور ہم انہی کے نقش قدم پر چلنے والے ہیں۔
وَكَذَٰلِكَ مَآ أَرْسَلْنَا مِن قَبْلِكَ فِي قَرْيَةٖ مِّن نَّذِيرٍ إِلَّا قَالَ مُتْرَفُوهَآ إِنَّا وَجَدْنَآ ءَابَآءَنَا عَلَىٰٓ أُمَّةٖ وَإِنَّا عَلَىٰٓ ءَاثَـٰرِهِم مُّقْتَدُونَ ﴿23﴾
اور اسی طرح ہم نے آپ سے پہلے جس بستی میں بھی کوئی ڈرانے والا بھیجا، وہاں کے خوشحال لوگوں نے یہی کہا: ہم نے اپنے باپ دادا کو ایک طریقے پر پایا اور ہم انہی کے نقش قدم پر چلنے والے ہیں۔
قَـٰلَ أَوَلَوْ جِئْتُكُم بِأَهْدَىٰ مِمَّا وَجَدتُّمْ عَلَيْهِ ءَابَآءَكُمۡۖ قَالُوٓا۟ إِنَّا بِمَآ أُرْسِلْتُم بِهِۦ كَـٰفِرُونَ ﴿24﴾
(نبی نے) کہا: کیا اگر میں تمہارے پاس اس سے بہتر ہدایت لے کر آؤں جس پر تم نے اپنے باپ دادا کو پایا؟ وہ بولے: ہم تو اس چیز کے منکر ہیں جس کے ساتھ تم بھیجے گئے ہو۔
فَٱنتَقَمْنَا مِنْهُمْ فَٱنظُرْ كَيْفَ كَانَ عَـٰقِبَةُ ٱلْمُكَذِّبِينَ ﴿25﴾
تو ہم نے ان سے انتقام لیا، سو دیکھو جھٹلانے والوں کا انجام کیسا ہوا۔
وَإِذْ قَالَ إِبْرَٰهِيمُ لِأَبِيهِ وَقَوْمِهِۦٓ إِنَّنِي بَرَآءٞ مِّمَّا تَعْبُدُونَ ﴿26﴾
اور جب ابراہیم نے اپنے باپ اور قوم سے کہا: بے شک میں ان چیزوں سے بیزار ہوں جنہیں تم پوجتے ہو۔
إِلَّا ٱلَّذِي فَطَرَنِي فَإِنَّهُۥ سَيَهْدِينِ ﴿27﴾
سوائے اس کے جس نے مجھے پیدا کیا، وہی مجھے ہدایت دے گا۔
وَجَعَلَهَا كَلِمَةَۢ بَاقِيَةٗ فِي عَقِبِهِۦ لَعَلَّهُمْ يَرْجِعُونَ ﴿28﴾
اور اس نے اسے اپنی نسل میں باقی رہنے والا کلمہ بنایا تاکہ وہ رجوع کریں۔
بَلْ مَتَّعْتُ هَـٰٓؤُلَآءِ وَءَابَآءَهُمْ حَتَّىٰ جَآءَهُمُ ٱلْحَقُّ وَرَسُولٞ مُّبِينٞ ﴿29﴾
بلکہ میں نے ان کو اور ان کے باپ دادا کو فائدہ پہنچایا یہاں تک کہ ان کے پاس حق اور واضح رسول آ گیا۔
وَلَمَّا جَآءَهُمُ ٱلْحَقُّ قَالُوا۟ هَـٰذَا سِحْرٞ وَإِنَّا بِهِۦ كَـٰفِرُونَ ﴿30﴾
اور جب ان کے پاس حق آیا تو انہوں نے کہا: یہ تو جادو ہے اور ہم اس کے منکر ہیں۔
وَقَالُوا۟ لَوْلَا نُزِّلَ هَـٰذَا ٱلْقُرْءَانُ عَلَىٰ رَجُلٖ مِّنَ ٱلْقَرْيَتَيْنِ عَظِيمٍ ﴿31﴾
اور انہوں نے کہا: یہ قرآن ان دو بستیوں کے کسی بڑے آدمی پر کیوں نہ نازل کیا گیا؟
أَهُمْ يَقْسِمُونَ رَحْمَةَ رَبِّكَۚ نَحْنُ قَسَمْنَـٰ بَيْنَهُم مَّعِيشَتَهُمْ فِي ٱلْحَيَوٰةِ ٱلدُّنْيَاۚ وَرَفَعْنَا بَعْضَهُمْ فَوْقَ بَعْضٖ دَرَجَـٰتٖ لِّيَتَّخِذَ بَعْضُهُم بَعْضٗا سُخْرِيّٗاۗ وَرَحْمَةُ رَبِّكَ خَيْرٞ مِّمَّا يَجْمَعُونَ ﴿32﴾
کیا وہ آپ کے رب کی رحمت تقسیم کرتے ہیں؟ ہم نے ان کے درمیان دنیا کی زندگی میں روزی تقسیم کی ہے، اور ہم نے ان میں سے بعض کو بعض پر درجے دیے تاکہ ایک دوسرے کو خدمت میں لگائیں، اور آپ کے رب کی رحمت اس سے بہتر ہے جو وہ جمع کرتے ہیں۔
وَلَوْلَآ أَن يَكُونَ ٱلنَّاسُ أُمَّةٗ وَٰحِدَةٗ لَّجَعَلْنَا لِمَن يَكْفُرُ بِٱلرَّحْمَـٰنِ لِبُيُوتِهِمْ سُقُفٗا مِّن فِضَّةٖ وَمَعَارِجَ عَلَيْهَا يَظْهَرُونَ ﴿33﴾
اور اگر یہ بات نہ ہوتی کہ سب لوگ ایک ہی امت بن جائیں گے، تو ہم رحمٰن کے منکروں کے گھروں کے لیے چاندی کی چھتیں بنا دیتے اور سیڑھیاں جن پر وہ چڑھتے۔
وَلِبُيُوتِهِمْ أَبْوَٰبٗا وَسُرُرًا عَلَيْهَا يَتَّكِـُٔونَ ﴿34﴾
اور ان کے گھروں کے لیے دروازے اور تخت جن پر وہ تکیہ لگاتے۔
وَزُخْرُفٗاۚ وَإِن كُلُّ ذَٰلِكَ لَمَّا مَتَـٰعُ ٱلْحَيَوٰةِ ٱلدُّنْيَاۚ وَٱلْـَٔاخِرَةُ عِندَ رَبِّكَ لِلْمُتَّقِينَ ﴿35﴾
اور سونے کے زیور، اور یہ سب دنیا کی زندگی کا سامان ہے، اور آخرت آپ کے رب کے پاس صرف پرہیزگاروں کے لیے ہے۔
وَمَن يَعْشُ عَن ذِكْرِ ٱلرَّحْمَـٰنِ نُقَيِّضْ لَهُۥ شَيْطَـٰنٗا فَهُوَ لَهُۥ قَرِينٞ ﴿36﴾
اور جو رحمٰن کے ذکر سے غافل ہوتا ہے، ہم اس کے لیے ایک شیطان مقرر کر دیتے ہیں، جو اس کا ساتھی ہوتا ہے۔
وَإِنَّهُمْ لَيَصُدُّونَهُمْ عَنِ ٱلسَّبِيلِ وَيَحْسَبُونَ أَنَّهُم مُّهْتَدُونَ ﴿37﴾
اور وہ انہیں راستے سے روکتے ہیں، اور وہ سمجھتے ہیں کہ وہ ہدایت پر ہیں۔
حَتَّىٰٓ إِذَا جَآءَنَا قَالَ يَـٰلَيْتَ بَيْنِي وَبَيْنَكَ بُعْدَ ٱلْمَشْرِقَيْنِ فَبِئْسَ ٱلْقَرِينُ ﴿38﴾
یہاں تک کہ جب وہ ہمارے پاس آئے گا، کہے گا: اے کاش میرے اور تیرے درمیان مشرق و مغرب کا فاصلہ ہوتا، تو برا ساتھی ہے۔
وَلَن يَنفَعَكُمُ ٱلْيَوْمَ إِذْ ظَلَمْتُمْ أَنَّكُمْ فِي ٱلْعَذَابِ مُشْتَرِكُونَ ﴿39﴾
اور آج تمہیں کوئی فائدہ نہ دے گا، جب تم نے ظلم کیا، کہ تم سب عذاب میں شریک ہو۔
أَفَأَنتَ تُسْمِعُ ٱلصُّمَّ أَوْ تَهْدِي ٱلْعُمْيَ وَمَن كَانَ فِي ضَلَـٰلٖ مُّبِينٖ ﴿40﴾
تو کیا آپ بہروں کو سنا سکتے ہیں یا اندھوں کو ہدایت دے سکتے ہیں اور انہیں جو کھلی گمراہی میں ہیں؟
فَإِمَّا نَذْهَبَنَّ بِكَ فَإِنَّا مِنْهُم مُّنتَقِمُونَ ﴿41﴾
پس اگر ہم آپ کو لے جائیں تو ہم ان سے ضرور انتقام لینے والے ہیں۔
أَوْ نُرِيَنَّكَ ٱلَّذِي وَعَدْنَـٰهُمْ فَإِنَّا عَلَيْهِم مُّقْتَدِرُونَ ﴿42﴾
یا ہم آپ کو وہ چیز دکھا دیں جس کا ہم نے ان سے وعدہ کیا ہے، تو ہم ان پر پوری قدرت رکھتے ہیں۔
فَٱسْتَمْسِكْ بِٱلَّذِيٓ أُوحِيَ إِلَيْكَۖ إِنَّكَ عَلَىٰ صِرَٰطٖ مُّسْتَقِيمٖ ﴿43﴾
پس آپ کو جو وحی کی گئی ہے اسے مضبوطی سے تھامے رکھیں، بے شک آپ سیدھے راستے پر ہیں۔
وَإِنَّهُۥ لَذِكْرٞ لَّكَ وَلِقَوْمِكَۖ وَسَوْفَ تُسْـَٔلُونَ ﴿44﴾
اور بے شک یہ آپ کے لیے اور آپ کی قوم کے لیے نصیحت ہے، اور تم سے ضرور پوچھا جائے گا۔
وَسْـَٔلْ مَنْ أَرْسَلْنَا مِن قَبْلِكَ مِن رُّسُلِنَآ أَجَعَلْنَا مِن دُونِ ٱلرَّحْمَـٰنِ ءَالِهَةٗ يُعْبَدُونَ ﴿45﴾
اور آپ ان رسولوں سے پوچھیں جنہیں ہم نے آپ سے پہلے بھیجا تھا: کیا ہم نے رحمٰن کے سوا کوئی معبود مقرر کیے تھے جن کی عبادت کی جائے؟
وَلَقَدْ أَرْسَلْنَا مُوسَىٰ بِـَٔايَـٰتِنَآ إِلَىٰ فِرْعَوْنَ وَمَلَإِيهِۦ فَقَالَ إِنِّي رَسُولُ رَبِّ ٱلْعَـٰلَمِينَ ﴿46﴾
اور ہم نے موسیٰ کو اپنی آیات کے ساتھ فرعون اور اس کے سرداروں کی طرف بھیجا، تو اس نے کہا: میں رب العالمین کا رسول ہوں۔
فَلَمَّا جَآءَهُم بِـَٔايَـٰتِنَآ إِذَا هُم مِّنْهَا يَضْحَكُونَ ﴿47﴾
پھر جب وہ ان کے پاس ہماری آیات لے کر آیا تو وہ ان پر ہنسنے لگے۔
وَمَا نُرِيهِم مِّنْ ءَايَةٍ إِلَّا هِيَ أَكْبَرُ مِنْ أُخْتِهَاۚ وَأَخَذْنَـٰهُم بِٱلْعَذَابِ لَعَلَّهُمْ يَرْجِعُونَ ﴿48﴾
اور ہم انہیں جو بھی نشانی دکھاتے وہ اپنی پچھلی سے بڑی ہوتی، اور ہم نے انہیں عذاب میں پکڑا تاکہ وہ رجوع کریں۔
وَقَالُوا۟ يَـٰٓأَيُّهَ ٱلسَّاحِرُ ٱدْعُ لَنَا رَبَّكَ بِمَا عَهِدَ عِندَكَ إِنَّنَا لَمُهْتَدُونَ ﴿49﴾
اور وہ کہنے لگے: اے جادوگر! اپنے رب سے ہمارے لیے دعا کرو جس کا اس نے تم سے وعدہ کیا ہے، بے شک ہم ضرور ہدایت پانے والے ہیں۔
فَلَمَّا كَشَفْنَا عَنْهُمُ ٱلْعَذَابَ إِذَا هُمْ يَنكُثُونَ ﴿50﴾
پھر جب ہم نے ان سے عذاب دور کیا تو وہ فوراً عہد توڑنے لگے۔
وَنَادَىٰ فِرْعَوْنُ فِي قَوْمِهِۦ قَالَ يَـٰقَوْمِ أَلَيْسَ لِي مُلْكُ مِصْرَ وَهَـٰذِهِ ٱلْأَنْهَـٰرُ تَجْرِي مِن تَحْتِيٓۖ أَفَلَا تُبْصِرُونَ ﴿51﴾
اور فرعون نے اپنی قوم میں پکارا: اے میری قوم! کیا مصر کی بادشاہی اور یہ نہریں جو میرے نیچے بہتی ہیں، میرے لیے نہیں؟ کیا تم دیکھتے نہیں؟
أَمْ أَنَا۠ خَيْرٞ مِّنْ هَـٰذَا ٱلَّذِي هُوَ مَهِينٞ وَلَا يَكَادُ يُبِينُ ﴿52﴾
یا میں بہتر ہوں اس شخص سے جو ذلیل ہے اور صاف بات بھی نہیں کر سکتا؟
فَلَوْلَآ أُلْقِيَ عَلَيْهِ أَسْوِرَةٞ مِّن ذَهَبٍ أَوْ جَآءَ مَعَهُ ٱلْمَلَـٰٓئِكَةُ مُقْتَرِنِينَ ﴿53﴾
تو اس پر سونے کے کنگن کیوں نہ ڈالے گئے؟ یا فرشتے اس کے ساتھ آتے؟
فَٱسْتَخَفَّ قَوْمَهُۥ فَأَطَاعُوهُۚ إِنَّهُمْ كَانُوا۟ قَوْمٗا فَـٰسِقِينَ ﴿54﴾
تو اس نے اپنی قوم کو ہلکا سمجھا اور وہ اس کی بات مان گئے، بے شک وہ نافرمان قوم تھے۔
فَلَمَّآ ءَاسَفُونَا ٱنتَقَمْنَا مِنْهُمْ فَأَغْرَقْنَـٰهُمْ أَجْمَعِينَ ﴿55﴾
پھر جب انہوں نے ہمیں غصہ دلایا تو ہم نے ان سے انتقام لیا اور سب کو غرق کر دیا۔
فَجَعَلْنَـٰهُمْ سَلَفٗا وَمَثَلٗا لِّلْـَٔاخِرِينَ ﴿56﴾
تو ہم نے انہیں پہلوں کے لیے مثال بنا دیا اور بعد والوں کے لیے عبرت۔
وَلَمَّا ضُرِبَ ٱبْنُ مَرْيَمَ مَثَلٗا إِذَا قَوْمُكَ مِنْهُ يَصِدُّونَ ﴿57﴾
اور جب ابن مریم کی مثال دی گئی تو آپ کی قوم اس پر شور مچانے لگی۔
وَقَالُوٓا۟ ءَأَالِهَتُنَا خَيْرٌ أَمْ هُوَۚ مَا ضَرَبُوهُ لَكَ إِلَّا جَدَلَۢاۚ بَلْ هُمْ قَوْمٌ خَصِمُونَ ﴿58﴾
اور انہوں نے کہا: کیا ہمارے معبود بہتر ہیں یا وہ؟ انہوں نے یہ مثال آپ کے لیے صرف جھگڑے کے لیے دی، بلکہ وہ جھگڑالو قوم ہیں۔
إِنْ هُوَ إِلَّا عَبْدٌ أَنْعَمْنَا عَلَيْهِ وَجَعَلْنَـٰهُ مَثَلٗا لِّبَنِيٓ إِسْرَٰٓءِيلَ ﴿59﴾
وہ تو صرف ایک بندہ تھا جس پر ہم نے انعام کیا اور اسے بنی اسرائیل کے لیے مثال بنایا۔
وَلَوْ نَشَآءُ لَجَعَلْنَا مِنكُم مَّلَـٰٓئِكَةٗ فِي ٱلْأَرْضِ يَخْلُفُونَ ﴿60﴾
اور اگر ہم چاہتے تو تم میں سے فرشتے زمین میں جانشین بنا دیتے۔
وَإِنَّهُۥ لَعِلْمٞ لِّلسَّاعَةِ فَلَا تَمْتَرُنَّ بِهَا وَٱتَّبِعُونِۚ هَـٰذَا صِرَٰطٞ مُّسْتَقِيمٞ ﴿61﴾
اور بے شک وہ قیامت کی نشانی ہے، پس تم اس میں شک نہ کرو اور میری پیروی کرو، یہی سیدھا راستہ ہے۔
وَلَا يَصُدَّنَّكُمُ ٱلشَّيْطَـٰنُۖ إِنَّهُۥ لَكُمْ عَدُوّٞ مُّبِينٞ ﴿62﴾
اور شیطان تمہیں روک نہ دے، بے شک وہ تمہارا کھلا دشمن ہے۔
وَلَمَّا جَآءَ عِيسَىٰ بِٱلْبَيِّنَـٰتِ قَالَ قَدْ جِئْتُكُم بِٱلْحِكْمَةِ وَلِأُبَيِّنَ لَكُم بَعْضَ ٱلَّذِي تَخْتَلِفُونَ فِيهِۖ فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿63﴾
اور جب عیسیٰ واضح دلائل لے کر آئے تو کہا: میں تمہارے پاس حکمت لے کر آیا ہوں اور تاکہ تمہیں بعض وہ باتیں واضح کر دوں جن میں تم اختلاف کرتے ہو، پس اللہ سے ڈرو اور میری اطاعت کرو۔
إِنَّ ٱللَّهَ هُوَ رَبِّي وَرَبُّكُمْ فَٱعْبُدُوهُۚ هَـٰذَا صِرَٰطٞ مُّسْتَقِيمٞ ﴿64﴾
بے شک اللہ ہی میرا رب اور تمہارا رب ہے، پس اسی کی عبادت کرو، یہی سیدھا راستہ ہے۔
فَٱخْتَلَفَ ٱلْأَحْزَابُ مِنۢ بَيْنِهِمْۖ فَوَيْلٞ لِّلَّذِينَ ظَلَمُوا۟ مِنْ عَذَابِ يَوْمٍ أَلِيمٍ ﴿65﴾
پھر گروہوں نے آپس میں اختلاف کیا، سو ظالموں کے لیے دردناک دن کے عذاب کی ہلاکت ہے۔
هَلْ يَنظُرُونَ إِلَّا ٱلسَّاعَةَ أَن تَأْتِيَهُم بَغْتَةٗ وَهُمْ لَا يَشْعُرُونَ ﴿66﴾
کیا وہ صرف قیامت کا انتظار کر رہے ہیں کہ وہ ان پر اچانک آ جائے اور انہیں خبر بھی نہ ہو؟
ٱلْأَخِلَّآءُ يَوْمَئِذِۭ بَعْضُهُمْ لِبَعْضٍ عَدُوٌّ إِلَّا ٱلْمُتَّقِينَ ﴿67﴾
اس دن دوست ایک دوسرے کے دشمن ہوں گے، سوائے پرہیزگاروں کے۔
يَـٰعِبَادِ لَا خَوْفٌ عَلَيْكُمُ ٱلْيَوْمَ وَلَآ أَنتُمْ تَحْزَنُونَ ﴿68﴾
اے میرے بندو! آج تم پر نہ کوئی خوف ہے اور نہ تم غمگین ہو گے۔
ٱلَّذِينَ ءَامَنُوا۟ بِـَٔايَـٰتِنَا وَكَانُوا۟ مُسْلِمِينَ ﴿69﴾
وہ لوگ جو ہماری آیات پر ایمان لائے اور فرمانبردار بنے۔
ٱدْخُلُوا۟ ٱلْجَنَّةَ أَنتُمْ وَأَزْوَٰجُكُمْ تُحْبَرُونَ ﴿70﴾
تم اور تمہاری بیویاں جنت میں داخل ہو جاؤ، تم خوش رکھے جاؤ گے۔
يُطَافُ عَلَيْهِمْ بِصِحَافٖ مِّن ذَهَبٖ وَأَكْوَابٖۖ وَفِيهَا مَا تَشْتَهِيهِ ٱلْأَنفُسُ وَتَلَذُّ ٱلْأَعْيُنُۖ وَأَنتُمْ فِيهَا خَـٰلِدُونَ ﴿71﴾
ان پر سونے کی پلیٹیں اور پیالے گردش کریں گے، اور اس میں وہ سب کچھ ہوگا جو دل چاہے اور آنکھوں کو لذت دے، اور تم اس میں ہمیشہ رہو گے۔
وَتِلْكَ ٱلْجَنَّةُ ٱلَّتِيٓ أُورِثْتُمُوهَا بِمَا كُنتُمْ تَعْمَلُونَ ﴿72﴾
اور یہ وہ جنت ہے جس کے تم وارث بنائے گئے ہو، ان اعمال کے سبب جو تم کرتے تھے۔
لَكُمْ فِيهَا فَـٰكِهَةٞ كَثِيرَةٞ مِّنْهَا تَأْكُلُونَ ﴿73﴾
تمہارے لیے اس میں بہت سے پھل ہوں گے جن میں سے تم کھاؤ گے۔
إِنَّ ٱلْمُجْرِمِينَ فِي عَذَابِ جَهَنَّمَ خَـٰلِدُونَ ﴿74﴾
بے شک مجرم جہنم کے عذاب میں ہمیشہ رہیں گے۔
لَا يُفَتَّرُ عَنْهُمْ وَهُمْ فِيهِ مُبْلِسُونَ ﴿75﴾
ان سے عذاب ہلکا نہ کیا جائے گا اور وہ اس میں مایوس ہوں گے۔
وَمَا ظَلَمْنَـٰهُمْ وَلَـٰكِن كَانُوا۟ هُمُ ٱلظَّـٰلِمِينَ ﴿76﴾
اور ہم نے ان پر ظلم نہیں کیا بلکہ وہ خود ہی ظالم تھے۔
وَنَادَوْا۟ يَـٰمَـٰلِكُ لِيَقْضِ عَلَيْنَا رَبُّكَۖ قَالَ إِنَّكُم مَّاكِثُونَ ﴿77﴾
اور وہ پکاریں گے: اے مالک! تیرا رب ہمارے ساتھ فیصلہ کر دے، وہ کہے گا: تم یہی رہنے والے ہو۔
لَقَدْ جِئْنَـٰكُم بِٱلْحَقِّ وَلَـٰكِنَّ أَكْثَرَكُمْ لِلْحَقِّ كَـٰرِهُونَ ﴿78﴾
بے شک ہم تمہارے پاس حق لے کر آئے تھے، لیکن تم میں سے اکثر حق کو ناپسند کرتے تھے۔
أَمْ أَبْرَمُوٓا۟ أَمْرٗا فَإِنَّا مُبْرِمُونَ ﴿79﴾
کیا انہوں نے کوئی فیصلہ کر لیا ہے؟ تو ہم بھی فیصلہ کرنے والے ہیں۔
أَمْ يَحْسَبُونَ أَنَّا لَا نَسْمَعُ سِرَّهُمْ وَنَجْوَىٰهُمۚ بَلَىٰ وَرُسُلُنَا لَدَيْهِمْ يَكْتُبُونَ ﴿80﴾
کیا وہ گمان کرتے ہیں کہ ہم ان کی چھپی باتیں اور سرگوشیاں نہیں سنتے؟ کیوں نہیں! اور ہمارے فرستادہ ان کے پاس لکھ رہے ہیں۔
قُلْ إِن كَانَ لِلرَّحْمَـٰنِ وَلَدٞ فَأَنَا۠ أَوَّلُ ٱلْعَـٰبِدِينَ ﴿81﴾
کہہ دو: اگر رحمٰن کے لیے کوئی اولاد ہوتی تو میں سب سے پہلے عبادت کرنے والا ہوتا۔
سُبْحَـٰنَ رَبِّ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ رَبِّ ٱلْعَرْشِ عَمَّا يَصِفُونَ ﴿82﴾
آسمانوں اور زمین کے رب، عرش کے رب کی پاکی ہے ان باتوں سے جو وہ بیان کرتے ہیں۔
فَذَرْهُمْ يَخُوضُوا۟ وَيَلْعَبُوا۟ حَتَّىٰ يُلَـٰقُوا۟ يَوْمَهُمُ ٱلَّذِي يُوعَدُونَ ﴿83﴾
تو انہیں چھوڑ دو کہ وہ بحث کریں اور کھیلتے رہیں یہاں تک کہ اپنے اس دن سے جا ملیں جس کا ان سے وعدہ کیا گیا ہے۔
وَهُوَ ٱلَّذِي فِي ٱلسَّمَآءِ إِلَـٰهٞ وَفِي ٱلْأَرْضِ إِلَـٰهٞۚ وَهُوَ ٱلْحَكِيمُ ٱلْعَلِيمُ ﴿84﴾
اور وہی آسمان میں معبود ہے اور زمین میں بھی معبود ہے، اور وہی حکمت والا، علم والا ہے۔
وَتَبَارَكَ ٱلَّذِي لَهُۥ مُلْكُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَا وَعِندَهُۥ عِلْمُ ٱلسَّاعَةِ وَإِلَيْهِ تُرْجَعُونَ ﴿85﴾
اور بابرکت ہے وہ جس کے لیے آسمانوں اور زمین اور جو کچھ ان کے درمیان ہے، سب کی بادشاہی ہے، اور اسی کے پاس قیامت کا علم ہے، اور اسی کی طرف تم لوٹائے جاؤ گے۔
وَلَا يَمْلِكُ ٱلَّذِينَ يَدْعُونَ مِن دُونِهِۦ ٱلشَّفَـٰعَةَ إِلَّا مَن شَهِدَ بِٱلْحَقِّ وَهُمْ يَعْلَمُونَ ﴿86﴾
اور جنہیں وہ اس کے سوا پکارتے ہیں وہ شفاعت کا اختیار نہیں رکھتے، سوائے اس کے جو حق کی گواہی دے اور وہ جانتے ہوں۔
وَلَئِن سَأَلْتَهُم مَّنْ خَلَقَهُمْ لَيَقُولُنَّ ٱللَّهُۖ فَأَنَّىٰ يُؤْفَكُونَ ﴿87﴾
اور اگر آپ ان سے پوچھیں کہ انہیں کس نے پیدا کیا؟ تو ضرور کہیں گے: اللہ نے، پھر وہ کہاں بہکائے جا رہے ہیں؟
وَقِيلِهِۦ يَـٰرَبِّ إِنَّ هَـٰٓؤُلَآءِ قَوْمٞ لَّا يُؤْمِنُونَ ﴿88﴾
اور اس کی بات: اے میرے رب! یہ وہ قوم ہے جو ایمان نہیں لاتی۔
فَٱصْفَحْ عَنْهُمْ وَقُلْ سَلَـٰمٞۚ فَسَوْفَ يَعْلَمُونَ ﴿89﴾
تو ان سے درگزر کرو اور کہہ دو: سلام ہو، پھر وہ جلد جان لیں گے۔

وَٱلْكِتَـٰبِ ٱلْمُبِينِ ﴿2﴾
قسم ہے واضح کتاب کی۔
إِنَّآ أَنزَلْنَـٰهُ فِي لَيْلَةٖ مُّبَـٰرَكَةٍ إِنَّا كُنَّا مُنذِرِينَ ﴿3﴾
بے شک ہم نے اسے ایک بابرکت رات میں نازل کیا، بے شک ہم ہی ڈرانے والے ہیں۔
فِيهَا يُفْرَقُ كُلُّ أَمْرٍ حَكِيمٍ ﴿4﴾
اسی رات میں ہر حکمت والے کام کا فیصلہ کیا جاتا ہے۔
أَمْرٗا مِّنْ عِندِنَآۚ إِنَّا كُنَّا مُرْسِلِينَ ﴿5﴾
ہمارے حکم سے، بے شک ہم ہی بھیجنے والے ہیں۔
رَحْمَةٗ مِّن رَّبِّكَۚ إِنَّهُۥ هُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ ﴿6﴾
یہ آپ کے رب کی رحمت ہے، بے شک وہی سننے والا، جاننے والا ہے۔
رَبِّ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَآۖ إِن كُنتُم مُّوقِنِينَ ﴿7﴾
آسمانوں اور زمین اور جو کچھ ان کے درمیان ہے، سب کا رب ہے، اگر تم یقین رکھتے ہو۔
لَآ إِلَـٰهَ إِلَّا هُوَۖ يُحْيِۦ وَيُمِيتُۖ رَبُّكُمْ وَرَبُّ ءَابَآئِكُمُ ٱلْأَوَّلِينَ ﴿8﴾
اس کے سوا کوئی معبود نہیں، وہی زندہ کرتا ہے اور مارتا ہے، وہی تمہارا رب اور تمہارے پہلے باپ دادا کا رب ہے۔
بَلْ هُمْ فِي شَكّٖ يَلْعَبُونَ ﴿9﴾
بلکہ وہ شک میں پڑے کھیل رہے ہیں۔
فَٱرْتَقِبْ يَوْمَ تَأْتِي ٱلسَّمَآءُ بِدُخَانٖ مُّبِينٖ ﴿10﴾
پس انتظار کرو اس دن کا جب آسمان واضح دھوئیں لے کر آئے گا۔
يَغْشَى ٱلنَّاسَ ۖ هَـٰذَا عَذَابٌ أَلِيمٌ ﴿11﴾
وہ لوگوں کو ڈھانپ لے گا، یہ دردناک عذاب ہے۔
رَبَّنَا ٱكْشِفْ عَنَّا ٱلْعَذَابَ إِنَّا مُؤْمِنُونَ ﴿12﴾
اے ہمارے رب! ہم سے عذاب دور کر دے، بے شک ہم ایمان لاتے ہیں۔
أَنَّىٰ لَهُمُ ٱلذِّكْرَىٰ وَقَدْ جَآءَهُمْ رَسُولٞ مُّبِينٞ ﴿13﴾
انہیں نصیحت کہاں فائدہ دے گی جبکہ ان کے پاس واضح رسول آ چکا تھا؟
ثُمَّ تَوَلَّوْا۟ عَنْهُ وَقَالُوا۟ مُعَلَّمٞ مَّجْنُونٌ ﴿14﴾
پھر انہوں نے اس سے منہ موڑ لیا اور کہا: یہ تو سکھایا ہوا دیوانہ ہے۔
إِنَّا كَاشِفُوا۟ ٱلْعَذَابِ قَلِيلًا إِنَّكُمْ عَآئِدُونَ ﴿15﴾
ہم تھوڑے وقت کے لیے عذاب کو دور کر دیتے ہیں، تم ضرور پھر لوٹ آؤ گے۔
يَوْمَ نَبْطِشُ ٱلْبَطْشَةَ ٱلْكُبْرَىٰٓ إِنَّا مُنتَقِمُونَ ﴿16﴾
جس دن ہم بڑی پکڑ کریں گے، ہم ضرور انتقام لینے والے ہیں۔
وَلَقَدْ فَتَنَّا قَبْلَهُمْ قَوْمَ فِرْعَوْنَ وَجَآءَهُمْ رَسُولٞ كَرِيمٌ ﴿17﴾
اور ہم نے ان سے پہلے فرعون کی قوم کو آزمائش میں ڈالا، اور ان کے پاس ایک معزز رسول آیا۔
أَنْ أَدُّوٓا۟ إِلَيَّ عِبَادَ ٱللَّهِ ۖ إِنِّي لَكُمْ رَسُولٌ أَمِينٞ ﴿18﴾
کہ اللہ کے بندوں کو میرے حوالے کر دو، بے شک میں تمہارے لیے امانت دار رسول ہوں۔
وَأَلَّا تَعْلُوا۟ عَلَى ٱللَّهِ ۖ إِنِّيٓ ءَاتِيكُم بِسُلْطَـٰنٖ مُّبِينٖ ﴿19﴾
اور اللہ کے مقابلے میں سرکشی نہ کرو، بے شک میں تمہارے پاس واضح دلیل کے ساتھ آیا ہوں۔
وَإِنِّي عُذْتُ بِرَبِّي وَرَبِّكُمۡ أَن تَرْجُمُونِ ﴿20﴾
اور میں اپنے رب اور تمہارے رب کی پناہ لیتا ہوں کہ تم مجھے سنگسار کرو۔
فَإِن لَّمْ تُؤْمِنُوا۟ لِي فَٱعْتَزِلُونِ ﴿21﴾
پس اگر تم مجھ پر ایمان نہیں لاتے تو مجھ سے الگ ہو جاؤ۔
فَدَعَا رَبَّهُۥٓ أَنَّ هَـٰٓؤُلَآءِ قَوْمٞ مُّجْرِمُونَ ﴿22﴾
پھر اس نے اپنے رب سے دعا کی: یہ لوگ مجرم ہیں۔
فَأَسْرِ بِعِبَادِي لَيْلًا إِنَّكُم مُّتَّبَعُونَ ﴿23﴾
تو میرے بندوں کو راتوں رات لے چلو، بے شک تمہارا پیچھا کیا جائے گا۔
وَٱتْرُكِ ٱلْبَحْرَ رَهْوًاۖ إِنَّهُمْ جُندٞ مُّغْرَقُونَ ﴿24﴾
اور سمندر کو ساکن چھوڑ دو، بے شک وہ لشکر غرق ہونے والا ہے۔
كَمۡ تَرَكُوا۟ مِن جَنَّـٰتٖ وَعُيُونٖ ﴿25﴾
کتنے ہی باغات اور چشمے وہ چھوڑ گئے۔
وَزُرُوعٖ وَمَقَامٖ كَرِيمٖ ﴿26﴾
اور کھیتیاں اور عزت والی رہائشیں۔
وَنَعْمَةٖ كَانُوا۟ فِيهَا فَـٰكِهِينَ ﴿27﴾
اور نعمتیں جن میں وہ مزے کرتے تھے۔
كَذَٰلِكَۖ وَأَوْرَثْنَـٰهَا قَوْمًا ءَاخَرِينَ ﴿28﴾
اسی طرح ہوا، اور ہم نے ان چیزوں کو دوسری قوم کو وارث بنا دیا۔
فَمَا بَكَتْ عَلَيْهِمُ ٱلسَّمَآءُ وَٱلْأَرْضُ وَمَا كَانُوا۟ مُنظَرِينَ ﴿29﴾
تو آسمان اور زمین ان پر روئے نہیں، اور نہ ہی انہیں مہلت دی گئی۔
وَلَقَدْ نَجَّيْنَا بَنِيٓ إِسْرَٰٓءِيلَ مِنَ ٱلْعَذَابِ ٱلْمُهِينِ ﴿30﴾
اور ہم نے بنی اسرائیل کو ذلت والے عذاب سے نجات دی۔
مِّن فِرْعَوْنَۚ إِنَّهُۥ كَانَ عَالِيٗا مِّنَ ٱلْمُسْرِفِينَ ﴿31﴾
فرعون سے، بے شک وہ سرکش لوگوں میں سے تھا۔
وَلَقَدِ ٱخْتَرْنَـٰهُمْ عَلَىٰ عِلْمٍ عَلَى ٱلْعَـٰلَمِينَ ﴿32﴾
اور ہم نے انہیں علم کے ساتھ جہان والوں پر چن لیا تھا۔
وَءَاتَيْنَـٰهُم مِّنَ ٱلْءَايَـٰتِ مَا فِيهِ بَلَـٰٓؤٞاۢ مُّبِينٞ ﴿33﴾
اور ہم نے انہیں ایسی نشانیاں دیں جن میں واضح آزمائش تھی۔
إِنَّ هَـٰٓؤُلَآءِ لَيَقُولُونَ ﴿34﴾
بے شک یہ لوگ کہتے ہیں۔
إِنْ هِيَ إِلَّا مَوْتَتُنَا ٱلْأُولَىٰ وَمَا نَحْنُ بِمُنشَرِينَ ﴿35﴾
یہ تو بس ہماری پہلی موت ہے، اور ہم دوبارہ نہیں اٹھائے جائیں گے۔
فَأْتُوا۟ بِـَٔابَآئِنَآ إِن كُنتُمْ صَـٰدِقِينَ ﴿36﴾
تو ہمارے باپ دادا کو لے آؤ اگر تم سچے ہو۔
أَهُمۡ خَيْرٌ أَمْ قَوْمُ تُبَّعٖ وَٱلَّذِينَ مِن قَبْلِهِمۡۚ أَهْلَكْنَـٰهُمْ إِنَّهُمۡ كَانُوا۟ مُجْرِمِينَ ﴿37﴾
کیا یہ بہتر ہیں یا قومِ تُبّع اور ان سے پہلے والے؟ ہم نے انہیں ہلاک کر دیا، بے شک وہ مجرم تھے۔
وَمَا خَلَقْنَا ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ وَمَا بَيْنَهُمَا لَـٰعِبِينَ ﴿38﴾
اور ہم نے آسمانوں اور زمین کو اور جو کچھ ان کے درمیان ہے کھیل کے طور پر نہیں بنایا۔
مَا خَلَقْنَـٰهُمَآ إِلَّا بِٱلْحَقِّ وَلَـٰكِنَّ أَكْثَرَهُمْ لَا يَعْلَمُونَ ﴿39﴾
ہم نے انہیں حق کے ساتھ پیدا کیا ہے، لیکن ان میں سے اکثر نہیں جانتے۔
إِنَّ يَوْمَ ٱلْفَصْلِ مِيقَـٰتُهُمْ أَجْمَعِينَ ﴿40﴾
بے شک فیصلے کا دن ان سب کا مقرر وقت ہے۔
يَوْمَ لَا يُغْنِي مَوْلٗى عَن مَّوْلٗى شَيْـٔٗا وَلَا هُمْ يُنصَرُونَ ﴿41﴾
جس دن کوئی دوست کسی دوست کے کام نہ آئے گا، اور نہ ہی ان کی مدد کی جائے گی۔
إِلَّا مَن رَّحِمَ ٱللَّهُۚ إِنَّهُۥ هُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿42﴾
سوائے اس کے جس پر اللہ رحم کرے، بے شک وہی زبردست، رحم کرنے والا ہے۔
إِنَّ شَجَرَتَ ٱلزَّقُّومِ ﴿43﴾
بے شک زقوم کا درخت۔
طَعَامُ ٱلْأَثِيمِ ﴿44﴾
گناہگار کا کھانا ہے۔
كَٱلْمُهْلِ يَغْلِي فِي ٱلْبُطُونِ ﴿45﴾
پگھلے ہوئے تانبے کی طرح پیٹوں میں کھولتا ہے۔
كَغَلْيِ ٱلْحَمِيمِ ﴿46﴾
جیسے کھولتا ہوا پانی کھولتا ہے۔
خُذُوهُ فَٱعْتِلُوهُ إِلَىٰ سَوَآءِ ٱلْجَحِيمِ ﴿47﴾
پکڑو اسے اور گھسیٹتے ہوئے جہنم کے وسط تک لے جاؤ۔
ثُمَّ صُبُّوا۟ فَوْقَ رَأْسِهِۦ مِنْ عَذَابِ ٱلْحَمِيمِ ﴿48﴾
پھر اس کے سر پر کھولتے ہوئے پانی کا عذاب انڈیل دو۔
ذُقْ إِنَّكَ أَنتَ ٱلْعَزِيزُ ٱلْكَرِيمُ ﴿49﴾
چکھ! بے شک تو ہی عزت والا، بزرگی والا تھا۔
إِنَّ هَـٰذَا مَا كُنتُم بِهِۦ تَمْتَرُونَ ﴿50﴾
بے شک یہ وہی چیز ہے جس میں تم شک کرتے تھے۔
إِنَّ ٱلْمُتَّقِينَ فِي مَقَامٍ أَمِينٖ ﴿51﴾
بے شک پرہیزگار لوگ امن والی جگہ میں ہوں گے۔
فِي جَنَّـٰتٖ وَعُيُونٖ ﴿52﴾
باغوں اور چشموں میں۔
يَلْبَسُونَ مِن سُندُسٖ وَإِسْتَبْرَقٖ مُّتَقَـٰبِلِينَ ﴿53﴾
وہ باریک اور موٹے ریشم کے لباس پہنیں گے، آمنے سامنے بیٹھے ہوں گے۔
كَذَٰلِكَ وَزَوَّجْنَـٰهُم بِحُورٍ عِينٖ ﴿54﴾
اسی طرح ہوگا، اور ہم انہیں بڑی آنکھوں والی حوروں سے جوڑ دیں گے۔
يَدْعُونَ فِيهَا بِكُلِّ فَـٰكِهَةٍ ءَامِنِينَ ﴿55﴾
وہ وہاں ہر قسم کے پھل طلب کریں گے، امن کے ساتھ۔
لَا يَذُوقُونَ فِيهَا ٱلْمَوْتَ إِلَّا ٱلْمَوْتَةَ ٱلْأُولَىٰۖ وَوَقَىٰهُمْ عَذَابَ ٱلْجَحِيمِ ﴿56﴾
وہ وہاں موت کا ذائقہ نہیں چکھیں گے، سوائے پہلی موت کے، اور اللہ انہیں جہنم کے عذاب سے بچا لے گا۔
فَضْلٗا مِّن رَّبِّكَۚ ذَٰلِكَ هُوَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿57﴾
یہ آپ کے رب کا فضل ہے، یہی بڑی کامیابی ہے۔
فَإِنَّمَا يَسَّرْنَـٰهُ بِلِسَانِكَ لَعَلَّهُمْ يَتَذَكَّرُونَ ﴿58﴾
پس ہم نے اسے آپ کی زبان میں آسان کر دیا تاکہ وہ نصیحت حاصل کریں۔
فَٱرْتَقِبْ إِنَّهُم مُّرْتَقِبُونَ ﴿59﴾
پس انتظار کرو، بے شک وہ بھی انتظار کر رہے ہیں۔
سورة الجاثية
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
حمٓ ﴿1﴾
حٰمٓ
تَنزِيلُ ٱلْكِتَـٰبِ مِنَ ٱللَّهِ ٱلْعَزِيزِ ٱلْحَكِيمِ ﴿2﴾
اس کتاب کا نزول اللہ کی طرف سے ہے جو زبردست، حکمت والا ہے۔
إِنَّ فِي ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ لَـَٔايَـٰتٖ لِّلْمُؤْمِنِينَ ﴿3﴾
بے شک آسمانوں اور زمین میں ایمان والوں کے لیے نشانیاں ہیں۔
وَفِي خَلْقِكُمْ وَمَا يَبُثُّ مِن دَآبَّةٍ ءَايَـٰتٞ لِّقَوْمٖ يُوقِنُونَ ﴿4﴾
اور تمہاری پیدائش میں اور ان جانوروں میں جو وہ پھیلاتا ہے، یقین رکھنے والوں کے لیے نشانیاں ہیں۔
وَٱخْتِلَافِ ٱلَّيْلِ وَٱلنَّهَارِ وَمَآ أَنزَلَ ٱللَّهُ مِنَ ٱلسَّمَآءِ مِن رِّزْقٖ فَأَحْيَا بِهِ ٱلْأَرْضَ بَعْدَ مَوْتِهَا وَتَصْرِيفِ ٱلرِّيَـٰحِ ءَايَـٰتٞ لِّقَوْمٖ يَعْقِلُونَ ﴿5﴾
اور رات اور دن کے بدلنے میں، اور جو رزق اللہ نے آسمان سے نازل کیا پھر اس سے زمین کو اس کی موت کے بعد زندہ کیا، اور ہواؤں کے چلانے میں عقل والوں کے لیے نشانیاں ہیں۔
تِلْكَ ءَايَـٰتُ ٱللَّهِ نَتْلُوهَا عَلَيْكَ بِٱلْحَقِّ ۖ فَبِأَيِّ حَدِيثِۭ بَعْدَ ٱللَّهِ وَءَايَـٰتِهِۦ يُؤْمِنُونَ ﴿6﴾
یہ اللہ کی آیات ہیں جو ہم آپ پر حق کے ساتھ پڑھتے ہیں، تو اللہ اور اس کی آیات کے بعد وہ کس بات پر ایمان لائیں گے؟
وَيْلٞ لِّكُلِّ أَفَّاكٍ أَثِيمٖ ﴿7﴾
ہر جھوٹے گناہگار کے لیے ہلاکت ہے۔
يَسْمَعُ ءَايَـٰتِ ٱللَّهِ تُتْلَىٰ عَلَيْهِ ثُمَّ يُصِرُّ مُسْتَكْبِرٗا كَأَن لَّمْ يَسْمَعْهَاۚ فَبَشِّرْهُ بِعَذَابٍ أَلِيمٍ ﴿8﴾
وہ اللہ کی آیات سنتا ہے جو اس پر پڑھی جاتی ہیں، پھر تکبر سے اصرار کرتا ہے گویا اس نے انہیں سنا ہی نہیں، تو اسے دردناک عذاب کی خوشخبری دے دو۔
وَإِذَا عَلِمَ مِنْ ءَايَـٰتِنَا شَيْـٔٗا ٱتَّخَذَهَا هُزُوٗاۚ أُو۟لَـٰٓئِكَ لَهُمْ عَذَابٞ مُّهِينٞ ﴿9﴾
اور جب وہ ہماری آیات میں سے کسی کو جانتا ہے تو اسے مذاق بنا لیتا ہے، یہی لوگ ہیں جن کے لیے ذلت کا عذاب ہے۔
مِّن وَرَآئِهِمْ جَهَنَّمُ وَلَا يُغْنِي عَنْهُم مَّا كَسَبُوا۟ شَيْـٔٗا وَلَا مَا ٱتَّخَذُوا۟ مِن دُونِ ٱللَّهِ أَوْلِيَآءَۚ وَلَهُمْ عَذَابٌ عَظِيمٌ ﴿10﴾
ان کے پیچھے جہنم ہے، اور جو کچھ انہوں نے کمایا وہ ان کے کچھ کام نہ آئے گا، اور نہ وہ مددگار جو انہوں نے اللہ کے سوا بنا رکھے تھے، اور ان کے لیے بڑا عذاب ہے۔
هَـٰذَا هُدٗىۖ وَٱلَّذِينَ كَفَرُوا۟ بِـَٔايَـٰتِ رَبِّهِمْ لَهُمْ عَذَابٞ مِّن رِّجْزٍ أَلِيمٌ ﴿11﴾
یہ (قرآن) ہدایت ہے، اور جنہوں نے اپنے رب کی آیات کا انکار کیا ان کے لیے دردناک عذاب ہے۔
ٱللَّهُ ٱلَّذِي سَخَّرَ لَكُمُ ٱلْبَحْرَ لِتَجْرِيَ ٱلْفُلْكُ فِيهِ بِأَمْرِهِۦ وَلِتَبْتَغُوا۟ مِن فَضْلِهِۦ وَلَعَلَّكُمْ تَشْكُرُونَ ﴿12﴾
اللہ وہ ہے جس نے تمہارے لیے سمندر کو مسخر کیا تاکہ اس میں کشتیاں اس کے حکم سے چلیں، اور تاکہ تم اس کے فضل سے کچھ تلاش کرو، اور تاکہ تم شکر ادا کرو۔
وَسَخَّرَ لَكُم مَّا فِي ٱلسَّمَـٰوَٰتِ وَمَا فِي ٱلْأَرْضِ جَمِيعٗا مِّنْهُۚ إِنَّ فِي ذَٰلِكَ لَـَٔايَـٰتٖ لِّقَوْمٖ يَتَفَكَّرُونَ ﴿13﴾
اور اس نے تمہارے لیے جو کچھ آسمانوں اور زمین میں ہے سب کو اپنی طرف سے مسخر کر دیا، بے شک اس میں ان لوگوں کے لیے نشانیاں ہیں جو غور و فکر کرتے ہیں۔
قُل لِّلَّذِينَ ءَامَنُوا۟ يَغْفِرُوا۟ لِلَّذِينَ لَا يَرْجُونَ أَيَّامَ ٱللَّهِ لِيَجْزِيَ قَوْمَۢا بِمَا كَانُوا۟ يَكْسِبُونَ ﴿14﴾
ایمان والوں سے کہہ دو کہ وہ ان لوگوں کو معاف کر دیں جو اللہ کے دنوں کی امید نہیں رکھتے، تاکہ اللہ ایک قوم کو ان کے اعمال کے مطابق جزا دے۔
مَنْ عَمِلَ صَـٰلِحٗا فَلِنَفْسِهِۦۖ وَمَنْ أَسَآءَ فَعَلَيْهَاۖ ثُمَّ إِلَىٰ رَبِّكُمْ تُرْجَعُونَ ﴿15﴾
جو کوئی نیک عمل کرے گا تو وہ اپنے ہی فائدے کے لیے کرے گا، اور جو برائی کرے گا تو وہ اسی پر پڑے گی، پھر تم سب اپنے رب کی طرف لوٹائے جاؤ گے۔
وَلَقَدْ ءَاتَيْنَا بَنِيٓ إِسْرَٰٓءِيلَ ٱلْكِتَـٰبَ وَٱلْحُكْمَ وَٱلنُّبُوَّةَ وَرَزَقْنَـٰهُم مِّنَ ٱلطَّيِّبَـٰتِ وَفَضَّلْنَـٰهُمْ عَلَى ٱلْعَـٰلَمِينَ ﴿16﴾
اور ہم نے بنی اسرائیل کو کتاب، حکومت اور نبوت عطا کی، اور انہیں پاکیزہ چیزوں سے رزق دیا، اور انہیں جہان والوں پر فضیلت دی۔
وَءَاتَيْنَـٰهُم بَيِّنَـٰتٖ مِّنَ ٱلْأَمْرِ ۖ فَمَا ٱخْتَلَفُوٓا۟ إِلَّا مِنۢ بَعْدِ مَا جَآءَهُمُ ٱلْعِلْمُ بَغْيَۢا بَيْنَهُمْۚ إِنَّ رَبَّكَ يَقْضِي بَيْنَهُمْ يَوْمَ ٱلْقِيَـٰمَةِ فِيمَا كَانُوا۟ فِيهِ يَخْتَلِفُونَ ﴿17﴾
اور ہم نے انہیں دین کے معاملے میں واضح دلائل دیے، پھر انہوں نے اختلاف نہیں کیا مگر اس کے بعد کہ ان کے پاس علم آ چکا تھا، آپس کی ضد کی وجہ سے۔ بے شک آپ کا رب قیامت کے دن ان کے درمیان ان باتوں کا فیصلہ کرے گا جن میں وہ اختلاف کرتے تھے۔
ثُمَّ جَعَلْنَـٰكَ عَلَىٰ شَرِيعَةٖ مِّنَ ٱلْأَمْرِ فَٱتَّبِعْهَا وَلَا تَتَّبِعْ أَهْوَآءَ ٱلَّذِينَ لَا يَعْلَمُونَ ﴿18﴾
پھر ہم نے آپ کو دین کے ایک واضح راستے پر قائم کیا، تو آپ اسی کی پیروی کریں اور ان لوگوں کی خواہشات کی پیروی نہ کریں جو علم نہیں رکھتے۔
إِنَّهُمْ لَن يُغْنُوا۟ عَنكَ مِنَ ٱللَّهِ شَيْـٔٗاۚ وَإِنَّ ٱلظَّـٰلِمِينَ بَعْضُهُمْ أَوْلِيَآءُ بَعْضٖۖ وَٱللَّهُ وَلِيُّ ٱلْمُتَّقِينَ ﴿19﴾
بے شک وہ اللہ کے مقابلے میں آپ کے کچھ بھی کام نہیں آ سکتے، اور بے شک ظالم ایک دوسرے کے دوست ہیں، اور اللہ پرہیزگاروں کا دوست ہے۔
هَـٰذَا بَصَـٰٓئِرُ لِلنَّاسِ وَهُدٗى وَرَحْمَةٞ لِّقَوْمٖ يُوقِنُونَ ﴿20﴾
یہ (قرآن) لوگوں کے لیے بصیرتیں ہیں، اور ہدایت اور رحمت ہے ان لوگوں کے لیے جو یقین رکھتے ہیں۔
أَمْ حَسِبَ ٱلَّذِينَ ٱجْتَرَحُوا۟ ٱلسَّيِّـَٔاتِ أَن نَّجْعَلَهُمْ كَٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ سَوَآءٗ مَّحْيَاهُمْ وَمَمَاتُهُمْۚ سَآءَ مَا يَحْكُمُونَ ﴿21﴾
کیا وہ لوگ جنہوں نے برے کام کیے یہ گمان کرتے ہیں کہ ہم انہیں ان لوگوں کی طرح کر دیں گے جو ایمان لائے اور نیک عمل کیے، کہ ان کی زندگی اور موت برابر ہو جائے؟ بہت برا ہے جو وہ فیصلہ کرتے ہیں۔
وَخَلَقَ ٱللَّهُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ بِٱلْحَقِّ وَلِتُجْزَىٰ كُلُّ نَفْسِۭ بِمَا كَسَبَتْ وَهُمْ لَا يُظْلَمُونَ ﴿22﴾
اور اللہ نے آسمانوں اور زمین کو حق کے ساتھ پیدا کیا تاکہ ہر شخص کو اس کے کیے کا بدلہ دیا جائے، اور ان پر ظلم نہیں کیا جائے گا۔
أَفَرَءَيْتَ مَنِ ٱتَّخَذَ إِلَـٰهَهُۥ هَوَىٰهُ وَأَضَلَّهُ ٱللَّهُ عَلَىٰ عِلْمٖ وَخَتَمَ عَلَىٰ سَمْعِهِۦ وَقَلْبِهِۦ وَجَعَلَ عَلَىٰ بَصَرِهِۦ غِشَـٰوَةٗۚ فَمَن يَهْدِيهِ مِنۢ بَعْدِ ٱللَّهِۚ أَفَلَا تَذَكَّرُونَ ﴿23﴾
تو کیا آپ نے اس شخص کو دیکھا جس نے اپنی خواہش کو اپنا معبود بنا لیا، اور اللہ نے علم کے باوجود اسے گمراہ کر دیا، اور اس کے کان اور دل پر مہر لگا دی، اور اس کی آنکھ پر پردہ ڈال دیا؟ تو اللہ کے بعد کون اسے ہدایت دے سکتا ہے؟ کیا تم نصیحت نہیں پکڑتے؟
وَقَالُوا۟ مَا هِيَ إِلَّا حَيَاتُنَا ٱلدُّنْيَا نَمُوتُ وَنَحْيَا وَمَا يُهْلِكُنَآ إِلَّا ٱلدَّهْرُۚ وَمَا لَهُم بِذَٰلِكَ مِنْ عِلْمٍ إِنْ هُمْ إِلَّا يَظُنُّونَ ﴿24﴾
اور وہ کہتے ہیں: ہماری زندگی تو بس دنیا کی زندگی ہے، ہم مرتے ہیں اور جیتے ہیں، اور ہمیں صرف زمانہ ہی ہلاک کرتا ہے۔ حالانکہ انہیں اس کا کوئی علم نہیں، وہ صرف گمان ہی کرتے ہیں۔
وَإِذَا تُتْلَىٰ عَلَيْهِمْ ءَايَـٰتُنَا بَيِّنَـٰتٖ مَا كَانَ حُجَّتَهُمْ إِلَّآ أَن قَالُوا۟ ٱئْتُوا۟ بِـَٔابَآئِنَآ إِن كُنتُمْ صَـٰدِقِينَ ﴿25﴾
اور جب ان پر ہماری واضح آیات پڑھی جاتی ہیں تو ان کی دلیل بس یہی ہوتی ہے کہ وہ کہتے ہیں: اگر تم سچے ہو تو ہمارے باپ دادا کو لے آؤ۔
قُلِ ٱللَّهُ يُحْيِيكُمْ ثُمَّ يُمِيتُكُمْ ثُمَّ يَجْمَعُكُمْ إِلَىٰ يَوْمِ ٱلْقِيَـٰمَةِ لَا رَيْبَ فِيهِ وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَعْلَمُونَ ﴿26﴾
کہہ دو: اللہ ہی تمہیں زندہ کرتا ہے، پھر تمہیں موت دیتا ہے، پھر تمہیں قیامت کے دن جمع کرے گا جس میں کوئی شک نہیں، لیکن اکثر لوگ نہیں جانتے۔
وَلِلَّهِ مُلْكُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۚ وَيَوْمَ تَقُومُ ٱلسَّاعَةُ يَوْمَئِذٖ يَخْسَرُ ٱلْمُبْطِلُونَ ﴿27﴾
اور اللہ ہی کے لیے ہے آسمانوں اور زمین کی بادشاہی، اور جس دن قیامت قائم ہوگی، اس دن باطل پرست خسارے میں ہوں گے۔
وَتَرَىٰ كُلَّ أُمَّةٖ جَاثِيَةٗۖ كُلُّ أُمَّةٖ تُدْعَىٰٓ إِلَىٰ كِتَـٰبِهَا ٱلْيَوْمَ تُجْزَوْنَ مَا كُنتُمْ تَعْمَلُونَ ﴿28﴾
اور آپ ہر امت کو گھٹنوں کے بل گرا ہوا دیکھیں گے، ہر امت کو اس کی کتاب کی طرف بلایا جائے گا: آج تمہیں وہ بدلہ دیا جائے گا جو تم کرتے رہے۔
هَـٰذَا كِتَـٰبُنَا يَنطِقُ عَلَيْكُم بِٱلْحَقِّۚ إِنَّا كُنَّا نَسْتَنسِخُ مَا كُنتُمْ تَعْمَلُونَ ﴿29﴾
یہ ہماری کتاب ہے جو تمہارے خلاف حق کے ساتھ بولے گی، ہم وہ سب کچھ لکھواتے رہے جو تم کرتے تھے۔
فَأَمَّا ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ فَيُدْخِلُهُمْ رَبُّهُمْ فِي رَحْمَتِهِۦۚ ذَٰلِكَ هُوَ ٱلْفَوْزُ ٱلْمُبِينُ ﴿30﴾
پس جو لوگ ایمان لائے اور نیک عمل کیے، ان کو ان کا رب اپنی رحمت میں داخل کرے گا، یہی کھلی کامیابی ہے۔
وَأَمَّا ٱلَّذِينَ كَفَرُوٓا۟ أَفَلَمْ تَكُنْ ءَايَـٰتِي تُتْلَىٰ عَلَيْكُمْ فَٱسْتَكْبَرْتُمْ وَكُنتُمْ قَوْمٗا مُّجْرِمِينَ ﴿31﴾
اور جنہوں نے کفر کیا، (ان سے کہا جائے گا): کیا تم پر میری آیات نہیں پڑھی جاتی تھیں؟ پھر تم نے تکبر کیا اور تم مجرم قوم بن گئے۔
وَإِذَا قِيلَ إِنَّ وَعْدَ ٱللَّهِ حَقّٞ وَٱلسَّاعَةُ لَا رَيْبَ فِيهَا قُلْتُم مَّا نَدْرِي مَا ٱلسَّاعَةُ إِن نَّظُنُّ إِلَّا ظَنّٗا وَمَا نَحْنُ بِمُسْتَيْقِنِينَ ﴿32﴾
اور جب کہا جاتا تھا کہ اللہ کا وعدہ سچا ہے اور قیامت میں کوئی شک نہیں، تو تم کہتے تھے: ہمیں نہیں معلوم قیامت کیا ہے، ہم تو صرف گمان کرتے ہیں، اور ہم یقین کرنے والے نہیں۔
وَبَدَا لَهُمْ سَيِّـَٔاتُ مَا عَمِلُوا۟ وَحَاقَ بِهِم مَّا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ ﴿33﴾
اور ان کے سامنے ان کے برے اعمال ظاہر ہو گئے، اور جس چیز کا وہ مذاق اڑاتے تھے وہی ان پر آ پڑی۔
وَقِيلَ ٱلْيَوْمَ نَنسَىٰكُمْ كَمَا نَسِيتُمْ لِقَآءَ يَوْمِكُمْ هَـٰذَا وَمَأْوَىٰكُمُ ٱلنَّارُ وَمَا لَكُم مِّن نَّـٰصِرِينَ ﴿34﴾
اور کہا جائے گا: آج ہم تمہیں بھلا دیں گے جیسے تم نے اپنے اس دن کی ملاقات کو بھلا دیا تھا، اور تمہارا ٹھکانہ آگ ہے، اور تمہارے لیے کوئی مددگار نہیں۔
ذَٰلِكُم بِأَنَّكُمُ ٱتَّخَذْتُمْ ءَايَـٰتِ ٱللَّهِ هُزُوٗا وَغَرَّتْكُمُ ٱلْحَيَوٰةُ ٱلدُّنْيَاۚ فَٱلْيَوْمَ لَا يُخْرَجُونَ مِنْهَا وَلَا هُمْ يُسْتَعْتَبُونَ ﴿35﴾
یہ اس لیے کہ تم نے اللہ کی آیات کو مذاق بنا لیا تھا اور دنیا کی زندگی نے تمہیں دھوکے میں ڈال دیا تھا، پس آج وہ اس سے نہیں نکالے جائیں گے اور نہ ہی ان سے معذرت قبول کی جائے گی۔
فَلِلَّهِ ٱلْحَمْدُ رَبِّ ٱلسَّمَـٰوَٰتِ وَرَبِّ ٱلْأَرْضِ رَبِّ ٱلْعَـٰلَمِينَ ﴿36﴾
پس اللہ ہی کے لیے حمد ہے، جو آسمانوں کا رب ہے، زمین کا رب ہے، اور تمام جہانوں کا رب ہے۔
وَلَهُ ٱلْكِبْرِيَآءُ فِي ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۚ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿37﴾
اور اسی کے لیے آسمانوں اور زمین میں عظمت ہے، اور وہی زبردست، حکمت والا ہے۔
