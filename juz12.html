12 پارہ


بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
وَمَا مِن دَآبَّةٖ فِي ٱلْأَرْضِ إِلَّا عَلَى ٱللَّهِ رِزْقُهَا وَيَعْلَمُ مُسْتَقَرَّهَا وَمُسْتَوْدَعَهَاۚ كُلّٞ فِي كِتَٰبٖ مُّبِينٖ ﴿6﴾
اور زمین میں کوئی چلنے والا ایسا نہیں جس کا رزق اللہ کے ذمے نہ ہو، اور وہ اس کے ٹھہرنے کی جگہ اور اس کے سونپے جانے کی جگہ کو جانتا ہے، سب کچھ ایک واضح کتاب میں ہے۔
وَهُوَ ٱلَّذِي خَلَقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ فِي سِتَّةِ أَيَّامٖ وَكَانَ عَرْشُهُۥ عَلَى ٱلْمَآءِ لِيَبْلُوَكُمْ أَيُّكُمْ أَحْسَنُ عَمَلٗاۗ وَلَئِن قُلْتَ إِنَّكُم مَّبْعُوثُونَ مِنۢ بَعْدِ ٱلْمَوْتِ لَيَقُولَنَّ ٱلَّذِينَ كَفَرُوٓا۟ إِنْ هَـٰذَآ إِلَّا سِحْرٞ مُّبِينٞ ﴿7﴾
اور وہی ہے جس نے آسمانوں اور زمین کو چھ دنوں میں پیدا کیا، اور اس کا عرش پانی پر تھا، تاکہ وہ تمہیں آزمائے کہ تم میں سے کون اچھے عمل کرتا ہے، اور اگر تم کہو کہ تم مرنے کے بعد اٹھائے جاؤ گے تو کافر ضرور کہیں گے: یہ تو کھلا جادو ہے۔
وَلَئِنْ أَخَّرْنَا عَنْهُمُ ٱلْعَذَابَ إِلَىٰٓ أُمَّةٖ مَّعْدُودَةٖ لَّيَقُولُنَّ مَا يَحْبِسُهُۥٓۚ أَلَا يَوْمَ يَأْتِيهِمْ لَيْسَ مَصْرُوفًا عَنْهُمْ وَحَاقَ بِهِم مَّا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ ﴿8﴾
اور اگر ہم ان سے عذاب کو ایک گنی چنی مدت تک مؤخر کر دیں تو وہ ضرور کہیں گے: اسے کیا چیز روک رہی ہے؟ خبردار! جس دن وہ ان پر آ جائے گا تو وہ ان سے ہٹایا نہ جائے گا، اور وہی چیز ان کو گھیر لے گی جس کا وہ مذاق اڑاتے تھے۔
وَلَئِنْ أَذَقْنَا ٱلْإِنسَـٰنَ مِنَّا رَحْمَةٗ ثُمَّ نَزَعْنَـٰهَا مِنْهُ إِنَّهُۥ لَيَـُٔوسٞ كَفُورٞ ﴿9﴾
اور اگر ہم انسان کو اپنی طرف سے کوئی رحمت چکھائیں، پھر اسے اس سے چھین لیں، تو وہ ضرور مایوس اور ناشکرا ہو جاتا ہے۔
وَلَئِنْ أَذَقْنَـٰهُ نَعْمَآءَ بَعْدَ ضَرَّآءَ مَسَّتْهُ لَيَقُولَنَّ ذَهَبَ ٱلسَّيِّـَٔاتُ عَنِّيٓۚ إِنَّهُۥ لَفَرِحٞ فَخُورٌ ﴿10﴾
اور اگر ہم اسے تکلیف کے بعد نعمت چکھائیں تو وہ ضرور کہتا ہے: مجھ سے برائیاں دور ہو گئیں، بے شک وہ خوشی میں اترانے والا ہے۔
إِلَّا ٱلَّذِينَ صَبَرُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ أُوْلَـٰٓئِكَ لَهُم مَّغْفِرَةٞ وَأَجْرٞ كَبِيرٞ ﴿11﴾
سوائے ان لوگوں کے جو صبر کرتے ہیں اور نیک عمل کرتے ہیں، ان کے لیے بخشش اور بڑا اجر ہے۔
فَلَعَلَّكَ تَارِكٞ بَعْضَ مَا يُوحَىٰٓ إِلَيْكَ وَضَآئِقُۢ بِهِۦ صَدْرُكَ أَن يَقُولُوا۟ لَوْلَآ أُنزِلَ عَلَيْهِ كَنزٌ أَوْ جَآءَ مَعَهُۥ مَلَكٞۚ إِنَّمَآ أَنتَ نَذِيرٞۚ وَٱللَّهُ عَلَىٰ كُلِّ شَيْءٖ وَكِيلٌ ﴿12﴾
تو شاید تم اس چیز کے کچھ حصے کو چھوڑ دو جو تمہاری طرف وحی کی گئی ہے، اور تمہارا سینہ اس سے تنگ ہو کہ وہ کہیں: اس پر کوئی خزانہ کیوں نہ اتارا گیا یا اس کے ساتھ کوئی فرشتہ کیوں نہ آیا؟ تم تو صرف ڈرانے والے ہو، اور اللہ ہر چیز پر نگہبان ہے۔
أَمْ يَقُولُونَ ٱفْتَرَىٰهُۖ قُلْ فَأْتُوا۟ بِعَشْرِ سُوَرٖ مِّثْلِهِۦ مُفْتَرَيَـٰتٖ وَٱدْعُوا۟ مَنِ ٱسْتَطَعْتُم مِّن دُونِ ٱللَّهِ إِن كُنتُمْ صَـٰدِقِينَ ﴿13﴾
کیا وہ کہتے ہیں: اس نے اسے خود گھڑ لیا ہے؟ کہہ دو: تم بھی اس جیسے دس سورتیں گھڑی ہوئی لے آؤ، اور اللہ کے سوا جسے چاہو بلا لو، اگر تم سچے ہو۔
فَإِلَّمْ يَسْتَجِيبُوا۟ لَكُمْ فَٱعْلَمُوٓا۟ أَنَّمَآ أُنزِلَ بِعِلْمِ ٱللَّهِ وَأَن لَّآ إِلَـٰهَ إِلَّا هُوَۖ فَهَلْ أَنتُم مُّسْلِمُونَ ﴿14﴾
پھر اگر وہ تمہاری بات قبول نہ کریں تو جان لو کہ یہ اللہ کے علم سے نازل ہوا ہے، اور اس کے سوا کوئی معبود نہیں، تو کیا تم مسلمان ہو گئے؟
مَن كَانَ يُرِيدُ ٱلْحَيَوٰةَ ٱلدُّنْيَا وَزِينَتَهَا نُوَفِّ إِلَيْهِمْ أَعْمَـٰلَهُمْ فِيهَا وَهُمْ فِيهَا لَا يُبْخَسُونَ ﴿15﴾
جو لوگ دنیا کی زندگی اور اس کی زینت چاہتے ہیں، ہم ان کے اعمال انہیں وہیں پورے دے دیتے ہیں، اور ان کے ساتھ کمی نہیں کی جاتی۔
أُوْلَـٰٓئِكَ ٱلَّذِينَ لَيْسَ لَهُمْ فِي ٱلْـَٔاخِرَةِ إِلَّا ٱلنَّارُۖ وَحَبِطَ مَا صَنَعُوا۟ فِيهَا وَبَٰطِلٞ مَّا كَانُوا۟ يَعْمَلُونَ ﴿16﴾
یہی لوگ ہیں جن کے لیے آخرت میں صرف آگ ہے، اور جو کچھ انہوں نے کیا وہ برباد ہو گیا، اور جو عمل وہ کرتے تھے وہ باطل ہو گئے۔
أَفَمَن كَانَ عَلَىٰ بَيِّنَةٖ مِّن رَّبِّهِۦ وَيَتْلُوهُ شَاهِدٞ مِّنْهُۥ وَمِن قَبْلِهِۦ كِتَٰبُ مُوسَىٰٓ إِمَامٗا وَرَحْمَةًۚ أُوْلَـٰٓئِكَ يُؤْمِنُونَ بِهِۦۚ وَمَن يَكْفُرْ بِهِۦ مِنَ ٱلْأَحْزَابِ فَٱلنَّارُ مَوْعِدُهُۥۚ فَلَا تَكُ فِي مِرْيَةٖ مِّنْهُۚ إِنَّهُ ٱلْحَقُّ مِن رَّبِّكَ وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يُؤْمِنُونَ ﴿17﴾
تو کیا وہ شخص جو اپنے رب کی طرف سے واضح دلیل پر ہو، اور اس کے پیچھے ایک گواہ بھی ہو، اور اس سے پہلے موسیٰ کی کتاب بھی ہو جو امام اور رحمت ہے — یہی لوگ اس پر ایمان لاتے ہیں، اور جو اس کا انکار کرے ان گروہوں میں سے، تو آگ اس کا وعدہ ہے، پس تم اس کے بارے میں شک نہ کرو، بے شک وہ تمہارے رب کی طرف سے حق ہے، لیکن اکثر لوگ ایمان نہیں لاتے۔
وَمَنْ أَظْلَمُ مِمَّنِ ٱفْتَرَىٰ عَلَى ٱللَّهِ كَذِبًاۚ أُوْلَـٰٓئِكَ يُعْرَضُونَ عَلَىٰ رَبِّهِمْ وَيَقُولُ ٱلْأَشْهَـٰدُ هَـٰٓؤُلَآءِ ٱلَّذِينَ كَذَبُوا۟ عَلَىٰ رَبِّهِمْۚ أَلَا لَعْنَةُ ٱللَّهِ عَلَى ٱلظَّـٰلِمِينَ ﴿18﴾
اور اس سے بڑھ کر ظالم کون ہے جو اللہ پر جھوٹ باندھے؟ یہی لوگ اپنے رب کے سامنے پیش کیے جائیں گے، اور گواہ کہیں گے: یہی ہیں جنہوں نے اپنے رب پر جھوٹ باندھا، خبردار! اللہ کی لعنت ہو ظالموں پر۔
ٱلَّذِينَ يَصُدُّونَ عَن سَبِيلِ ٱللَّهِ وَيَبْغُونَهَا عِوَجٗا وَهُم بِٱلْـَٔاخِرَةِ هُمْ كَـٰفِرُونَ ﴿19﴾
جو لوگ اللہ کے راستے سے روکتے ہیں اور اس میں کجی چاہتے ہیں، اور وہ آخرت کے منکر ہیں۔
أُوْلَـٰٓئِكَ لَمْ يَكُونُوا۟ مُعْجِزِينَ فِي ٱلْأَرْضِ وَمَا كَانَ لَهُم مِّن دُونِ ٱللَّهِ مِنْ أَوْلِيَآءَۚ يُضَٰعَفُ لَهُمُ ٱلْعَذَابُۚ مَا كَانُوا۟ يَسْتَطِيعُونَ ٱلسَّمْعَ وَمَا كَانُوا۟ يُبْصِرُونَ ﴿20﴾
یہ لوگ زمین میں اللہ کو عاجز کرنے والے نہ تھے، اور ان کے لیے اللہ کے سوا کوئی مددگار نہ تھا، ان کے لیے عذاب دوگنا کیا جائے گا، وہ نہ سن سکتے تھے اور نہ دیکھ سکتے تھے۔
أُوْلَـٰٓئِكَ ٱلَّذِينَ خَسِرُوٓا۟ أَنفُسَهُمْ وَضَلَّ عَنْهُم مَّا كَانُوا۟ يَفْتَرُونَ ﴿21﴾
یہی لوگ ہیں جنہوں نے اپنے آپ کو نقصان پہنچایا، اور جو کچھ وہ جھوٹ گھڑتے تھے وہ ان سے گم ہو گیا۔
لَا جَرَمَ أَنَّهُمْ فِي ٱلْـَٔاخِرَةِ هُمُ ٱلْأَخْسَرُونَ ﴿22﴾
یقیناً وہی لوگ آخرت میں سب سے زیادہ نقصان اٹھانے والے ہیں۔
إِنَّ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ وَأَخْبَتُوٓا۟ إِلَىٰ رَبِّهِمْ أُوْلَـٰٓئِكَ أَصْحَـٰبُ ٱلْجَنَّةِ هُمْ فِيهَا خَـٰلِدُونَ ﴿23﴾
بے شک جو لوگ ایمان لائے اور نیک عمل کیے اور اپنے رب کے سامنے عاجزی اختیار کی، وہی جنت والے ہیں، وہ اس میں ہمیشہ رہیں گے۔
مَثَلُ ٱلْفَرِيقَيْنِ كَٱلْأَعْمَىٰ وَٱلْأَصَمِّ وَٱلْبَصِيرِ وَٱلسَّمِيعِۚ هَلْ يَسْتَوِيَانِ مَثَلًاۚ أَفَلَا تَذَكَّرُونَ ﴿24﴾
دونوں گروہوں کی مثال اندھے اور بہرے، اور دیکھنے والے اور سننے والے کی سی ہے، کیا یہ دونوں مثال میں برابر ہو سکتے ہیں؟ تو کیا تم نصیحت حاصل نہیں کرتے؟
وَلَقَدْ أَرْسَلْنَا نُوحًا إِلَىٰ قَوْمِهِۦٓ إِنِّي لَكُمْ نَذِيرٞ مُّبِينٌ ﴿25﴾
اور بے شک ہم نے نوح کو اس کی قوم کی طرف بھیجا، کہ میں تمہارے لیے کھلا ہوا ڈرانے والا ہوں۔
أَن لَّا تَعْبُدُوٓا۟ إِلَّا ٱللَّهَۖ إِنِّيٓ أَخَافُ عَلَيْكُمْ عَذَابَ يَوْمٍ أَلِيمٖ ﴿26﴾
کہ تم اللہ کے سوا کسی کی عبادت نہ کرو، بے شک میں تم پر ایک دردناک دن کے عذاب سے ڈرتا ہوں۔
فَقَالَ ٱلْمَلَأُ ٱلَّذِينَ كَفَرُوا۟ مِن قَوْمِهِۦ مَا نَرَىٰكَ إِلَّا بَشَرٗا مِّثْلَنَا وَمَا نَرَىٰكَ ٱتَّبَعَكَ إِلَّا ٱلَّذِينَ هُمۡ أَرَاذِلُنَا بَادِيَ ٱلرَّأْيِۖ وَمَا نَرَىٰ لَكُمْ عَلَيْنَا مِن فَضْلِۢ بَلْ نَظُنُّكُمْ كَـٰذِبِينَ ﴿27﴾
تو اس کی قوم کے سرداروں نے جو کافر تھے کہا: ہم تمہیں اپنے جیسا انسان ہی دیکھتے ہیں، اور ہم نہیں دیکھتے کہ تمہاری پیروی کی ہو مگر ان لوگوں نے جو ہم میں سے کمینے ہیں، وہ بھی سرسری رائے سے، اور ہم تمہارے لیے اپنے اوپر کوئی فضیلت نہیں دیکھتے، بلکہ ہم تمہیں جھوٹا سمجھتے ہیں۔
قَالَ يَـٰقَوْمِ أَرَءَيْتُمْ إِن كُنتُ عَلَىٰ بَيِّنَةٖ مِّن رَّبِّي وَءَاتَىٰنِي رَحْمَةٗ مِّنْ عِندِهِۦ فَعُمِّيَتْ عَلَيْكُمۡ أَنُلْزِمُكُمُوهَا وَأَنتُمْ لَهَا كَـٰرِهُونَ ﴿28﴾
اس نے کہا: اے میری قوم! کیا تم نے دیکھا اگر میں اپنے رب کی طرف سے ایک دلیل پر ہوں، اور اس نے مجھے اپنی طرف سے رحمت عطا کی ہو، پھر وہ تم پر چھپا دی گئی ہو، تو کیا ہم اسے تم پر زبردستی تھوپ دیں حالانکہ تم اسے ناپسند کرتے ہو؟
وَيَـٰقَوْمِ لَآ أَسْـَٔلُكُمْ عَلَيْهِ مَالًاۖ إِنْ أَجْرِيَ إِلَّا عَلَى ٱللَّهِۚ وَمَآ أَنَا۠ بِطَارِدِ ٱلَّذِينَ ءَامَنُوٓا۟ۚ إِنَّهُم مُّلَـٰقُوا۟ رَبِّهِمْ وَلَـٰكِنِّيٓ أَرَىٰكُمْ قَوْمٗا تَجْهَلُونَ ﴿29﴾
اور اے میری قوم! میں تم سے اس (دعوت) پر کوئی مال نہیں مانگتا، میرا اجر تو صرف اللہ پر ہے، اور میں ایمان والوں کو نکالنے والا نہیں ہوں، بے شک وہ اپنے رب سے ملنے والے ہیں، لیکن میں تمہیں ایک جاہل قوم دیکھتا ہوں۔
وَيَـٰقَوْمِ مَن يَنصُرُنِي مِنَ ٱللَّهِ إِن طَرَدتُّهُمۡۚ أَفَلَا تَذَكَّرُونَ ﴿30﴾
اور اے میری قوم! اگر میں انہیں نکال دوں تو اللہ کے مقابلے میں کون میری مدد کرے گا؟ تو کیا تم نصیحت حاصل نہیں کرتے؟
وَلَآ أَقُولُ لَكُمْ عِندِي خَزَآئِنُ ٱللَّهِ وَلَآ أَعْلَمُ ٱلْغَيْبَ وَلَآ أَقُولُ إِنِّي مَلَكٌ وَلَآ أَقُولُ لِلَّذِينَ تَزْدَرِيٓ أَعْيُنُكُمْ لَن يُؤْتِيَهُمُ ٱللَّهُ خَيْرٗاۖ ٱللَّهُ أَعْلَمُ بِمَا فِيٓ أَنفُسِهِمْۖ إِنِّيٓ إِذٗا لَّمِنَ ٱلظَّـٰلِمِينَ ﴿31﴾
اور میں تم سے یہ نہیں کہتا کہ میرے پاس اللہ کے خزانے ہیں، اور نہ میں غیب جانتا ہوں، اور نہ یہ کہتا ہوں کہ میں فرشتہ ہوں، اور نہ ان لوگوں کے بارے میں جنہیں تم حقیر سمجھتے ہو یہ کہتا ہوں کہ اللہ انہیں کوئی بھلائی نہیں دے گا، اللہ خوب جانتا ہے جو ان کے دلوں میں ہے، اگر میں ایسا کہوں تو میں ظالموں میں سے ہوں۔
قَالُوا۟ يَـٰنُوحُ قَدْ جَـٰدَلْتَنَا فَأَكْثَرْتَ جِدَٰلَنَا فَأْتِنَا بِمَا تَعِدُنَآ إِن كُنتَ مِنَ ٱلصَّـٰدِقِينَ ﴿32﴾
انہوں نے کہا: اے نوح! تم نے ہم سے بحث کی اور بہت بحث کی، تو اب وہ چیز لے آؤ جس کا تم وعدہ کرتے ہو، اگر تم سچے ہو۔
قَالَ إِنَّمَا يَأْتِيكُم بِهِ ٱللَّهُ إِن شَآءَ وَمَآ أَنتُم بِمُعْجِزِينَ ﴿33﴾
نوح نے کہا: وہ تو اللہ ہی تم پر لائے گا اگر وہ چاہے، اور تم اسے روکنے والے نہیں ہو۔
وَلَا يَنفَعُكُمْ نُصْحِيٓ إِنْ أَرَدتُّ أَنْ أَنصَحَ لَكُمْ إِن كَانَ ٱللَّهُ يُرِيدُ أَن يُغْوِيَكُمْۚ هُوَ رَبُّكُمْ وَإِلَيْهِ تُرْجَعُونَ ﴿34﴾
اور اگر میں تمہیں نصیحت کرنا چاہوں تو میری نصیحت تمہیں فائدہ نہیں دے گی، اگر اللہ تمہیں گمراہ کرنا چاہے، وہی تمہارا رب ہے، اور اسی کی طرف تم لوٹائے جاؤ گے۔
أَمْ يَقُولُونَ ٱفْتَرَىٰهُۖ قُلْ إِنِ ٱفْتَرَيْتُهُۥ فَعَلَيَّ إِجْرَامِي وَأَنَا۠ بَرِيٓءٞ مِّمَّا تُجْرِمُونَ ﴿35﴾
کیا وہ کہتے ہیں: اس نے اسے گھڑ لیا ہے؟ کہہ دو: اگر میں نے اسے گھڑ لیا ہے تو میرا جرم میرے ذمے ہے، اور میں تمہارے جرم سے بری ہوں۔
وَأُوحِيَ إِلَىٰ نُوحٍ أَنَّهُۥ لَن يُؤْمِنَ مِن قَوْمِكَ إِلَّا مَن قَدْ ءَامَنَ فَلَا تَبْتَئِسْ بِمَا كَانُوا۟ يَفْعَلُونَ ﴿36﴾
اور نوح کی طرف وحی کی گئی کہ تمہاری قوم میں سے جو ایمان لا چکے ان کے سوا کوئی ایمان نہیں لائے گا، پس ان کے اعمال پر غم نہ کرو۔
وَٱصْنَعِ ٱلْفُلْكَ بِأَعْيُنِنَا وَوَحْيِنَا وَلَا تُخَـٰطِبْنِي فِي ٱلَّذِينَ ظَلَمُوٓا۟ إِنَّهُم مُّغْرَقُونَ ﴿37﴾
اور ہماری نگرانی اور وحی کے مطابق کشتی بناؤ، اور ان لوگوں کے بارے میں مجھ سے بات نہ کرنا جنہوں نے ظلم کیا، بے شک وہ غرق ہونے والے ہیں۔
وَيَصْنَعُ ٱلْفُلْكَ وَكُلَّمَا مَرَّ عَلَيْهِ مَلَأٞ مِّن قَوْمِهِۦ سَخِرُوا۟ مِنْهُۚ قَالَ إِن تَسْخَرُوا۟ مِنَّا فَإِنَّا نَسْخَرُ مِنكُمْ كَمَا تَسْخَرُونَ ﴿38﴾
اور وہ کشتی بناتا رہا، اور جب بھی اس کی قوم کے سردار اس کے پاس سے گزرتے تو اس کا مذاق اڑاتے، اس نے کہا: اگر تم ہم سے مذاق کرتے ہو تو ہم بھی تم سے اسی طرح مذاق کریں گے جیسے تم کرتے ہو۔
فَسَوْفَ تَعْلَمُونَ مَن يَأْتِيهِ عَذَابٞ يُخْزِيهِ وَيَحِلُّ عَلَيْهِ عَذَابٞ مُّقِيمٌ ﴿39﴾
پھر تم جلد جان لو گے کہ کس پر وہ عذاب آتا ہے جو اسے رسوا کرتا ہے، اور کس پر وہ دائمی عذاب نازل ہوتا ہے۔
حَتَّىٰٓ إِذَا جَآءَ أَمْرُنَا وَفَارَ ٱلتَّنُّورُ قُلْنَا ٱحْمِلْ فِيهَا مِن كُلّٖ زَوْجَيْنِ ٱثْنَيْنِ وَأَهْلَكَ إِلَّا مَن سَبَقَ عَلَيْهِ ٱلْقَوْلُ وَمَنْ ءَامَنَۚ وَمَآ ءَامَنَ مَعَهُۥٓ إِلَّا قَلِيلٞ ﴿40﴾
یہاں تک کہ جب ہمارا حکم آ گیا اور تنور ابلنے لگا، ہم نے کہا: اس میں ہر قسم کے جوڑے دو دو لے لو، اور اپنے گھر والوں کو بھی، سوائے اس کے جس پر فیصلہ ہو چکا، اور ایمان والوں کو بھی، اور اس کے ساتھ بہت کم لوگ ایمان لائے۔
وَقَالَ ٱرْكَبُوا۟ فِيهَا بِسْمِ ٱللَّهِ مَجْرَىٰهَا وَمُرْسَىٰهَآۚ إِنَّ رَبِّي لَغَفُورٞ رَّحِيمٞ ﴿41﴾
اور اس نے کہا: اس میں سوار ہو جاؤ، اللہ کے نام سے ہی اس کا چلنا اور ٹھہرنا ہے، بے شک میرا رب بخشنے والا، رحم کرنے والا ہے۔
وَهِيَ تَجْرِي بِهِمْ فِي مَوْجٖۢ كَٱلْجِبَالِ وَنَادَىٰ نُوحٌ ٱبْنَهُۥ وَكَانَ فِي مَعْزِلٖ يَـٰبُنَيَّ ٱرْكَب مَّعَنَا وَلَا تَكُن مَّعَ ٱلْكَـٰفِرِينَ ﴿42﴾
اور وہ کشتی انہیں لے کر پہاڑوں جیسے موجوں میں چلنے لگی، اور نوح نے اپنے بیٹے کو پکارا، جو الگ تھا: اے میرے بیٹے! ہمارے ساتھ سوار ہو جاؤ اور کافروں کے ساتھ نہ رہو۔
قَالَ سَـَٔاوِيٓ إِلَىٰ جَبَلٖ يَعْصِمُنِي مِنَ ٱلْمَآءِۚ قَالَ لَا عَاصِمَ ٱلْيَوْمَ مِنْ أَمْرِ ٱللَّهِ إِلَّا مَن رَّحِمَۚ وَحَالَ بَيْنَهُمَا ٱلْمَوْجُ فَكَانَ مِنَ ٱلْمُغْرَقِينَ ﴿43﴾
اس نے کہا: میں کسی پہاڑ پر چڑھ کر پانی سے بچ جاؤں گا، نوح نے کہا: آج اللہ کے حکم سے کوئی بچانے والا نہیں، سوائے جس پر وہ رحم کرے، اور ان دونوں کے درمیان موج آ گئی، تو وہ ڈوبنے والوں میں سے ہو گیا۔
وَقِيلَ يَـٰٓأَرْضُ ٱبْلَعِي مَآءَكِ وَيَـٰسَمَآءُ أَقْلِعِي وَغِيضَ ٱلْمَآءُ وَقُضِيَ ٱلْأَمْرُ وَٱسْتَوَتْ عَلَى ٱلْجُودِيِّۖ وَقِيلَ بُعْدٗا لِّلْقَوْمِ ٱلظَّـٰلِمِينَ ﴿44﴾
اور کہا گیا: اے زمین! اپنا پانی نگل لے، اور اے آسمان! تھم جا، اور پانی خشک کر دیا گیا، اور کام پورا ہو گیا، اور کشتی جودی پر ٹھہر گئی، اور کہا گیا: دور ہو جائے ظالم قوم۔
وَنَادَىٰ نُوحٞ رَّبَّهُۥ فَقَالَ رَبِّ إِنَّ ٱبْنِي مِنْ أَهْلِي وَإِنَّ وَعْدَكَ ٱلْحَقُّ وَأَنتَ أَحْكَمُ ٱلْحَـٰكِمِينَ ﴿45﴾
اور نوح نے اپنے رب کو پکارا، کہا: اے میرے رب! میرا بیٹا میرے گھر والوں میں سے ہے، اور بے شک تیرا وعدہ سچا ہے، اور تو سب فیصلہ کرنے والوں سے بہتر ہے۔
قَالَ يَـٰنُوحُ إِنَّهُۥ لَيْسَ مِنْ أَهْلِكَ إِنَّهُۥ عَمَلٌ غَيْرُ صَـٰلِحٖ فَلَا تَسْـَٔلْنِ مَا لَيْسَ لَكَ بِهِۦ عِلْمٌۖ إِنِّيٓ أَعِظُكَ أَن تَكُونَ مِنَ ٱلْجَـٰهِلِينَ ﴿46﴾
اللہ نے فرمایا: اے نوح! وہ تیرے گھر والوں میں سے نہیں ہے، بے شک وہ ناپسندیدہ عمل ہے، پس مجھ سے وہ چیز نہ مانگو جس کا تمہیں علم نہیں، میں تمہیں نصیحت کرتا ہوں کہ جاہلوں میں نہ ہو جانا۔
قَالَ رَبِّ إِنِّيٓ أَعُوذُ بِكَ أَنْ أَسْـَٔلَكَ مَا لَيْسَ لِي بِهِۦ عِلْمٞۖ وَإِلَّا تَغْفِرْ لِي وَتَرْحَمْنِيٓ أَكُن مِّنَ ٱلْخَـٰسِرِينَ ﴿47﴾
نوح نے کہا: اے میرے رب! میں تیری پناہ مانگتا ہوں کہ تجھ سے وہ چیز مانگوں جس کا مجھے علم نہیں، اور اگر تو مجھے معاف نہ کرے اور مجھ پر رحم نہ کرے تو میں نقصان اٹھانے والوں میں سے ہو جاؤں گا۔
قِيلَ يَـٰنُوحُ ٱهْبِطْ بِسَلَـٰمٖ مِّنَّا وَبَرَكَـٰتٍ عَلَيْكَ وَعَلَىٰٓ أُمَمٖ مِّمَّن مَّعَكَۚ وَأُمَمٞ سَنُمَتِّعُهُمْ ثُمَّ يَمَسُّهُم مِّنَّا عَذَابٌ أَلِيمٌ ﴿48﴾
کہا گیا: اے نوح! اترو ہماری طرف سے سلامتی اور برکتوں کے ساتھ تم پر اور ان امتوں پر جو تمہارے ساتھ ہیں، اور کچھ امتیں ہیں جنہیں ہم فائدہ دیں گے، پھر انہیں ہماری طرف سے دردناک عذاب پہنچے گا۔
تِلْكَ مِنْ أَنۢبَآءِ ٱلْغَيْبِ نُوحِيهَآ إِلَيْكَۖ مَا كُنتَ تَعْلَمُهَآ أَنتَ وَلَا قَوْمُكَ مِن قَبْلِ هَـٰذَاۖ فَٱصْبِرْۖ إِنَّ ٱلْعَـٰقِبَةَ لِلْمُتَّقِينَ ﴿49﴾
یہ غیب کی خبریں ہیں جو ہم تمہیں وحی کے ذریعے بتاتے ہیں، تم انہیں اس سے پہلے نہ جانتے تھے، نہ تم اور نہ تمہاری قوم، پس صبر کرو، بے شک انجام متقیوں کے لیے ہے۔
وَإِلَىٰ عَادٍ أَخَاهُمْ هُودٗاۚ قَالَ يَـٰقَوْمِ ٱعْبُدُوا۟ ٱللَّهَ مَا لَكُم مِّنْ إِلَـٰهٍ غَيْرُهُۚ إِنْ أَنتُمْ إِلَّا مُفْتَرُونَ ﴿50﴾
اور ہم نے عاد کی طرف ان کے بھائی ہود کو بھیجا، اس نے کہا: اے میری قوم! اللہ کی عبادت کرو، اس کے سوا تمہارا کوئی معبود نہیں، تم تو صرف جھوٹ گھڑنے والے ہو۔
يَـٰقَوْمِ لَآ أَسْـَٔلُكُمْ عَلَيْهِ أَجْرًاۖ إِنْ أَجْرِيَ إِلَّا عَلَى ٱلَّذِي فَطَرَنِيٓۚ أَفَلَا تَعْقِلُونَ ﴿51﴾
اے میری قوم! میں تم سے اس پر کوئی اجر نہیں مانگتا، میرا اجر تو صرف اس پر ہے جس نے مجھے پیدا کیا، تو کیا تم عقل نہیں رکھتے؟
وَيَـٰقَوْمِ ٱسْتَغْفِرُوا۟ رَبَّكُمْ ثُمَّ تُوبُوٓا۟ إِلَيْهِ يُرْسِلِ ٱلسَّمَآءَ عَلَيْكُم مِّدْرَارٗا وَيَزِدْكُمْ قُوَّةً إِلَىٰ قُوَّتِكُمْ وَلَا تَتَوَلَّوْا۟ مُجْرِمِينَ ﴿52﴾
اور اے میری قوم! اپنے رب سے معافی مانگو، پھر اس کی طرف رجوع کرو، وہ تم پر آسمان سے خوب بارش برسائے گا، اور تمہاری قوت پر مزید قوت بڑھا دے گا، اور مجرم بن کر منہ نہ موڑو۔
قَالُوا۟ يَـٰهُودُ مَا جِئْتَنَا بِبَيِّنَةٖ وَمَا نَحْنُ بِتَارِكِيٓ ءَالِهَتِنَا عَنْ قَوْلِكَ وَمَا نَحْنُ لَكَ بِمُؤْمِنِينَ ﴿53﴾
انہوں نے کہا: اے ہود! تم ہمارے پاس کوئی واضح دلیل نہیں لائے، اور ہم تمہارے کہنے سے اپنے معبودوں کو چھوڑنے والے نہیں، اور ہم تم پر ایمان لانے والے نہیں۔
إِن نَّقُولُ إِلَّا ٱعْتَرَىٰكَ بَعْضُ ءَالِهَتِنَا بِسُوٓءٖۚ قَالَ إِنِّيٓ أُشْهِدُ ٱللَّهَ وَٱشْهَدُوٓا۟ أَنِّي بَرِيٓءٞ مِّمَّا تُشْرِكُونَ ﴿54﴾
ہم تو یہی کہتے ہیں کہ ہمارے کسی معبود نے تمہیں کچھ نقصان پہنچایا ہے، ہود نے کہا: میں اللہ کو گواہ بناتا ہوں، اور تم بھی گواہ رہو کہ میں ان سب سے بری ہوں جنہیں تم شریک ٹھہراتے ہو۔
مِن دُونِهِۦ فَكِيدُونِي جَمِيعٗا ثُمَّ لَا تُنظِرُونِ ﴿55﴾
اللہ کے سوا، پس تم سب مل کر میرے خلاف چالیں چل لو، پھر مجھے مہلت نہ دو۔
إِنِّي تَوَكَّلْتُ عَلَى ٱللَّهِ رَبِّي وَرَبِّكُمۡۚ مَّا مِن دَآبَّةٍ إِلَّا هُوَ ءَاخِذُۢ بِنَاصِيَتِهَآۚ إِنَّ رَبِّي عَلَىٰ صِرَٰطٖ مُّسْتَقِيمٖ ﴿56﴾
میں نے اللہ پر بھروسہ کیا ہے، جو میرا بھی رب ہے اور تمہارا بھی رب، کوئی بھی جاندار ایسا نہیں جس کی پیشانی اللہ کے قبضے میں نہ ہو، بے شک میرا رب سیدھے راستے پر ہے۔
فَإِن تَوَلَّوْا۟ فَقَدْ أَبْلَغْتُكُم مَّآ أُرْسِلْتُ بِهِۦٓ إِلَيْكُمْۚ وَيَسْتَخْلِفُ رَبِّي قَوْمًا غَيْرَكُمْ وَلَا تَضُرُّونَهُۥ شَيْـٔٗاۚ إِنَّ رَبِّي عَلَىٰ كُلِّ شَيْءٍ حَفِيظٞ ﴿57﴾
پھر اگر تم منہ موڑو تو میں نے تمہیں وہ بات پہنچا دی ہے جس کے ساتھ میں تمہاری طرف بھیجا گیا تھا، اور میرا رب تمہارے بعد کسی اور قوم کو تمہاری جگہ لے آئے گا، اور تم اس کا کچھ بھی نقصان نہیں کر سکو گے، بے شک میرا رب ہر چیز پر نگہبان ہے۔
وَلَمَّا جَآءَ أَمْرُنَا نَجَّيْنَا هُودٗا وَٱلَّذِينَ ءَامَنُوا۟ مَعَهُۥ بِرَحْمَةٖ مِّنَّا وَنَجَّيْنَـٰهُم مِّنْ عَذَابٍ غَلِيظٖ ﴿58﴾
اور جب ہمارا حکم آ پہنچا تو ہم نے ہود اور ان لوگوں کو جو اس کے ساتھ ایمان لائے تھے، اپنی رحمت سے نجات دی، اور ہم نے انہیں سخت عذاب سے بچا لیا۔
وَتِلْكَ عَادٞۚ جَحَدُوا۟ بِـَٔايَـٰتِ رَبِّهِمْ وَعَصَوْا۟ رُسُلَهُۥ وَٱتَّبَعُوٓا۟ أَمْرَ كُلِّ جَبَّارٍ عَنِيدٖ ﴿59﴾
اور یہ عاد تھے، جنہوں نے اپنے رب کی آیات کا انکار کیا، اور اس کے رسولوں کی نافرمانی کی، اور ہر سرکش ضدی کے حکم کی پیروی کی۔
وَأُتْبِعُوا۟ فِي هَـٰذِهِ ٱلدُّنْيَا لَعْنَةٗ وَيَوْمَ ٱلْقِيَـٰمَةِۚ أَلَآ إِنَّ عَادٗا كَفَرُوا۟ رَبَّهُمْۚ أَلَا بُعْدٗا لِّعَادٍ قَوْمِ هُودٖ ﴿60﴾
اور ان پر دنیا میں لعنت مسلط کی گئی، اور قیامت کے دن بھی، خبردار! بے شک عاد نے اپنے رب کا انکار کیا، خبردار! ہود کی قوم عاد کے لیے دوری ہے۔
وَإِلَىٰ ثَمُودَ أَخَاهُمْ صَـٰلِحٗاۚ قَالَ يَـٰقَوْمِ ٱعْبُدُوا۟ ٱللَّهَ مَا لَكُم مِّنْ إِلَـٰهٍ غَيْرُهُۖ هُوَ أَنشَأَكُم مِّنَ ٱلْأَرْضِ وَٱسْتَعْمَرَكُمْ فِيهَا فَٱسْتَغْفِرُوهُ ثُمَّ تُوبُوٓا۟ إِلَيْهِۚ إِنَّ رَبِّي قَرِيبٞ مُّجِيبٞ ﴿61﴾
اور ہم نے ثمود کی طرف ان کے بھائی صالح کو بھیجا، اس نے کہا: اے میری قوم! اللہ کی عبادت کرو، اس کے سوا تمہارا کوئی معبود نہیں، اسی نے تمہیں زمین سے پیدا کیا اور اس میں تمہیں بسایا، پس اس سے معافی مانگو، پھر اس کی طرف رجوع کرو، بے شک میرا رب قریب ہے، قبول کرنے والا ہے۔
قَالُوا۟ يَـٰصَـٰلِحُ قَدْ كُنتَ فِينَا مَرْجُوّٗا قَبْلَ هَـٰذَآۖ أَتَنْهَىٰنَآ أَن نَّعْبُدَ مَا يَعْبُدُ ءَابَآؤُنَا وَإِنَّنَا لَفِي شَكّٖ مِّمَّا تَدْعُونَآ إِلَيْهِ مُرِيبٖ ﴿62﴾
انہوں نے کہا: اے صالح! اس سے پہلے تو ہم تم سے امید رکھتے تھے، کیا تم ہمیں اس سے روکتے ہو کہ ہم ان کی عبادت کریں جن کی عبادت ہمارے باپ دادا کرتے تھے؟ اور بے شک ہم اس چیز کے بارے میں شک میں ہیں جس کی طرف تم ہمیں بلا رہے ہو، جو ہمیں بے چین کیے ہوئے ہے۔
قَالَ يَـٰقَوْمِ أَرَءَيْتُمْ إِن كُنتُ عَلَىٰ بَيِّنَةٖ مِّن رَّبِّي وَءَاتَىٰنِي مِنْهُ رَحْمَةٗ فَمَن يَنصُرُنِي مِنَ ٱللَّهِ إِنْ عَصَيْتُهُۥۖ فَمَا تَزِيدُونَنِي غَيْرَ تَخْسِيرٖ ﴿63﴾
اس نے کہا: اے میری قوم! کیا تم نے دیکھا اگر میں اپنے رب کی طرف سے ایک دلیل پر ہوں اور اس نے مجھے اپنی طرف سے رحمت عطا کی ہو، پھر اگر میں اس کی نافرمانی کروں تو کون مجھے اللہ سے بچا سکتا ہے؟ تو تم مجھے صرف نقصان ہی پہنچا رہے ہو۔
وَيَـٰقَوْمِ هَـٰذِهِۦ نَاقَةُ ٱللَّهِ لَكُمْ ءَايَةٗۖ فَذَرُوهَا تَأْكُلْ فِيٓ أَرْضِ ٱللَّهِ وَلَا تَمَسُّوهَا بِسُوٓءٖ فَيَأْخُذَكُمْ عَذَابٞ قَرِيبٞ ﴿64﴾
اور اے میری قوم! یہ اللہ کی اونٹنی تمہارے لیے نشانی ہے، پس اسے چھوڑ دو کہ اللہ کی زمین میں کھاتی رہے، اور اسے برائی سے ہاتھ نہ لگانا، ورنہ تمہیں قریب کا عذاب آ پکڑے گا۔
فَعَقَرُوهَا فَقَالَ تَمَتَّعُوا۟ فِي دَارِكُمْ ثَلَـٰثَةَ أَيَّامٖۖ ذَٰلِكَ وَعْدٌ غَيْرُ مَكْذُوبٖ ﴿65﴾
پھر انہوں نے اسے مار ڈالا، تو صالح نے کہا: اپنے گھروں میں تین دن تک فائدہ اٹھا لو، یہ وعدہ جھوٹا نہیں۔
فَلَمَّا جَآءَ أَمْرُنَا نَجَّيْنَا صَـٰلِحٗا وَٱلَّذِينَ ءَامَنُوا۟ مَعَهُۥ بِرَحْمَةٖ مِّنَّا وَمِنْ خِزْيِ يَوْمِئِذٍۚ إِنَّ رَبَّكَ هُوَ ٱلْقَوِيُّ ٱلْعَزِيزُ ﴿66﴾
پھر جب ہمارا حکم آ پہنچا تو ہم نے صالح اور ان لوگوں کو جو اس کے ساتھ ایمان لائے تھے، اپنی رحمت سے نجات دی، اور اس دن کی رسوائی سے بھی، بے شک تیرا رب ہی زبردست، غالب ہے۔
وَأَخَذَ ٱلَّذِينَ ظَلَمُوا۟ ٱلصَّيْحَةُ فَأَصْبَحُوا۟ فِي دِيَـٰرِهِمْ جَـٰثِمِينَ ﴿67﴾
اور ظالموں کو ایک چنگھاڑ نے آ پکڑا، تو وہ اپنے گھروں میں اوندھے پڑے رہ گئے۔
كَأَن لَّمْ يَغْنَوْا۟ فِيهَآۗ أَلَآ إِنَّ ثَمُودَ كَفَرُوا۟ رَبَّهُمْۗ أَلَا بُعْدٗا لِّثَمُودَ ﴿68﴾
گویا وہ کبھی وہاں بسے ہی نہ تھے، خبردار! ثمود نے اپنے رب کا انکار کیا، خبردار! ثمود کے لیے دوری ہے۔
وَلَقَدْ جَآءَتْ رُسُلُنَآ إِبْرَٰهِيمَ بِٱلْبُشْرَىٰ قَالُوا۟ سَلَـٰمٗاۖ قَالَ سَلَـٰمٞ فَمَا لَبِثَ أَن جَآءَ بِعِجْلٍ حَنِيذٖ ﴿69﴾
اور بے شک ہمارے فرشتے ابراہیم کے پاس خوشخبری لے کر آئے، انہوں نے کہا: سلام ہو، اس نے کہا: سلام ہو، پھر کچھ دیر نہ گزری کہ وہ بھنا ہوا بچھڑا لے آیا۔
فَلَمَّا رَءَآ أَيْدِيَهُمْ لَا تَصِلُ إِلَيْهِ نَكِرَهُمْ وَأَوْجَسَ مِنْهُمْ خِيفَةٗۖ قَالُوا۟ لَا تَخَفْ إِنَّآ أُرْسِلْنَآ إِلَىٰ قَوْمِ لُوطٖ ﴿70﴾
پھر جب اس نے دیکھا کہ ان کے ہاتھ کھانے کی طرف نہیں بڑھتے تو انہیں اجنبی سمجھا اور ان سے دل میں خوف محسوس کیا، انہوں نے کہا: مت ڈرو، ہم لوط کی قوم کی طرف بھیجے گئے ہیں۔
وَٱمْرَأَتُهُۥ قَآئِمَةٞ فَضَحِكَتْ فَبَشَّرْنَـٰهَا بِإِسْحَـٰقَ وَمِن وَرَآءِ إِسْحَـٰقَ يَعْقُوبَ ﴿71﴾
اور اس کی بیوی کھڑی تھی، تو وہ ہنس پڑی، تو ہم نے اسے اسحٰق کی خوشخبری دی، اور اسحٰق کے بعد یعقوب کی۔
قَالَتْ يَـٰوَيْلَتَىٰٓ أَأَلِدُ وَأَنَا۠ عَجُوزٞ وَهَـٰذَا بَعْلِي شَيْخًاۖ إِنَّ هَـٰذَا لَشَيْءٌ عَجِيبٌ ﴿72﴾
اس نے کہا: ہائے میری مصیبت! کیا میں بچہ جَنوں گی جبکہ میں بوڑھی ہوں، اور یہ میرا شوہر بھی بوڑھا ہے؟ بے شک یہ تو بڑی عجیب بات ہے۔
قَالُوٓا۟ أَتَعْجَبِينَ مِنْ أَمْرِ ٱللَّهِۖ رَحْمَتُ ٱللَّهِ وَبَرَكَـٰتُهُۥ عَلَيْكُمْ أَهْلَ ٱلْبَيْتِۚ إِنَّهُۥ حَمِيدٞ مَّجِيدٞ ﴿73﴾
انہوں نے کہا: کیا تم اللہ کے حکم پر تعجب کرتی ہو؟ اللہ کی رحمت اور برکتیں تم پر ہوں، اے اہلِ بیت! بے شک وہ تعریف کے لائق، بزرگی والا ہے۔
فَلَمَّا ذَهَبَ عَنْ إِبْرَٰهِيمَ ٱلرَّوْعُ وَجَآءَتْهُ ٱلْبُشْرَىٰ يُجَـٰدِلُنَا فِي قَوْمِ لُوطٍ ﴿74﴾
پھر جب ابراہیم سے خوف دور ہو گیا اور اسے خوشخبری آ گئی، تو وہ لوط کی قوم کے بارے میں ہم سے بحث کرنے لگا۔
إِنَّ إِبْرَٰهِيمَ لَحَلِيمٌ أَوَّاهٞ مُّنِيبٞ ﴿75﴾
بے شک ابراہیم بہت بردبار، آہ و زاری کرنے والا، اور رجوع کرنے والا تھا۔
يَـٰٓإِبْرَٰهِيمُ أَعْرِضْ عَنْ هَـٰذَآۖ إِنَّهُۥ قَدْ جَآءَ أَمْرُ رَبِّكَۖ وَإِنَّهُمْ ءَاتِيهِمْ عَذَابٌ غَيْرُ مَرْدُودٖ ﴿76﴾
اے ابراہیم! اس بات سے درگزر کرو، بے شک تمہارے رب کا حکم آ چکا ہے، اور ان پر وہ عذاب آ کر رہے گا جو ٹالا نہیں جا سکتا۔
وَلَمَّا جَآءَتْ رُسُلُنَا لُوطٗا سِيٓءَ بِهِمْ وَضَاقَ بِهِمْ ذَرْعٗا وَقَالَ هَـٰذَا يَوْمٌ عَصِيبٞ ﴿77﴾
اور جب ہمارے فرشتے لوط کے پاس آئے تو وہ ان کی وجہ سے پریشان ہوا اور ان سے دل تنگ ہوا، اور کہا: یہ تو بڑا سخت دن ہے۔
وَجَآءَهُۥ قَوْمُهُۥ يُهْرَعُونَ إِلَيْهِۖ وَمِن قَبْلُ كَانُوا۟ يَعْمَلُونَ ٱلسَّيِّـَٔاتِۚ قَالَ يَـٰقَوْمِ هَـٰٓؤُلَآءِ بَنَاتِي هُنَّ أَطْهَرُ لَكُمْۖ فَٱتَّقُوا۟ ٱللَّهَ وَلَا تُخْزُونِ فِي ضَيْفِيٓۖ أَلَيْسَ مِنكُم رَجُلٞ رَّشِيدٞ ﴿78﴾
اور اس کی قوم اس کے پاس دوڑتی ہوئی آئی، اور وہ پہلے سے ہی برے کام کیا کرتے تھے، اس نے کہا: اے میری قوم! یہ میری بیٹیاں ہیں، یہ تمہارے لیے پاکیزہ تر ہیں، پس اللہ سے ڈرو اور میرے مہمانوں کے بارے میں مجھے رسوا نہ کرو، کیا تم میں کوئی سمجھدار مرد نہیں؟
قَالُوا۟ لَقَدْ عَلِمْتَ مَا لَنَا فِي بَنَاتِكَ مِنْ حَقّٖۖ وَإِنَّكَ لَتَعْلَمُ مَا نُرِيدُ ﴿79﴾
انہوں نے کہا: تم خوب جانتے ہو کہ ہمیں تمہاری بیٹیوں میں کوئی حق نہیں، اور تم خوب جانتے ہو کہ ہم کیا چاہتے ہیں۔
قَالَ لَوْ أَنَّ لِي بِكُمْ قُوَّةً أَوْ ءَاوِيٓ إِلَىٰ رُكْنٖ شَدِيدٖ ﴿80﴾
اس نے کہا: کاش میرے پاس تمہارے مقابلے میں طاقت ہوتی، یا میں کسی مضبوط سہارے کی طرف پناہ لے سکتا۔
قَالُوا۟ يَـٰلُوطُ إِنَّا رُسُلُ رَبِّكَ لَن يَصِلُوٓا۟ إِلَيْكَۖ فَأَسْرِ بِأَهْلِكَ بِقِطْعٖ مِّنَ ٱلَّيْلِ وَلَا يَلْتَفِتْ مِنكُمْ أَحَدٌ إِلَّا ٱمْرَأَتَكَۖ إِنَّهُۥ مُصِيبُهَا مَآ أَصَابَهُمْۚ إِنَّ مَوْعِدَهُمُ ٱلصُّبْحُۚ أَلَيْسَ ٱلصُّبْحُ بِقَرِيبٖ ﴿81﴾
انہوں نے کہا: اے لوط! ہم تمہارے رب کے فرشتے ہیں، وہ تم تک ہرگز نہیں پہنچ سکیں گے، پس رات کے کسی حصے میں اپنے گھر والوں کو لے کر نکل جاؤ، اور تم میں سے کوئی پیچھے نہ دیکھے، سوائے تمہاری بیوی کے، بے شک اسے وہی مصیبت پہنچے گی جو انہیں پہنچے گی، بے شک ان کا وعدہ صبح ہے، کیا صبح قریب نہیں؟
فَلَمَّا جَآءَ أَمْرُنَا جَعَلْنَا عَـٰلِيَهَا سَافِلَهَا وَأَمْطَرْنَا عَلَيْهَا حِجَارَةٗ مِّن سِجِّيلٖ مَّنضُودٖ ﴿82﴾
پھر جب ہمارا حکم آ پہنچا تو ہم نے اس بستی کو الٹ دیا، اور اس پر تہ بہ تہ کنکریوں والے پتھر برسائے۔
مُّسَوَّمَةً عِندَ رَبِّكَۖ وَمَا هِيَ مِنَ ٱلظَّـٰلِمِينَ بِبَعِيدٖ ﴿83﴾
وہ تمہارے رب کے ہاں نشان زدہ ہیں، اور وہ ظالموں سے کچھ دور نہیں۔
وَإِلَىٰ مَدْيَنَ أَخَاهُمْ شُعَيْبٗاۚ قَالَ يَـٰقَوْمِ ٱعْبُدُوا۟ ٱللَّهَ مَا لَكُم مِّنْ إِلَـٰهٍ غَيْرُهُۖ وَلَا تَنقُصُوا۟ ٱلْمِكْيَالَ وَٱلْمِيزَانَۖ إِنِّيٓ أَرَىٰكُم بِخَيْرٖ وَإِنِّيٓ أَخَافُ عَلَيْكُمْ عَذَابَ يَوْمٖ مُّحِيطٖ ﴿84﴾
اور ہم نے مدین کی طرف ان کے بھائی شعیب کو بھیجا، اس نے کہا: اے میری قوم! اللہ کی عبادت کرو، اس کے سوا تمہارا کوئی معبود نہیں، اور ناپ تول میں کمی نہ کرو، بے شک میں تمہیں خوشحال دیکھتا ہوں، اور میں تم پر ایک گھیر لینے والے دن کے عذاب سے ڈرتا ہوں۔
وَيَـٰقَوْمِ أَوْفُوا۟ ٱلْمِكْيَالَ وَٱلْمِيزَانَ بِٱلْقِسْطِ وَلَا تَبْخَسُوا۟ ٱلنَّاسَ أَشْيَآءَهُمْ وَلَا تَعْثَوْا۟ فِي ٱلْأَرْضِ مُفْسِدِينَ ﴿85﴾
اور اے میری قوم! انصاف کے ساتھ ناپ تول پورا کرو، اور لوگوں کو ان کی چیزیں کم نہ دو، اور زمین میں فساد نہ پھیلاؤ۔
بَقِيَّتُ ٱللَّهِ خَيْرٞ لَّكُمْ إِن كُنتُم مُّؤْمِنِينَۚ وَمَآ أَنَا۠ عَلَيْكُم بِحَفِيظٖ ﴿86﴾
اللہ کی باقی رہنے والی چیز تمہارے لیے بہتر ہے اگر تم ایمان رکھتے ہو، اور میں تم پر نگہبان نہیں ہوں۔
قَالُوا۟ يَـٰشُعَيْبُ أَصَلَوٰتُكَ تَأْمُرُكَ أَن نَّتْرُكَ مَا يَعْبُدُ ءَابَآؤُنَآ أَوْ أَن نَّفْعَلَ فِيٓ أَمْوَٰلِنَا مَا نَشَـٰٓؤُا۟ۚ إِنَّكَ لَأَنتَ ٱلْحَلِيمُ ٱلرَّشِيدُ ﴿87﴾
انہوں نے کہا: اے شعیب! کیا تمہاری نماز تمہیں یہ حکم دیتی ہے کہ ہم ان چیزوں کو چھوڑ دیں جن کی عبادت ہمارے باپ دادا کرتے تھے، یا ہم اپنے مال میں جو چاہیں نہ کریں؟ بے شک تم ہی بڑے بردبار، سمجھدار ہو۔
قَالَ يَـٰقَوْمِ أَرَءَيْتُمْ إِن كُنتُ عَلَىٰ بَيِّنَةٖ مِّن رَّبِّي وَرَزَقَنِي مِنْهُ رِزْقًا حَسَنٗاۚ وَمَآ أُرِيدُ أَنْ أُخَالِفَكُمْ إِلَىٰ مَآ أَنْهَىٰكُمْ عَنْهُۚ إِنْ أُرِيدُ إِلَّا ٱلْإِصْلَـٰحَ مَا ٱسْتَطَعْتُۚ وَمَا تَوْفِيقِيٓ إِلَّا بِٱللَّهِۚ عَلَيْهِ تَوَكَّلْتُ وَإِلَيْهِ أُنِيبُ ﴿88﴾
اس نے کہا: اے میری قوم! کیا تم نے دیکھا اگر میں اپنے رب کی طرف سے ایک دلیل پر ہوں، اور اس نے مجھے اپنی طرف سے اچھا رزق دیا ہو؟ اور میں تمہیں جس چیز سے روکتا ہوں اس کے خلاف خود نہ چلوں، میں تو صرف اصلاح چاہتا ہوں جہاں تک مجھ سے ہو سکے، اور میری توفیق صرف اللہ ہی سے ہے، اسی پر میں نے بھروسہ کیا، اور اسی کی طرف رجوع کرتا ہوں۔
وَيَـٰقَوْمِ لَا يَجْرِمَنَّكُمْ شِقَاقِيٓ أَن يُصِيبَكُم مِّثْلُ مَآ أَصَابَ قَوْمَ نُوحٍ أَوْ قَوْمَ هُودٍ أَوْ قَوْمَ صَـٰلِحٖۚ وَمَا قَوْمُ لُوطٖ مِّنكُم بِبَعِيدٖ ﴿89﴾
اور اے میری قوم! میری مخالفت تمہیں اس بات پر آمادہ نہ کرے کہ تمہیں وہی مصیبت پہنچے جو قومِ نوح، قومِ ہود یا قومِ صالح کو پہنچی، اور قومِ لوط تم سے کچھ دور نہیں۔
وَٱسْتَغْفِرُوا۟ رَبَّكُمْ ثُمَّ تُوبُوٓا۟ إِلَيْهِۚ إِنَّ رَبِّي رَحِيمٞ وَدُودٞ ﴿90﴾
اور اپنے رب سے معافی مانگو، پھر اس کی طرف رجوع کرو، بے شک میرا رب رحم کرنے والا، محبت کرنے والا ہے۔
قَالُوا۟ يَـٰشُعَيْبُ مَا نَفْقَهُ كَثِيرٗا مِّمَّا تَقُولُ وَإِنَّا لَنَرَىٰكَ فِينَا ضَعِيفٗاۖ وَلَوْلَا رَهْطُكَ لَرَجَمْنَـٰكَۖ وَمَآ أَنتَ عَلَيْنَا بِعَزِيزٖ ﴿91﴾
انہوں نے کہا: اے شعیب! ہم تمہاری بہت سی باتیں سمجھ نہیں پاتے، اور ہم تمہیں اپنے درمیان کمزور دیکھتے ہیں، اور اگر تمہارا خاندان نہ ہوتا تو ہم تمہیں سنگسار کر دیتے، اور تم ہمارے لیے کوئی باعزت نہیں ہو۔
قَالَ يَـٰقَوْمِ أَرَهْطِيٓ أَعَزُّ عَلَيْكُم مِّنَ ٱللَّهِ وَٱتَّخَذْتُمُوهُ وَرَآءَكُمْ ظِهْرِيًّاۚ إِنَّ رَبِّي بِمَا تَعْمَلُونَ مُحِيطٞ ﴿92﴾
اس نے کہا: اے میری قوم! کیا میرا خاندان تمہیں اللہ سے زیادہ عزیز ہے؟ اور تم نے اسے پسِ پشت ڈال دیا؟ بے شک میرا رب تمہارے اعمال کو گھیرے ہوئے ہے۔
وَيَـٰقَوْمِ ٱعْمَلُوا۟ عَلَىٰ مَكَانَتِكُمْ إِنِّي عَامِلٞۖ سَوْفَ تَعْلَمُونَ مَن يَأْتِيهِ عَذَابٞ يُخْزِيهِ وَمَنْ هُوَ كَاذِبٞۖ وَٱرْتَقِبُوٓا۟ إِنِّي مَعَكُمْ رَقِيبٞ ﴿93﴾
اور اے میری قوم! تم اپنی جگہ عمل کرو، بے شک میں بھی عمل کر رہا ہوں، عنقریب تم جان لو گے کہ کس پر وہ عذاب آتا ہے جو اسے رسوا کرتا ہے، اور کون جھوٹا ہے، پس انتظار کرو، بے شک میں بھی تمہارے ساتھ انتظار کرنے والا ہوں۔
وَلَمَّا جَآءَ أَمْرُنَا نَجَّيْنَا شُعَيْبٗا وَٱلَّذِينَ ءَامَنُوا۟ مَعَهُۥ بِرَحْمَةٖ مِّنَّا وَأَخَذَتِ ٱلَّذِينَ ظَلَمُوا۟ ٱلصَّيْحَةُ فَأَصْبَحُوا۟ فِي دِيَـٰرِهِمْ جَـٰثِمِينَ ﴿94﴾
اور جب ہمارا حکم آ پہنچا تو ہم نے شعیب اور ان لوگوں کو جو اس کے ساتھ ایمان لائے تھے، اپنی رحمت سے نجات دی، اور ظالموں کو ایک چنگھاڑ نے آ پکڑا، تو وہ اپنے گھروں میں اوندھے پڑے رہ گئے۔
كَأَن لَّمْ يَغْنَوْا۟ فِيهَآۗ أَلَآ بُعْدٗا لِّمَدْيَنَ كَمَا بَعِدَتْ ثَمُودُ ﴿95﴾
گویا وہ کبھی وہاں بسے ہی نہ تھے، خبردار! مدین کے لیے بھی دوری ہے، جیسے ثمود دور ہو گئے۔
وَلَقَدْ أَرْسَلْنَا مُوسَىٰ بِـَٔايَـٰتِنَا وَسُلْطَـٰنٖ مُّبِينٍ ﴿96﴾
اور بے شک ہم نے موسیٰ کو اپنی آیات اور واضح دلیل کے ساتھ بھیجا۔
إِلَىٰ فِرْعَوْنَ وَمَلَإِيهِۦ فَٱتَّبَعُوا۟ أَمْرَ فِرْعَوْنَۖ وَمَآ أَمْرُ فِرْعَوْنَ بِرَشِيدٍ ﴿97﴾
فرعون اور اس کے سرداروں کی طرف، تو انہوں نے فرعون کے حکم کی پیروی کی، حالانکہ فرعون کا حکم درست نہ تھا۔
يَقْدُمُ قَوْمَهُۥ يَوْمَ ٱلْقِيَـٰمَةِ فَأَوْرَدَهُمُ ٱلنَّارَۖ وَبِئْسَ ٱلْوِرْدُ ٱلْمَوْرُودُ ﴿98﴾
وہ قیامت کے دن اپنی قوم کے آگے آگے ہو گا، پھر انہیں آگ میں داخل کرے گا، اور وہ داخل ہونے کی جگہ بہت بری ہے۔
وَأُتْبِعُوا۟ فِي هَـٰذِهِۦ لَعْنَةٗ وَيَوْمَ ٱلْقِيَـٰمَةِۚ بِئْسَ ٱلرِّفْدُ ٱلْمَرْفُودُ ﴿99﴾
اور ان پر دنیا میں لعنت مسلط کی گئی، اور قیامت کے دن بھی، اور وہ مدد بہت بری ہے جو انہیں دی گئی۔
ذَٰلِكَ مِنْ أَنۢبَآءِ ٱلْقُرَىٰ نَقُصُّهُۥ عَلَيْكَۖ مِنْهَا قَآئِمٞ وَحَصِيدٞ ﴿100﴾
یہ ان بستیوں کی خبریں ہیں جو ہم تمہیں سناتے ہیں، ان میں کچھ باقی ہیں اور کچھ کٹ چکی ہیں۔
وَمَا ظَلَمْنَـٰهُمْ وَلَـٰكِن ظَلَمُوٓا۟ أَنفُسَهُمْ فَمَآ أَغْنَتْ عَنْهُمْ ءَالِهَتُهُمُ ٱلَّتِي يَدْعُونَ مِن دُونِ ٱللَّهِ مِن شَيْءٍ لَّمَّا جَآءَ أَمْرُ رَبِّكَۖ وَمَا زَادُوهُمْ غَيْرَ تَتْبِيبٖ ﴿101﴾
اور ہم نے ان پر ظلم نہیں کیا، بلکہ انہوں نے خود اپنے اوپر ظلم کیا، پس جب تمہارے رب کا حکم آ گیا تو ان کے وہ معبود جنہیں وہ اللہ کے سوا پکارتے تھے، ان کے کچھ کام نہ آئے، اور انہوں نے ان کی ہلاکت ہی میں اضافہ کیا۔
وَكَذَٰلِكَ أَخْذُ رَبِّكَ إِذَآ أَخَذَ ٱلْقُرَىٰ وَهِيَ ظَـٰلِمَةٌۚ إِنَّ أَخْذَهُۥٓ أَلِيمٞ شَدِيدٌ ﴿102﴾
اور اسی طرح تمہارے رب کی پکڑ ہوتی ہے جب وہ بستیوں کو پکڑتا ہے جبکہ وہ ظالم ہوتی ہیں، بے شک اس کی پکڑ دردناک، سخت ہے۔
إِنَّ فِي ذَٰلِكَ لَـَٔايَةٗ لِّمَنْ خَافَ عَذَابَ ٱلْـَٔاخِرَةِۚ ذَٰلِكَ يَوْمٞ مَّجْمُوعٞ لَّهُ ٱلنَّاسُ وَذَٰلِكَ يَوْمٞ مَّشْهُودٞ ﴿103﴾
بے شک اس میں عبرت ہے اس کے لیے جو آخرت کے عذاب سے ڈرتا ہے، وہ دن ہے جس میں سب لوگ جمع کیے جائیں گے، اور وہ دن حاضر ہونے والا ہے۔
وَمَا نُؤَخِّرُهُۥٓ إِلَّا لِأَجَلٖ مَّعْدُودٖ ﴿104﴾
اور ہم اسے مؤخر نہیں کرتے مگر ایک گنی چنی مدت کے لیے۔
يَوْمَ يَأْتِ لَا تَكَلَّمُ نَفْسٌ إِلَّا بِإِذْنِهِۦۚ فَمِنْهُمۡ شَقِيّٞ وَسَعِيدٞ ﴿105﴾
جس دن وہ آ جائے گا، کوئی شخص اس کی اجازت کے بغیر بات نہیں کرے گا، پھر ان میں کچھ بدبخت ہوں گے اور کچھ نیک بخت۔
فَأَمَّا ٱلَّذِينَ شَقُوا۟ فَفِي ٱلنَّارِ لَهُمْ فِيهَا زَفِيرٞ وَشَهِيقٌ ﴿106﴾
تو جو بدبخت ہوں گے وہ آگ میں ہوں گے، ان کے لیے اس میں چیخ و پکار ہو گی۔
خَـٰلِدِينَ فِيهَا مَا دَامَتِ ٱلسَّمَـٰوَٰتُ وَٱلْأَرْضُ إِلَّا مَا شَآءَ رَبُّكَۚ إِنَّ رَبَّكَ فَعَّالٞ لِّمَا يُرِيدُ ﴿107﴾
وہ اس میں ہمیشہ رہیں گے، جب تک آسمان و زمین قائم ہیں، مگر جو تمہارا رب چاہے، بے شک تمہارا رب جو چاہے کرتا ہے۔
وَأَمَّا ٱلَّذِينَ سُعِدُوا۟ فَفِي ٱلْجَنَّةِ خَـٰلِدِينَ فِيهَا مَا دَامَتِ ٱلسَّمَـٰوَٰتُ وَٱلْأَرْضُ إِلَّا مَا شَآءَ رَبُّكَۖ عَطَآءً غَيْرَ مَجْذُوذٖ ﴿108﴾
اور جو نیک بخت ہوں گے وہ جنت میں ہوں گے، اس میں ہمیشہ رہیں گے، جب تک آسمان و زمین قائم ہیں، مگر جو تمہارا رب چاہے، یہ عطا ہے جو کبھی ختم نہ ہو گی۔
فَلَا تَكُ فِي مِرْيَةٖ مِّمَّا يَعْبُدُ هَـٰٓؤُلَآءِۚ مَا يَعْبُدُونَ إِلَّا كَمَا يَعْبُدُ ءَابَآؤُهُم مِّن قَبْلُۚ وَإِنَّا لَمُوَفُّوهُمْ نَصِيبَهُمۡ غَيْرَ مَنقُوصٖ ﴿109﴾
پس تم اس چیز میں شک نہ کرو جس کی یہ لوگ عبادت کرتے ہیں، یہ صرف اسی کی عبادت کرتے ہیں جس کی ان کے باپ دادا پہلے کرتے تھے، اور ہم انہیں ان کا حصہ پورا پورا دیں گے، جس میں کوئی کمی نہ ہو گی۔
وَلَقَدْ ءَاتَيْنَا مُوسَى ٱلْكِتَـٰبَ فَٱخْتُلِفَ فِيهِۚ وَلَوْلَا كَلِمَةٞ سَبَقَتْ مِن رَّبِّكَ لَقُضِيَ بَيْنَهُمْۚ وَإِنَّهُمْ لَفِي شَكّٖ مِّنْهُ مُرِيبٖ ﴿110﴾
اور بے شک ہم نے موسیٰ کو کتاب دی، پھر اس میں اختلاف کیا گیا، اور اگر تمہارے رب کی طرف سے ایک بات پہلے نہ ہو چکی ہوتی تو ان کے درمیان فیصلہ کر دیا جاتا، اور بے شک وہ اس کے بارے میں شک میں پڑے ہوئے ہیں جو بے چین کرنے والا ہے۔
وَإِنَّ كُلّٗا لَّمَّا لَيُوَفِّيَنَّهُمْ رَبُّكَ أَعْمَـٰلَهُمۡۚ إِنَّهُۥ بِمَا يَعْمَلُونَ خَبِيرٞ ﴿111﴾
اور بے شک تمہارا رب ہر ایک کو ان کے اعمال کا پورا بدلہ دے گا، بے شک وہ ان کے اعمال سے خوب باخبر ہے۔
فَٱسْتَقِمْ كَمَآ أُمِرْتَ وَمَن تَابَ مَعَكَ وَلَا تَطْغَوْا۟ۚ إِنَّهُۥ بِمَا تَعْمَلُونَ بَصِيرٞ ﴿112﴾
پس تم قائم رہو جیسے تمہیں حکم دیا گیا ہے، اور وہ بھی جو تمہارے ساتھ توبہ کریں، اور حد سے نہ بڑھو، بے شک وہ تمہارے اعمال کو دیکھنے والا ہے۔
وَلَا تَرْكَنُوٓا۟ إِلَى ٱلَّذِينَ ظَلَمُوا۟ فَتَمَسَّكُمُ ٱلنَّارُۖ وَمَا لَكُم مِّن دُونِ ٱللَّهِ مِنْ أَوْلِيَآءَ ثُمَّ لَا تُنصَرُونَ ﴿113﴾
اور ظالموں کی طرف نہ جھکو، ورنہ تمہیں آگ چھو لے گی، اور تمہارے لیے اللہ کے سوا کوئی مددگار نہ ہو گا، پھر تم مدد نہ دیے جاؤ گے۔
وَأَقِمِ ٱلصَّلَوٰةَ طَرَفَيِ ٱلنَّهَارِ وَزُلَفٗا مِّنَ ٱلَّيْلِۚ إِنَّ ٱلْحَسَنَـٰتِ يُذْهِبْنَ ٱلسَّيِّـَٔاتِۚ ذَٰلِكَ ذِكْرَىٰ لِلذَّٰكِرِينَ ﴿114﴾
اور نماز قائم کرو دن کے دونوں کناروں پر اور رات کے کچھ حصے میں، بے شک نیکیاں برائیوں کو مٹا دیتی ہیں، یہ نصیحت ہے یاد رکھنے والوں کے لیے۔
وَٱصْبِرْ فَإِنَّ ٱللَّهَ لَا يُضِيعُ أَجْرَ ٱلْمُحْسِنِينَ ﴿115﴾
اور صبر کرو، بے شک اللہ نیکی کرنے والوں کا اجر ضائع نہیں کرتا۔
فَلَوْلَا كَانَ مِنَ ٱلْقُرُونِ مِن قَبْلِكُمْ أُوْلُوا۟ بَقِيَّةٖ يَنْهَوْنَ عَنِ ٱلْفَسَادِ فِي ٱلْأَرْضِ إِلَّا قَلِيلٗا مِّمَّنْ أَنجَيْنَا مِنْهُمْۗ وَاتَّبَعَ ٱلَّذِينَ ظَلَمُوا۟ مَآ أُتْرِفُوا۟ فِيهِ وَكَانُوا۟ مُجْرِمِينَ ﴿116﴾
تو کیوں نہ تم سے پہلے کی نسلوں میں کچھ ایسے لوگ ہوئے جو زمین میں فساد سے روکتے؟ مگر تھوڑے سے جنہیں ہم نے نجات دی، اور ظالم لوگ تو اسی عیش و عشرت کے پیچھے لگے رہے جس میں وہ ڈالے گئے تھے، اور وہ مجرم تھے۔
وَمَا كَانَ رَبُّكَ لِيُهْلِكَ ٱلْقُرَىٰ بِظُلْمٖ وَأَهْلُهَا مُصْلِحُونَ ﴿117﴾
اور تمہارا رب ایسا نہیں کہ بستیوں کو ظلم سے ہلاک کرے جبکہ ان کے لوگ اصلاح کرنے والے ہوں۔
وَلَوْ شَآءَ رَبُّكَ لَجَعَلَ ٱلنَّاسَ أُمَّةٗ وَٰحِدَةٗۖ وَلَا يَزَالُونَ مُخْتَلِفِينَ ﴿118﴾
اور اگر تمہارا رب چاہتا تو سب لوگوں کو ایک امت بنا دیتا، مگر وہ ہمیشہ اختلاف کرتے رہیں گے۔
إِلَّا مَن رَّحِمَ رَبُّكَۚ وَلِذَٰلِكَ خَلَقَهُمْۗ وَتَمَّتْ كَلِمَةُ رَبِّكَ لَأَمْلَأَنَّ جَهَنَّمَ مِنَ ٱلْجِنَّةِ وَٱلنَّاسِ أَجْمَعِينَ ﴿119﴾
سوائے ان کے جن پر تمہارا رب رحم کرے، اور اسی لیے اس نے انہیں پیدا کیا، اور تمہارے رب کا فرمان پورا ہو چکا کہ میں جہنم کو جنوں اور انسانوں سب سے بھر دوں گا۔
وَكُلّٗا نَّقُصُّ عَلَيْكَ مِنْ أَنۢبَآءِ ٱلرُّسُلِ مَا نُثَبِّتُ بِهِۦ فُؤَادَكَۚ وَجَآءَكَ فِي هَـٰذِهِ ٱلْحَقُّ وَمَوْعِظَةٞ وَذِكْرَىٰ لِلْمُؤْمِنِينَ ﴿120﴾
اور ہم رسولوں کی خبریں تمہیں اس لیے سناتے ہیں تاکہ تمہارے دل کو مضبوط کریں، اور اس میں تمہارے پاس حق، نصیحت اور ایمان والوں کے لیے یاد دہانی آ چکی ہے۔
وَقُل لِّلَّذِينَ لَا يُؤْمِنُونَ ٱعْمَلُوا۟ عَلَىٰ مَكَانَتِكُمْ إِنَّا عَـٰمِلُونَ ﴿121﴾
اور ان لوگوں سے کہہ دو جو ایمان نہیں لاتے: تم اپنی جگہ عمل کرو، ہم بھی عمل کرنے والے ہیں۔
وَٱنتَظِرُوٓا۟ إِنَّا مُنتَظِرُونَ ﴿122﴾
اور انتظار کرو، ہم بھی انتظار کرنے والے ہیں۔
وَلِلَّهِ غَيْبُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَإِلَيْهِ يُرْجَعُ ٱلْأَمْرُ كُلُّهُۥ فَٱعْبُدْهُ وَتَوَكَّلْ عَلَيْهِۚ وَمَا رَبُّكَ بِغَـٰفِلٍ عَمَّا تَعْمَلُونَ ﴿123﴾
اور آسمانوں اور زمین کا غیب اللہ ہی کے لیے ہے، اور اسی کی طرف تمام معاملات لوٹائے جاتے ہیں، پس اسی کی عبادت کرو اور اسی پر بھروسہ کرو، اور تمہارا رب اس سے غافل نہیں جو تم کرتے ہو۔
سورة يوسف
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
الٓرۚ تِلْكَ ءَايَاتُ ٱلْكِتَـٰبِ ٱلْمُبِينِ ﴿1﴾
الٓر۔ یہ روشن کتاب کی آیات ہیں۔
إِنَّآ أَنزَلْنَـٰهُ قُرْءَانًا عَرَبِيّٗا لَّعَلَّكُمْ تَعْقِلُونَ ﴿2﴾
بے شک ہم نے اسے عربی قرآن بنا کر نازل کیا تاکہ تم سمجھو۔
نَحْنُ نَقُصُّ عَلَيْكَ أَحْسَنَ ٱلْقَصَصِ بِمَآ أَوْحَيْنَآ إِلَيْكَ هَـٰذَا ٱلْقُرْءَانَ وَإِن كُنتَ مِن قَبْلِهِۦ لَمِنَ ٱلْغَـٰفِلِينَ ﴿3﴾
ہم تمہیں بہترین قصہ سناتے ہیں اس قرآن کے ذریعے جو ہم نے تمہاری طرف وحی کیا، حالانکہ تم اس سے پہلے بے خبر لوگوں میں سے تھے۔
إِذْ قَالَ يُوسُفُ لِأَبِيهِ يَـٰٓأَبَتِ إِنِّي رَأَيْتُ أَحَدَ عَشَرَ كَوْكَبٗا وَٱلشَّمْسَ وَٱلْقَمَرَ رَأَيْتُهُمْ لِي سَـٰجِدِينَ ﴿4﴾
جب یوسف نے اپنے باپ سے کہا: اے میرے ابا! میں نے خواب میں گیارہ ستارے، سورج اور چاند دیکھا، میں نے انہیں مجھے سجدہ کرتے دیکھا۔
قَالَ يَـٰبُنَيَّ لَا تَقْصُصْ رُءْيَاكَ عَلَىٰٓ إِخْوَتِكَ فَيَكِيدُوا۟ لَكَ كَيْدًاۚ إِنَّ ٱلشَّيْطَـٰنَ لِلْإِنسَـٰنِ عَدُوّٞ مُّبِينٞ ﴿5﴾
اس نے کہا: اے میرے بیٹے! اپنا خواب اپنے بھائیوں کو نہ سنانا، ورنہ وہ تیرے خلاف کوئی چال چلیں گے، بے شک شیطان انسان کا کھلا دشمن ہے۔
وَكَذَٰلِكَ يَجْتَبِيكَ رَبُّكَ وَيُعَلِّمُكَ مِن تَأْوِيلِ ٱلْأَحَادِيثِ وَيُتِمُّ نِعْمَتَهُۥ عَلَيْكَ وَعَلَىٰٓ ءَالِ يَعْقُوبَ كَمَآ أَتَمَّهَا عَلَىٰٓ أَبَوَيْكَ إِبْرَٰهِيمَ وَإِسْحَـٰقَۚ إِنَّ رَبَّكَ عَلِيمٌ حَكِيمٞ ﴿6﴾
اور اسی طرح تیرا رب تجھے چن لے گا، اور تجھے باتوں کی تعبیر سکھائے گا، اور اپنی نعمت تجھ پر اور یعقوب کے گھر والوں پر پوری کرے گا، جیسے اس نے تیرے باپ دادا ابراہیم اور اسحاق پر پوری کی، بے شک تیرا رب علم والا، حکمت والا ہے۔
لَّقَدْ كَانَ فِي يُوسُفَ وَإِخْوَتِهِۦٓ ءَايَـٰتٞ لِّلسَّآئِلِينَ ﴿7﴾
یقیناً یوسف اور اس کے بھائیوں کے قصے میں پوچھنے والوں کے لیے نشانیاں ہیں۔
إِذْ قَالُوا۟ لَيُوسُفُ وَأَخُوهُ أَحَبُّ إِلَىٰٓ أَبِينَا مِنَّا وَنَحْنُ عُصْبَةٌ إِنَّ أَبَانَا لَفِي ضَلَـٰلٖ مُّبِينٍ ﴿8﴾
جب انہوں نے کہا: یوسف اور اس کا بھائی ہمارے باپ کو ہم سے زیادہ محبوب ہیں، حالانکہ ہم طاقتور جماعت ہیں، بے شک ہمارا باپ کھلی گمراہی میں ہے۔
ٱقْتُلُوا۟ يُوسُفَ أَوِ ٱطْرَحُوهُ أَرْضٗا يَخْلُ لَكُمْ وَجْهُ أَبِيكُمْ وَتَكُونُوا۟ مِنۢ بَعْدِهِۦ قَوْمٗا صَـٰلِحِينَ ﴿9﴾
یوسف کو قتل کر دو یا اسے کسی جگہ پھینک دو تاکہ تمہارے باپ کی توجہ صرف تم پر ہو جائے، اور اس کے بعد تم نیک لوگ بن جانا۔
قَالَ قَآئِلٞ مِّنْهُمْ لَا تَقْتُلُوا۟ يُوسُفَ وَأَلْقُوهُ فِي غَيَـٰبَتِ ٱلْجُبِّ يَلْتَقِطْهُۥ بَعْضُ ٱلسَّيَّارَةِ إِن كُنتُمْ فَـٰعِلِينَ ﴿10﴾
ان میں سے ایک کہنے والے نے کہا: یوسف کو قتل نہ کرو، بلکہ اسے کسی اندھے کنویں میں ڈال دو، کوئی قافلہ اسے نکال لے جائے گا، اگر تم کچھ کرنا ہی چاہتے ہو۔
قَالُوا۟ يَـٰٓأَبَانَا مَا لَكَ لَا تَأْمَنَّا عَلَىٰ يُوسُفَ وَإِنَّا لَهُۥ لَنَـٰصِحُونَ ﴿11﴾
انہوں نے کہا: اے ہمارے باپ! آپ یوسف کے بارے میں ہم پر اعتماد کیوں نہیں کرتے، حالانکہ ہم اس کے خیرخواہ ہیں۔
أَرْسِلْهُ مَعَنَا غَدٗا يَرْتَعْ وَيَلْعَبْ وَإِنَّا لَهُۥ لَحَـٰفِظُونَ ﴿12﴾
کل اسے ہمارے ساتھ بھیج دیں، وہ خوب کھائے پیے اور کھیلے، اور ہم اس کے محافظ ہیں۔
قَالَ إِنِّي لَيَحْزُنُنِيٓ أَن تَذْهَبُوا۟ بِهِۦ وَأَخَافُ أَن يَأْكُلَهُ ٱلذِّئْبُ وَأَنتُمْ عَنْهُ غَـٰفِلُونَ ﴿13﴾
اس نے کہا: مجھے یہ غم ہوگا کہ تم اسے لے جاؤ، اور مجھے ڈر ہے کہ کہیں اسے بھیڑیا نہ کھا جائے جبکہ تم اس سے غافل ہو۔
قَالُوا۟ لَئِنْ أَكَلَهُ ٱلذِّئْبُ وَنَحْنُ عُصْبَةٌ إِنَّآ إِذٗا لَّخَـٰسِرُونَ ﴿14﴾
انہوں نے کہا: اگر اسے بھیڑیا کھا جائے جبکہ ہم ایک مضبوط جماعت ہیں، تو ہم ضرور نقصان اٹھانے والے ہوں گے۔
فَلَمَّا ذَهَبُوا۟ بِهِۦ وَأَجْمَعُوٓا۟ أَن يَجْعَلُوهُ فِي غَيَـٰبَتِ ٱلْجُبِّۚ وَأَوْحَيْنَآ إِلَيْهِ لَتُنَبِّئَنَّهُم بِأَمْرِهِمْ هَـٰذَا وَهُمْ لَا يَشْعُرُونَ ﴿15﴾
پھر جب وہ اسے لے گئے اور اس بات پر متفق ہو گئے کہ اسے اندھے کنویں میں ڈال دیں، تو ہم نے یوسف کو وحی کی: تم انہیں ان کے اس کام کی خبر دو گے جبکہ وہ (تمہیں) پہچان نہ سکیں گے۔
وَجَآءُوٓ أَبَاهُمْ عِشَآءٗ يَبْكُونَ ﴿16﴾
اور وہ رات کو روتے ہوئے اپنے باپ کے پاس آئے۔
قَالُوا۟ يَـٰٓأَبَانَآ إِنَّا ذَهَبْنَا نَسْتَبِقُ وَتَرَكْنَا يُوسُفَ عِندَ مَتَـٰعِنَا فَأَكَلَهُ ٱلذِّئْبُۖ وَمَآ أَنتَ بِمُؤْمِنٖ لَّنَا وَلَوْ كُنَّا صَـٰدِقِينَ ﴿17﴾
انہوں نے کہا: اے ہمارے باپ! ہم دوڑ میں لگے ہوئے تھے اور یوسف کو اپنے سامان کے پاس چھوڑ گئے، تو اسے بھیڑیا کھا گیا، اور آپ ہماری بات کا یقین نہیں کریں گے چاہے ہم سچے ہی ہوں۔
وَجَآءُوا۟ عَلَىٰ قَمِيصِهِۦ بِدَمٖ كَذِبٖۚ قَالَ بَلْ سَوَّلَتْ لَكُمْ أَنفُسُكُمْ أَمْرٗاۖ فَصَبْرٞ جَمِيلٞۖ وَٱللَّهُ ٱلْمُسْتَعَانُ عَلَىٰ مَا تَصِفُونَ ﴿18﴾
اور وہ یوسف کی قمیص پر جھوٹا خون لگا کر لائے، یعقوب نے کہا: بلکہ تمہارے نفس نے تمہیں کوئی بات سجھائی ہے، پس صبر ہی بہتر ہے، اور اللہ ہی ہے جس سے مدد لی جاتی ہے اس بات پر جو تم بیان کرتے ہو۔
وَجَآءَتْ سَيَّارَةٞ فَأَرْسَلُوا۟ وَارِدَهُمْ فَأَدْلَىٰ دَلْوَهُۥۖ قَالَ يَـٰبُشْرَىٰ هَـٰذَا غُلَـٰمٞۚ وَأَسَرُّوهُ بِضَـٰعَةٗۚ وَٱللَّهُ عَلِيمُۢ بِمَا يَعْمَلُونَ ﴿19﴾
اور ایک قافلہ آیا، تو انہوں نے اپنا پانی لانے والا بھیجا، اس نے اپنا ڈول لٹکایا، تو کہا: خوشخبری ہو! یہ تو ایک لڑکا ہے، اور انہوں نے اسے مالِ تجارت سمجھ کر چھپا لیا، اور اللہ خوب جاننے والا ہے جو وہ کرتے ہیں۔
وَشَرَوْهُ بِثَمَنِۭ بَخْسٖ دَرَٰهِمَ مَعْدُودَةٖ وَكَانُوا۟ فِيهِ مِنَ ٱلزَّـٰهِدِينَ ﴿20﴾
اور انہوں نے اسے تھوڑی قیمت پر چند درہموں کے عوض بیچ دیا، اور وہ اس کے بارے میں بے رغبت تھے۔
وَقَالَ ٱلَّذِي ٱشْتَرَىٰهُ مِن مِّصْرَ لِٱمْرَأَتِهِۦٓ أَكْرِمِي مَثْوَىٰهُ عَسَىٰٓ أَن يَنفَعَنَآ أَوْ نَتَّخِذَهُۥ وَلَدٗاۚ وَكَذَٰلِكَ مَكَّنَّا لِيُوسُفَ فِي ٱلْأَرْضِ وَلِنُعَلِّمَهُۥ مِن تَأْوِيلِ ٱلْأَحَادِيثِۚ وَٱللَّهُ غَالِبٌ عَلَىٰٓ أَمْرِهِۦ وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَعْلَمُونَ ﴿21﴾
اور مصر کے جس شخص نے اسے خریدا، اس نے اپنی بیوی سے کہا: اسے عزت سے رکھو، شاید یہ ہمیں فائدہ دے یا ہم اسے بیٹا بنا لیں۔ اور اسی طرح ہم نے یوسف کو زمین میں جگہ دی، اور اسے خوابوں کی تعبیر سکھائی، اور اللہ اپنے کام پر غالب ہے، لیکن اکثر لوگ نہیں جانتے۔
وَلَمَّا بَلَغَ أَشُدَّهُۥٓ ءَاتَيْنَـٰهُ حُكْمٗا وَعِلْمٗاۚ وَكَذَٰلِكَ نَجْزِي ٱلْمُحْسِنِينَ ﴿22﴾
اور جب وہ اپنی جوانی کو پہنچا، ہم نے اسے حکمت اور علم عطا کیا، اور اسی طرح ہم نیکوکاروں کو جزا دیتے ہیں۔
وَرَٰوَدَتْهُ ٱلَّتِي هُوَ فِي بَيْتِهَا عَن نَّفْسِهِۦ وَغَلَّقَتِ ٱلْأَبْوَٰبَ وَقَالَتْ هَيْتَ لَكَۚ قَالَ مَعَاذَ ٱللَّهِۖ إِنَّهُۥ رَبِّيٓ أَحْسَنَ مَثْوَايَۖ إِنَّهُۥ لَا يُفْلِحُ ٱلظَّـٰلِمُونَ ﴿23﴾
اور جس عورت کے گھر میں وہ تھا، اس نے اسے اپنی طرف مائل کرنا چاہا، اور دروازے بند کر دیے، اور کہا: آؤ! اس نے کہا: اللہ کی پناہ! بے شک وہ میرا آقا ہے، اس نے مجھے اچھی جگہ دی ہے، بے شک ظالم کامیاب نہیں ہوتے۔
وَلَقَدْ هَمَّتْ بِهِۦۖ وَهَمَّ بِهَا لَوْلَآ أَن رَّءَا بُرْهَـٰنَ رَبِّهِۦۚ كَذَٰلِكَ لِنَصْرِفَ عَنْهُ ٱلسُّوٓءَ وَٱلْفَحْشَآءَۚ إِنَّهُۥ مِنْ عِبَادِنَا ٱلْمُخْلَصِينَ ﴿24﴾
اور بے شک وہ اس کی طرف مائل ہوئی، اور وہ بھی اس کی طرف مائل ہوتا اگر اپنے رب کی دلیل نہ دیکھتا، اسی طرح ہم نے اس سے برائی اور بے حیائی کو دور کیا، بے شک وہ ہمارے چنے ہوئے بندوں میں سے تھا۔
وَٱسْتَبَقَا ٱلْبَابَ وَقَدَّتْ قَمِيصَهُۥ مِن دُبُرٖۖ وَأَلْفَيَا سَيِّدَهَا لَدَا ٱلْبَابِۚ قَالَتْ مَا جَزَآءُ مَنْ أَرَادَ بِأَهْلِكَ سُوٓءًا إِلَّآ أَن يُسْجَنَ أَوْ عَذَابٌ أَلِيمٞ ﴿25﴾
اور دونوں دروازے کی طرف دوڑے، اور اس نے یوسف کا قمیص پیچھے سے پھاڑ دیا، اور دونوں نے اس کے آقا کو دروازے پر پایا، اس نے کہا: جو شخص تمہاری بیوی کے ساتھ برائی کا ارادہ کرے، اس کی سزا قید یا دردناک عذاب ہے۔
قَالَ هِيَ رَٰوَدَتْنِي عَن نَّفْسِيۚ وَشَهِدَ شَاهِدٞ مِّنْ أَهْلِهَآ إِن كَانَ قَمِيصُهُۥ قُدَّ مِن قُبُلٖ فَصَدَقَتْ وَهُوَ مِنَ ٱلْكَـٰذِبِينَ ﴿26﴾
یوسف نے کہا: اس نے مجھے اپنی طرف مائل کرنا چاہا، اور اس کے خاندان میں سے ایک گواہ نے گواہی دی: اگر اس کا قمیص آگے سے پھٹا ہے تو وہ سچی ہے اور یوسف جھوٹا ہے۔
وَإِن كَانَ قَمِيصُهُۥ قُدَّ مِن دُبُرٖ فَكَذَبَتْ وَهُوَ مِنَ ٱلصَّـٰدِقِينَ ﴿27﴾
اور اگر اس کا قمیص پیچھے سے پھٹا ہے تو وہ جھوٹی ہے اور یوسف سچا ہے۔
فَلَمَّا رَءَا قَمِيصَهُۥ قُدَّ مِن دُبُرٖ قَالَ إِنَّهُۥ مِن كَيْدِكُنَّۖ إِنَّ كَيْدَكُنَّ عَظِيمٞ ﴿28﴾
پھر جب اس نے دیکھا کہ یوسف کا قمیص پیچھے سے پھٹا ہے، تو کہا: یہ تم عورتوں کی چال ہے، بے شک تمہاری چال بہت بڑی ہے۔
يُوسُفُ أَعْرِضْ عَنْ هَـٰذَاۚ وَٱسْتَغْفِرِي لِذَنۢبِكِۖ إِنَّكِ كُنتِ مِنَ ٱلْخَاطِـِٔينَ ﴿29﴾
یوسف! اس بات کو نظر انداز کرو، اور (اے عورت!) تو اپنے گناہ کی معافی مانگ، بے شک تو خطا کاروں میں سے ہے۔
وَقَالَ نِسْوَةٞ فِي ٱلْمَدِينَةِ ٱمْرَأَتُ ٱلْعَزِيزِ تُرَٰوِدُ فَتَىٰهَا عَن نَّفْسِهِۦۖ قَدْ شَغَفَهَا حُبًّاۖ إِنَّا لَنَرَىٰهَا فِي ضَلَـٰلٖ مُّبِينٖ ﴿30﴾
اور شہر کی عورتوں نے کہا: عزیز کی بیوی اپنے غلام کو اپنی طرف مائل کرنا چاہتی ہے، بے شک اس کی محبت اس کے دل میں گھر کر گئی ہے، ہم تو اسے کھلی گمراہی میں دیکھتے ہیں۔
فَلَمَّا سَمِعَتْ بِمَكْرِهِنَّ أَرْسَلَتْ إِلَيْهِنَّ وَأَعْتَدَتْ لَهُنَّ مُتَّكَـٔٗا وَءَاتَتْ كُلَّ وَٰحِدَةٖ مِّنْهُنَّ سِكِّينٗا وَقَالَتِ ٱخْرُجْ عَلَيْهِنَّۖ فَلَمَّا رَأَيْنَهُۥٓ أَكْبَرْنَهُۥ وَقَطَّعْنَ أَيْدِيَهُنَّ وَقُلْنَ حَـٰشَ لِلَّهِ مَا هَـٰذَا بَشَرًا إِنْ هَـٰذَآ إِلَّا مَلَكٞ كَرِيمٞ ﴿31﴾
پھر جب اس نے ان کی چالاکی کی بات سنی، تو انہیں بلا بھیجا، اور ان کے لیے ایک مجلس تیار کی، اور ہر ایک کو ایک چھری دی، اور کہا: (یوسف!) ان کے سامنے آ جاؤ، پھر جب انہوں نے اسے دیکھا تو اسے بڑا سمجھا اور اپنے ہاتھ کاٹ لیے، اور کہا: اللہ کی پناہ! یہ انسان نہیں، یہ تو کوئی معزز فرشتہ ہے۔
قَالَتْ فَذَٰلِكُنَّ ٱلَّذِي لُمْتُنَّنِي فِيهِۖ وَلَقَدْ رَٰوَدتُّهُۥ عَن نَّفْسِهِۦ فَٱسْتَعْصَمَۖ وَلَئِن لَّمْ يَفْعَلْ مَآ ءَامُرُهُۥ لَيُسْجَنَنَّ وَلَيَكُونٗا مِّنَ ٱلصَّـٰغِرِينَ ﴿32﴾
اس نے کہا: یہی ہے وہ جس کے بارے میں تم مجھے ملامت کرتی تھیں، اور بے شک میں نے اسے اپنی طرف مائل کرنا چاہا، مگر اس نے بچاؤ کیا، اور اگر اس نے وہ نہ کیا جو میں کہتی ہوں تو وہ ضرور قید کیا جائے گا اور ذلیل ہوگا۔
قَالَ رَبِّ ٱلسِّجْنُ أَحَبُّ إِلَيَّ مِمَّا يَدْعُونَنِيٓ إِلَيْهِۖ وَإِلَّا تَصْرِفْ عَنِّي كَيْدَهُنَّ أَصْبُ إِلَيْهِنَّ وَأَكُن مِّنَ ٱلْجَـٰهِلِينَ ﴿33﴾
یوسف نے کہا: اے میرے رب! قید مجھے ان چیزوں سے زیادہ پسند ہے جن کی طرف یہ مجھے بلا رہی ہیں، اور اگر تو ان کی چال مجھ سے نہ ہٹائے تو میں ان کی طرف مائل ہو جاؤں گا اور جاہلوں میں سے ہو جاؤں گا۔
فَٱسْتَجَابَ لَهُۥ رَبُّهُۥ فَصَرَفَ عَنْهُ كَيْدَهُنَّۚ إِنَّهُۥ هُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ ﴿34﴾
تو اس کے رب نے اس کی دعا قبول کی، اور ان کی چالاکی اس سے دور کر دی، بے شک وہ سننے والا، جاننے والا ہے۔
ثُمَّ بَدَا لَهُم مِّنۢ بَعْدِ مَا رَأَوُا۟ ٱلْـَٔايَـٰتِ لَيَسْجُنُنَّهُۥ حَتَّىٰ حِينٖ ﴿35﴾
پھر ان کو یہ بات سوجھی کہ باوجود اس کے کہ وہ نشانیاں دیکھ چکے تھے، اسے کچھ مدت کے لیے قید کر دیں۔
وَدَخَلَ مَعَهُ ٱلسِّجْنَ فَتَيَانِۖ قَالَ أَحَدُهُمَآ إِنِّيٓ أَرَىٰنِيٓ أَعْصِرُ خَمْرٗاۖ وَقَالَ ٱلْـَٔاخَرُ إِنِّيٓ أَرَىٰنِيٓ أَحْمِلُ فَوْقَ رَأْسِي خُبْزٗا تَأْكُلُ ٱلطَّيْرُ مِنْهُۖ نَبِّئْنَا بِتَأْوِيلِهِۦٓ إِنَّا نَرَىٰكَ مِنَ ٱلْمُحْسِنِينَ ﴿36﴾
اور اس کے ساتھ دو جوان قید خانے میں داخل ہوئے، ان میں سے ایک نے کہا: میں خواب میں دیکھتا ہوں کہ شراب نچوڑ رہا ہوں، اور دوسرے نے کہا: میں دیکھتا ہوں کہ اپنے سر پر روٹی اٹھائے ہوئے ہوں جس میں سے پرندے کھا رہے ہیں، ہمیں اس کی تعبیر بتاؤ، بے شک ہم تمہیں نیکوکاروں میں سے دیکھتے ہیں۔
قَالَ لَا يَأْتِيكُمَا طَعَامٞ تُرْزَقَانِهِۦٓ إِلَّا نَبَّأْتُكُمَا بِتَأْوِيلِهِۦ قَبْلَ أَن يَأْتِيَكُمَاۚ ذَٰلِكُمَا مِمَّا عَلَّمَنِي رَبِّيٓۚ إِنِّي تَرَكْتُ مِلَّةَ قَوْمٖ لَّا يُؤْمِنُونَ بِٱللَّهِ وَهُم بِٱلْـَٔاخِرَةِ هُمْ كَـٰفِرُونَ ﴿37﴾
یوسف نے کہا: تم دونوں کو جو کھانا دیا جاتا ہے، وہ آنے سے پہلے میں تمہیں اس کی تعبیر بتا دوں گا، یہ اس علم میں سے ہے جو میرے رب نے مجھے سکھایا ہے، بے شک میں نے ان لوگوں کا دین چھوڑ دیا ہے جو اللہ پر ایمان نہیں رکھتے اور وہ آخرت کے منکر ہیں۔
وَٱتَّبَعْتُ مِلَّةَ ءَابَآءِيٓ إِبْرَٰهِيمَ وَإِسْحَـٰقَ وَيَعْقُوبَۚ مَا كَانَ لَنَآ أَن نُّشْرِكَ بِٱللَّهِ مِن شَيْءٖۚ ذَٰلِكَ مِن فَضْلِ ٱللَّهِ عَلَيْنَا وَعَلَى ٱلنَّاسِ وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَشْكُرُونَ ﴿38﴾
اور میں نے اپنے باپ دادا ابراہیم، اسحاق اور یعقوب کا دین اختیار کیا ہے، ہمارے لیے جائز نہیں کہ ہم اللہ کے ساتھ کسی کو شریک کریں، یہ اللہ کا فضل ہے ہم پر اور لوگوں پر، لیکن اکثر لوگ شکر نہیں کرتے۔
يَـٰصَـٰحِبَيِ ٱلسِّجْنِ ءَأَرْبَابٞ مُّتَفَرِّقُونَ خَيْرٌ أَمِ ٱللَّهُ ٱلْوَٰحِدُ ٱلْقَهَّارُ ﴿39﴾
اے میرے قید خانے کے ساتھیو! کیا مختلف رب بہتر ہیں یا اللہ، جو واحد اور زبردست ہے؟
مَا تَعْبُدُونَ مِن دُونِهِۦٓ إِلَّآ أَسْمَآءٗ سَمَّيْتُمُوهَآ أَنتُمْ وَءَابَآؤُكُم مَّآ أَنزَلَ ٱللَّهُ بِهَا مِن سُلْطَـٰنٍۚ إِنِ ٱلْحُكْمُ إِلَّا لِلَّهِۚ أَمَرَ أَلَّا تَعْبُدُوٓا۟ إِلَّآ إِيَّاهُۚ ذَٰلِكَ ٱلدِّينُ ٱلْقَيِّمُ وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَعْلَمُونَ ﴿40﴾
تم اللہ کے سوا صرف ناموں کی عبادت کرتے ہو جو تم نے اور تمہارے باپ دادا نے رکھ لیے ہیں، اللہ نے ان کے لیے کوئی دلیل نازل نہیں کی، حکم تو صرف اللہ ہی کا ہے، اس نے حکم دیا ہے کہ صرف اسی کی عبادت کرو، یہی سیدھا دین ہے، لیکن اکثر لوگ نہیں جانتے۔
يَـٰصَـٰحِبَيِ ٱلسِّجْنِ أَمَّآ أَحَدُكُمَا فَيَسْقِي رَبَّهُۥ خَمْرٗاۖ وَأَمَّا ٱلْـَٔاخَرُ فَيُصْلَبُ فَتَأْكُلُ ٱلطَّيْرُ مِن رَّأْسِهِۦۚ قُضِيَ ٱلْأَمْرُ ٱلَّذِي فِيهِ تَسْتَفْتِيَانِ ﴿41﴾
اے میرے قید خانے کے ساتھیو! تم میں سے ایک اپنے آقا کو شراب پلائے گا، اور دوسرا سولی پر چڑھایا جائے گا، اور پرندے اس کے سر سے کھائیں گے، وہ فیصلہ ہو چکا جس کے بارے میں تم دونوں نے پوچھا۔
وَقَالَ لِلَّذِي ظَنَّ أَنَّهُۥ نَاجٖ مِّنْهُمَا ٱذْكُرْنِي عِندَ رَبِّكَ فَأَنسَىٰهُ ٱلشَّيْطَـٰنُ ذِكْرَ رَبِّهِۦ فَلَبِثَ فِي ٱلسِّجْنِ بِضْعَ سِنِينَ ﴿42﴾
اور یوسف نے اس شخص سے کہا جس کے بارے میں اسے یقین تھا کہ وہ رہا ہو جائے گا: اپنے آقا سے میرا ذکر کرنا، لیکن شیطان نے اسے اپنے آقا سے ذکر کرنا بھلا دیا، تو یوسف کئی سال قید میں رہا۔
وَقَالَ ٱلْمَلِكُ إِنِّيٓ أَرَىٰ سَبْعَ بَقَرَٰتٖ سِمَانٖ يَأْكُلُهُنَّ سَبْعٌ عِجَافٞ وَسَبْعَ سُنبُلَـٰتٍ خُضْرٖ وَأُخَرَ يَابِسَـٰتٖۖ يَـٰٓأَيُّهَا ٱلْمَلَأُ أَفْتُونِي فِي رُءْيَـٰيَ إِن كُنتُمْ لِلرُّءْيَا تَعْبُرُونَ ﴿43﴾
اور بادشاہ نے کہا: میں خواب میں دیکھتا ہوں کہ سات موٹی گائیں ہیں جنہیں سات دبلی گائیں کھا رہی ہیں، اور سات سبز خوشے ہیں اور سات خشک، اے درباریو! میرے خواب کی تعبیر بتاؤ اگر تم خواب کی تعبیر جانتے ہو۔
قَالُوٓا۟ أَضْغَـٰثُ أَحْلَـٰمٖۖ وَمَا نَحْنُ بِتَأْوِيلِ ٱلْأَحْلَـٰمِ بِعَـٰلِمِينَ ﴿44﴾
انہوں نے کہا: یہ تو پریشان خواب ہیں، اور ہم خوابوں کی تعبیر نہیں جانتے۔
وَقَالَ ٱلَّذِي نَجَا مِنْهُمَا وَٱدَّكَرَ بَعْدَ أُمَّةٍ أَنَا۠ أُنَبِّئُكُم بِتَأْوِيلِهِۦ فَأَرْسِلُونِ ﴿45﴾
اور ان دونوں میں سے جو بچ گیا تھا، اس نے ایک مدت کے بعد یاد کیا، اور کہا: میں تمہیں اس کی تعبیر بتاتا ہوں، مجھے (یوسف کے پاس) بھیج دو۔
يُوسُفُ أَيُّهَا ٱلصِّدِّيقُ أَفْتِنَا فِي سَبْعِ بَقَرَٰتٖ سِمَانٖ يَأْكُلُهُنَّ سَبْعٌ عِجَافٞ وَسَبْعِ سُنبُلَـٰتٍ خُضْرٖ وَأُخَرَ يَابِسَـٰتٖ لَّعَلِّيٓ أَرْجِعُ إِلَى ٱلنَّاسِ لَعَلَّهُمْ يَعْلَمُونَ ﴿46﴾
یوسف! اے سچے! ہمیں سات موٹی گایوں کے بارے میں بتاؤ جنہیں سات دبلی گائیں کھا رہی ہیں، اور سات سبز خوشوں اور سات خشک کے بارے میں، تاکہ میں لوگوں کے پاس واپس جاؤں، شاید وہ جان لیں۔
قَالَ تَزْرَعُونَ سَبْعَ سِنِينَ دَأَبٗا فَمَا حَصَدتُّمۡ فَذَرُوهُ فِي سُنبُلِهِۦٓ إِلَّا قَلِيلٗا مِّمَّا تَأْكُلُونَ ﴿47﴾
یوسف نے کہا: تم سات سال لگاتار کھیتی کرو گے، پھر جو کچھ تم کاٹو اسے اس کے خوشوں میں رہنے دو، سوائے تھوڑے سے جو تم کھاؤ۔
ثُمَّ يَأْتِي مِنۢ بَعْدِ ذَٰلِكَ سَبْعٞ شِدَادٞ يَأْكُلْنَ مَا قَدَّمْتُمْ لَهُنَّ إِلَّا قَلِيلٗا مِّمَّا تُحْصِنُونَ ﴿48﴾
پھر اس کے بعد سات سخت سال آئیں گے جو وہ سب کچھ کھا جائیں گے جو تم نے ان کے لیے ذخیرہ کیا ہوگا، سوائے تھوڑے سے جو تم محفوظ رکھو گے۔
ثُمَّ يَأْتِي مِنۢ بَعْدِ ذَٰلِكَ عَامٞ فِيهِ يُغَاثُ ٱلنَّاسُ وَفِيهِ يَعْصِرُونَ ﴿49﴾
پھر اس کے بعد ایک سال آئے گا جس میں لوگوں کو بارش دی جائے گی اور وہ رس نچوڑیں گے۔
وَقَالَ ٱلْمَلِكُ ٱئْتُونِي بِهِۦۖ فَلَمَّا جَآءَهُ ٱلرَّسُولُ قَالَ ٱرْجِعْ إِلَىٰ رَبِّكَ فَسْـَٔلْهُۥ مَا بَالُ ٱلنِّسْوَةِ ٱلَّـٰتِي قَطَّعْنَ أَيْدِيَهُنَّۚ إِنَّ رَبِّي بِكَيْدِهِنَّ عَلِيمٞ ﴿50﴾
اور بادشاہ نے کہا: اسے میرے پاس لاؤ، پھر جب قاصد یوسف کے پاس آیا تو یوسف نے کہا: اپنے آقا کے پاس واپس جاؤ اور اس سے پوچھو کہ ان عورتوں کا کیا معاملہ ہے جنہوں نے اپنے ہاتھ کاٹ لیے تھے، بے شک میرا رب ان کی چالاکی کو خوب جانتا ہے۔
قَالَ مَا خَطْبُكُنَّ إِذْ رَٰوَدتُّنَّ يُوسُفَ عَن نَّفْسِهِۦۚ قُلْنَ حَـٰشَ لِلَّهِ مَا عَلِمْنَا عَلَيْهِ سُوٓءٗاۚ قَالَتِ ٱمْرَأَتُ ٱلْعَزِيزِ ٱلْـَٔـٰنَ حَصْحَصَ ٱلْحَقُّ أَنَا۠ رَٰوَدتُّهُۥ عَن نَّفْسِهِۦ وَإِنَّهُۥ لَمِنَ ٱلصَّـٰدِقِينَ ﴿51﴾
بادشاہ نے کہا: تمہارا کیا معاملہ ہے جب تم نے یوسف کو اپنی طرف مائل کرنا چاہا؟ انہوں نے کہا: اللہ کی پناہ! ہم نے اس میں کوئی برائی نہیں دیکھی۔ عزیز کی بیوی نے کہا: اب حق ظاہر ہو گیا، میں نے ہی اسے اپنی طرف مائل کرنا چاہا تھا، اور بے شک وہ سچوں میں سے ہے۔
ذَٰلِكَ لِيَعْلَمَ أَنِّي لَمْ أَخُنْهُۥ بِٱلْغَيْبِ وَأَنَّ ٱللَّهَ لَا يَهْدِي كَيْدَ ٱلْخَـٰٓئِنِينَ ﴿52﴾
یہ اس لیے ہے تاکہ وہ جان لے کہ میں نے اس کی غیر موجودگی میں اس سے خیانت نہیں کی، اور بے شک اللہ خیانت کرنے والوں کی چال کو کامیاب نہیں کرتا۔
