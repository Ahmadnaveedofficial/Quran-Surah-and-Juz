سورة آل عمران

بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
الٓمٓ ﴿1﴾
الف، لام، میم۔
ٱللَّهُ لَآ إِلَـٰهَ إِلَّا هُوَ ٱلْحَيُّ ٱلْقَيُّومُ ﴿2﴾
اللہ، اس کے سوا کوئی معبود نہیں، وہ زندہ ہے، قائم رکھنے والا ہے۔
نَزَّلَ عَلَيْكَ ٱلْكِتَـٰبَ بِٱلْحَقِّ مُصَدِّقٗا لِّمَا بَيْنَ يَدَيْهِ وَأَنزَلَ ٱلتَّوْرَىٰةَ وَٱلْإِنجِيلَ ﴿3﴾
اس نے آپ پر حق کے ساتھ کتاب نازل کی، جو اس سے پہلے کی تصدیق کرنے والی ہے، اور اس نے تورات اور انجیل نازل کی۔
مِن قَبْلُ هُدٗى لِّلنَّاسِۚ وَأَنزَلَ ٱلْفُرْقَانَۗ إِنَّ ٱلَّذِينَ كَفَرُوا۟ بِـَٔايَـٰتِ ٱللَّهِ لَهُمْ عَذَابٞ شَدِيدٞۗ وَٱللَّهُ عَزِيزٞ ذُو ٱنتِقَامٍ ﴿4﴾
اس سے پہلے لوگوں کی ہدایت کے لیے، اور اس نے فرقان نازل کی، بے شک جو لوگ اللہ کی آیات کا انکار کرتے ہیں ان کے لیے سخت عذاب ہے، اور اللہ زبردست ہے، انتقام لینے والا ہے۔
إِنَّ ٱللَّهَ لَا يَخْفَىٰ عَلَيْهِ شَيْءٞ فِي ٱلْأَرْضِ وَلَا فِي ٱلسَّمَآءِ ﴿5﴾
بے شک اللہ سے کوئی چیز چھپی نہیں، نہ زمین میں اور نہ آسمان میں۔
هُوَ ٱلَّذِي يُصَوِّرُكُمْ فِي ٱلْأَرْحَامِ كَيْفَ يَشَآءُۚ لَآ إِلَـٰهَ إِلَّا هُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿6﴾
وہی ہے جو تمہیں رحموں میں جس طرح چاہتا ہے صورت دیتا ہے، اس کے سوا کوئی معبود نہیں، وہ زبردست، حکمت والا ہے۔
هُوَ ٱلَّذِيٓ أَنزَلَ عَلَيْكَ ٱلْكِتَـٰبَ مِنْهُ ءَايَـٰتٞ مُّحْكَمَـٰتٌ هُنَّ أُمُّ ٱلْكِتَـٰبِ وَأُخَرُ مُتَشَـٰبِهَـٰتٞۖ فَأَمَّا ٱلَّذِينَ فِي قُلُوبِهِمْ زَيْغٞ فَيَتَّبِعُونَ مَا تَشَـٰبَهَ مِنْهُ ٱبْتِغَآءَ ٱلْفِتْنَةِ وَٱبْتِغَآءَ تَأْوِيلِهِۦۗ وَمَا يَعْلَمُ تَأْوِيلَهُۥٓ إِلَّا ٱللَّهُۗ وَٱلرَّٰسِخُونَ فِي ٱلْعِلْمِ يَقُولُونَ ءَامَنَّا بِهِۦ كُلّٞ مِّنْ عِندِ رَبِّنَاۗ وَمَا يَذَّكَّرُ إِلَّآ أُو۟لُوا۟ ٱلْأَلْبَـٰبِ ﴿7﴾
وہی ہے جس نے آپ پر کتاب نازل کی، اس میں کچھ آیات محکم ہیں، وہی اصل کتاب ہیں، اور کچھ متشابہ ہیں، پس جن کے دلوں میں کجی ہے وہ متشابہات کے پیچھے پڑتے ہیں، فتنہ انگیزی اور تاویل کی تلاش میں، حالانکہ اس کی تاویل اللہ کے سوا کوئی نہیں جانتا، اور جو علم میں پختہ ہیں وہ کہتے ہیں: ہم اس پر ایمان لائے، سب کچھ ہمارے رب کی طرف سے ہے، اور نصیحت صرف عقل والے ہی حاصل کرتے ہیں۔
رَبَّنَا لَا تُزِغْ قُلُوبَنَا بَعْدَ إِذْ هَدَيْتَنَا وَهَبْ لَنَا مِن لَّدُنكَ رَحْمَةًۚ إِنَّكَ أَنتَ ٱلْوَهَّابُ ﴿8﴾
اے ہمارے رب! ہمارے دلوں کو ہدایت دینے کے بعد ٹیڑھا نہ کر، اور ہمیں اپنی طرف سے رحمت عطا فرما، بے شک تو ہی بہت عطا کرنے والا ہے۔
رَبَّنَآ إِنَّكَ جَامِعُ ٱلنَّاسِ لِيَوْمٖ لَّا رَيْبَ فِيهِۚ إِنَّ ٱللَّهَ لَا يُخْلِفُ ٱلْمِيعَادَ ﴿9﴾
اے ہمارے رب! بے شک تو لوگوں کو اس دن جمع کرنے والا ہے جس میں کوئی شک نہیں، بے شک اللہ وعدہ خلافی نہیں کرتا۔
إِنَّ ٱلَّذِينَ كَفَرُوا۟ لَن تُغْنِيَ عَنْهُمْ أَمْوَٰلُهُمْ وَلَآ أَوْلَـٰدُهُم مِّنَ ٱللَّهِ شَيْـٔٗاۖ وَأُو۟لَـٰٓئِكَ هُمْ وَقُودُ ٱلنَّارِ ﴿10﴾
بے شک جنہوں نے کفر کیا، ان کے مال اور ان کی اولاد اللہ کے مقابلے میں ان کے کچھ بھی کام نہ آئیں گے، اور وہی لوگ دوزخ کا ایندھن ہیں۔
كَدَأْبِ ءَالِ فِرْعَوْنَ وَٱلَّذِينَ مِن قَبْلِهِمۚ كَذَّبُوا۟ بِـَٔايَـٰتِنَا فَأَخَذَهُمُ ٱللَّهُ بِذُنُوبِهِمۗ وَٱللَّهُ شَدِيدُ ٱلْعِقَابِ ﴿11﴾
جیسے فرعون والوں اور ان سے پہلے لوگوں کا حال تھا، انہوں نے ہماری آیات کو جھٹلایا، تو اللہ نے انہیں ان کے گناہوں کی وجہ سے پکڑ لیا، اور اللہ سخت سزا دینے والا ہے۔
قُل لِّلَّذِينَ كَفَرُوا۟ سَتُغْلَبُونَ وَتُحْشَرُونَ إِلَىٰ جَهَنَّمَۖ وَبِئْسَ ٱلْمِهَادُ ﴿12﴾
کہہ دیجیے ان لوگوں سے جنہوں نے کفر کیا: تم مغلوب کیے جاؤ گے اور جہنم کی طرف جمع کیے جاؤ گے، اور وہ بہت ہی برا ٹھکانہ ہے۔
قَدْ كَانَ لَكُمْ ءَايَةٞ فِي فِئَتَيْنِ ٱلْتَقَتَاۖ فِئَةٞ تُقَـٰتِلُ فِي سَبِيلِ ٱللَّهِ وَأُخْرَىٰ كَافِرَةٞ يَرَوْنَهُم مِّثْلَيْهِمْ رَأْيَ ٱلْعَيْنِۚ وَٱللَّهُ يُؤَيِّدُ بِنَصْرِهِۦ مَن يَشَآءُۚ إِنَّ فِي ذَٰلِكَ لَعِبْرَةٗ لِّأُو۟لِي ٱلْأَبْصَـٰرِ ﴿13﴾
تمہارے لیے دو گروہوں کے مقابلے میں ایک نشانی تھی، ایک گروہ اللہ کی راہ میں لڑ رہا تھا اور دوسرا کافر تھا، وہ انہیں اپنی آنکھوں سے دوگنا دیکھ رہے تھے، اور اللہ اپنی مدد سے جسے چاہتا ہے تقویت دیتا ہے، بے شک اس میں آنکھوں والوں کے لیے عبرت ہے۔
زُيِّنَ لِلنَّاسِ حُبُّ ٱلشَّهَوَٰتِ مِنَ ٱلنِّسَآءِ وَٱلْبَنِينَ وَٱلْقَنَـٰطِيرِ ٱلْمُقَنطَرَةِ مِنَ ٱلذَّهَبِ وَٱلْفِضَّةِ وَٱلْخَيْلِ ٱلْمُسَوَّمَةِ وَٱلْأَنْعَـٰمِ وَٱلْحَرْثِۗ ذَٰلِكَ مَتَـٰعُ ٱلْحَيَوٰةِ ٱلدُّنْيَاۖ وَٱللَّهُ عِندَهُۥ حُسْنُ ٱلْمَـَٔابِ ﴿14﴾
لوگوں کے لیے خواہشات کی چیزیں خوشنما بنا دی گئی ہیں: عورتیں، بیٹے، سونے چاندی کے جمع کیے ہوئے خزانے، نشان زدہ گھوڑے، مویشی اور کھیتیاں — یہ سب دنیا کی زندگی کا سامان ہے، اور اللہ ہی کے پاس بہترین ٹھکانہ ہے۔
قُلْ أَؤُنَبِّئُكُم بِخَيْرٖ مِّن ذَٰلِكُمۚ لِلَّذِينَ ٱتَّقَوْا۟ عِندَ رَبِّهِمْ جَنَّـٰتٞ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُ خَـٰلِدِينَ فِيهَا وَأَزْوَٰجٞ مُّطَهَّرَةٞ وَرِضْوَٰنٞ مِّنَ ٱللَّهِۗ وَٱللَّهُ بَصِيرُۢ بِٱلْعِبَادِ ﴿15﴾
کہہ دیجیے: کیا میں تمہیں اس سے بہتر چیز کی خبر دوں؟ جو لوگ تقویٰ اختیار کرتے ہیں ان کے لیے ان کے رب کے پاس باغات ہیں جن کے نیچے نہریں بہتی ہیں، وہ ان میں ہمیشہ رہیں گے، اور پاکیزہ بیویاں ہوں گی، اور اللہ کی رضا مندی، اور اللہ بندوں کو خوب دیکھنے والا ہے۔
ٱلَّذِينَ يَقُولُونَ رَبَّنَآ إِنَّنَآ ءَامَنَّا فَٱغْفِرْ لَنَا ذُنُوبَنَا وَقِنَا عَذَابَ ٱلنَّارِ ﴿16﴾
وہ لوگ جو کہتے ہیں: اے ہمارے رب! بے شک ہم ایمان لائے، پس ہمارے گناہ بخش دے اور ہمیں آگ کے عذاب سے بچا۔
ٱلصَّـٰبِرِينَ وَٱلصَّـٰدِقِينَ وَٱلْقَـٰنِتِينَ وَٱلْمُنفِقِينَ وَٱلْمُسْتَغْفِرِينَ بِٱلْأَسْحَارِ ﴿17﴾
(یہ لوگ) صبر کرنے والے، سچے، فرمانبردار، اللہ کی راہ میں خرچ کرنے والے، اور سحر کے وقت استغفار کرنے والے ہیں۔
شَهِدَ ٱللَّهُ أَنَّهُۥ لَآ إِلَـٰهَ إِلَّا هُوَ وَٱلْمَلَـٰٓئِكَةُ وَأُو۟لُوا۟ ٱلْعِلْمِ قَآئِمَۢا بِٱلْقِسْطِۚ لَآ إِلَـٰهَ إِلَّا هُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿18﴾
اللہ نے گواہی دی کہ اس کے سوا کوئی معبود نہیں، اور فرشتوں نے اور علم والوں نے بھی، وہ انصاف پر قائم ہے، اس کے سوا کوئی معبود نہیں، وہ زبردست، حکمت والا ہے۔
إِنَّ ٱلدِّينَ عِندَ ٱللَّهِ ٱلْإِسْلَـٰمُۗ وَمَا ٱخْتَلَفَ ٱلَّذِينَ أُوتُوا۟ ٱلْكِتَـٰبَ إِلَّا مِنۢ بَعْدِ مَا جَآءَهُمُ ٱلْعِلْمُ بَغْيَۢا بَيْنَهُمْۚ وَمَن يَكْفُرْ بِـَٔايَـٰتِ ٱللَّهِ فَإِنَّ ٱللَّهَ سَرِيعُ ٱلْحِسَابِ ﴿19﴾
بے شک اللہ کے نزدیک دین اسلام ہی ہے، اور اہلِ کتاب نے اختلاف نہیں کیا مگر اس کے بعد کہ ان کے پاس علم آ چکا تھا، آپس کی ضد کی وجہ سے، اور جو کوئی اللہ کی آیات کا انکار کرے تو بے شک اللہ جلد حساب لینے والا ہے۔
فَإِنْ حَآجُّوكَ فَقُلْ أَسْلَمْتُ وَجْهِيَ لِلَّهِ وَمَنِ ٱتَّبَعَنِۗ وَقُل لِّلَّذِينَ أُوتُوا۟ ٱلْكِتَـٰبَ وَٱلْأُمِّيِّـۧنَ أَأَسْلَمْتُمْۚ فَإِنْ أَسْلَمُوا۟ فَقَدِ ٱهْتَدَوْا۟ۖ وَإِن تَوَلَّوْا۟ فَإِنَّمَا عَلَيْكَ ٱلْبَلَـٰغُۗ وَٱللَّهُ بَصِيرُۢ بِٱلْعِبَادِ ﴿20﴾
پھر اگر وہ آپ سے جھگڑا کریں تو کہہ دیجیے: میں نے اپنا چہرہ اللہ کے لیے جھکا دیا ہے، اور وہ بھی جنہوں نے میری پیروی کی، اور اہلِ کتاب اور ان پڑھ لوگوں سے کہہ دیجیے: کیا تم نے اسلام قبول کیا؟ پھر اگر وہ اسلام لے آئیں تو ہدایت پا گئے، اور اگر منہ موڑیں تو آپ پر صرف پہنچا دینا ہے، اور اللہ بندوں کو خوب دیکھنے والا ہے۔
إِنَّ ٱلَّذِينَ يَكْفُرُونَ بِـَٔايَـٰتِ ٱللَّهِ وَيَقْتُلُونَ ٱلنَّبِيِّـۧنَ بِغَيْرِ حَقّٖ وَيَقْتُلُونَ ٱلَّذِينَ يَأْمُرُونَ بِٱلْقِسْطِ مِنَ ٱلنَّاسِ فَبَشِّرْهُم بِعَذَابٍ أَلِيمٍ ﴿21﴾
بے شک جو لوگ اللہ کی آیات کا انکار کرتے ہیں، اور نبیوں کو ناحق قتل کرتے ہیں، اور لوگوں میں سے جو انصاف کا حکم دیتے ہیں انہیں قتل کرتے ہیں — تو انہیں دردناک عذاب کی خوشخبری دے دیجیے۔
أُو۟لَـٰٓئِكَ ٱلَّذِينَ حَبِطَتْ أَعْمَـٰلُهُمْ فِي ٱلدُّنْيَا وَٱلْـَٔاخِرَةِۖ وَمَا لَهُم مِّن نَّـٰصِرِينَ ﴿22﴾
یہی وہ لوگ ہیں جن کے اعمال دنیا اور آخرت میں ضائع ہو گئے، اور ان کے لیے کوئی مددگار نہیں ہوگا۔
أَلَمْ تَرَ إِلَى ٱلَّذِينَ أُوتُوا۟ نَصِيبٗا مِّنَ ٱلْكِتَـٰبِ يُدْعَوْنَ إِلَىٰ كِتَـٰبِ ٱللَّهِ لِيَحْكُمَ بَيْنَهُمْ ثُمَّ يَتَوَلَّىٰ فَرِيقٞ مِّنْهُمْ وَهُم مُّعْرِضُونَ ﴿23﴾
کیا آپ نے ان لوگوں کو نہیں دیکھا جنہیں کتاب کا کچھ حصہ دیا گیا، انہیں اللہ کی کتاب کی طرف بلایا جاتا ہے تاکہ وہ ان کے درمیان فیصلہ کرے، پھر ان میں سے ایک گروہ منہ موڑ لیتا ہے حالانکہ وہ اعراض کرنے والے ہیں۔
ذَٰلِكَ بِأَنَّهُمْ قَالُوا۟ لَن تَمَسَّنَا ٱلنَّارُ إِلَّآ أَيَّامٗا مَّعْدُودَةٗۚ وَغَرَّهُمْ فِي دِينِهِم مَّا كَانُوا۟ يَفْتَرُونَ ﴿24﴾
یہ اس لیے کہ وہ کہتے ہیں: ہمیں آگ صرف چند گنتی کے دن ہی چھوئے گی، اور ان کے دین میں ان کی گھڑی ہوئی باتوں نے انہیں دھوکے میں ڈال رکھا ہے۔
فَكَيْفَ إِذَا جَمَعْنَـٰهُمْ لِيَوْمٖ لَّا رَيْبَ فِيهِ وَوُفِّيَتْ كُلُّ نَفْسٖ مَّا كَسَبَتْ وَهُمْ لَا يُظْلَمُونَ ﴿25﴾
تو کیا حال ہوگا جب ہم انہیں اس دن جمع کریں گے جس میں کوئی شک نہیں، اور ہر شخص کو اس کے کیے کا پورا بدلہ دیا جائے گا، اور ان پر ظلم نہیں کیا جائے گا؟
قُلِ ٱللَّهُمَّ مَـٰلِكَ ٱلْمُلْكِ تُؤْتِي ٱلْمُلْكَ مَن تَشَآءُ وَتَنزِعُ ٱلْمُلْكَ مِمَّن تَشَآءُۖ وَتُعِزُّ مَن تَشَآءُ وَتُذِلُّ مَن تَشَآءُۖ بِيَدِكَ ٱلْخَيْرُۖ إِنَّكَ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٞ ﴿26﴾
کہہ دیجیے: اے اللہ! بادشاہی کے مالک! تو جسے چاہے بادشاہی دیتا ہے اور جس سے چاہے بادشاہی چھین لیتا ہے، اور جسے چاہے عزت دیتا ہے اور جسے چاہے ذلیل کرتا ہے، تیرے ہی ہاتھ میں ہر بھلائی ہے، بے شک تو ہر چیز پر قادر ہے۔
تُولِجُ ٱلَّيْلَ فِي ٱلنَّهَارِ وَتُولِجُ ٱلنَّهَارَ فِي ٱلَّيْلِۖ وَتُخْرِجُ ٱلْحَيَّ مِنَ ٱلْمَيِّتِ وَتُخْرِجُ ٱلْمَيِّتَ مِنَ ٱلْحَيِّۖ وَتَرْزُقُ مَن تَشَآءُ بِغَيْرِ حِسَابٖ ﴿27﴾
تو رات کو دن میں داخل کرتا ہے اور دن کو رات میں داخل کرتا ہے، اور زندہ کو مردہ سے نکالتا ہے اور مردہ کو زندہ سے نکالتا ہے، اور جسے چاہے بے حساب رزق دیتا ہے۔
لَّا يَتَّخِذِ ٱلْمُؤْمِنُونَ ٱلْكَـٰفِرِينَ أَوْلِيَآءَ مِن دُونِ ٱلْمُؤْمِنِينَۖ وَمَن يَفْعَلْ ذَٰلِكَ فَلَيْسَ مِنَ ٱللَّهِ فِي شَيْءٍ إِلَّآ أَن تَتَّقُوا۟ مِنْهُمْ تُقَىٰةٗۗ وَيُحَذِّرُكُمُ ٱللَّهُ نَفْسَهُۚ وَإِلَى ٱللَّهِ ٱلْمَصِيرُ ﴿28﴾
مؤمنوں کو چاہیے کہ کافروں کو مؤمنوں کے سوا دوست نہ بنائیں، اور جو ایسا کرے گا اس کا اللہ سے کوئی تعلق نہیں، سوائے اس کے کہ تم ان سے بچاؤ کے طور پر (ظاہری تعلق) رکھو، اور اللہ تمہیں اپنی ذات سے ڈراتا ہے، اور اللہ ہی کی طرف لوٹ کر جانا ہے۔
قُلْ إِن تُخْفُوا۟ مَا فِي صُدُورِكُمْ أَوْ تُبْدُوهُ يَعْلَمْهُ ٱللَّهُۗ وَيَعْلَمُ مَا فِي ٱلسَّمَـٰوَٰتِ وَمَا فِي ٱلْأَرْضِۗ وَٱللَّهُ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٞ ﴿29﴾
کہہ دیجیے: اگر تم اپنے سینوں کی بات چھپاؤ یا ظاہر کرو، اللہ اسے جانتا ہے، اور وہ جانتا ہے جو کچھ آسمانوں اور زمین میں ہے، اور اللہ ہر چیز پر قادر ہے۔
يَوْمَ تَجِدُ كُلُّ نَفْسٖ مَّا عَمِلَتْ مِنْ خَيْرٖ مُّحْضَرٗا وَمَا عَمِلَتْ مِن سُوٓءٖۖ تَوَدُّ لَوْ أَنَّ بَيْنَهَا وَبَيْنَهُۥٓ أَمَدَۢا بَعِيدٗاۗ وَيُحَذِّرُكُمُ ٱللَّهُ نَفْسَهُۗ وَٱللَّهُ رَءُوفُۢ بِٱلْعِبَادِ ﴿30﴾
جس دن ہر شخص اپنے کیے ہوئے نیک عمل کو حاضر پائے گا، اور جو برائیاں کی ہوں گی وہ بھی، تو وہ چاہے گا کہ کاش اس کے اور ان کے درمیان بہت دوری ہوتی، اور اللہ تمہیں اپنی ذات سے ڈراتا ہے، اور اللہ بندوں پر بہت مہربان ہے۔
قُلْ إِن كُنتُمْ تُحِبُّونَ ٱللَّهَ فَٱتَّبِعُونِي يُحْبِبْكُمُ ٱللَّهُ وَيَغْفِرْ لَكُمْ ذُنُوبَكُمْۗ وَٱللَّهُ غَفُورٞ رَّحِيمٌ ﴿31﴾
کہہ دیجیے: اگر تم اللہ سے محبت کرتے ہو تو میری پیروی کرو، اللہ تم سے محبت کرے گا اور تمہارے گناہ معاف کر دے گا، اور اللہ بخشنے والا، مہربان ہے۔
قُلْ أَطِيعُوا۟ ٱللَّهَ وَٱلرَّسُولَۖ فَإِن تَوَلَّوْا۟ فَإِنَّ ٱللَّهَ لَا يُحِبُّ ٱلْكَـٰفِرِينَ ﴿32﴾
کہہ دیجیے: اللہ اور رسول کی اطاعت کرو، پھر اگر وہ منہ موڑیں تو بے شک اللہ کافروں کو پسند نہیں کرتا۔
إِنَّ ٱللَّهَ ٱصْطَفَىٰٓ ءَادَمَ وَنُوحٗا وَءَالَ إِبْرَٰهِيمَ وَءَالَ عِمْرَٰنَ عَلَى ٱلْعَـٰلَمِينَ ﴿33﴾
بے شک اللہ نے آدم، نوح، آلِ ابراہیم اور آلِ عمران کو تمام جہان والوں پر چن لیا۔
ذُرِّيَّةَۢ بَعْضُهَا مِنۢ بَعْضٖۗ وَٱللَّهُ سَمِيعٌ عَلِيمٌ ﴿34﴾
یہ نسلیں ایک دوسرے سے ہیں، اور اللہ سب کچھ سننے والا، جاننے والا ہے۔
إِذْ قَالَتِ ٱمْرَأَتُ عِمْرَٰنَ رَبِّ إِنِّي نَذَرْتُ لَكَ مَا فِي بَطْنِي مُحَرَّرٗا فَتَقَبَّلْ مِنِّيٓۖ إِنَّكَ أَنتَ ٱلسَّمِيعُ ٱلْعَلِيمُ ﴿35﴾
جب عمران کی بیوی نے کہا: اے میرے رب! میں نے اپنے پیٹ میں جو ہے اسے تیرے لیے نذر کیا ہے، آزاد (خالص) طور پر، پس تو اسے مجھ سے قبول فرما، بے شک تو ہی سننے والا، جاننے والا ہے۔
فَلَمَّا وَضَعَتْهَا قَالَتْ رَبِّ إِنِّي وَضَعْتُهَآ أُنثَىٰ وَٱللَّهُ أَعْلَمُ بِمَا وَضَعَتْۖ وَلَيْسَ ٱلذَّكَرُ كَٱلْأُنثَىٰۖ وَإِنِّي سَمَّيْتُهَا مَرْيَمَ وَإِنِّيٓ أُعِيذُهَا بِكَ وَذُرِّيَّتَهَا مِنَ ٱلشَّيْطَـٰنِ ٱلرَّجِيمِ ﴿36﴾
پھر جب اس نے اسے جنا تو کہا: اے میرے رب! میں نے تو اسے لڑکی جنا ہے، اور اللہ خوب جانتا ہے جو اس نے جنا، اور لڑکا لڑکی جیسا نہیں ہوتا، اور میں نے اس کا نام مریم رکھا ہے، اور میں اسے اور اس کی اولاد کو تیرے سپرد کرتی ہوں شیطان مردود سے بچانے کے لیے۔
فَتَقَبَّلَهَا رَبُّهَا بِقَبُولٍ حَسَنٖ وَأَنبَتَهَا نَبَاتًا حَسَنٗا وَكَفَّلَهَا زَكَرِيَّاۖ كُلَّمَا دَخَلَ عَلَيْهَا زَكَرِيَّا ٱلْمِحْرَابَ وَجَدَ عِندَهَا رِزْقٗاۖ قَالَ يَـٰمَرْيَمُ أَنَّىٰ لَكِ هَـٰذَاۖ قَالَتْ هُوَ مِنْ عِندِ ٱللَّهِۖ إِنَّ ٱللَّهَ يَرْزُقُ مَن يَشَآءُ بِغَيْرِ حِسَابٍ ﴿37﴾
تو اس کے رب نے اسے اچھی قبولیت کے ساتھ قبول کیا، اور اسے اچھی طرح اگایا، اور زکریا کو اس کی کفالت دی، جب بھی زکریا اس کے پاس محراب میں آتے تو اس کے پاس کوئی رزق پاتے، وہ کہتے: اے مریم! یہ تمہارے پاس کہاں سے آیا؟ وہ کہتی: یہ اللہ کی طرف سے ہے، بے شک اللہ جسے چاہے بے حساب رزق دیتا ہے۔
هُنَالِكَ دَعَا زَكَرِيَّا رَبَّهُۥۖ قَالَ رَبِّ هَبْ لِي مِن لَّدُنكَ ذُرِّيَّةٗ طَيِّبَةًۖ إِنَّكَ سَمِيعُ ٱلدُّعَآءِ ﴿38﴾
وہیں زکریا نے اپنے رب سے دعا کی، کہا: اے میرے رب! مجھے اپنی طرف سے پاکیزہ اولاد عطا فرما، بے شک تو دعا سننے والا ہے۔
فَنَادَتْهُ ٱلْمَلَـٰٓئِكَةُ وَهُوَ قَآئِمٞ يُصَلِّي فِي ٱلْمِحْرَابِ أَنَّ ٱللَّهَ يُبَشِّرُكَ بِيَحْيَىٰ مُصَدِّقَۢا بِكَلِمَةٖ مِّنَ ٱللَّهِ وَسَيِّدٗا وَحَصُورٗا وَنَبِيّٗا مِّنَ ٱلصَّـٰلِحِينَ ﴿39﴾
تو فرشتوں نے اسے آواز دی جب وہ محراب میں کھڑے نماز پڑھ رہے تھے: بے شک اللہ تمہیں یحییٰ کی خوشخبری دیتا ہے، جو اللہ کے ایک کلمہ کی تصدیق کرنے والا ہوگا، سردار، پاکدامن، اور نبی ہوگا صالحین میں سے۔
قَالَ رَبِّ أَنَّىٰ يَكُونُ لِي غُلَـٰمٞ وَقَدْ بَلَغَنِيَ ٱلْكِبَرُ وَٱمْرَأَتِي عَاقِرٞۖ قَالَ كَذَٰلِكَ ٱللَّهُ يَفْعَلُ مَا يَشَآءُ ﴿40﴾
زکریا نے کہا: اے میرے رب! مجھے لڑکا کیسے ہوگا جبکہ مجھے بڑھاپا آ چکا ہے اور میری بیوی بانجھ ہے؟ فرمایا: اسی طرح اللہ جو چاہے کرتا ہے۔
قَالَ رَبِّ ٱجْعَل لِّيٓ ءَايَةٗۚ قَالَ ءَايَتُكَ أَلَّا تُكَلِّمَ ٱلنَّاسَ ثَلَـٰثَةَ أَيَّامٍ إِلَّا رَمْزٗاۗ وَٱذْكُر رَّبَّكَ كَثِيرٗا وَسَبِّحْ بِٱلْعَشِيِّ وَٱلْإِبْكَـٰرِ ﴿41﴾
زکریا نے کہا: اے میرے رب! میرے لیے کوئی نشانی مقرر فرما، فرمایا: تمہاری نشانی یہ ہے کہ تم تین دن لوگوں سے بات نہ کر سکو گے مگر اشارے سے، اور اپنے رب کو کثرت سے یاد کرو اور شام و صبح تسبیح کرو۔
وَإِذْ قَالَتِ ٱلْمَلَـٰٓئِكَةُ يَـٰمَرْيَمُ إِنَّ ٱللَّهَ ٱصْطَفَىٰكِ وَطَهَّرَكِ وَٱصْطَفَىٰكِ عَلَىٰ نِسَآءِ ٱلْعَـٰلَمِينَ ﴿42﴾
اور جب فرشتوں نے کہا: اے مریم! بے شک اللہ نے تمہیں چن لیا، اور تمہیں پاک کیا، اور تمہیں تمام جہانوں کی عورتوں پر فضیلت دی۔
يَـٰمَرْيَمُ ٱقْنُتِي لِرَبِّكِ وَٱسْجُدِي وَٱرْكَعِي مَعَ ٱلرَّٰكِعِينَ ﴿43﴾
اے مریم! اپنے رب کے لیے عاجزی سے عبادت کرو، اور سجدہ کرو، اور رکوع کرنے والوں کے ساتھ رکوع کرو۔
ذَٰلِكَ مِنْ أَنبَآءِ ٱلْغَيْبِ نُوحِيهِ إِلَيْكَۚ وَمَا كُنتَ لَدَيْهِمْ إِذْ يُلْقُونَ أَقْلَـٰمَهُمْ أَيُّهُمْ يَكْفُلُ مَرْيَمَۖ وَمَا كُنتَ لَدَيْهِمْ إِذْ يَخْتَصِمُونَ ﴿44﴾
یہ غیب کی خبروں میں سے ہے جو ہم آپ کی طرف وحی کرتے ہیں، اور آپ ان کے پاس نہ تھے جب وہ اپنی قلمیں ڈال رہے تھے کہ ان میں سے کون مریم کی کفالت کرے، اور نہ آپ ان کے پاس تھے جب وہ جھگڑ رہے تھے۔
إِذْ قَالَتِ ٱلْمَلَـٰٓئِكَةُ يَـٰمَرْيَمُ إِنَّ ٱللَّهَ يُبَشِّرُكِ بِكَلِمَةٖ مِّنْهُ ٱسْمُهُ ٱلْمَسِيحُ عِيسَى ٱبْنُ مَرْيَمَ وَجِيهٗا فِي ٱلدُّنْيَا وَٱلْـَٔاخِرَةِ وَمِنَ ٱلْمُقَرَّبِينَ ﴿45﴾
جب فرشتوں نے کہا: اے مریم! بے شک اللہ تمہیں اپنی طرف سے ایک کلمہ کی بشارت دیتا ہے، جس کا نام مسیح عیسیٰ ابن مریم ہے، دنیا اور آخرت میں باعزت ہوگا، اور مقربین میں سے ہوگا۔
وَيُكَلِّمُ ٱلنَّاسَ فِي ٱلْمَهْدِ وَكَهْلٗا وَمِنَ ٱلصَّـٰلِحِينَ ﴿46﴾
اور وہ لوگوں سے گہوارے میں بھی بات کرے گا اور بڑی عمر میں بھی، اور وہ صالحین میں سے ہوگا۔
قَالَتْ رَبِّ أَنَّىٰ يَكُونُ لِي وَلَدٞ وَلَمْ يَمْسَسْنِي بَشَرٞۖ قَالَ كَذَٰلِكِ ٱللَّهُ يَخْلُقُ مَا يَشَآءُۚ إِذَا قَضَىٰٓ أَمْرٗا فَإِنَّمَا يَقُولُ لَهُۥ كُن فَيَكُونُ ﴿47﴾
مریم نے کہا: اے میرے رب! میرے لیے بچہ کیسے ہوگا جبکہ مجھے کسی انسان نے چھوا بھی نہیں؟ فرمایا: اسی طرح اللہ جو چاہے پیدا کرتا ہے، جب وہ کسی کام کا فیصلہ کرتا ہے تو کہتا ہے: "ہو جا" تو وہ ہو جاتا ہے۔
وَيُعَلِّمُهُ ٱلْكِتَـٰبَ وَٱلْحِكْمَةَ وَٱلتَّوْرَىٰةَ وَٱلْإِنجِيلَ ﴿48﴾
اور وہ اسے کتاب، حکمت، تورات اور انجیل سکھائے گا۔
وَرَسُولًا إِلَىٰ بَنِيٓ إِسْرَٰٓءِيلَ أَنِّي قَدْ جِئْتُكُم بِـَٔايَةٖ مِّن رَّبِّكُمْۖ أَنِّيٓ أَخْلُقُ لَكُم مِّنَ ٱلطِّينِ كَهَيْـَٔةِ ٱلطَّيْرِ فَأَنفُخُ فِيهِ فَيَكُونُ طَيْرَۢا بِإِذْنِ ٱللَّهِۖ وَأُبْرِئُ ٱلْأَكْمَهَ وَٱلْأَبْرَصَ وَأُحْيِ ٱلْمَوْتَىٰ بِإِذْنِ ٱللَّهِۖ وَأُنَبِّئُكُم بِمَا تَأْكُلُونَ وَمَا تَدَّخِرُونَ فِي بُيُوتِكُمۚ إِنَّ فِي ذَٰلِكَ لَـَٔايَةٗ لَّكُمْ إِن كُنتُم مُّؤْمِنِينَ ﴿49﴾
اور وہ بنی اسرائیل کی طرف رسول ہوگا، کہ میں تمہارے پاس تمہارے رب کی طرف سے نشانی لے کر آیا ہوں، میں تمہارے لیے مٹی سے پرندے کی شکل بناتا ہوں، پھر اس میں پھونک مارتا ہوں تو وہ اللہ کے حکم سے پرندہ بن جاتا ہے، اور میں مادرزاد اندھے اور کوڑھی کو اچھا کرتا ہوں، اور مردوں کو زندہ کرتا ہوں اللہ کے حکم سے، اور میں تمہیں بتاتا ہوں جو تم کھاتے ہو اور جو اپنے گھروں میں ذخیرہ کرتے ہو، بے شک اس میں تمہارے لیے نشانی ہے اگر تم ایمان رکھتے ہو۔
وَمُصَدِّقٗا لِّمَا بَيْنَ يَدَيَّ مِنَ ٱلتَّوْرَىٰةِ وَلِأُحِلَّ لَكُم بَعْضَ ٱلَّذِي حُرِّمَ عَلَيْكُمۚ وَجِئْتُكُم بِـَٔايَةٖ مِّن رَّبِّكُمْ فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿50﴾
اور میں اس تورات کی تصدیق کرنے والا ہوں جو میرے سامنے ہے، اور تاکہ میں تمہارے لیے بعض وہ چیزیں حلال کر دوں جو تم پر حرام کی گئی تھیں، اور میں تمہارے پاس تمہارے رب کی طرف سے نشانی لے کر آیا ہوں، پس اللہ سے ڈرو اور میری اطاعت کرو۔
إِنَّ ٱللَّهَ رَبِّي وَرَبُّكُمْ فَٱعْبُدُوهُۚ هَـٰذَا صِرَٰطٞ مُّسْتَقِيمٞ ﴿51﴾
بے شک اللہ میرا رب بھی ہے اور تمہارا رب بھی، پس اسی کی عبادت کرو، یہی سیدھا راستہ ہے۔
فَلَمَّآ أَحَسَّ عِيسَىٰ مِنْهُمُ ٱلْكُفْرَ قَالَ مَنْ أَنصَارِيٓ إِلَى ٱللَّهِۚ قَالَ ٱلْحَوَارِيُّونَ نَحْنُ أَنصَارُ ٱللَّهِۖ ءَامَنَّا بِٱللَّهِ وَٱشْهَدْ بِأَنَّا مُسْلِمُونَ ﴿52﴾
پھر جب عیسیٰ نے ان کی طرف سے کفر محسوس کیا تو کہا: کون ہے جو اللہ کی طرف میرا مددگار ہو؟ حواریوں نے کہا: ہم اللہ کے مددگار ہیں، ہم اللہ پر ایمان لائے، اور آپ گواہ رہیں کہ ہم مسلمان ہیں۔
رَبَّنَآ ءَامَنَّا بِمَآ أَنزَلْتَ وَٱتَّبَعْنَا ٱلرَّسُولَ فَٱكْتُبْنَا مَعَ ٱلشَّـٰهِدِينَ ﴿53﴾
اے ہمارے رب! ہم اس پر ایمان لائے جو تو نے نازل فرمایا، اور ہم نے رسول کی پیروی کی، پس ہمیں گواہی دینے والوں میں لکھ لے۔
وَمَكَرُوا۟ وَمَكَرَ ٱللَّهُۖ وَٱللَّهُ خَيْرُ ٱلْمَـٰكِرِينَ ﴿54﴾
اور انہوں نے چال چلی، اور اللہ نے بھی چال چلی، اور اللہ سب چال چلنے والوں سے بہتر ہے۔
إِذْ قَالَ ٱللَّهُ يَـٰعِيسَىٰٓ إِنِّي مُتَوَفِّيكَ وَرَافِعُكَ إِلَيَّ وَمُطَهِّرُكَ مِنَ ٱلَّذِينَ كَفَرُوا۟ وَجَاعِلُ ٱلَّذِينَ ٱتَّبَعُوكَ فَوْقَ ٱلَّذِينَ كَفَرُوٓا۟ إِلَىٰ يَوْمِ ٱلْقِيَـٰمَةِۖ ثُمَّ إِلَيَّ مَرْجِعُكُمْ فَأَحْكُمُ بَيْنَكُمْ فِيمَا كُنتُمْ فِيهِ تَخْتَلِفُونَ ﴿55﴾
جب اللہ نے فرمایا: اے عیسیٰ! میں تجھے وفات دینے والا ہوں اور اپنی طرف اٹھانے والا ہوں، اور تجھے کافروں سے پاک کرنے والا ہوں، اور تیرے پیروکاروں کو قیامت کے دن تک کافروں پر غالب کرنے والا ہوں، پھر میری ہی طرف تم سب کو لوٹنا ہے، پھر میں تمہارے درمیان ان باتوں کا فیصلہ کروں گا جن میں تم اختلاف کرتے تھے۔
فَأَمَّا ٱلَّذِينَ كَفَرُوا۟ فَأُعَذِّبُهُمْ عَذَابٗا شَدِيدٗا فِي ٱلدُّنْيَا وَٱلْـَٔاخِرَةِۖ وَمَا لَهُم مِّن نَّـٰصِرِينَ ﴿56﴾
پھر جو لوگ کفر کریں گے، میں انہیں دنیا اور آخرت میں سخت عذاب دوں گا، اور ان کا کوئی مددگار نہ ہوگا۔
وَأَمَّا ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ فَيُوَفِّيهِمْ أُجُورَهُمْۗ وَٱللَّهُ لَا يُحِبُّ ٱلظَّـٰلِمِينَ ﴿57﴾
اور جو لوگ ایمان لائے اور نیک عمل کیے، اللہ انہیں ان کا پورا اجر دے گا، اور اللہ ظالموں کو پسند نہیں کرتا۔
ذَٰلِكَ نَتْلُوهُ عَلَيْكَ مِنَ ٱلْـَٔايَـٰتِ وَٱلذِّكْرِ ٱلْحَكِيمِ ﴿58﴾
یہ ہم آپ پر آیات اور حکمت والی نصیحت میں سے پڑھ کر سناتے ہیں۔
إِنَّ مَثَلَ عِيسَىٰ عِندَ ٱللَّهِ كَمَثَلِ ءَادَمَۖ خَلَقَهُۥ مِن تُرَابٖ ثُمَّ قَالَ لَهُۥ كُن فَيَكُونُ ﴿59﴾
بے شک عیسیٰ کی مثال اللہ کے نزدیک آدم کی مثال کی طرح ہے، اسے مٹی سے پیدا کیا، پھر اسے کہا: "ہو جا" تو وہ ہو گیا۔
ٱلْحَقُّ مِن رَّبِّكَ فَلَا تَكُن مِّنَ ٱلْمُمْتَرِينَ ﴿60﴾
یہ حق ہے تیرے رب کی طرف سے، پس تو شک کرنے والوں میں سے نہ ہو۔
فَمَنْ حَآجَّكَ فِيهِ مِنۢ بَعْدِ مَا جَآءَكَ مِنَ ٱلْعِلْمِ فَقُلْ تَعَالَوْا۟ نَدْعُ أَبْنَآءَنَا وَأَبْنَآءَكُمْ وَنِسَآءَنَا وَنِسَآءَكُمْ وَأَنفُسَنَا وَأَنفُسَكُمْ ثُمَّ نَبْتَهِلْ فَنَجْعَل لَّعْنَتَ ٱللَّهِ عَلَى ٱلْكَـٰذِبِينَ ﴿61﴾
پھر جو کوئی اس کے بارے میں آپ سے جھگڑا کرے بعد اس کے کہ آپ کے پاس علم آ چکا ہو، تو کہہ دیجیے: آؤ ہم اپنے بیٹوں کو اور تمہارے بیٹوں کو، اپنی عورتوں کو اور تمہاری عورتوں کو، اپنے آپ کو اور تمہارے آپ کو بلائیں، پھر دعا کریں اور جھوٹوں پر اللہ کی لعنت ڈالیں۔
إِنَّ هَـٰذَا لَهُوَ ٱلْقَصَصُ ٱلْحَقُّۚ وَمَا مِنْ إِلَـٰهٍ إِلَّا ٱللَّهُۚ وَإِنَّ ٱللَّهَ لَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿62﴾
بے شک یہ سچا بیان ہے، اور اللہ کے سوا کوئی معبود نہیں، اور بے شک اللہ ہی زبردست، حکمت والا ہے۔
فَإِن تَوَلَّوْا۟ فَإِنَّ ٱللَّهَ عَلِيمُۢ بِٱلْمُفْسِدِينَ ﴿63﴾
پھر اگر وہ منہ موڑیں تو بے شک اللہ فساد کرنے والوں کو خوب جانتا ہے۔
قُلْ يَـٰٓأَهْلَ ٱلْكِتَـٰبِ تَعَالَوْا۟ إِلَىٰ كَلِمَةٖ سَوَآءِۭ بَيْنَنَا وَبَيْنَكُمْ أَلَّا نَعْبُدَ إِلَّا ٱللَّهَ وَلَا نُشْرِكَ بِهِۦ شَيْـٔٗا وَلَا يَتَّخِذَ بَعْضُنَا بَعْضًا أَرْبَابٗا مِّن دُونِ ٱللَّهِۚ فَإِن تَوَلَّوْا۟ فَقُولُوا۟ ٱشْهَدُوا۟ بِأَنَّا مُسْلِمُونَ ﴿64﴾
کہہ دیجیے: اے اہلِ کتاب! آؤ ایک ایسی بات کی طرف جو ہمارے اور تمہارے درمیان یکساں ہے، کہ ہم اللہ کے سوا کسی کی عبادت نہ کریں، اور اس کے ساتھ کسی کو شریک نہ ٹھہرائیں، اور ہم میں سے کوئی دوسرے کو اللہ کے سوا رب نہ بنائے، پھر اگر وہ منہ موڑیں تو کہہ دیجیے: گواہ رہو کہ ہم مسلمان ہیں۔
يَـٰٓأَهْلَ ٱلْكِتَـٰبِ لِمَ تُحَآجُّونَ فِيٓ إِبْرَٰهِيمَ وَمَآ أُنزِلَتِ ٱلتَّوْرَىٰةُ وَٱلْإِنجِيلُ إِلَّا مِنۢ بَعْدِهِۦٓۚ أَفَلَا تَعْقِلُونَ ﴿65﴾
اے اہلِ کتاب! تم ابراہیم کے بارے میں کیوں جھگڑتے ہو، حالانکہ تورات اور انجیل ان کے بعد ہی نازل کی گئی تھیں؟ کیا تم عقل نہیں رکھتے؟
هَـٰٓأَنتُمْ هَـٰٓؤُلَآءِ حَـٰجَجْتُمْ فِيمَا لَكُم بِهِۦ عِلْمٞ فَلِمَ تُحَآجُّونَ فِيمَا لَيْسَ لَكُم بِهِۦ عِلْمٞۚ وَٱللَّهُ يَعْلَمُ وَأَنتُمْ لَا تَعْلَمُونَ ﴿66﴾
تم وہی لوگ ہو جو ان باتوں میں جھگڑ چکے جن کا تمہیں کچھ علم تھا، پھر ان باتوں میں کیوں جھگڑتے ہو جن کا تمہیں کچھ بھی علم نہیں؟ اور اللہ جانتا ہے، تم نہیں جانتے۔
مَا كَانَ إِبْرَٰهِيمُ يَهُودِيّٗا وَلَا نَصْرَانِيّٗا وَلَـٰكِن كَانَ حَنِيفٗا مُّسْلِمٗاۚ وَمَا كَانَ مِنَ ٱلْمُشْرِكِينَ ﴿67﴾
ابراہیم نہ یہودی تھے اور نہ نصرانی، بلکہ وہ ایک یکسو مسلمان تھے، اور وہ مشرکوں میں سے نہ تھے۔
إِنَّ أَوْلَى ٱلنَّاسِ بِإِبْرَٰهِيمَ لَلَّذِينَ ٱتَّبَعُوهُ وَهَـٰذَا ٱلنَّبِيُّ وَٱلَّذِينَ ءَامَنُوا۟ۗ وَٱللَّهُ وَلِيُّ ٱلْمُؤْمِنِينَ ﴿68﴾
بے شک ابراہیم سے نسبت رکھنے کے سب سے زیادہ حق دار وہ لوگ ہیں جنہوں نے ان کی پیروی کی، اور یہ نبی اور وہ لوگ جو ایمان لائے، اور اللہ ایمان والوں کا دوست ہے۔
وَدَّت طَّآئِفَةٞ مِّنْ أَهْلِ ٱلْكِتَـٰبِ لَوْ يُضِلُّونَكُمۖ وَمَا يُضِلُّونَ إِلَّآ أَنفُسَهُمۖ وَمَا يَشْعُرُونَ ﴿69﴾
اہلِ کتاب میں سے ایک گروہ چاہتا ہے کہ تمہیں گمراہ کر دے، حالانکہ وہ صرف اپنے آپ کو گمراہ کر رہے ہیں، اور وہ شعور نہیں رکھتے۔
يَـٰٓأَهْلَ ٱلْكِتَـٰبِ لِمَ تَكْفُرُونَ بِـَٔايَـٰتِ ٱللَّهِ وَأَنتُمْ تَشْهَدُونَ ﴿70﴾
اے اہلِ کتاب! تم اللہ کی آیات کا انکار کیوں کرتے ہو حالانکہ تم خود گواہی دیتے ہو؟
يَـٰٓأَهْلَ ٱلْكِتَـٰبِ لِمَ تَلْبِسُونَ ٱلْحَقَّ بِٱلْبَـٰطِلِ وَتَكْتُمُونَ ٱلْحَقَّ وَأَنتُمْ تَعْلَمُونَ ﴿71﴾
اے اہلِ کتاب! تم حق کو باطل کے ساتھ کیوں ملاتے ہو اور حق کو چھپاتے ہو حالانکہ تم جانتے ہو؟
وَقَالَت طَّآئِفَةٞ مِّنْ أَهْلِ ٱلْكِتَـٰبِ ءَامِنُوا۟ بِٱلَّذِيٓ أُنزِلَ عَلَى ٱلَّذِينَ ءَامَنُوا۟ وَجْهَ ٱلنَّهَارِ وَٱكْفُرُوٓا۟ ءَاخِرَهُۥ لَعَلَّهُمْ يَرْجِعُونَ ﴿72﴾
اور اہلِ کتاب میں سے ایک گروہ نے کہا: دن کے آغاز میں اس چیز پر ایمان لاؤ جو ایمان والوں پر نازل ہوئی ہے، اور دن کے آخر میں انکار کرو، شاید وہ (ایمان والے) پلٹ جائیں۔
وَلَا تُؤْمِنُوٓا۟ إِلَّا لِمَن تَبِعَ دِينَكُمۖ قُلْ إِنَّ ٱلْهُدَىٰ هُدَى ٱللَّهِۗ أَن يُؤْتَىٰٓ أَحَدٞ مِّثْلَ مَآ أُوتِيتُمْ أَوْ يُحَآجُّوكُمْ عِندَ رَبِّكُمۗ قُلْ إِنَّ ٱلْفَضْلَ بِيَدِ ٱللَّهِ يُؤْتِيهِ مَن يَشَآءُۗ وَٱللَّهُ وَٰسِعٌ عَلِيمٞ ﴿73﴾
اور (انہوں نے کہا:) اپنے دین کے پیروکاروں کے سوا کسی پر ایمان نہ لاؤ، کہہ دیجیے: بے شک ہدایت تو اللہ کی ہدایت ہے، (اور یہ بھی کہا:) کہ کسی کو وہی کچھ دیا جائے جو تمہیں دیا گیا، یا وہ تمہارے رب کے پاس تم سے جھگڑا کرے، کہہ دیجیے: بے شک فضل اللہ کے ہاتھ میں ہے، وہ جسے چاہے عطا کرتا ہے، اور اللہ وسعت والا، جاننے والا ہے۔
يَخْتَصُّ بِرَحْمَتِهِۦ مَن يَشَآءُۚ وَٱللَّهُ ذُو ٱلْفَضْلِ ٱلْعَظِيمِ ﴿74﴾
وہ اپنی رحمت کے لیے جسے چاہے خاص کر لیتا ہے، اور اللہ بڑے فضل والا ہے۔
وَمِنْ أَهْلِ ٱلْكِتَـٰبِ مَنْ إِن تَأْمَنْهُ بِقِنطَارٖ يُؤَدِّهِۦٓ إِلَيْكَ وَمِنْهُم مَّنْ إِن تَأْمَنْهُ بِدِينَارٖ لَّا يُؤَدِّهِۦٓ إِلَيْكَ إِلَّا مَا دُمْتَ عَلَيْهِ قَآئِمٗاۗ ذَٰلِكَ بِأَنَّهُمْ قَالُوا۟ لَيْسَ عَلَيْنَا فِي ٱلْأُمِّيِّـۧنَ سَبِيلٞ وَيَقُولُونَ عَلَى ٱللَّهِ ٱلْكَذِبَ وَهُمْ يَعْلَمُونَ ﴿75﴾
اور اہلِ کتاب میں سے کچھ ایسے ہیں کہ اگر تم انہیں خزانے کا امین بناؤ تو وہ تمہیں واپس کر دیں گے، اور ان میں سے کچھ ایسے بھی ہیں کہ اگر تم انہیں ایک دینار کا امین بناؤ تو وہ تمہیں واپس نہ کریں گے، مگر جب تک تم ان کے سر پر کھڑے رہو، یہ اس لیے کہ وہ کہتے ہیں: ہم پر اَن پڑھ لوگوں کے بارے میں کوئی گناہ نہیں، اور وہ اللہ پر جھوٹ بولتے ہیں حالانکہ وہ جانتے ہیں۔
بَلَىٰۚ مَنْ أَوْفَىٰ بِعَهْدِهِۦ وَٱتَّقَىٰ فَإِنَّ ٱللَّهَ يُحِبُّ ٱلْمُتَّقِينَ ﴿76﴾
کیوں نہیں! جو شخص اپنا وعدہ پورا کرے اور تقویٰ اختیار کرے تو بے شک اللہ متقیوں سے محبت کرتا ہے۔
إِنَّ ٱلَّذِينَ يَشْتَرُونَ بِعَهْدِ ٱللَّهِ وَأَيْمَـٰنِهِمْ ثَمَنٗا قَلِيلًا أُو۟لَـٰٓئِكَ لَا خَلَـٰقَ لَهُمْ فِي ٱلْـَٔاخِرَةِ وَلَا يُكَلِّمُهُمُ ٱللَّهُ وَلَا يَنظُرُ إِلَيْهِمْ يَوْمَ ٱلْقِيَـٰمَةِ وَلَا يُزَكِّيهِمْۚ وَلَهُمْ عَذَابٌ أَلِيمٞ ﴿77﴾
بے شک جو لوگ اللہ کے عہد اور اپنی قسموں کے بدلے تھوڑی قیمت لیتے ہیں، ان کا آخرت میں کوئی حصہ نہیں، اور اللہ نہ ان سے بات کرے گا، نہ ان کی طرف دیکھے گا قیامت کے دن، اور نہ انہیں پاک کرے گا، اور ان کے لیے دردناک عذاب ہے۔
وَإِنَّ مِنْهُمْ لَفَرِيقٗا يَلْوُۥنَ أَلْسِنَتَهُم بِٱلْكِتَـٰبِ لِتَحْسَبُوهُ مِنَ ٱلْكِتَـٰبِ وَمَا هُوَ مِنَ ٱلْكِتَـٰبِ وَيَقُولُونَ هُوَ مِنْ عِندِ ٱللَّهِ وَمَا هُوَ مِنْ عِندِ ٱللَّهِۖ وَيَقُولُونَ عَلَى ٱللَّهِ ٱلْكَذِبَ وَهُمْ يَعْلَمُونَ ﴿78﴾
اور بے شک ان میں ایک گروہ ایسا بھی ہے جو کتاب کو پڑھتے ہوئے اپنی زبانوں کو مروڑتا ہے تاکہ تم اسے کتاب کا حصہ سمجھو، حالانکہ وہ کتاب کا حصہ نہیں، اور وہ کہتے ہیں: یہ اللہ کی طرف سے ہے، حالانکہ وہ اللہ کی طرف سے نہیں، اور وہ اللہ پر جھوٹ بولتے ہیں حالانکہ وہ جانتے ہیں۔
مَا كَانَ لِبَشَرٍ أَن يُؤْتِيَهُ ٱللَّهُ ٱلْكِتَـٰبَ وَٱلْحُكْمَ وَٱلنُّبُوَّةَ ثُمَّ يَقُولَ لِلنَّاسِ كُونُوا۟ عِبَادٗا لِّي مِن دُونِ ٱللَّهِ وَلَـٰكِن كُونُوا۟ رَبَّـٰنِيِّـۧنَ بِمَا كُنتُمْ تُعَلِّمُونَ ٱلْكِتَـٰبَ وَبِمَا كُنتُمْ تَدْرُسُونَ ﴿79﴾
کسی بشر کے لیے جائز نہیں کہ اللہ اسے کتاب، حکم اور نبوت دے، پھر وہ لوگوں سے کہے: اللہ کو چھوڑ کر میرے بندے بن جاؤ، بلکہ (وہ کہے گا:) رب والے بن جاؤ، اس وجہ سے کہ تم کتاب سکھاتے ہو اور اس کا درس دیتے ہو۔
وَلَا يَأْمُرَكُمْ أَن تَتَّخِذُوا۟ ٱلْمَلَـٰٓئِكَةَ وَٱلنَّبِيِّـۧنَ أَرْبَابًاۗ أَيَأْمُرُكُم بِٱلْكُفْرِ بَعْدَ إِذْ أَنتُم مُّسْلِمُونَ ﴿80﴾
اور نہ وہ تمہیں یہ حکم دے گا کہ فرشتوں اور نبیوں کو رب بنا لو، کیا وہ تمہیں کفر کا حکم دے گا بعد اس کے کہ تم مسلمان ہو چکے ہو؟
وَإِذْ أَخَذَ ٱللَّهُ مِيثَـٰقَ ٱلنَّبِيِّـۧنَ لَمَآ ءَاتَيْتُكُم مِّن كِتَـٰبٖ وَحِكْمَةٖ ثُمَّ جَآءَكُمْ رَسُولٞ مُّصَدِّقٞ لِّمَا مَعَكُمْ لَتُؤْمِنُنَّ بِهِۦ وَلَتَنصُرُنَّهُۚ قَالَ ءَأَقْرَرْتُمْ وَأَخَذْتُمْ عَلَىٰ ذَٰلِكُمْ إِصْرِيۖ قَالُوٓا۟ أَقْرَرْنَاۚ قَالَ فَٱشْهَدُوا۟ وَأَنَا۠ مَعَكُم مِّنَ ٱلشَّـٰهِدِينَ ﴿81﴾
اور جب اللہ نے نبیوں سے عہد لیا کہ: جو کچھ میں تمہیں کتاب اور حکمت دوں، پھر تمہارے پاس کوئی رسول آئے جو اس کی تصدیق کرے جو تمہارے پاس ہے، تو تم ضرور اس پر ایمان لاؤ گے اور ضرور اس کی مدد کرو گے۔ فرمایا: کیا تم نے اقرار کیا اور اس پر میرا بھاری عہد لیا؟ انہوں نے کہا: ہم نے اقرار کیا۔ فرمایا: تو گواہ رہو، اور میں بھی تمہارے ساتھ گواہوں میں سے ہوں۔
فَمَن تَوَلَّىٰ بَعْدَ ذَٰلِكَ فَأُو۟لَـٰٓئِكَ هُمُ ٱلْفَـٰسِقُونَ ﴿82﴾
پھر جو اس کے بعد منہ موڑے تو وہی لوگ فاسق ہیں۔
أَفَغَيْرَ دِينِ ٱللَّهِ يَبْغُونَ وَلَهُۥٓ أَسْلَمَ مَن فِي ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ طَوْعٗا وَكَرْهٗا وَإِلَيْهِ يُرْجَعُونَ ﴿83﴾
کیا وہ اللہ کے دین کے سوا کچھ اور چاہتے ہیں؟ حالانکہ جو کچھ آسمانوں اور زمین میں ہے سب اس کے تابع ہے، خوشی سے یا مجبوری سے، اور اسی کی طرف وہ لوٹائے جائیں گے۔
قُلْ ءَامَنَّا بِٱللَّهِ وَمَآ أُنزِلَ عَلَيْنَا وَمَآ أُنزِلَ عَلَىٰٓ إِبْرَٰهِيمَ وَإِسْمَـٰعِيلَ وَإِسْحَـٰقَ وَيَعْقُوبَ وَٱلْأَسْبَاطِ وَمَآ أُوتِيَ مُوسَىٰ وَعِيسَىٰ وَٱلنَّبِيُّونَ مِن رَّبِّهِمْ لَا نُفَرِّقُ بَيْنَ أَحَدٖ مِّنْهُمْ وَنَحْنُ لَهُۥ مُسْلِمُونَ ﴿84﴾
کہہ دیجیے: ہم ایمان لائے اللہ پر، اور اس پر جو ہم پر نازل کیا گیا، اور جو ابراہیم، اسماعیل، اسحاق، یعقوب اور اولادِ یعقوب پر نازل کیا گیا، اور جو موسیٰ، عیسیٰ اور دیگر نبیوں کو ان کے رب کی طرف سے دیا گیا، ہم ان میں سے کسی کے درمیان فرق نہیں کرتے، اور ہم اسی کے فرمانبردار ہیں۔
وَمَن يَبْتَغِ غَيْرَ ٱلْإِسْلَـٰمِ دِينٗا فَلَن يُقْبَلَ مِنْهُ وَهُوَ فِي ٱلْـَٔاخِرَةِ مِنَ ٱلْخَـٰسِرِينَ ﴿85﴾
اور جو کوئی اسلام کے سوا کوئی اور دین چاہے، تو وہ اس سے ہرگز قبول نہ کیا جائے گا، اور وہ آخرت میں نقصان اٹھانے والوں میں سے ہوگا۔
كَيْفَ يَهْدِي ٱللَّهُ قَوْمٗا كَفَرُوا۟ بَعْدَ إِيمَـٰنِهِمْ وَشَهِدُوٓا۟ أَنَّ ٱلرَّسُولَ حَقّٞ وَجَآءَهُمُ ٱلْبَيِّنَـٰتُۚ وَٱللَّهُ لَا يَهْدِي ٱلْقَوْمَ ٱلظَّـٰلِمِينَ ﴿86﴾
اللہ ایسے لوگوں کو کیسے ہدایت دے گا جنہوں نے ایمان لانے کے بعد کفر کیا، اور گواہی دی کہ رسول حق ہے، اور ان کے پاس واضح نشانیاں آ چکی تھیں؟ اور اللہ ظالم قوم کو ہدایت نہیں دیتا۔
أُو۟لَـٰٓئِكَ جَزَآؤُهُمْ أَنَّ عَلَيْهِمْ لَعْنَةَ ٱللَّهِ وَٱلْمَلَـٰٓئِكَةِ وَٱلنَّاسِ أَجْمَعِينَ ﴿87﴾
ان لوگوں کی جزا یہ ہے کہ ان پر اللہ، فرشتوں اور تمام انسانوں کی لعنت ہے۔
خَـٰلِدِينَ فِيهَا لَا يُخَفَّفُ عَنْهُمُ ٱلْعَذَابُ وَلَا هُمْ يُنظَرُونَ ﴿88﴾
وہ ہمیشہ اس (لعنت) میں رہیں گے، ان سے عذاب ہلکا نہ کیا جائے گا اور نہ انہیں مہلت دی جائے گی۔
إِلَّا ٱلَّذِينَ تَابُوا۟ مِنۢ بَعْدِ ذَٰلِكَ وَأَصْلَحُوا۟ فَإِنَّ ٱللَّهَ غَفُورٞ رَّحِيمٌ ﴿89﴾
سوائے ان لوگوں کے جنہوں نے اس کے بعد توبہ کی اور اصلاح کی، تو بے شک اللہ بخشنے والا، مہربان ہے۔
إِنَّ ٱلَّذِينَ كَفَرُوا۟ بَعْدَ إِيمَـٰنِهِمْ ثُمَّ ٱزْدَادُوا۟ كُفْرٗا لَّن تُقْبَلَ تَوْبَتُهُمْۗ وَأُو۟لَـٰٓئِكَ هُمُ ٱلضَّآلُّونَ ﴿90﴾
بے شک جنہوں نے ایمان کے بعد کفر کیا، پھر کفر میں بڑھتے گئے، ان کی توبہ ہرگز قبول نہ کی جائے گی، اور وہی لوگ گمراہ ہیں۔
إِنَّ ٱلَّذِينَ كَفَرُوا۟ وَمَاتُوا۟ وَهُمْ كُفَّارٞ فَلَن يُقْبَلَ مِنْ أَحَدِهِم مِّلْءُ ٱلْأَرْضِ ذَهَبٗا وَلَوِ ٱفْتَدَىٰ بِهِۦٓۚ أُو۟لَـٰٓئِكَ لَهُمْ عَذَابٌ أَلِيمٞ وَمَا لَهُم مِّن نَّـٰصِرِينَ ﴿91﴾
بے شک جنہوں نے کفر کیا اور اسی حالت میں مر گئے، ان میں سے کسی سے زمین بھر سونا بھی قبول نہ کیا جائے گا، اگرچہ وہ اس کے ذریعے فدیہ دینا چاہے، ان کے لیے دردناک عذاب ہے، اور ان کا کوئی مددگار نہ ہوگا۔
لَن تَنَالُوا۟ ٱلْبِرَّ حَتَّىٰ تُنفِقُوا۟ مِمَّا تُحِبُّونَۚ وَمَا تُنفِقُوا۟ مِن شَيْءٖ فَإِنَّ ٱللَّهَ بِهِۦ عَلِيمٞ ﴿92﴾
تم ہرگز نیکی کو نہ پہنچو گے جب تک اپنی پسندیدہ چیزوں میں سے خرچ نہ کرو، اور جو کچھ تم خرچ کرو، بے شک اللہ اسے خوب جانتا ہے۔
كُلُّ ٱلطَّعَامِ كَانَ حِلّٗا لِّبَنِيٓ إِسْرَٰٓءِيلَ إِلَّا مَا حَرَّمَ إِسْرَٰٓءِيلُ عَلَىٰ نَفْسِهِۦ مِن قَبْلِ أَن تُنَزَّلَ ٱلتَّوْرَىٰةُۚ قُلْ فَأْتُوا۟ بِٱلتَّوْرَىٰةِ فَٱتْلُوهَآ إِن كُنتُمْ صَـٰدِقِينَ ﴿93﴾
تمام کھانے بنی اسرائیل کے لیے حلال تھے، سوائے ان کے جو اسرائیل نے اپنے اوپر خود حرام کر لیے تھے، تورات کے نازل ہونے سے پہلے۔ کہہ دیجیے: تورات لاؤ اور اسے پڑھو اگر تم سچے ہو۔
فَمَنِ ٱفْتَرَىٰ عَلَى ٱللَّهِ ٱلْكَذِبَ مِنۢ بَعْدِ ذَٰلِكَ فَأُو۟لَـٰٓئِكَ هُمُ ٱلظَّـٰلِمُونَ ﴿94﴾
پھر جو اس کے بعد اللہ پر جھوٹ باندھے، وہی لوگ ظالم ہیں۔
قُلْ صَدَقَ ٱللَّهُ فَٱتَّبِعُوا۟ مِلَّةَ إِبْرَٰهِيمَ حَنِيفٗاۚ وَمَا كَانَ مِنَ ٱلْمُشْرِكِينَ ﴿95﴾
کہہ دیجیے: اللہ نے سچ فرمایا، پس ابراہیم کے دین کی پیروی کرو جو یکسو تھے، اور وہ مشرکوں میں سے نہ تھے۔
إِنَّ أَوَّلَ بَيْتٖ وُضِعَ لِلنَّاسِ لَلَّذِي بِبَكَّةَ مُبَارَكٗا وَهُدٗى لِّلْعَـٰلَمِينَ ﴿96﴾
بے شک سب سے پہلا گھر جو لوگوں کے لیے مقرر کیا گیا وہی ہے جو مکہ میں ہے، برکت والا اور تمام جہان والوں کے لیے ہدایت ہے۔
فِيهِ ءَايَـٰتُۢ بَيِّنَـٰتٞ مَّقَامُ إِبْرَٰهِيمَۖ وَمَن دَخَلَهُۥ كَانَ ءَامِنٗاۗ وَلِلَّهِ عَلَى ٱلنَّاسِ حِجُّ ٱلْبَيْتِ مَنِ ٱسْتَطَاعَ إِلَيْهِ سَبِيلٗاۚ وَمَن كَفَرَ فَإِنَّ ٱللَّهَ غَنِيٌّ عَنِ ٱلْعَـٰلَمِينَ ﴿97﴾
اس میں کھلی نشانیاں ہیں، مقامِ ابراہیم ہے، اور جو اس میں داخل ہوا وہ امن میں ہے، اور اللہ کے لیے لوگوں پر اس گھر کا حج فرض ہے جو اس تک پہنچنے کی استطاعت رکھتا ہو، اور جو انکار کرے تو بے شک اللہ تمام جہانوں سے بے نیاز ہے۔
قُلْ يَـٰٓأَهْلَ ٱلْكِتَـٰبِ لِمَ تَكْفُرُونَ بِـَٔايَـٰتِ ٱللَّهِ وَٱللَّهُ شَهِيدٌ عَلَىٰ مَا تَعْمَلُونَ ﴿98﴾
کہہ دیجیے: اے اہلِ کتاب! تم اللہ کی آیات کا انکار کیوں کرتے ہو، حالانکہ اللہ تمہارے اعمال پر گواہ ہے؟
قُلْ يَـٰٓأَهْلَ ٱلْكِتَـٰبِ لِمَ تَصُدُّونَ عَن سَبِيلِ ٱللَّهِ مَنْ ءَامَنَ تَبْغُونَهَا عِوَجٗا وَأَنتُمْ شُهَدَآءُۚ وَمَا ٱللَّهُ بِغَـٰفِلٍ عَمَّا تَعْمَلُونَ ﴿99﴾
کہہ دیجیے: اے اہلِ کتاب! تم ایمان لانے والوں کو اللہ کے راستے سے کیوں روکتے ہو، اس میں کجی تلاش کرتے ہو حالانکہ تم گواہ ہو؟ اور اللہ تمہارے اعمال سے غافل نہیں۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ إِن تُطِيعُوا۟ فَرِيقٗا مِّنَ ٱلَّذِينَ أُوتُوا۟ ٱلْكِتَـٰبَ يَرُدُّوكُم بَعْدَ إِيمَـٰنِكُمْ كَـٰفِرِينَ ﴿100﴾
اے ایمان والو! اگر تم اہلِ کتاب میں سے ایک گروہ کی بات مانو گے تو وہ تمہیں ایمان کے بعد کفر کی طرف لوٹا دیں گے۔
وَكَيْفَ تَكْفُرُونَ وَأَنتُمْ تُتْلَىٰ عَلَيْكُمْ ءَايَـٰتُ ٱللَّهِ وَفِيكُمْ رَسُولُهُۥۗ وَمَن يَعْتَصِمۢ بِٱللَّهِ فَقَدْ هُدِيَ إِلَىٰ صِرَٰطٖ مُّسْتَقِيمٖ ﴿101﴾
اور تم کیسے کفر کرو گے حالانکہ تم پر اللہ کی آیات پڑھی جاتی ہیں اور تم میں اس کا رسول موجود ہے؟ اور جو اللہ کو مضبوطی سے تھام لے تو وہ سیدھے راستے کی طرف ہدایت پا گیا۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ ٱتَّقُوا۟ ٱللَّهَ حَقَّ تُقَاتِهِۦ وَلَا تَمُوتُنَّ إِلَّا وَأَنتُم مُّسْلِمُونَ ﴿102﴾
اے ایمان والو! اللہ سے ڈرو جیسا کہ اس سے ڈرنے کا حق ہے، اور تمہیں موت نہ آئے مگر اس حال میں کہ تم مسلمان ہو۔
وَٱعْتَصِمُوا۟ بِحَبْلِ ٱللَّهِ جَمِيعٗا وَلَا تَفَرَّقُواۚ وَٱذْكُرُوا۟ نِعْمَتَ ٱللَّهِ عَلَيْكُمْ إِذْ كُنتُمْ أَعْدَآءٗ فَأَلَّفَ بَيْنَ قُلُوبِكُمْ فَأَصْبَحْتُم بِنِعْمَتِهِۦٓ إِخْوَٰنٗاۖ وَكُنتُمْ عَلَىٰ شَفَا حُفْرَةٖ مِّنَ ٱلنَّارِ فَأَنقَذَكُم مِّنْهَاۗ كَذَٰلِكَ يُبَيِّنُ ٱللَّهُ لَكُمْ ءَايَـٰتِهِۦ لَعَلَّكُمْ تَهْتَدُونَ ﴿103﴾
اور سب مل کر اللہ کی رسی کو مضبوطی سے تھام لو اور تفرقہ نہ ڈالو، اور اللہ کی وہ نعمت یاد کرو جو اس نے تم پر کی، جب تم ایک دوسرے کے دشمن تھے تو اس نے تمہارے دلوں میں الفت ڈال دی، پھر تم اس کی نعمت سے بھائی بھائی بن گئے، اور تم آگ کے گڑھے کے کنارے پر تھے تو اس نے تمہیں اس سے بچا لیا، اسی طرح اللہ تمہارے لیے اپنی آیات کھول کر بیان کرتا ہے تاکہ تم ہدایت پاؤ۔
وَلْتَكُن مِّنكُمْ أُمَّةٞ يَدْعُونَ إِلَى ٱلْخَيْرِ وَيَأْمُرُونَ بِٱلْمَعْرُوفِ وَيَنْهَوْنَ عَنِ ٱلْمُنكَرِۚ وَأُو۟لَـٰٓئِكَ هُمُ ٱلْمُفْلِحُونَ ﴿104﴾
اور تم میں ایک جماعت ایسی ہونی چاہیے جو بھلائی کی طرف بلائے، نیکی کا حکم دے اور برائی سے روکے، اور وہی لوگ فلاح پانے والے ہیں۔
وَلَا تَكُونُوا۟ كَٱلَّذِينَ تَفَرَّقُوا۟ وَٱخْتَلَفُوا۟ مِنۢ بَعْدِ مَا جَآءَهُمُ ٱلْبَيِّنَـٰتُۚ وَأُو۟لَـٰٓئِكَ لَهُمْ عَذَابٌ عَظِيمٞ ﴿105﴾
اور ان لوگوں کی طرح نہ ہو جانا جو تفرقہ میں پڑے اور اختلاف کرنے لگے بعد اس کے کہ ان کے پاس واضح نشانیاں آ چکی تھیں، اور انہی کے لیے بڑا عذاب ہے۔
يَوْمَ تَبْيَضُّ وُجُوهٞ وَتَسْوَدُّ وُجُوهٞۚ فَأَمَّا ٱلَّذِينَ ٱسْوَدَّتْ وُجُوهُهُمْ أَكَفَرْتُم بَعْدَ إِيمَـٰنِكُمْ فَذُوقُوا۟ ٱلْعَذَابَ بِمَا كُنتُمْ تَكْفُرُونَ ﴿106﴾
جس دن کچھ چہرے روشن ہوں گے اور کچھ چہرے سیاہ، تو جن کے چہرے سیاہ ہوں گے (ان سے کہا جائے گا:) کیا تم نے ایمان کے بعد کفر کیا؟ پس چکھو عذاب کو اس کفر کے بدلے جو تم کرتے تھے۔
وَأَمَّا ٱلَّذِينَ ٱبْيَضَّتْ وُجُوهُهُمْ فَفِي رَحْمَةِ ٱللَّهِ هُمْ فِيهَا خَـٰلِدُونَ ﴿107﴾
اور جن کے چہرے روشن ہوں گے، وہ اللہ کی رحمت میں ہوں گے، وہ اس میں ہمیشہ رہیں گے۔
تِلْكَ ءَايَـٰتُ ٱللَّهِ نَتْلُوهَا عَلَيْكَ بِٱلْحَقِّۚ وَمَا ٱللَّهُ يُرِيدُ ظُلْمٗا لِّلْعَـٰلَمِينَ ﴿108﴾
یہ اللہ کی آیات ہیں جو ہم آپ پر حق کے ساتھ پڑھ کر سناتے ہیں، اور اللہ جہان والوں پر ظلم نہیں چاہتا۔
وَلِلَّهِ مَا فِي ٱلسَّمَـٰوَٰتِ وَمَا فِي ٱلْأَرْضِۚ وَإِلَى ٱللَّهِ تُرْجَعُ ٱلْأُمُورُ ﴿109﴾
اور اللہ ہی کا ہے جو کچھ آسمانوں اور زمین میں ہے، اور تمام امور اللہ ہی کی طرف لوٹائے جاتے ہیں۔
كُنتُمْ خَيْرَ أُمَّةٍ أُخْرِجَتْ لِلنَّاسِ تَأْمُرُونَ بِٱلْمَعْرُوفِ وَتَنْهَوْنَ عَنِ ٱلْمُنكَرِ وَتُؤْمِنُونَ بِٱللَّهِۗ وَلَوْ ءَامَنَ أَهْلُ ٱلْكِتَـٰبِ لَكَانَ خَيْرٗا لَّهُمۚ مِّنْهُمُ ٱلْمُؤْمِنُونَ وَأَكْثَرُهُمُ ٱلْفَـٰسِقُونَ ﴿110﴾
تم بہترین امت ہو جو لوگوں کے لیے نکالی گئی، تم نیکی کا حکم دیتے ہو، برائی سے روکتے ہو، اور اللہ پر ایمان رکھتے ہو، اور اگر اہلِ کتاب ایمان لے آتے تو ان کے لیے بہتر ہوتا، ان میں کچھ ایمان والے ہیں، لیکن ان میں سے اکثر فاسق ہیں۔
لَن يَضُرُّوكُمْ إِلَّآ أَذٗىۖ وَإِن يُقَـٰتِلُوكُمْ يُوَلُّوكُمُ ٱلْأَدْبَارَ ثُمَّ لَا يُنصَرُونَ ﴿111﴾
وہ تمہیں کچھ نقصان نہیں پہنچا سکتے مگر تھوڑی اذیت، اور اگر وہ تم سے لڑیں گے تو تمہارے سامنے پیٹھ پھیر دیں گے، پھر ان کی مدد نہ کی جائے گی۔
ضُرِبَتْ عَلَيْهِمُ ٱلذِّلَّةُ أَيْنَ مَا ثُقِفُوٓا۟ إِلَّا بِحَبْلٖ مِّنَ ٱللَّهِ وَحَبْلٖ مِّنَ ٱلنَّاسِ وَبَآءُو بِغَضَبٖ مِّنَ ٱللَّهِ وَضُرِبَتْ عَلَيْهِمُ ٱلْمَسْكَنَةُۚ ذَٰلِكَ بِأَنَّهُمْ كَانُوا۟ يَكْفُرُونَ بِـَٔايَـٰتِ ٱللَّهِ وَيَقْتُلُونَ ٱلْأَنبِيَآءَ بِغَيْرِ حَقّٖۚ ذَٰلِكَ بِمَا عَصَوا۟ وَّكَانُوا۟ يَعْتَدُونَ ﴿112﴾
ان پر ذلت مسلط کر دی گئی جہاں کہیں بھی پائے جائیں، سوائے اس کے کہ اللہ کی یا لوگوں کی پناہ میں ہوں، اور وہ اللہ کے غضب کے سزاوار ہوئے، اور ان پر محتاجی مسلط کر دی گئی، یہ اس لیے کہ وہ اللہ کی آیات کا انکار کرتے تھے اور نبیوں کو ناحق قتل کرتے تھے، یہ اس سبب سے کہ وہ نافرمانی کرتے اور حد سے بڑھ جاتے تھے۔
لَيْسُوا۟ سَوَآءٗۗ مِّنْ أَهْلِ ٱلْكِتَـٰبِ أُمَّةٞ قَآئِمَةٞ يَتْلُونَ ءَايَـٰتِ ٱللَّهِ ءَانَآءَ ٱلَّيْلِ وَهُمْ يَسْجُدُونَ ﴿113﴾
وہ سب برابر نہیں، اہلِ کتاب میں ایک جماعت ایسی بھی ہے جو حق پر قائم ہے، وہ اللہ کی آیات رات کے وقت پڑھتے ہیں اور سجدہ کرتے ہیں۔
يُؤْمِنُونَ بِٱللَّهِ وَٱلْيَوْمِ ٱلْـَٔاخِرِ وَيَأْمُرُونَ بِٱلْمَعْرُوفِ وَيَنْهَوْنَ عَنِ ٱلْمُنكَرِ وَيُسَـٰرِعُونَ فِي ٱلْخَيْرَٰتِۖ وَأُو۟لَـٰٓئِكَ مِنَ ٱلصَّـٰلِحِينَ ﴿114﴾
وہ اللہ اور یومِ آخرت پر ایمان رکھتے ہیں، نیکی کا حکم دیتے ہیں، برائی سے روکتے ہیں، اور بھلائی کے کاموں میں سبقت کرتے ہیں، اور وہی لوگ صالحین میں سے ہیں۔
وَمَا يَفْعَلُوا۟ مِنْ خَيْرٖ فَلَن يُكْفَرُوهُۚ وَٱللَّهُ عَلِيمُۢ بِٱلْمُتَّقِينَ ﴿115﴾
اور جو کچھ وہ نیکی کریں گے، اس کی ناقدری نہیں کی جائے گی، اور اللہ پرہیزگاروں کو خوب جانتا ہے۔
إِنَّ ٱلَّذِينَ كَفَرُوا۟ لَن تُغْنِيَ عَنْهُمْ أَمْوَٰلُهُمْ وَلَآ أَوْلَـٰدُهُم مِّنَ ٱللَّهِ شَيْـٔٗاۚ وَأُو۟لَـٰٓئِكَ أَصْحَـٰبُ ٱلنَّارِۖ هُمْ فِيهَا خَـٰلِدُونَ ﴿116﴾
بے شک جنہوں نے کفر کیا، ان کے مال اور ان کی اولاد اللہ کے مقابلے میں ان کے کچھ کام نہ آئیں گے، اور وہی لوگ دوزخ والے ہیں، وہ اس میں ہمیشہ رہیں گے۔
مَّثَلُ مَا يُنفِقُونَ فِي هَـٰذِهِ ٱلْحَيَوٰةِ ٱلدُّنْيَا كَمَثَلِ رِيحٖ فِيهَا صِرٌّ أَصَابَتْ حَرْثَ قَوْمٖ ظَلَمُوٓا۟ أَنفُسَهُمْ فَأَهْلَكَتْهُۚ وَمَا ظَلَمَهُمُ ٱللَّهُ وَلَـٰكِنْ أَنفُسَهُمْ يَظْلِمُونَ ﴿117﴾
جو کچھ وہ اس دنیا کی زندگی میں خرچ کرتے ہیں، اس کی مثال ایسی ہے جیسے ایک سردی والی ہوا ہو جس میں سخت ٹھنڈ ہو، جو ایسے لوگوں کی کھیتی پر چلے جنہوں نے اپنی جانوں پر ظلم کیا، تو وہ اسے تباہ کر دے، اور اللہ نے ان پر ظلم نہیں کیا بلکہ وہ خود اپنی جانوں پر ظلم کرتے ہیں۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تَتَّخِذُوا۟ بِطَانَةٗ مِّن دُونِكُمْ لَا يَأْلُونَكُمْ خَبَالٗاۚ وَدُّوا۟ مَا عَنِتُّمۚ قَدْ بَدَتِ ٱلْبَغْضَآءُ مِنْ أَفْوَٰهِهِمْ وَمَا تُخْفِي صُدُورُهُمْ أَكْبَرُۚ قَدْ بَيَّنَّا لَكُمُ ٱلْـَٔايَـٰتِ إِن كُنتُمْ تَعْقِلُونَ ﴿118﴾
اے ایمان والو! اپنے سوا کسی کو رازدار نہ بناؤ، وہ تمہیں نقصان پہنچانے میں کوئی کمی نہیں کرتے، وہ چاہتے ہیں کہ تم تکلیف میں پڑو، ان کے منہ سے دشمنی ظاہر ہو چکی ہے، اور جو کچھ ان کے سینوں میں چھپا ہے وہ اس سے بھی بڑا ہے، ہم نے تمہارے لیے آیات کھول کر بیان کر دی ہیں اگر تم عقل رکھتے ہو۔
هَـٰٓأَنتُمْ أُو۟لَـٰٓئِ تُحِبُّونَهُمْ وَلَا يُحِبُّونَكُمْ وَتُؤْمِنُونَ بِٱلْكِتَـٰبِ كُلِّهِۦۚ وَإِذَا لَقُوكُمْ قَالُوٓا۟ ءَامَنَّا وَإِذَا خَلَوْا۟ عَضُّوا۟ عَلَيْكُمُ ٱلْأَنَامِلَ مِنَ ٱلْغَيْظِۚ قُلْ مُوتُوا۟ بِغَيْظِكُمْۗ إِنَّ ٱللَّهَ عَلِيمُۢ بِذَاتِ ٱلصُّدُورِ ﴿119﴾
تم وہ لوگ ہو جو ان سے محبت کرتے ہو، اور وہ تم سے محبت نہیں کرتے، اور تم تمام کتاب پر ایمان رکھتے ہو، اور جب وہ تم سے ملتے ہیں تو کہتے ہیں: ہم ایمان لائے، اور جب تنہا ہوتے ہیں تو غصے سے اپنی انگلیاں کاٹتے ہیں، کہہ دو: اپنے غصے سے مر جاؤ، بے شک اللہ دلوں کے بھید خوب جانتا ہے۔
إِن تَمْسَسْكُمْ حَسَنَةٞ تَسُؤْهُمْ وَإِن تُصِبْكُمْ سَيِّئَةٞ يَفْرَحُوا۟ بِهَاۖ وَإِن تَصْبِرُوا۟ وَتَتَّقُوا۟ لَا يَضُرُّكُمْ كَيْدُهُمْ شَيْـًٔاۗ إِنَّ ٱللَّهَ بِمَا يَعْمَلُونَ مُحِيطٞ ﴿120﴾
اگر تمہیں کوئی بھلائی پہنچے تو انہیں بُرا لگتا ہے، اور اگر تمہیں کوئی مصیبت پہنچے تو وہ خوش ہوتے ہیں، اور اگر تم صبر کرو اور تقویٰ اختیار کرو تو ان کی چال تمہیں کچھ نقصان نہ دے سکے گی، بے شک اللہ ان کے اعمال کو گھیرے ہوئے ہے۔
وَإِذْ غَدَوْتَ مِنْ أَهْلِكَ تُبَوِّئُ ٱلْمُؤْمِنِينَ مَقَـٰعِدَ لِلْقِتَالِۗ وَٱللَّهُ سَمِيعٌ عَلِيمٌ ﴿121﴾
اور جب آپ صبح سویرے اپنے گھر والوں سے نکلے تاکہ مومنوں کو جنگ کے لیے مورچوں پر مقرر کریں، اور اللہ سب کچھ سننے والا، جاننے والا ہے۔
إِذْ هَمَّت طَّآئِفَتَانِ مِنكُمْ أَن تَفْشَلَا وَٱللَّهُ وَلِيُّهُمَاۗ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُؤْمِنُونَ ﴿122﴾
جب تم میں سے دو گروہ بزدلی کا ارادہ کر چکے تھے، حالانکہ اللہ ان کا مددگار تھا، اور مومنوں کو اللہ ہی پر بھروسا کرنا چاہیے۔
وَلَقَدْ نَصَرَكُمُ ٱللَّهُ بِبَدْرٖ وَأَنتُمْ أَذِلَّةٞۖ فَٱتَّقُوا۟ ٱللَّهَ لَعَلَّكُمْ تَشْكُرُونَ ﴿123﴾
اور بے شک اللہ نے بدر میں تمہاری مدد کی جبکہ تم کمزور تھے، پس اللہ سے ڈرو تاکہ تم شکر گزار بنو۔
إِذْ تَقُولُ لِلْمُؤْمِنِينَ أَلَن يَكْفِيَكُمْ أَن يُمِدَّكُمْ رَبُّكُم بِثَلَـٰثَةِ ءَالَافٖ مِّنَ ٱلْمَلَـٰٓئِكَةِ مُنزَلِينَ ﴿124﴾
جب آپ مومنوں سے کہہ رہے تھے: کیا تمہارے لیے یہ کافی نہیں کہ تمہارا رب تین ہزار نازل کیے گئے فرشتوں سے تمہاری مدد کرے؟
بَلَىٰٓۚ إِن تَصْبِرُوا۟ وَتَتَّقُوا۟ وَيَأْتُوكُم مِّن فَوْرِهِمْ هَـٰذَا يُمْدِدْكُمْ رَبُّكُم بِخَمْسَةِ ءَالَافٖ مِّنَ ٱلْمَلَـٰٓئِكَةِ مُسَوِّمِينَ ﴿125﴾
کیوں نہیں! اگر تم صبر کرو اور تقویٰ اختیار کرو اور وہ دشمن اسی وقت تم پر آ جائیں، تو تمہارا رب تمہاری مدد پانچ ہزار نشان زدہ فرشتوں سے کرے گا۔
وَمَا جَعَلَهُ ٱللَّهُ إِلَّا بُشْرَىٰ لَكُمْ وَلِتَطْمَئِنَّ قُلُوبُكُم بِهِۦۗ وَمَا ٱلنَّصْرُ إِلَّا مِنْ عِندِ ٱللَّهِ ٱلْعَزِيزِ ٱلْحَكِيمِ ﴿126﴾
اور اللہ نے یہ (مدد) تمہارے لیے صرف خوشخبری بنائی، اور تاکہ تمہارے دل اس سے مطمئن ہو جائیں، اور مدد تو صرف اللہ ہی کی طرف سے ہے، جو زبردست، حکمت والا ہے۔
لِيَقْطَعَ طَرَفٗا مِّنَ ٱلَّذِينَ كَفَرُوٓا۟ أَوْ يَكْبِتَهُمْ فَيَنقَلِبُوا۟ خَآئِبِينَ ﴿127﴾
تاکہ وہ کافروں کے ایک گروہ کو کاٹ دے یا انہیں ذلیل کرے، تو وہ ناکام ہو کر لوٹ جائیں۔
لَيْسَ لَكَ مِنَ ٱلْأَمْرِ شَيْءٌ أَوْ يَتُوبَ عَلَيْهِمْ أَوْ يُعَذِّبَهُمْ فَإِنَّهُمْ ظَـٰلِمُونَ ﴿128﴾
آپ کے اختیار میں کچھ نہیں، چاہے اللہ ان پر رحم کرے یا انہیں عذاب دے، کیونکہ وہ ظالم ہیں۔
وَلِلَّهِ مَا فِي ٱلسَّمَـٰوَٰتِ وَمَا فِي ٱلْأَرْضِۚ يَغْفِرُ لِمَن يَشَآءُ وَيُعَذِّبُ مَن يَشَآءُۚ وَٱللَّهُ غَفُورٞ رَّحِيمٞ ﴿129﴾
اور اللہ ہی کا ہے جو کچھ آسمانوں اور زمین میں ہے، وہ جسے چاہے بخش دیتا ہے اور جسے چاہے عذاب دیتا ہے، اور اللہ بخشنے والا، مہربان ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تَأْكُلُوا۟ ٱلرِّبَوٰا۟ أَضْعَـٰفٗا مُّضَـٰعَفَةٗۖ وَٱتَّقُوا۟ ٱللَّهَ لَعَلَّكُمْ تُفْلِحُونَ ﴿130﴾
اے ایمان والو! سود کو دوگنا چوگنا کر کے نہ کھاؤ، اور اللہ سے ڈرو تاکہ تم فلاح پاؤ۔
وَٱتَّقُوا۟ ٱلنَّارَ ٱلَّتِيٓ أُعِدَّتْ لِلْكَـٰفِرِينَ ﴿131﴾
اور اس آگ سے بچو جو کافروں کے لیے تیار کی گئی ہے۔
وَأَطِيعُوا۟ ٱللَّهَ وَٱلرَّسُولَ لَعَلَّكُمْ تُرْحَمُونَ ﴿132﴾
اور اللہ اور رسول کی اطاعت کرو تاکہ تم پر رحم کیا جائے۔
وَسَارِعُوٓا۟ إِلَىٰ مَغْفِرَةٖ مِّن رَّبِّكُمْ وَجَنَّةٍ عَرْضُهَا ٱلسَّمَـٰوَٰتُ وَٱلْأَرْضُ أُعِدَّتْ لِلْمُتَّقِينَ ﴿133﴾
اور اپنے رب کی بخشش اور اس جنت کی طرف دوڑو جس کی چوڑائی آسمانوں اور زمین کے برابر ہے، جو متقیوں کے لیے تیار کی گئی ہے۔
ٱلَّذِينَ يُنفِقُونَ فِي ٱلسَّرَّآءِ وَٱلضَّرَّآءِ وَٱلْكَـٰظِمِينَ ٱلْغَيْظَ وَٱلْعَافِينَ عَنِ ٱلنَّاسِۗ وَٱللَّهُ يُحِبُّ ٱلْمُحْسِنِينَ ﴿134﴾
جو لوگ خوشحالی اور تنگی میں خرچ کرتے ہیں، اور غصے کو ضبط کرنے والے اور لوگوں کو معاف کرنے والے ہیں، اور اللہ نیکی کرنے والوں سے محبت کرتا ہے۔
وَٱلَّذِينَ إِذَا فَعَلُوا۟ فَـٰحِشَةً أَوْ ظَلَمُوٓا۟ أَنفُسَهُمْ ذَكَرُوا۟ ٱللَّهَ فَٱسْتَغْفَرُوا۟ لِذُنُوبِهِمْۖ وَمَن يَغْفِرُ ٱلذُّنُوبَ إِلَّا ٱللَّهُ وَلَمْ يُصِرُّوا۟ عَلَىٰ مَا فَعَلُوا۟ وَهُمْ يَعْلَمُونَ ﴿135﴾
اور وہ لوگ کہ جب کوئی بے حیائی کریں یا اپنے نفس پر ظلم کریں تو اللہ کو یاد کرتے ہیں اور اپنے گناہوں کی بخشش مانگتے ہیں — اور اللہ کے سوا کون گناہوں کو بخش سکتا ہے؟ — اور وہ جانتے بوجھتے ہوئے اپنے کیے پر اصرار نہیں کرتے۔
أُو۟لَـٰٓئِكَ جَزَآؤُهُم مَّغْفِرَةٞ مِّن رَّبِّهِمْ وَجَنَّـٰتٞ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُ خَـٰلِدِينَ فِيهَاۚ وَنِعْمَ أَجْرُ ٱلْعَـٰمِلِينَ ﴿136﴾
ان لوگوں کی جزا ان کے رب کی طرف سے بخشش ہے اور وہ جنتیں ہیں جن کے نیچے نہریں بہتی ہیں، وہ ان میں ہمیشہ رہیں گے، اور عمل کرنے والوں کا کیا ہی اچھا اجر ہے۔
قَدْ خَلَتْ مِن قَبْلِكُمْ سُنَنٞ فَسِيرُوا۟ فِي ٱلْأَرْضِ فَٱنظُرُوا۟ كَيْفَ كَانَ عَـٰقِبَةُ ٱلْمُكَذِّبِينَ ﴿137﴾
تم سے پہلے کئی طریقے گزر چکے ہیں، پس زمین میں چلو پھرو اور دیکھو جھٹلانے والوں کا انجام کیسا ہوا۔
هَـٰذَا بَيَانٞ لِّلنَّاسِ وَهُدٗى وَمَوْعِظَةٞ لِّلْمُتَّقِينَ ﴿138﴾
یہ لوگوں کے لیے ایک بیان ہے، اور ہدایت اور نصیحت ہے پرہیزگاروں کے لیے۔
وَلَا تَهِنُوا۟ وَلَا تَحْزَنُوا۟ وَأَنتُمُ ٱلْأَعْلَوْنَ إِن كُنتُم مُّؤْمِنِينَ ﴿139﴾
اور نہ کمزور پڑو اور نہ غم کرو، اور تم ہی غالب رہو گے اگر تم مؤمن ہو۔
إِن يَمْسَسْكُمْ قَرْحٞ فَقَدْ مَسَّ ٱلْقَوْمَ قَرْحٞ مِّثْلُهُۥۚ وَتِلْكَ ٱلْأَيَّامُ نُدَاوِلُهَا بَيْنَ ٱلنَّاسِ وَلِيَعْلَمَ ٱللَّهُ ٱلَّذِينَ ءَامَنُوا۟ وَيَتَّخِذَ مِنكُمْ شُهَدَآءَۗ وَٱللَّهُ لَا يُحِبُّ ٱلظَّـٰلِمِينَ ﴿140﴾
اگر تمہیں زخم لگا ہے تو اس قوم کو بھی ایسا ہی زخم لگ چکا ہے، اور یہ دن ہم لوگوں کے درمیان بدلتے رہتے ہیں، تاکہ اللہ ایمان والوں کو جان لے اور تم میں سے کچھ کو شہید بنائے، اور اللہ ظالموں کو پسند نہیں کرتا۔
وَلِيُمَحِّصَ ٱللَّهُ ٱلَّذِينَ ءَامَنُوا۟ وَيَمْحَقَ ٱلْكَـٰفِرِينَ ﴿141﴾
اور تاکہ اللہ ایمان والوں کو پاک کرے اور کافروں کو مٹا دے۔
أَمْ حَسِبْتُمْ أَن تَدْخُلُوا۟ ٱلْجَنَّةَ وَلَمَّا يَعْلَمِ ٱللَّهُ ٱلَّذِينَ جَـٰهَدُوا۟ مِنكُمْ وَيَعْلَمَ ٱلصَّـٰبِرِينَ ﴿142﴾
کیا تم نے یہ گمان کیا کہ تم جنت میں داخل ہو جاؤ گے حالانکہ ابھی اللہ نے تم میں سے جہاد کرنے والوں کو نہیں جانچا، اور نہ ہی صبر کرنے والوں کو؟
وَلَقَدْ كُنتُمْ تَمَنَّوْنَ ٱلْمَوْتَ مِن قَبْلِ أَن تَلْقَوْهُ فَقَدْ رَأَيْتُمُوهُ وَأَنتُمْ تَنظُرُونَ ﴿143﴾
اور بے شک تم موت کی تمنا کرتے تھے اس سے پہلے کہ تم اس سے ملو، تو اب تم نے اسے دیکھ لیا جبکہ تم دیکھ رہے تھے۔
وَمَا مُحَمَّدٌ إِلَّا رَسُولٞۚ قَدْ خَلَتْ مِن قَبْلِهِ ٱلرُّسُلُۚ أَفَإِيْن مَّاتَ أَوْ قُتِلَ ٱنقَلَبْتُمْ عَلَىٰٓ أَعْقَـٰبِكُمْۚ وَمَن يَنقَلِبْ عَلَىٰ عَقِبَيْهِ فَلَن يَضُرَّ ٱللَّهَ شَيْـٔٗاۗ وَسَيَجْزِي ٱللَّهُ ٱلشَّـٰكِرِينَ ﴿144﴾
اور محمد (صلى الله عليه وسلم) تو صرف ایک رسول ہیں، ان سے پہلے بھی بہت سے رسول گزر چکے ہیں، تو کیا اگر وہ وفات پا جائیں یا قتل کر دیے جائیں تو تم الٹے پاؤں پھر جاؤ گے؟ اور جو کوئی الٹے پاؤں پھرے گا تو وہ اللہ کا کچھ بھی نقصان نہ کرے گا، اور اللہ شکر گزاروں کو جلد جزا دے گا۔
وَمَا كَانَ لِنَفْسٍ أَن تَمُوتَ إِلَّا بِإِذْنِ ٱللَّهِ كِتَـٰبٗا مُّؤَجَّلٗاۗ وَمَن يُرِدْ ثَوَابَ ٱلدُّنْيَا نُؤْتِهِۦ مِنْهَاۖ وَمَن يُرِدْ ثَوَابَ ٱلْـَٔاخِرَةِ نُؤْتِهِۦ مِنْهَاۚ وَسَنَجْزِي ٱلشَّـٰكِرِينَ ﴿145﴾
اور کوئی جان اللہ کے حکم کے بغیر نہیں مر سکتی، مقررہ وقت کے مطابق، اور جو دنیا کا اجر چاہے ہم اسے اس میں سے دیں گے، اور جو آخرت کا اجر چاہے ہم اسے اس میں سے دیں گے، اور ہم شکر گزاروں کو جلد جزا دیں گے۔
وَكَأَيِّن مِّن نَّبِيّٖ قَـٰتَلَ مَعَهُۥ رِبِّيُّونَ كَثِيرٞ فَمَا وَهَنُوا۟ لِمَآ أَصَابَهُمْ فِي سَبِيلِ ٱللَّهِ وَمَا ضَعُفُوا۟ وَمَا ٱسْتَكَانُوا۟ۗ وَٱللَّهُ يُحِبُّ ٱلصَّـٰبِرِينَ ﴿146﴾
اور کتنے ہی نبی ایسے گزرے جن کے ساتھ بہت سے اللہ والے لڑے، پھر وہ اللہ کی راہ میں جو مصیبت انہیں پہنچی اس سے نہ کمزور ہوئے، نہ تھکے، اور نہ جھکے، اور اللہ صبر کرنے والوں سے محبت کرتا ہے۔
وَمَا كَانَ قَوْلَهُمْ إِلَّآ أَن قَالُوا۟ رَبَّنَا ٱغْفِرْ لَنَا ذُنُوبَنَا وَإِسْرَافَنَا فِيٓ أَمْرِنَا وَثَبِّتْ أَقْدَامَنَا وَٱنصُرْنَا عَلَى ٱلْقَوْمِ ٱلْكَـٰفِرِينَ ﴿147﴾
اور ان کا قول صرف یہ تھا کہ: اے ہمارے رب! ہمارے گناہ اور ہمارے کام میں زیادتی کو بخش دے، اور ہمارے قدم جما دے، اور ہمیں کافر قوم پر فتح دے۔
فَـَٔاتَىٰهُمُ ٱللَّهُ ثَوَابَ ٱلدُّنْيَا وَحُسْنَ ثَوَابِ ٱلْـَٔاخِرَةِۚ وَٱللَّهُ يُحِبُّ ٱلْمُحْسِنِينَ ﴿148﴾
تو اللہ نے انہیں دنیا کا اجر دیا اور آخرت کا بہترین اجر بھی، اور اللہ نیکی کرنے والوں سے محبت کرتا ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ إِن تُطِيعُوا۟ ٱلَّذِينَ كَفَرُوا۟ يَرُدُّوكُمْ عَلَىٰٓ أَعْقَـٰبِكُمْ فَتَنقَلِبُوا۟ خَـٰسِرِينَ ﴿149﴾
اے ایمان والو! اگر تم کافروں کی بات مانو گے تو وہ تمہیں الٹے پاؤں پھیر دیں گے، پھر تم نقصان اٹھانے والے بن جاؤ گے۔
بَلِ ٱللَّهُ مَوْلَىٰكُمْۖ وَهُوَ خَيْرُ ٱلنَّـٰصِرِينَ ﴿150﴾
بلکہ اللہ تمہارا مددگار ہے، اور وہ سب مدد کرنے والوں سے بہتر ہے۔
سَنُلْقِي فِي قُلُوبِ ٱلَّذِينَ كَفَرُوا۟ ٱلرُّعْبَ بِمَآ أَشْرَكُوا۟ بِٱللَّهِ مَا لَمْ يُنَزِّلْ بِهِۦ سُلْطَـٰنٗاۖ وَمَأْوَىٰهُمُ ٱلنَّارُۖ وَبِئْسَ مَثْوَى ٱلظَّـٰلِمِينَ ﴿151﴾
ہم عنقریب کافروں کے دلوں میں رعب ڈال دیں گے، اس وجہ سے کہ انہوں نے اللہ کے ساتھ اس چیز کو شریک بنایا جس کی کوئی دلیل اللہ نے نازل نہیں کی، اور ان کا ٹھکانہ جہنم ہے، اور وہ ظالموں کا برا ٹھکانہ ہے۔
وَلَقَدْ صَدَقَكُمُ ٱللَّهُ وَعْدَهُۥٓ إِذْ تَحُسُّونَهُم بِإِذْنِهِۦۖ حَتَّىٰٓ إِذَا فَشِلْتُمْ وَتَنَـٰزَعْتُمْ فِي ٱلْأَمْرِ وَعَصَيْتُم مِّنۢ بَعْدِ مَآ أَرَىٰكُم مَّا تُحِبُّونَۚ مِنكُم مَّن يُرِيدُ ٱلدُّنْيَا وَمِنكُم مَّن يُرِيدُ ٱلْـَٔاخِرَةَۚ ثُمَّ صَرَفَكُمْ عَنْهُمْ لِيَبْتَلِيَكُمْۖ وَلَقَدْ عَفَا عَنكُمْۗ وَٱللَّهُ ذُو فَضْلٍ عَلَى ٱلْمُؤْمِنِينَ ﴿152﴾
اور بے شک اللہ نے تم سے اپنا وعدہ سچ کر دکھایا جب تم ان کو اس کے حکم سے قتل کر رہے تھے، یہاں تک کہ تم نے کمزوری دکھائی اور اس معاملے میں اختلاف کیا اور نافرمانی کی، بعد اس کے کہ اللہ نے تمہیں وہ چیز دکھا دی جو تم چاہتے تھے، تم میں سے کچھ دنیا چاہتے تھے اور کچھ آخرت، پھر اللہ نے تمہیں ان سے پھیر دیا تاکہ تمہیں آزمائے، اور بے شک اللہ نے تمہیں معاف کر دیا، اور اللہ ایمان والوں پر فضل والا ہے۔
إِذْ تُصْعِدُونَ وَلَا تَلْوُۥنَ عَلَىٰٓ أَحَدٖ وَٱلرَّسُولُ يَدْعُوكُمْ فِيٓ أُخْرَىٰكُمْ فَأَثَـٰبَكُمْ غَمًّۢا بِغَمّٖ لِّكَيْلَا تَحْزَنُوا۟ عَلَىٰ مَا فَاتَكُمْ وَلَا مَآ أَصَـٰبَكُمْۗ وَٱللَّهُ خَبِيرُۢ بِمَا تَعْمَلُونَ ﴿153﴾
جب تم بھاگے جا رہے تھے اور کسی کی طرف پلٹ کر نہیں دیکھ رہے تھے، اور رسول تمہیں پیچھے سے پکار رہے تھے، تو اللہ نے تمہیں غم پر غم دیا تاکہ تم اس پر غم نہ کرو جو تمہارے ہاتھ سے نکل گیا اور نہ اس پر جو تمہیں پہنچا، اور اللہ تمہارے اعمال سے خوب باخبر ہے۔
ثُمَّ أَنزَلَ عَلَيْكُم مِّنۢ بَعْدِ ٱلْغَمِّ أَمَنَةٗ نُّعَاسٗا يَغْشَىٰ طَآئِفَةٗ مِّنكُمْۖ وَطَآئِفَةٞ قَدْ أَهَمَّتْهُمْ أَنفُسُهُمْ يَظُنُّونَ بِٱللَّهِ غَيْرَ ٱلْحَقِّ ظَنَّ ٱلْجَـٰهِلِيَّةِۖ يَقُولُونَ هَل لَّنَا مِنَ ٱلْأَمْرِ مِن شَيْءٖۚ قُلْ إِنَّ ٱلْأَمْرَ كُلَّهُۥ لِلَّهِۗ يُخْفُونَ فِيٓ أَنفُسِهِم مَّا لَا يُبْدُونَ لَكَۖ يَقُولُونَ لَوْ كَانَ لَنَا مِنَ ٱلْأَمْرِ شَيْءٞ مَّا قُتِلْنَا هَـٰهُنَاۗ قُل لَّوْ كُنتُمْ فِي بُيُوتِكُمْ لَبَرَزَ ٱلَّذِينَ كُتِبَ عَلَيْهِمُ ٱلْقَتْلُ إِلَىٰ مَضَاجِعِهِمْۗ وَلِيَبْتَلِيَ ٱللَّهُ مَا فِي صُدُورِكُمْ وَلِيُمَحِّصَ مَا فِي قُلُوبِكُمْۗ وَٱللَّهُ عَلِيمُۢ بِذَاتِ ٱلصُّدُورِ ﴿154﴾
پھر اس غم کے بعد اللہ نے تم پر ایک امانت والی نیند نازل کی جو تم میں سے ایک گروہ پر چھا گئی، اور ایک گروہ وہ تھا جنہیں اپنی جان کی فکر لاحق تھی، وہ اللہ کے بارے میں جاہلیت کا گمان کر رہے تھے، وہ کہتے تھے: کیا ہمارے لیے کچھ اختیار ہے؟ کہہ دو: سارا اختیار اللہ ہی کا ہے، وہ اپنے دلوں میں وہ بات چھپاتے ہیں جو وہ آپ پر ظاہر نہیں کرتے، وہ کہتے ہیں: اگر ہمارے اختیار میں کچھ ہوتا تو ہم یہاں قتل نہ کیے جاتے، کہہ دو: اگر تم اپنے گھروں میں بھی ہوتے تو جن پر قتل لکھا جا چکا تھا وہ ضرور اپنی قتل گاہوں کی طرف نکل آتے، اور یہ اس لیے ہوا تاکہ اللہ تمہارے سینوں کی بات کو آزمائے اور تمہارے دلوں کو پاک کرے، اور اللہ دلوں کے بھید خوب جانتا ہے۔
إِنَّ ٱلَّذِينَ تَوَلَّوْا۟ مِنكُمْ يَوْمَ ٱلْتَقَى ٱلْجَمْعَانِ إِنَّمَا ٱسْتَزَلَّهُمُ ٱلشَّيْطَـٰنُ بِبَعْضِ مَا كَسَبُوا۟ۖ وَلَقَدْ عَفَا ٱللَّهُ عَنْهُمْۚ إِنَّ ٱللَّهَ غَفُورٌ حَلِيمٞ ﴿155﴾
بے شک تم میں سے جنہوں نے اس دن پیٹھ پھیری جب دونوں لشکر آپس میں ٹکرائے، انہیں شیطان نے ان کے بعض اعمال کے سبب لغزش دی، اور بے شک اللہ نے انہیں معاف کر دیا، بے شک اللہ بخشنے والا، بردبار ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تَكُونُوا۟ كَٱلَّذِينَ كَفَرُوا۟ وَقَالُوا۟ لِإِخْوَٰنِهِمْ إِذَا ضَرَبُوا۟ فِي ٱلْأَرْضِ أَوْ كَانُوا۟ غُزّٗى لَّوْ كَانُوا۟ عِندَنَا مَا مَاتُوا۟ وَمَا قُتِلُوا۟ لِيَجْعَلَ ٱللَّهُ ذَٰلِكَ حَسْرَةٗ فِي قُلُوبِهِمۗ وَٱللَّهُ يُحْيِۦ وَيُمِيتُۗ وَٱللَّهُ بِمَا تَعْمَلُونَ بَصِيرٞ ﴿156﴾
اے ایمان والو! ان لوگوں کی طرح نہ ہو جانا جنہوں نے کفر کیا اور اپنے بھائیوں کے بارے میں کہا جب وہ زمین میں سفر پر نکلے یا جہاد میں گئے: اگر وہ ہمارے پاس ہوتے تو نہ مرتے اور نہ قتل کیے جاتے، تاکہ اللہ اس بات کو ان کے دلوں میں حسرت بنا دے، اور اللہ ہی زندگی دیتا ہے اور موت دیتا ہے، اور اللہ تمہارے اعمال کو خوب دیکھنے والا ہے۔
وَلَئِن قُتِلْتُمْ فِي سَبِيلِ ٱللَّهِ أَوْ مُتُّمْ لَمَغْفِرَةٞ مِّنَ ٱللَّهِ وَرَحْمَةٌ خَيْرٞ مِّمَّا يَجْمَعُونَ ﴿157﴾
اور اگر تم اللہ کی راہ میں قتل کیے جاؤ یا مر جاؤ تو اللہ کی طرف سے مغفرت اور رحمت ان چیزوں سے بہتر ہے جو وہ جمع کرتے ہیں۔
وَلَئِن مُّتُّمْ أَوْ قُتِلْتُمْ لَإِلَى ٱللَّهِ تُحْشَرُونَ ﴿158﴾
اور اگر تم مر جاؤ یا قتل کیے جاؤ تو تم سب اللہ ہی کی طرف جمع کیے جاؤ گے۔
فَبِمَا رَحْمَةٖ مِّنَ ٱللَّهِ لِنتَ لَهُمْۖ وَلَوْ كُنتَ فَظّٗا غَلِيظَ ٱلْقَلْبِ لَٱنفَضُّوا۟ مِنْ حَوْلِكَۖ فَٱعْفُ عَنْهُمْ وَٱسْتَغْفِرْ لَهُمْ وَشَاوِرْهُمْ فِي ٱلْأَمْرِۖ فَإِذَا عَزَمْتَ فَتَوَكَّلْ عَلَى ٱللَّهِۚ إِنَّ ٱللَّهَ يُحِبُّ ٱلْمُتَوَكِّلِينَ ﴿159﴾
پس اللہ کی رحمت سے آپ ان کے لیے نرم ہو گئے، اور اگر آپ سخت مزاج اور سخت دل ہوتے تو وہ آپ کے گرد سے منتشر ہو جاتے، سو ان سے درگزر کریں، ان کے لیے بخشش مانگیں، اور کام میں ان سے مشورہ کریں، پھر جب ارادہ کر لیں تو اللہ پر بھروسا کریں، بے شک اللہ بھروسا کرنے والوں کو پسند کرتا ہے۔
إِن يَنصُرْكُمُ ٱللَّهُ فَلَا غَالِبَ لَكُمْۖ وَإِن يَخْذُلْكُمْ فَمَن ذَا ٱلَّذِي يَنصُرُكُم مِّنۢ بَعْدِهِۦۗ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُؤْمِنُونَ ﴿160﴾
اگر اللہ تمہاری مدد کرے تو تم پر کوئی غالب نہیں آ سکتا، اور اگر وہ تمہیں چھوڑ دے تو اس کے بعد کون ہے جو تمہاری مدد کرے؟ اور مؤمنوں کو اللہ ہی پر بھروسا کرنا چاہیے۔
وَمَا كَانَ لِنَبِيٍّ أَن يَغُلَّۚ وَمَن يَغْلُلْ يَأْتِ بِمَا غَلَّ يَوْمَ ٱلْقِيَـٰمَةِۚ ثُمَّ تُوَفَّىٰ كُلُّ نَفْسٖ مَّا كَسَبَتْ وَهُمْ لَا يُظْلَمُونَ ﴿161﴾
کسی نبی کے لیے جائز نہیں کہ خیانت کرے، اور جو خیانت کرے گا وہ قیامت کے دن وہ چیز لے کر آئے گا جس میں اس نے خیانت کی، پھر ہر شخص کو اس کے کیے کا پورا بدلہ دیا جائے گا، اور ان پر ظلم نہیں کیا جائے گا۔
أَفَمَنِ ٱتَّبَعَ رِضْوَٰنَ ٱللَّهِ كَمَنۢ بَآءَ بِسَخَطٖ مِّنَ ٱللَّهِ وَمَأْوَىٰهُ جَهَنَّمُۚ وَبِئْسَ ٱلْمَصِيرُ ﴿162﴾
تو کیا وہ شخص جو اللہ کی رضا کا پیرو ہے اس شخص کی طرح ہو سکتا ہے جو اللہ کے غضب کا سزاوار ہوا اور اس کا ٹھکانہ جہنم ہے؟ اور وہ بہت برا ٹھکانہ ہے۔
هُمْ دَرَجَـٰتٌ عِندَ ٱللَّهِۗ وَٱللَّهُ بَصِيرُۢ بِمَا يَعْمَلُونَ ﴿163﴾
ان کے درجے اللہ کے ہاں مختلف ہیں، اور اللہ ان کے اعمال کو خوب دیکھنے والا ہے۔
لَقَدْ مَنَّ ٱللَّهُ عَلَى ٱلْمُؤْمِنِينَ إِذْ بَعَثَ فِيهِمْ رَسُولٗا مِّنْ أَنفُسِهِمْ يَتْلُوا۟ عَلَيْهِمْ ءَايَـٰتِهِۦ وَيُزَكِّيهِمْ وَيُعَلِّمُهُمُ ٱلْكِتَـٰبَ وَٱلْحِكْمَةَۚ وَإِن كَانُوا۟ مِن قَبْلُ لَفِي ضَلَـٰلٖ مُّبِينٖ ﴿164﴾
بے شک اللہ نے مؤمنوں پر بڑا احسان کیا جب اس نے ان میں انہی میں سے ایک رسول بھیجا، جو ان پر اس کی آیات پڑھتا ہے، انہیں پاک کرتا ہے، اور انہیں کتاب و حکمت سکھاتا ہے، حالانکہ وہ اس سے پہلے کھلی گمراہی میں تھے۔
أَوَلَمَّآ أَصَـٰبَتْكُم مُّصِيبَةٞ قَدْ أَصَبْتُم مِّثْلَيْهَا قُلْتُمْ أَنَّىٰ هَـٰذَاۖ قُلْ هُوَ مِنْ عِندِ أَنفُسِكُمْۗ إِنَّ ٱللَّهَ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٞ ﴿165﴾
کیا جب تمہیں مصیبت پہنچی حالانکہ تم اس سے دوگنی پہنچا چکے تھے، تو تم نے کہا: یہ کہاں سے آ گئی؟ کہہ دو: یہ تمہاری اپنی طرف سے ہے، بے شک اللہ ہر چیز پر قادر ہے۔
وَمَآ أَصَـٰبَكُمۡ يَوۡمَ ٱلۡتَقَى ٱلۡجَمۡعَانِ فَبِإِذۡنِ ٱللَّهِ وَلِيَعۡلَمَ ٱلۡمُؤۡمِنِينَ ﴿166﴾
اور جو کچھ تمہیں اس دن پہنچا جب دونوں لشکر آمنے سامنے ہوئے، وہ اللہ کے حکم سے تھا، اور تاکہ وہ ایمان والوں کو جان لے۔
وَلِيَعۡلَمَ ٱلَّذِينَ نَافَقُواْۚ وَقِيلَ لَهُمۡ تَعَالَوۡاْ قَـٰتِلُواْ فِي سَبِيلِ ٱللَّهِ أَوِ ٱدۡفَعُواْۖ قَالُواْ لَوۡ نَعۡلَمُ قِتَالٗا لَّٱتَّبَعۡنَـٰكُمۡۗ هُمۡ لِلۡكُفۡرِ يَوۡمَئِذٍ أَقۡرَبُ مِنۡهُمۡ لِلۡإِيمَـٰنِۚ يَقُولُونَ بِأَفۡوَٰهِهِم مَّا لَيۡسَ فِي قُلُوبِهِمۡۚ وَٱللَّهُ أَعۡلَمُ بِمَا يَكۡتُمُونَ ﴿167﴾
اور تاکہ وہ منافقوں کو بھی جان لے، اور ان سے کہا گیا: آؤ اللہ کی راہ میں لڑو یا دفاع کرو، تو انہوں نے کہا: اگر ہمیں لڑائی کا علم ہوتا تو ہم ضرور تمہارے ساتھ چلتے، وہ اس دن ایمان کی نسبت کفر سے زیادہ قریب تھے، وہ اپنے منہ سے وہ بات کہتے ہیں جو ان کے دلوں میں نہیں ہوتی، اور اللہ خوب جانتا ہے جو وہ چھپاتے ہیں۔
ٱلَّذِينَ قَالُواْ لِإِخۡوَٰنِهِمۡ وَقَعَدُواْ لَوۡ أَطَاعُونَا مَا قُتِلُواْۗ قُلۡ فَٱدۡرَءُواْ عَنۡ أَنفُسِكُمُ ٱلۡمَوۡتَ إِن كُنتُمۡ صَـٰدِقِينَ ﴿168﴾
جنہوں نے اپنے بھائیوں کے بارے میں کہا حالانکہ وہ خود بیٹھے رہے: اگر وہ ہماری بات مانتے تو قتل نہ ہوتے، کہہ دو: اگر تم سچے ہو تو اپنی جانوں سے موت کو ہٹا لو۔
وَلَا تَحۡسَبَنَّ ٱلَّذِينَ قُتِلُواْ فِي سَبِيلِ ٱللَّهِ أَمۡوَٰتَۢاۚ بَلۡ أَحۡيَآءٌ عِندَ رَبِّهِمۡ يُرۡزَقُونَ ﴿169﴾
اور جو لوگ اللہ کی راہ میں قتل کیے گئے انہیں مردہ نہ سمجھو، بلکہ وہ زندہ ہیں، اپنے رب کے پاس رزق دیے جاتے ہیں۔
فَرِحِينَ بِمَآ ءَاتَىٰهُمُ ٱللَّهُ مِن فَضۡلِهِۦ وَيَسۡتَبۡشِرُونَ بِٱلَّذِينَ لَمۡ يَلۡحَقُواْ بِهِم مِّنۢ خَلۡفِهِمۡ أَلَّا خَوۡفٌ عَلَيۡهِمۡ وَلَا هُمۡ يَحۡزَنُونَ ﴿170﴾
وہ اللہ کے فضل سے جو انہیں ملا ہے خوش ہیں، اور ان لوگوں کے بارے میں خوش خبری دیتے ہیں جو ابھی ان سے نہیں ملے، کہ ان پر نہ کوئی خوف ہوگا اور نہ وہ غمگین ہوں گے۔
يَسۡتَبۡشِرُونَ بِنِعۡمَةٖ مِّنَ ٱللَّهِ وَفَضۡلٖ وَأَنَّ ٱللَّهَ لَا يُضِيعُ أَجۡرَ ٱلۡمُؤۡمِنِينَ ﴿171﴾
وہ اللہ کی نعمت اور فضل کی خوش خبری دیتے ہیں، اور یہ کہ اللہ مؤمنوں کا اجر ضائع نہیں کرتا۔
ٱلَّذِينَ ٱسۡتَجَابُواْ لِلَّهِ وَٱلرَّسُولِ مِنۢ بَعۡدِ مَآ أَصَابَهُمُ ٱلۡقَرۡحُۚ لِلَّذِينَ أَحۡسَنُواْ مِنۡهُمۡ وَٱتَّقَوۡاْ أَجۡرٌ عَظِيمٌ ﴿172﴾
جنہوں نے زخم پہنچنے کے بعد اللہ اور رسول کی بات مانی، ان میں سے جنہوں نے نیکی کی اور تقویٰ اختیار کیا ان کے لیے بڑا اجر ہے۔
ٱلَّذِينَ قَالَ لَهُمُ ٱلنَّاسُ إِنَّ ٱلنَّاسَ قَدْ جَمَعُوا۟ لَكُمْ فَٱخْشَوْهُمْ فَزَادَهُمْ إِيمَـٰنٗا وَقَالُوا۟ حَسْبُنَا ٱللَّهُ وَنِعْمَ ٱلْوَكِيلُ ﴿173﴾
وہ لوگ جن سے لوگوں نے کہا: دشمنوں نے تمہارے خلاف لشکر جمع کر لیا ہے، ان سے ڈرو، تو اس (بات) نے ان کا ایمان اور بڑھا دیا، اور انہوں نے کہا: ہمارے لیے اللہ کافی ہے، اور وہی بہترین کارساز ہے۔
فَٱنقَلَبُوا۟ بِنِعْمَةٖ مِّنَ ٱللَّهِ وَفَضْلٖ لَّمْ يَمْسَسْهُمْ سُوٓءٞۚ وَٱتَّبَعُوا۟ رِضْوَٰنَ ٱللَّهِۗ وَٱللَّهُ ذُو فَضْلٍ عَظِيمٍ ﴿174﴾
تو وہ اللہ کے فضل اور اس کی نعمت کے ساتھ واپس لوٹے، انہیں کوئی برائی نہ پہنچی، اور انہوں نے اللہ کی رضا کی پیروی کی، اور اللہ بڑے فضل والا ہے۔
إِنَّمَا ذَٰلِكُمُ ٱلشَّيْطَـٰنُ يُخَوِّفُ أَوْلِيَآءَهُۥۖ فَلَا تَخَافُوهُمْ وَخَافُونِ إِن كُنتُم مُّؤْمِنِينَ ﴿175﴾
یہ تو صرف شیطان ہے جو اپنے دوستوں سے تمہیں ڈراتا ہے، پس ان سے نہ ڈرو بلکہ مجھ سے ڈرو اگر تم مؤمن ہو۔
وَلَا يَحْزُنكَ ٱلَّذِينَ يُسَـٰرِعُونَ فِي ٱلْكُفْرِۚ إِنَّهُمْ لَن يَضُرُّوا۟ ٱللَّهَ شَيْـٔٗاۚ يُرِيدُ ٱللَّهُ أَلَّا يَجْعَلَ لَهُمْ حَظّٗا فِي ٱلْـَٔاخِرَةِۖ وَلَهُمْ عَذَابٌ عَظِيمٌ ﴿176﴾
جو لوگ کفر میں جلدی کرتے ہیں وہ آپ کو غمگین نہ کریں، بے شک وہ اللہ کا کچھ بھی نقصان نہیں کر سکتے، اللہ چاہتا ہے کہ ان کے لیے آخرت میں کوئی حصہ نہ رکھے، اور ان کے لیے بڑا عذاب ہے۔
إِنَّ ٱلَّذِينَ ٱشْتَرَوُا۟ ٱلْكُفْرَ بِٱلْإِيمَـٰنِ لَن يَضُرُّوا۟ ٱللَّهَ شَيْـٔٗا وَلَهُمْ عَذَابٌ أَلِيمٞ ﴿177﴾
بے شک جنہوں نے ایمان کے بدلے کفر خریدا، وہ اللہ کا کچھ بھی نقصان نہیں کر سکتے، اور ان کے لیے دردناک عذاب ہے۔
وَلَا يَحْسَبَنَّ ٱلَّذِينَ كَفَرُوٓا۟ أَنَّمَا نُمۡلِي لَهُمۡ خَيۡرٞ لِّأَنفُسِهِمۡۚ إِنَّمَا نُمۡلِي لَهُمۡ لِيَزۡدَادُوٓا۟ إِثۡمٗاۚ وَلَهُمۡ عَذَابٞ مُّهِينٞ ﴿178﴾
اور جو لوگ کفر کرتے ہیں وہ ہرگز یہ نہ سمجھیں کہ ہم جو انہیں مہلت دیتے ہیں وہ ان کے لیے بہتر ہے، ہم تو انہیں اس لیے مہلت دیتے ہیں تاکہ وہ گناہ میں اور بڑھ جائیں، اور ان کے لیے رسوا کن عذاب ہے۔
مَا كَانَ ٱللَّهُ لِيَذَرَ ٱلْمُؤْمِنِينَ عَلَىٰ مَآ أَنتُمْ عَلَيْهِ حَتَّىٰ يَمِيزَ ٱلْخَبِيثَ مِنَ ٱلطَّيِّبِۗ وَمَا كَانَ ٱللَّهُ لِيُطْلِعَكُمْ عَلَى ٱلْغَيْبِ وَلَـٰكِنَّ ٱللَّهَ يَجْتَبِي مِن رُّسُلِهِۦ مَن يَشَآءُۖ فَـَٔامِنُوا۟ بِٱللَّهِ وَرُسُلِهِۦۚ وَإِن تُؤْمِنُوا۟ وَتَتَّقُوا۟ فَلَكُمْ أَجْرٌ عَظِيمٞ ﴿179﴾
اللہ ایسا نہیں کہ مؤمنوں کو اس حالت پر چھوڑ دے جس پر تم ہو، جب تک کہ ناپاک کو پاک سے جدا نہ کر دے، اور اللہ ایسا نہیں کہ تمہیں غیب پر مطلع کرے، لیکن اللہ اپنے رسولوں میں سے جسے چاہے چن لیتا ہے، پس اللہ اور اس کے رسولوں پر ایمان لاؤ، اور اگر تم ایمان لاؤ اور تقویٰ اختیار کرو تو تمہارے لیے بڑا اجر ہے۔
إِنَّ ٱلَّذِينَ يَبْخَلُونَ بِمَآ ءَاتَىٰهُمُ ٱللَّهُ مِن فَضْلِهِۦ يَحْسَبُونَهُۥ خَيْرٗاۚ بَلْ هُوَ شَرّٞ لَّهُمْۚ سَيُطَوَّقُونَ مَا بَخِلُوا۟ بِهِۦ يَوْمَ ٱلْقِيَـٰمَةِۗ وَلِلَّهِ مِيرَٰثُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۗ وَٱللَّهُ بِمَا تَعْمَلُونَ خَبِيرٞ ﴿180﴾
بے شک جو لوگ اللہ کے فضل سے دی گئی چیزوں میں بخل کرتے ہیں، وہ اسے اپنے لیے بہتر سمجھتے ہیں، بلکہ وہ ان کے لیے بُرا ہے، قیامت کے دن وہی چیز ان کے گلے کا طوق بنائی جائے گی، اور اللہ ہی کے لیے ہے آسمانوں اور زمین کا وارث، اور اللہ تمہارے اعمال سے خوب باخبر ہے۔
لَقَدْ سَمِعَ ٱللَّهُ قَوْلَ ٱلَّذِينَ قَالُوٓا۟ إِنَّ ٱللَّهَ فَقِيرٞ وَنَحْنُ أَغْنِيَآءُۘ سَنَكْتُبُ مَا قَالُوا۟ وَقَتْلَهُمُ ٱلْأَنبِيَآءَ بِغَيْرِ حَقّٖ وَنَقُولُ ذُوقُوا۟ عَذَابَ ٱلْحَرِيقِ ﴿181﴾
اللہ نے ان لوگوں کی بات سن لی جنہوں نے کہا: اللہ فقیر ہے اور ہم امیر ہیں، ہم ان کی بات اور ان کا نبیوں کو ناحق قتل کرنا لکھ رکھیں گے، اور کہیں گے: جلنے والے عذاب کا مزہ چکھو۔
ذَٰلِكَ بِمَا قَدَّمَتْ أَيْدِيكُمْ وَأَنَّ ٱللَّهَ لَيْسَ بِظَلَّـٰمٖ لِّلْعَبِيدِ ﴿182﴾
یہ اس وجہ سے ہے جو تمہارے ہاتھوں نے آگے بھیجا، اور بے شک اللہ بندوں پر ظلم کرنے والا نہیں۔
ٱلَّذِينَ قَالُوٓا۟ إِنَّ ٱللَّهَ عَهِدَ إِلَيْنَآ أَلَّا نُؤْمِنَ لِرَسُولٍ حَتَّىٰ يَأْتِيَنَا بِقُرْبَانٖ تَأْكُلُهُ ٱلنَّارُۗ قُلْ قَدْ جَآءَكُمْ رُسُلٞ مِّن قَبْلِي بِٱلْبَيِّنَـٰتِ وَبِٱلَّذِي قُلْتُمْ فَلِمَ قَتَلْتُمُوهُمْ إِن كُنتُمْ صَـٰدِقِينَ ﴿183﴾
جنہوں نے کہا: اللہ نے ہم سے وعدہ کیا ہے کہ ہم کسی رسول پر ایمان نہ لائیں جب تک وہ ہمارے پاس ایسی قربانی نہ لائے جسے آگ کھا جائے، کہہ دو: مجھ سے پہلے تمہارے پاس رسول واضح نشانیاں اور وہ چیز بھی لے کر آئے جو تم نے کہی، پھر تم نے انہیں کیوں قتل کیا اگر تم سچے ہو؟
فَإِن كَذَّبُوكَ فَقَدْ كُذِّبَ رُسُلٞ مِّن قَبْلِكَ جَآءُوا۟ بِٱلْبَيِّنَـٰتِ وَٱلزُّبُرِ وَٱلْكِتَـٰبِ ٱلْمُنِيرِ ﴿184﴾
پھر اگر وہ آپ کو جھٹلاتے ہیں تو آپ سے پہلے بھی کئی رسول جھٹلائے گئے، جو واضح دلائل، صحیفے اور روشن کتاب لے کر آئے تھے۔
كُلُّ نَفْسٖ ذَآئِقَةُ ٱلْمَوْتِۗ وَإِنَّمَا تُوَفَّوْنَ أُجُورَكُمْ يَوْمَ ٱلْقِيَـٰمَةِۖ فَمَن زُحْزِحَ عَنِ ٱلنَّارِ وَأُدْخِلَ ٱلْجَنَّةَ فَقَدْ فَازَۗ وَمَا ٱلْحَيَوٰةُ ٱلدُّنْيَآ إِلَّا مَتَـٰعُ ٱلْغُرُورِ ﴿185﴾
ہر جان کو موت کا مزہ چکھنا ہے، اور تمہیں تمہارے اجر قیامت کے دن ہی پورے دیے جائیں گے، پس جو شخص آگ سے بچا لیا گیا اور جنت میں داخل کر دیا گیا وہی کامیاب ہوا، اور دنیا کی زندگی تو دھوکے کا سامان ہے۔
لَتُبْلَوُنَّ فِيٓ أَمْوَٰلِكُمْ وَأَنفُسِكُمْ وَلَتَسْمَعُنَّ مِنَ ٱلَّذِينَ أُوتُوا۟ ٱلْكِتَـٰبَ مِن قَبْلِكُمْ وَمِنَ ٱلَّذِينَ أَشْرَكُوٓا۟ أَذٗى كَثِيرٗاۚ وَإِن تَصْبِرُوا۟ وَتَتَّقُوا۟ فَإِنَّ ذَٰلِكَ مِنْ عَزْمِ ٱلْأُمُورِ ﴿186﴾
تم ضرور اپنے مالوں اور جانوں میں آزمائے جاؤ گے، اور تم ضرور اہلِ کتاب اور مشرکوں سے بہت سی اذیت کی باتیں سنو گے، اور اگر تم صبر کرو اور تقویٰ اختیار کرو تو یہ بڑے حوصلے کے کاموں میں سے ہے۔
وَإِذْ أَخَذَ ٱللَّهُ مِيثَـٰقَ ٱلَّذِينَ أُوتُوا۟ ٱلْكِتَـٰبَ لَتُبَيِّنُنَّهُۥ لِلنَّاسِ وَلَا تَكْتُمُونَهُۥ فَنَبَذُوهُ وَرَآءَ ظُهُورِهِمْ وَٱشْتَرَوْا۟ بِهِۦ ثَمَنٗا قَلِيلٗاۚ فَبِئْسَ مَا يَشْتَرُونَ ﴿187﴾
اور جب اللہ نے اہلِ کتاب سے عہد لیا کہ تم اسے لوگوں کے لیے ضرور واضح کرو گے اور اسے چھپاؤ گے نہیں، تو انہوں نے اسے اپنی پیٹھوں کے پیچھے پھینک دیا اور اس کے بدلے تھوڑی قیمت لے لی، تو کیا ہی برا سودا ہے جو وہ کرتے ہیں۔
لَا تَحْسَبَنَّ ٱلَّذِينَ يَفْرَحُونَ بِمَآ أَتَوْا۟ وَّيُحِبُّونَ أَن يُحْمَدُوا۟ بِمَا لَمْ يَفْعَلُوا۟ فَلَا تَحْسَبْهُم بِمَفَازَةٖ مِّنَ ٱلْعَذَابِۖ وَلَهُمْ عَذَابٌ أَلِيمٌ ﴿188﴾
جو لوگ اپنے کیے پر خوش ہوتے ہیں اور چاہتے ہیں کہ ان کاموں پر ان کی تعریف کی جائے جو انہوں نے کیے ہی نہیں، انہیں عذاب سے نجات یافتہ نہ سمجھو، اور ان کے لیے دردناک عذاب ہے۔
وَلِلَّهِ مُلْكُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۗ وَٱللَّهُ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٌ ﴿189﴾
اور آسمانوں اور زمین کی بادشاہی اللہ ہی کے لیے ہے، اور اللہ ہر چیز پر قادر ہے۔
إِنَّ فِي خَلْقِ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَٱخْتِلَـٰفِ ٱلَّيْلِ وَٱلنَّهَارِ لَءَايَـٰتٖ لِّأُو۟لِي ٱلْأَلْبَـٰبِ ﴿190﴾
بے شک آسمانوں اور زمین کی پیدائش میں، اور رات دن کے بدلنے میں عقل والوں کے لیے نشانیاں ہیں۔
ٱلَّذِينَ يَذْكُرُونَ ٱللَّهَ قِيَـٰمٗا وَقُعُودٗا وَعَلَىٰ جُنُوبِهِمْ وَيَتَفَكَّرُونَ فِي خَلْقِ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ رَبَّنَا مَا خَلَقْتَ هَـٰذَا بَـٰطِلٗاۖ سُبْحَـٰنَكَ فَقِنَا عَذَابَ ٱلنَّارِ ﴿191﴾
جو لوگ اللہ کو یاد کرتے ہیں کھڑے، بیٹھے اور لیٹے ہوئے، اور آسمانوں اور زمین کی پیدائش میں غور کرتے ہیں (کہتے ہیں:) اے ہمارے رب! تو نے یہ سب بے مقصد نہیں بنایا، تو پاک ہے، پس ہمیں آگ کے عذاب سے بچا۔
رَبَّنَآ إِنَّكَ مَن تُدْخِلِ ٱلنَّارَ فَقَدْ أَخْزَيْتَهُۥۖ وَمَا لِلظَّـٰلِمِينَ مِنْ أَنصَارٖ ﴿192﴾
اے ہمارے رب! جسے تو آگ میں داخل کرے، تو یقیناً اسے رسوا کر دیا، اور ظالموں کا کوئی مددگار نہیں۔
رَبَّنَآ إِنَّنَا سَمِعْنَا مُنَادِيٗا يُنَادِي لِلْإِيمَـٰنِ أَنْ ءَامِنُوا۟ بِرَبِّكُمْ فَـَٔامَنَّاۚ رَبَّنَا فَٱغْفِرْ لَنَا ذُنُوبَنَا وَكَفِّرْ عَنَّا سَيِّـَٔاتِنَا وَتَوَفَّنَا مَعَ ٱلْأَبْرَارِ ﴿193﴾
اے ہمارے رب! ہم نے ایک پکارنے والے کو سنا جو ایمان کی طرف بلاتا تھا کہ اپنے رب پر ایمان لاؤ، تو ہم ایمان لے آئے، اے ہمارے رب! ہمارے گناہ بخش دے، ہماری برائیاں دور کر دے، اور ہمیں نیک لوگوں کے ساتھ موت دے۔
رَبَّنَا وَءَاتِنَا مَا وَعَدتَّنَا عَلَىٰ رُسُلِكَ وَلَا تُخْزِنَا يَوْمَ ٱلْقِيَـٰمَةِۚ إِنَّكَ لَا تُخْلِفُ ٱلْمِيعَادَ ﴿194﴾
اے ہمارے رب! ہمیں وہ سب عطا فرما جو تو نے اپنے رسولوں کے ذریعے وعدہ فرمایا ہے، اور ہمیں قیامت کے دن رسوا نہ کرنا، بے شک تو وعدہ خلافی نہیں کرتا۔
فَٱسْتَجَابَ لَهُمْ رَبُّهُمْ أَنِّي لَآ أُضِيعُ عَمَلَ عَامِلٖ مِّنكُم مِّن ذَكَرٍ أَوْ أُنثَىٰۖ بَعْضُكُم مِّنۢ بَعْضٖۖ فَٱلَّذِينَ هَاجَرُوا۟ وَأُخْرِجُوا۟ مِن دِيَـٰرِهِمْ وَأُوذُوا۟ فِي سَبِيلِي وَقَـٰتَلُوا۟ وَقُتِلُوا۟ لَأُكَفِّرَنَّ عَنْهُمْ سَيِّـَٔاتِهِمْ وَلَأُدْخِلَنَّهُمْ جَنَّـٰتٖ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُۖ ثَوَابٗا مِّنْ عِندِ ٱللَّهِۗ وَٱللَّهُ عِندَهُۥ حُسْنُ ٱلثَّوَابِ ﴿195﴾
تو ان کے رب نے ان کی دعا قبول کر لی: کہ میں تم میں سے کسی عمل کرنے والے کا عمل ضائع نہیں کرتا، خواہ مرد ہو یا عورت، تم سب ایک دوسرے سے ہو، پس جنہوں نے ہجرت کی، اپنے گھروں سے نکالے گئے، میری راہ میں اذیتیں اٹھائیں، لڑے اور قتل کیے گئے، میں ان کی برائیاں ضرور دور کروں گا اور انہیں ایسی جنتوں میں داخل کروں گا جن کے نیچے نہریں بہتی ہیں، یہ اللہ کی طرف سے انعام ہے، اور اللہ ہی کے پاس بہترین اجر ہے۔
لَا يَغُرَّنَّكَ تَقَلُّبُ ٱلَّذِينَ كَفَرُوا۟ فِي ٱلْبِلَـٰدِ ﴿196﴾
کافروں کا شہروں میں چلنا پھرنا تمہیں ہرگز دھوکے میں نہ ڈالے۔
مَتَـٰعٞ قَلِيلٞ ثُمَّ مَأْوَىٰهُمْ جَهَنَّمُۚ وَبِئْسَ ٱلْمِهَادُ ﴿197﴾
یہ تھوڑا سا فائدہ ہے، پھر ان کا ٹھکانہ جہنم ہے، اور وہ بہت ہی برا ٹھکانہ ہے۔
لَـٰكِنِ ٱلَّذِينَ ٱتَّقَوْا۟ رَبَّهُمْ لَهُمْ جَنَّـٰتٞ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُ خَـٰلِدِينَ فِيهَا نُزُلٗا مِّنْ عِندِ ٱللَّهِۗ وَمَا عِندَ ٱللَّهِ خَيْرٞ لِّلْأَبْرَارِ ﴿198﴾
لیکن جو لوگ اپنے رب سے ڈرتے رہے، ان کے لیے جنتیں ہیں جن کے نیچے نہریں بہتی ہیں، وہ ان میں ہمیشہ رہیں گے، یہ اللہ کی طرف سے مہمانی ہے، اور جو کچھ اللہ کے پاس ہے وہ نیک لوگوں کے لیے بہتر ہے۔
وَإِنَّ مِنْ أَهْلِ ٱلْكِتَـٰبِ لَمَن يُؤْمِنُ بِٱللَّهِ وَمَآ أُنزِلَ إِلَيْكُمْ وَمَآ أُنزِلَ إِلَيْهِمْ خَـٰشِعِينَ لِلَّهِ لَا يَشْتَرُونَ بِـَٔايَـٰتِ ٱللَّهِ ثَمَنٗا قَلِيلًاۗ أُو۟لَـٰٓئِكَ لَهُمْ أَجْرُهُمْ عِندَ رَبِّهِمْۗ إِنَّ ٱللَّهَ سَرِيعُ ٱلْحِسَابِ ﴿199﴾
اور بے شک اہلِ کتاب میں کچھ ایسے بھی ہیں جو اللہ پر ایمان رکھتے ہیں، اور اس پر بھی جو تم پر نازل کیا گیا اور جو ان پر نازل کیا گیا، وہ اللہ کے آگے جھکنے والے ہیں، اور اللہ کی آیات کے بدلے تھوڑی قیمت نہیں لیتے، یہی لوگ ہیں جن کا اجر ان کے رب کے پاس ہے، بے شک اللہ جلد حساب لینے والا ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ ٱصْبِرُوا۟ وَصَابِرُوا۟ وَرَابِطُوا۟ وَٱتَّقُوا۟ ٱللَّهَ لَعَلَّكُمْ تُفْلِحُونَ ﴿200﴾
اے ایمان والو! صبر کرو، اور صبر میں دوسروں سے بڑھنے کی کوشش کرو، اور محاذ پر جمے رہو، اور اللہ سے ڈرو تاکہ تم فلاح پاؤ۔
