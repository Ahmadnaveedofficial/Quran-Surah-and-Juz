سورة فاطر

بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
ٱلْحَمْدُ لِلَّهِ فَاطِرِ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ جَاعِلِ ٱلْمَلَـٰٓئِكَةِ رُسُلًا أُوْلِيٓ أَجْنِحَةٖ مَّثْنَىٰ وَثُلَـٰثَ وَرُبَـٰعَۚ يَزِيدُ فِي ٱلْخَلْقِ مَا يَشَآءُۚ إِنَّ ٱللَّهَ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٌ ﴿1﴾
سب تعریف اللہ کے لیے ہے جو آسمانوں اور زمین کا پیدا کرنے والا ہے، جو فرشتوں کو پیغام رساں بناتا ہے، جن کے دو دو، تین تین، چار چار پر ہیں۔ وہ پیدائش میں جو چاہے اضافہ کرتا ہے۔ بے شک اللہ ہر چیز پر قادر ہے۔
مَا يَفْتَحِ ٱللَّهُ لِلنَّاسِ مِن رَّحْمَةٖ فَلَا مُمْسِكَ لَهَاۖ وَمَا يُمْسِكْ فَلَا مُرْسِلَ لَهُۥ مِنۢ بَعْدِهِۦۚ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿2﴾
اللہ جو رحمت لوگوں کے لیے کھول دے، اسے کوئی روکنے والا نہیں، اور جسے وہ روک لے، تو اس کے بعد کوئی اسے بھیجنے والا نہیں۔ اور وہی زبردست، حکمت والا ہے۔
يَـٰٓأَيُّهَا ٱلنَّاسُ ٱذْكُرُوا۟ نِعْمَتَ ٱللَّهِ عَلَيْكُمْۚ هَلْ مِنْ خَـٰلِقٍ غَيْرُ ٱللَّهِ يَرْزُقُكُم مِّنَ ٱلسَّمَآءِ وَٱلْأَرْضِۚ لَآ إِلَـٰهَ إِلَّا هُوَۖ فَأَنَّىٰ تُؤْفَكُونَ ﴿3﴾
اے لوگو! اللہ کی نعمت کو یاد کرو جو تم پر ہے۔ کیا اللہ کے سوا کوئی اور خالق ہے جو تمہیں آسمان اور زمین سے رزق دیتا ہے؟ کوئی معبود نہیں مگر وہی، تو تم کہاں بہکائے جا رہے ہو؟
وَإِن يُكَذِّبُوكَ فَقَدْ كُذِّبَتْ رُسُلٞ مِّن قَبْلِكَۖ وَإِلَى ٱللَّهِ تُرْجَعُ ٱلْأُمُورُ ﴿4﴾
اور اگر وہ آپ کو جھٹلاتے ہیں تو آپ سے پہلے بھی رسول جھٹلائے گئے، اور تمام امور اللہ ہی کی طرف لوٹائے جاتے ہیں۔
يَـٰٓأَيُّهَا ٱلنَّاسُ إِنَّ وَعْدَ ٱللَّهِ حَقّٞۖ فَلَا تَغُرَّنَّكُمُ ٱلْحَيَوٰةُ ٱلدُّنْيَا وَلَا يَغُرَّنَّكُم بِٱللَّهِ ٱلْغَرُورُ ﴿5﴾
اے لوگو! بے شک اللہ کا وعدہ سچا ہے، تو دنیا کی زندگی تمہیں دھوکے میں نہ ڈالے، اور نہ ہی دھوکہ دینے والا تمہیں اللہ کے بارے میں دھوکے میں ڈالے۔
إِنَّ ٱلشَّيْطَـٰنَ لَكُمْ عَدُوّٞ فَٱتَّخِذُوهُ عَدُوًّاۚ إِنَّمَا يَدْعُوا۟ حِزْبَهُۥ لِيَكُونُوا۟ مِنْ أَصْحَـٰبِ ٱلسَّعِيرِ ﴿6﴾
بے شک شیطان تمہارا دشمن ہے، تو اسے دشمن ہی بناؤ۔ وہ تو اپنے گروہ کو صرف اس لیے بلاتا ہے کہ وہ بھڑکتی آگ والوں میں سے ہو جائیں۔
ٱلَّذِينَ كَفَرُوا۟ لَهُمْ عَذَابٞ شَدِيدٞۖ وَٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ لَهُم مَّغْفِرَةٞ وَأَجْرٞ كَبِيرٌ ﴿7﴾
جنہوں نے کفر کیا ان کے لیے سخت عذاب ہے، اور جنہوں نے ایمان لایا اور نیک عمل کیے ان کے لیے بخشش اور بڑا اجر ہے۔
أَفَمَن زُيِّنَ لَهُۥ سُوٓءُ عَمَلِهِۦ فَرَءَاهُ حَسَنٗاۖ فَإِنَّ ٱللَّهَ يُضِلُّ مَن يَشَآءُ وَيَهْدِي مَن يَشَآءُۖ فَلَا تَذْهَبْ نَفْسُكَ عَلَيْهِمْ حَسَرَٰتٍۚ إِنَّ ٱللَّهَ عَلِيمُۢ بِمَا يَصْنَعُونَ ﴿8﴾
تو کیا وہ شخص جس کے لیے اس کا برا عمل خوشنما بنا دیا گیا، پھر وہ اسے اچھا سمجھتا ہے؟ بے شک اللہ جسے چاہے گمراہ کرتا ہے اور جسے چاہے ہدایت دیتا ہے۔ پس آپ ان پر حسرتیں کرتے ہوئے اپنی جان نہ گھلائیں۔ بے شک اللہ خوب جانتا ہے جو وہ کرتے ہیں۔
وَٱللَّهُ ٱلَّذِيٓ أَرْسَلَ ٱلرِّيَـٰحَ فَتُثِيرُ سَحَابٗا فَسُقْنَـٰهُ إِلَىٰ بَلَدٖ مَّيِّتٖ فَأَحْيَيْنَا بِهِ ٱلْأَرْضَ بَعْدَ مَوْتِهَاۚ كَذَٰلِكَ ٱلنُّشُورُ ﴿9﴾
اور اللہ ہی ہے جس نے ہوائیں بھیجیں، پھر وہ بادل اٹھاتی ہیں، پھر ہم اسے مردہ شہر کی طرف لے جاتے ہیں، پھر ہم اس کے ذریعے زمین کو اس کی موت کے بعد زندہ کرتے ہیں۔ اسی طرح دوبارہ اٹھایا جانا ہے۔
مَن كَانَ يُرِيدُ ٱلْعِزَّةَ فَلِلَّهِ ٱلْعِزَّةُ جَمِيعًاۚ إِلَيْهِ يَصْعَدُ ٱلْكَلِمُ ٱلطَّيِّبُ وَٱلْعَمَلُ ٱلصَّـٰلِحُ يَرْفَعُهُۥۚ وَٱلَّذِينَ يَمْكُرُونَ ٱلسَّيِّـَٔاتِ لَهُمْ عَذَابٞ شَدِيدٞۖ وَمَكْرُ أُوْلَـٰٓئِكَ هُوَ يَبُورُ ﴿10﴾
جو کوئی عزت چاہتا ہے تو اللہ ہی کے لیے ساری عزت ہے۔ اسی کی طرف پاکیزہ باتیں چڑھتی ہیں اور نیک عمل انہیں بلند کرتا ہے۔ اور جو لوگ برے مکر کرتے ہیں ان کے لیے سخت عذاب ہے، اور ان کا مکر برباد ہو جاتا ہے۔
وَٱللَّهُ خَلَقَكُم مِّن تُرَابٖ ثُمَّ مِن نُّطْفَةٖ ثُمَّ جَعَلَكُمْ أَزْوَٰجٗاۚ وَمَا تَحْمِلُ مِنۢ أُنثَىٰ وَلَا تَضَعُ إِلَّا بِعِلْمِهِۦۚ وَمَا يُعَمَّرُ مِن مُّعَمَّرٖ وَلَا يُنقَصُ مِنْ عُمُرِهِۦٓ إِلَّا فِي كِتَـٰبٍۚ إِنَّ ذَٰلِكَ عَلَى ٱللَّهِ يَسِيرٞ ﴿11﴾
اور اللہ ہی نے تمہیں مٹی سے پیدا کیا، پھر نطفہ سے، پھر تمہیں جوڑے بنا دیا۔ اور کوئی عورت حاملہ نہیں ہوتی اور نہ ہی وضع حمل کرتی ہے مگر اس کے علم سے۔ اور نہ کسی عمر والے کو عمر دی جاتی ہے اور نہ اس کی عمر کم کی جاتی ہے مگر ایک کتاب میں۔ بے شک یہ اللہ کے لیے آسان ہے۔
وَمَا يَسْتَوِي ٱلْبَحْرَانِ هَـٰذَا عَذْبٞ فُرَاتٞ سَآئِغٞ شَرَابُهُۥ وَهَـٰذَا مِلْحٌ أُجَاجٞۖ وَمِن كُلّٖ تَأْكُلُونَ لَحْمٗا طَرِيّٗا وَتَسْتَخْرِجُونَ حِلْيَةٗ تَلْبَسُونَهَاۖ وَتَرَى ٱلْفُلْكَ فِيهِ مَوَاخِرَ لِتَبْتَغُوا۟ مِن فَضْلِهِۦ وَلَعَلَّكُمْ تَشْكُرُونَ ﴿12﴾
اور دونوں سمندر برابر نہیں، یہ میٹھا خوشگوار ہے، اس کا پانی پینے کے قابل ہے، اور یہ کھاری کڑوا ہے۔ اور تم دونوں سے تازہ گوشت کھاتے ہو اور زیور نکالتے ہو جسے تم پہنتے ہو۔ اور تم کشتیوں کو اس میں پانی چیرتے ہوئے دیکھتے ہو تاکہ تم اس کے فضل سے رزق تلاش کرو اور تاکہ تم شکر ادا کرو۔
يُولِجُ ٱلَّيْلَ فِي ٱلنَّهَارِ وَيُولِجُ ٱلنَّهَارَ فِي ٱلَّيْلِۚ وَسَخَّرَ ٱلشَّمْسَ وَٱلْقَمَرَ كُلّٞ يَجْرِي لِأَجَلٖ مُّسَمّٗىۚ ذَٰلِكُمُ ٱللَّهُ رَبُّكُمْۖ لَهُ ٱلْمُلْكُۚ وَٱلَّذِينَ تَدْعُونَ مِن دُونِهِۦ مَا يَمْلِكُونَ مِن قِطْمِيرٍ ﴿13﴾
وہ رات کو دن میں داخل کرتا ہے اور دن کو رات میں داخل کرتا ہے۔ اور سورج اور چاند کو مسخر کر دیا، ہر ایک مقررہ وقت تک چلتا ہے۔ یہی اللہ تمہارا رب ہے، اسی کے لیے بادشاہی ہے۔ اور جنہیں تم اس کے سوا پکارتے ہو وہ کھجور کی گٹھلی کے چھلکے کے بھی مالک نہیں۔
إِن تَدْعُوهُمْ لَا يَسْمَعُوا۟ دُعَآءَكُمْۖ وَلَوْ سَمِعُوا۟ مَا ٱسْتَجَابُوا۟ لَكُمْۖ وَيَوْمَ ٱلْقِيَـٰمَةِ يَكْفُرُونَ بِشِرْكِكُمْۚ وَلَا يُنَبِّئُكَ مِثْلُ خَبِيرٖ ﴿14﴾
اگر تم انہیں پکارو تو وہ تمہاری پکار نہیں سن سکتے، اور اگر سن بھی لیں تو تمہیں جواب نہیں دے سکتے، اور قیامت کے دن تمہارے شرک کا انکار کریں گے۔ اور خبردار کرنے والا ایسا نہیں جیسے جاننے والا۔
يَـٰٓأَيُّهَا ٱلنَّاسُ أَنتُمُ ٱلْفُقَرَآءُ إِلَى ٱللَّهِۖ وَٱللَّهُ هُوَ ٱلْغَنِيُّ ٱلْحَمِيدُ ﴿15﴾
اے لوگو! تم اللہ کے محتاج ہو، اور اللہ ہی بے نیاز، تعریف کے لائق ہے۔
إِن يَشَأْ يُذْهِبْكُمْ وَيَأْتِ بِخَلْقٖ جَدِيدٖ ﴿16﴾
اگر وہ چاہے تو تمہیں لے جائے اور نئی مخلوق لے آئے۔
وَمَا ذَٰلِكَ عَلَى ٱللَّهِ بِعَزِيزٖ ﴿17﴾
اور یہ اللہ کے لیے کچھ مشکل نہیں۔
وَلَا تَزِرُ وَازِرَةٞ وِزْرَ أُخْرَىٰۚ وَإِن تَدْعُ مُثْقَلَةٌ إِلَىٰ حِمْلِهَا لَا يُحْمَلْ مِنْهُ شَيْءٞ وَلَوْ كَانَ ذَا قُرْبَىٰٓۚ إِنَّمَا تُنذِرُ ٱلَّذِينَ يَخْشَوْنَ رَبَّهُم بِٱلْغَيْبِ وَأَقَامُوا۟ ٱلصَّلَوٰةَۚ وَمَن تَزَكَّىٰ فَإِنَّمَا يَتَزَكَّىٰ لِنَفْسِهِۦۚ وَإِلَى ٱللَّهِ ٱلْمَصِيرُ ﴿18﴾
کوئی بوجھ اٹھانے والا دوسرے کا بوجھ نہیں اٹھائے گا۔ اور اگر کوئی بھاری بوجھ والا اپنے بوجھ کے لیے پکارے، تو اس میں سے کچھ بھی نہ اٹھایا جائے گا، اگرچہ وہ قریبی رشتہ دار ہی ہو۔ آپ تو صرف ان لوگوں کو خبردار کرتے ہیں جو اپنے رب سے غیب میں ڈرتے ہیں اور نماز قائم کرتے ہیں۔ اور جو پاکیزگی اختیار کرتا ہے، وہ اپنے ہی فائدے کے لیے پاکیزگی اختیار کرتا ہے۔ اور اللہ ہی کی طرف لوٹنا ہے۔
وَمَا يَسْتَوِي ٱلْأَعْمَىٰ وَٱلْبَصِيرُ ﴿19﴾
اندھا اور دیکھنے والا برابر نہیں۔
وَلَا ٱلظُّلُمَـٰتُ وَلَا ٱلنُّورُ ﴿20﴾
اور نہ اندھیرے اور نہ روشنی۔
وَلَا ٱلظِّلُّ وَلَا ٱلْحَرُورُ ﴿21﴾
اور نہ سایہ اور نہ گرمی۔
وَمَا يَسْتَوِي ٱلْأَحْيَآءُ وَلَا ٱلْأَمْوَٰتُۚ إِنَّ ٱللَّهَ يُسْمِعُ مَن يَشَآءُۖ وَمَآ أَنتَ بِمُسْمِعٖ مَّن فِي ٱلْقُبُورِ ﴿22﴾
اور زندہ اور مردہ برابر نہیں۔ بے شک اللہ جسے چاہے سنوا دیتا ہے، اور آپ ان کو نہیں سنا سکتے جو قبروں میں ہیں۔
إِنْ أَنتَ إِلَّا نَذِيرٌ ﴿23﴾
آپ تو صرف خبردار کرنے والے ہیں۔
إِنَّآ أَرْسَلْنَـٰكَ بِٱلْحَقِّ بَشِيرٗا وَنَذِيرٗاۚ وَإِن مِّنْ أُمَّةٍ إِلَّا خَلَا فِيهَا نَذِيرٞ ﴿24﴾
بے شک ہم نے آپ کو حق کے ساتھ خوشخبری دینے والا اور ڈرانے والا بنا کر بھیجا ہے، اور کوئی امت ایسی نہیں جس میں کوئی ڈرانے والا نہ آیا ہو۔
وَإِن يُكَذِّبُوكَ فَقَدْ كَذَّبَ ٱلَّذِينَ مِن قَبْلِهِمْ جَآءَتْهُمْ رُسُلُهُم بِٱلْبَيِّنَـٰتِ وَبِٱلزُّبُرِ وَبِٱلْكِتَـٰبِ ٱلْمُنِيرِ ﴿25﴾
اور اگر وہ آپ کو جھٹلاتے ہیں تو ان سے پہلے والوں نے بھی جھٹلایا، ان کے پاس ان کے رسول واضح دلائل، صحیفے اور روشن کتاب لے کر آئے۔
ثُمَّ أَخَذْتُ ٱلَّذِينَ كَفَرُوا۟ فَكَيْفَ كَانَ نَكِيرِ ﴿26﴾
پھر میں نے کافروں کو پکڑ لیا، تو کیسا تھا میرا انکار؟
أَلَمْ تَرَ أَنَّ ٱللَّهَ أَنزَلَ مِنَ ٱلسَّمَآءِ مَآءٗ فَأَخْرَجْنَا بِهِۦ ثَمَرَٰتٖ مُّخْتَلِفًا أَلْوَٰنُهَاۚ وَمِنَ ٱلْجِبَالِ جُدَدٌۭ بِيضٞ وَحُمْرٞ مُّخْتَلِفٌ أَلْوَٰنُهَا وَغَرَابِيبُ سُودٞ ﴿27﴾
کیا آپ نے نہیں دیکھا کہ اللہ نے آسمان سے پانی نازل کیا، پھر ہم نے اس سے مختلف رنگوں کے پھل نکالے؟ اور پہاڑوں میں سفید، سرخ اور مختلف رنگوں کی دھاریاں ہیں، اور کچھ گہرے سیاہ ہیں۔
وَمِنَ ٱلنَّاسِ وَٱلدَّوَآبِّ وَٱلْأَنْعَـٰمِ مُخْتَلِفٌ أَلْوَٰنُهُۥۚ إِنَّمَا يَخْشَى ٱللَّهَ مِنْ عِبَادِهِ ٱلْعُلَمَـٰٓؤُا۟ۗ إِنَّ ٱللَّهَ عَزِيزٌ غَفُورٌ ﴿28﴾
اور انسانوں، جانوروں اور چوپایوں میں بھی مختلف رنگ ہیں۔ اللہ سے اس کے بندوں میں صرف علم والے ہی ڈرتے ہیں۔ بے شک اللہ زبردست، بخشنے والا ہے۔
إِنَّ ٱلَّذِينَ يَتْلُونَ كِتَـٰبَ ٱللَّهِ وَأَقَامُوا۟ ٱلصَّلَوٰةَ وَأَنفَقُوا۟ مِمَّا رَزَقْنَـٰهُمْ سِرّٗا وَعَلَانِيَةٗ يَرْجُونَ تِجَـٰرَةٗ لَّن تَبُورَ ﴿29﴾
بے شک جو لوگ اللہ کی کتاب پڑھتے ہیں، نماز قائم کرتے ہیں، اور جو کچھ ہم نے انہیں دیا ہے اس میں سے چھپ کر اور ظاہر خرچ کرتے ہیں، وہ ایسی تجارت کی امید رکھتے ہیں جو کبھی خسارے میں نہیں جائے گی۔
لِيُوَفِّيَهُمْ أُجُورَهُمْ وَيَزِيدَهُم مِّن فَضْلِهِۦٓۚ إِنَّهُۥ غَفُورٞ شَكُورٞ ﴿30﴾
تاکہ وہ انہیں ان کے اجر پورے دے اور اپنے فضل سے انہیں زیادہ دے۔ بے شک وہ بخشنے والا، قدر دان ہے۔
وَٱلَّذِيٓ أَوْحَيْنَآ إِلَيْكَ مِنَ ٱلْكِتَـٰبِ هُوَ ٱلْحَقُّ مُصَدِّقٗا لِّمَا بَيْنَ يَدَيْهِۚ إِنَّ ٱللَّهَ بِعِبَادِهِۦ لَخَبِيرُۢ بَصِيرٞ ﴿31﴾
اور وہ کتاب جو ہم نے آپ کی طرف وحی کی ہے، وہی حق ہے، اپنے سے پہلے کی تصدیق کرنے والی ہے۔ بے شک اللہ اپنے بندوں سے خوب خبردار، خوب دیکھنے والا ہے۔
ثُمَّ أَوْرَثْنَا ٱلْكِتَـٰبَ ٱلَّذِينَ ٱصْطَفَيْنَا مِنْ عِبَادِنَاۖ فَمِنْهُمْ ظَالِمٞ لِّنَفْسِهِۦ وَمِنْهُم مُّقْتَصِدٞ وَمِنْهُمْ سَابِقُۢ بِٱلْخَيْرَٰتِ بِإِذْنِ ٱللَّهِۚ ذَٰلِكَ هُوَ ٱلْفَضْلُ ٱلْكَبِيرُ ﴿32﴾
پھر ہم نے اپنے بندوں میں سے چنے ہوئے لوگوں کو کتاب کا وارث بنایا، تو ان میں سے کوئی اپنے نفس پر ظلم کرنے والا ہے، اور کوئی درمیانی راہ پر ہے، اور کوئی اللہ کے حکم سے نیکیوں میں سبقت کرنے والا ہے۔ یہی بڑا فضل ہے۔
جَنَّـٰتُ عَدْنٖ يَدْخُلُونَهَا يُحَلَّوْنَ فِيهَا مِنْ أَسَاوِرَ مِن ذَهَبٖ وَلُؤْلُؤٗاۖ وَلِبَاسُهُمْ فِيهَا حَرِيرٞ ﴿33﴾
ہمیشگی کے باغات جن میں وہ داخل ہوں گے، انہیں وہاں سونے اور موتیوں کے کنگن پہنائے جائیں گے، اور ان کا لباس وہاں ریشم ہوگا۔
وَقَالُوا۟ ٱلْحَمْدُ لِلَّهِ ٱلَّذِيٓ أَذْهَبَ عَنَّا ٱلْحَزَنَۖ إِنَّ رَبَّنَا لَغَفُورٞ شَكُورٌ ﴿34﴾
اور وہ کہیں گے: سب تعریف اللہ کے لیے ہے جس نے ہم سے غم دور کر دیا۔ بے شک ہمارا رب بخشنے والا، قدر دان ہے۔
ٱلَّذِيٓ أَحَلَّنَا دَارَ ٱلْمُقَامَةِ مِن فَضْلِهِۦ لَا يَمَسُّنَا فِيهَا نَصَبٞ وَلَا يَمَسُّنَا فِيهَا لُغُوبٞ ﴿35﴾
جس نے ہمیں اپنے فضل سے ہمیشہ رہنے والی جگہ میں اتارا، وہاں ہمیں نہ کوئی تھکن پہنچے گی اور نہ کوئی تکلیف۔
وَٱلَّذِينَ كَفَرُوا۟ لَهُمْ نَارُ جَهَنَّمَۖ لَا يُقْضَىٰ عَلَيْهِمْ فَيَمُوتُوا۟ وَلَا يُخَفَّفُ عَنْهُم مِّنْ عَذَابِهَاۚ كَذَٰلِكَ نَجْزِي كُلَّ كَفُورٍ ﴿36﴾
اور جنہوں نے کفر کیا ان کے لیے جہنم کی آگ ہے، نہ ان پر فیصلہ کیا جائے گا کہ وہ مر جائیں، اور نہ ہی ان سے اس کا عذاب ہلکا کیا جائے گا۔ اسی طرح ہم ہر ناشکرے کو بدلہ دیتے ہیں۔
وَهُمْ يَصْطَرِخُونَ فِيهَا رَبَّنَآ أَخْرِجْنَا نَعْمَلْ صَـٰلِحٗا غَيْرَ ٱلَّذِي كُنَّا نَعْمَلُۚ أَوَلَمْ نُعَمِّرْكُم مَّا يَتَذَكَّرُ فِيهِ مَن تَذَكَّرَ وَجَآءَكُمُ ٱلنَّذِيرُۖ فَذُوقُوا۟ فَمَا لِلظَّـٰلِمِينَ مِن نَّصِيرٍ ﴿37﴾
اور وہ اس میں چیخیں گے: اے ہمارے رب! ہمیں نکال لے تاکہ ہم نیک عمل کریں، اس کے علاوہ جو ہم کرتے تھے۔ کیا ہم نے تمہیں اتنی عمر نہیں دی تھی کہ جس میں نصیحت حاصل کرنے والا نصیحت حاصل کر لیتا؟ اور تمہارے پاس خبردار کرنے والا بھی آیا تھا۔ پس اب چکھو، اور ظالموں کے لیے کوئی مددگار نہیں۔
إِنَّ ٱللَّهَ عَـٰلِمُ غَيْبِ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۚ إِنَّهُۥ عَلِيمُۢ بِذَاتِ ٱلصُّدُورِ ﴿38﴾
بے شک اللہ آسمانوں اور زمین کے غیب کا جاننے والا ہے، بے شک وہ سینوں کے رازوں کو خوب جاننے والا ہے۔
هُوَ ٱلَّذِي جَعَلَكُمْ خَلَـٰٓئِفَ فِي ٱلْأَرْضِۖ فَمَن كَفَرَ فَعَلَيْهِ كُفْرُهُۥۖ وَلَا يَزِيدُ ٱلْكَـٰفِرِينَ كُفْرُهُمْ عِندَ رَبِّهِمْ إِلَّا مَقْتٗاۖ وَلَا يَزِيدُ ٱلْكَـٰفِرِينَ كُفْرُهُمْ إِلَّا خَسَارٗا ﴿39﴾
وہی ہے جس نے تمہیں زمین میں جانشین بنایا، پس جس نے کفر کیا تو اس کا کفر اسی پر ہے، اور کافروں کا کفر ان کے رب کے نزدیک صرف ناراضی میں اضافہ کرتا ہے، اور ان کا کفر صرف نقصان میں اضافہ کرتا ہے۔
قُلْ أَرَءَيْتُمْ شُرَكَآءَكُمُ ٱلَّذِينَ تَدْعُونَ مِن دُونِ ٱللَّهِ أَرُونِي مَاذَا خَلَقُوا۟ مِنَ ٱلْأَرْضِ أَمْ لَهُمْ شِرْكٞ فِي ٱلسَّمَـٰوَٰتِۖ أَمْ ءَاتَيْنَـٰهُمْ كِتَـٰبٗا فَهُمۡ عَلَىٰ بَيِّنَتٖ مِّنْهُۚ بَلْ إِن يَعِدُ ٱلظَّـٰلِمُونَ بَعْضُهُم بَعْضًا إِلَّا غُرُورًا ﴿40﴾
کہہ دیجیے: کیا تم نے اپنے شریکوں کو دیکھا جنہیں تم اللہ کے سوا پکارتے ہو؟ مجھے دکھاؤ کہ انہوں نے زمین میں کیا پیدا کیا ہے، یا ان کا آسمانوں میں کوئی حصہ ہے؟ یا ہم نے انہیں کوئی کتاب دی ہے کہ وہ اس کی دلیل پر ہوں؟ بلکہ ظالم ایک دوسرے سے صرف دھوکے کا وعدہ کرتے ہیں۔
إِنَّ ٱللَّهَ يُمْسِكُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ أَن تَزُولَاۚ وَلَئِن زَالَتَآ إِنْ أَمْسَكَهُمَا مِنْ أَحَدٖ مِّنۢ بَعْدِهِۦٓۚ إِنَّهُۥ كَانَ حَلِيمٗا غَفُورٗا ﴿41﴾
بے شک اللہ آسمانوں اور زمین کو تھامے ہوئے ہے کہ وہ ٹل نہ جائیں، اور اگر وہ ٹل جائیں تو ان کے بعد کوئی انہیں تھامنے والا نہیں۔ بے شک وہ بردبار، بخشنے والا ہے۔
وَأَقْسَمُوا۟ بِٱللَّهِ جَهْدَ أَيْمَـٰنِهِمْ لَئِن جَآءَهُمْ نَذِيرٞ لَّيَكُونُنَّ أَهْدَىٰ مِنْ إِحْدَى ٱلْأُمَمِۖ فَلَمَّا جَآءَهُمْ نَذِيرٞ مَّا زَادَهُمْ إِلَّا نُفُورٗا ﴿42﴾
اور انہوں نے اللہ کی سخت قسمیں کھائیں کہ اگر ان کے پاس کوئی خبردار کرنے والا آئے تو وہ ضرور سب امتوں سے زیادہ ہدایت یافتہ ہوں گے۔ پھر جب ان کے پاس خبردار کرنے والا آیا تو اس نے انہیں صرف نفرت ہی میں بڑھایا۔
ٱسْتِكْبَارٗا فِي ٱلْأَرْضِ وَمَكْرَ ٱلسَّيِّئِۚ وَلَا يَحِيقُ ٱلْمَكْرُ ٱلسَّيِّئُ إِلَّا بِأَهْلِهِۦۚ فَهَلْ يَنظُرُونَ إِلَّا سُنَّتَ ٱلْأَوَّلِينَۚ فَلَن تَجِدَ لِسُنَّتِ ٱللَّهِ تَبْدِيلٗاۖ وَلَن تَجِدَ لِسُنَّتِ ٱللَّهِ تَحْوِيلًا ﴿43﴾
زمین میں تکبر اور برے مکر کی وجہ سے۔ اور برا مکر صرف اسی پر پڑتا ہے جو اس کا مرتکب ہو۔ تو کیا وہ صرف پہلے لوگوں کی سنت کا انتظار کر رہے ہیں؟ پس آپ اللہ کی سنت میں کوئی تبدیلی نہ پائیں گے، اور نہ ہی اللہ کی سنت میں کوئی تبدیلی پائیں گے۔
أَوَلَمْ يَسِيرُوا۟ فِي ٱلْأَرْضِ فَيَنظُرُوا۟ كَيْفَ كَانَ عَـٰقِبَةُ ٱلَّذِينَ كَانُوا۟ مِن قَبْلِهِمْ وَكَانُوا۟ أَشَدَّ مِنْهُمْ قُوَّةٗۚ وَمَا كَانَ ٱللَّهُ لِيُعْجِزَهُۥ مِن شَيْءٖ فِي ٱلسَّمَـٰوَٰتِ وَلَا فِي ٱلْأَرْضِۚ إِنَّهُۥ كَانَ عَلِيمٗا قَدِيرٗا ﴿44﴾
کیا انہوں نے زمین میں سفر نہیں کیا تاکہ دیکھیں کہ ان سے پہلے والوں کا انجام کیسا ہوا؟ حالانکہ وہ ان سے زیادہ قوت والے تھے۔ اور اللہ کو آسمانوں اور زمین میں کوئی چیز عاجز نہیں کر سکتی۔ بے شک وہ علم والا، قدرت والا ہے۔
وَلَوْ يُؤَاخِذُ ٱللَّهُ ٱلنَّاسَ بِمَا كَسَبُوا۟ مَا تَرَكَ عَلَىٰ ظَهْرِهَا مِن دَآبَّةٖ وَلَـٰكِن يُؤَخِّرُهُمْ إِلَىٰٓ أَجَلٖ مُّسَمّٗىۖ فَإِذَا جَآءَ أَجَلُهُمْ فَإِنَّ ٱللَّهَ كَانَ بِعِبَادِهِۦ بَصِيرَۢا ﴿45﴾
اور اگر اللہ لوگوں کو ان کے اعمال پر پکڑتا تو زمین کی پشت پر کوئی جاندار باقی نہ رہتا، لیکن وہ انہیں ایک مقرر وقت تک مہلت دیتا ہے۔ پھر جب ان کا وقت آ جاتا ہے تو بے شک اللہ اپنے بندوں کو خوب دیکھنے والا ہے۔
