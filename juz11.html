11 پارہ


بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
إِنَّمَا ٱلسَّبِيلُ عَلَى ٱلَّذِينَ يَسْتَـْٔذِنُونَكَ وَهُمْ أَغْنِيَآءُۚ رَضُوا۟ بِأَن يَكُونُوا۟ مَعَ ٱلْخَوَالِفِۖ وَطَبَعَ ٱللَّهُ عَلَىٰ قُلُوبِهِمْ فَهُمْ لَا يَعْلَمُونَ ﴿93﴾
الزام تو ان پر ہے جو تم سے اجازت مانگتے ہیں حالانکہ وہ مالدار ہیں، وہ اس پر راضی ہو گئے کہ پیچھے رہنے والوں کے ساتھ رہیں، اور اللہ نے ان کے دلوں پر مہر لگا دی ہے، پس وہ نہیں جانتے۔
يَعْتَذِرُونَ إِلَيْكُمْ إِذَا رَجَعْتُمْ إِلَيْهِمْۚ قُل لَّا تَعْتَذِرُوا۟ لَن نُّؤْمِنَ لَكُمْۖ قَدْ نَبَّأَنَا ٱللَّهُ مِنْ أَخْبَارِكُمْۚ وَسَيَرَى ٱللَّهُ عَمَلَكُمْ وَرَسُولُهُۥۖ ثُمَّ تُرَدُّونَ إِلَىٰ عَـٰلِمِ ٱلْغَيْبِ وَٱلشَّهَـٰدَةِ فَيُنَبِّئُكُم بِمَا كُنتُمْ تَعْمَلُونَ ﴿94﴾
جب تم ان کے پاس واپس جاؤ گے تو وہ تم سے عذر پیش کریں گے، کہہ دو: عذر مت پیش کرو، ہم تم پر یقین نہیں کریں گے، اللہ نے ہمیں تمہاری خبریں دے دی ہیں، اور اللہ اور اس کا رسول تمہارے عمل کو دیکھے گا، پھر تم غیب اور ظاہر کے جاننے والے کی طرف لوٹائے جاؤ گے، پس وہ تمہیں بتائے گا جو تم کرتے رہے۔
سَيَحْلِفُونَ بِٱللَّهِ لَكُمْ إِذَا ٱنقَلَبْتُمْ إِلَيْهِمْ لِتُعْرِضُوا۟ عَنْهُمْۖ فَأَعْرِضُوا۟ عَنْهُمْۖ إِنَّهُمْ رِجْسٞۖ وَمَأْوَىٰهُمْ جَهَنَّمُۖ جَزَآءٗۢ بِمَا كَانُوا۟ يَكْسِبُونَ ﴿95﴾
جب تم ان کے پاس واپس جاؤ گے تو وہ اللہ کی قسمیں کھائیں گے تاکہ تم ان سے منہ پھیر لو، پس تم ان سے منہ پھیر لو، بے شک وہ ناپاک ہیں، اور ان کا ٹھکانہ جہنم ہے، اس بدلے میں جو وہ کماتے رہے۔
يَحْلِفُونَ لَكُمْ لِتَرْضَوْا۟ عَنْهُمْۖ فَإِن تَرْضَوْا۟ عَنْهُمْ فَإِنَّ ٱللَّهَ لَا يَرْضَىٰ عَنِ ٱلْقَوْمِ ٱلْفَـٰسِقِينَ ﴿96﴾
وہ تمہیں راضی کرنے کے لیے قسمیں کھاتے ہیں، پس اگر تم ان سے راضی ہو بھی جاؤ تو بے شک اللہ فاسق قوم سے راضی نہیں ہوتا۔
ٱلْأَعْرَابُ أَشَدُّ كُفْرٗا وَنِفَاقٗا وَأَجْدَرُ أَلَّا يَعْلَمُوا۟ حُدُودَ مَآ أَنزَلَ ٱللَّهُ عَلَىٰ رَسُولِهِۦۗ وَٱللَّهُ عَلِيمٌ حَكِيمٞ ﴿97﴾
دیہاتی لوگ کفر و نفاق میں زیادہ سخت ہیں، اور اس بات کے زیادہ لائق ہیں کہ وہ ان حدود کو نہ جانیں جو اللہ نے اپنے رسول پر نازل کی ہیں، اور اللہ سب کچھ جاننے والا، حکمت والا ہے۔
وَمِنَ ٱلْأَعْرَابِ مَن يَتَّخِذُ مَا يُنفِقُ مَغْرَمٗا وَيَتَرَبَّصُ بِكُمُ ٱلدَّوَآئِرَۚ عَلَيْهِمْ دَآئِرَةُ ٱلسَّوْءِۗ وَٱللَّهُ سَمِيعٌ عَلِيمٞ ﴿98﴾
اور دیہاتیوں میں سے کچھ ایسے ہیں جو جو کچھ خرچ کرتے ہیں اسے تاوان سمجھتے ہیں، اور تم پر گردشِ زمانہ کے منتظر رہتے ہیں، ان پر بری گردش آئے، اور اللہ سننے والا، جاننے والا ہے۔
وَمِنَ ٱلْأَعْرَابِ مَن يُؤْمِنُ بِٱللَّهِ وَٱلْيَوْمِ ٱلْـَٔاخِرِ وَيَتَّخِذُ مَا يُنفِقُ قُرُبَـٰتٍ عِندَ ٱللَّهِ وَصَلَوَٰتِ ٱلرَّسُولِۚ أَلَآ إِنَّهَا قُرْبَةٞ لَّهُمْۚ سَيُدْخِلُهُمُ ٱللَّهُ فِي رَحْمَتِهِۦٓۚ إِنَّ ٱللَّهَ غَفُورٞ رَّحِيمٌ ﴿99﴾
اور دیہاتیوں میں سے کچھ ایسے بھی ہیں جو اللہ اور آخرت کے دن پر ایمان رکھتے ہیں، اور جو کچھ خرچ کرتے ہیں اسے اللہ کے نزدیک قربت اور رسول کی دعاؤں کا ذریعہ سمجھتے ہیں، خبردار! بے شک وہ ان کے لیے قربت ہے، عنقریب اللہ انہیں اپنی رحمت میں داخل کرے گا، بے شک اللہ بخشنے والا، رحم کرنے والا ہے۔
وَٱلسَّـٰبِقُونَ ٱلْأَوَّلُونَ مِنَ ٱلْمُهَـٰجِرِينَ وَٱلْأَنصَارِ وَٱلَّذِينَ ٱتَّبَعُوهُم بِإِحْسَـٰنٖ رَّضِيَ ٱللَّهُ عَنْهُمْ وَرَضُوا۟ عَنْهُۖ وَأَعَدَّ لَهُمْ جَنَّـٰتٗ تَجْرِي تَحْتَهَا ٱلْأَنْهَـٰرُ خَـٰلِدِينَ فِيهَآ أَبَدٗاۚ ذَٰلِكَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿100﴾
اور مہاجرین اور انصار میں سے سبقت لے جانے والے پہلے لوگ، اور وہ جنہوں نے نیکی کے ساتھ ان کی پیروی کی، اللہ ان سے راضی ہوا اور وہ اللہ سے راضی ہوئے، اور اس نے ان کے لیے ایسی جنتیں تیار کیں جن کے نیچے نہریں بہتی ہیں، جن میں وہ ہمیشہ ہمیشہ رہیں گے، یہی ہے بڑی کامیابی۔
وَمِمَّنْ حَوْلَكُم مِّنَ ٱلْأَعْرَابِ مُنَـٰفِقُونَ وَمِنْ أَهْلِ ٱلْمَدِينَةِۖ مَرَدُوا۟ عَلَى ٱلنِّفَاقِ لَا تَعْلَمُهُمْۖ نَّحْنُ نَعْلَمُهُمْۚ سَنُعَذِّبُهُم مَّرَّتَيْنِ ثُمَّ يُرَدُّونَ إِلَىٰ عَذَابٍ عَظِيمٖ ﴿101﴾
اور تمہارے گرد و نواح کے دیہاتیوں میں اور مدینہ والوں میں بھی منافق ہیں، جو نفاق پر جمے ہوئے ہیں، تم انہیں نہیں جانتے، ہم انہیں جانتے ہیں، ہم انہیں دو بار عذاب دیں گے، پھر وہ بڑے عذاب کی طرف لوٹائے جائیں گے۔
وَءَاخَرُونَ ٱعْتَرَفُوا۟ بِذُنُوبِهِمْ خَلَطُوا۟ عَمَلٗا صَـٰلِحٗا وَءَاخَرَ سَيِّئًا عَسَى ٱللَّهُ أَن يَتُوبَ عَلَيْهِمْۚ إِنَّ ٱللَّهَ غَفُورٞ رَّحِيمٌ ﴿102﴾
اور کچھ دوسرے ہیں جنہوں نے اپنے گناہوں کا اعتراف کیا، انہوں نے نیک عمل کو برے عمل کے ساتھ ملا دیا، امید ہے کہ اللہ ان پر توبہ قبول کرے گا، بے شک اللہ بخشنے والا، رحم کرنے والا ہے۔
خُذْ مِنْ أَمْوَٰلِهِمْ صَدَقَةٗ تُطَهِّرُهُمْ وَتُزَكِّيهِم بِهَا وَصَلِّ عَلَيْهِمْۖ إِنَّ صَلَوٰتَكَ سَكَنٞ لَّهُمْۗ وَٱللَّهُ سَمِيعٌ عَلِيمٌ ﴿103﴾
ان کے مالوں میں سے صدقہ لے لو تاکہ تم انہیں پاک کرو اور ان کا تزکیہ کرو، اور ان کے لیے دعا کرو، بے شک تمہاری دعا ان کے لیے سکون ہے، اور اللہ سننے والا، جاننے والا ہے۔
أَلَمْ يَعْلَمُوٓا۟ أَنَّ ٱللَّهَ هُوَ يَقْبَلُ ٱلتَّوْبَةَ عَنْ عِبَادِهِۦ وَيَأْخُذُ ٱلصَّدَقَـٰتِ وَأَنَّ ٱللَّهَ هُوَ ٱلتَّوَّابُ ٱلرَّحِيمُ ﴿104﴾
کیا وہ نہیں جانتے کہ اللہ ہی اپنے بندوں سے توبہ قبول کرتا ہے اور صدقات لیتا ہے، اور بے شک اللہ ہی توبہ قبول کرنے والا، رحم کرنے والا ہے؟
وَقُلِ ٱعْمَلُوا۟ فَسَيَرَى ٱللَّهُ عَمَلَكُمْ وَرَسُولُهُۥ وَٱلْمُؤْمِنُونَۖ وَسَتُرَدُّونَ إِلَىٰ عَـٰلِمِ ٱلْغَيْبِ وَٱلشَّهَـٰدَةِ فَيُنَبِّئُكُم بِمَا كُنتُمْ تَعْمَلُونَ ﴿105﴾
اور کہہ دو: عمل کرو، پس اللہ تمہارے عمل کو دیکھے گا، اور اس کا رسول اور مؤمن بھی، اور تم غیب و ظاہر کے جاننے والے کی طرف لوٹائے جاؤ گے، پھر وہ تمہیں بتائے گا جو تم کرتے رہے۔
وَءَاخَرُونَ مُرْجَوْنَ لِأَمْرِ ٱللَّهِ إِمَّا يُعَذِّبُهُمْ وَإِمَّا يَتُوبُ عَلَيْهِمْۗ وَٱللَّهُ عَلِيمٌ حَكِيمٌ ﴿106﴾
اور کچھ دوسرے ہیں جن کا معاملہ اللہ کے فیصلے پر چھوڑا گیا ہے، چاہے وہ انہیں عذاب دے یا ان پر توبہ قبول کرے، اور اللہ سب کچھ جاننے والا، حکمت والا ہے۔
وَٱلَّذِينَ ٱتَّخَذُوا۟ مَسْجِدٗا ضِرَارٗا وَكُفْرٗا وَتَفْرِيقٗا بَيْنَ ٱلْمُؤْمِنِينَ وَإِرْصَادٗا لِّمَنْ حَارَبَ ٱللَّهَ وَرَسُولَهُۥ مِن قَبْلُۚ وَلَيَحْلِفُنَّ إِنْ أَرَدْنَآ إِلَّا ٱلْحُسْنَىٰۖ وَٱللَّهُ يَشْهَدُ إِنَّهُمْ لَكَـٰذِبُونَ ﴿107﴾
اور وہ لوگ جنہوں نے مسجد بنائی نقصان پہنچانے، کفر کرنے، مؤمنوں میں تفرقہ ڈالنے اور اس کے لیے گھات لگانے کی نیت سے جو پہلے اللہ اور اس کے رسول سے جنگ کر چکا ہے، اور وہ ضرور قسمیں کھائیں گے کہ ہمارا ارادہ تو صرف بھلائی کا تھا، اور اللہ گواہی دیتا ہے کہ وہ یقیناً جھوٹے ہیں۔
لَا تَقُمْ فِيهِ أَبَدٗاۚ لَمَسْجِدٌ أُسِّسَ عَلَى ٱلتَّقْوَىٰ مِنْ أَوَّلِ يَوْمٍ أَحَقُّ أَن تَقُومَ فِيهِۚ فِيهِ رِجَالٞ يُحِبُّونَ أَن يَتَطَهَّرُوا۟ۚ وَٱللَّهُ يُحِبُّ ٱلْمُطَّهِّرِينَ ﴿108﴾
اس میں کبھی کھڑے نہ ہونا، وہ مسجد جو پہلے دن سے تقویٰ پر قائم کی گئی ہے، وہ اس لائق ہے کہ تم اس میں کھڑے ہو، اس میں ایسے لوگ ہیں جو پاکیزگی کو پسند کرتے ہیں، اور اللہ پاکیزگی اختیار کرنے والوں کو پسند کرتا ہے۔
أَفَمَنْ أَسَّسَ بُنْيَـٰنَهُۥ عَلَىٰ تَقْوَىٰ مِنَ ٱللَّهِ وَرِضْوَٰنٍ خَيْرٌ أَم مَّنْ أَسَّسَ بُنْيَـٰنَهُۥ عَلَىٰ شَفَا جُرُفٍ هَارٖ فَٱنْهَارَ بِهِۦ فِي نَارِ جَهَنَّمَۗ وَٱللَّهُ لَا يَهْدِي ٱلْقَوْمَ ٱلظَّـٰلِمِينَ ﴿109﴾
کیا وہ شخص بہتر ہے جس نے اپنی عمارت اللہ کے تقویٰ اور رضا پر رکھی، یا وہ جس نے اپنی عمارت ایک گرنے والے کنارے پر رکھی، پھر وہ اسے جہنم کی آگ میں لے گیا؟ اور اللہ ظالم قوم کو ہدایت نہیں دیتا۔
لَا يَزَالُ بُنْيَـٰنُهُمُ ٱلَّذِي بَنَوْا۟ رِيبَةٗ فِي قُلُوبِهِمْ إِلَّآ أَن تَقْطَعَ قُلُوبُهُمْۗ وَٱللَّهُ عَلِيمٌ حَكِيمٌ ﴿110﴾
ان کی وہ عمارت جو انہوں نے بنائی، ان کے دلوں میں شک ہی بنی رہے گی، یہاں تک کہ ان کے دل ٹکڑے ہو جائیں، اور اللہ سب کچھ جاننے والا، حکمت والا ہے۔
إِنَّ ٱللَّهَ ٱشْتَرَىٰ مِنَ ٱلْمُؤْمِنِينَ أَنفُسَهُمْ وَأَمْوَٰلَهُم بِأَنَّ لَهُمُ ٱلْجَنَّةَۚ يُقَـٰتِلُونَ فِي سَبِيلِ ٱللَّهِ فَيَقْتُلُونَ وَيُقْتَلُونَۖ وَعْدًا عَلَيْهِ حَقّٗا فِي ٱلتَّوْرَىٰةِ وَٱلْإِنجِيلِ وَٱلْقُرْءَانِۚ وَمَنْ أَوْفَىٰ بِعَهْدِهِۦ مِنَ ٱللَّهِۚ فَٱسْتَبْشِرُوا۟ بِبَيْعِكُمُ ٱلَّذِي بَايَعْتُم بِهِۦۚ وَذَٰلِكَ هُوَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿111﴾
بے شک اللہ نے مؤمنوں سے ان کی جانیں اور ان کے مال خرید لیے ہیں اس کے بدلے کہ ان کے لیے جنت ہے، وہ اللہ کی راہ میں لڑتے ہیں، پس قتل کرتے ہیں اور قتل کیے جاتے ہیں، یہ وعدہ ہے اللہ پر سچا تورات، انجیل اور قرآن میں، اور اللہ سے زیادہ اپنے وعدے کو پورا کرنے والا کون ہے؟ پس خوش ہو جاؤ اپنے اس سودے پر جو تم نے کیا، اور یہی بڑی کامیابی ہے۔
ٱلتَّـٰٓئِبُونَ ٱلْعَـٰبِدُونَ ٱلْحَـٰمِدُونَ ٱلسَّـٰٓئِحُونَ ٱلرَّ‌ٰكِعُونَ ٱلسَّـٰجِدُونَ ٱلْـَٔامِرُونَ بِٱلْمَعْرُوفِ وَٱلنَّاهُونَ عَنِ ٱلْمُنكَرِ وَٱلْحَـٰفِظُونَ لِحُدُودِ ٱللَّهِۗ وَبَشِّرِ ٱلْمُؤْمِنِينَ ﴿112﴾
(یہ کامیابی) ان لوگوں کے لیے ہے جو توبہ کرنے والے، عبادت کرنے والے، حمد کرنے والے، سفر کرنے والے، رکوع کرنے والے، سجدہ کرنے والے، نیکی کا حکم دینے والے، برائی سے روکنے والے اور اللہ کی حدود کی حفاظت کرنے والے ہیں — اور مؤمنوں کو خوشخبری دے دو۔
مَا كَانَ لِلنَّبِيِّ وَٱلَّذِينَ ءَامَنُوٓا۟ أَن يَسْتَغْفِرُوا۟ لِلْمُشْرِكِينَ وَلَوْ كَانُوٓا۟ أُوْلِي قُرْبَىٰ مِنۢ بَعْدِ مَا تَبَيَّنَ لَهُمْ أَنَّهُمْ أَصْحَـٰبُ ٱلْجَحِيمِ ﴿113﴾
نبی اور ایمان والوں کے لیے جائز نہیں کہ وہ مشرکوں کے لیے بخشش مانگیں، اگرچہ وہ قرابت دار ہی ہوں، بعد اس کے کہ ان پر واضح ہو جائے کہ وہ جہنم والے ہیں۔
وَمَا كَانَ ٱسْتِغْفَارُ إِبْرَٰهِيمَ لِأَبِيهِ إِلَّا عَن مَّوْعِدَةٖ وَعَدَهَآ إِيَّاهُۖ فَلَمَّا تَبَيَّنَ لَهُۥٓ أَنَّهُۥ عَدُوّٞ لِّلَّهِ تَبَرَّأَ مِنْهُۚ إِنَّ إِبْرَٰهِيمَ لَأَوَّاهٌ حَلِيمٞ ﴿114﴾
اور ابراہیم کا اپنے باپ کے لیے استغفار کرنا صرف ایک وعدے کی وجہ سے تھا جو اس نے اس سے کیا تھا، پھر جب اسے معلوم ہو گیا کہ وہ اللہ کا دشمن ہے تو اس سے بیزاری ظاہر کی، بے شک ابراہیم بہت نرم دل، بردبار تھا۔
وَمَا كَانَ ٱللَّهُ لِيُضِلَّ قَوْمَۢا بَعْدَ إِذْ هَدَىٰهُمْ حَتَّىٰ يُبَيِّنَ لَهُم مَّا يَتَّقُونَۚ إِنَّ ٱللَّهَ بِكُلِّ شَيْءٍ عَلِيمٌ ﴿115﴾
اور اللہ ایسا نہیں کہ کسی قوم کو ہدایت دینے کے بعد گمراہ کر دے جب تک کہ وہ ان پر واضح نہ کر دے کہ وہ کس چیز سے بچیں، بے شک اللہ ہر چیز کو جاننے والا ہے۔
إِنَّ ٱللَّهَ لَهُۥ مُلْكُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ يُحْيِۦ وَيُمِيتُۚ وَمَا لَكُم مِّن دُونِ ٱللَّهِ مِن وَلِيّٖ وَلَا نَصِيرٖ ﴿116﴾
بے شک اللہ ہی کے لیے آسمانوں اور زمین کی بادشاہی ہے، وہی زندگی دیتا ہے اور موت دیتا ہے، اور اللہ کے سوا تمہارا نہ کوئی کارساز ہے اور نہ مددگار۔
لَقَد تَّابَ ٱللَّهُ عَلَى ٱلنَّبِيِّ وَٱلْمُهَـٰجِرِينَ وَٱلْأَنصَارِ ٱلَّذِينَ ٱتَّبَعُوهُ فِي سَاعَةِ ٱلْعُسْرَةِ مِنۢ بَعْدِ مَا كَادَ يَزِيغُ قُلُوبُ فَرِيقٖ مِّنْهُمْ ثُمَّ تَابَ عَلَيْهِمْۚ إِنَّهُۥ بِهِمْ رَءُوفٞ رَّحِيمٞ ﴿117﴾
بے شک اللہ نے نبی پر اور مہاجرین و انصار پر توبہ فرمائی، جنہوں نے سختی کے وقت اس کا ساتھ دیا، بعد اس کے کہ ان میں سے ایک گروہ کے دل کج ہونے کے قریب ہو گئے تھے، پھر اللہ نے ان پر توبہ فرمائی، بے شک وہ ان پر بہت مہربان، نہایت رحم والا ہے۔
وَعَلَى ٱلثَّلَـٰثَةِ ٱلَّذِينَ خُلِّفُوا۟ حَتَّىٰٓ إِذَا ضَاقَتْ عَلَيْهِمُ ٱلْأَرْضُ بِمَا رَحُبَتْ وَضَاقَتْ عَلَيْهِمْ أَنفُسُهُمْ وَظَنُّوٓا۟ أَن لَّا مَلْجَأَ مِنَ ٱللَّهِ إِلَّآ إِلَيْهِ ثُمَّ تَابَ عَلَيْهِمْ لِيَتُوبُوٓا۟ۚ إِنَّ ٱللَّهَ هُوَ ٱلتَّوَّابُ ٱلرَّحِيمُ ﴿118﴾
اور ان تینوں پر بھی (اللہ نے توبہ فرمائی) جنہیں پیچھے چھوڑ دیا گیا تھا، یہاں تک کہ ان پر زمین اپنی وسعت کے باوجود تنگ ہو گئی، اور ان کے دل بھی تنگ ہو گئے، اور انہوں نے سمجھ لیا کہ اللہ سے بچنے کی کوئی جگہ نہیں مگر اسی کی طرف، پھر اللہ نے ان پر توبہ فرمائی تاکہ وہ توبہ کریں، بے شک اللہ ہی توبہ قبول کرنے والا، رحم کرنے والا ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ ٱتَّقُوا۟ ٱللَّهَ وَكُونُوا۟ مَعَ ٱلصَّـٰدِقِينَ ﴿119﴾
اے ایمان والو! اللہ سے ڈرو اور سچوں کے ساتھ ہو جاؤ۔
مَا كَانَ لِأَهْلِ ٱلْمَدِينَةِ وَمَنْ حَوْلَهُم مِّنَ ٱلْأَعْرَابِ أَن يَتَخَلَّفُوا۟ عَن رَّسُولِ ٱللَّهِ وَلَا يَرْغَبُوا۟ بِأَنفُسِهِمْ عَن نَّفْسِهِۦۚ ذَٰلِكَ بِأَنَّهُمْ لَا يُصِيبُهُمْ ظَمَأٞ وَلَا نَصَبٞ وَلَا مَخْمَصَةٞ فِي سَبِيلِ ٱللَّهِ وَلَا يَطَـُٔونَ مَوْطِئٗا يَغِيظُ ٱلْكُفَّارَ وَلَا يَنَالُونَ مِنْ عَدُوّٖ نَّيْلًا إِلَّا كُتِبَ لَهُم بِهِۦ عَمَلٞ صَـٰلِحٌۚ إِنَّ ٱللَّهَ لَا يُضِيعُ أَجْرَ ٱلْمُحْسِنِينَ ﴿120﴾
مدینہ والوں اور ان کے گرد و نواح کے دیہاتیوں کے لیے جائز نہیں کہ وہ رسول اللہ سے پیچھے رہ جائیں یا اپنی جانوں کو اس کی جان سے زیادہ عزیز سمجھیں، یہ اس لیے کہ وہ اللہ کی راہ میں جو بھی پیاس، تھکن یا بھوک برداشت کرتے ہیں، یا کوئی ایسی جگہ قدم رکھتے ہیں جو کافروں کو غصہ دلاتی ہے، یا کسی دشمن سے کچھ حاصل کرتے ہیں، تو ان سب پر ان کے لیے نیک عمل لکھا جاتا ہے، بے شک اللہ نیکی کرنے والوں کا اجر ضائع نہیں کرتا۔
وَلَا يُنفِقُونَ نَفَقَةٗ صَغِيرَةٗ وَلَا كَبِيرَةٗ وَلَا يَقْطَعُونَ وَادِيٗا إِلَّا كُتِبَ لَهُمْ لِيَجْزِيَهُمُ ٱللَّهُ أَحْسَنَ مَا كَانُوا۟ يَعْمَلُونَ ﴿121﴾
اور وہ کوئی چھوٹا یا بڑا خرچ نہیں کرتے اور نہ کوئی وادی طے کرتے ہیں مگر ان کے لیے لکھ لیا جاتا ہے تاکہ اللہ انہیں ان کے بہترین اعمال کا بدلہ دے۔
وَمَا كَانَ ٱلْمُؤْمِنُونَ لِيَنفِرُوا۟ كَآفَّةٗۚ فَلَوْلَا نَفَرَ مِن كُلِّ فِرْقَةٖ مِّنْهُمْ طَآئِفَةٞ لِّيَتَفَقَّهُوا۟ فِي ٱلدِّينِ وَلِيُنذِرُوا۟ قَوْمَهُمْ إِذَا رَجَعُوٓا۟ إِلَيْهِمْ لَعَلَّهُمْ يَحْذَرُونَ ﴿122﴾
اور مؤمنوں کے لیے یہ مناسب نہیں کہ سب کے سب نکل کھڑے ہوں، تو کیوں نہ ہر گروہ میں سے کچھ لوگ نکلیں تاکہ دین کی سمجھ حاصل کریں اور جب اپنی قوم کی طرف لوٹیں تو انہیں خبردار کریں تاکہ وہ بچ جائیں۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ قَـٰتِلُوا۟ ٱلَّذِينَ يَلُونَكُم مِّنَ ٱلْكُفَّارِ وَلْيَجِدُوا۟ فِيكُمْ غِلْظَةٗۚ وَٱعْلَمُوٓا۟ أَنَّ ٱللَّهَ مَعَ ٱلْمُتَّقِينَ ﴿123﴾
اے ایمان والو! ان کافروں سے لڑو جو تمہارے قریب ہیں، اور انہیں تم میں سختی پائیں، اور جان لو کہ اللہ متقیوں کے ساتھ ہے۔
وَإِذَآ أُنزِلَتْ سُورَةٞ فَمِنْهُم مَّن يَقُولُ أَيُّكُمْ زَادَتْهُ هَـٰذِهِۦٓ إِيمَـٰنٗاۚ فَأَمَّا ٱلَّذِينَ ءَامَنُوا۟ فَزَادَتْهُمْ إِيمَـٰنٗا وَهُمْ يَسْتَبْشِرُونَ ﴿124﴾
اور جب کوئی سورت نازل کی جاتی ہے تو ان میں سے کچھ کہتے ہیں: تم میں سے کس کا ایمان اس نے بڑھایا؟ تو جو لوگ ایمان لائے ان کا ایمان بڑھا دیا اور وہ خوش ہوتے ہیں۔
وَأَمَّا ٱلَّذِينَ فِي قُلُوبِهِم مَّرَضٞ فَزَادَتْهُمْ رِجْسًا إِلَىٰ رِجْسِهِمْ وَمَاتُوا۟ وَهُمْ كَـٰفِرُونَ ﴿125﴾
اور جن کے دلوں میں بیماری ہے، اس نے ان کی نجاست میں اور اضافہ کر دیا، اور وہ کفر کی حالت میں مر گئے۔
أَوَلَا يَرَوْنَ أَنَّهُمْ يُفْتَنُونَ فِي كُلِّ عَامٖ مَّرَّةً أَوْ مَرَّتَيْنِ ثُمَّ لَا يَتُوبُونَ وَلَا هُمْ يَذَّكَّرُونَ ﴿126﴾
کیا وہ نہیں دیکھتے کہ ہر سال ایک یا دو بار آزمائش میں ڈالے جاتے ہیں، پھر بھی نہ وہ توبہ کرتے ہیں اور نہ نصیحت پکڑتے ہیں؟
وَإِذَا مَآ أُنزِلَتْ سُورَةٞ نَّظَرَ بَعْضُهُمْ إِلَىٰ بَعْضٍ هَلْ يَرَىٰكُم مِّنْ أَحَدٖ ثُمَّ ٱنصَرَفُوا۟ۚ صَرَفَ ٱللَّهُ قُلُوبَهُم بِأَنَّهُمْ قَوْمٞ لَّا يَفْقَهُونَ ﴿127﴾
اور جب کوئی سورت نازل کی جاتی ہے تو وہ ایک دوسرے کی طرف دیکھتے ہیں (کہ) کیا کوئی تمہیں دیکھ رہا ہے؟ پھر وہ پلٹ جاتے ہیں، اللہ نے ان کے دل پھیر دیے کیونکہ وہ ایسے لوگ ہیں جو سمجھتے نہیں۔
لَقَدْ جَآءَكُمْ رَسُولٞ مِّنْ أَنفُسِكُمْ عَزِيزٌ عَلَيْهِ مَا عَنِتُّمْ حَرِيصٌ عَلَيْكُم بِٱلْمُؤْمِنِينَ رَءُوفٞ رَّحِيمٞ ﴿128﴾
بے شک تمہارے پاس تم ہی میں سے ایک رسول آیا ہے، تمہاری تکلیف اس پر گراں گزرتی ہے، وہ تمہاری بھلائی کا خواہاں ہے، مؤمنوں پر بہت شفیق، نہایت مہربان ہے۔
فَإِن تَوَلَّوْا۟ فَقُلْ حَسْبِيَ ٱللَّهُ لَآ إِلَـٰهَ إِلَّا هُوَۖ عَلَيْهِ تَوَكَّلْتُۖ وَهُوَ رَبُّ ٱلْعَرْشِ ٱلْعَظِيمِ ﴿129﴾
پھر اگر وہ منہ موڑیں تو کہہ دو: میرے لیے اللہ کافی ہے، اس کے سوا کوئی معبود نہیں، میں نے اسی پر بھروسہ کیا، اور وہی عرش عظیم کا رب ہے۔

الٓرۚ تِلْكَ ءَايَاتُ ٱلْكِتَـٰبِ ٱلْحَكِيمِ ﴿1﴾
الٓر۔ یہ حکمت والی کتاب کی آیات ہیں۔
أَكَانَ لِلنَّاسِ عَجَبًا أَنْ أَوْحَيْنَآ إِلَىٰ رَجُلٖ مِّنْهُمْ أَنْ أَنذِرِ ٱلنَّاسَ وَبَشِّرِ ٱلَّذِينَ ءَامَنُوٓا۟ أَنَّ لَهُمْ قَدَمَ صِدْقٍ عِندَ رَبِّهِمۗۚ قَالَ ٱلْكَـٰفِرُونَ إِنَّ هَـٰذَا لَسَـٰحِرٞ مُّبِينٌ ﴿2﴾
کیا لوگوں کے لیے تعجب کی بات ہے کہ ہم نے ان ہی میں سے ایک مرد پر وحی کی کہ لوگوں کو ڈراؤ، اور ایمان والوں کو خوشخبری دو کہ ان کے لیے ان کے رب کے پاس سچی عزت ہے؟ کافروں نے کہا: بے شک یہ تو کھلا جادوگر ہے۔
إِنَّ رَبَّكُمُ ٱللَّهُ ٱلَّذِي خَلَقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ فِي سِتَّةِ أَيَّامٖ ثُمَّ ٱسْتَوَىٰ عَلَى ٱلْعَرْشِۖ يُدَبِّرُ ٱلْأَمْرَۖ مَا مِن شَفِيعٍ إِلَّا مِنۢ بَعْدِ إِذْنِهِۦۚ ذَٰلِكُمُ ٱللَّهُ رَبُّكُمْ فَٱعْبُدُوهُۚ أَفَلَا تَذَكَّرُونَ ﴿3﴾
بے شک تمہارا رب اللہ ہے جس نے آسمانوں اور زمین کو چھ دنوں میں پیدا کیا، پھر عرش پر قائم ہوا، وہی ہر کام کا انتظام کرتا ہے، اس کی اجازت کے بغیر کوئی سفارش کرنے والا نہیں، یہی اللہ تمہارا رب ہے، پس اسی کی عبادت کرو، کیا تم نصیحت حاصل نہیں کرتے؟
إِلَيْهِ مَرْجِعُكُمْ جَمِيعٗاۖ وَعْدَ ٱللَّهِ حَقًّاۚ إِنَّهُۥ يَبْدَؤُا۟ ٱلْخَلْقَ ثُمَّ يُعِيدُهُۥ لِيَجْزِيَ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ بِٱلْقِسْطِۚ وَٱلَّذِينَ كَفَرُوا۟ لَهُمْ شَرَابٞ مِّنْ حَمِيمٖ وَعَذَابٌ أَلِيمُۢ بِمَا كَانُوا۟ يَكْفُرُونَ ﴿4﴾
تم سب کا لوٹنا اسی کی طرف ہے، اللہ کا وعدہ سچا ہے، وہی پہلی بار پیدا کرتا ہے، پھر دوبارہ پیدا کرے گا تاکہ ایمان والوں اور نیک عمل کرنے والوں کو انصاف کے ساتھ بدلہ دے، اور جنہوں نے کفر کیا ان کے لیے کھولتا ہوا پانی ہے اور دردناک عذاب، اس وجہ سے کہ وہ کفر کرتے تھے۔
هُوَ ٱلَّذِي جَعَلَ ٱلشَّمْسَ ضِيَآءٗ وَٱلْقَمَرَ نُورٗا وَقَدَّرَهُۥ مَنَازِلَ لِتَعْلَمُوا۟ عَدَدَ ٱلسِّنِينَ وَٱلْحِسَابَۚ مَا خَلَقَ ٱللَّهُ ذَٰلِكَ إِلَّا بِٱلْحَقِّۚ يُفَصِّلُ ٱلْـَٔايَـٰتِ لِقَوْمٖ يَعْلَمُونَ ﴿5﴾
وہی ہے جس نے سورج کو چمکدار بنایا اور چاند کو نور، اور اس کے لیے منزلیں مقرر کیں تاکہ تم سالوں کی گنتی اور حساب جان سکو، اللہ نے یہ سب کچھ حق کے ساتھ پیدا کیا، وہ نشانیاں کھول کر بیان کرتا ہے ان لوگوں کے لیے جو علم رکھتے ہیں۔
إِنَّ فِي ٱخْتِلَـٰفِ ٱلَّيْلِ وَٱلنَّهَارِ وَمَا خَلَقَ ٱللَّهُ فِي ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ لَـَٔايَـٰتٖ لِّقَوْمٖ يَتَّقُونَ ﴿6﴾
بے شک رات اور دن کے بدلنے میں، اور جو کچھ اللہ نے آسمانوں اور زمین میں پیدا کیا ہے، ان سب میں ان لوگوں کے لیے نشانیاں ہیں جو تقویٰ اختیار کرتے ہیں۔
إِنَّ ٱلَّذِينَ لَا يَرْجُونَ لِقَآءَنَا وَرَضُوا۟ بِٱلْحَيَوٰةِ ٱلدُّنْيَا وَٱطْمَأَنُّوا۟ بِهَا وَٱلَّذِينَ هُمْ عَنْ ءَايَـٰتِنَا غَـٰفِلُونَ ﴿7﴾
بے شک وہ لوگ جو ہماری ملاقات کی امید نہیں رکھتے، اور دنیا کی زندگی پر راضی ہو گئے اور اسی پر مطمئن ہو گئے، اور جو ہماری آیات سے غافل ہیں۔
أُوْلَـٰٓئِكَ مَأْوَىٰهُمُ ٱلنَّارُ بِمَا كَانُوا۟ يَكْسِبُونَ ﴿8﴾
یہی لوگ ہیں جن کا ٹھکانا آگ ہے، اس وجہ سے جو وہ کماتے تھے۔
إِنَّ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ يَهْدِيهِمْ رَبُّهُم بِإِيمَـٰنِهِمۖ تَجْرِي مِن تَحْتِهِمُ ٱلْأَنْهَـٰرُ فِي جَنَّـٰتِ ٱلنَّعِيمِ ﴿9﴾
بے شک جو لوگ ایمان لائے اور نیک عمل کیے، ان کا رب ان کے ایمان کی وجہ سے انہیں ہدایت دے گا، ان کے نیچے نہریں بہتی ہوں گی نعمتوں والی جنتوں میں۔
دَعْوَىٰهُمْ فِيهَا سُبْحَـٰنَكَ ٱللَّهُمَّ وَتَحِيَّتُهُمْ فِيهَا سَلَـٰمٞۚ وَءَاخِرُ دَعْوَىٰهُمْ أَنِ ٱلْحَمْدُ لِلَّهِ رَبِّ ٱلْعَـٰلَمِينَ ﴿10﴾
ان کی دعا وہاں یہ ہو گی: اے اللہ! تو پاک ہے، اور ان کی ملاقات کا کلمہ سلام ہو گا، اور ان کی آخری دعا یہ ہو گی: سب تعریف اللہ کے لیے ہے جو رب ہے تمام جہانوں کا۔
وَلَوْ يُعَجِّلُ ٱللَّهُ لِلنَّاسِ ٱلشَّرَّ ٱسْتِعْجَالَهُم بِٱلْخَيْرِ لَقُضِيَ إِلَيْهِمْ أَجَلُهُمْۖ فَنَذَرُ ٱلَّذِينَ لَا يَرْجُونَ لِقَآءَنَا فِي طُغْيَـٰنِهِمْ يَعْمَهُونَ ﴿11﴾
اور اگر اللہ لوگوں کے لیے برائی کو جلدی لے آئے جیسے وہ بھلائی کے لیے جلدی کرتے ہیں، تو ان کی مدت پوری کر دی جاتی، پس ہم ان لوگوں کو جو ہماری ملاقات کی امید نہیں رکھتے، ان کی سرکشی میں چھوڑ دیتے ہیں کہ وہ بھٹکتے رہیں۔
وَإِذَا مَسَّ ٱلْإِنسَـٰنَ ٱلضُّرُّ دَعَانَا لِجَنبِهِۦٓ أَوْ قَاعِدًا أَوْ قَآئِمٗاۖ فَلَمَّا كَشَفْنَا عَنْهُ ضُرَّهُۥ مَرَّ كَأَن لَّمْ يَدْعُنَآ إِلَىٰ ضُرّٖ مَّسَّهُۥۚ كَذَٰلِكَ زُيِّنَ لِلْمُسْرِفِينَ مَا كَانُوا۟ يَعْمَلُونَ ﴿12﴾
اور جب انسان کو تکلیف پہنچتی ہے تو وہ ہمیں پکارتا ہے لیٹے ہوئے، بیٹھے ہوئے یا کھڑے ہوئے، پھر جب ہم اس سے اس کی تکلیف دور کر دیتے ہیں تو وہ ایسے چلتا ہے گویا اس نے ہمیں کسی تکلیف کے وقت پکارا ہی نہ تھا، اسی طرح حد سے بڑھنے والوں کے لیے ان کے اعمال خوشنما بنا دیے گئے ہیں۔
وَلَقَدْ أَهْلَكْنَا ٱلْقُرُونَ مِن قَبْلِكُمْ لَمَّا ظَلَمُوا۟ وَجَآءَتْهُمْ رُسُلُهُم بِٱلْبَيِّنَـٰتِ وَمَا كَانُوا۟ لِيُؤْمِنُوا۟ۚ كَذَٰلِكَ نَجْزِي ٱلْقَوْمَ ٱلْمُجْرِمِينَ ﴿13﴾
اور بے شک ہم نے تم سے پہلے کی قوموں کو ہلاک کر دیا جب انہوں نے ظلم کیا، اور ان کے پاس ان کے رسول واضح دلائل لے کر آئے، مگر وہ ایمان لانے والے نہ تھے، اسی طرح ہم مجرم قوم کو بدلہ دیتے ہیں۔
ثُمَّ جَعَلْنَـٰكُمْ خَلَـٰٓئِفَ فِي ٱلْأَرْضِ مِنۢ بَعْدِهِمۡ لِنَنظُرَ كَيْفَ تَعْمَلُونَ ﴿14﴾
پھر ہم نے تمہیں زمین میں ان کے بعد جانشین بنایا تاکہ دیکھیں تم کیسے عمل کرتے ہو۔
وَإِذَا تُتْلَىٰ عَلَيْهِمْ ءَايَاتُنَا بَيِّنَـٰتٖ قَالَ ٱلَّذِينَ لَا يَرْجُونَ لِقَآءَنَا ٱئْتِ بِقُرْءَانٍ غَيْرِ هَـٰذَآ أَوْ بَدِّلْهُۚ قُلْ مَا يَكُونُ لِيٓ أَنْ أُبَدِّلَهُۥ مِن تِلْقَآئِ نَفْسِيٓۖ إِنْ أَتَّبِعُ إِلَّا مَا يُوحَىٰٓ إِلَيَّۚ إِنِّيٓ أَخَافُ إِنْ عَصَيْتُ رَبِّي عَذَابَ يَوْمٍ عَظِيمٖ ﴿15﴾
اور جب ان پر ہماری واضح آیات پڑھی جاتی ہیں تو وہ لوگ جو ہماری ملاقات کی امید نہیں رکھتے کہتے ہیں: اس قرآن کے علاوہ کوئی اور لے آؤ یا اسے بدل دو، کہہ دو: میرے لیے یہ ممکن نہیں کہ میں اسے اپنی طرف سے بدل دوں، میں تو صرف اس کی پیروی کرتا ہوں جو میری طرف وحی کی جاتی ہے، اگر میں اپنے رب کی نافرمانی کروں تو مجھے بڑے دن کے عذاب کا خوف ہے۔
قُل لَّوْ شَآءَ ٱللَّهُ مَا تَلَوْتُهُۥ عَلَيْكُمْ وَلَآ أَدْرَىٰكُم بِهِۦۖ فَقَدْ لَبِثْتُ فِيكُمْ عُمُرٗا مِّن قَبْلِهِۦٓۚ أَفَلَا تَعْقِلُونَ ﴿16﴾
کہہ دو: اگر اللہ چاہتا تو میں تم پر یہ نہ پڑھتا اور نہ ہی وہ تمہیں اس سے آگاہ کرتا، میں تو اس سے پہلے تمہارے درمیان ایک عمر گزار چکا ہوں، کیا تم عقل نہیں رکھتے؟
فَمَنْ أَظْلَمُ مِمَّنِ ٱفْتَرَىٰ عَلَى ٱللَّهِ كَذِبًا أَوْ كَذَّبَ بِـَٔايَـٰتِهِۦٓۚ إِنَّهُۥ لَا يُفْلِحُ ٱلْمُجْرِمُونَ ﴿17﴾
تو اس سے بڑھ کر ظالم کون ہے جو اللہ پر جھوٹ باندھے یا اس کی آیات کو جھٹلائے؟ بے شک مجرم لوگ فلاح نہیں پاتے۔
وَيَعْبُدُونَ مِن دُونِ ٱللَّهِ مَا لَا يَضُرُّهُمْ وَلَا يَنفَعُهُمْ وَيَقُولُونَ هَـٰٓؤُلَآءِ شُفَعَـٰٓؤُنَا عِندَ ٱللَّهِۚ قُلْ أَتُنَبِّـُٔونَ ٱللَّهَ بِمَا لَا يَعْلَمُ فِي ٱلسَّمَـٰوَٰتِ وَلَا فِي ٱلْأَرْضِۚ سُبْحَـٰنَهُۥ وَتَعَـٰلَىٰ عَمَّا يُشْرِكُونَ ﴿18﴾
اور وہ اللہ کے سوا ان کی عبادت کرتے ہیں جو نہ انہیں نقصان پہنچا سکتے ہیں اور نہ فائدہ، اور کہتے ہیں: یہ اللہ کے ہاں ہمارے سفارشی ہیں، کہہ دو: کیا تم اللہ کو ایسی چیز کی خبر دیتے ہو جسے وہ آسمانوں اور زمین میں نہیں جانتا؟ وہ پاک ہے اور بلند ہے اس سے جو وہ شریک ٹھہراتے ہیں۔
وَمَا كَانَ ٱلنَّاسُ إِلَّآ أُمَّةٗ وَٰحِدَةٗ فَٱخْتَلَفُوا۟ۚ وَلَوْلَا كَلِمَةٞ سَبَقَتْ مِن رَّبِّكَ لَقُضِيَ بَيْنَهُمْ فِيمَا فِيهِ يَخْتَلِفُونَ ﴿19﴾
اور لوگ ایک ہی امت تھے، پھر انہوں نے اختلاف کیا، اور اگر تمہارے رب کی طرف سے ایک بات پہلے نہ ہو چکی ہوتی تو ان کے درمیان ان کے اختلافات میں فیصلہ کر دیا جاتا۔
وَيَقُولُونَ لَوْلَآ أُنزِلَ عَلَيْهِ ءَايَةٞ مِّن رَّبِّهِۦۖ فَقُلْ إِنَّمَا ٱلْغَيْبُ لِلَّهِ فَٱنتَظِرُوٓا۟ إِنِّي مَعَكُم مِّنَ ٱلْمُنتَظِرِينَ ﴿20﴾
اور وہ کہتے ہیں: اس پر اس کے رب کی طرف سے کوئی نشانی کیوں نہیں اتاری گئی؟ کہہ دو: غیب تو صرف اللہ ہی کے لیے ہے، پس انتظار کرو، میں بھی تمہارے ساتھ انتظار کرنے والوں میں سے ہوں۔
وَإِذَآ أَذَقْنَا ٱلنَّاسَ رَحْمَةٗ مِّنۢ بَعْدِ ضَرَّآءَ مَسَّتْهُمْ إِذَا لَهُمْ مَّكْرٞ فِيٓ ءَايَاتِنَاۚ قُلِ ٱللَّهُ أَسْرَعُ مَكْرٗاۚ إِنَّ رُسُلَنَا يَكْتُبُونَ مَا تَمْكُرُونَ ﴿21﴾
اور جب ہم لوگوں کو تکلیف کے بعد رحمت چکھاتے ہیں تو فوراً وہ ہماری آیات کے بارے میں چالیں چلنے لگتے ہیں، کہہ دو: اللہ سب سے زیادہ چالاک ہے، بے شک ہمارے فرشتے تمہاری چالاکیاں لکھ رہے ہیں۔
هُوَ ٱلَّذِي يُسَيِّرُكُمْ فِي ٱلْبَرِّ وَٱلْبَحْرِۖ حَتَّىٰٓ إِذَا كُنتُمْ فِي ٱلْفُلْكِ وَجَرَيْنَ بِهِم بِرِيحٖ طَيِّبَةٖ وَفَرِحُوا۟ بِهَا جَآءَتْهَا رِيحٌ عَاصِفٞ وَجَآءَهُمُ ٱلْمَوْجُ مِن كُلِّ مَكَانٖ وَظَنُّوٓا۟ أَنَّهُمْ أُحِيطَ بِهِمْ دَعَوُا۟ ٱللَّهَ مُخْلِصِينَ لَهُ ٱلدِّينَ لَئِنْ أَنجَيْتَنَا مِنْ هَـٰذِهِۦ لَنَكُونَنَّ مِنَ ٱلشَّـٰكِرِينَ ﴿22﴾
وہی ہے جو تمہیں خشکی اور سمندر میں چلاتا ہے، یہاں تک کہ جب تم کشتیوں میں ہوتے ہو اور وہ انہیں خوشگوار ہوا کے ساتھ لے جاتی ہے اور وہ خوش ہوتے ہیں، تو اچانک ان پر تیز آندھی آتی ہے اور ہر طرف سے موجیں آ جاتی ہیں، اور وہ سمجھتے ہیں کہ وہ گھیر لیے گئے ہیں، تو وہ اللہ کو پکارتے ہیں دین کو اسی کے لیے خالص کرتے ہوئے: اگر تو ہمیں اس سے نجات دے دے تو ہم ضرور شکر گزاروں میں سے ہوں گے۔
فَلَمَّآ أَنجَىٰهُمْ إِذَا هُمْ يَبْغُونَ فِي ٱلْأَرْضِ بِغَيْرِ ٱلْحَقِّۚ يَـٰٓأَيُّهَا ٱلنَّاسُ إِنَّمَا بَغْيُكُمْ عَلَىٰٓ أَنفُسِكُمۡۖ مَّتَـٰعَ ٱلْحَيَوٰةِ ٱلدُّنْيَاۖ ثُمَّ إِلَيْنَا مَرْجِعُكُمْ فَنُنَبِّئُكُم بِمَا كُنتُمْ تَعْمَلُونَ ﴿23﴾
پھر جب وہ انہیں نجات دیتا ہے تو وہ زمین میں ناحق سرکشی کرنے لگتے ہیں، اے لوگو! تمہاری سرکشی تمہارے ہی خلاف ہے، یہ دنیا کی زندگی کا تھوڑا سا فائدہ ہے، پھر ہماری طرف تمہیں لوٹنا ہے، پھر ہم تمہیں بتائیں گے جو تم کرتے رہے۔
إِنَّمَا مَثَلُ ٱلْحَيَوٰةِ ٱلدُّنْيَا كَمَآءٍ أَنزَلْنَـٰهُ مِنَ ٱلسَّمَآءِ فَٱخْتَلَطَ بِهِۦ نَبَاتُ ٱلْأَرْضِ مِمَّا يَأْكُلُ ٱلنَّاسُ وَٱلْأَنْعَـٰمُۖ حَتَّىٰٓ إِذَآ أَخَذَتِ ٱلْأَرْضُ زُخْرُفَهَا وَٱزَّيَّنَتْ وَظَنَّ أَهْلُهَآ أَنَّهُمْ قَادِرُونَ عَلَيْهَآ أَتَىٰهَآ أَمْرُنَا لَيْلًا أَوْ نَهَارٗا فَجَعَلْنَـٰهَا حَصِيدٗا كَأَن لَّمْ تَغْنَ بِٱلْأَمْسِۚ كَذَٰلِكَ نُفَصِّلُ ٱلْـَٔايَـٰتِ لِقَوْمٖ يَتَفَكَّرُونَ ﴿24﴾
دنیا کی زندگی کی مثال اس پانی کی طرح ہے جسے ہم نے آسمان سے اتارا، پھر اس سے زمین کی نباتات اگتی ہیں جنہیں انسان اور جانور کھاتے ہیں، یہاں تک کہ جب زمین اپنی رونق اختیار کر لیتی ہے اور خوب آراستہ ہو جاتی ہے اور اس کے لوگ سمجھتے ہیں کہ وہ اس پر قابو پا چکے ہیں، تو ہمارا حکم رات یا دن میں آ جاتا ہے، پھر ہم اسے ایسا صاف کر دیتے ہیں گویا کل وہاں کچھ تھا ہی نہیں، اسی طرح ہم نشانیاں کھول کر بیان کرتے ہیں ان لوگوں کے لیے جو غور و فکر کرتے ہیں۔
وَٱللَّهُ يَدْعُوٓا۟ إِلَىٰ دَارِ ٱلسَّلَـٰمِ وَيَهْدِي مَن يَشَآءُ إِلَىٰ صِرَٰطٖ مُّسْتَقِيمٖ ﴿25﴾
اور اللہ دارالسلام کی طرف بلاتا ہے، اور جسے چاہتا ہے سیدھے راستے کی طرف ہدایت دیتا ہے۔
لِلَّذِينَ أَحْسَنُوا۟ ٱلْحُسْنَىٰ وَزِيَادَةٞۖ وَلَا يَرْهَقُ وُجُوهَهُمْ قَتَرٞ وَلَا ذِلَّةٌۚ أُوْلَـٰٓئِكَ أَصْحَـٰبُ ٱلْجَنَّةِ هُمْ فِيهَا خَـٰلِدُونَ ﴿26﴾
جنہوں نے نیکی کی ان کے لیے بہترین بدلہ ہے اور مزید بھی، اور ان کے چہروں پر نہ سیاہی چھائے گی اور نہ ذلت، یہی جنت والے ہیں، وہ اس میں ہمیشہ رہیں گے۔
وَٱلَّذِينَ كَسَبُوا۟ ٱلسَّيِّـَٔاتِ جَزَآءُ سَيِّئَةِۭ بِمِثْلِهَا وَتَرْهَقُهُمْ ذِلَّةٞۖ مَّا لَهُم مِّنَ ٱللَّهِ مِنْ عَاصِمٖۖ كَأَنَّمَآ أُغْشِيَتْ وُجُوهُهُمْ قِطَعٗا مِّنَ ٱلَّيْلِ مُظْلِمًاۚ أُوْلَـٰٓئِكَ أَصْحَـٰبُ ٱلنَّارِ هُمْ فِيهَا خَـٰلِدُونَ ﴿27﴾
اور جنہوں نے برائیاں کمائیں، ان کی برائی کا بدلہ اسی جیسا ہو گا، اور ان پر ذلت چھا جائے گی، ان کے لیے اللہ کی طرف سے کوئی بچانے والا نہ ہو گا، گویا ان کے چہروں پر اندھیری رات کے ٹکڑے ڈال دیے گئے ہوں، یہی آگ والے ہیں، وہ اس میں ہمیشہ رہیں گے۔
وَيَوْمَ نَحْشُرُهُمْ جَمِيعٗا ثُمَّ نَقُولُ لِلَّذِينَ أَشْرَكُوا۟ مَكَانَكُمْ أَنتُمْ وَشُرَكَآؤُكُمْۚ فَزَيَّلْنَا بَيْنَهُمْۖ وَقَالَ شُرَكَآؤُهُم مَّا كُنتُمْ إِيَّانَا تَعْبُدُونَ ﴿28﴾
اور جس دن ہم سب کو جمع کریں گے، پھر ہم مشرکوں سے کہیں گے: تم اور تمہارے شریک اپنی جگہ ٹھہر جاؤ، پھر ہم ان کے درمیان جدائی ڈال دیں گے، اور ان کے شریک کہیں گے: تم ہماری عبادت نہیں کرتے تھے۔
فَكَفَىٰ بِٱللَّهِ شَهِيدَۭا بَيْنَنَا وَبَيْنَكُمۖ إِن كُنَّا عَنْ عِبَادَتِكُمْ لَغَـٰفِلِينَ ﴿29﴾
پس ہمارے اور تمہارے درمیان اللہ کافی گواہ ہے، بے شک ہم تمہاری عبادت سے بالکل غافل تھے۔
هُنَالِكَ تَبْلُوا۟ كُلُّ نَفْسٖ مَّآ أَسْلَفَتْۖ وَرُدُّوٓا۟ إِلَى ٱللَّهِ مَوْلَىٰهُمُ ٱلْحَقِّۖ وَضَلَّ عَنْهُم مَّا كَانُوا۟ يَفْتَرُونَ ﴿30﴾
اس وقت ہر شخص اپنے کیے ہوئے عمل کو جانچ لے گا، اور وہ اللہ کی طرف لوٹائے جائیں گے جو ان کا سچا مالک ہے، اور وہ سب کچھ ان سے گم ہو جائے گا جو وہ جھوٹ گھڑتے تھے۔
قُلْ مَن يَرْزُقُكُم مِّنَ ٱلسَّمَآءِ وَٱلْأَرْضِ أَمَّن يَمْلِكُ ٱلسَّمْعَ وَٱلْأَبْصَـٰرَ وَمَن يُخْرِجُ ٱلْحَيَّ مِنَ ٱلْمَيِّتِ وَيُخْرِجُ ٱلْمَيِّتَ مِنَ ٱلْحَيِّ وَمَن يُدَبِّرُ ٱلْأَمْرَۚ فَسَيَقُولُونَ ٱللَّهُۚ فَقُلْ أَفَلَا تَتَّقُونَ ﴿31﴾
کہہ دو: کون تمہیں آسمان اور زمین سے رزق دیتا ہے؟ اور کون تمہارے کانوں اور آنکھوں کا مالک ہے؟ اور کون زندہ کو مردہ سے نکالتا ہے اور مردہ کو زندہ سے نکالتا ہے؟ اور کون تمام امور کی تدبیر کرتا ہے؟ وہ کہیں گے: اللہ، تو کہہ دو: کیا تم تقویٰ اختیار نہیں کرتے؟
فَذَٰلِكُمُ ٱللَّهُ رَبُّكُمُ ٱلْحَقُّۖ فَمَاذَا بَعْدَ ٱلْحَقِّ إِلَّا ٱلضَّلَـٰلُۖ فَأَنَّىٰ تُصْرَفُونَ ﴿32﴾
پس یہی اللہ تمہارا سچا رب ہے، تو حق کے بعد گمراہی کے سوا کیا باقی رہ جاتا ہے؟ تو تم کہاں پھیرے جا رہے ہو؟
كَذَٰلِكَ حَقَّتْ كَلِمَتُ رَبِّكَ عَلَى ٱلَّذِينَ فَسَقُوٓا۟ أَنَّهُمْ لَا يُؤْمِنُونَ ﴿33﴾
اسی طرح تمہارے رب کا فرمان ان نافرمانوں پر ثابت ہو چکا ہے کہ وہ ایمان نہیں لائیں گے۔
قُلْ هَلْ مِن شُرَكَآئِكُم مَّن يَبْدَؤُا۟ ٱلْخَلْقَ ثُمَّ يُعِيدُهُۥۚ قُلِ ٱللَّهُ يَبْدَؤُا۟ ٱلْخَلْقَ ثُمَّ يُعِيدُهُۥۖ فَأَنَّىٰ تُؤْفَكُونَ ﴿34﴾
کہہ دو: کیا تمہارے شریکوں میں کوئی ایسا ہے جو پہلی بار پیدا کرے پھر اسے دوبارہ لوٹائے؟ کہہ دو: اللہ ہی پہلی بار پیدا کرتا ہے پھر اسے لوٹاتا ہے، تو تم کہاں بہکائے جا رہے ہو؟
قُلْ هَلْ مِن شُرَكَآئِكُم مَّن يَهْدِيٓ إِلَى ٱلْحَقِّۚ قُلِ ٱللَّهُ يَهْدِي لِلْحَقِّۚ أَفَمَن يَهْدِيٓ إِلَى ٱلْحَقِّ أَحَقُّ أَن يُتَّبَعَ أَمَّن لَّا يَهِدِّيٓ إِلَّآ أَن يُهْدَىٰۖ فَمَا لَكُمْ كَيْفَ تَحْكُمُونَ ﴿35﴾
کہہ دو: کیا تمہارے شریکوں میں کوئی ایسا ہے جو حق کی طرف ہدایت دے؟ کہہ دو: اللہ ہی حق کی طرف ہدایت دیتا ہے، تو جو حق کی طرف ہدایت دے وہ پیروی کے لائق ہے یا وہ جو خود ہدایت نہیں دیتا جب تک اسے ہدایت نہ دی جائے؟ تو تمہیں کیا ہو گیا ہے، کیسے فیصلہ کرتے ہو؟
وَمَا يَتَّبِعُ أَكْثَرُهُمْ إِلَّا ظَنًّاۚ إِنَّ ٱلظَّنَّ لَا يُغْنِي مِنَ ٱلْحَقِّ شَيْـًٔاۚ إِنَّ ٱللَّهَ عَلِيمُۢ بِمَا يَفْعَلُونَ ﴿36﴾
اور ان میں سے اکثر صرف گمان کی پیروی کرتے ہیں، بے شک گمان حق کے مقابلے میں کچھ فائدہ نہیں دیتا، بے شک اللہ ان کے اعمال کو خوب جانتا ہے۔
وَمَا كَانَ هَـٰذَا ٱلْقُرْءَانُ أَن يُفْتَرَىٰ مِن دُونِ ٱللَّهِ وَلَـٰكِن تَصْدِيقَ ٱلَّذِي بَيْنَ يَدَيْهِ وَتَفْصِيلَ ٱلْكِتَـٰبِ لَا رَيْبَ فِيهِ مِن رَّبِّ ٱلْعَـٰلَمِينَ ﴿37﴾
اور یہ قرآن ایسا نہیں کہ اللہ کے سوا کوئی اسے گھڑ لے، بلکہ یہ اس کی تصدیق ہے جو اس سے پہلے ہے، اور کتاب کی تفصیل ہے، اس میں کوئی شک نہیں، رب العالمین کی طرف سے ہے۔
أَمْ يَقُولُونَ ٱفْتَرَىٰهُۖ قُلْ فَأْتُوا۟ بِسُورَةٖ مِّثْلِهِۦ وَٱدْعُوا۟ مَنِ ٱسْتَطَعْتُم مِّن دُونِ ٱللَّهِ إِن كُنتُمْ صَـٰدِقِينَ ﴿38﴾
کیا وہ کہتے ہیں: اس نے اسے گھڑ لیا ہے؟ کہہ دو: تم اس جیسی ایک سورت لے آؤ، اور اللہ کے سوا جسے چاہو بلا لو اگر تم سچے ہو۔
بَلْ كَذَّبُوا۟ بِمَا لَمْ يُحِيطُوا۟ بِعِلْمِهِۦ وَلَمَّا يَأْتِهِمْ تَأْوِيلُهُۥۚ كَذَٰلِكَ كَذَّبَ ٱلَّذِينَ مِن قَبْلِهِمۡۖ فَٱنظُرْ كَيْفَ كَانَ عَـٰقِبَةُ ٱلظَّـٰلِمِينَ ﴿39﴾
بلکہ انہوں نے اس چیز کو جھٹلا دیا جس کا انہیں علم نہ تھا، اور ابھی اس کی حقیقت ان کے سامنے نہیں آئی تھی، اسی طرح ان سے پہلے لوگوں نے بھی جھٹلایا، پس دیکھو ظالموں کا انجام کیسا ہوا۔
وَمِنْهُم مَّن يُؤْمِنُ بِهِۦ وَمِنْهُم مَّن لَّا يُؤْمِنُ بِهِۦۚ وَرَبُّكَ أَعْلَمُ بِٱلْمُفْسِدِينَ ﴿40﴾
اور ان میں سے کچھ ایسے ہیں جو اس پر ایمان لاتے ہیں، اور کچھ ایسے ہیں جو ایمان نہیں لاتے، اور تمہارا رب فساد کرنے والوں کو خوب جانتا ہے۔
وَإِن كَذَّبُوكَ فَقُل لِّي عَمَلِي وَلَكُمْ عَمَلُكُمْۖ أَنتُم بَرِيٓـُٔونَ مِمَّآ أَعْمَلُ وَأَنَا۠ بَرِيٓءٞ مِّمَّا تَعْمَلُونَ ﴿41﴾
اور اگر وہ تمہیں جھٹلائیں تو کہہ دو: میرے لیے میرا عمل ہے اور تمہارے لیے تمہارا عمل، تم میرے عمل سے بری ہو اور میں تمہارے عمل سے بری ہوں۔
وَمِنْهُم مَّن يَسْتَمِعُ إِلَيْكَۚ أَفَأَنتَ تُسْمِعُ ٱلصُّمَّ وَلَوْ كَانُوا۟ لَا يَعْقِلُونَ ﴿42﴾
اور ان میں سے کچھ ایسے ہیں جو تمہاری بات سنتے ہیں، تو کیا تم بہروں کو سنا سکتے ہو، اگرچہ وہ عقل نہ رکھتے ہوں؟
وَمِنْهُم مَّن يَنظُرُ إِلَيْكَۚ أَفَأَنتَ تَهْدِي ٱلْعُمْيَ وَلَوْ كَانُوا۟ لَا يُبْصِرُونَ ﴿43﴾
اور ان میں سے کچھ ایسے ہیں جو تمہیں دیکھتے ہیں، تو کیا تم اندھوں کو راستہ دکھا سکتے ہو، اگرچہ وہ نہ دیکھتے ہوں؟
إِنَّ ٱللَّهَ لَا يَظْلِمُ ٱلنَّاسَ شَيْـٔٗا وَلَـٰكِنَّ ٱلنَّاسَ أَنفُسَهُمْ يَظْلِمُونَ ﴿44﴾
بے شک اللہ لوگوں پر کچھ بھی ظلم نہیں کرتا، بلکہ لوگ خود ہی اپنے اوپر ظلم کرتے ہیں۔
وَيَوْمَ يَحْشُرُهُمْ كَأَن لَّمْ يَلْبَثُوٓا۟ إِلَّا سَاعَةٗ مِّنَ ٱلنَّهَارِ يَتَعَارَفُونَ بَيْنَهُمْۚ قَدْ خَسِرَ ٱلَّذِينَ كَذَّبُوا۟ بِلِقَآءِ ٱللَّهِ وَمَا كَانُوا۟ مُهْتَدِينَ ﴿45﴾
اور جس دن وہ انہیں جمع کرے گا، گویا وہ دن کی ایک گھڑی سے زیادہ نہ ٹھہرے ہوں، وہ آپس میں ایک دوسرے کو پہچانیں گے، بے شک وہ لوگ خسارے میں رہے جنہوں نے اللہ کی ملاقات کو جھٹلایا، اور وہ ہدایت یافتہ نہ تھے۔
وَإِمَّا نُرِيَنَّكَ بَعْضَ ٱلَّذِي نَعِدُهُمْ أَوْ نَتَوَفَّيَنَّكَ فَإِلَيْنَا مَرْجِعُهُمْ ثُمَّ ٱللَّهُ شَهِيدٌ عَلَىٰ مَا يَفْعَلُونَ ﴿46﴾
اور ہم تمہیں ان میں سے کچھ وہ چیز دکھا دیں جس کا ہم ان سے وعدہ کر رہے ہیں، یا تمہیں وفات دے دیں، تو ان کا لوٹنا ہماری طرف ہی ہے، پھر اللہ ان کے اعمال پر گواہ ہے۔
وَلِكُلِّ أُمَّةٖ رَّسُولٞۖ فَإِذَا جَآءَ رَسُولُهُمْ قُضِيَ بَيْنَهُم بِٱلْقِسْطِ وَهُمْ لَا يُظْلَمُونَ ﴿47﴾
اور ہر امت کے لیے ایک رسول ہے، پھر جب ان کا رسول آ جاتا ہے تو ان کے درمیان انصاف کے ساتھ فیصلہ کر دیا جاتا ہے، اور ان پر ظلم نہیں کیا جاتا۔
وَيَقُولُونَ مَتَىٰ هَـٰذَا ٱلْوَعْدُ إِن كُنتُمْ صَـٰدِقِينَ ﴿48﴾
اور وہ کہتے ہیں: یہ وعدہ کب پورا ہو گا اگر تم سچے ہو؟
قُل لَّآ أَمْلِكُ لِنَفْسِي ضَرّٗا وَلَا نَفْعًا إِلَّا مَا شَآءَ ٱللَّهُۗ لِكُلِّ أُمَّةٍ أَجَلٌۖ إِذَا جَآءَ أَجَلُهُمْ فَلَا يَسْتَـْٔخِرُونَ سَاعَةٗ وَلَا يَسْتَقْدِمُونَ ﴿49﴾
کہہ دو: میں اپنے لیے نہ نقصان کا مالک ہوں اور نہ نفع کا، مگر جو اللہ چاہے، ہر امت کے لیے ایک وقت مقرر ہے، جب ان کا وقت آ جاتا ہے تو نہ وہ ایک گھڑی پیچھے ہوتے ہیں اور نہ آگے بڑھتے ہیں۔
قُلْ أَرَءَيْتُمْ إِنْ أَتَىٰكُمْ عَذَابُهُۥ بَيَـٰتًا أَوْ نَهَارٗا مَّاذَا يَسْتَعْجِلُ مِنْهُ ٱلْمُجْرِمُونَ ﴿50﴾
کہہ دو: کیا تم نے دیکھا اگر تم پر اس کا عذاب رات کو یا دن کو آ جائے، تو مجرم اس سے کس چیز کی جلدی مچا رہے ہیں؟
أَثُمَّ إِذَا مَا وَقَعَ ءَامَنتُم بِهِۦٓۚ ءَٟلْـَٔـٰنَ وَقَدْ كُنتُم بِهِۦ تَسْتَعْجِلُونَ ﴿51﴾
کیا پھر جب وہ واقع ہو جائے گا تو تم اس پر ایمان لاؤ گے؟ اب؟ حالانکہ تم اس کی جلدی مچا رہے تھے۔
ثُمَّ قِيلَ لِلَّذِينَ ظَلَمُوا۟ ذُوقُوا۟ عَذَابَ ٱلْخُلْدِ هَلْ تُجْزَوْنَ إِلَّا بِمَا كُنتُمْ تَكْسِبُونَ ﴿52﴾
پھر ظالموں سے کہا جائے گا: ہمیشہ کا عذاب چکھو، تمہیں صرف اسی کا بدلہ دیا جا رہا ہے جو تم کماتے تھے۔
وَيَسْتَنبِـُٔونَكَ أَحَقٌّ هُوَۖ قُلْ إِي وَرَبِّيٓ إِنَّهُۥ لَحَقّٞۖ وَمَآ أَنتُم بِمُعْجِزِينَ ﴿53﴾
اور وہ تم سے پوچھتے ہیں: کیا واقعی یہ سچ ہے؟ کہہ دو: ہاں! میرے رب کی قسم! بے شک یہ سچ ہے، اور تم عاجز کرنے والے نہیں ہو۔
وَلَوْ أَنَّ لِكُلِّ نَفْسٖ ظَلَمَتْ مَا فِي ٱلْأَرْضِ لَٱفْتَدَتْ بِهِۦۗ وَأَسَرُّوا۟ ٱلنَّدَامَةَ لَمَّا رَأَوُا۟ ٱلْعَذَابَۖ وَقُضِيَ بَيْنَهُم بِٱلْقِسْطِ وَهُمْ لَا يُظْلَمُونَ ﴿54﴾
اور اگر ہر ظلم کرنے والے شخص کے پاس زمین کی ساری چیزیں ہوں تو وہ ضرور اسے فدیہ میں دے دے، اور جب وہ عذاب دیکھیں گے تو ندامت چھپائیں گے، اور ان کے درمیان انصاف کے ساتھ فیصلہ کر دیا جائے گا، اور ان پر ظلم نہیں کیا جائے گا۔
أَلَآ إِنَّ لِلَّهِ مَا فِي ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۗ أَلَآ إِنَّ وَعْدَ ٱللَّهِ حَقّٞ وَلَـٰكِنَّ أَكْثَرَهُمْ لَا يَعْلَمُونَ ﴿55﴾
خبردار! بے شک اللہ ہی کے لیے ہے جو کچھ آسمانوں اور زمین میں ہے، خبردار! بے شک اللہ کا وعدہ سچا ہے، لیکن ان میں سے اکثر نہیں جانتے۔
هُوَ يُحْيِۦ وَيُمِيتُ وَإِلَيْهِ تُرْجَعُونَ ﴿56﴾
وہی زندہ کرتا ہے اور مارتا ہے، اور اسی کی طرف تم لوٹائے جاؤ گے۔
يَـٰٓأَيُّهَا ٱلنَّاسُ قَدْ جَآءَتْكُم مَّوْعِظَةٞ مِّن رَّبِّكُمْ وَشِفَآءٞ لِّمَا فِي ٱلصُّدُورِ وَهُدٗى وَرَحْمَةٞ لِّلْمُؤْمِنِينَ ﴿57﴾
اے لوگو! تمہارے رب کی طرف سے تمہارے پاس نصیحت آ چکی ہے، اور دلوں کی بیماریوں کے لیے شفا، اور ایمان والوں کے لیے ہدایت اور رحمت۔
قُلْ بِفَضْلِ ٱللَّهِ وَبِرَحْمَتِهِۦ فَبِذَٰلِكَ فَلْيَفْرَحُوا۟ۚ هُوَ خَيْرٞ مِّمَّا يَجْمَعُونَ ﴿58﴾
کہہ دو: اللہ کے فضل اور اس کی رحمت پر ہی انہیں خوش ہونا چاہیے، وہ اس سے بہتر ہے جو وہ جمع کرتے ہیں۔
قُلْ أَرَءَيْتُم مَّآ أَنزَلَ ٱللَّهُ لَكُم مِّن رِّزْقٖ فَجَعَلْتُم مِّنْهُ حَرَامٗا وَحَلَـٰلٗاۚ قُلْ ءَآللَّهُ أَذِنَ لَكُمۡ أَمْ عَلَى ٱللَّهِ تَفْتَرُونَ ﴿59﴾
کہہ دو: کیا تم نے دیکھا جو رزق اللہ نے تمہیں دیا، پھر تم نے اس میں سے کچھ کو حرام اور کچھ کو حلال بنا لیا؟ کہہ دو: کیا اللہ نے تمہیں اس کی اجازت دی ہے یا تم اللہ پر جھوٹ باندھتے ہو؟
وَمَا ظَنُّ ٱلَّذِينَ يَفْتَرُونَ عَلَى ٱللَّهِ ٱلْكَذِبَ يَوْمَ ٱلْقِيَـٰمَةِۚ إِنَّ ٱللَّهَ لَذُو فَضْلٍ عَلَى ٱلنَّاسِ وَلَـٰكِنَّ أَكْثَرَهُمْ لَا يَشْكُرُونَ ﴿60﴾
اور جو لوگ اللہ پر جھوٹ باندھتے ہیں، قیامت کے دن ان کا کیا گمان ہے؟ بے شک اللہ لوگوں پر فضل کرنے والا ہے، لیکن ان میں سے اکثر شکر نہیں کرتے۔
وَمَا تَكُونُ فِي شَأْنٖ وَمَا تَتْلُوا۟ مِنْهُ مِن قُرْءَانٖ وَلَا تَعْمَلُونَ مِنْ عَمَلٍ إِلَّا كُنَّا عَلَيْكُمْ شُهُودًا إِذْ تُفِيضُونَ فِيهِۚ وَمَا يَعْزُبُ عَن رَّبِّكَ مِن مِّثْقَالِ ذَرَّةٖ فِي ٱلْأَرْضِ وَلَا فِي ٱلسَّمَآءِ وَلَآ أَصْغَرَ مِن ذَٰلِكَ وَلَآ أَكْبَرَ إِلَّا فِي كِتَـٰبٖ مُّبِينٖ ﴿61﴾
اور تم جس حال میں بھی ہوتے ہو، اور قرآن میں سے جو کچھ بھی پڑھتے ہو، اور جو عمل بھی کرتے ہو، ہم تم پر گواہ ہوتے ہیں جب تم اس میں مشغول ہوتے ہو، اور تمہارے رب سے ذرہ برابر بھی کوئی چیز چھپی نہیں ہے، نہ زمین میں اور نہ آسمان میں، اور نہ اس سے چھوٹی اور نہ بڑی، مگر وہ سب واضح کتاب میں ہے۔
أَلَآ إِنَّ أَوْلِيَآءَ ٱللَّهِ لَا خَوْفٌ عَلَيْهِمْ وَلَا هُمْ يَحْزَنُونَ ﴿62﴾
خبردار! بے شک اللہ کے دوستوں پر نہ کوئی خوف ہے اور نہ وہ غمگین ہوں گے۔
ٱلَّذِينَ ءَامَنُوا۟ وَكَانُوا۟ يَتَّقُونَ ﴿63﴾
وہ لوگ جو ایمان لائے اور تقویٰ اختیار کرتے رہے۔
لَهُمُ ٱلْبُشْرَىٰ فِي ٱلْحَيَوٰةِ ٱلدُّنْيَا وَفِي ٱلْـَٔاخِرَةِۚ لَا تَبْدِيلَ لِكَلِمَـٰتِ ٱللَّهِۚ ذَٰلِكَ هُوَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿64﴾
ان کے لیے دنیا کی زندگی میں اور آخرت میں خوشخبری ہے، اللہ کی باتوں میں کوئی تبدیلی نہیں، یہی بڑی کامیابی ہے۔
وَلَا يَحْزُنكَ قَوْلُهُمْۘ إِنَّ ٱلْعِزَّةَ لِلَّهِ جَمِيعًاۚ هُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ ﴿65﴾
اور ان کی بات تمہیں غمگین نہ کرے، بے شک عزت تو سب کی سب اللہ ہی کے لیے ہے، وہی سننے والا، جاننے والا ہے۔
أَلَآ إِنَّ لِلَّهِ مَن فِي ٱلسَّمَـٰوَٰتِ وَمَن فِي ٱلْأَرْضِۗ وَمَا يَتَّبِعُ ٱلَّذِينَ يَدْعُونَ مِن دُونِ ٱللَّهِ شُرَكَآءَۚ إِن يَتَّبِعُونَ إِلَّا ٱلظَّنَّ وَإِنْ هُمْ إِلَّا يَخْرُصُونَ ﴿66﴾
خبردار! بے شک اللہ ہی کے لیے ہے جو آسمانوں میں ہے اور جو زمین میں ہے، اور جو لوگ اللہ کے سوا دوسروں کو شریک پکارتے ہیں وہ کسی چیز کی پیروی نہیں کرتے، وہ صرف گمان کی پیروی کرتے ہیں، اور وہ صرف جھوٹ گھڑتے ہیں۔
هُوَ ٱلَّذِي جَعَلَ لَكُمُ ٱلَّيْلَ لِتَسْكُنُوا۟ فِيهِ وَٱلنَّهَارَ مُبْصِرًاۚ إِنَّ فِي ذَٰلِكَ لَـَٔايَـٰتٖ لِّقَوْمٖ يَسْمَعُونَ ﴿67﴾
وہی ہے جس نے تمہارے لیے رات بنائی تاکہ تم اس میں آرام کرو، اور دن کو روشن بنایا، بے شک اس میں ان لوگوں کے لیے نشانیاں ہیں جو سنتے ہیں۔
قَالُوا۟ ٱتَّخَذَ ٱللَّهُ وَلَدٗاۗ سُبْحَـٰنَهُۥۖ هُوَ ٱلْغَنِيُّۖ لَهُۥ مَا فِي ٱلسَّمَـٰوَٰتِ وَمَا فِي ٱلْأَرْضِۚ إِنْ عِندَكُم مِّن سُلْطَـٰنٖۭ بِهَـٰذَآۚ أَتَقُولُونَ عَلَى ٱللَّهِ مَا لَا تَعْلَمُونَ ﴿68﴾
انہوں نے کہا: اللہ نے بیٹا بنایا ہے؟ وہ پاک ہے! وہی بے نیاز ہے، اسی کے لیے ہے جو کچھ آسمانوں اور زمین میں ہے، تمہارے پاس اس بات کی کوئی دلیل نہیں، کیا تم اللہ پر وہ بات کہتے ہو جو تم نہیں جانتے؟
قُلْ إِنَّ ٱلَّذِينَ يَفْتَرُونَ عَلَى ٱللَّهِ ٱلْكَذِبَ لَا يُفْلِحُونَ ﴿69﴾
کہہ دو: بے شک جو لوگ اللہ پر جھوٹ باندھتے ہیں وہ کامیاب نہیں ہوتے۔
مَتَـٰعٞ فِي ٱلدُّنْيَا ثُمَّ إِلَيْنَا مَرْجِعُهُمْ ثُمَّ نُذِيقُهُمُ ٱلْعَذَابَ ٱلشَّدِيدَ بِمَا كَانُوا۟ يَكْفُرُونَ ﴿70﴾
یہ دنیا میں تھوڑا سا فائدہ ہے، پھر ان کا لوٹنا ہماری طرف ہے، پھر ہم انہیں سخت عذاب چکھائیں گے اس وجہ سے کہ وہ کفر کرتے تھے۔
وَٱتْلُ عَلَيْهِمْ نَبَأَ نُوحٍ إِذْ قَالَ لِقَوْمِهِۦ يَـٰقَوْمِ إِن كَانَ كَبُرَ عَلَيْكُم مَّقَامِي وَتَذْكِيرِي بِـَٔايَـٰتِ ٱللَّهِ فَعَلَى ٱللَّهِ تَوَكَّلْتُ فَأَجْمِعُوٓا۟ أَمْرَكُمْ وَشُرَكَآءَكُمْ ثُمَّ لَا يَكُنْ أَمْرُكُمْ عَلَيْكُمْ غُمَّةٗ ثُمَّ ٱقْضُوٓا۟ إِلَيَّ وَلَا تُنظِرُونِ ﴿71﴾
اور انہیں نوح کی خبر سناؤ، جب اس نے اپنی قوم سے کہا: اے میری قوم! اگر تمہیں میرا ٹھہرنا اور اللہ کی آیات کے ساتھ نصیحت کرنا بھاری لگتا ہے، تو میں نے اللہ پر بھروسہ کیا ہے، پس تم اپنے کام اور اپنے شریکوں کو جمع کر لو، پھر تمہارا کام تم پر پوشیدہ نہ رہے، پھر میرے ساتھ فیصلہ کر لو اور مجھے مہلت نہ دو۔
فَإِن تَوَلَّيْتُمْ فَمَا سَأَلْتُكُم مِّنْ أَجْرٍۖ إِنْ أَجْرِيَ إِلَّا عَلَى ٱللَّهِۖ وَأُمِرْتُ أَنْ أَكُونَ مِنَ ٱلْمُسْلِمِينَ ﴿72﴾
پھر اگر تم منہ موڑو تو میں نے تم سے کوئی اجر نہیں مانگا، میرا اجر تو صرف اللہ پر ہے، اور مجھے حکم دیا گیا ہے کہ میں مسلمانوں میں سے ہوں۔
فَكَذَّبُوهُ فَنَجَّيْنَـٰهُ وَمَن مَّعَهُۥ فِي ٱلْفُلْكِ وَجَعَلْنَـٰهُمْ خَلَـٰٓئِفَ وَأَغْرَقْنَا ٱلَّذِينَ كَذَّبُوا۟ بِـَٔايَـٰتِنَاۚ فَٱنظُرْ كَيْفَ كَانَ عَـٰقِبَةُ ٱلْمُنذَرِينَ ﴿73﴾
پھر انہوں نے اسے جھٹلایا، تو ہم نے اسے اور اس کے ساتھیوں کو کشتی میں نجات دی، اور انہیں جانشین بنایا، اور جنہوں نے ہماری آیات کو جھٹلایا انہیں غرق کر دیا، پس دیکھو ڈرائے جانے والوں کا انجام کیسا ہوا۔
ثُمَّ بَعَثْنَا مِنۢ بَعْدِهِۦ رُسُلًا إِلَىٰ قَوْمِهِمْ فَجَآءُوهُم بِٱلْبَيِّنَـٰتِ فَمَا كَانُوا۟ لِيُؤْمِنُوا۟ بِمَا كَذَّبُوا۟ بِهِۦ مِن قَبْلُۚ كَذَٰلِكَ نَطْبَعُ عَلَىٰ قُلُوبِ ٱلْمُعْتَدِينَ ﴿74﴾
پھر ہم نے اس کے بعد ان کی قوموں کی طرف رسول بھیجے، تو وہ ان کے پاس واضح نشانیاں لے کر آئے، مگر وہ اس پر ایمان نہ لائے جسے وہ پہلے جھٹلا چکے تھے، اسی طرح ہم حد سے بڑھنے والوں کے دلوں پر مہر لگا دیتے ہیں۔
ثُمَّ بَعَثْنَا مِنۢ بَعْدِهِم مُّوسَىٰ وَهَـٰرُونَ إِلَىٰ فِرْعَوْنَ وَمَلَإِيهِۦ بِـَٔايَـٰتِنَا فَٱسْتَكْبَرُوا۟ وَكَانُوا۟ قَوْمٗا مُّجْرِمِينَ ﴿75﴾
پھر ہم نے ان کے بعد موسیٰ اور ہارون کو فرعون اور اس کے سرداروں کی طرف اپنی نشانیاں دے کر بھیجا، تو انہوں نے تکبر کیا اور وہ مجرم قوم تھے۔
فَلَمَّا جَآءَهُمُ ٱلْحَقُّ مِنْ عِندِنَا قَالُوٓا۟ إِنَّ هَـٰذَا لَسِحْرٞ مُّبِينٞ ﴿76﴾
پھر جب ان کے پاس ہماری طرف سے حق آیا تو انہوں نے کہا: بے شک یہ تو کھلا جادو ہے۔
قَالَ مُوسَىٰٓ أَتَقُولُونَ لِلْحَقِّ لَمَّا جَآءَكُمْ أَسِحْرٌ هَـٰذَا وَلَا يُفْلِحُ ٱلسَّـٰحِرُونَ ﴿77﴾
موسیٰ نے کہا: کیا تم حق کے بارے میں جب وہ تمہارے پاس آیا کہتے ہو کہ یہ جادو ہے؟ حالانکہ جادوگر فلاح نہیں پاتے۔
قَالُوٓا۟ أَجِئْتَنَا لِتَلْفِتَنَا عَمَّا وَجَدْنَا عَلَيْهِ ءَابَآءَنَا وَتَكُونَ لَكُمَا ٱلْكِبْرِيَآءُ فِي ٱلْأَرْضِۖ وَمَا نَحْنُ لَكُمَا بِمُؤْمِنِينَ ﴿78﴾
انہوں نے کہا: کیا تم ہمارے پاس اس لیے آئے ہو کہ ہمیں اس سے پھیر دو جس پر ہم نے اپنے باپ دادا کو پایا، اور تم دونوں کو زمین میں بڑائی حاصل ہو؟ اور ہم تم دونوں پر ایمان لانے والے نہیں۔
وَقَالَ فِرْعَوْنُ ٱئْتُونِي بِكُلِّ سَـٰحِرٍ عَلِيمٖ ﴿79﴾
اور فرعون نے کہا: میرے پاس ہر ماہر جادوگر کو لے آؤ۔
فَلَمَّا جَآءَ ٱلسَّحَرَةُ قَالَ لَهُم مُّوسَىٰٓ أَلْقُوا۟ مَآ أَنتُم مُّلْقُونَ ﴿80﴾
پھر جب جادوگر آ گئے تو موسیٰ نے ان سے کہا: تم جو کچھ ڈالنے والے ہو، ڈال دو۔
فَلَمَّآ أَلْقَوْا۟ قَالَ مُوسَىٰ مَا جِئْتُم بِهِ ٱلسِّحْرُۖ إِنَّ ٱللَّهَ سَيُبْطِلُهُۥٓۖ إِنَّ ٱللَّهَ لَا يُصْلِحُ عَمَلَ ٱلْمُفْسِدِينَ ﴿81﴾
پھر جب انہوں نے (اپنا جادو) ڈال دیا تو موسیٰ نے کہا: جو کچھ تم لائے ہو وہ جادو ہے، بے شک اللہ اسے باطل کرے گا، بے شک اللہ فساد کرنے والوں کے عمل کو درست نہیں کرتا۔
وَيُحِقُّ ٱللَّهُ ٱلْحَقَّ بِكَلِمَـٰتِهِۦ وَلَوْ كَرِهَ ٱلْمُجْرِمُونَ ﴿82﴾
اور اللہ اپنے کلمات سے حق کو ثابت کرتا ہے، اگرچہ مجرم اسے ناپسند کریں۔
فَمَآ ءَامَنَ لِمُوسَىٰٓ إِلَّا ذُرِّيَّةٞ مِّن قَوْمِهِۦ عَلَىٰ خَوْفٖ مِّن فِرْعَوْنَ وَمَلَإِهِمْ أَن يَفْتِنَهُمْۚ وَإِنَّ فِرْعَوْنَ لَعَالٖ فِي ٱلْأَرْضِ وَإِنَّهُۥ لَمِنَ ٱلْمُسْرِفِينَ ﴿83﴾
تو موسیٰ پر اس کی قوم میں سے صرف ایک نسل نے ایمان لایا، فرعون اور اس کے سرداروں سے خوف کی وجہ سے کہ وہ انہیں آزمائش میں مبتلا نہ کر دیں، اور بے شک فرعون زمین میں سرکش تھا، اور وہ حد سے بڑھنے والوں میں سے تھا۔
وَقَالَ مُوسَىٰ يَـٰقَوْمِ إِن كُنتُمْ ءَامَنتُم بِٱللَّهِ فَعَلَيْهِ تَوَكَّلُوٓا۟ إِن كُنتُم مُّسْلِمِينَ ﴿84﴾
اور موسیٰ نے کہا: اے میری قوم! اگر تم اللہ پر ایمان لائے ہو تو اسی پر بھروسہ کرو، اگر تم مسلمان ہو۔
فَقَالُوا۟ عَلَى ٱللَّهِ تَوَكَّلْنَاۚ رَبَّنَا لَا تَجْعَلْنَا فِتْنَةٗ لِّلْقَوْمِ ٱلظَّـٰلِمِينَ ﴿85﴾
انہوں نے کہا: ہم نے اللہ پر بھروسہ کیا، اے ہمارے رب! ہمیں ظالم قوم کے لیے آزمائش نہ بنا۔
وَنَجِّنَا بِرَحْمَتِكَ مِنَ ٱلْقَوْمِ ٱلْكَـٰفِرِينَ ﴿86﴾
اور ہمیں اپنی رحمت سے کافر قوم سے نجات دے۔
وَأَوْحَيْنَآ إِلَىٰ مُوسَىٰ وَأَخِيهِ أَن تَبَوَّءَا لِقَوْمِكُمَا بِمِصْرَ بُيُوتٗا وَٱجْعَلُوا۟ بُيُوتَكُمْ قِبْلَةٗ وَأَقِيمُوا۟ ٱلصَّلَوٰةَۚ وَبَشِّرِ ٱلْمُؤْمِنِينَ ﴿87﴾
اور ہم نے موسیٰ اور اس کے بھائی کو وحی کی کہ اپنی قوم کے لیے مصر میں گھر بناؤ، اور اپنے گھروں کو قبلہ بناؤ، اور نماز قائم کرو، اور ایمان والوں کو خوشخبری دو۔
وَقَالَ مُوسَىٰ رَبَّنَآ إِنَّكَ ءَاتَيْتَ فِرْعَوْنَ وَمَلَأَهُۥ زِينَةٗ وَأَمْوَٰلٗا فِي ٱلْحَيَوٰةِ ٱلدُّنْيَا رَبَّنَا لِيُضِلُّوا۟ عَن سَبِيلِكَۚ رَبَّنَا ٱطْمِسْ عَلَىٰٓ أَمْوَٰلِهِمْ وَٱشْدُدْ عَلَىٰ قُلُوبِهِمْ فَلَا يُؤْمِنُوا۟ حَتَّىٰ يَرَوُا۟ ٱلْعَذَابَ ٱلْأَلِيمَ ﴿88﴾
اور موسیٰ نے کہا: اے ہمارے رب! بے شک تو نے فرعون اور اس کے سرداروں کو دنیا کی زندگی میں زینت اور مال دیا ہے، اے ہمارے رب! تاکہ وہ تیرے راستے سے گمراہ کریں، اے ہمارے رب! ان کے مال کو مٹا دے، اور ان کے دلوں کو سخت کر دے، تاکہ وہ ایمان نہ لائیں جب تک دردناک عذاب نہ دیکھ لیں۔
قَالَ قَدْ أُجِيبَتْ دَّعْوَتُكُمَا فَٱسْتَقِيمَا وَلَا تَتَّبِعَانِّ سَبِيلَ ٱلَّذِينَ لَا يَعْلَمُونَ ﴿89﴾
اللہ نے فرمایا: تم دونوں کی دعا قبول کر لی گئی، پس تم ثابت قدم رہو، اور ان لوگوں کے راستے کی پیروی نہ کرو جو علم نہیں رکھتے۔
وَجَاوَزْنَا بِبَنِيٓ إِسْرَٰٓءِيلَ ٱلْبَحْرَ فَأَتْبَعَهُمْ فِرْعَوْنُ وَجُنُودُهُۥ بَغْيٗا وَعَدْوٗاۖ حَتَّىٰٓ إِذَآ أَدْرَكَهُ ٱلْغَرَقُ قَالَ ءَامَنتُ أَنَّهُۥ لَآ إِلَـٰهَ إِلَّا ٱلَّذِيٓ ءَامَنَتْ بِهِۦ بَنُوا۟ إِسْرَٰٓءِيلَ وَأَنَا۠ مِنَ ٱلْمُسْلِمِينَ ﴿90﴾
اور ہم نے بنی اسرائیل کو سمندر سے پار کر دیا، تو فرعون اور اس کے لشکر نے ظلم اور زیادتی سے ان کا پیچھا کیا، یہاں تک کہ جب وہ ڈوبنے لگا تو کہنے لگا: میں ایمان لایا کہ کوئی معبود نہیں مگر وہی جس پر بنی اسرائیل ایمان لائے، اور میں مسلمانوں میں سے ہوں۔
ءَٟلْـَٔـٰنَ وَقَدْ عَصَيْتَ قَبْلُ وَكُنتَ مِنَ ٱلْمُفْسِدِينَ ﴿91﴾
(جواب دیا گیا:) اب؟ حالانکہ تو پہلے نافرمانی کرتا رہا اور فساد کرنے والوں میں سے تھا۔
فَٱلْيَوْمَ نُنَجِّيكَ بِبَدَنِكَ لِتَكُونَ لِمَنْ خَلْفَكَ ءَايَةٗۚ وَإِنَّ كَثِيرٗا مِّنَ ٱلنَّاسِ عَنْ ءَايَـٰتِنَا لَغَـٰفِلُونَ ﴿92﴾
پس آج ہم تیرے جسم کو بچا لیں گے تاکہ تو اپنے بعد والوں کے لیے نشانی بن جائے، اور بے شک بہت سے لوگ ہماری نشانیوں سے غافل ہیں۔
وَلَقَدْ بَوَّأْنَا بَنِيٓ إِسْرَٰٓءِيلَ مُبَوَّأَ صِدْقٖ وَرَزَقْنَـٰهُم مِّنَ ٱلطَّيِّبَـٰتِ فَمَا ٱخْتَلَفُوا۟ حَتَّىٰ جَآءَهُمُ ٱلْعِلْمُۚ إِنَّ رَبَّكَ يَقْضِي بَيْنَهُمۡ يَوْمَ ٱلْقِيَـٰمَةِ فِيمَا كَانُوا۟ فِيهِ يَخْتَلِفُونَ ﴿93﴾
اور بے شک ہم نے بنی اسرائیل کو سچی جگہ میں ٹھہرایا، اور انہیں پاکیزہ چیزوں سے رزق دیا، پھر انہوں نے اختلاف نہیں کیا یہاں تک کہ ان کے پاس علم آ گیا، بے شک تمہارا رب قیامت کے دن ان کے درمیان ان باتوں کا فیصلہ کرے گا جن میں وہ اختلاف کرتے تھے۔
فَإِن كُنتَ فِي شَكّٖ مِّمَّآ أَنزَلْنَآ إِلَيْكَ فَسْـَٔلِ ٱلَّذِينَ يَقْرَءُونَ ٱلْكِتَـٰبَ مِن قَبْلِكَۚ لَقَدْ جَآءَكَ ٱلْحَقُّ مِن رَّبِّكَ فَلَا تَكُونَنَّ مِنَ ٱلْمُمْتَرِينَ ﴿94﴾
پس اگر تمہیں اس چیز میں شک ہو جو ہم نے تمہاری طرف نازل کی ہے تو ان لوگوں سے پوچھ لو جو تم سے پہلے کتاب پڑھتے ہیں، بے شک تمہارے رب کی طرف سے تمہارے پاس حق آ چکا ہے، پس تم شک کرنے والوں میں سے نہ ہو جانا۔
وَلَا تَكُونَنَّ مِنَ ٱلَّذِينَ كَذَّبُوا۟ بِـَٔايَـٰتِ ٱللَّهِ فَتَكُونَ مِنَ ٱلْخَـٰسِرِينَ ﴿95﴾
اور ان لوگوں میں سے نہ ہو جانا جنہوں نے اللہ کی آیات کو جھٹلایا، ورنہ تم نقصان اٹھانے والوں میں سے ہو جاؤ گے۔
إِنَّ ٱلَّذِينَ حَقَّتْ عَلَيْهِمْ كَلِمَتُ رَبِّكَ لَا يُؤْمِنُونَ ﴿96﴾
بے شک جن پر تمہارے رب کا فرمان ثابت ہو چکا ہے، وہ ایمان نہیں لائیں گے۔
وَلَوْ جَآءَتْهُمْ كُلُّ ءَايَةٍ حَتَّىٰ يَرَوُا۟ ٱلْعَذَابَ ٱلْأَلِيمَ ﴿97﴾
اگرچہ ان کے پاس ہر نشانی آ جائے، یہاں تک کہ وہ دردناک عذاب دیکھ لیں۔
فَلَوْلَا كَانَتْ قَرْيَةٌ ءَامَنَتْ فَنَفَعَهَآ إِيمَـٰنُهَآ إِلَّا قَوْمَ يُونُسَۚ لَمَّآ ءَامَنُوا۟ كَشَفْنَا عَنْهُمْ عَذَابَ ٱلْخِزْيِ فِي ٱلْحَيَوٰةِ ٱلدُّنْيَا وَمَتَّعْنَـٰهُمْ إِلَىٰ حِينٖ ﴿98﴾
تو کیوں نہ کوئی بستی ایسی ہوئی جو ایمان لائی اور اس کا ایمان اسے نفع دے گیا؟ سوائے یونس کی قوم کے، جب وہ ایمان لائے تو ہم نے ان سے دنیا کی زندگی میں رسوائی کا عذاب دور کر دیا اور انہیں ایک وقت تک فائدہ پہنچایا۔
وَلَوْ شَآءَ رَبُّكَ لَـَٔامَنَ مَن فِي ٱلْأَرْضِ كُلُّهُمْ جَمِيعًاۚ أَفَأَنتَ تُكْرِهُ ٱلنَّاسَ حَتَّىٰ يَكُونُوا۟ مُؤْمِنِينَ ﴿99﴾
اور اگر تمہارا رب چاہتا تو زمین میں سب لوگ ایمان لے آتے، تو کیا تم لوگوں کو مجبور کرو گے کہ وہ ایمان لے آئیں؟
وَمَا كَانَ لِنَفْسٍ أَن تُؤْمِنَ إِلَّا بِإِذْنِ ٱللَّهِۚ وَيَجْعَلُ ٱلرِّجْسَ عَلَى ٱلَّذِينَ لَا يَعْقِلُونَ ﴿100﴾
اور کسی جان کے لیے ممکن نہیں کہ وہ ایمان لائے مگر اللہ کے حکم سے، اور وہ گندگی ان پر ڈال دیتا ہے جو عقل نہیں رکھتے۔
قُلِ ٱنظُرُوا۟ مَاذَا فِي ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۚ وَمَا تُغْنِي ٱلْـَٔايَـٰتُ وَٱلنُّذُرُ عَن قَوْمٖ لَّا يُؤْمِنُونَ ﴿101﴾
کہہ دو: دیکھو آسمانوں اور زمین میں کیا کچھ ہے، اور نشانیاں اور ڈرانے والے ان لوگوں کو کچھ فائدہ نہیں دیتے جو ایمان نہیں لاتے۔
فَهَلْ يَنتَظِرُونَ إِلَّا مِثْلَ أَيَّامِ ٱلَّذِينَ خَلَوْا۟ مِن قَبْلِهِمۡۚ قُلْ فَٱنتَظِرُوٓا۟ إِنِّي مَعَكُم مِّنَ ٱلْمُنتَظِرِينَ ﴿102﴾
تو کیا وہ اس کے سوا کسی چیز کا انتظار کر رہے ہیں جو ان سے پہلے لوگوں کے دنوں کی طرح ہو؟ کہہ دو: انتظار کرو، میں بھی تمہارے ساتھ انتظار کرنے والوں میں سے ہوں۔
ثُمَّ نُنَجِّي رُسُلَنَا وَٱلَّذِينَ ءَامَنُوا۟ۚ كَذَٰلِكَ حَقًّا عَلَيْنَا نُنجِ ٱلْمُؤْمِنِينَ ﴿103﴾
پھر ہم اپنے رسولوں کو اور ایمان والوں کو نجات دیتے ہیں، اسی طرح ہم پر حق ہے کہ ہم ایمان والوں کو نجات دیں۔
قُلْ يَـٰٓأَيُّهَا ٱلنَّاسُ إِن كُنتُمْ فِي شَكّٖ مِّن دِينِي فَلَآ أَعْبُدُ ٱلَّذِينَ تَعْبُدُونَ مِن دُونِ ٱللَّهِ وَلَـٰكِنْ أَعْبُدُ ٱللَّهَ ٱلَّذِي يَتَوَفَّىٰكُمْۖ وَأُمِرْتُ أَنْ أَكُونَ مِنَ ٱلْمُؤْمِنِينَ ﴿104﴾
کہہ دو: اے لوگو! اگر تم میرے دین کے بارے میں شک میں ہو تو میں ان کی عبادت نہیں کرتا جنہیں تم اللہ کے سوا پوجتے ہو، بلکہ میں اس اللہ کی عبادت کرتا ہوں جو تمہیں موت دیتا ہے، اور مجھے حکم دیا گیا ہے کہ میں ایمان والوں میں سے ہوں۔
وَأَنْ أَقِمْ وَجْهَكَ لِلدِّينِ حَنِيفٗا وَلَا تَكُونَنَّ مِنَ ٱلْمُشْرِكِينَ ﴿105﴾
اور یہ کہ تم اپنے چہرے کو دین کی طرف یکسو کر لو، اور مشرکوں میں سے نہ ہو جاؤ۔
وَلَا تَدْعُ مِن دُونِ ٱللَّهِ مَا لَا يَنفَعُكَ وَلَا يَضُرُّكَۖ فَإِن فَعَلْتَ فَإِنَّكَ إِذٗا مِّنَ ٱلظَّـٰلِمِينَ ﴿106﴾
اور اللہ کے سوا کسی ایسی چیز کو نہ پکارو جو نہ تمہیں نفع دے سکتی ہے اور نہ نقصان، پھر اگر تم نے ایسا کیا تو یقیناً تم ظالموں میں سے ہو جاؤ گے۔
وَإِن يَمْسَسْكَ ٱللَّهُ بِضُرّٖ فَلَا كَاشِفَ لَهُۥٓ إِلَّا هُوَۖ وَإِن يُرِدْكَ بِخَيْرٖ فَلَا رَآدَّ لِفَضْلِهِۦۚ يُصِيبُ بِهِۦ مَن يَشَآءُ مِنْ عِبَادِهِۦۚ وَهُوَ ٱلْغَفُورُ ٱلرَّحِيمُ ﴿107﴾
اور اگر اللہ تمہیں کسی تکلیف میں مبتلا کرے تو اس کے سوا کوئی اسے دور کرنے والا نہیں، اور اگر وہ تمہیں کسی بھلائی سے نوازنا چاہے تو اس کے فضل کو کوئی روکنے والا نہیں، وہ اپنے بندوں میں سے جسے چاہے پہنچاتا ہے، اور وہی بخشنے والا، رحم کرنے والا ہے۔
قُلْ يَـٰٓأَيُّهَا ٱلنَّاسُ قَدْ جَآءَكُمُ ٱلْحَقُّ مِن رَّبِّكُمْۖ فَمَنِ ٱهْتَدَىٰ فَإِنَّمَا يَهْتَدِي لِنَفْسِهِۦۖ وَمَن ضَلَّ فَإِنَّمَا يَضِلُّ عَلَيْهَاۖ وَمَآ أَنَا۠ عَلَيْكُم بِوَكِيلٖ ﴿108﴾
کہہ دو: اے لوگو! تمہارے رب کی طرف سے تمہارے پاس حق آ چکا ہے، پس جو ہدایت پاتا ہے وہ اپنے ہی فائدے کے لیے ہدایت پاتا ہے، اور جو گمراہ ہوتا ہے وہ اپنے ہی نقصان کے لیے گمراہ ہوتا ہے، اور میں تم پر نگہبان نہیں ہوں۔
وَٱتَّبِعْ مَا يُوحَىٰٓ إِلَيْكَ وَٱصْبِرْ حَتَّىٰ يَحْكُمَ ٱللَّهُۚ وَهُوَ خَيْرُ ٱلْحَـٰكِمِينَ ﴿109﴾
اور اس چیز کی پیروی کرو جو تمہاری طرف وحی کی جاتی ہے، اور صبر کرو یہاں تک کہ اللہ فیصلہ کرے، اور وہ سب سے بہتر فیصلہ کرنے والا ہے۔
سورة هود
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
الٓرۚ كِتَٰبٌ أُحْكِمَتْ ءَايَاتُهُۥ ثُمَّ فُصِّلَتْ مِن لَّدُنْ حَكِيمٍ خَبِيرٍ ﴿1﴾
الٓر۔ یہ ایک ایسی کتاب ہے جس کی آیتیں محکم کی گئیں، پھر تفصیل سے بیان کی گئیں، ایک حکمت والے، خبردار کی طرف سے۔
أَلَّا تَعْبُدُوٓا۟ إِلَّا ٱللَّهَۚ إِنَّنِي لَكُم مِّنْهُ نَذِيرٞ وَبَشِيرٞ ﴿2﴾
کہ تم اللہ کے سوا کسی کی عبادت نہ کرو، بے شک میں تمہارے لیے اس کی طرف سے ڈرانے والا اور خوشخبری دینے والا ہوں۔
وَأَنِ ٱسْتَغْفِرُوا۟ رَبَّكُمْ ثُمَّ تُوبُوٓا۟ إِلَيْهِ يُمَتِّعْكُم مَّتَـٰعًا حَسَنًا إِلَىٰٓ أَجَلٖ مُّسَمّٗى وَيُؤْتِ كُلَّ ذِي فَضْلٖ فَضْلَهُۥۖ وَإِن تَوَلَّوْا۟ فَإِنِّيٓ أَخَافُ عَلَيْكُمْ عَذَابَ يَوْمٖ كَبِيرٍ ﴿3﴾
اور یہ کہ اپنے رب سے معافی مانگو، پھر اس کی طرف رجوع کرو، وہ تمہیں ایک مقررہ وقت تک اچھا فائدہ دے گا، اور ہر صاحبِ فضل کو اس کا فضل عطا کرے گا، اور اگر تم منہ موڑو گے تو میں تم پر ایک بڑے دن کے عذاب سے ڈرتا ہوں۔
إِلَى ٱللَّهِ مَرْجِعُكُمْۖ وَهُوَ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٌ ﴿4﴾
تم سب کو اللہ ہی کی طرف لوٹنا ہے، اور وہ ہر چیز پر قادر ہے۔
أَلَآ إِنَّهُمْ يَثْنُونَ صُدُورَهُمْ لِيَسْتَخْفُوا۟ مِنْهُۚ أَلَا حِينَ يَسْتَغْشُونَ ثِيَابَهُمْ يَعْلَمُ مَا يُسِرُّونَ وَمَا يُعْلِنُونَۚ إِنَّهُۥ عَلِيمُۢ بِذَاتِ ٱلصُّدُورِ ﴿5﴾
خبردار! وہ اپنے سینے موڑتے ہیں تاکہ اس سے چھپ جائیں، خبردار! جب وہ اپنے کپڑے اوڑھ لیتے ہیں، وہ جانتا ہے جو کچھ وہ چھپاتے ہیں اور جو ظاہر کرتے ہیں، بے شک وہ سینوں کے بھیدوں کو جاننے والا ہے۔
