28 پارہ


سورة المجادلة
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
قَدْ سَمِعَ ٱللَّهُ قَوْلَ ٱلَّتِى تُجَـٰدِلُكَ فِى زَوْجِهَا وَتَشْتَكِىٓ إِلَى ٱللَّهِ وَٱللَّهُ يَسْمَعُ تَحَاوُرَكُمَآ ۚ إِنَّ ٱللَّهَ سَمِيعٌۢ بَصِيرٌ ﴿1﴾
اللہ نے اس عورت کی بات سن لی جو اپنے شوہر کے بارے میں آپ سے جھگڑ رہی تھی اور اللہ سے فریاد کر رہی تھی، اور اللہ تم دونوں کی گفتگو سن رہا تھا۔ بے شک اللہ سننے والا، دیکھنے والا ہے۔
ٱلَّذِينَ يُظَـٰهِرُونَ مِنكُم مِّن نِّسَآئِهِمْ مَا هُنَّ أُمَّهَـٰتِهِمْ ۖ إِنْ أُمَّهَـٰتُهُمْ إِلَّا ٱلَّـٰٓـِٔى وَلَدْنَهُمْ ۚ وَإِنَّهُمْ لَيَقُولُونَ مُنكَرًۭا مِّنَ ٱلْقَوْلِ وَزُورًۭا ۚ وَإِنَّ ٱللَّهَ لَعَفُوٌّ غَفُورٌۭ ﴿2﴾
جو لوگ اپنی بیویوں سے ظہار کرتے ہیں، وہ ان کی مائیں نہیں بن جاتیں۔ ان کی مائیں تو وہی ہیں جنہوں نے انہیں جنا۔ وہ ایک ناپسندیدہ اور جھوٹی بات کہتے ہیں، اور بے شک اللہ معاف کرنے والا، بخشنے والا ہے۔
وَٱلَّذِينَ يُظَـٰهِرُونَ مِن نِّسَآئِهِمْ ثُمَّ يَعُودُونَ لِمَا قَالُوا۟ فَتَحْرِيرُ رَقَبَةٍۢ مِّن قَبْلِ أَن يَتَمَآسَّا ۚ ذَٰلِكُمْ تُوعَظُونَ بِهِۦ ۚ وَٱللَّهُ بِمَا تَعْمَلُونَ خَبِيرٌۭ ﴿3﴾
اور جو لوگ اپنی بیویوں سے ظہار کریں پھر اپنی بات سے رجوع کریں، تو ان پر لازم ہے کہ دونوں کے تعلق سے پہلے ایک غلام آزاد کریں۔ یہ وہ ہے جس سے تم نصیحت حاصل کرو، اور اللہ تمہارے اعمال سے خوب باخبر ہے۔
فَمَن لَّمْ يَجِدْ فَصِيَامُ شَهْرَيْنِ مُتَتَابِعَيْنِ مِن قَبْلِ أَن يَتَمَآسَّا ۖ فَمَن لَّمْ يَسْتَطِعْ فَإِطْعَامُ سِتِّينَ مِسْكِينًۭا ۚ ذَٰلِكَ لِتُؤْمِنُوا۟ بِٱللَّهِ وَرَسُولِهِۦ ۚ وَتِلْكَ حُدُودُ ٱللَّهِ ۗ وَلِلْكَـٰفِرِينَ عَذَابٌ أَلِيمٌۭ ﴿4﴾
پھر جو غلام نہ پائے، وہ دو مہینے لگاتار روزے رکھے اس سے پہلے کہ دونوں تعلق قائم کریں۔ اور جو اس کی بھی طاقت نہ رکھے، وہ ساٹھ مسکینوں کو کھانا کھلائے۔ یہ اس لیے ہے تاکہ تم اللہ اور اس کے رسول پر ایمان لاؤ۔ اور یہ اللہ کی حدود ہیں، اور کافروں کے لیے دردناک عذاب ہے۔
إِنَّ ٱلَّذِينَ يُحَآدُّونَ ٱللَّهَ وَرَسُولَهُۥ كُبِتُوا۟ كَمَا كُبِتَ ٱلَّذِينَ مِن قَبْلِهِمْ ۚ وَقَدْ أَنزَلْنَآ ءَايَـٰتٍۢ بَيِّنَـٰتٍۢ ۚ وَلِلْكَـٰفِرِينَ عَذَابٌۭ مُّهِينٌ ﴿5﴾
بے شک جو لوگ اللہ اور اس کے رسول کی مخالفت کرتے ہیں وہ ذلیل کیے گئے جیسے ان سے پہلے لوگ ذلیل کیے گئے تھے۔ اور ہم نے واضح آیات نازل کی ہیں، اور کافروں کے لیے رسوا کن عذاب ہے۔
يَوْمَ يَبْعَثُهُمُ ٱللَّهُ جَمِيعًۭا فَيُنَبِّئُهُم بِمَا عَمِلُوٓا۟ ۚ أَحْصَىٰهُ ٱللَّهُ وَنَسُوهُ ۚ وَٱللَّهُ عَلَىٰ كُلِّ شَىْءٍۢ شَهِيدٌ ﴿6﴾
جس دن اللہ ان سب کو اٹھائے گا، پھر انہیں ان کے اعمال سے آگاہ کرے گا۔ اللہ نے ان سب کو شمار کر رکھا ہے، اور وہ بھول گئے ہیں۔ اور اللہ ہر چیز پر گواہ ہے۔
أَلَمْ تَرَ أَنَّ ٱللَّهَ يَعْلَمُ مَا فِى ٱلسَّمَـٰوَٰتِ وَمَا فِى ٱلْأَرْضِ ۖ مَا يَكُونُ مِن نَّجْوَىٰ ثَلَـٰثَةٍ إِلَّا هُوَ رَابِعُهُمْ وَلَا خَمْسَةٍ إِلَّا هُوَ سَادِسُهُمْ وَلَآ أَدْنَىٰ مِن ذَٰلِكَ وَلَآ أَكْثَرَ إِلَّا هُوَ مَعَهُمْ أَيْنَ مَا كَانُوا۟ ۖ ثُمَّ يُن... عَمِلُوا۟ يَوْمَ ٱلْقِيَـٰمَةِ ۚ إِنَّ ٱللَّهَ بِكُلِّ شَىْءٍ عَلِيمٌ ﴿7﴾
کیا تم نے نہیں دیکھا کہ اللہ جانتا ہے جو کچھ آسمانوں اور زمین میں ہے؟ تین آدمیوں کی سرگوشی نہیں ہوتی مگر وہ چوتھا ہوتا ہے، اور پانچ کی نہیں مگر وہ چھٹا ہوتا ہے، اور نہ اس سے کم اور نہ زیادہ مگر وہ ان کے ساتھ ہوتا ہے جہاں کہیں بھی ہوں۔ پھر قیامت کے دن وہ انہیں ان کے اعمال سے آگاہ کرے گا۔ بے شک اللہ ہر چیز کو جاننے والا ہے۔
أَلَمْ تَرَ إِلَى ٱلَّذِينَ نُهُوا۟ عَنِ ٱلنَّجْوَىٰ ثُمَّ يَعُودُونَ لِمَا نُهُوا۟ عَنْهُ وَيَتَنَـٰجَوْنَ بِٱلْإِثْمِ وَٱلْعُدْوَٰنِ وَمَعْصِيَتِ ٱلرَّسُولِ ۖ وَإِذَا جَآءُوكَ حَيَّوْكَ بِمَا لَمْ يُحَيِّكَ بِهِ ٱللَّهُ وَيَقُولُونَ فِىٓ أَنفُسِهِمْ لَوْلَا يُعَذِّبُنَا ٱللَّهُ بِمَا نَقُولُ ۚ حَسْبُهُمْ جَهَنَّمُ ۚ يَصْلَوْنَهَا ۖ فَبِئْسَ ٱلْمَصِيرُ ﴿8﴾
کیا تم نے ان لوگوں کو نہیں دیکھا جنہیں سرگوشی سے روکا گیا تھا، پھر وہ اسی چیز کی طرف لوٹتے ہیں جس سے روکا گیا تھا، اور وہ گناہ، زیادتی اور رسول کی نافرمانی کی باتیں کرتے ہیں۔ اور جب وہ آپ کے پاس آتے ہیں تو آپ کو اس طرح سلام کرتے ہیں جس طرح اللہ نے سلام نہیں فرمایا، اور اپنے دلوں میں کہتے ہیں: اللہ ہمیں اس بات پر کیوں عذاب نہیں دیتا جو ہم کہتے ہیں؟ ان کے لیے جہنم کافی ہے، وہ اس میں داخل ہوں گے، اور وہ بہت برا ٹھکانہ ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ إِذَا تَنَـٰجَيْتُمْ فَلَا تَتَنَـٰجَوْا۟ بِٱلْإِثْمِ وَٱلْعُدْوَٰنِ وَمَعْصِيَتِ ٱلرَّسُولِ وَتَنَـٰجَوْا۟ بِٱلْبِرِّ وَٱلتَّقْوَىٰ ۖ وَٱتَّقُوا۟ ٱللَّهَ ٱلَّذِىٓ إِلَيْهِ تُحْشَرُونَ ﴿9﴾
اے ایمان والو! جب تم آپس میں سرگوشی کرو تو گناہ، زیادتی اور رسول کی نافرمانی کی بات نہ کرو، بلکہ نیکی اور تقویٰ کی بات کرو، اور اس اللہ سے ڈرو جس کی طرف تم جمع کیے جاؤ گے۔
إِنَّمَا ٱلنَّجْوَىٰ مِنَ ٱلشَّيْطَـٰنِ لِيَحْزُنَ ٱلَّذِينَ ءَامَنُوا۟ وَلَيْسَ بِضَآرِّهِمْ شَيْـًٔا إِلَّا بِإِذْنِ ٱللَّهِ ۚ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُؤْمِنُونَ ﴿10﴾
بے شک سرگوشی شیطان کی طرف سے ہوتی ہے تاکہ ایمان والوں کو غمگین کرے، حالانکہ وہ انہیں کچھ نقصان نہیں پہنچا سکتا مگر اللہ کے حکم سے۔ اور ایمان والوں کو اللہ ہی پر بھروسہ کرنا چاہیے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ إِذَا قِيلَ لَكُمْ تَفَسَّحُوا۟ فِى ٱلْمَجَـٰلِسِ فَٱفْسَحُوا۟ يَفْسَحِ ٱللَّهُ لَكُمْ ۖ وَإِذَا قِيلَ ٱنشُزُوا۟ فَٱنشُزُوا۟ ۖ يَرْفَعِ ٱللَّهُ ٱلَّذِينَ ءَامَنُوا۟ مِنكُمْ وَٱلَّذِينَ أُوتُوا۟ ٱلْعِلْمَ دَرَجَـٰتٍۢ ۚ وَٱللَّهُ بِمَا تَعْمَلُونَ خَبِيرٌۭ ﴿11﴾
اے ایمان والو! جب تم سے کہا جائے مجلس میں جگہ دو تو جگہ دے دیا کرو، اللہ تمہیں کشادگی دے گا۔ اور جب کہا جائے اٹھو تو اٹھ جایا کرو، اللہ تم میں سے ایمان والوں اور علم والوں کے درجے بلند کرے گا، اور اللہ تمہارے اعمال سے خوب باخبر ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ إِذَا نَـٰجَيْتُمُ ٱلرَّسُولَ فَقَدِّمُوا۟ بَيْنَ يَدَىۡ نَجۡوَىٰكُمۡ صَدَقَةٗ ۚ ذَٰلِكَ خَيۡرٞ لَّكُمۡ وَأَطۡهَرُۚ فَإِن لَّمۡ تَجِدُوا۟ فَإِنَّ ٱللَّهَ غَفُورٌۭ رَّحِيمٌ ﴿12﴾
اے ایمان والو! جب تم رسول سے تنہائی میں بات کرو تو اپنی بات سے پہلے صدقہ دیا کرو، یہ تمہارے لیے بہتر اور پاکیزہ ہے۔ پھر اگر تم نہ پاؤ تو بے شک اللہ بخشنے والا، رحم کرنے والا ہے۔
أَأَشْفَقْتُمْ أَن تُقَدِّمُوا۟ بَيْنَ يَدَىۡ نَجۡوَىٰكُمۡ صَدَقَـٰتٖۚ فَإِذۡ لَمۡ تَفۡعَلُوا۟ وَتَابَ ٱللَّهُ عَلَيْكُمۡ فَأَقِيمُوا۟ ٱلصَّلَوٰةَ وَءَاتُوا۟ ٱلزَّكَوٰةَ وَأَطِيعُوا۟ ٱللَّهَ وَرَسُولَهُۥۚ وَٱللَّهُ خَبِيرُۢ بِمَا تَعۡمَلُونَ ﴿13﴾
کیا تم صدقہ دینے سے گھبرا گئے اس سے پہلے کہ رسول سے تنہائی میں بات کرو؟ پھر جب تم نے ایسا نہ کیا اور اللہ نے تم پر رجوع فرمایا، تو نماز قائم کرو، زکوٰۃ دو، اور اللہ و رسول کی اطاعت کرو۔ اور اللہ تمہارے اعمال سے خوب باخبر ہے۔
أَلَمْ تَرَ إِلَى ٱلَّذِينَ تَوَلَّوْا۟ قَوْمًۭا غَضِبَ ٱللَّهُ عَلَيْهِم ۚ مَّا هُم مِّنكُمْ وَلَا مِنْهُمْ وَيَحْلِفُونَ عَلَى ٱلْكَذِبِ وَهُمْ يَعْلَمُونَ ﴿14﴾
کیا آپ نے ان لوگوں کو نہیں دیکھا جنہوں نے ایسے لوگوں سے دوستی کی جن پر اللہ غضبناک ہوا؟ وہ نہ تم میں سے ہیں اور نہ ان میں سے، اور وہ جانتے ہوئے جھوٹی قسمیں کھاتے ہیں۔
أَعَدَّ ٱللَّهُ لَهُمْ عَذَابًۭا شَدِيدًا ۖ إِنَّهُمْ سَآءَ مَا كَانُوا۟ يَعْمَلُونَ ﴿15﴾
اللہ نے ان کے لیے سخت عذاب تیار کر رکھا ہے، بے شک وہ بہت برا عمل کرتے رہے۔
ٱتَّخَذُوٓا۟ أَيْمَـٰنَهُمْ جُنَّةًۭ فَصَدُّوا۟ عَن سَبِيلِ ٱللَّهِ فَلَهُمْ عَذَابٌۭ مُّهِينٌۭ ﴿16﴾
انہوں نے اپنی قسموں کو ڈھال بنا لیا اور اللہ کے راستے سے روکا، تو ان کے لیے رسوا کن عذاب ہے۔
لَّن تُغْنِىَ عَنْهُمْ أَمْوَٰلُهُمْ وَلَآ أَوْلَـٰدُهُم مِّنَ ٱللَّهِ شَيْـًۭٔا ۚ أُو۟لَـٰٓئِكَ أَصْحَـٰبُ ٱلنَّارِ ۖ هُمْ فِيهَا خَـٰلِدُونَ ﴿17﴾
نہ ان کے مال اور نہ ان کی اولاد اللہ کے مقابلے میں ان کے کچھ کام آئیں گے، یہی لوگ جہنم والے ہیں، وہ اس میں ہمیشہ رہیں گے۔
يَوْمَ يَبْعَثُهُمُ ٱللَّهُ جَمِيعًۭا فَيَحْلِفُونَ لَهُۥ كَمَا يَحْلِفُونَ لَكُمْ وَيَحْسَبُونَ أَنَّهُمْ عَلَىٰ شَىْءٍ ۚ أَلَآ إِنَّهُمْ هُمُ ٱلْكَـٰذِبُونَ ﴿18﴾
جس دن اللہ انہیں سب کو اٹھائے گا، تو وہ اس کے سامنے بھی قسمیں کھائیں گے جیسے تمہارے سامنے کھاتے ہیں، اور وہ سمجھتے ہیں کہ وہ کسی چیز پر ہیں۔ خبردار! بے شک وہی جھوٹے ہیں۔
ٱسْتَحْوَذَ عَلَيْهِمُ ٱلشَّيْطَـٰنُ فَأَنسَىٰهُمْ ذِكْرَ ٱللَّهِ ۚ أُو۟لَـٰٓئِكَ حِزْبُ ٱلشَّيْطَـٰنِ ۚ أَلَآ إِنَّ حِزْبَ ٱلشَّيْطَـٰنِ هُمُ ٱلْخَـٰسِرُونَ ﴿19﴾
شیطان نے ان پر قابو پا لیا اور انہیں اللہ کا ذکر بھلا دیا۔ یہی لوگ شیطان کی جماعت ہیں۔ خبردار! بے شک شیطان کی جماعت ہی نقصان اٹھانے والی ہے۔
إِنَّ ٱلَّذِينَ يُحَآدُّونَ ٱللَّهَ وَرَسُولَهُۥٓ أُو۟لَـٰٓئِكَ فِى ٱلْأَذَلِّينَ ﴿20﴾
بے شک جو لوگ اللہ اور اس کے رسول کی مخالفت کرتے ہیں، وہی سب سے زیادہ ذلیل ہیں۔
كَتَبَ ٱللَّهُ لَأَغْلِبَنَّ أَنَا۠ وَرُسُلِىٓ ۚ إِنَّ ٱللَّهَ قَوِىٌّ عَزِيزٌ ﴿21﴾
اللہ نے لکھ دیا ہے: میں اور میرے رسول ضرور غالب آئیں گے۔ بے شک اللہ زبردست، غالب ہے۔
لَا تَجِدُ قَوْمًا يُؤْمِنُونَ بِٱللَّهِ وَٱلْيَوْمِ ٱلْـَٔاخِرِ يُوَآدُّونَ مَنْ حَآدَّ ٱللَّهَ وَرَسُولَهُۥ وَلَوْ كَانُوٓا۟ ءَابَآءَهُمْ أَوْ أَبْنَآءَهُمْ أَوْ إِخْوَٰنَهُمْ أَوْ عَشِيرَتَهُمْ ۚ أُو۟لَـٰٓئِكَ كَتَبَ فِى قُلُوبِهِمُ ٱلْإِيمَـٰنَ وَأَيَّدَهُم بِرُوحٍۢ مِّنْهُ ۖ وَيُدْخِلُهُمْ جَنَّـٰتٍۢ تَجْرِى مِن تَحْتِهَا ٱلْأَنْهَـٰرُ خَـٰلِدِينَ فِيهَا ۚ رَضِىَ ٱللَّهُ عَنْهُمْ وَرَضُوا۟ عَنْهُ ۚ أُو۟لَـٰٓئِكَ حِزْبُ ٱللَّهِ ۚ أَلَآ إِنَّ حِزْبَ ٱللَّهِ هُمُ ٱلْمُفْلِحُونَ ﴿22﴾
آپ ایسے لوگوں کو نہیں پائیں گے جو اللہ اور آخرت کے دن پر ایمان رکھتے ہوں اور ان لوگوں سے محبت کریں جو اللہ اور اس کے رسول کی مخالفت کرتے ہیں، خواہ وہ ان کے باپ ہوں یا بیٹے یا بھائی یا خاندان والے۔ یہی وہ لوگ ہیں جن کے دلوں میں اللہ نے ایمان لکھ دیا ہے اور انہیں اپنی طرف سے روح کے ذریعے قوت دی ہے، اور وہ انہیں ایسی جنتوں میں داخل کرے گا جن کے نیچے نہریں بہتی ہیں، وہ ان میں ہمیشہ رہیں گے۔ اللہ ان سے راضی ہوا اور وہ اللہ سے راضی ہوئے۔ یہی اللہ کی جماعت ہے۔ خبردار! بے شک اللہ کی جماعت ہی کامیاب ہے۔
سورة الحشر
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
سَبَّحَ لِلَّهِ مَا فِى ٱلسَّمَـٰوَٰتِ وَمَا فِى ٱلْأَرْضِ ۖ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿1﴾
اللہ کی تسبیح کرتی ہیں وہ سب چیزیں جو آسمانوں اور زمین میں ہیں، اور وہی زبردست، حکمت والا ہے۔
هُوَ ٱلَّذِىٓ أَخْرَجَ ٱلَّذِينَ كَفَرُوا۟ مِنْ أَهْلِ ٱلْكِتَـٰبِ مِن دِيَـٰرِهِمْ لِأَوَّلِ ٱلْحَشْرِ ۚ مَا ظَنَنتُمْ أَن يَخْرُجُوا۟ ۖ وَظَنُّوا۟ أَنَّهُم مَّانِعَتُهُمْ حُصُونُهُم مِّنَ ٱللَّهِ فَأَتَىٰهُمُ ٱللَّهُ مِنْ حَيْثُ لَمْ يَحْتَسِبُوا۟ وَقَذَفَ فِى قُلُوبِهِمُ ٱلرُّعْبَ يُخْرِبُونَ بُيُوتَهُم بِأَيْدِيهِمْ وَأَيْدِى ٱلْمُؤْمِنِينَ فَٱعْتَبِرُوا۟ يَـٰٓأُو۟لِى ٱلْأَبْصَـٰرِ ﴿2﴾
وہی ہے جس نے اہلِ کتاب میں سے کافروں کو ان کے گھروں سے پہلے ہی جمع ہونے پر نکال باہر کیا۔ تم نے گمان بھی نہ کیا تھا کہ وہ نکلیں گے، اور وہ سمجھتے تھے کہ ان کے قلعے انہیں اللہ سے بچا لیں گے، مگر اللہ ان پر ایسی جگہ سے آیا جہاں کا انہیں گمان بھی نہ تھا، اور اس نے ان کے دلوں میں رعب ڈال دیا۔ وہ اپنے گھروں کو اپنے ہاتھوں اور مومنوں کے ہاتھوں سے برباد کر رہے تھے، پس اے بصیرت والو! عبرت حاصل کرو۔
وَلَوْلَآ أَن كَتَبَ ٱللَّهُ عَلَيْهِمُ ٱلْجَلَآءَ لَعَذَّبَهُمْ فِى ٱلدُّنْيَا ۖ وَلَهُمْ فِى ٱلْـَٔاخِرَةِ عَذَابُ ٱلنَّارِ ﴿3﴾
اور اگر اللہ نے ان کے لیے جلاوطنی نہ لکھی ہوتی تو وہ انہیں دنیا میں ضرور عذاب دیتا، اور آخرت میں ان کے لیے آگ کا عذاب ہے۔
ذَٰلِكَ بِأَنَّهُمْ شَآقُّوا۟ ٱللَّهَ وَرَسُولَهُۥ ۚ وَمَن يُشَآقِّ ٱللَّهَ فَإِنَّ ٱللَّهَ شَدِيدُ ٱلْعِقَابِ ﴿4﴾
یہ اس لیے کہ انہوں نے اللہ اور اس کے رسول کی مخالفت کی، اور جو اللہ کی مخالفت کرے تو بے شک اللہ سخت سزا دینے والا ہے۔
مَا قَطَعْتُم مِّن لِّينَةٍ أَوْ تَرَكْتُمُوهَا قَآئِمَةً عَلَىٰٓ أُصُولِهَا فَبِإِذْنِ ٱللَّهِ وَلِيُخْزِىَ ٱلْفَـٰسِقِينَ ﴿5﴾
تم نے جو بھی کھجور کے درخت کاٹا یا اسے جڑوں پر قائم چھوڑا، تو وہ اللہ کے حکم سے تھا، اور تاکہ وہ نافرمانوں کو رسوا کرے۔
وَمَآ أَفَآءَ ٱللَّهُ عَلَىٰ رَسُولِهِۦ مِنْهُمْ فَمَآ أَوْجَفْتُمْ عَلَيْهِ مِنْ خَيْلٍۢ وَلَا رِكَابٍ وَلَـٰكِنَّ ٱللَّهَ يُسَلِّطُ رُسُلَهُۥ عَلَىٰ مَن يَشَآءُ ۚ وَٱللَّهُ عَلَىٰ كُلِّ شَىْءٍۢ قَدِيرٌ ﴿6﴾
جو مال اللہ نے ان سے اپنے رسول کو دلایا، اس پر تم نے نہ گھوڑے دوڑائے نہ اونٹ، لیکن اللہ اپنے رسولوں کو جس پر چاہے غالب کر دیتا ہے، اور اللہ ہر چیز پر قدرت رکھتا ہے۔
مَّآ أَفَآءَ ٱللَّهُ عَلَىٰ رَسُولِهِۦ مِنْ أَهْلِ ٱلْقُرَىٰ فَلِلَّهِ وَلِلرَّسُولِ وَلِذِى ٱلْقُرْبَىٰ وَٱلْيَتَـٰمَىٰ وَٱلْمَسَـٰكِينِ وَٱبْنِ ٱلسَّبِيلِ كَىۡ لَا يَكُونَ دُولَةًۢ بَيۡنَ ٱلۡأَغۡنِيَآءِ مِنكُمۡ ۚ وَمَآ ءَاتَىٰكُمُ ٱلرَّسُولُ فَخُذُوهُ ۖ وَمَا نَهَىٰكُمۡ عَنۡهُ فَٱنتَهُوا۟ ۚ وَٱتَّقُوا۟ ٱللَّهَ ۖ إِنَّ ٱللَّهَ شَدِيدُ ٱلۡعِقَابِ ﴿7﴾
جو مال اللہ نے اپنے رسول کو ان بستیوں کے لوگوں سے دلایا، وہ اللہ، رسول، قرابت داروں، یتیموں، مسکینوں اور مسافروں کے لیے ہے — تاکہ وہ مال تمہارے مالداروں کے درمیان گردش نہ کرتا رہے۔ اور جو کچھ رسول تمہیں دے، وہ لے لو، اور جس چیز سے منع کرے، اس سے رک جاؤ۔ اور اللہ سے ڈرو، بے شک اللہ سخت سزا دینے والا ہے۔
لِلۡفُقَرَآءِ ٱلۡمُهَـٰجِرِينَ ٱلَّذِينَ أُخۡرِجُوا۟ مِن دِيَـٰرِهِمۡ وَأَمۡوَٰلِهِمۡ يَبۡتَغُونَ فَضۡلٗا مِّنَ ٱللَّهِ وَرِضۡوَٰنٗا وَيَنصُرُونَ ٱللَّهَ وَرَسُولَهُۥٓ ۚ أُو۟لَـٰٓئِكَ هُمُ ٱلصَّـٰدِقُونَ ﴿8﴾
وہ مال ان مہاجر فقراء کے لیے ہے جو اپنے گھروں اور مالوں سے نکالے گئے، وہ اللہ کا فضل اور رضا چاہتے ہیں، اور اللہ اور اس کے رسول کی مدد کرتے ہیں — یہی لوگ سچے ہیں۔
وَٱلَّذِينَ تَبَوَّءُوا۟ ٱلدَّارَ وَٱلْإِيمَـٰنَ مِن قَبْلِهِمْ يُحِبُّونَ مَنْ هَاجَرَ إِلَيْهِمْ وَلَا يَجِدُونَ فِى صُدُورِهِمْ حَاجَةًۭ مِّمَّآ أُوتُوا۟ وَيُؤْثِرُونَ عَلَىٰٓ أَنفُسِهِمْ وَلَوْ كَانَ بِهِمْ خَصَاصَةٌۭ ۚ وَمَن يُوقَ شُحَّ نَفْسِهِۦ فَأُو۟لَـٰٓئِكَ هُمُ ٱلْمُفْلِحُونَ ﴿9﴾
اور وہ لوگ جنہوں نے ان سے پہلے دارِ ہجرت اور ایمان کو ٹھکانہ بنایا، وہ ان لوگوں سے محبت کرتے ہیں جو ان کے پاس ہجرت کر کے آئے، اور جو کچھ انہیں دیا گیا اس پر اپنے دلوں میں کوئی خلش نہیں پاتے، اور وہ خود پر دوسروں کو ترجیح دیتے ہیں، اگرچہ انہیں خود سخت حاجت ہو۔ اور جو اپنے نفس کے بخل سے بچا لیا گیا، وہی کامیاب ہیں۔
وَٱلَّذِينَ جَآءُوا۟ مِنۢ بَعْدِهِمْ يَقُولُونَ رَبَّنَا ٱغْفِرْ لَنَا وَلِإِخْوَٰنِنَا ٱلَّذِينَ سَبَقُونَا بِٱلْإِيمَـٰنِ وَلَا تَجْعَلْ فِى قُلُوبِنَا غِلًّۭا لِّلَّذِينَ ءَامَنُوا۟ ۚ رَبَّنَآ إِنَّكَ رَءُوفٌۭ رَّحِيمٌۭ ﴿10﴾
اور وہ لوگ جو ان کے بعد آئے، کہتے ہیں: اے ہمارے رب! ہمیں اور ہمارے ان بھائیوں کو بخش دے جو ہم سے پہلے ایمان لائے، اور ہمارے دلوں میں ایمان والوں کے لیے کوئی کینہ نہ رکھ۔ اے ہمارے رب! بے شک تو بہت مہربان، رحم کرنے والا ہے۔
أَلَمْ تَرَ إِلَى ٱلَّذِينَ نَافَقُوا۟ يَقُولُونَ لِإِخْوَٰنِهِمُ ٱلَّذِينَ كَفَرُوا۟ مِنْ أَهْلِ ٱلْكِتَـٰبِ لَئِنْ أُخْرِجْتُمْ لَنَخْرُجَنَّ مَعَكُمْ وَلَا نُطِيعُ فِيكُمْ أَحَدًا أَبَدًۭا وَإِن قُوتِلْتُمْ لَنَنصُرَنَّكُمْ ۚ وَٱللَّهُ يَشْهَدُ إِنَّهُمْ لَكَـٰذِبُونَ ﴿11﴾
کیا آپ نے ان منافقوں کو نہیں دیکھا جو اپنے اہلِ کتاب کافر بھائیوں سے کہتے ہیں: اگر تم نکالے گئے تو ہم ضرور تمہارے ساتھ نکلیں گے، اور تمہارے معاملے میں ہم کبھی کسی کی بات نہیں مانیں گے، اور اگر تم سے جنگ کی گئی تو ہم ضرور تمہاری مدد کریں گے۔ اور اللہ گواہی دیتا ہے کہ وہ یقیناً جھوٹے ہیں۔
لَئِنْ أُخْرِجُوا۟ لَا يَخْرُجُونَ مَعَهُمْ وَلَئِن قُوتِلُوا۟ لَا يَنصُرُونَهُمْ وَلَئِن نَّصَرُوهُمْ لَيُوَلُّنَّ ٱلْأَدْبَـٰرَ ثُمَّ لَا يُنصَرُونَ ﴿12﴾
اگر وہ نکالے گئے تو یہ ان کے ساتھ نہیں نکلیں گے، اور اگر ان سے جنگ کی گئی تو یہ ان کی مدد نہیں کریں گے، اور اگر مدد کریں بھی تو ضرور پیٹھ پھیر کر بھاگ جائیں گے، پھر انہیں مدد نہیں دی جائے گی۔
لَأَنتُمْ أَشَدُّ رَهْبَةًۭ فِى صُدُورِهِم مِّنَ ٱللَّهِ ۚ ذَٰلِكَ بِأَنَّهُمْ قَوْمٌۭ لَّا يَفْقَهُونَ ﴿13﴾
تم ان کے دلوں میں اللہ سے زیادہ خوفناک ہو، یہ اس لیے کہ وہ سمجھ نہیں رکھتے۔
لَا يُقَـٰتِلُونَكُمْ جَمِيعًا إِلَّا فِى قُرًۭى مُّحَصَّنَةٍ أَوْ مِن وَرَآءِ جُدُرٍۢ ۚ بَأْسُهُم بَيْنَهُمْ شَدِيدٌۭ ۚ تَحْسَبُهُمْ جَمِيعًۭا وَقُلُوبُهُمْ شَتَّىٰ ۚ ذَٰلِكَ بِأَنَّهُمْ قَوْمٌۭ لَّا يَعْقِلُونَ ﴿14﴾
یہ سب مل کر تم سے لڑائی نہیں کریں گے مگر قلعہ بند بستیوں میں یا دیواروں کے پیچھے سے۔ ان کے آپس کے جھگڑے سخت ہیں۔ تم انہیں اکٹھا سمجھتے ہو، حالانکہ ان کے دل جدا جدا ہیں۔ یہ اس لیے کہ وہ عقل نہیں رکھتے۔
كَمَثَلِ ٱلَّذِينَ مِن قَبْلِهِمْ قَرِيبًۭا ذَاقُوا۟ وَبَالَ أَمْرِهِمْ وَلَهُمْ عَذَابٌ أَلِيمٌۭ ﴿15﴾
ان کی مثال ان لوگوں کی سی ہے جو ان سے تھوڑے ہی عرصے پہلے تھے، انہوں نے اپنے کام کا وبال چکھا اور ان کے لیے دردناک عذاب ہے۔
كَمَثَلِ ٱلشَّيْطَـٰنِ إِذْ قَالَ لِلْإِنسَـٰنِ ٱكْفُرْ فَلَمَّا كَفَرَ قَالَ إِنِّى بَرِىٓءٌۭ مِّنكَ إِنِّىٓ أَخَافُ ٱللَّهَ رَبَّ ٱلْعَـٰلَمِينَ ﴿16﴾
ان کی مثال شیطان کی سی ہے، جب اس نے انسان سے کہا: کفر کر، پھر جب اس نے کفر کیا تو کہا: میں تجھ سے بیزار ہوں، میں تو رب العالمین سے ڈرتا ہوں۔
فَكَانَ عَـٰقِبَتَهُمَآ أَنَّهُمَا فِى ٱلنَّارِ خَـٰلِدَيْنِ فِيهَا ۚ وَذَٰلِكَ جَزَٰٓؤُا۟ ٱلظَّـٰلِمِينَ ﴿17﴾
تو ان دونوں کا انجام یہ ہوا کہ وہ آگ میں ہمیشہ رہیں گے، اور یہی ظالموں کی جزا ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ ٱتَّقُوا۟ ٱللَّهَ وَلْتَنظُرْ نَفْسٌۭ مَّا قَدَّمَتْ لِغَدٍۢ ۖ وَٱتَّقُوا۟ ٱللَّهَ ۚ إِنَّ ٱللَّهَ خَبِيرٌۭ بِمَا تَعْمَلُونَ ﴿18﴾
اے ایمان والو! اللہ سے ڈرو، اور ہر شخص دیکھے کہ اس نے کل کے لیے کیا آگے بھیجا ہے، اور اللہ سے ڈرو، بے شک اللہ تمہارے اعمال سے خوب باخبر ہے۔
وَلَا تَكُونُوا۟ كَٱلَّذِينَ نَسُوا۟ ٱللَّهَ فَأَنسَىٰهُمْ أَنفُسَهُمْ ۚ أُو۟لَـٰٓئِكَ هُمُ ٱلْفَـٰسِقُونَ ﴿19﴾
اور ان لوگوں کی طرح نہ ہو جاؤ جنہوں نے اللہ کو بھلا دیا، تو اللہ نے انہیں خود ان کا نفس بھلا دیا۔ یہی لوگ فاسق ہیں۔
لَا يَسْتَوِىٓ أَصْحَـٰبُ ٱلنَّارِ وَأَصْحَـٰبُ ٱلْجَنَّةِ ۚ أَصْحَـٰبُ ٱلْجَنَّةِ هُمُ ٱلْفَآئِزُونَ ﴿20﴾
جہنمی اور جنتی برابر نہیں ہو سکتے، جنتی ہی کامیاب ہیں۔
لَوْ أَنزَلْنَا هَـٰذَا ٱلْقُرْءَانَ عَلَىٰ جَبَلٍۢ لَّرَأَيْتَهُۥ خَـٰشِعًۭا مُّتَصَدِّعًۭا مِّنْ خَشْيَةِ ٱللَّهِ ۚ وَتِلْكَ ٱلْأَمْثَـٰلُ نَضْرِبُهَا لِلنَّاسِ لَعَلَّهُمْ يَتَفَكَّرُونَ ﴿21﴾
اگر ہم یہ قرآن کسی پہاڑ پر نازل کرتے تو تم اسے اللہ کے خوف سے جھکا ہوا اور پھٹا ہوا دیکھتے۔ ہم یہ مثالیں لوگوں کے لیے بیان کرتے ہیں تاکہ وہ غور کریں۔
هُوَ ٱللَّهُ ٱلَّذِى لَآ إِلَـٰهَ إِلَّا هُوَ ۖ عَـٰلِمُ ٱلْغَيْبِ وَٱلشَّهَـٰدَةِ ۖ هُوَ ٱلرَّحْمَـٰنُ ٱلرَّحِيمُ ﴿22﴾
وہی اللہ ہے جس کے سوا کوئی معبود نہیں، غیب اور ظاہر کا جاننے والا، وہی رحمٰن، رحیم ہے۔
هُوَ ٱللَّهُ ٱلَّذِى لَآ إِلَـٰهَ إِلَّا هُوَ ٱلْمَلِكُ ٱلْقُدُّوسُ ٱلسَّلَـٰمُ ٱلْمُؤْمِنُ ٱلْمُهَيْمِنُ ٱلْعَزِيزُ ٱلْجَبَّارُ ٱلْمُتَكَبِّرُ ۚ سُبْحَـٰنَ ٱللَّهِ عَمَّا يُشْرِكُونَ ﴿23﴾
وہی اللہ ہے جس کے سوا کوئی معبود نہیں، بادشاہ، نہایت پاک، سلامتی دینے والا، امان دینے والا، نگہبان، زبردست، جبر کرنے والا، بڑائی والا۔ اللہ پاک ہے اس سے جو وہ شریک ٹھہراتے ہیں۔
هُوَ ٱللَّهُ ٱلْخَـٰلِقُ ٱلْبَارِئُ ٱلْمُصَوِّرُ ۖ لَهُ ٱلْأَسْمَآءُ ٱلْحُسْنَىٰ ۚ يُسَبِّحُ لَهُۥ مَا فِى ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ ۚ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿24﴾
وہی اللہ ہے، پیدا کرنے والا، وجود بخشنے والا، صورت بنانے والا۔ اسی کے لیے سب اچھے نام ہیں۔ اس کی تسبیح کرتی ہیں وہ سب چیزیں جو آسمانوں اور زمین میں ہیں، اور وہی زبردست، حکمت والا ہے۔
سورة الممتحنة
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تَتَّخِذُوا۟ عَدُوِّى وَعَدُوَّكُمْ أَوْلِيَآءَ تُلْقُونَ إِلَيْهِم بِٱلْمَوَدَّةِ وَقَدْ كَفَرُوا۟ بِمَا جَآءَكُم مِّنَ ٱلْحَقِّ يُخْرِجُونَ ٱلرَّسُولَ وَإِيَّاكُمْ أَن تُؤْمِنُوا۟ بِٱللَّهِ رَبِّكُمْ ۚ إِن كُنتُمْ خَرَجْتُمْ جِهَـٰدًۭا فِى سَبِيلِى وَٱبْتِغَآءَ مَرْضَاتِى تُسِرُّونَ إِلَيْهِم بِٱلْمَوَدَّةِ وَأَنَا۠ أَعْلَمُ بِمَآ أَخْفَيْتُمْ وَمَآ أَعْلَنتُمْ ۚ وَمَن يَفْعَلْهُ مِنكُمْ فَقَدْ ضَلَّ سَوَآءَ ٱلسَّبِيلِ ﴿1﴾
اے ایمان والو! میرے اور اپنے دشمنوں کو دوست نہ بناؤ، تم ان سے محبت کرتے ہو حالانکہ وہ اس حق کا انکار کر چکے ہیں جو تمہارے پاس آیا ہے۔ وہ رسول کو اور تمہیں نکال دیتے ہیں صرف اس وجہ سے کہ تم اپنے رب اللہ پر ایمان لائے ہو۔ اگر تم میری راہ میں جہاد کرنے اور میری رضا حاصل کرنے کے لیے نکلے ہو تو ان سے خفیہ محبت نہ رکھو۔ میں خوب جانتا ہوں جو تم چھپاتے ہو اور جو ظاہر کرتے ہو۔ اور جو تم میں سے ایسا کرے وہ سیدھے راستے سے بھٹک گیا۔
إِن يَثْقَفُوكُمْ يَكُونُوا۟ لَكُمْ أَعْدَآءًۭ وَيَبْسُطُوٓا۟ إِلَيْكُمْ أَيْدِيَهُمْ وَأَلْسِنَتَهُم بِٱلسُّوٓءِ وَوَدُّوا۟ لَوْ تَكْفُرُونَ ﴿2﴾
اگر وہ تم پر قابو پا جائیں تو تمہارے دشمن بن جائیں گے، اور تمہیں برائی سے ہاتھ اور زبان سے اذیت دیں گے، اور وہ چاہتے ہیں کہ تم کفر کرو۔
لَن تَنفَعَكُمْ أَرْحَامُكُمْ وَلَآ أَوْلَـٰدُكُمْ ۚ يَوْمَ ٱلْقِيَـٰمَةِ يَفْصِلُ بَيْنَكُمْ ۚ وَٱللَّهُ بِمَا تَعْمَلُونَ بَصِيرٌ ﴿3﴾
قیامت کے دن تمہارے رشتہ دار اور تمہاری اولاد تمہیں کچھ فائدہ نہ دیں گے، اللہ تمہارے درمیان فیصلہ کرے گا، اور اللہ تمہارے اعمال کو خوب دیکھنے والا ہے۔
قَدْ كَانَتْ لَكُمْ أُسْوَةٌۭ حَسَنَةٌۭ فِىٓ إِبْرَٰهِيمَ وَٱلَّذِينَ مَعَهُۥٓ إِذْ قَالُوا۟ لِقَوْمِهِمْ إِنَّا بُرَءَٰٓؤُا۟ مِنكُمْ وَمِمَّا تَعْبُدُونَ مِن دُونِ ٱللَّهِ ۖ كَفَرْنَا بِكُمْ وَبَدَا بَيْنَنَا وَبَيْنَكُمُ ٱلْعَدَٰوَةُ وَٱلْبَغْضَآءُ أَبَدًۭا حَتَّىٰ تُؤْمِنُوا۟ بِٱللَّهِ وَحْدَهُۥٓ ۤ إِلَّا قَوْلَ إِبْرَٰهِيمَ لِأَبِيهِ لَأَسْتَغْفِرَنَّ لَكَ وَمَآ أَمْلِكُ لَكَ مِنَ ٱللَّهِ مِن شَىْءٍۢ ۚ رَبَّنَا عَلَيْكَ تَوَكَّلْنَا وَإِلَيْكَ أَنَبْنَا وَإِلَيْكَ ٱلْمَصِيرُ ﴿4﴾
تمہارے لیے ابراہیم اور ان کے ساتھیوں میں بہترین نمونہ ہے، جب انہوں نے اپنی قوم سے کہا: ہم تم سے اور ان چیزوں سے جنہیں تم اللہ کے سوا پوجتے ہو، بیزار ہیں۔ ہم نے تم سے کفر کیا، اور ہمارے درمیان اور تمہارے درمیان ہمیشہ کے لیے دشمنی اور نفرت ظاہر ہو گئی، یہاں تک کہ تم صرف اللہ پر ایمان لاؤ۔ سوائے ابراہیم کے اپنے باپ سے کہنے کے: میں ضرور تیرے لیے بخشش مانگوں گا، اور اللہ کے مقابلے میں تیرے لیے کچھ اختیار نہیں رکھتا۔ اے ہمارے رب! ہم نے تجھ پر بھروسہ کیا، اور تیری ہی طرف رجوع کیا، اور تیری ہی طرف لوٹنا ہے۔
رَبَّنَا لَا تَجْعَلْنَا فِتْنَةًۭ لِّلَّذِينَ كَفَرُوا۟ وَٱغْفِرْ لَنَا رَبَّنَآ ۖ إِنَّكَ أَنتَ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿5﴾
اے ہمارے رب! ہمیں کافروں کے لیے آزمائش نہ بنا، اور ہمیں بخش دے، اے ہمارے رب! بے شک تو زبردست، حکمت والا ہے۔
لَقَدْ كَانَ لَكُمْ فِيهِمْ أُسْوَةٌۭ حَسَنَةٌۭ لِّمَن كَانَ يَرْجُوا۟ ٱللَّهَ وَٱلْيَوْمَ ٱلْـَٔاخِرَ ۚ وَمَن يَتَوَلَّ فَإِنَّ ٱللَّهَ هُوَ ٱلْغَنِىُّ ٱلْحَمِيدُ ﴿6﴾
یقیناً ان (ابراہیم اور ان کے ساتھیوں) میں تمہارے لیے بہترین نمونہ ہے، ہر اس شخص کے لیے جو اللہ اور آخرت کے دن کی امید رکھتا ہے۔ اور جو منہ موڑے تو بے شک اللہ بے نیاز، تعریف کے لائق ہے۔
عَسَى ٱللَّهُ أَن يَجْعَلَ بَيْنَكُمْ وَبَيْنَ ٱلَّذِينَ عَادَيْتُم مِّنْهُم مَّوَدَّةًۭ ۚ وَٱللَّهُ قَدِيرٌۭ ۚ وَٱللَّهُ غَفُورٌۭ رَّحِيمٌۭ ﴿7﴾
امید ہے کہ اللہ تمہارے اور ان لوگوں کے درمیان جن سے تم دشمنی رکھتے ہو، محبت پیدا کر دے گا۔ اور اللہ ہر چیز پر قدرت رکھتا ہے، اور اللہ بخشنے والا، رحم کرنے والا ہے۔
لَا يَنْهَىٰكُمُ ٱللَّهُ عَنِ ٱلَّذِينَ لَمْ يُقَـٰتِلُوكُمْ فِى ٱلدِّينِ وَلَمْ يُخْرِجُوكُم مِّن دِيَـٰرِكُمْ أَن تَبَرُّوهُمْ وَتُقْسِطُوٓا۟ إِلَيْهِمْ ۚ إِنَّ ٱللَّهَ يُحِبُّ ٱلْمُقْسِطِينَ ﴿8﴾
اللہ تمہیں ان لوگوں سے نیکی کرنے اور انصاف سے پیش آنے سے نہیں روکتا جو تم سے دین کے معاملے میں نہیں لڑے اور تمہیں تمہارے گھروں سے نہیں نکالا۔ بے شک اللہ انصاف کرنے والوں کو پسند کرتا ہے۔
إِنَّمَا يَنْهَىٰكُمُ ٱللَّهُ عَنِ ٱلَّذِينَ قَـٰتَلُوكُمْ فِى ٱلدِّينِ وَأَخْرَجُوكُم مِّن دِيَـٰرِكُمْ وَظَـٰهَرُوا۟ عَلَىٰٓ إِخْرَاجِكُمْ أَن تَوَلَّوْهُمْ ۚ وَمَن يَتَوَلَّهُمْ فَأُو۟لَـٰٓئِكَ هُمُ ٱلظَّـٰلِمُونَ ﴿9﴾
اللہ تمہیں صرف ان لوگوں سے دوستی رکھنے سے روکتا ہے جنہوں نے تم سے دین کے معاملے میں جنگ کی، تمہیں تمہارے گھروں سے نکالا، اور تمہیں نکالنے میں دوسروں کی مدد کی۔ اور جو ان سے دوستی رکھے گا وہی ظالم ہیں۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ إِذَا جَآءَكُمُ ٱلْمُؤْمِنَـٰتُ مُهَـٰجِرَٰتٍۢ فَٱمْتَحِنُوهُنَّ ۖ ٱللَّهُ أَعْلَمُ بِإِيمَـٰنِهِنَّ ۖ فَإِنْ عَلِمْتُمُوهُنَّ مُؤْمِنَـٰتٍۢ فَلَا تَرْجِعُوهُنَّ إِلَى ٱلْكُفَّارِ ۖ لَا هُنَّ حِلٌّۭ لَّهُمْ وَلَا هُمْ يَحِلُّونَ لَهُنَّ ۖ وَءَاتُوهُم مَّآ أَنفَقُوا۟ ۚ وَلَا جُنَاحَ عَلَيْكُمْ أَن تَنكِحُوهُنَّ إِذَآ ءَاتَيْتُمُوهُنَّ أُجُورَهُنَّ ۚ وَلَا تُمْسِكُوا۟ بِعِصَمِ ٱلْكَوَافِرِ وَسْـَٔلُوا۟ مَآ أَنفَقْتُمْ وَلْيَسْـَٔلُوا۟ مَآ أَنفَقُوا۟ ۚ ذَٰلِكُمْ حُكْمُ ٱللَّهِ ۖ يَحْكُمُ بَيْنَكُمْ ۚ وَٱللَّهُ عَلِيمٌ حَكِيمٌ ﴿10﴾
اے ایمان والو! جب تمہارے پاس مومن عورتیں ہجرت کر کے آئیں تو ان کا امتحان لو، اللہ ان کے ایمان کو خوب جانتا ہے۔ پھر اگر تم انہیں مومن سمجھو تو انہیں کافروں کی طرف واپس نہ کرو، نہ وہ ان کے لیے حلال ہیں اور نہ وہ ان کے لیے۔ اور جو کچھ (کافر شوہروں نے) ان پر خرچ کیا ہو وہ انہیں واپس دو۔ اور تم پر کوئی گناہ نہیں کہ تم ان عورتوں سے نکاح کر لو جب تم انہیں ان کا مہر دے دو۔ اور کافر عورتوں کو اپنے نکاح میں نہ رکھو، اور جو کچھ تم نے خرچ کیا ہو وہ واپس مانگو، اور وہ بھی جو انہوں نے خرچ کیا ہو واپس مانگیں۔ یہ اللہ کا حکم ہے، وہ تمہارے درمیان فیصلہ کرتا ہے، اور اللہ سب کچھ جاننے والا، حکمت والا ہے۔
وَإِن جَآءَكُمْ أَزْوَٰجٌۭ مِّنَ ٱلْكُفَّارِ فَعَاقَبْتُمْ فَـَٔاتُوا۟ ٱلَّذِينَ ذَهَبَتْ أَزْوَٰجُهُمْ مِّثْلَ مَآ أَنفَقُوا۟ ۚ وَٱتَّقُوا۟ ٱللَّهَ ٱلَّذِىٓ أَنتُم بِهِۦ مُؤْمِنُونَ ﴿11﴾
اور اگر تمہاری بیویوں میں سے کوئی کافر مردوں کے پاس چلی جائے، پھر تمہیں موقع ملے (بدلہ لینے کا)، تو ان لوگوں کو وہی مال واپس دو جو انہوں نے خرچ کیا تھا جن کی بیویاں چلی گئیں۔ اور اللہ سے ڈرو جس پر تم ایمان رکھتے ہو۔
يَـٰٓأَيُّهَا ٱلنَّبِىُّ إِذَا جَآءَكَ ٱلْمُؤْمِنَـٰتُ يُبَايِعْنَكَ عَلَىٰٓ أَن لَّا يُشْرِكْنَ بِٱللَّهِ شَيْـًۭٔا وَلَا يَسْرِقْنَ وَلَا يَزْنِينَ وَلَا يَقْتُلْنَ أَوْلَـٰدَهُنَّ وَلَا يَأْتِينَ بِبُهْتَـٰنٍۢ يَفْتَرِينَهُۥ بَيْنَ أَيْدِيهِنَّ وَأَرْجُلِهِنَّ وَلَا يَعْصِينَكَ فِى مَعْرُوفٍۢ فَبَايِعْهُنَّ وَٱسْتَغْفِرْ لَهُنَّ ٱللَّهَ ۖ إِنَّ ٱللَّهَ غَفُورٌۭ رَّحِيمٌۭ ﴿12﴾
اے نبی! جب مومن عورتیں آپ کے پاس آئیں اس شرط پر بیعت کرنے کے لیے کہ وہ اللہ کے ساتھ کسی کو شریک نہیں کریں گی، چوری نہیں کریں گی، زنا نہیں کریں گی، اپنی اولاد کو قتل نہیں کریں گی، جھوٹا بہتان نہیں باندھیں گی جو وہ اپنے ہاتھوں اور پاؤں کے درمیان گھڑیں، اور کسی نیک کام میں آپ کی نافرمانی نہیں کریں گی — تو ان سے بیعت لے لیجیے اور ان کے لیے اللہ سے بخشش مانگیے۔ بے شک اللہ بخشنے والا، رحم کرنے والا ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تَتَوَلَّوْا۟ قَوْمًۭا غَضِبَ ٱللَّهُ عَلَيْهِمْ ۙ قَدْ يَئِسُوا۟ مِنَ ٱلْـَٔاخِرَةِ كَمَا يَئِسَ ٱلْكُفَّارُ مِنْ أَصْحَـٰبِ ٱلْقُبُورِ ﴿13﴾
اے ایمان والو! ان لوگوں سے دوستی نہ رکھو جن پر اللہ کا غضب نازل ہوا۔ وہ آخرت سے ایسے مایوس ہیں جیسے کفار قبروں والوں سے مایوس ہیں۔
سورة الصف
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
سَبَّحَ لِلَّهِ مَا فِى ٱلسَّمَـٰوَٰتِ وَمَا فِى ٱلْأَرْضِ ۖ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿1﴾
اللہ کی تسبیح کرتی ہیں وہ سب چیزیں جو آسمانوں اور زمین میں ہیں، اور وہی زبردست، حکمت والا ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لِمَ تَقُولُونَ مَا لَا تَفْعَلُونَ ﴿2﴾
اے ایمان والو! تم وہ بات کیوں کہتے ہو جو کرتے نہیں؟
كَبُرَ مَقْتًۭا عِندَ ٱللَّهِ أَن تَقُولُوا۟ مَا لَا تَفْعَلُونَ ﴿3﴾
اللہ کے نزدیک یہ سخت ناپسندیدہ بات ہے کہ تم وہ کہو جو کرتے نہیں۔
إِنَّ ٱللَّهَ يُحِبُّ ٱلَّذِينَ يُقَـٰتِلُونَ فِى سَبِيلِهِۦ صَفًّۭا كَأَنَّهُم بُنْيَـٰنٌۭ مَّرْصُوصٌۭ ﴿4﴾
بے شک اللہ ان لوگوں سے محبت کرتا ہے جو اس کی راہ میں صف باندھ کر لڑتے ہیں گویا وہ سیسہ پلائی ہوئی دیوار ہیں۔
وَإِذْ قَالَ مُوسَىٰ لِقَوْمِهِۦ يَـٰقَوْمِ لِمَ تُؤْذُونَنِى وَقَد تَّعْلَمُونَ أَنِّى رَسُولُ ٱللَّهِ إِلَيْكُمْ ۖ فَلَمَّا زَاغُوٓا۟ أَزَاغَ ٱللَّهُ قُلُوبَهُمْ ۚ وَٱللَّهُ لَا يَهْدِى ٱلْقَوْمَ ٱلْفَـٰسِقِينَ ﴿5﴾
اور جب موسیٰ نے اپنی قوم سے کہا: اے میری قوم! تم مجھے کیوں اذیت دیتے ہو حالانکہ تم جانتے ہو کہ میں تمہاری طرف اللہ کا رسول ہوں؟ پھر جب وہ ٹیڑھے ہو گئے تو اللہ نے ان کے دلوں کو بھی ٹیڑھا کر دیا، اور اللہ فاسق قوم کو ہدایت نہیں دیتا۔
وَإِذْ قَالَ عِيسَى ٱبْنُ مَرْيَمَ يَـٰبَنِىٓ إِسْرَٰٓءِيلَ إِنِّى رَسُولُ ٱللَّهِ إِلَيْكُم مُّصَدِّقًۭا لِّمَا بَيْنَ يَدَىَّ مِنَ ٱلتَّوْرَىٰةِ وَمُبَشِّرًۭا بِرَسُولٍۢ يَأْتِى مِنۢ بَعْدِى ٱسْمُهُۥٓ أَحْمَدُ ۖ فَلَمَّا جَآءَهُم بِٱلْبَيِّنَـٰتِ قَالُوا۟ هَـٰذَا سِحْرٌۭ مُّبِينٌۭ ﴿6﴾
اور جب عیسیٰ ابن مریم نے کہا: اے بنی اسرائیل! میں تمہاری طرف اللہ کا رسول ہوں، تورات کی تصدیق کرنے والا جو میرے سامنے ہے، اور ایک رسول کی بشارت دینے والا جو میرے بعد آئے گا، اس کا نام احمد ہوگا۔ پھر جب وہ ان کے پاس واضح نشانیاں لے کر آیا تو انہوں نے کہا: یہ تو کھلا جادو ہے۔
وَمَنْ أَظْلَمُ مِمَّنِ ٱفْتَرَىٰ عَلَى ٱللَّهِ ٱلْكَذِبَ وَهُوَ يُدْعَىٰٓ إِلَى ٱلْإِسْلَـٰمِ ۚ وَٱللَّهُ لَا يَهْدِى ٱلْقَوْمَ ٱلظَّـٰلِمِينَ ﴿7﴾
اور اس سے بڑھ کر ظالم کون ہے جو اللہ پر جھوٹ باندھے حالانکہ اسے اسلام کی طرف بلایا جا رہا ہو؟ اور اللہ ظالم قوم کو ہدایت نہیں دیتا۔
يُرِيدُونَ لِيُطْفِـُٔوا۟ نُورَ ٱللَّهِ بِأَفْوَٰهِهِمْ وَٱللَّهُ مُتِمُّ نُورِهِۦ وَلَوْ كَرِهَ ٱلْكَـٰفِرُونَ ﴿8﴾
وہ چاہتے ہیں کہ اللہ کے نور کو اپنے منہ سے بجھا دیں، اور اللہ اپنے نور کو مکمل کر کے رہے گا، چاہے کافر ناپسند کریں۔
هُوَ ٱلَّذِىٓ أَرْسَلَ رَسُولَهُۥ بِٱلْهُدَىٰ وَدِينِ ٱلْحَقِّ لِيُظْهِرَهُۥ عَلَى ٱلدِّينِ كُلِّهِۦ وَلَوْ كَرِهَ ٱلْمُشْرِكُونَ ﴿9﴾
وہی ہے جس نے اپنے رسول کو ہدایت اور دینِ حق کے ساتھ بھیجا تاکہ اسے تمام دینوں پر غالب کرے، چاہے مشرک ناپسند کریں۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ هَلْ أَدُلُّكُمْ عَلَىٰ تِجَـٰرَةٍۢ تُنجِيكُم مِّنْ عَذَابٍ أَلِيمٍۢ ﴿10﴾
اے ایمان والو! کیا میں تمہیں ایسی تجارت بتاؤں جو تمہیں دردناک عذاب سے نجات دے؟
تُؤْمِنُونَ بِٱللَّهِ وَرَسُولِهِۦ وَتُجَـٰهِدُونَ فِى سَبِيلِ ٱللَّهِ بِأَمْوَٰلِكُمْ وَأَنفُسِكُمْ ۚ ذَٰلِكُمْ خَيْرٌۭ لَّكُمْ إِن كُنتُمْ تَعْلَمُونَ ﴿11﴾
اللہ اور اس کے رسول پر ایمان لاؤ، اور اللہ کی راہ میں اپنے مالوں اور جانوں سے جہاد کرو، یہ تمہارے لیے بہتر ہے اگر تم جانتے ہو۔
يَغْفِرْ لَكُمْ ذُنُوبَكُمْ وَيُدْخِلْكُمْ جَنَّـٰتٍۢ تَجْرِى مِن تَحْتِهَا ٱلْأَنْهَـٰرُ وَمَسَـٰكِنَ طَيِّبَةًۭ فِى جَنَّـٰتِ عَدْنٍۢ ۚ ذَٰلِكَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿12﴾
وہ تمہارے گناہ بخش دے گا اور تمہیں ایسی جنتوں میں داخل کرے گا جن کے نیچے نہریں بہتی ہیں، اور ہمیشہ رہنے والی جنتوں میں پاکیزہ گھر دے گا، یہی بڑی کامیابی ہے۔
وَأُخْرَىٰ تُحِبُّونَهَا نَصْرٌۭ مِّنَ ٱللَّهِ وَفَتْحٌۭ قَرِيبٌۭ ۖ وَبَشِّرِ ٱلْمُؤْمِنِينَ ﴿13﴾
اور ایک اور چیز جو تم پسند کرتے ہو: اللہ کی طرف سے مدد اور قریب فتح، اور ایمان والوں کو خوشخبری دے دو۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ كُونُوٓا۟ أَنصَارَ ٱللَّهِ كَمَا قَالَ عِيسَى ٱبْنُ مَرْيَمَ لِلْحَوَارِيِّـۧنَ مَنْ أَنصَارِىٓ إِلَى ٱللَّهِ ۖ قَالَ ٱلْحَوَارِيُّونَ نَحْنُ أَنصَارُ ٱللَّهِ ۖ فَـَٔامَنَت طَّآئِفَةٌۭ مِّنۢ بَنِىٓ إِسْرَٰٓءِيلَ وَكَفَرَت طَّآئِفَةٌۭ ۖ فَأَيَّدْنَا ٱلَّذِينَ ءَامَنُوا۟ عَلَىٰ عَدُوِّهِمْ فَأَصْبَحُوا۟ ظَـٰهِرِينَ ﴿14﴾
اے ایمان والو! اللہ کے مددگار بنو، جیسے عیسیٰ ابن مریم نے حواریوں سے کہا: کون ہے جو اللہ کی طرف میرا مددگار ہو؟ حواریوں نے کہا: ہم اللہ کے مددگار ہیں۔ پھر بنی اسرائیل میں ایک گروہ ایمان لایا اور ایک گروہ نے کفر کیا، تو ہم نے ایمان والوں کو ان کے دشمنوں پر مدد دی، تو وہ غالب آ گئے۔
سورة الجمعة
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
يُسَبِّحُ لِلَّهِ مَا فِى ٱلسَّمَـٰوَٰتِ وَمَا فِى ٱلْأَرْضِ ٱلْمَلِكِ ٱلْقُدُّوسِ ٱلْعَزِيزِ ٱلْحَكِيمِ ﴿1﴾
اللہ کی تسبیح کرتی ہیں وہ سب چیزیں جو آسمانوں اور زمین میں ہیں، بادشاہ، پاک، زبردست، حکمت والے کی۔
هُوَ ٱلَّذِى بَعَثَ فِى ٱلْأُمِّيِّـۧنَ رَسُولًۭا مِّنْهُمْ يَتْلُوا۟ عَلَيْهِمْ ءَايَـٰتِهِۦ وَيُزَكِّيهِمْ وَيُعَلِّمُهُمُ ٱلْكِتَـٰبَ وَٱلْحِكْمَةَ ۖ وَإِن كَانُوا۟ مِن قَبْلُ لَفِى ضَلَـٰلٍۢ مُّبِينٍ ﴿2﴾
وہی ہے جس نے ان پڑھ لوگوں میں ایک رسول بھیجا انہی میں سے، جو ان پر اس کی آیات پڑھتا ہے، انہیں پاک کرتا ہے، اور انہیں کتاب و حکمت سکھاتا ہے، حالانکہ وہ اس سے پہلے کھلی گمراہی میں تھے۔
وَءَاخَرِينَ مِنْهُمْ لَمَّا يَلْحَقُوا۟ بِهِمْ ۚ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿3﴾
اور ان کے علاوہ کچھ اور لوگوں میں بھی (یہ رسول بھیجا گیا ہے) جو ابھی ان سے نہیں ملے، اور وہ زبردست، حکمت والا ہے۔
ذَٰلِكَ فَضْلُ ٱللَّهِ يُؤْتِيهِ مَن يَشَآءُ ۚ وَٱللَّهُ ذُو ٱلْفَضْلِ ٱلْعَظِيمِ ﴿4﴾
یہ اللہ کا فضل ہے، وہ جسے چاہے عطا کرتا ہے، اور اللہ بڑے فضل والا ہے۔
مَثَلُ ٱلَّذِينَ حُمِّلُوا۟ ٱلتَّوْرَىٰةَ ثُمَّ لَمْ يَحْمِلُوهَا كَمَثَلِ ٱلْحِمَارِ يَحْمِلُ أَسْفَارًۭا ۚ بِئْسَ مَثَلُ ٱلْقَوْمِ ٱلَّذِينَ كَذَّبُوا۟ بِـَٔايَـٰتِ ٱللَّهِ ۚ وَٱللَّهُ لَا يَهْدِى ٱلْقَوْمَ ٱلظَّـٰلِمِينَ ﴿5﴾
ان لوگوں کی مثال جن پر تورات کا بوجھ ڈالا گیا پھر انہوں نے اسے نہ اٹھایا، گدھے کی مثال ہے جو کتابیں اٹھائے ہوئے ہو۔ بہت بری مثال ہے ان لوگوں کی جنہوں نے اللہ کی آیات کو جھٹلایا، اور اللہ ظالم قوم کو ہدایت نہیں دیتا۔
قُلْ يَـٰٓأَيُّهَا ٱلَّذِينَ هَادُوا۟ إِن زَعَمْتُمْ أَنَّكُمْ أَوْلِيَآءُ لِلَّهِ مِن دُونِ ٱلنَّاسِ فَتَمَنَّوُا۟ ٱلْمَوْتَ إِن كُنتُمْ صَـٰدِقِينَ ﴿6﴾
کہہ دو: اے یہودیو! اگر تم دعویٰ کرتے ہو کہ تم ہی اللہ کے دوست ہو لوگوں کے سوا، تو موت کی تمنا کرو اگر تم سچے ہو۔
وَلَا يَتَمَنَّوْنَهُۥٓ أَبَدًۭا بِمَا قَدَّمَتْ أَيْدِيهِمْ ۚ وَٱللَّهُ عَلِيمٌۢ بِٱلظَّـٰلِمِينَ ﴿7﴾
اور وہ کبھی موت کی تمنا نہیں کریں گے، اس وجہ سے جو ان کے ہاتھوں نے آگے بھیجا ہے، اور اللہ ظالموں کو خوب جانتا ہے۔
قُلْ إِنَّ ٱلْمَوْتَ ٱلَّذِى تَفِرُّونَ مِنْهُ فَإِنَّهُۥ مُلَـٰقِيكُمْ ۖ ثُمَّ تُرَدُّونَ إِلَىٰ عَـٰلِمِ ٱلْغَيْبِ وَٱلشَّهَـٰدَةِ فَيُنَبِّئُكُم بِمَا كُنتُمْ تَعْمَلُونَ ﴿8﴾
کہہ دو: وہ موت جس سے تم بھاگتے ہو، وہ تمہیں ضرور آ کر ملے گی، پھر تم لوٹائے جاؤ گے اس کی طرف جو غیب اور ظاہر کا جاننے والا ہے، پھر وہ تمہیں بتائے گا جو کچھ تم کرتے رہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ إِذَا نُودِىَ لِلصَّلَوٰةِ مِن يَوْمِ ٱلْجُمُعَةِ فَٱسْعَوْا۟ إِلَىٰ ذِكْرِ ٱللَّهِ وَذَرُوا۟ ٱلْبَيْعَ ۚ ذَٰلِكُمْ خَيْرٌۭ لَّكُمْ إِن كُنتُمْ تَعْلَمُونَ ﴿9﴾
اے ایمان والو! جب جمعہ کے دن نماز کے لیے اذان دی جائے تو اللہ کے ذکر کی طرف دوڑو اور خرید و فروخت چھوڑ دو، یہ تمہارے لیے بہتر ہے اگر تم جانتے ہو۔
فَإِذَا قُضِيَتِ ٱلصَّلَوٰةُ فَٱنتَشِرُوا۟ فِى ٱلْأَرْضِ وَٱبْتَغُوا۟ مِن فَضْلِ ٱللَّهِ وَٱذْكُرُوا۟ ٱللَّهَ كَثِيرًۭا لَّعَلَّكُمْ تُفْلِحُونَ ﴿10﴾
پھر جب نماز پوری ہو جائے تو زمین میں پھیل جاؤ اور اللہ کا فضل تلاش کرو، اور اللہ کو بہت یاد کرو تاکہ تم کامیاب ہو جاؤ۔
وَإِذَا رَأَوْا۟ تِجَـٰرَةً أَوْ لَهْوًا ٱنفَضُّوٓا۟ إِلَيْهَا وَتَرَكُوكَ قَآئِمًۭا ۚ قُلْ مَا عِندَ ٱللَّهِ خَيْرٌۭ مِّنَ ٱللَّهْوِ وَمِنَ ٱلتِّجَـٰرَةِ ۚ وَٱللَّهُ خَيْرُ ٱلرَّٰزِقِينَ ﴿11﴾
اور جب انہوں نے تجارت یا کھیل تماشا دیکھا تو اس کی طرف دوڑ گئے اور آپ کو کھڑا چھوڑ دیا۔ کہہ دو: جو اللہ کے پاس ہے وہ کھیل اور تجارت سے بہتر ہے، اور اللہ سب سے بہتر رزق دینے والا ہے۔
سورة المنافقون
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
إِذَا جَآءَكَ ٱلْمُنَـٰفِقُونَ قَالُوا۟ نَشْهَدُ إِنَّكَ لَرَسُولُ ٱللَّهِ ۗ وَٱللَّهُ يَعْلَمُ إِنَّكَ لَرَسُولُهُۥ ۙ وَٱللَّهُ يَشْهَدُ إِنَّ ٱلْمُنَـٰفِقِينَ لَكَـٰذِبُونَ ﴿1﴾
جب منافق آپ کے پاس آتے ہیں تو کہتے ہیں: ہم گواہی دیتے ہیں کہ آپ اللہ کے رسول ہیں۔ اور اللہ جانتا ہے کہ آپ اس کے رسول ہیں، اور اللہ گواہی دیتا ہے کہ منافق جھوٹے ہیں۔
ٱتَّخَذُوٓا۟ أَيْمَـٰنَهُمْ جُنَّةًۭ فَصَدُّوا۟ عَن سَبِيلِ ٱللَّهِ ۚ إِنَّهُمْ سَآءَ مَا كَانُوا۟ يَعْمَلُونَ ﴿2﴾
انہوں نے اپنی قسموں کو ڈھال بنا لیا اور اللہ کے راستے سے روکنے لگے، بے شک وہ بہت برا کام کرتے ہیں۔
ذَٰلِكَ بِأَنَّهُمْ ءَامَنُوا۟ ثُمَّ كَفَرُوا۟ فَطُبِعَ عَلَىٰ قُلُوبِهِمْ فَهُمْ لَا يَفْقَهُونَ ﴿3﴾
یہ اس لیے کہ وہ ایمان لائے پھر کفر کیا، تو ان کے دلوں پر مہر لگا دی گئی، پس وہ سمجھتے نہیں۔
وَإِذَا رَأَيْتَهُمْ تُعْجِبُكَ أَجْسَـٰمُهُمْ ۖ وَإِن يَقُولُوا۟ تَسْمَعْ لِقَوْلِهِمْ ۖ كَأَنَّهُمْ خُشُبٌۭ مُّسَنَّدَةٌۭ ۖ يَحْسَبُونَ كُلَّ صَيْحَةٍ عَلَيْهِمْ ۚ هُمُ ٱلْعَدُوُّ فَٱحْذَرْهُمْ ۚ قَـٰتَلَهُمُ ٱللَّهُ ۖ أَنَّىٰ يُؤْفَكُونَ ﴿4﴾
اور جب آپ انہیں دیکھیں تو ان کے جسم آپ کو خوشنما لگیں، اور اگر وہ بات کریں تو آپ ان کی بات سنیں، گویا وہ دیوار سے ٹیک لگائے ہوئے لکڑیاں ہیں۔ ہر آواز کو اپنے خلاف سمجھتے ہیں۔ وہی دشمن ہیں، پس ان سے بچو۔ اللہ انہیں ہلاک کرے، کہاں بہکائے جا رہے ہیں۔
وَإِذَا قِيلَ لَهُمْ تَعَالَوْا۟ يَسْتَغْفِرْ لَكُمْ رَسُولُ ٱللَّهِ لَوَّوْا۟ رُءُوسَهُمْ وَرَأَيْتَهُمْ يَصُدُّونَ وَهُم مُّسْتَكْبِرُونَ ﴿5﴾
اور جب ان سے کہا جائے کہ آؤ رسول اللہ تمہارے لیے استغفار کریں، تو وہ اپنے سر موڑ لیتے ہیں، اور آپ دیکھتے ہیں کہ وہ روگردانی کرتے ہیں اور تکبر کرتے ہیں۔
سَوَآءٌ عَلَيْهِمْ أَسْتَغْفَرْتَ لَهُمْ أَمْ لَمْ تَسْتَغْفِرْ لَهُمْ لَن يَغْفِرَ ٱللَّهُ لَهُمْ ۚ إِنَّ ٱللَّهَ لَا يَهْدِى ٱلْقَوْمَ ٱلْفَـٰسِقِينَ ﴿6﴾
ان کے لیے برابر ہے کہ آپ ان کے لیے استغفار کریں یا نہ کریں، اللہ انہیں ہرگز نہیں بخشے گا۔ بے شک اللہ فاسق قوم کو ہدایت نہیں دیتا۔
هُمُ ٱلَّذِينَ يَقُولُونَ لَا تُنفِقُوا۟ عَلَىٰ مَنْ عِندَ رَسُولِ ٱللَّهِ حَتَّىٰ يَنفَضُّوا۟ ۗ وَلِلَّهِ خَزَآئِنُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَلَـٰكِنَّ ٱلْمُنَـٰفِقِينَ لَا يَفْقَهُونَ ﴿7﴾
یہ وہ لوگ ہیں جو کہتے ہیں: رسول اللہ کے پاس جو لوگ ہیں ان پر خرچ نہ کرو تاکہ وہ منتشر ہو جائیں۔ حالانکہ آسمانوں اور زمین کے خزانے اللہ ہی کے ہیں، لیکن منافق سمجھتے نہیں۔
يَقُولُونَ لَئِن رَّجَعْنَآ إِلَى ٱلْمَدِينَةِ لَيُخْرِجَنَّ ٱلْأَعَزُّ مِنْهَا ٱلْأَذَلَّ ۚ وَلِلَّهِ ٱلْعِزَّةُ وَلِرَسُولِهِۦ وَلِلْمُؤْمِنِينَ وَلَـٰكِنَّ ٱلْمُنَـٰفِقِينَ لَا يَعْلَمُونَ ﴿8﴾
وہ کہتے ہیں: اگر ہم مدینہ واپس گئے تو عزت والا ذلت والے کو وہاں سے نکال دے گا۔ حالانکہ عزت تو اللہ، اس کے رسول اور مومنوں کے لیے ہے، لیکن منافق نہیں جانتے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تُلْهِكُمْ أَمْوَٰلُكُمْ وَلَآ أَوْلَـٰدُكُمْ عَن ذِكْرِ ٱللَّهِ ۚ وَمَن يَفْعَلْ ذَٰلِكَ فَأُو۟لَـٰٓئِكَ هُمُ ٱلْخَـٰسِرُونَ ﴿9﴾
اے ایمان والو! تمہارے مال اور تمہاری اولاد تمہیں اللہ کے ذکر سے غافل نہ کریں، اور جو ایسا کرے وہی نقصان اٹھانے والے ہیں۔
وَأَنفِقُوا۟ مِن مَّا رَزَقْنَـٰكُم مِّن قَبْلِ أَن يَأْتِىَ أَحَدَكُمُ ٱلْمَوْتُ فَيَقُولَ رَبِّ لَوْلَآ أَخَّرْتَنِىٓ إِلَىٰٓ أَجَلٍۢ قَرِيبٍۢ فَأَصَّدَّقَ وَأَكُن مِّنَ ٱلصَّـٰلِحِينَ ﴿10﴾
اور جو کچھ ہم نے تمہیں دیا ہے اس میں سے خرچ کرو اس سے پہلے کہ تم میں سے کسی کو موت آ جائے، تو وہ کہے: اے میرے رب! مجھے تھوڑی مہلت دے دے تاکہ میں صدقہ دے سکوں اور نیک لوگوں میں شامل ہو جاؤں۔
وَلَن يُؤَخِّرَ ٱللَّهُ نَفْسًا إِذَا جَآءَ أَجَلُهَا ۚ وَٱللَّهُ خَبِيرٌۢ بِمَا تَعْمَلُونَ ﴿11﴾
اور جب کسی کی موت کا وقت آ جائے تو اللہ ہرگز مہلت نہیں دیتا، اور اللہ تمہارے اعمال سے خوب باخبر ہے۔
سورة التغابن
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
يُسَبِّحُ لِلَّهِ مَا فِى ٱلسَّمَـٰوَٰتِ وَمَا فِى ٱلْأَرْضِ ۖ لَهُ ٱلْمُلْكُ وَلَهُ ٱلْحَمْدُ ۖ وَهُوَ عَلَىٰ كُلِّ شَىْءٍۢ قَدِيرٌ ﴿1﴾
اللہ کی تسبیح کرتی ہیں وہ سب چیزیں جو آسمانوں اور زمین میں ہیں، اسی کے لیے بادشاہی ہے اور اسی کے لیے حمد ہے، اور وہ ہر چیز پر قدرت رکھتا ہے۔
هُوَ ٱلَّذِى خَلَقَكُمْ فَمِنكُمْ كَافِرٌۭ وَمِنكُم مُّؤْمِنٌۭ ۚ وَٱللَّهُ بِمَا تَعْمَلُونَ بَصِيرٌ ﴿2﴾
وہی ہے جس نے تمہیں پیدا کیا، پھر تم میں سے کوئی کافر ہے اور کوئی مومن، اور اللہ تمہارے اعمال کو خوب دیکھنے والا ہے۔
خَلَقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ بِٱلْحَقِّ ۖ وَصَوَّرَكُمْ فَأَحْسَنَ صُوَرَكُمْ ۚ وَإِلَيْهِ ٱلْمَصِيرُ ﴿3﴾
اس نے آسمانوں اور زمین کو حق کے ساتھ پیدا کیا، اور تمہاری صورت بنائی تو بہترین صورت بنائی، اور اسی کی طرف لوٹنا ہے۔
يَعْلَمُ مَا فِى ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَيَعْلَمُ مَا تُسِرُّونَ وَمَا تُعْلِنُونَ ۚ وَٱللَّهُ عَلِيمٌۢ بِذَاتِ ٱلصُّدُورِ ﴿4﴾
وہ جانتا ہے جو کچھ آسمانوں اور زمین میں ہے، اور وہ جانتا ہے جو کچھ تم چھپاتے ہو اور جو ظاہر کرتے ہو، اور اللہ دلوں کے بھیدوں کو خوب جاننے والا ہے۔
أَلَمْ يَأْتِكُمْ نَبَؤُا۟ ٱلَّذِينَ كَفَرُوا۟ مِن قَبْلُ فَذَاقُوا۟ وَبَالَ أَمْرِهِمْ وَلَهُمْ عَذَابٌ أَلِيمٌۭ ﴿5﴾
کیا تمہارے پاس ان لوگوں کی خبر نہیں آئی جنہوں نے اس سے پہلے کفر کیا، تو انہوں نے اپنے کام کا وبال چکھا، اور ان کے لیے دردناک عذاب ہے۔
ذَٰلِكَ بِأَنَّهُۥ كَانَت تَّأْتِيهِمْ رُسُلُهُم بِٱلْبَيِّنَـٰتِ فَقَالُوٓا۟ أَبَشَرٌۭ يَهْدُونَنَا فَكَفَرُوا۟ وَتَوَلَّوا۟ ۚ وَّٱسْتَغْنَى ٱللَّهُ ۚ وَٱللَّهُ غَنِىٌّ حَمِيدٌ ﴿6﴾
یہ اس لیے کہ ان کے پاس ان کے رسول واضح دلائل لے کر آتے تھے، تو انہوں نے کہا: کیا انسان ہمیں ہدایت دیں گے؟ پس انہوں نے کفر کیا اور منہ موڑ لیا، اور اللہ بے نیاز ہے، اور اللہ تعریف کے لائق ہے۔
زَعَمَ ٱلَّذِينَ كَفَرُوٓا۟ أَن لَّن يُبْعَثُوا۟ ۚ قُلْ بَلَىٰ وَرَبِّى لَتُبْعَثُنَّ ثُمَّ لَتُنَبَّؤُنَّ بِمَا عَمِلْتُمْ ۚ وَذَٰلِكَ عَلَى ٱللَّهِ يَسِيرٌۭ ﴿7﴾
جن لوگوں نے کفر کیا وہ گمان کرتے ہیں کہ وہ دوبارہ نہیں اٹھائے جائیں گے۔ کہہ دو: کیوں نہیں! میرے رب کی قسم، تم ضرور اٹھائے جاؤ گے، پھر تمہیں بتایا جائے گا جو کچھ تم نے کیا، اور یہ اللہ پر آسان ہے۔
فَـَٔامِنُوا۟ بِٱللَّهِ وَرَسُولِهِۦ وَٱلنُّورِ ٱلَّذِىٓ أَنزَلْنَا ۚ وَٱللَّهُ بِمَا تَعْمَلُونَ خَبِيرٌۭ ﴿8﴾
پس اللہ اور اس کے رسول پر ایمان لاؤ اور اس نور پر جو ہم نے نازل کیا، اور اللہ تمہارے اعمال سے خوب خبردار ہے۔
يَوْمَ يَجْمَعُكُمْ لِيَوْمِ ٱلْجَمْعِ ۚ ذَٰلِكَ يَوْمُ ٱلتَّغَابُنِ ۗ وَمَن يُؤْمِنۢ بِٱللَّهِ وَيَعْمَلْ صَـٰلِحًۭا يُكَفِّرْ عَنْهُ سَيِّـَٔاتِهِۦ وَيُدْخِلْهُ جَنَّـٰتٍۢ تَجْرِى مِن تَحْتِهَا ٱلْأَنْهَـٰرُ خَـٰلِدِينَ فِيهَآ أَبَدًۭا ۚ ذَٰلِكَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿9﴾
جس دن وہ تمہیں جمع کرے گا جمع ہونے کے دن کے لیے، وہی دن ہے نقصان اور فائدے کا۔ اور جو اللہ پر ایمان لائے اور نیک عمل کرے، اللہ اس سے اس کے گناہ دور کر دے گا اور اسے ایسی جنتوں میں داخل کرے گا جن کے نیچے نہریں بہتی ہیں، وہ ان میں ہمیشہ رہیں گے۔ یہی بڑی کامیابی ہے۔
وَٱلَّذِينَ كَفَرُوا۟ وَكَذَّبُوا۟ بِـَٔايَـٰتِنَآ أُو۟لَـٰٓئِكَ أَصْحَـٰبُ ٱلنَّارِ خَـٰلِدِينَ فِيهَا ۖ وَبِئْسَ ٱلْمَصِيرُ ﴿10﴾
اور جنہوں نے کفر کیا اور ہماری آیات کو جھٹلایا، وہی لوگ جہنم والے ہیں، وہ اس میں ہمیشہ رہیں گے، اور وہ بہت ہی برا ٹھکانہ ہے۔
مَآ أَصَابَ مِن مُّصِيبَةٍ إِلَّا بِإِذْنِ ٱللَّهِ ۗ وَمَن يُؤْمِنۢ بِٱللَّهِ يَهْدِ قَلْبَهُۥ ۚ وَٱللَّهُ بِكُلِّ شَىْءٍ عَلِيمٌۭ ﴿11﴾
کوئی مصیبت نہیں آتی مگر اللہ کے حکم سے، اور جو اللہ پر ایمان رکھتا ہے، اللہ اس کے دل کو ہدایت دیتا ہے، اور اللہ ہر چیز کو خوب جاننے والا ہے۔
وَأَطِيعُوا۟ ٱللَّهَ وَأَطِيعُوا۟ ٱلرَّسُولَ ۖ فَإِن تَوَلَّيْتُمْ فَإِنَّمَا عَلَىٰ رَسُولِنَا ٱلْبَلَـٰغُ ٱلْمُبِينُ ﴿12﴾
اور اللہ کی اطاعت کرو اور رسول کی اطاعت کرو، پھر اگر تم منہ موڑو تو ہمارے رسول پر صرف واضح طور پر پہنچا دینا ہے۔
ٱللَّهُ لَآ إِلَـٰهَ إِلَّا هُوَ ۚ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُؤْمِنُونَ ﴿13﴾
اللہ ہی ہے جس کے سوا کوئی معبود نہیں، پس ایمان والوں کو اللہ ہی پر بھروسہ کرنا چاہیے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ إِنَّ مِنْ أَزْوَٰجِكُمْ وَأَوْلَـٰدِكُمْ عَدُوًّۭا لَّكُمْ فَٱحْذَرُوهُمْ ۚ وَإِن تَعْفُوا۟ وَتَصْفَحُوا۟ وَتَغْفِرُوا۟ فَإِنَّ ٱللَّهَ غَفُورٌۭ رَّحِيمٌ ﴿14﴾
اے ایمان والو! تمہاری بیویوں اور اولاد میں سے بعض تمہارے دشمن ہیں، پس ان سے بچو۔ اور اگر تم معاف کر دو، درگزر کرو اور بخش دو تو بے شک اللہ بخشنے والا، رحم کرنے والا ہے۔
إِنَّمَآ أَمْوَٰلُكُمْ وَأَوْلَـٰدُكُمْ فِتْنَةٌۭ ۚ وَٱللَّهُ عِندَهُۥٓ أَجْرٌ عَظِيمٌۭ ﴿15﴾
تمہارے مال اور تمہاری اولاد تو آزمائش ہیں، اور اللہ کے پاس بڑا اجر ہے۔
فَٱتَّقُوا۟ ٱللَّهَ مَا ٱسْتَطَعْتُمْ وَٱسْمَعُوا۟ وَأَطِيعُوا۟ وَأَنفِقُوا۟ خَيْرًۭا لِّأَنفُسِكُمْ ۗ وَمَن يُوقَ شُحَّ نَفْسِهِۦ فَأُو۟لَـٰٓئِكَ هُمُ ٱلْمُفْلِحُونَ ﴿16﴾
پس اللہ سے ڈرو جتنا تم سے ہو سکے، اور سنو، اطاعت کرو، اور خرچ کرو، یہ تمہارے لیے بہتر ہے۔ اور جو اپنے نفس کی بخل سے بچا لیا گیا، وہی کامیاب ہیں۔
إِن تُقْرِضُوا۟ ٱللَّهَ قَرْضًا حَسَنًۭا يُضَـٰعِفْهُ لَكُمْ وَيَغْفِرْ لَكُمْ ۚ وَٱللَّهُ شَكُورٌ حَلِيمٌ ﴿17﴾
اگر تم اللہ کو اچھا قرض دو گے تو وہ تمہارے لیے اسے کئی گنا بڑھا دے گا اور تمہیں بخش دے گا، اور اللہ قدر دان، بردبار ہے۔
عَـٰلِمُ ٱلْغَيْبِ وَٱلشَّهَـٰدَةِ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿18﴾
وہ غیب اور ظاہر کا جاننے والا ہے، زبردست، حکمت والا ہے۔
سورة الطلاق
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
يَـٰٓأَيُّهَا ٱلنَّبِىُّ إِذَا طَلَّقْتُمُ ٱلنِّسَآءَ فَطَلِّقُوهُنَّ لِعِدَّتِهِنَّ وَأَحْصُوا۟ ٱلْعِدَّةَ ۖ وَٱتَّقُوا۟ ٱللَّهَ رَبَّكُمْ ۖ لَا تُخْرِجُوهُنَّ مِنۢ بُيُوتِهِنَّ وَلَا يَخْرُجْنَ إِلَّآ أَن يَأْتِينَ بِفَـٰحِشَةٍۢ مُّبَيِّنَةٍۢ ۚ وَتِلْكَ حُدُودُ ٱللَّهِ ۚ وَمَن يَتَعَدَّ حُدُودَ ٱللَّهِ فَقَدْ ظَلَمَ نَفْسَهُۥ ۚ لَا تَدْرِى لَعَلَّ ٱللَّهَ يُحْدِثُ بَعْدَ ذَٰلِكَ أَمْرًۭا ﴿1﴾
اے نبی! جب تم عورتوں کو طلاق دو تو انہیں ان کی عدت کے لیے طلاق دو، اور عدت کا شمار رکھو، اور اللہ سے ڈرو جو تمہارا رب ہے۔ انہیں ان کے گھروں سے نہ نکالو اور نہ وہ خود نکلیں، مگر یہ کہ وہ کھلی بے حیائی کریں۔ یہ اللہ کی حدیں ہیں، اور جو اللہ کی حدوں سے تجاوز کرے وہ اپنے آپ پر ظلم کرتا ہے۔ تم نہیں جانتے شاید اللہ اس کے بعد کوئی نئی بات پیدا کر دے۔
فَإِذَا بَلَغْنَ أَجَلَهُنَّ فَأَمْسِكُوهُنَّ بِمَعْرُوفٍ أَوْ فَارِقُوهُنَّ بِمَعْرُوفٍ ۚ وَأَشْهِدُوا۟ ذَوَىْ عَدْلٍۢ مِّنكُمْ وَأَقِيمُوا۟ ٱلشَّهَـٰدَةَ لِلَّهِ ۚ ذَٰلِكُمْ يُوعَظُ بِهِۦ مَن كَانَ يُؤْمِنُ بِٱللَّهِ وَٱلْيَوْمِ ٱلْـَٔاخِرِ ۚ وَمَن يَتَّقِ ٱللَّهَ يَجْعَلْ لَهُۥ مَخْرَجًۭا ﴿2﴾
پھر جب وہ اپنی عدت کے اختتام پر پہنچ جائیں تو انہیں بھلے طریقے سے روک لو یا بھلے طریقے سے جدا کر دو، اور اپنے میں سے دو انصاف والے گواہ بنا لو، اور اللہ کے لیے گواہی قائم کرو۔ یہ نصیحت ہے اس کے لیے جو اللہ اور آخرت کے دن پر ایمان رکھتا ہے۔ اور جو اللہ سے ڈرتا ہے اللہ اس کے لیے نکلنے کا راستہ بنا دیتا ہے۔
وَيَرْزُقْهُ مِنْ حَيْثُ لَا يَحْتَسِبُ ۚ وَمَن يَتَوَكَّلْ عَلَى ٱللَّهِ فَهُوَ حَسْبُهُۥٓ ۚ إِنَّ ٱللَّهَ بَـٰلِغُ أَمْرِهِۦ ۚ قَدْ جَعَلَ ٱللَّهُ لِكُلِّ شَىْءٍۢ قَدْرًۭا ﴿3﴾
اور اسے وہاں سے رزق دے گا جہاں سے اسے گمان بھی نہ ہو، اور جو اللہ پر بھروسہ کرے تو وہ اس کے لیے کافی ہے۔ بے شک اللہ اپنا کام پورا کرنے والا ہے، اللہ نے ہر چیز کے لیے ایک اندازہ مقرر کر رکھا ہے۔
وَٱلَّـٰٓـِٔى يَئِسْنَ مِنَ ٱلْمَحِيضِ مِن نِّسَآئِكُمْ إِنِ ٱرْتَبْتُمْ فَعِدَّتُهُنَّ ثَلَـٰثَةُ أَشْهُرٍۢ وَٱلَّـٰٓـِٔى لَمْ يَحِضْنَ ۚ وَأُو۟لَـٰتُ ٱلْأَحْمَالِ أَجَلُهُنَّ أَن يَضَعْنَ حَمْلَهُنَّ ۚ وَمَن يَتَّقِ ٱللَّهَ يَجْعَلْ لَهُۥ مِنْ أَمْرِهِۦ يُسْرًۭا ﴿4﴾
اور تمہاری عورتوں میں سے جو حیض سے ناامید ہو چکی ہوں، اگر تمہیں شک ہو تو ان کی عدت تین مہینے ہے، اور ان کی بھی جنہیں ابھی حیض نہیں آیا۔ اور حاملہ عورتوں کی عدت ان کا بچہ جننا ہے۔ اور جو اللہ سے ڈرتا ہے اللہ اس کے معاملے میں آسانی پیدا کر دیتا ہے۔
ذَٰلِكَ أَمْرُ ٱللَّهِ أَنزَلَهُۥٓ إِلَيْكُمْ ۚ وَمَن يَتَّقِ ٱللَّهَ يُكَفِّرْ عَنْهُ سَيِّـَٔاتِهِۦ وَيُعْظِمْ لَهُۥٓ أَجْرًۭا ﴿5﴾
یہ اللہ کا حکم ہے جو اس نے تمہاری طرف نازل کیا ہے، اور جو اللہ سے ڈرتا ہے اللہ اس سے اس کے گناہ دور کر دیتا ہے اور اسے بڑا اجر عطا کرتا ہے۔
أَسْكِنُوهُنَّ مِنْ حَيْثُ سَكَنْتُم مِّن وُجْدِكُمْ وَلَا تُضَآرُّوهُنَّ لِتُضَيِّقُوا۟ عَلَيْهِنَّ ۚ وَإِن كُنَّ أُو۟لَـٰتِ حَمْلٍۢ فَأَنفِقُوا۟ عَلَيْهِنَّ حَتَّىٰ يَضَعْنَ حَمْلَهُنَّ ۚ فَإِنْ أَرْضَعْنَ لَكُمْ فَـَٔاتُوهُنَّ أُجُورَهُنَّ ۚ وَأْتَمِرُوا۟ بَيْنَكُم بِمَعْرُوفٍۢ ۖ وَإِن تَعَاسَرْتُمْ فَسَتُرْضِعُ لَهُۥٓ أُخْرَىٰ ﴿6﴾
انہیں وہیں رکھو جہاں تم رہتے ہو اپنی استطاعت کے مطابق، اور انہیں تنگ کرنے کے لیے نقصان نہ پہنچاؤ۔ اور اگر وہ حاملہ ہوں تو ان پر خرچ کرو یہاں تک کہ وہ اپنا حمل وضع کریں۔ پھر اگر وہ تمہارے لیے دودھ پلائیں تو انہیں ان کی اجرت دو، اور آپس میں بھلے طریقے سے مشورہ کرو۔ اور اگر تمہارا آپس میں اختلاف ہو جائے تو کوئی اور عورت دودھ پلائے گی۔
لِيُنفِقْ ذُو سَعَةٍۢ مِّن سَعَتِهِۦ ۖ وَمَن قُدِرَ عَلَيْهِ رِزْقُهُۥ فَلْيُنفِقْ مِمَّآ ءَاتَىٰهُ ٱللَّهُ ۚ لَا يُكَلِّفُ ٱللَّهُ نَفْسًا إِلَّا مَآ ءَاتَىٰهَا ۗ سَيَجْعَلُ ٱللَّهُ بَعْدَ عُسْرٍۢ يُسْرًۭا ﴿7﴾
جس کے پاس کشادگی ہو وہ اپنی کشادگی کے مطابق خرچ کرے، اور جس پر اس کا رزق تنگ ہو وہ اسی میں سے خرچ کرے جو اللہ نے اسے دیا ہے۔ اللہ کسی جان کو اس کی طاقت سے زیادہ تکلیف نہیں دیتا۔ اللہ تنگی کے بعد آسانی پیدا کر دے گا۔
وَكَأَيِّن مِّن قَرْيَةٍ عَتَتْ عَنْ أَمْرِ رَبِّهَا وَرُسُلِهِۦ فَحَاسَبْنَـٰهَا حِسَابًۭا شَدِيدًۭا وَعَذَّبْنَـٰهَا عَذَابًۭا نُّكْرًۭا ﴿8﴾
اور کتنی ہی بستیاں تھیں جنہوں نے اپنے رب اور اس کے رسولوں کے حکم سے سرکشی کی، تو ہم نے ان کا سخت حساب لیا اور انہیں سخت عذاب دیا۔
فَذَاقَتْ وَبَالَ أَمْرِهَا وَكَانَ عَـٰقِبَةُ أَمْرِهَا خُسْرًۭا ﴿9﴾
تو انہوں نے اپنے کام کا وبال چکھ لیا، اور ان کے کام کا انجام خسارہ ہی ہوا۔
أَعَدَّ ٱللَّهُ لَهُمْ عَذَابًۭا شَدِيدًۭا ۖ فَٱتَّقُوا۟ ٱللَّهَ يَـٰٓأُو۟لِى ٱلْأَلْبَـٰبِ ٱلَّذِينَ ءَامَنُوا۟ ۚ قَدْ أَنزَلَ ٱللَّهُ إِلَيْكُمْ ذِكْرًۭا ﴿10﴾
اللہ نے ان کے لیے سخت عذاب تیار کر رکھا ہے، پس اے عقل والو! اللہ سے ڈرو، جو ایمان لائے ہو۔ اللہ نے تمہاری طرف نصیحت نازل کی ہے۔
رَّسُولًۭا يَتْلُوا۟ عَلَيْكُمْ ءَايَـٰتِ ٱللَّهِ مُبَيِّنَـٰتٍۢ لِّيُخْرِجَ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ مِنَ ٱلظُّلُمَـٰتِ إِلَى ٱلنُّورِ ۚ وَمَن يُؤْمِنۢ بِٱللَّهِ وَيَعْمَلْ صَـٰلِحًۭا يُدْخِلْهُ جَنَّـٰتٍۢ تَجْرِى مِن تَحْتِهَا ٱلْأَنْهَـٰرُ خَـٰلِدِينَ فِيهَآ أَبَدًۭا ۖ قَدْ أَحْسَنَ ٱللَّهُ لَهُۥ رِزْقًۭا ﴿11﴾
ایک رسول تم پر اللہ کی واضح آیات پڑھتا ہے تاکہ وہ ایمان لانے والوں اور نیک عمل کرنے والوں کو اندھیروں سے نکال کر روشنی کی طرف لے آئے۔ اور جو اللہ پر ایمان لائے اور نیک عمل کرے، اللہ اسے ایسی جنتوں میں داخل کرے گا جن کے نیچے نہریں بہتی ہیں، وہ ان میں ہمیشہ رہیں گے۔ اللہ نے اس کے لیے بہترین رزق تیار کر رکھا ہے۔
ٱللَّهُ ٱلَّذِى خَلَقَ سَبْعَ سَمَـٰوَٰتٍۢ وَمِنَ ٱلْأَرْضِ مِثْلَهُنَّ ۖ يَتَنَزَّلُ ٱلْأَمْرُ بَيْنَهُنَّ لِتَعْلَمُوٓا۟ أَنَّ ٱللَّهَ عَلَىٰ كُلِّ شَىْءٍۢ قَدِيرٌۭ وَأَنَّ ٱللَّهَ قَدْ أَحَاطَ بِكُلِّ شَىْءٍ عِلْمًۭا ﴿12﴾
اللہ وہ ہے جس نے سات آسمان پیدا کیے اور زمین سے بھی ان جیسی، اس کا حکم ان کے درمیان نازل ہوتا ہے تاکہ تم جان لو کہ اللہ ہر چیز پر قدرت رکھتا ہے اور اللہ نے ہر چیز کو اپنے علم سے گھیر رکھا ہے۔
سورة التحريم
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
يَـٰٓأَيُّهَا ٱلنَّبِىُّ لِمَ تُحَرِّمُ مَآ أَحَلَّ ٱللَّهُ لَكَ ۖ تَبْتَغِى مَرْضَاتَ أَزْوَٰجِكَ ۚ وَٱللَّهُ غَفُورٌۭ رَّحِيمٌۭ ﴿1﴾
اے نبی! تم کیوں اس چیز کو حرام کرتے ہو جو اللہ نے تمہارے لیے حلال کی ہے؟ تم اپنی بیویوں کی خوشنودی چاہتے ہو، اور اللہ بخشنے والا، رحم کرنے والا ہے۔
قَدْ فَرَضَ ٱللَّهُ لَكُمْ تَحِلَّةَ أَيْمَـٰنِكُمْ ۚ وَٱللَّهُ مَوْلَىٰكُمْ ۖ وَهُوَ ٱلْعَلِيمُ ٱلْحَكِيمُ ﴿2﴾
اللہ نے تمہارے لیے اپنی قسموں کا کفارہ مقرر کر دیا ہے، اور اللہ تمہارا مالک ہے، اور وہی علم والا، حکمت والا ہے۔
وَإِذْ أَسَرَّ ٱلنَّبِىُّ إِلَىٰ بَعْضِ أَزْوَٰجِهِۦ حَدِيثًۭا فَلَمَّا نَبَّأَتْ بِهِۦ وَأَظْهَرَهُ ٱللَّهُ عَلَيْهِ عَرَّفَ بَعْضَهُۥ وَأَعْرَضَ عَنۢ بَعْضٍۢ ۖ فَلَمَّا نَبَّ...َأَكَ هَـٰذَا ۖ قَالَ نَبَّأَنِىَ ٱلْعَلِيمُ ٱلْخَبِيرُ ﴿3﴾
اور جب نبی نے اپنی ایک بیوی سے ایک راز کی بات کی، پھر جب اس نے اسے ظاہر کر دیا اور اللہ نے نبی کو اس کی اطلاع دے دی، تو نبی نے اس کا کچھ حصہ بتا دیا اور کچھ سے چشم پوشی کی، پھر جب نبی نے اسے بتایا تو اس نے کہا: آپ کو یہ کس نے بتایا؟ نبی نے کہا: مجھے علم والے، خبردار نے بتایا۔
إِن تَتُوبَآ إِلَى ٱللَّهِ فَقَدْ صَغَتْ قُلُوبُكُمَا ۖ وَإِن تَظَـٰهَرَا عَلَيْهِ فَإِنَّ ٱللَّهَ هُوَ مَوْلَىٰهُ وَجِبْرِيلُ وَصَـٰلِحُ ٱلْمُؤْمِنِينَ ۖ وَٱلْمَلَـٰٓئِكَةُ بَعْدَ ذَٰلِكَ ظَهِيرٌ ﴿4﴾
اگر تم دونوں اللہ سے توبہ کرو تو تمہارے دل جھک چکے ہیں، اور اگر تم نبی کے خلاف ایک دوسرے کی مدد کرو تو اللہ اس کا مددگار ہے، اور جبرائیل، اور نیک مومن، اور اس کے بعد فرشتے بھی مددگار ہیں۔
عَسَىٰ رَبُّهُۥٓ إِن طَلَّقَكُنَّ أَن يُبْدِلَهُۥٓ أَزْوَٰجًۭا خَيْرًۭا مِّنكُنَّ مُسْلِمَـٰتٍۢ مُّؤْمِنَـٰتٍۢ قَـٰنِتَـٰتٍۢ تَـٰٓئِبَـٰتٍ عَـٰبِدَٰتٍۢ سَـٰٓئِحَـٰتٍۢ ثَيِّبَـٰتٍۢ وَأَبْكَارًۭا ﴿5﴾
اگر نبی تمہیں طلاق دے دے تو امید ہے کہ اس کا رب اسے تم سے بہتر بیویاں دے دے: مسلمان، مومن، فرمانبردار، توبہ کرنے والی، عبادت گزار، سفر کرنے والی، بیوہ اور کنواری۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ قُوا۟ أَنفُسَكُمْ وَأَهْلِيكُمْ نَارًۭا وَقُودُهَا ٱلنَّاسُ وَٱلْحِجَارَةُ عَلَيْهَا مَلَـٰٓئِكَةٌۭ غِلَاظٌۭ شِدَادٌۭ لَّا يَعْصُونَ ٱللَّهَ مَآ أَمَرَهُمْ وَيَفْعَلُونَ مَا يُؤْمَرُونَ ﴿6﴾
اے ایمان والو! اپنے آپ کو اور اپنے گھر والوں کو اس آگ سے بچاؤ جس کا ایندھن انسان اور پتھر ہیں، اس پر سخت اور طاقتور فرشتے مقرر ہیں، جو اللہ کے حکم کی نافرمانی نہیں کرتے اور وہی کرتے ہیں جو انہیں حکم دیا جاتا ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ كَفَرُوا۟ لَا تَعْتَذِرُوا۟ ٱلْيَوْمَ ۖ إِنَّمَا تُجْزَوْنَ مَا كُنتُمْ تَعْمَلُونَ ﴿7﴾
اے کافرو! آج معذرت نہ کرو، تمہیں وہی بدلہ دیا جائے گا جو تم عمل کرتے تھے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ تُوبُوٓا۟ إِلَى ٱللَّهِ تَوْبَةًۭ نَّصُوحًا عَسَىٰ رَبُّكُمْ أَن يُكَفِّرَ عَنكُمْ سَيِّـَٔاتِكُمْ وَيُدْخِلَكُمْ جَنَّـٰتٍۢ تَجْرِى مِن تَحْتِهَا ٱلْأَنْهَـٰرُ ۚ يَوْمَ لَا يُخْزِى ٱللَّهُ ٱلنَّبِىَّ وَٱلَّذِينَ ءَامَنُوا۟ مَعَهُ ۚ نُورُهُمْ يَسْعَىٰ بَيْنَ أَيْدِيهِمْ وَبِأَيْمَـٰنِهِمْ يَقُولُونَ رَبَّنَآ أَتْمِمْ لَنَا نُورَنَا وَٱغْفِرْ لَنَآ ۖ إِنَّكَ عَلَىٰ كُلِّ شَىْءٍۢ قَدِيرٌ ﴿8﴾
اے ایمان والو! اللہ کی طرف سچی توبہ کرو، امید ہے کہ تمہارا رب تم سے تمہارے گناہ دور کر دے گا اور تمہیں ان جنتوں میں داخل کرے گا جن کے نیچے نہریں بہتی ہیں۔ اس دن اللہ نبی کو اور ان لوگوں کو جو اس کے ساتھ ایمان لائے ہیں، رسوا نہ کرے گا، ان کا نور ان کے آگے اور دائیں جانب دوڑ رہا ہوگا۔ وہ کہیں گے: اے ہمارے رب! ہمارا نور ہمارے لیے مکمل کر دے اور ہمیں بخش دے، بے شک تو ہر چیز پر قدرت رکھتا ہے۔
يَـٰٓأَيُّهَا ٱلنَّبِىُّ جَـٰهِدِ ٱلْكُفَّارَ وَٱلْمُنَـٰفِقِينَ وَٱغْلُظْ عَلَيْهِمْ ۚ وَمَأْوَىٰهُمْ جَهَنَّمُ ۖ وَبِئْسَ ٱلْمَصِيرُ ﴿9﴾
اے نبی! کافروں اور منافقوں سے جہاد کرو اور ان پر سختی کرو، ان کا ٹھکانہ جہنم ہے، اور وہ بہت ہی برا ٹھکانہ ہے۔
ضَرَبَ ٱللَّهُ مَثَلًۭا لِّلَّذِينَ كَفَرُوا۟ ٱمْرَأَتَ نُوحٍۢ وَٱمْرَأَتَ لُوطٍۢ ۖ كَانَتَا تَحْتَ عَبْدَيْنِ مِنْ عِبَادِنَا صَـٰلِحَيْنِ فَخَانَتَاهُمَا فَلَمْ يُغْنِيَا عَنْهُمَا مِنَ ٱللَّهِ شَيْـًۭٔا وَقِيلَ ٱدْخُلَا ٱلنَّارَ مَعَ ٱلدَّاخِلِينَ ﴿10﴾
اللہ نے کافروں کے لیے نوح کی بیوی اور لوط کی بیوی کی مثال بیان کی، وہ دونوں ہمارے دو نیک بندوں کے زیرِ نکاح تھیں، مگر انہوں نے ان سے خیانت کی، تو وہ دونوں ان کے لیے اللہ کے مقابلے میں کچھ کام نہ آئے، اور کہا گیا: تم دونوں جہنم میں داخل ہو جاؤ داخل ہونے والوں کے ساتھ۔
وَضَرَبَ ٱللَّهُ مَثَلًۭا لِّلَّذِينَ ءَامَنُوا۟ ٱمْرَأَتَ فِرْعَوْنَ إِذْ قَالَتْ رَبِّ ٱبْنِ لِى عِندَكَ بَيْتًۭا فِى ٱلْجَنَّةِ وَنَجِّنِى مِن فِرْعَوْنَ وَعَمَلِهِۦ وَنَجِّنِى مِنَ ٱلْقَوْمِ ٱلظَّـٰلِمِينَ ﴿11﴾
اور اللہ نے ایمان والوں کے لیے فرعون کی بیوی کی مثال بیان کی، جب اس نے کہا: اے میرے رب! میرے لیے اپنے پاس جنت میں ایک گھر بنا دے، اور مجھے فرعون اور اس کے عمل سے نجات دے، اور مجھے ظالم قوم سے بچا۔
وَمَرْيَمَ ٱبْنَتَ عِمْرَٰنَ ٱلَّتِىٓ أَحْصَنَتْ فَرْجَهَا فَنَفَخْنَا فِيهِ مِن رُّوحِنَا وَصَدَّقَتْ بِكَلِمَـٰتِ رَبِّهَا وَكُتُبِهِۦ وَكَانَتْ مِنَ ٱلْقَـٰنِتِينَ ﴿12﴾
اور مریم بنتِ عمران کی مثال، جس نے اپنی شرمگاہ کی حفاظت کی، تو ہم نے اس میں اپنی روح پھونکی، اور اس نے اپنے رب کے کلمات اور اس کی کتابوں کی تصدیق کی، اور وہ فرمانبرداروں میں سے تھی۔
