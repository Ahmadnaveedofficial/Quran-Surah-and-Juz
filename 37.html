سورة الصافات

بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
وَٱلصَّـٰفَّـٰتِ صَفّٗا ﴿1﴾
قسم ہے ان (فرشتوں) کی جو صف باندھ کر کھڑے ہوتے ہیں۔
فَٱلزَّٰجِرَٰتِ زَجْرٗا ﴿2﴾
پھر ان کی جو جھڑک کر چلاتی ہیں۔
فَٱلتَّٰلِيَٰتِ ذِكْرًا ﴿3﴾
پھر ان کی جو ذکر پڑھتی ہیں۔
إِنَّ إِلَـٰهَكُمْ لَوَٰحِدٞ ﴿4﴾
بے شک تمہارا معبود ایک ہی ہے۔
رَّبُّ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَا وَرَبُّ ٱلْمَشَـٰرِقِ ﴿5﴾
آسمانوں اور زمین اور جو کچھ ان کے درمیان ہے سب کا رب، اور مشرقوں کا رب۔
إِنَّا زَيَّنَّا ٱلسَّمَآءَ ٱلدُّنْيَا بِزِينَةٍ ٱلْكَوَاكِبِ ﴿6﴾
بے شک ہم نے آسمانِ دنیا کو ستاروں کی زینت سے آراستہ کیا۔
وَحِفْظٗا مِّن كُلِّ شَيْطَٰنٖ مَّارِدٖ ﴿7﴾
اور ہر سرکش شیطان سے حفاظت کے لیے۔
لَا يَسَّمَّعُونَ إِلَى ٱلْمَلَإِ ٱلْأَعْلَىٰ وَيُقْذَفُونَ مِن كُلِّ جَانِبٖ ﴿8﴾
وہ بلند جماعت (فرشتوں) کی باتیں نہیں سن سکتے، اور ہر طرف سے ان پر پھینکا جاتا ہے۔
دُحُورٗاۖ وَلَهُمْ عَذَابٞ وَاصِبٌ ﴿9﴾
دھتکارے ہوئے، اور ان کے لیے مسلسل عذاب ہے۔
إِلَّا مَنْ خَطِفَ ٱلْخَطْفَةَ فَأَتْبَعَهُۥ شِهَابٞ ثَاقِبٞ ﴿10﴾
سوائے اس کے جو چوری سے کچھ لے اُڑے، تو اس کے پیچھے چمکتا ہوا شعلہ لگا دیا جاتا ہے۔
فَٱسْتَفْتِهِمْ أَهُمْ أَشَدُّ خَلْقًا أَم مَّنْ خَلَقْنَآۚ إِنَّا خَلَقْنَـٰهُم مِّن طِينٖ لَّازِبٖ ﴿11﴾
تو ان سے پوچھو: کیا ان کا بنانا زیادہ مشکل ہے یا وہ جنہیں ہم نے پیدا کیا؟ ہم نے انہیں چپکنے والی مٹی سے پیدا کیا۔
بَلْ عَجِبْتَ وَيَسْخَرُونَ ﴿12﴾
بلکہ تم تعجب کرتے ہو اور وہ مذاق اڑاتے ہیں۔
وَإِذَا ذُكِّرُوا۟ لَا يَذْكُرُونَ ﴿13﴾
اور جب انہیں نصیحت کی جاتی ہے تو وہ نصیحت نہیں پکڑتے۔
وَإِذَا رَأَوْا۟ ءَايَةٗ يَسْتَسْخِرُونَ ﴿14﴾
اور جب کوئی نشانی دیکھتے ہیں تو مذاق بناتے ہیں۔
وَقَالُوٓا۟ إِنْ هَـٰذَآ إِلَّا سِحْرٞ مُّبِينٌ ﴿15﴾
اور کہتے ہیں: یہ تو کھلا جادو ہے۔
أَءِذَا مِتْنَا وَكُنَّا تُرَٰبٗا وَعِظَـٰمًا أَءِنَّا لَمَبْعُوثُونَ ﴿16﴾
کیا جب ہم مر جائیں گے اور مٹی اور ہڈیاں ہو جائیں گے، کیا ہم پھر اٹھائے جائیں گے؟
أَوَ ءَابَآؤُنَا ٱلْأَوَّلُونَ ﴿17﴾
یا ہمارے پہلے باپ دادا بھی؟
قُلْ نَعَمْ وَأَنتُمْ دَٰخِرُونَ ﴿18﴾
کہہ دو: ہاں! اور تم ذلیل ہو کر اٹھائے جاؤ گے۔
فَإِنَّمَا هِيَ زَجْرَةٞ وَٰحِدَةٞ فَإِذَا هُمۡ يَنظُرُونَ ﴿19﴾
بس وہ ایک ہی چیخ ہوگی، پھر وہ سب دیکھنے لگیں گے۔
وَقَالُوا۟ يَـٰوَيْلَنَا هَـٰذَا يَوْمُ ٱلدِّينِ ﴿20﴾
اور وہ کہیں گے: ہائے ہماری شامت! یہ تو حساب کا دن ہے۔
هَٰذَا يَوْمُ ٱلْفَصْلِ ٱلَّذِي كُنتُم بِهِۦ تُكَذِّبُونَ ﴿21﴾
یہی ہے فیصلے کا دن جسے تم جھٹلایا کرتے تھے۔
ٱحْشُرُوا۟ ٱلَّذِينَ ظَلَمُوا۟ وَأَزْوَٰجَهُمْ وَمَا كَانُوا۟ يَعْبُدُونَ ﴿22﴾
جمع کرو ان لوگوں کو جنہوں نے ظلم کیا اور ان کے ساتھیوں کو اور جنہیں وہ پوجتے تھے۔
مِن دُونِ ٱللَّهِ فَٱهْدُوهُمْ إِلَىٰ صِرَٰطِ ٱلْجَحِيمِ ﴿23﴾
اللہ کے سوا، پھر انہیں جہنم کے راستے کی طرف لے چلو۔
وَقِفُوهُمْ إِنَّهُم مَّسْـُٔولُونَ ﴿24﴾
اور انہیں روک لو، بے شک ان سے سوال کیا جائے گا۔
مَا لَكُمْ لَا تَنَاصَرُونَ ﴿25﴾
تمہیں کیا ہوا کہ تم ایک دوسرے کی مدد نہیں کرتے؟
بَلْ هُمُ ٱلْيَوْمَ مُسْتَسْلِمُونَ ﴿26﴾
بلکہ آج وہ سب مطیع ہو گئے ہیں۔
وَأَقْبَلَ بَعْضُهُمْ عَلَىٰ بَعْضٖ يَتَسَآءَلُونَ ﴿27﴾
اور وہ ایک دوسرے کی طرف متوجہ ہو کر سوال کریں گے۔
قَالُوٓا۟ إِنَّكُمْ كُنتُمْ تَأْتُونَنَا عَنِ ٱلْيَمِينِ ﴿28﴾
وہ کہیں گے: تم تو ہمیں دائیں طرف سے آ کر بہکاتے تھے۔
قَالُوا۟ بَل لَّمْ تَكُونُوا۟ مُؤْمِنِينَ ﴿29﴾
وہ کہیں گے: بلکہ تم خود ایمان لانے والے نہ تھے۔
وَمَا كَانَ لَنَا عَلَيْكُم مِّن سُلْطَٰنِۢ بَلْ كُنتُمْ قَوْمٗا طَاغِينَ ﴿30﴾
اور ہمارا تم پر کوئی زور نہ تھا، بلکہ تم خود سرکش قوم تھے۔
فَحَقَّ عَلَيْنَا قَوْلُ رَبِّنَآ إِنَّا لَذَآئِقُونَ ﴿31﴾
تو ہمارے رب کا قول ہم پر ثابت ہو گیا، بے شک ہم ضرور عذاب چکھنے والے ہیں۔
فَأَغْوَيْنَـٰكُمْ إِنَّا كُنَّا غَـٰوِينَ ﴿32﴾
تو ہم نے تمہیں گمراہ کیا، بے شک ہم خود گمراہ تھے۔
فَإِنَّهُمْ يَوْمَ ٱلْقِيَـٰمَةِ فِي ٱلْعَذَابِ مُشْتَرِكُونَ ﴿33﴾
پس وہ قیامت کے دن عذاب میں شریک ہوں گے۔
إِنَّا كَذَٰلِكَ نَفْعَلُ بِٱلْمُجْرِمِينَ ﴿34﴾
بے شک ہم مجرموں کے ساتھ اسی طرح کرتے ہیں۔
إِنَّهُمْ كَانُوٓا۟ إِذَا قِيلَ لَهُمْ لَآ إِلَـٰهَ إِلَّا ٱللَّهُ يَسْتَكْبِرُونَ ﴿35﴾
بے شک جب ان سے کہا جاتا: اللہ کے سوا کوئی معبود نہیں، تو وہ تکبر کرتے۔
وَيَقُولُونَ أَءِنَّا لَتَارِكُوا۟ ءَالِهَتِنَا لِشَاعِرٖ مَّجْنُونِۭ ﴿36﴾
اور کہتے: کیا ہم اپنے معبودوں کو ایک دیوانے شاعر کے لیے چھوڑ دیں؟
بَلْ جَآءَ بِٱلْحَقِّ وَصَدَّقَ ٱلْمُرْسَلِينَ ﴿37﴾
بلکہ وہ حق لے کر آیا اور رسولوں کی تصدیق کی۔
إِنَّكُمْ لَذَآئِقُوا۟ ٱلْعَذَابِ ٱلْأَلِيمِ ﴿38﴾
بے شک تم دردناک عذاب چکھنے والے ہو۔
وَمَا تُجْزَوْنَ إِلَّا مَا كُنتُمْ تَعْمَلُونَ ﴿39﴾
اور تمہیں وہی بدلہ دیا جائے گا جو تم عمل کرتے تھے۔
إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿40﴾
سوائے اللہ کے چنے ہوئے بندوں کے۔
أُو۟لَـٰٓئِكَ لَهُمْ رِزْقٞ مَّعْلُومٞ ﴿41﴾
یہی لوگ ہیں جن کے لیے مقررہ رزق ہے۔
فَوَٰكِهُۥۤۖ وَهُم مُّكْرَمُونَ ﴿42﴾
پھل ہوں گے، اور وہ عزت دیے جائیں گے۔
فِي جَنَّـٰتِ ٱلنَّعِيمِ ﴿43﴾
نعمتوں والی جنتوں میں۔
عَلَىٰ سُرُرٖ مُّتَقَـٰبِلِينَ ﴿44﴾
تختوں پر آمنے سامنے بیٹھے ہوں گے۔
يُطَافُ عَلَيْهِم بِكَأْسٖ مِّن مَّعِينِ ﴿45﴾
ان پر صاف بہنے والی شراب کے جام گردش کریں گے۔
بَيْضَآءَ لَذَّةٖ لِّلشَّـٰرِبِينَ ﴿46﴾
سفید رنگ کی، پینے والوں کے لیے لذت بخش۔
لَا فِيهَا غَوْلٞ وَلَا هُمْ عَنْهَا يُنزَفُونَ ﴿47﴾
نہ اس میں کوئی نقصان ہے اور نہ وہ اس سے بے ہوش ہوں گے۔
وَعِندَهُمْ قَـٰصِرَٰتُ ٱلطَّرْفِ عِينٞ ﴿48﴾
اور ان کے پاس نظریں جھکائے ہوئے بڑی آنکھوں والی (بیویاں) ہوں گی۔
كَأَنَّهُنَّ بَيْضٞ مَّكْنُونٞ ﴿49﴾
گویا وہ چھپائے ہوئے انڈے ہیں۔
فَأَقْبَلَ بَعْضُهُمْ عَلَىٰ بَعْضٖ يَتَسَآءَلُونَ ﴿50﴾
تو وہ ایک دوسرے کی طرف متوجہ ہو کر سوال کریں گے۔
قَالَ قَآئِلٞ مِّنْهُمْ إِنِّي كَانَ لِي قَرِينٞ ﴿51﴾
ان میں سے ایک کہنے والا کہے گا: میرا ایک ساتھی تھا۔
يَقُولُ أَءِنَّكَ لَمِنَ ٱلْمُصَدِّقِينَ ﴿52﴾
کہتا تھا: کیا تو واقعی ان لوگوں میں سے ہے جو تصدیق کرتے ہیں؟
أَءِذَا مِتْنَا وَكُنَّا تُرَٰبٗا وَعِظَـٰمًا أَءِنَّا لَمَدِينُونَ ﴿53﴾
کیا جب ہم مر جائیں گے اور مٹی اور ہڈیاں ہو جائیں گے، تو ہم ضرور بدلہ دیے جائیں گے؟
قَالَ هَلْ أَنتُم مُّطَّلِعُونَ ﴿54﴾
وہ کہے گا: کیا تم جھانک کر دیکھنا چاہتے ہو؟
فَٱطَّلَعَ فَرَءَاهُ فِي سَوَآءِ ٱلْجَحِيمِ ﴿55﴾
تو وہ جھانکے گا اور اسے جہنم کے بیچ میں دیکھے گا۔
قَالَ تَٱللَّهِ إِن كِدتَّ لَتُرْدِينِ ﴿56﴾
کہے گا: اللہ کی قسم! تو تو مجھے بھی ہلاک کرنے والا تھا۔
وَلَوْلَا نِعْمَةُ رَبِّي لَكُنتُ مِنَ ٱلْمُحْضَرِينَ ﴿57﴾
اور اگر میرے رب کا فضل نہ ہوتا تو میں بھی حاضر کیے جانے والوں میں سے ہوتا۔
أَفَمَا نَحْنُ بِمَيِّتِينَ ﴿58﴾
تو کیا ہم اب مرنے والے نہیں؟
إِلَّا مَوْتَتَنَا ٱلْأُولَىٰ وَمَا نَحْنُ بِمُعَذَّبِينَ ﴿59﴾
سوائے اپنی پہلی موت کے، اور ہم عذاب نہیں دیے جائیں گے۔
إِنَّ هَـٰذَا لَهُوَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿60﴾
بے شک یہی بڑی کامیابی ہے۔
لِمِثْلِ هَـٰذَا فَلْيَعْمَلِ ٱلْعَـٰمِلُونَ ﴿61﴾
ایسی کامیابی کے لیے عمل کرنے والوں کو عمل کرنا چاہیے۔
أَذَٰلِكَ خَيْرٞ نُّزُلًا أَمْ شَجَرَةُ ٱلزَّقُّومِ ﴿62﴾
کیا یہ بہتر مہمانی ہے یا زقوم کا درخت؟
إِنَّا جَعَلْنَـٰهَا فِتْنَةٗ لِّلظَّـٰلِمِينَ ﴿63﴾
ہم نے اسے ظالموں کے لیے آزمائش بنایا ہے۔
إِنَّهَا شَجَرَةٞ تَخْرُجُ فِيٓ أَصْلِ ٱلْجَحِيمِ ﴿64﴾
بے شک وہ ایک درخت ہے جو جہنم کی جڑ سے نکلتا ہے۔
طَلْعُهَا كَأَنَّهُۥ رُءُوسُ ٱلشَّيَـٰطِينِ ﴿65﴾
اس کے پھل گویا شیطانوں کے سر ہیں۔
فَإِنَّهُمْ لَءَاكِلُونَ مِنْهَا فَمَالِـُٔونَ مِنْهَا ٱلْبُطُونَ ﴿66﴾
پس وہ اس میں سے کھائیں گے اور اپنے پیٹ بھر لیں گے۔
ثُمَّ إِنَّ لَهُمْ عَلَيْهَا لَشَوْبٗا مِّنْ حَمِيمٖ ﴿67﴾
پھر ان کے لیے اس پر کھولتا ہوا پانی ملایا جائے گا۔
ثُمَّ إِنَّ مَرْجِعَهُمْ لَإِلَى ٱلْجَحِيمِ ﴿68﴾
پھر ان کی واپسی جہنم ہی کی طرف ہوگی۔
إِنَّهُمْ أَلْفَوْا۟ ءَابَآءَهُمْ ضَآلِّينَ ﴿69﴾
بے شک انہوں نے اپنے باپ دادا کو گمراہ پایا۔
فَهُمْ عَلَىٰٓ ءَاثَـٰرِهِمْ يُهْرَعُونَ ﴿70﴾
تو وہ انہی کے نقش قدم پر دوڑتے رہے۔
وَلَقَدْ ضَلَّ قَبْلَهُمْ أَكْثَرُ ٱلْأَوَّلِينَ ﴿71﴾
اور ان سے پہلے اکثر لوگ گمراہ ہو چکے ہیں۔
وَلَقَدْ أَرْسَلْنَا فِيهِم مُّنذِرِينَ ﴿72﴾
اور ہم نے ان میں ڈرانے والے بھیجے۔
فَٱنظُرْ كَيْفَ كَانَ عَـٰقِبَةُ ٱلْمُنذَرِينَ ﴿73﴾
تو دیکھ لو کہ ڈرائے جانے والوں کا انجام کیسا ہوا۔
إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿74﴾
سوائے اللہ کے چنے ہوئے بندوں کے۔
وَلَقَدْ نَادَىٰنَا نُوحٞ فَلَنِعْمَ ٱلْمُجِيبُونَ ﴿75﴾
اور بے شک نوح نے ہمیں پکارا، تو ہم بہترین جواب دینے والے ہیں۔
وَنَجَّيْنَـٰهُ وَأَهْلَهُۥ مِنَ ٱلْكَرْبِ ٱلْعَظِيمِ ﴿76﴾
اور ہم نے اسے اور اس کے گھر والوں کو بڑی مصیبت سے نجات دی۔
وَجَعَلْنَا ذُرِّيَّتَهُۥ هُمُ ٱلْبَاقِينَ ﴿77﴾
اور ہم نے اس کی اولاد کو باقی رہنے والوں میں کر دیا۔
وَتَرَكْنَا عَلَيْهِ فِي ٱلْـَٔاخِرِينَ ﴿78﴾
اور ہم نے اس کا ذکر بعد والوں میں باقی رکھا۔
سَلَـٰمٌ عَلَىٰ نُوحٖ فِي ٱلْعَـٰلَمِينَ ﴿79﴾
سلام ہو نوح پر تمام جہان والوں میں۔
إِنَّا كَذَٰلِكَ نَجْزِي ٱلْمُحْسِنِينَ ﴿80﴾
بے شک ہم نیکی کرنے والوں کو اسی طرح بدلہ دیتے ہیں۔
إِنَّهُۥ مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿81﴾
بے شک وہ ہمارے مؤمن بندوں میں سے تھا۔
ثُمَّ أَغْرَقْنَا ٱلْـَٔاخَرِينَ ﴿82﴾
پھر ہم نے باقی سب کو غرق کر دیا۔
وَإِنَّ مِن شِيعَتِهِۦ لَإِبْرَٰهِيمَ ﴿83﴾
اور بے شک اس کے گروہ میں ابراہیم بھی تھا۔
إِذْ جَآءَ رَبَّهُۥ بِقَلْبٖ سَلِيمٍ ﴿84﴾
جب وہ اپنے رب کے پاس پاک دل لے کر آیا۔
إِذْ قَالَ لِأَبِيهِ وَقَوْمِهِۦ مَاذَا تَعْبُدُونَ ﴿85﴾
جب اس نے اپنے باپ اور قوم سے کہا: تم کیا عبادت کرتے ہو؟
أَئِفْكًا ءَالِهَةٗ دُونَ ٱللَّهِ تُرِيدُونَ ﴿86﴾
کیا تم جھوٹ گھڑ کر اللہ کے سوا دوسرے معبود چاہتے ہو؟
فَمَا ظَنُّكُم بِرَبِّ ٱلْعَـٰلَمِينَ ﴿87﴾
تو تم رب العالمین کے بارے میں کیا گمان رکھتے ہو؟
فَنَظَرَ نَظْرَةٗ فِي ٱلنُّجُومِ ﴿88﴾
پھر اس نے ستاروں کی طرف ایک نظر ڈالی۔
فَقَالَ إِنِّي سَقِيمٞ ﴿89﴾
اور کہا: میں بیمار ہوں۔
فَتَوَلَّوْا۟ عَنْهُ مُدْبِرِينَ ﴿90﴾
تو وہ اس سے منہ موڑ کر چلے گئے۔
فَرَاغَ إِلَىٰٓ ءَالِهَتِهِمْ فَقَالَ أَلَا تَأْكُلُونَ ﴿91﴾
پھر وہ ان کے معبودوں کی طرف چپکے سے گیا اور کہا: کیا تم کھاتے نہیں؟
فَرَاغَ إِلَيْهِمْ ضَرْبٗا بِٱلْيَمِينِ ﴿92﴾
پھر اس نے ان (بتوں) کو دائیں ہاتھ سے مارنا شروع کیا۔
فَأَقْبَلُوٓا۟ إِلَيْهِ يَزِفُّونَ ﴿93﴾
تو وہ اس کی طرف دوڑتے ہوئے آئے۔
قَالَ أَتَعْبُدُونَ مَا تَنْحِتُونَ ﴿94﴾
اس نے کہا: کیا تم ان چیزوں کی عبادت کرتے ہو جنہیں تم خود تراشتے ہو؟
وَٱللَّهُ خَلَقَكُمْ وَمَا تَعْمَلُونَ ﴿95﴾
حالانکہ اللہ نے تمہیں اور تمہارے اعمال کو پیدا کیا ہے۔
قَالُوا۟ ٱبْنُوا۟ لَهُۥ بُنْيَـٰنٗا فَأَلْقُوهُ فِي ٱلْجَحِيمِ ﴿96﴾
انہوں نے کہا: اس کے لیے ایک عمارت بناؤ اور اسے دہکتی آگ میں ڈال دو۔
فَأَرَادُوا۟ بِهِۦ كَيْدٗا فَجَعَلْنَـٰهُمُ ٱلْأَسْفَلِينَ ﴿97﴾
انہوں نے اس کے ساتھ چالاکی کرنا چاہی، تو ہم نے انہیں نیچا دکھا دیا۔
وَقَالَ إِنِّي ذَاهِبٌ إِلَىٰ رَبِّي سَيَهْدِينِ ﴿98﴾
اور اس نے کہا: میں اپنے رب کی طرف جا رہا ہوں، وہ ضرور میری رہنمائی کرے گا۔
رَبِّ هَبْ لِي مِنَ ٱلصَّـٰلِحِينَ ﴿99﴾
اے میرے رب! مجھے نیک لوگوں میں سے (اولاد) عطا فرما۔
فَبَشَّرْنَـٰهُ بِغُلَـٰمٍ حَلِيمٖ ﴿100﴾
تو ہم نے اسے ایک بردبار لڑکے کی خوشخبری دی۔
فَلَمَّا بَلَغَ مَعَهُ ٱلسَّعْيَ قَالَ يَـٰبُنَيَّ إِنِّيٓ أَرَىٰ فِي ٱلْمَنَامِ أَنِّيٓ أَذْبَحُكَ فَٱنظُرْ مَاذَا تَرَىٰۚ قَالَ يَـٰٓأَبَتِ ٱفْعَلْ مَا تُؤْمَرُۖ سَتَجِدُنِيٓ إِن شَآءَ ٱللَّهُ مِنَ ٱلصَّـٰبِرِينَ ﴿101﴾
پھر جب وہ اس کے ساتھ دوڑ دھوپ کرنے کی عمر کو پہنچا، تو اس نے کہا: اے میرے بیٹے! میں خواب میں دیکھتا ہوں کہ میں تجھے ذبح کر رہا ہوں، تو دیکھ تو کیا رائے رکھتا ہے؟ اس نے کہا: اے ابا جان! جو حکم دیا گیا ہے وہی کیجیے، ان شاء اللہ آپ مجھے صبر کرنے والوں میں پائیں گے۔
فَلَمَّا بَلَغَ مَعَهُ ٱلسَّعْيَ قَالَ يَـٰبُنَيَّ إِنِّيٓ أَرَىٰ فِي ٱلْمَنَامِ أَنِّيٓ أَذْبَحُكَ فَٱنظُرْ مَاذَا تَرَىٰۚ قَالَ يَـٰٓأَبَتِ ٱفْعَلْ مَا تُؤْمَرُۖ سَتَجِدُنِيٓ إِن شَآءَ ٱللَّهُ مِنَ ٱلصَّـٰبِرِينَ ﴿102﴾
پھر جب وہ اس کے ساتھ دوڑنے (کی عمر) کو پہنچا تو ابراہیم نے کہا: اے میرے بیٹے! میں خواب میں دیکھتا ہوں کہ میں تجھے ذبح کر رہا ہوں، تو بتا تیرا کیا خیال ہے؟ اس نے کہا: اے ابا جان! جو حکم دیا گیا ہے وہی کیجیے، ان شاء اللہ آپ مجھے صبر کرنے والوں میں پائیں گے۔
فَلَمَّآ أَسْلَمَا وَتَلَّهُۥ لِلْجَبِينِ ﴿103﴾
پھر جب دونوں نے فرمانبرداری کی اور اس نے اسے ماتھے کے بل لٹا دیا۔
وَنَـٰدَيْنَـٰهُ أَن يَـٰٓإِبْرَٰهِيمُ ﴿104﴾
اور ہم نے اسے پکارا: اے ابراہیم۔
قَدْ صَدَّقْتَ ٱلرُّءْيَآۚ إِنَّا كَذَٰلِكَ نَجْزِي ٱلْمُحْسِنِينَ ﴿105﴾
تو نے خواب کو سچ کر دکھایا، بے شک ہم نیکی کرنے والوں کو اسی طرح جزا دیتے ہیں۔
إِنَّ هَـٰذَا لَهُوَ ٱلْبَلَـٰٓؤُا۟ ٱلْمُبِينُ ﴿106﴾
بے شک یہ کھلی آزمائش تھی۔
وَفَدَيْنَـٰهُ بِذِبْحٍ عَظِيمٖ ﴿107﴾
اور ہم نے اسے ایک عظیم قربانی سے بدل دیا۔
وَتَرَكْنَا عَلَيْهِ فِي ٱلْـَٔاخِرِينَ ﴿108﴾
اور ہم نے اس کا ذکر بعد والوں میں باقی رکھا۔
سَلَـٰمٌ عَلَىٰٓ إِبْرَٰهِيمَ ﴿109﴾
سلام ہو ابراہیم پر۔
كَذَٰلِكَ نَجْزِي ٱلْمُحْسِنِينَ ﴿110﴾
ہم نیکی کرنے والوں کو اسی طرح جزا دیتے ہیں۔
إِنَّهُۥ مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿111﴾
بے شک وہ ہمارے مؤمن بندوں میں سے تھا۔
وَبَشَّرْنَـٰهُ بِإِسْحَـٰقَ نَبِيّٗا مِّنَ ٱلصَّـٰلِحِينَ ﴿112﴾
اور ہم نے اسے اسحاق کی خوشخبری دی، جو نیک لوگوں میں سے نبی تھا۔
وَبَـٰرَكْنَا عَلَيْهِ وَعَلَىٰٓ إِسْحَـٰقَۚ وَمِن ذُرِّيَّتِهِمَا مُحْسِنٞ وَظَالِمٞ لِّنَفْسِهِۦ مُبِينٞ ﴿113﴾
اور ہم نے اس پر اور اسحاق پر برکت دی، اور ان دونوں کی اولاد میں کچھ نیک اور کچھ اپنے نفس پر کھلا ظلم کرنے والے ہیں۔
وَلَقَدْ مَنَنَّا عَلَىٰ مُوسَىٰ وَهَـٰرُونَ ﴿114﴾
اور بے شک ہم نے موسیٰ اور ہارون پر احسان کیا۔
وَنَجَّيْنَـٰهُمَا وَقَوْمَهُمَا مِنَ ٱلْكَرْبِ ٱلْعَظِيمِ ﴿115﴾
اور ہم نے ان دونوں کو اور ان کی قوم کو بڑی مصیبت سے نجات دی۔
وَنَصَرْنَـٰهُمْ فَكَانُوا۟ هُمُ ٱلْغَـٰلِبِينَ ﴿116﴾
اور ہم نے ان کی مدد کی، تو وہی غالب رہے۔
وَءَاتَيْنَـٰهُمَا ٱلْكِتَـٰبَ ٱلْمُسْتَبِينَ ﴿117﴾
اور ہم نے ان دونوں کو واضح کتاب عطا کی۔
وَهَدَيْنَـٰهُمَا ٱلصِّرَٰطَ ٱلْمُسْتَقِيمَ ﴿118﴾
اور ہم نے ان دونوں کو سیدھا راستہ دکھایا۔
وَتَرَكْنَا عَلَيْهِمَا فِي ٱلْـَٔاخِرِينَ ﴿119﴾
اور ہم نے ان دونوں کا ذکر بعد والوں میں باقی رکھا۔
سَلَـٰمٌ عَلَىٰ مُوسَىٰ وَهَـٰرُونَ ﴿120﴾
سلام ہو موسیٰ اور ہارون پر۔
كَذَٰلِكَ نَجْزِي ٱلْمُحْسِنِينَ ﴿121﴾
ہم نیکی کرنے والوں کو اسی طرح جزا دیتے ہیں۔
إِنَّهُمَا مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿122﴾
بے شک وہ دونوں ہمارے مؤمن بندوں میں سے تھے۔
وَإِنَّ إِلْيَاسَ لَمِنَ ٱلْمُرْسَلِينَ ﴿123﴾
اور بے شک الیاس رسولوں میں سے تھا۔
إِذْ قَالَ لِقَوْمِهِۦٓ أَلَا تَتَّقُونَ ﴿124﴾
جب اس نے اپنی قوم سے کہا: کیا تم تقویٰ نہیں اختیار کرتے؟
أَتَدْعُونَ بَعْلٗا وَتَذَرُونَ أَحْسَنَ ٱلْخَـٰلِقِينَ ﴿125﴾
کیا تم بعل کو پکارتے ہو اور سب سے بہتر خالق کو چھوڑ دیتے ہو؟
ٱللَّهَ رَبَّكُمْ وَرَبَّ ءَابَآئِكُمُ ٱلْأَوَّلِينَ ﴿126﴾
اللہ کو، جو تمہارا رب ہے اور تمہارے پہلے باپ دادا کا رب ہے۔
فَكَذَّبُوهُ فَإِنَّهُمْ لَمُحْضَرُونَ ﴿127﴾
تو انہوں نے اسے جھٹلایا، پس وہ ضرور حاضر کیے جانے والے ہیں۔
إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿128﴾
سوائے اللہ کے چنے ہوئے بندوں کے۔
وَتَرَكْنَا عَلَيْهِ فِي ٱلْـَٔاخِرِينَ ﴿129﴾
اور ہم نے اس کا ذکر بعد والوں میں باقی رکھا۔
سَلَـٰمٌ عَلَىٰٓ إِلْيَاسِينَ ﴿130﴾
سلام ہو الیاس پر۔
كَذَٰلِكَ نَجْزِي ٱلْمُحْسِنِينَ ﴿131﴾
ہم نیکی کرنے والوں کو اسی طرح جزا دیتے ہیں۔
إِنَّهُۥ مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿132﴾
بے شک وہ ہمارے مؤمن بندوں میں سے تھا۔
وَإِنَّ لُوطٗا لَّمِنَ ٱلْمُرْسَلِينَ ﴿133﴾
اور بے شک لوط رسولوں میں سے تھا۔
إِذْ نَجَّيْنَـٰهُ وَأَهْلَهُۥٓ أَجْمَعِينَ ﴿134﴾
جب ہم نے اسے اور اس کے سب گھر والوں کو نجات دی۔
إِلَّا عَجُوزٗا فِي ٱلْغَٰبِرِينَ ﴿135﴾
سوائے ایک بوڑھی عورت کے جو پیچھے رہ جانے والوں میں تھی۔
ثُمَّ دَمَّرْنَا ٱلْـَٔاخَرِينَ ﴿136﴾
پھر ہم نے باقی سب کو تباہ کر دیا۔
وَإِنَّكُمْ لَتَمُرُّونَ عَلَيْهِم مُّصْبِحِينَ ﴿137﴾
اور بے شک تم صبح کے وقت ان پر سے گزرتے ہو۔
وَبِٱلَّيْلِۚ أَفَلَا تَعْقِلُونَ ﴿138﴾
اور رات کے وقت بھی، تو کیا تم عقل نہیں رکھتے؟
وَإِنَّ يُونُسَ لَمِنَ ٱلْمُرْسَلِينَ ﴿139﴾
اور بے شک یونس رسولوں میں سے تھا۔
إِذْ أَبَقَ إِلَى ٱلْفُلْكِ ٱلْمَشْحُونِ ﴿140﴾
جب وہ بھاگ کر بھری ہوئی کشتی کی طرف گیا۔
فَسَاهَمَ فَكَانَ مِنَ ٱلْمُدْحَضِينَ ﴿141﴾
تو اس نے قرعہ اندازی کی، اور وہ ہارنے والوں میں سے ہو گیا۔
فَٱلْتَقَمَهُ ٱلْحُوتُ وَهُوَ مُلِيمٞ ﴿142﴾
تو مچھلی نے اسے نگل لیا، اور وہ قابلِ ملامت تھا۔
فَلَوْلَآ أَنَّهُۥ كَانَ مِنَ ٱلْمُسَبِّحِينَ ﴿143﴾
اگر وہ تسبیح کرنے والوں میں سے نہ ہوتا۔
لَلَبِثَ فِي بَطْنِهِۦٓ إِلَىٰ يَوْمِ يُبْعَثُونَ ﴿144﴾
تو وہ اس کے پیٹ میں قیامت کے دن تک رہتا۔
فَنَبَذْنَـٰهُ بِٱلْعَرَآءِ وَهُوَ سَقِيمٞ ﴿145﴾
پھر ہم نے اسے چٹیل میدان میں پھینک دیا، اور وہ بیمار تھا۔
وَأَنۢبَتْنَا عَلَيْهِ شَجَرَةٗ مِّن يَقْطِينٖ ﴿146﴾
اور ہم نے اس پر کدو کا درخت اگا دیا۔
وَأَرْسَلْنَـٰهُ إِلَىٰ مِاْئَةِ أَلْفٍ أَوْ يَزِيدُونَ ﴿147﴾
اور ہم نے اسے ایک لاکھ یا اس سے زیادہ لوگوں کی طرف بھیجا۔
فَـَٔامَنُوا۟ فَمَتَّعْنَـٰهُمْ إِلَىٰ حِينٖ ﴿148﴾
تو وہ ایمان لے آئے، اور ہم نے انہیں ایک وقت تک فائدہ پہنچایا۔
فَٱسْتَفْتِهِمْ أَلِرَبِّكَ ٱلْبَنَاتُ وَلَهُمُ ٱلْبَنُونَ ﴿149﴾
تو ان سے پوچھو: کیا تمہارے رب کے لیے بیٹیاں ہیں اور ان کے لیے بیٹے؟
أَمْ خَلَقْنَا ٱلْمَلَـٰٓئِكَةَ إِنَـٰثٗا وَهُمْ شَـٰهِدُونَ ﴿150﴾
کیا ہم نے فرشتوں کو عورتیں بنایا ہے اور وہ دیکھ رہے تھے؟
أَلَآ إِنَّهُم مِّنْ إِفْكِهِمْ لَيَقُولُونَ ﴿151﴾
خبردار! وہ اپنے جھوٹ میں یہ کہتے ہیں۔
وَلَدَ ٱللَّهُ وَإِنَّهُمْ لَكَـٰذِبُونَ ﴿152﴾
کہ اللہ نے اولاد پیدا کی، حالانکہ وہ جھوٹے ہیں۔
أَصْطَفَى ٱلْبَنَاتِ عَلَى ٱلْبَنِينَ ﴿153﴾
کیا اس نے بیٹوں پر بیٹیاں پسند کیں؟
مَا لَكُمْ كَيْفَ تَحْكُمُونَ ﴿154﴾
تمہیں کیا ہوا ہے؟ کیسے فیصلے کرتے ہو؟
أَفَلَا تَذَكَّرُونَ ﴿155﴾
کیا تم نصیحت حاصل نہیں کرتے؟
أَمْ لَكُمْ سُلْطَـٰنٞ مُّبِينٞ ﴿156﴾
یا تمہارے پاس کوئی واضح دلیل ہے؟
فَأْتُوا۟ بِكِتَـٰبِكُمْ إِن كُنتُمْ صَـٰدِقِينَ ﴿157﴾
تو اپنی کتاب لاؤ اگر تم سچے ہو۔
وَجَعَلُوا۟ بَيْنَهُۥ وَبَيْنَ ٱلْجِنَّةِ نَسَبٗاۚ وَلَقَدْ عَلِمَتِ ٱلْجِنَّةُ إِنَّهُمْ لَمُحْضَرُونَ ﴿158﴾
اور انہوں نے اللہ اور جنات کے درمیان نسب بنا لیا، حالانکہ جنات جانتے ہیں کہ وہ ضرور حاضر کیے جائیں گے۔
سُبْحَـٰنَ ٱللَّهِ عَمَّا يَصِفُونَ ﴿159﴾
اللہ پاک ہے ان باتوں سے جو وہ بیان کرتے ہیں۔
إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿160﴾
سوائے اللہ کے چنے ہوئے بندوں کے۔
فَإِنَّكُمْ وَمَا تَعْبُدُونَ ﴿161﴾
تو بے شک تم اور جنہیں تم پوجتے ہو۔
مَآ أَنتُمْ عَلَيْهِ بِفَـٰتِنِينَ ﴿162﴾
تم ان پر قابو نہیں پا سکتے۔
إِلَّا مَنْ هُوَ صَالِ ٱلْجَحِيمِ ﴿163﴾
سوائے اس کے جو جہنم میں جانے والا ہے۔
وَمَا مِنَّآ إِلَّا لَهُۥ مَقَامٞ مَّعْلُومٞ ﴿164﴾
اور ہم میں سے ہر ایک کا ایک مقرر مقام ہے۔
وَإِنَّا لَنَحْنُ ٱلصَّآفُّونَ ﴿165﴾
اور بے شک ہم ہی صف باندھنے والے ہیں۔
وَإِنَّا لَنَحْنُ ٱلْمُسَبِّحُونَ ﴿166﴾
اور بے شک ہم ہی تسبیح کرنے والے ہیں۔
وَإِن كَانُوا۟ لَيَقُولُونَ ﴿167﴾
اور بے شک وہ کہا کرتے تھے۔
لَوْ أَنَّ عِندَنَا ذِكْرٗا مِّنَ ٱلْأَوَّلِينَ ﴿168﴾
کاش ہمارے پاس پہلے لوگوں کی کوئی نصیحت ہوتی۔
لَكُنَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿169﴾
تو ہم اللہ کے چنے ہوئے بندے ہوتے۔
فَكَفَرُوا۟ بِهِۦ فَسَوْفَ يَعْلَمُونَ ﴿170﴾
تو انہوں نے اس کا انکار کیا، پس عنقریب وہ جان لیں گے۔
وَلَقَدۡ سَبَقَتۡ كَلِمَتُنَا لِعِبَادِنَا ٱلۡمُرۡسَلِينَ ﴿171﴾
اور بے شک ہمارے رسول بندوں کے حق میں ہمارا وعدہ پہلے ہی ہو چکا ہے۔
إِنَّهُمۡ لَهُمُ ٱلۡمَنصُورُونَ ﴿172﴾
کہ بے شک وہی مدد دیے جائیں گے۔
وَإِنَّ جُندَنَا لَهُمُ ٱلۡغَـٰلِبُونَ ﴿173﴾
اور بے شک ہمارا لشکر ہی غالب رہے گا۔
فَتَوَلَّ عَنۡهُمۡ حَتَّىٰ حِينٖ ﴿174﴾
تو ان سے ایک وقت تک منہ پھیر لے۔
وَأَبۡصِرۡهُمۡ فَسَوۡفَ يُبۡصِرُونَ ﴿175﴾
اور انہیں دیکھتا رہ، عنقریب وہ بھی دیکھ لیں گے۔
أَفَبِعَذَابِنَا يَسۡتَعۡجِلُونَ ﴿176﴾
کیا وہ ہمارے عذاب کو جلدی چاہتے ہیں؟
فَإِذَا نَزَلَ بِسَاحَتِهِمۡ فَسَآءَ صَبَاحُ ٱلۡمُنذَرِينَ ﴿177﴾
پھر جب وہ ان کے صحن میں اترے گا تو ڈرائے جانے والوں کی صبح بری ہوگی۔
وَتَوَلَّ عَنۡهُمۡ حَتَّىٰ حِينٖ ﴿178﴾
اور ان سے ایک وقت تک منہ پھیر لے۔
وَأَبۡصِرۡ فَسَوۡفَ يُبۡصِرُونَ ﴿179﴾
اور دیکھتا رہ، عنقریب وہ بھی دیکھ لیں گے۔
سُبۡحَـٰنَ رَبِّكَ رَبِّ ٱلۡعِزَّةِ عَمَّا يَصِفُونَ ﴿180﴾
پاک ہے تیرے رب، عزت والے رب، ان باتوں سے جو وہ بیان کرتے ہیں۔
وَسَلَـٰمٌ عَلَى ٱلۡمُرۡسَلِينَ ﴿181﴾
اور سلام ہو رسولوں پر۔
وَٱلۡحَمۡدُ لِلَّهِ رَبِّ ٱلۡعَـٰلَمِينَ ﴿182﴾
اور تمام تعریف اللہ کے لیے ہے جو رب ہے تمام جہانوں کا۔
