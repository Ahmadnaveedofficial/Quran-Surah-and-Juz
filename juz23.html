23 پارہ


بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
وَمَا لِيَ لَآ أَعْبُدُ ٱلَّذِي فَطَرَنِي وَإِلَيْهِ تُرْجَعُونَ ﴿22﴾
اور مجھے کیا ہے کہ میں اس کی عبادت نہ کروں جس نے مجھے پیدا کیا، اور تم سب اسی کی طرف لوٹائے جاؤ گے؟
ءَأَتَّخِذُ مِن دُونِهِۦٓ ءَالِهَةً إِن يُرِدْنِ ٱلرَّحْمَـٰنُ بِضُرّٖ لَّا تُغْنِ عَنِّي شَفَـٰعَتُهُمْ شَيْـٔٗا وَلَا يُنقِذُونِ ﴿23﴾
کیا میں اس کے سوا دوسرے معبود بنا لوں؟ اگر رحمٰن مجھے کوئی نقصان پہنچانا چاہے تو ان کی سفارش مجھے کچھ فائدہ نہ دے سکے گی اور نہ ہی وہ مجھے بچا سکیں گے۔
إِنِّيٓ إِذٗا لَّفِي ضَلَـٰلٖ مُّبِينٍ ﴿24﴾
بے شک میں تو اس صورت میں کھلی گمراہی میں ہوں گا۔
إِنِّيٓ ءَامَنتُ بِرَبِّكُمْ فَٱسْمَعُونِ ﴿25﴾
بے شک میں تمہارے رب پر ایمان لایا، پس میری بات سنو۔
قِيلَ ٱدْخُلِ ٱلْجَنَّةَۚ قَالَ يَـٰلَيْتَ قَوْمِي يَعْلَمُونَ ﴿26﴾
کہا گیا: جنت میں داخل ہو جا، اس نے کہا: کاش میری قوم جانتی۔
بِمَا غَفَرَ لِي رَبِّي وَجَعَلَنِي مِنَ ٱلْمُكْرَمِينَ ﴿27﴾
کہ میرے رب نے مجھے بخش دیا اور مجھے عزت والوں میں شامل کیا۔
وَمَآ أَنزَلْنَا عَلَىٰ قَوْمِهِۦ مِنۢ بَعْدِهِۦ مِن جُندٖ مِّنَ ٱلسَّمَآءِ وَمَا كُنَّا مُنزِلِينَ ﴿28﴾
اور ہم نے اس کے بعد اس کی قوم پر آسمان سے کوئی لشکر نہیں اتارا، اور نہ ہم اتارنے والے تھے۔
إِن كَانَتْ إِلَّا صَيْحَةٗ وَٰحِدَةٗ فَإِذَا هُمۡ خَـٰمِدُونَ ﴿29﴾
بس ایک ہی چیخ تھی، پھر وہ سب بجھ کر رہ گئے۔
يَـٰحَسْرَةً عَلَى ٱلْعِبَادِۚ مَا يَأْتِيهِم مِّن رَّسُولٍ إِلَّا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ ﴿30﴾
افسوس ہے بندوں پر! ان کے پاس جو بھی رسول آتا ہے، وہ اس کا مذاق ہی اڑاتے ہیں۔
أَلَمْ يَرَوْا۟ كَمْ أَهْلَكْنَا قَبْلَهُم مِّنَ ٱلْقُرُونِ أَنَّهُمْ إِلَيْهِمْ لَا يَرْجِعُونَ ﴿31﴾
کیا انہوں نے نہیں دیکھا کہ ہم نے ان سے پہلے کتنی قومیں ہلاک کیں؟ وہ ان کی طرف لوٹ کر نہیں آئیں۔
وَإِن كُلّٞ لَّمَّا جَمِيعٌ لَّدَيْنَا مُحْضَرُونَ ﴿32﴾
اور سب کے سب ہمارے حضور حاضر کیے جائیں گے۔
وَءَايَةٞ لَّهُمُ ٱلْأَرْضُ ٱلْمَيْتَةُ أَحْيَيْنَـٰهَا وَأَخْرَجْنَا مِنْهَا حَبّٗا فَمِنْهُ يَأْكُلُونَ ﴿33﴾
اور ان کے لیے ایک نشانی مردہ زمین ہے، ہم نے اسے زندہ کیا اور اس سے دانہ نکالا، جسے وہ کھاتے ہیں۔
وَجَعَلْنَا فِيهَا جَنَّـٰتٖ مِّن نَّخِيلٖ وَأَعْنَـٰبٖ وَفَجَّرْنَا فِيهَا مِنَ ٱلْعُيُونِ ﴿34﴾
اور ہم نے اس میں کھجوروں اور انگوروں کے باغ بنائے، اور اس میں چشمے جاری کیے۔
لِيَأْكُلُوا۟ مِن ثَمَرِهِۦ وَمَا عَمِلَتْهُ أَيْدِيهِمۡۖ أَفَلَا يَشْكُرُونَ ﴿35﴾
تاکہ وہ اس کے پھلوں سے کھائیں، اور یہ ان کے ہاتھوں کا بنایا ہوا نہیں، تو کیا وہ شکر ادا نہیں کرتے؟
سُبْحَـٰنَ ٱلَّذِي خَلَقَ ٱلْأَزْوَٰجَ كُلَّهَا مِمَّا تُنۢبِتُ ٱلْأَرْضُ وَمِنْ أَنفُسِهِمْ وَمِمَّا لَا يَعْلَمُونَ ﴿36﴾
پاک ہے وہ ذات جس نے تمام جوڑے پیدا کیے، ان چیزوں میں سے جو زمین اگاتی ہے، اور خود ان کی ذاتوں میں سے، اور ان چیزوں میں سے جنہیں وہ نہیں جانتے۔
وَءَايَةٞ لَّهُمُ ٱلَّيْلُ نَسْلَخُ مِنْهُ ٱلنَّهَارَ فَإِذَا هُم مُّظْلِمُونَ ﴿37﴾
اور ان کے لیے ایک نشانی رات ہے، ہم اس سے دن کو کھینچ لیتے ہیں، تو وہ اندھیرے میں رہ جاتے ہیں۔
وَٱلشَّمْسُ تَجْرِي لِمُسْتَقَرّٖ لَّهَاۚ ذَٰلِكَ تَقْدِيرُ ٱلْعَزِيزِ ٱلْعَلِيمِ ﴿38﴾
اور سورج اپنے مقررہ ٹھکانے کی طرف چلتا ہے، یہ غالب، علم والے کا اندازہ ہے۔
وَٱلْقَمَرَ قَدَّرْنَـٰهُ مَنَازِلَ حَتَّىٰ عَادَ كَٱلْعُرْجُونِ ٱلْقَدِيمِ ﴿39﴾
اور ہم نے چاند کے لیے منزلیں مقرر کیں، یہاں تک کہ وہ پرانی کھجور کی شاخ کی طرح ہو جاتا ہے۔
لَا ٱلشَّمْسُ يَنۢبَغِي لَهَآ أَن تُدْرِكَ ٱلْقَمَرَ وَلَا ٱلَّيْلُ سَابِقُ ٱلنَّهَارِۚ وَكُلّٞ فِي فَلَكٖ يَسْبَحُونَ ﴿40﴾
نہ سورج کو یہ زیب دیتا ہے کہ وہ چاند کو جا پکڑے اور نہ رات دن سے آگے نکل سکتی ہے، اور سب ایک ایک مدار میں تیر رہے ہیں۔
وَءَايَةٞ لَّهُمْ أَنَّا حَمَلْنَا ذُرِّيَّتَهُمْ فِي ٱلْفُلْكِ ٱلْمَشْحُونِ ﴿41﴾
اور ان کے لیے ایک نشانی یہ ہے کہ ہم نے ان کی اولاد کو بھری ہوئی کشتی میں سوار کیا۔
وَخَلَقْنَا لَهُم مِّن مِّثْلِهِۦ مَا يَرْكَبُونَ ﴿42﴾
اور ہم نے ان کے لیے اس جیسی اور چیزیں پیدا کیں جن پر وہ سوار ہوتے ہیں۔
وَإِن نَّشَأْ نُغْرِقْهُمْ فَلَا صَرِيخَ لَهُمْ وَلَا هُمْ يُنقَذُونَ ﴿43﴾
اور اگر ہم چاہیں تو انہیں غرق کر دیں، پھر نہ کوئی فریاد رسی ہوگی اور نہ ہی وہ بچائے جائیں گے۔
إِلَّا رَحْمَةٗ مِّنَّا وَمَتَـٰعًا إِلَىٰ حِينٖ ﴿44﴾
مگر ہماری طرف سے رحمت اور ایک وقت تک فائدہ اٹھانا۔
وَإِذَا قِيلَ لَهُمُ ٱتَّقُوا۟ مَا بَيْنَ أَيْدِيكُمْ وَمَا خَلْفَكُمْ لَعَلَّكُمْ تُرْحَمُونَ ﴿45﴾
اور جب ان سے کہا جاتا ہے: اپنے آگے اور پیچھے (کے انجام) سے ڈرو تاکہ تم پر رحم کیا جائے۔
وَمَا تَأْتِيهِم مِّنْ ءَايَةٖ مِّنْ ءَايَـٰتِ رَبِّهِمْ إِلَّا كَانُوا۟ عَنْهَا مُعْرِضِينَ ﴿46﴾
اور ان کے پاس ان کے رب کی نشانیوں میں سے کوئی نشانی نہیں آتی مگر وہ اس سے منہ موڑ لیتے ہیں۔
وَإِذَا قِيلَ لَهُمْ أَنفِقُوا۟ مِمَّا رَزَقَكُمُ ٱللَّهُ قَالَ ٱلَّذِينَ كَفَرُوا۟ لِلَّذِينَ ءَامَنُوٓا۟ أَنُطْعِمُ مَن لَّوْ يَشَآءُ ٱللَّهُ أَطْعَمَهُۥٓ إِنْ أَنتُمْ إِلَّا فِي ضَلَـٰلٖ مُّبِينٖ ﴿47﴾
اور جب ان سے کہا جاتا ہے: اس میں سے خرچ کرو جو اللہ نے تمہیں دیا ہے، تو کافر ایمان والوں سے کہتے ہیں: کیا ہم اسے کھلائیں جسے اللہ چاہتا تو خود کھلا دیتا؟ تم تو کھلی گمراہی میں ہو۔
وَيَقُولُونَ مَتَىٰ هَـٰذَا ٱلْوَعْدُ إِن كُنتُمْ صَـٰدِقِينَ ﴿48﴾
اور وہ کہتے ہیں: یہ وعدہ کب پورا ہوگا اگر تم سچے ہو؟
مَا يَنظُرُونَ إِلَّا صَيْحَةٗ وَٰحِدَةٗ تَأْخُذُهُمْ وَهُمْ يَخِصِّمُونَ ﴿49﴾
وہ بس ایک ہی چیخ کا انتظار کر رہے ہیں جو انہیں آ لے گی، اور وہ جھگڑ رہے ہوں گے۔
فَلَا يَسْتَطِيعُونَ تَوْصِيَةٗ وَلَآ إِلَىٰٓ أَهْلِهِمْ يَرْجِعُونَ ﴿50﴾
پھر نہ وہ وصیت کر سکیں گے اور نہ ہی اپنے گھر والوں کی طرف لوٹ سکیں گے۔
وَنُفِخَ فِي ٱلصُّورِ فَإِذَا هُم مِّنَ ٱلْأَجْدَاثِ إِلَىٰ رَبِّهِمْ يَنسِلُونَ ﴿51﴾
اور صور پھونکا جائے گا، تو وہ قبروں سے نکل کر اپنے رب کی طرف دوڑیں گے۔
قَالُوا۟ يَـٰوَيْلَنَا مَنۢ بَعَثَنَا مِن مَّرْقَدِنَاۜ هَـٰذَا مَا وَعَدَ ٱلرَّحْمَـٰنُ وَصَدَقَ ٱلْمُرْسَلُونَ ﴿52﴾
وہ کہیں گے: ہائے ہماری شامت! کس نے ہمیں ہماری خواب گاہ سے اٹھا دیا؟ یہ وہی ہے جس کا رحمٰن نے وعدہ کیا تھا، اور رسولوں نے سچ کہا تھا۔
إِن كَانَتْ إِلَّا صَيْحَةٗ وَٰحِدَةٗ فَإِذَا هُمْ جَمِيعٞ لَّدَيْنَا مُحْضَرُونَ ﴿53﴾
بس ایک ہی چیخ ہوگی، پھر وہ سب ہمارے سامنے حاضر ہوں گے۔
فَٱلْيَوْمَ لَا تُظْلَمُ نَفْسٞ شَيْـٔٗا وَلَا تُجْزَوْنَ إِلَّا مَا كُنتُمْ تَعْمَلُونَ ﴿54﴾
پس آج کسی جان پر ظلم نہیں کیا جائے گا، اور تمہیں صرف وہی بدلہ دیا جائے گا جو تم کرتے رہے۔
إِنَّ أَصْحَـٰبَ ٱلْجَنَّةِ ٱلْيَوْمَ فِي شُغُلٖ فَـٰكِهُونَ ﴿55﴾
بے شک جنت والے آج خوشی کے مشغلے میں ہوں گے۔
هُمۡ وَأَزۡوَٰجُهُمۡ فِي ظِلَـٰلٍ عَلَى ٱلۡأَرَآئِكِ مُتَّكِـُٔونَ ﴿56﴾
وہ اور ان کی بیویاں سایوں میں تختوں پر تکیہ لگائے بیٹھے ہوں گے۔
لَهُمْ فِيهَا فَـٰكِهَةٞ وَلَهُم مَّا يَدَّعُونَ ﴿57﴾
ان کے لیے وہاں میوے ہوں گے، اور جو کچھ وہ مانگیں گے وہ ان کے لیے ہوگا۔
سَلَـٰمٞ قَوْلٞا مِّن رَّبّٖ رَّحِيمٖ ﴿58﴾
سلام ہو، یہ ربِ رحیم کی طرف سے فرمان ہوگا۔
وَٱمْتَـٰزُوا۟ ٱلۡيَوْمَ أَيُّهَا ٱلْمُجْرِمُونَ ﴿59﴾
اور (کہا جائے گا:) اے مجرمو! آج تم الگ ہو جاؤ۔
أَلَمْ أَعْهَدْ إِلَيْكُمْ يَـٰبَنِيٓ ءَادَمَ أَن لَّا تَعْبُدُوا۟ ٱلشَّيْطَـٰنَۖ إِنَّهُۥ لَكُمْ عَدُوّٞ مُّبِينٞ ﴿60﴾
اے اولادِ آدم! کیا میں نے تم سے یہ عہد نہیں لیا تھا کہ شیطان کی عبادت نہ کرنا، بے شک وہ تمہارا کھلا دشمن ہے؟
وَأَنِ ٱعْبُدُونِيۚ هَـٰذَا صِرَٰطٞ مُّسْتَقِيمٞ ﴿61﴾
اور یہ کہ میری عبادت کرو، یہی سیدھا راستہ ہے۔
وَلَقَدْ أَضَلَّ مِنكُمْ جِبِلّٗا كَثِيرًاۖ أَفَلَمْ تَكُونُوا۟ تَعْقِلُونَ ﴿62﴾
اور اس نے تم میں سے بہت سی جماعتوں کو گمراہ کر دیا، تو کیا تم عقل نہیں رکھتے تھے؟
هَـٰذِهِۦ جَهَنَّمُ ٱلَّتِي كُنتُمْ تُوعَدُونَ ﴿63﴾
یہی ہے جہنم جس کا تم سے وعدہ کیا جاتا تھا۔
ٱصْلَوْهَا ٱلۡيَوْمَ بِمَا كُنتُمْ تَكْفُرُونَ ﴿64﴾
آج اس میں داخل ہو جاؤ، اس کے بدلے جو تم کفر کرتے تھے۔
ٱلۡيَوْمَ نَخْتِمُ عَلَىٰٓ أَفْوَٰهِهِمْ وَتُكَلِّمُنَآ أَيْدِيهِمْ وَتَشْهَدُ أَرْجُلُهُم بِمَا كَانُوا۟ يَكْسِبُونَ ﴿65﴾
آج ہم ان کے منہ پر مہر لگا دیں گے، اور ان کے ہاتھ ہم سے بات کریں گے، اور ان کے پاؤں گواہی دیں گے جو وہ کرتے رہے۔
وَلَوْ نَشَآءُ لَطَمَسْنَا عَلَىٰٓ أَعْيُنِهِمْ فَٱسْتَبَقُوا۟ ٱلصِّرَٰطَ فَأَنَّىٰ يُبْصِرُونَ ﴿66﴾
اور اگر ہم چاہیں تو ان کی آنکھیں مٹا دیں، پھر وہ راستہ کی طرف دوڑیں، تو انہیں کہاں سے نظر آئے گا؟
وَلَوْ نَشَآءُ لَمَسَخْنَـٰهُمْ عَلَىٰ مَكَانَتِهِمْ فَمَا ٱسْتَطَـٰعُوا۟ مُضِيّٗا وَلَا يَرْجِعُونَ ﴿67﴾
اور اگر ہم چاہیں تو انہیں ان کی جگہ پر ہی مسخ کر دیں، پھر نہ وہ آگے جا سکیں اور نہ لوٹ سکیں۔
وَمَن نُّعَمِّرْهُ نُنَكِّسْهُ فِي ٱلْخَلْقِۚ أَفَلَا يَعْقِلُونَ ﴿68﴾
اور جسے ہم لمبی عمر دیتے ہیں، اسے ہم خلقت میں الٹا دیتے ہیں، تو کیا وہ عقل نہیں رکھتے؟
وَمَا عَلَّمْنَـٰهُ ٱلشِّعْرَ وَمَا يَنبَغِي لَهُۥٓۚ إِنْ هُوَ إِلَّا ذِكْرٞ وَقُرْءَانٞ مُّبِينٞ ﴿69﴾
اور ہم نے اسے شعر نہیں سکھایا اور نہ ہی یہ اس کے لائق ہے، یہ تو صرف نصیحت اور واضح قرآن ہے۔
لِّيُنذِرَ مَن كَانَ حَيّٗا وَيَحِقَّ ٱلْقَوْلُ عَلَى ٱلْكَـٰفِرِينَ ﴿70﴾
تاکہ وہ اسے خبردار کرے جو زندہ ہو، اور کافروں پر بات ثابت ہو جائے۔
أَوَلَمْ يَرَوْا۟ أَنَّا خَلَقْنَا لَهُم مِّمَّا عَمِلَتْ أَيْدِينَآ أَنْعَـٰمٗا فَهُمْ لَهَا مَـٰلِكُونَ ﴿71﴾
کیا انہوں نے نہیں دیکھا کہ ہم نے ان کے لیے اپنی ہاتھوں کی بنائی ہوئی چیزوں میں سے چوپائے پیدا کیے، پھر وہ ان کے مالک بن گئے؟
وَذَلَّلْنَـٰهَا لَهُمْ فَمِنْهَا رَكُوبُهُمْ وَمِنْهَا يَأْكُلُونَ ﴿72﴾
اور ہم نے انہیں ان کے تابع کر دیا، تو ان میں سے بعض پر وہ سوار ہوتے ہیں اور بعض کو کھاتے ہیں۔
وَلَهُمْ فِيهَا مَنَـٰفِعُ وَمَشَارِبُۖ أَفَلَا يَشْكُرُونَ ﴿73﴾
اور ان کے لیے ان میں فائدے اور مشروب ہیں، تو کیا وہ شکر ادا نہیں کرتے؟
وَٱتَّخَذُوا۟ مِن دُونِ ٱللَّهِ ءَالِهَةٗ لَّعَلَّهُمْ يُنصَرُونَ ﴿74﴾
اور انہوں نے اللہ کے سوا دوسرے معبود بنا لیے تاکہ وہ ان کی مدد کریں۔
لَا يَسْتَطِيعُونَ نَصْرَهُمْ وَهُمْ لَهُمْ جُندٞ مُّحْضَرُونَ ﴿75﴾
وہ ان کی مدد نہیں کر سکتے، اور وہ ان کے لیے حاضر کیے گئے لشکر ہیں۔
فَلَا يَحْزُنْكَ قَوْلُهُمْۘ إِنَّا نَعْلَمُ مَا يُسِرُّونَ وَمَا يُعْلِنُونَ ﴿76﴾
تو ان کی بات آپ کو غمگین نہ کرے، بے شک ہم جانتے ہیں جو وہ چھپاتے ہیں اور جو ظاہر کرتے ہیں۔
أَوَلَمْ يَرَ ٱلْإِنسَـٰنُ أَنَّا خَلَقْنَـٰهُ مِن نُّطْفَةٖ فَإِذَا هُوَ خَصِيمٞ مُّبِينٞ ﴿77﴾
کیا انسان نے نہیں دیکھا کہ ہم نے اسے نطفہ سے پیدا کیا، پھر وہ کھلا جھگڑالو بن گیا؟
وَضَرَبَ لَنَا مَثَلٗا وَنَسِيَ خَلْقَهُۥۖ قَالَ مَن يُحْيِ ٱلْعِظَـٰمَ وَهِيَ رَمِيمٞ ﴿78﴾
اور اس نے ہمارے لیے مثال بیان کی اور اپنی پیدائش کو بھول گیا، کہنے لگا: کون ان ہڈیوں کو زندہ کرے گا جب وہ بوسیدہ ہو چکی ہوں؟
قُلْ يُحْيِيهَا ٱلَّذِيٓ أَنشَأَهَآ أَوَّلَ مَرَّةٖۖ وَهُوَ بِكُلِّ خَلْقٍ عَلِيمٌ ﴿79﴾
کہہ دیجیے: انہیں وہی زندہ کرے گا جس نے انہیں پہلی بار پیدا کیا، اور وہ ہر قسم کی پیدائش کو خوب جاننے والا ہے۔
ٱلَّذِي جَعَلَ لَكُم مِّنَ ٱلشَّجَرِ ٱلْأَخْضَرِ نَارٗا فَإِذَآ أَنتُم مِّنْهُ تُوقِدُونَ ﴿80﴾
جس نے تمہارے لیے سبز درخت سے آگ پیدا کی، تو تم اس سے آگ سلگاتے ہو۔
أَوَلَيْسَ ٱلَّذِي خَلَقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ بِقَـٰدِرٍ عَلَىٰٓ أَن يَخْلُقَ مِثْلَهُمۚ بَلَىٰ وَهُوَ ٱلْخَلَّـٰقُ ٱلْعَلِيمُ ﴿81﴾
کیا وہ جس نے آسمانوں اور زمین کو پیدا کیا، اس پر قادر نہیں کہ ان جیسے پیدا کرے؟ کیوں نہیں! اور وہی سب کچھ پیدا کرنے والا، خوب جاننے والا ہے۔
إِنَّمَآ أَمْرُهُۥٓ إِذَآ أَرَادَ شَيْـًٔا أَن يَقُولَ لَهُۥ كُن فَيَكُونُ ﴿82﴾
اس کا حکم تو بس یہ ہے کہ جب وہ کسی چیز کا ارادہ کرتا ہے تو کہتا ہے: "ہو جا" تو وہ ہو جاتی ہے۔
فَسُبْحَـٰنَ ٱلَّذِي بِيَدِهِۦ مَلَكُوتُ كُلِّ شَيْءٖ وَإِلَيْهِ تُرْجَعُونَ ﴿83﴾
پس پاک ہے وہ ذات جس کے ہاتھ میں ہر چیز کی بادشاہی ہے، اور تم سب اسی کی طرف لوٹائے جاؤ گے۔
سورة الصافات
بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
وَٱلصَّـٰفَّـٰتِ صَفّٗا ﴿1﴾
قسم ہے ان (فرشتوں) کی جو صف باندھ کر کھڑے ہوتے ہیں۔
فَٱلزَّٰجِرَٰتِ زَجْرٗا ﴿2﴾
پھر ان کی جو جھڑک کر چلاتی ہیں۔
فَٱلتَّٰلِيَٰتِ ذِكْرًا ﴿3﴾
پھر ان کی جو ذکر پڑھتی ہیں۔
إِنَّ إِلَـٰهَكُمْ لَوَٰحِدٞ ﴿4﴾
بے شک تمہارا معبود ایک ہی ہے۔
رَّبُّ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَا وَرَبُّ ٱلْمَشَـٰرِقِ ﴿5﴾
آسمانوں اور زمین اور جو کچھ ان کے درمیان ہے سب کا رب، اور مشرقوں کا رب۔
إِنَّا زَيَّنَّا ٱلسَّمَآءَ ٱلدُّنْيَا بِزِينَةٍ ٱلْكَوَاكِبِ ﴿6﴾
بے شک ہم نے آسمانِ دنیا کو ستاروں کی زینت سے آراستہ کیا۔
وَحِفْظٗا مِّن كُلِّ شَيْطَٰنٖ مَّارِدٖ ﴿7﴾
اور ہر سرکش شیطان سے حفاظت کے لیے۔
لَا يَسَّمَّعُونَ إِلَى ٱلْمَلَإِ ٱلْأَعْلَىٰ وَيُقْذَفُونَ مِن كُلِّ جَانِبٖ ﴿8﴾
وہ بلند جماعت (فرشتوں) کی باتیں نہیں سن سکتے، اور ہر طرف سے ان پر پھینکا جاتا ہے۔
دُحُورٗاۖ وَلَهُمْ عَذَابٞ وَاصِبٌ ﴿9﴾
دھتکارے ہوئے، اور ان کے لیے مسلسل عذاب ہے۔
إِلَّا مَنْ خَطِفَ ٱلْخَطْفَةَ فَأَتْبَعَهُۥ شِهَابٞ ثَاقِبٞ ﴿10﴾
سوائے اس کے جو چوری سے کچھ لے اُڑے، تو اس کے پیچھے چمکتا ہوا شعلہ لگا دیا جاتا ہے۔
فَٱسْتَفْتِهِمْ أَهُمْ أَشَدُّ خَلْقًا أَم مَّنْ خَلَقْنَآۚ إِنَّا خَلَقْنَـٰهُم مِّن طِينٖ لَّازِبٖ ﴿11﴾
تو ان سے پوچھو: کیا ان کا بنانا زیادہ مشکل ہے یا وہ جنہیں ہم نے پیدا کیا؟ ہم نے انہیں چپکنے والی مٹی سے پیدا کیا۔
بَلْ عَجِبْتَ وَيَسْخَرُونَ ﴿12﴾
بلکہ تم تعجب کرتے ہو اور وہ مذاق اڑاتے ہیں۔
وَإِذَا ذُكِّرُوا۟ لَا يَذْكُرُونَ ﴿13﴾
اور جب انہیں نصیحت کی جاتی ہے تو وہ نصیحت نہیں پکڑتے۔
وَإِذَا رَأَوْا۟ ءَايَةٗ يَسْتَسْخِرُونَ ﴿14﴾
اور جب کوئی نشانی دیکھتے ہیں تو مذاق بناتے ہیں۔
وَقَالُوٓا۟ إِنْ هَـٰذَآ إِلَّا سِحْرٞ مُّبِينٌ ﴿15﴾
اور کہتے ہیں: یہ تو کھلا جادو ہے۔
أَءِذَا مِتْنَا وَكُنَّا تُرَٰبٗا وَعِظَـٰمًا أَءِنَّا لَمَبْعُوثُونَ ﴿16﴾
کیا جب ہم مر جائیں گے اور مٹی اور ہڈیاں ہو جائیں گے، کیا ہم پھر اٹھائے جائیں گے؟
أَوَ ءَابَآؤُنَا ٱلْأَوَّلُونَ ﴿17﴾
یا ہمارے پہلے باپ دادا بھی؟
قُلْ نَعَمْ وَأَنتُمْ دَٰخِرُونَ ﴿18﴾
کہہ دو: ہاں! اور تم ذلیل ہو کر اٹھائے جاؤ گے۔
فَإِنَّمَا هِيَ زَجْرَةٞ وَٰحِدَةٞ فَإِذَا هُمۡ يَنظُرُونَ ﴿19﴾
بس وہ ایک ہی چیخ ہوگی، پھر وہ سب دیکھنے لگیں گے۔
وَقَالُوا۟ يَـٰوَيْلَنَا هَـٰذَا يَوْمُ ٱلدِّينِ ﴿20﴾
اور وہ کہیں گے: ہائے ہماری شامت! یہ تو حساب کا دن ہے۔
هَٰذَا يَوْمُ ٱلْفَصْلِ ٱلَّذِي كُنتُم بِهِۦ تُكَذِّبُونَ ﴿21﴾
یہی ہے فیصلے کا دن جسے تم جھٹلایا کرتے تھے۔
ٱحْشُرُوا۟ ٱلَّذِينَ ظَلَمُوا۟ وَأَزْوَٰجَهُمْ وَمَا كَانُوا۟ يَعْبُدُونَ ﴿22﴾
جمع کرو ان لوگوں کو جنہوں نے ظلم کیا اور ان کے ساتھیوں کو اور جنہیں وہ پوجتے تھے۔
مِن دُونِ ٱللَّهِ فَٱهْدُوهُمْ إِلَىٰ صِرَٰطِ ٱلْجَحِيمِ ﴿23﴾
اللہ کے سوا، پھر انہیں جہنم کے راستے کی طرف لے چلو۔
وَقِفُوهُمْ إِنَّهُم مَّسْـُٔولُونَ ﴿24﴾
اور انہیں روک لو، بے شک ان سے سوال کیا جائے گا۔
مَا لَكُمْ لَا تَنَاصَرُونَ ﴿25﴾
تمہیں کیا ہوا کہ تم ایک دوسرے کی مدد نہیں کرتے؟
بَلْ هُمُ ٱلْيَوْمَ مُسْتَسْلِمُونَ ﴿26﴾
بلکہ آج وہ سب مطیع ہو گئے ہیں۔
وَأَقْبَلَ بَعْضُهُمْ عَلَىٰ بَعْضٖ يَتَسَآءَلُونَ ﴿27﴾
اور وہ ایک دوسرے کی طرف متوجہ ہو کر سوال کریں گے۔
قَالُوٓا۟ إِنَّكُمْ كُنتُمْ تَأْتُونَنَا عَنِ ٱلْيَمِينِ ﴿28﴾
وہ کہیں گے: تم تو ہمیں دائیں طرف سے آ کر بہکاتے تھے۔
قَالُوا۟ بَل لَّمْ تَكُونُوا۟ مُؤْمِنِينَ ﴿29﴾
وہ کہیں گے: بلکہ تم خود ایمان لانے والے نہ تھے۔
وَمَا كَانَ لَنَا عَلَيْكُم مِّن سُلْطَٰنِۢ بَلْ كُنتُمْ قَوْمٗا طَاغِينَ ﴿30﴾
اور ہمارا تم پر کوئی زور نہ تھا، بلکہ تم خود سرکش قوم تھے۔
فَحَقَّ عَلَيْنَا قَوْلُ رَبِّنَآ إِنَّا لَذَآئِقُونَ ﴿31﴾
تو ہمارے رب کا قول ہم پر ثابت ہو گیا، بے شک ہم ضرور عذاب چکھنے والے ہیں۔
فَأَغْوَيْنَـٰكُمْ إِنَّا كُنَّا غَـٰوِينَ ﴿32﴾
تو ہم نے تمہیں گمراہ کیا، بے شک ہم خود گمراہ تھے۔
فَإِنَّهُمْ يَوْمَ ٱلْقِيَـٰمَةِ فِي ٱلْعَذَابِ مُشْتَرِكُونَ ﴿33﴾
پس وہ قیامت کے دن عذاب میں شریک ہوں گے۔
إِنَّا كَذَٰلِكَ نَفْعَلُ بِٱلْمُجْرِمِينَ ﴿34﴾
بے شک ہم مجرموں کے ساتھ اسی طرح کرتے ہیں۔
إِنَّهُمْ كَانُوٓا۟ إِذَا قِيلَ لَهُمْ لَآ إِلَـٰهَ إِلَّا ٱللَّهُ يَسْتَكْبِرُونَ ﴿35﴾
بے شک جب ان سے کہا جاتا: اللہ کے سوا کوئی معبود نہیں، تو وہ تکبر کرتے۔
وَيَقُولُونَ أَءِنَّا لَتَارِكُوا۟ ءَالِهَتِنَا لِشَاعِرٖ مَّجْنُونِۭ ﴿36﴾
اور کہتے: کیا ہم اپنے معبودوں کو ایک دیوانے شاعر کے لیے چھوڑ دیں؟
بَلْ جَآءَ بِٱلْحَقِّ وَصَدَّقَ ٱلْمُرْسَلِينَ ﴿37﴾
بلکہ وہ حق لے کر آیا اور رسولوں کی تصدیق کی۔
إِنَّكُمْ لَذَآئِقُوا۟ ٱلْعَذَابِ ٱلْأَلِيمِ ﴿38﴾
بے شک تم دردناک عذاب چکھنے والے ہو۔
وَمَا تُجْزَوْنَ إِلَّا مَا كُنتُمْ تَعْمَلُونَ ﴿39﴾
اور تمہیں وہی بدلہ دیا جائے گا جو تم عمل کرتے تھے۔
إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿40﴾
سوائے اللہ کے چنے ہوئے بندوں کے۔
أُو۟لَـٰٓئِكَ لَهُمْ رِزْقٞ مَّعْلُومٞ ﴿41﴾
یہی لوگ ہیں جن کے لیے مقررہ رزق ہے۔
فَوَٰكِهُۥۤۖ وَهُم مُّكْرَمُونَ ﴿42﴾
پھل ہوں گے، اور وہ عزت دیے جائیں گے۔
فِي جَنَّـٰتِ ٱلنَّعِيمِ ﴿43﴾
نعمتوں والی جنتوں میں۔
عَلَىٰ سُرُرٖ مُّتَقَـٰبِلِينَ ﴿44﴾
تختوں پر آمنے سامنے بیٹھے ہوں گے۔
يُطَافُ عَلَيْهِم بِكَأْسٖ مِّن مَّعِينِ ﴿45﴾
ان پر صاف بہنے والی شراب کے جام گردش کریں گے۔
بَيْضَآءَ لَذَّةٖ لِّلشَّـٰرِبِينَ ﴿46﴾
سفید رنگ کی، پینے والوں کے لیے لذت بخش۔
لَا فِيهَا غَوْلٞ وَلَا هُمْ عَنْهَا يُنزَفُونَ ﴿47﴾
نہ اس میں کوئی نقصان ہے اور نہ وہ اس سے بے ہوش ہوں گے۔
وَعِندَهُمْ قَـٰصِرَٰتُ ٱلطَّرْفِ عِينٞ ﴿48﴾
اور ان کے پاس نظریں جھکائے ہوئے بڑی آنکھوں والی (بیویاں) ہوں گی۔
كَأَنَّهُنَّ بَيْضٞ مَّكْنُونٞ ﴿49﴾
گویا وہ چھپائے ہوئے انڈے ہیں۔
فَأَقْبَلَ بَعْضُهُمْ عَلَىٰ بَعْضٖ يَتَسَآءَلُونَ ﴿50﴾
تو وہ ایک دوسرے کی طرف متوجہ ہو کر سوال کریں گے۔
قَالَ قَآئِلٞ مِّنْهُمْ إِنِّي كَانَ لِي قَرِينٞ ﴿51﴾
ان میں سے ایک کہنے والا کہے گا: میرا ایک ساتھی تھا۔
يَقُولُ أَءِنَّكَ لَمِنَ ٱلْمُصَدِّقِينَ ﴿52﴾
کہتا تھا: کیا تو واقعی ان لوگوں میں سے ہے جو تصدیق کرتے ہیں؟
أَءِذَا مِتْنَا وَكُنَّا تُرَٰبٗا وَعِظَـٰمًا أَءِنَّا لَمَدِينُونَ ﴿53﴾
کیا جب ہم مر جائیں گے اور مٹی اور ہڈیاں ہو جائیں گے، تو ہم ضرور بدلہ دیے جائیں گے؟
قَالَ هَلْ أَنتُم مُّطَّلِعُونَ ﴿54﴾
وہ کہے گا: کیا تم جھانک کر دیکھنا چاہتے ہو؟
فَٱطَّلَعَ فَرَءَاهُ فِي سَوَآءِ ٱلْجَحِيمِ ﴿55﴾
تو وہ جھانکے گا اور اسے جہنم کے بیچ میں دیکھے گا۔
قَالَ تَٱللَّهِ إِن كِدتَّ لَتُرْدِينِ ﴿56﴾
کہے گا: اللہ کی قسم! تو تو مجھے بھی ہلاک کرنے والا تھا۔
وَلَوْلَا نِعْمَةُ رَبِّي لَكُنتُ مِنَ ٱلْمُحْضَرِينَ ﴿57﴾
اور اگر میرے رب کا فضل نہ ہوتا تو میں بھی حاضر کیے جانے والوں میں سے ہوتا۔
أَفَمَا نَحْنُ بِمَيِّتِينَ ﴿58﴾
تو کیا ہم اب مرنے والے نہیں؟
إِلَّا مَوْتَتَنَا ٱلْأُولَىٰ وَمَا نَحْنُ بِمُعَذَّبِينَ ﴿59﴾
سوائے اپنی پہلی موت کے، اور ہم عذاب نہیں دیے جائیں گے۔
إِنَّ هَـٰذَا لَهُوَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿60﴾
بے شک یہی بڑی کامیابی ہے۔
لِمِثْلِ هَـٰذَا فَلْيَعْمَلِ ٱلْعَـٰمِلُونَ ﴿61﴾
ایسی کامیابی کے لیے عمل کرنے والوں کو عمل کرنا چاہیے۔
أَذَٰلِكَ خَيْرٞ نُّزُلًا أَمْ شَجَرَةُ ٱلزَّقُّومِ ﴿62﴾
کیا یہ بہتر مہمانی ہے یا زقوم کا درخت؟
إِنَّا جَعَلْنَـٰهَا فِتْنَةٗ لِّلظَّـٰلِمِينَ ﴿63﴾
ہم نے اسے ظالموں کے لیے آزمائش بنایا ہے۔
إِنَّهَا شَجَرَةٞ تَخْرُجُ فِيٓ أَصْلِ ٱلْجَحِيمِ ﴿64﴾
بے شک وہ ایک درخت ہے جو جہنم کی جڑ سے نکلتا ہے۔
طَلْعُهَا كَأَنَّهُۥ رُءُوسُ ٱلشَّيَـٰطِينِ ﴿65﴾
اس کے پھل گویا شیطانوں کے سر ہیں۔
فَإِنَّهُمْ لَءَاكِلُونَ مِنْهَا فَمَالِـُٔونَ مِنْهَا ٱلْبُطُونَ ﴿66﴾
پس وہ اس میں سے کھائیں گے اور اپنے پیٹ بھر لیں گے۔
ثُمَّ إِنَّ لَهُمْ عَلَيْهَا لَشَوْبٗا مِّنْ حَمِيمٖ ﴿67﴾
پھر ان کے لیے اس پر کھولتا ہوا پانی ملایا جائے گا۔
ثُمَّ إِنَّ مَرْجِعَهُمْ لَإِلَى ٱلْجَحِيمِ ﴿68﴾
پھر ان کی واپسی جہنم ہی کی طرف ہوگی۔
إِنَّهُمْ أَلْفَوْا۟ ءَابَآءَهُمْ ضَآلِّينَ ﴿69﴾
بے شک انہوں نے اپنے باپ دادا کو گمراہ پایا۔
فَهُمْ عَلَىٰٓ ءَاثَـٰرِهِمْ يُهْرَعُونَ ﴿70﴾
تو وہ انہی کے نقش قدم پر دوڑتے رہے۔
وَلَقَدْ ضَلَّ قَبْلَهُمْ أَكْثَرُ ٱلْأَوَّلِينَ ﴿71﴾
اور ان سے پہلے اکثر لوگ گمراہ ہو چکے ہیں۔
وَلَقَدْ أَرْسَلْنَا فِيهِم مُّنذِرِينَ ﴿72﴾
اور ہم نے ان میں ڈرانے والے بھیجے۔
فَٱنظُرْ كَيْفَ كَانَ عَـٰقِبَةُ ٱلْمُنذَرِينَ ﴿73﴾
تو دیکھ لو کہ ڈرائے جانے والوں کا انجام کیسا ہوا۔
إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿74﴾
سوائے اللہ کے چنے ہوئے بندوں کے۔
وَلَقَدْ نَادَىٰنَا نُوحٞ فَلَنِعْمَ ٱلْمُجِيبُونَ ﴿75﴾
اور بے شک نوح نے ہمیں پکارا، تو ہم بہترین جواب دینے والے ہیں۔
وَنَجَّيْنَـٰهُ وَأَهْلَهُۥ مِنَ ٱلْكَرْبِ ٱلْعَظِيمِ ﴿76﴾
اور ہم نے اسے اور اس کے گھر والوں کو بڑی مصیبت سے نجات دی۔
وَجَعَلْنَا ذُرِّيَّتَهُۥ هُمُ ٱلْبَاقِينَ ﴿77﴾
اور ہم نے اس کی اولاد کو باقی رہنے والوں میں کر دیا۔
وَتَرَكْنَا عَلَيْهِ فِي ٱلْـَٔاخِرِينَ ﴿78﴾
اور ہم نے اس کا ذکر بعد والوں میں باقی رکھا۔
سَلَـٰمٌ عَلَىٰ نُوحٖ فِي ٱلْعَـٰلَمِينَ ﴿79﴾
سلام ہو نوح پر تمام جہان والوں میں۔
إِنَّا كَذَٰلِكَ نَجْزِي ٱلْمُحْسِنِينَ ﴿80﴾
بے شک ہم نیکی کرنے والوں کو اسی طرح بدلہ دیتے ہیں۔
إِنَّهُۥ مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿81﴾
بے شک وہ ہمارے مؤمن بندوں میں سے تھا۔
ثُمَّ أَغْرَقْنَا ٱلْـَٔاخَرِينَ ﴿82﴾
پھر ہم نے باقی سب کو غرق کر دیا۔
وَإِنَّ مِن شِيعَتِهِۦ لَإِبْرَٰهِيمَ ﴿83﴾
اور بے شک اس کے گروہ میں ابراہیم بھی تھا۔
إِذْ جَآءَ رَبَّهُۥ بِقَلْبٖ سَلِيمٍ ﴿84﴾
جب وہ اپنے رب کے پاس پاک دل لے کر آیا۔
إِذْ قَالَ لِأَبِيهِ وَقَوْمِهِۦ مَاذَا تَعْبُدُونَ ﴿85﴾
جب اس نے اپنے باپ اور قوم سے کہا: تم کیا عبادت کرتے ہو؟
أَئِفْكًا ءَالِهَةٗ دُونَ ٱللَّهِ تُرِيدُونَ ﴿86﴾
کیا تم جھوٹ گھڑ کر اللہ کے سوا دوسرے معبود چاہتے ہو؟
فَمَا ظَنُّكُم بِرَبِّ ٱلْعَـٰلَمِينَ ﴿87﴾
تو تم رب العالمین کے بارے میں کیا گمان رکھتے ہو؟
فَنَظَرَ نَظْرَةٗ فِي ٱلنُّجُومِ ﴿88﴾
پھر اس نے ستاروں کی طرف ایک نظر ڈالی۔
فَقَالَ إِنِّي سَقِيمٞ ﴿89﴾
اور کہا: میں بیمار ہوں۔
فَتَوَلَّوْا۟ عَنْهُ مُدْبِرِينَ ﴿90﴾
تو وہ اس سے منہ موڑ کر چلے گئے۔
فَرَاغَ إِلَىٰٓ ءَالِهَتِهِمْ فَقَالَ أَلَا تَأْكُلُونَ ﴿91﴾
پھر وہ ان کے معبودوں کی طرف چپکے سے گیا اور کہا: کیا تم کھاتے نہیں؟
فَرَاغَ إِلَيْهِمْ ضَرْبٗا بِٱلْيَمِينِ ﴿92﴾
پھر اس نے ان (بتوں) کو دائیں ہاتھ سے مارنا شروع کیا۔
فَأَقْبَلُوٓا۟ إِلَيْهِ يَزِفُّونَ ﴿93﴾
تو وہ اس کی طرف دوڑتے ہوئے آئے۔
قَالَ أَتَعْبُدُونَ مَا تَنْحِتُونَ ﴿94﴾
اس نے کہا: کیا تم ان چیزوں کی عبادت کرتے ہو جنہیں تم خود تراشتے ہو؟
وَٱللَّهُ خَلَقَكُمْ وَمَا تَعْمَلُونَ ﴿95﴾
حالانکہ اللہ نے تمہیں اور تمہارے اعمال کو پیدا کیا ہے۔
قَالُوا۟ ٱبْنُوا۟ لَهُۥ بُنْيَـٰنٗا فَأَلْقُوهُ فِي ٱلْجَحِيمِ ﴿96﴾
انہوں نے کہا: اس کے لیے ایک عمارت بناؤ اور اسے دہکتی آگ میں ڈال دو۔
فَأَرَادُوا۟ بِهِۦ كَيْدٗا فَجَعَلْنَـٰهُمُ ٱلْأَسْفَلِينَ ﴿97﴾
انہوں نے اس کے ساتھ چالاکی کرنا چاہی، تو ہم نے انہیں نیچا دکھا دیا۔
وَقَالَ إِنِّي ذَاهِبٌ إِلَىٰ رَبِّي سَيَهْدِينِ ﴿98﴾
اور اس نے کہا: میں اپنے رب کی طرف جا رہا ہوں، وہ ضرور میری رہنمائی کرے گا۔
رَبِّ هَبْ لِي مِنَ ٱلصَّـٰلِحِينَ ﴿99﴾
اے میرے رب! مجھے نیک لوگوں میں سے (اولاد) عطا فرما۔
فَبَشَّرْنَـٰهُ بِغُلَـٰمٍ حَلِيمٖ ﴿100﴾
تو ہم نے اسے ایک بردبار لڑکے کی خوشخبری دی۔
فَلَمَّا بَلَغَ مَعَهُ ٱلسَّعْيَ قَالَ يَـٰبُنَيَّ إِنِّيٓ أَرَىٰ فِي ٱلْمَنَامِ أَنِّيٓ أَذْبَحُكَ فَٱنظُرْ مَاذَا تَرَىٰۚ قَالَ يَـٰٓأَبَتِ ٱفْعَلْ مَا تُؤْمَرُۖ سَتَجِدُنِيٓ إِن شَآءَ ٱللَّهُ مِنَ ٱلصَّـٰبِرِينَ ﴿101﴾
پھر جب وہ اس کے ساتھ دوڑ دھوپ کرنے کی عمر کو پہنچا، تو اس نے کہا: اے میرے بیٹے! میں خواب میں دیکھتا ہوں کہ میں تجھے ذبح کر رہا ہوں، تو دیکھ تو کیا رائے رکھتا ہے؟ اس نے کہا: اے ابا جان! جو حکم دیا گیا ہے وہی کیجیے، ان شاء اللہ آپ مجھے صبر کرنے والوں میں پائیں گے۔
فَلَمَّا بَلَغَ مَعَهُ ٱلسَّعْيَ قَالَ يَـٰبُنَيَّ إِنِّيٓ أَرَىٰ فِي ٱلْمَنَامِ أَنِّيٓ أَذْبَحُكَ فَٱنظُرْ مَاذَا تَرَىٰۚ قَالَ يَـٰٓأَبَتِ ٱفْعَلْ مَا تُؤْمَرُۖ سَتَجِدُنِيٓ إِن شَآءَ ٱللَّهُ مِنَ ٱلصَّـٰبِرِينَ ﴿102﴾
پھر جب وہ اس کے ساتھ دوڑنے (کی عمر) کو پہنچا تو ابراہیم نے کہا: اے میرے بیٹے! میں خواب میں دیکھتا ہوں کہ میں تجھے ذبح کر رہا ہوں، تو بتا تیرا کیا خیال ہے؟ اس نے کہا: اے ابا جان! جو حکم دیا گیا ہے وہی کیجیے، ان شاء اللہ آپ مجھے صبر کرنے والوں میں پائیں گے۔
فَلَمَّآ أَسْلَمَا وَتَلَّهُۥ لِلْجَبِينِ ﴿103﴾
پھر جب دونوں نے فرمانبرداری کی اور اس نے اسے ماتھے کے بل لٹا دیا۔
وَنَـٰدَيْنَـٰهُ أَن يَـٰٓإِبْرَٰهِيمُ ﴿104﴾
اور ہم نے اسے پکارا: اے ابراہیم۔
قَدْ صَدَّقْتَ ٱلرُّءْيَآۚ إِنَّا كَذَٰلِكَ نَجْزِي ٱلْمُحْسِنِينَ ﴿105﴾
تو نے خواب کو سچ کر دکھایا، بے شک ہم نیکی کرنے والوں کو اسی طرح جزا دیتے ہیں۔
إِنَّ هَـٰذَا لَهُوَ ٱلْبَلَـٰٓؤُا۟ ٱلْمُبِينُ ﴿106﴾
بے شک یہ کھلی آزمائش تھی۔
وَفَدَيْنَـٰهُ بِذِبْحٍ عَظِيمٖ ﴿107﴾
اور ہم نے اسے ایک عظیم قربانی سے بدل دیا۔
وَتَرَكْنَا عَلَيْهِ فِي ٱلْـَٔاخِرِينَ ﴿108﴾
اور ہم نے اس کا ذکر بعد والوں میں باقی رکھا۔
سَلَـٰمٌ عَلَىٰٓ إِبْرَٰهِيمَ ﴿109﴾
سلام ہو ابراہیم پر۔
كَذَٰلِكَ نَجْزِي ٱلْمُحْسِنِينَ ﴿110﴾
ہم نیکی کرنے والوں کو اسی طرح جزا دیتے ہیں۔
إِنَّهُۥ مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿111﴾
بے شک وہ ہمارے مؤمن بندوں میں سے تھا۔
وَبَشَّرْنَـٰهُ بِإِسْحَـٰقَ نَبِيّٗا مِّنَ ٱلصَّـٰلِحِينَ ﴿112﴾
اور ہم نے اسے اسحاق کی خوشخبری دی، جو نیک لوگوں میں سے نبی تھا۔
وَبَـٰرَكْنَا عَلَيْهِ وَعَلَىٰٓ إِسْحَـٰقَۚ وَمِن ذُرِّيَّتِهِمَا مُحْسِنٞ وَظَالِمٞ لِّنَفْسِهِۦ مُبِينٞ ﴿113﴾
اور ہم نے اس پر اور اسحاق پر برکت دی، اور ان دونوں کی اولاد میں کچھ نیک اور کچھ اپنے نفس پر کھلا ظلم کرنے والے ہیں۔
وَلَقَدْ مَنَنَّا عَلَىٰ مُوسَىٰ وَهَـٰرُونَ ﴿114﴾
اور بے شک ہم نے موسیٰ اور ہارون پر احسان کیا۔
وَنَجَّيْنَـٰهُمَا وَقَوْمَهُمَا مِنَ ٱلْكَرْبِ ٱلْعَظِيمِ ﴿115﴾
اور ہم نے ان دونوں کو اور ان کی قوم کو بڑی مصیبت سے نجات دی۔
وَنَصَرْنَـٰهُمْ فَكَانُوا۟ هُمُ ٱلْغَـٰلِبِينَ ﴿116﴾
اور ہم نے ان کی مدد کی، تو وہی غالب رہے۔
وَءَاتَيْنَـٰهُمَا ٱلْكِتَـٰبَ ٱلْمُسْتَبِينَ ﴿117﴾
اور ہم نے ان دونوں کو واضح کتاب عطا کی۔
وَهَدَيْنَـٰهُمَا ٱلصِّرَٰطَ ٱلْمُسْتَقِيمَ ﴿118﴾
اور ہم نے ان دونوں کو سیدھا راستہ دکھایا۔
وَتَرَكْنَا عَلَيْهِمَا فِي ٱلْـَٔاخِرِينَ ﴿119﴾
اور ہم نے ان دونوں کا ذکر بعد والوں میں باقی رکھا۔
سَلَـٰمٌ عَلَىٰ مُوسَىٰ وَهَـٰرُونَ ﴿120﴾
سلام ہو موسیٰ اور ہارون پر۔
كَذَٰلِكَ نَجْزِي ٱلْمُحْسِنِينَ ﴿121﴾
ہم نیکی کرنے والوں کو اسی طرح جزا دیتے ہیں۔
إِنَّهُمَا مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿122﴾
بے شک وہ دونوں ہمارے مؤمن بندوں میں سے تھے۔
وَإِنَّ إِلْيَاسَ لَمِنَ ٱلْمُرْسَلِينَ ﴿123﴾
اور بے شک الیاس رسولوں میں سے تھا۔
إِذْ قَالَ لِقَوْمِهِۦٓ أَلَا تَتَّقُونَ ﴿124﴾
جب اس نے اپنی قوم سے کہا: کیا تم تقویٰ نہیں اختیار کرتے؟
أَتَدْعُونَ بَعْلٗا وَتَذَرُونَ أَحْسَنَ ٱلْخَـٰلِقِينَ ﴿125﴾
کیا تم بعل کو پکارتے ہو اور سب سے بہتر خالق کو چھوڑ دیتے ہو؟
ٱللَّهَ رَبَّكُمْ وَرَبَّ ءَابَآئِكُمُ ٱلْأَوَّلِينَ ﴿126﴾
اللہ کو، جو تمہارا رب ہے اور تمہارے پہلے باپ دادا کا رب ہے۔
فَكَذَّبُوهُ فَإِنَّهُمْ لَمُحْضَرُونَ ﴿127﴾
تو انہوں نے اسے جھٹلایا، پس وہ ضرور حاضر کیے جانے والے ہیں۔
إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿128﴾
سوائے اللہ کے چنے ہوئے بندوں کے۔
وَتَرَكْنَا عَلَيْهِ فِي ٱلْـَٔاخِرِينَ ﴿129﴾
اور ہم نے اس کا ذکر بعد والوں میں باقی رکھا۔
سَلَـٰمٌ عَلَىٰٓ إِلْيَاسِينَ ﴿130﴾
سلام ہو الیاس پر۔
كَذَٰلِكَ نَجْزِي ٱلْمُحْسِنِينَ ﴿131﴾
ہم نیکی کرنے والوں کو اسی طرح جزا دیتے ہیں۔
إِنَّهُۥ مِنْ عِبَادِنَا ٱلْمُؤْمِنِينَ ﴿132﴾
بے شک وہ ہمارے مؤمن بندوں میں سے تھا۔
وَإِنَّ لُوطٗا لَّمِنَ ٱلْمُرْسَلِينَ ﴿133﴾
اور بے شک لوط رسولوں میں سے تھا۔
إِذْ نَجَّيْنَـٰهُ وَأَهْلَهُۥٓ أَجْمَعِينَ ﴿134﴾
جب ہم نے اسے اور اس کے سب گھر والوں کو نجات دی۔
إِلَّا عَجُوزٗا فِي ٱلْغَٰبِرِينَ ﴿135﴾
سوائے ایک بوڑھی عورت کے جو پیچھے رہ جانے والوں میں تھی۔
ثُمَّ دَمَّرْنَا ٱلْـَٔاخَرِينَ ﴿136﴾
پھر ہم نے باقی سب کو تباہ کر دیا۔
وَإِنَّكُمْ لَتَمُرُّونَ عَلَيْهِم مُّصْبِحِينَ ﴿137﴾
اور بے شک تم صبح کے وقت ان پر سے گزرتے ہو۔
وَبِٱلَّيْلِۚ أَفَلَا تَعْقِلُونَ ﴿138﴾
اور رات کے وقت بھی، تو کیا تم عقل نہیں رکھتے؟
وَإِنَّ يُونُسَ لَمِنَ ٱلْمُرْسَلِينَ ﴿139﴾
اور بے شک یونس رسولوں میں سے تھا۔
إِذْ أَبَقَ إِلَى ٱلْفُلْكِ ٱلْمَشْحُونِ ﴿140﴾
جب وہ بھاگ کر بھری ہوئی کشتی کی طرف گیا۔
فَسَاهَمَ فَكَانَ مِنَ ٱلْمُدْحَضِينَ ﴿141﴾
تو اس نے قرعہ اندازی کی، اور وہ ہارنے والوں میں سے ہو گیا۔
فَٱلْتَقَمَهُ ٱلْحُوتُ وَهُوَ مُلِيمٞ ﴿142﴾
تو مچھلی نے اسے نگل لیا، اور وہ قابلِ ملامت تھا۔
فَلَوْلَآ أَنَّهُۥ كَانَ مِنَ ٱلْمُسَبِّحِينَ ﴿143﴾
اگر وہ تسبیح کرنے والوں میں سے نہ ہوتا۔
لَلَبِثَ فِي بَطْنِهِۦٓ إِلَىٰ يَوْمِ يُبْعَثُونَ ﴿144﴾
تو وہ اس کے پیٹ میں قیامت کے دن تک رہتا۔
فَنَبَذْنَـٰهُ بِٱلْعَرَآءِ وَهُوَ سَقِيمٞ ﴿145﴾
پھر ہم نے اسے چٹیل میدان میں پھینک دیا، اور وہ بیمار تھا۔
وَأَنۢبَتْنَا عَلَيْهِ شَجَرَةٗ مِّن يَقْطِينٖ ﴿146﴾
اور ہم نے اس پر کدو کا درخت اگا دیا۔
وَأَرْسَلْنَـٰهُ إِلَىٰ مِاْئَةِ أَلْفٍ أَوْ يَزِيدُونَ ﴿147﴾
اور ہم نے اسے ایک لاکھ یا اس سے زیادہ لوگوں کی طرف بھیجا۔
فَـَٔامَنُوا۟ فَمَتَّعْنَـٰهُمْ إِلَىٰ حِينٖ ﴿148﴾
تو وہ ایمان لے آئے، اور ہم نے انہیں ایک وقت تک فائدہ پہنچایا۔
فَٱسْتَفْتِهِمْ أَلِرَبِّكَ ٱلْبَنَاتُ وَلَهُمُ ٱلْبَنُونَ ﴿149﴾
تو ان سے پوچھو: کیا تمہارے رب کے لیے بیٹیاں ہیں اور ان کے لیے بیٹے؟
أَمْ خَلَقْنَا ٱلْمَلَـٰٓئِكَةَ إِنَـٰثٗا وَهُمْ شَـٰهِدُونَ ﴿150﴾
کیا ہم نے فرشتوں کو عورتیں بنایا ہے اور وہ دیکھ رہے تھے؟
أَلَآ إِنَّهُم مِّنْ إِفْكِهِمْ لَيَقُولُونَ ﴿151﴾
خبردار! وہ اپنے جھوٹ میں یہ کہتے ہیں۔
وَلَدَ ٱللَّهُ وَإِنَّهُمْ لَكَـٰذِبُونَ ﴿152﴾
کہ اللہ نے اولاد پیدا کی، حالانکہ وہ جھوٹے ہیں۔
أَصْطَفَى ٱلْبَنَاتِ عَلَى ٱلْبَنِينَ ﴿153﴾
کیا اس نے بیٹوں پر بیٹیاں پسند کیں؟
مَا لَكُمْ كَيْفَ تَحْكُمُونَ ﴿154﴾
تمہیں کیا ہوا ہے؟ کیسے فیصلے کرتے ہو؟
أَفَلَا تَذَكَّرُونَ ﴿155﴾
کیا تم نصیحت حاصل نہیں کرتے؟
أَمْ لَكُمْ سُلْطَـٰنٞ مُّبِينٞ ﴿156﴾
یا تمہارے پاس کوئی واضح دلیل ہے؟
فَأْتُوا۟ بِكِتَـٰبِكُمْ إِن كُنتُمْ صَـٰدِقِينَ ﴿157﴾
تو اپنی کتاب لاؤ اگر تم سچے ہو۔
وَجَعَلُوا۟ بَيْنَهُۥ وَبَيْنَ ٱلْجِنَّةِ نَسَبٗاۚ وَلَقَدْ عَلِمَتِ ٱلْجِنَّةُ إِنَّهُمْ لَمُحْضَرُونَ ﴿158﴾
اور انہوں نے اللہ اور جنات کے درمیان نسب بنا لیا، حالانکہ جنات جانتے ہیں کہ وہ ضرور حاضر کیے جائیں گے۔
سُبْحَـٰنَ ٱللَّهِ عَمَّا يَصِفُونَ ﴿159﴾
اللہ پاک ہے ان باتوں سے جو وہ بیان کرتے ہیں۔
إِلَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿160﴾
سوائے اللہ کے چنے ہوئے بندوں کے۔
فَإِنَّكُمْ وَمَا تَعْبُدُونَ ﴿161﴾
تو بے شک تم اور جنہیں تم پوجتے ہو۔
مَآ أَنتُمْ عَلَيْهِ بِفَـٰتِنِينَ ﴿162﴾
تم ان پر قابو نہیں پا سکتے۔
إِلَّا مَنْ هُوَ صَالِ ٱلْجَحِيمِ ﴿163﴾
سوائے اس کے جو جہنم میں جانے والا ہے۔
وَمَا مِنَّآ إِلَّا لَهُۥ مَقَامٞ مَّعْلُومٞ ﴿164﴾
اور ہم میں سے ہر ایک کا ایک مقرر مقام ہے۔
وَإِنَّا لَنَحْنُ ٱلصَّآفُّونَ ﴿165﴾
اور بے شک ہم ہی صف باندھنے والے ہیں۔
وَإِنَّا لَنَحْنُ ٱلْمُسَبِّحُونَ ﴿166﴾
اور بے شک ہم ہی تسبیح کرنے والے ہیں۔
وَإِن كَانُوا۟ لَيَقُولُونَ ﴿167﴾
اور بے شک وہ کہا کرتے تھے۔
لَوْ أَنَّ عِندَنَا ذِكْرٗا مِّنَ ٱلْأَوَّلِينَ ﴿168﴾
کاش ہمارے پاس پہلے لوگوں کی کوئی نصیحت ہوتی۔
لَكُنَّا عِبَادَ ٱللَّهِ ٱلْمُخْلَصِينَ ﴿169﴾
تو ہم اللہ کے چنے ہوئے بندے ہوتے۔
فَكَفَرُوا۟ بِهِۦ فَسَوْفَ يَعْلَمُونَ ﴿170﴾
تو انہوں نے اس کا انکار کیا، پس عنقریب وہ جان لیں گے۔
وَلَقَدۡ سَبَقَتۡ كَلِمَتُنَا لِعِبَادِنَا ٱلۡمُرۡسَلِينَ ﴿171﴾
اور بے شک ہمارے رسول بندوں کے حق میں ہمارا وعدہ پہلے ہی ہو چکا ہے۔
إِنَّهُمۡ لَهُمُ ٱلۡمَنصُورُونَ ﴿172﴾
کہ بے شک وہی مدد دیے جائیں گے۔
وَإِنَّ جُندَنَا لَهُمُ ٱلۡغَـٰلِبُونَ ﴿173﴾
اور بے شک ہمارا لشکر ہی غالب رہے گا۔
فَتَوَلَّ عَنۡهُمۡ حَتَّىٰ حِينٖ ﴿174﴾
تو ان سے ایک وقت تک منہ پھیر لے۔
وَأَبۡصِرۡهُمۡ فَسَوۡفَ يُبۡصِرُونَ ﴿175﴾
اور انہیں دیکھتا رہ، عنقریب وہ بھی دیکھ لیں گے۔
أَفَبِعَذَابِنَا يَسۡتَعۡجِلُونَ ﴿176﴾
کیا وہ ہمارے عذاب کو جلدی چاہتے ہیں؟
فَإِذَا نَزَلَ بِسَاحَتِهِمۡ فَسَآءَ صَبَاحُ ٱلۡمُنذَرِينَ ﴿177﴾
پھر جب وہ ان کے صحن میں اترے گا تو ڈرائے جانے والوں کی صبح بری ہوگی۔
وَتَوَلَّ عَنۡهُمۡ حَتَّىٰ حِينٖ ﴿178﴾
اور ان سے ایک وقت تک منہ پھیر لے۔
وَأَبۡصِرۡ فَسَوۡفَ يُبۡصِرُونَ ﴿179﴾
اور دیکھتا رہ، عنقریب وہ بھی دیکھ لیں گے۔
سُبۡحَـٰنَ رَبِّكَ رَبِّ ٱلۡعِزَّةِ عَمَّا يَصِفُونَ ﴿180﴾
پاک ہے تیرے رب، عزت والے رب، ان باتوں سے جو وہ بیان کرتے ہیں۔
وَسَلَـٰمٌ عَلَى ٱلۡمُرۡسَلِينَ ﴿181﴾
اور سلام ہو رسولوں پر۔
وَٱلۡحَمۡدُ لِلَّهِ رَبِّ ٱلۡعَـٰلَمِينَ ﴿182﴾
اور تمام تعریف اللہ کے لیے ہے جو رب ہے تمام جہانوں کا۔
سورة ص
بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
صٓۚ وَٱلْقُرْءَانِ ذِي ٱلذِّكْرِ ﴿1﴾
ص۔ قسم ہے قرآن کی جو نصیحت والا ہے۔
بَلِ ٱلَّذِينَ كَفَرُوا۟ فِي عِزَّةٖ وَشِقَاقٍ ﴿2﴾
بلکہ وہ لوگ جنہوں نے کفر کیا، وہ غرور اور مخالفت میں ہیں۔
كَمْ أَهْلَكْنَا مِن قَبْلِهِم مِّن قَرْنٖ فَنَادَوْا۟ وَّلَاتَ حِينَ مَنَاصٖ ﴿3﴾
ہم نے ان سے پہلے کتنی ہی قومیں ہلاک کر دیں، تو وہ فریاد کرتے رہے، مگر وہ وقت بچاؤ کا نہ تھا۔
وَعَجِبُوٓا۟ أَن جَآءَهُم مُّنذِرٞ مِّنْهُمْۖ وَقَالَ ٱلْكَـٰفِرُونَ هَـٰذَا سَـٰحِرٞ كَذَّابٌ ﴿4﴾
اور انہیں تعجب ہوا کہ ان کے پاس انہی میں سے ایک ڈرانے والا آیا، اور کافروں نے کہا: یہ تو جادوگر، جھوٹا ہے۔
أَجَعَلَ ٱلْـَٔاهِلَةَ إِلَـٰهٗا وَٰحِدًاۖ إِنَّ هَـٰذَا لَشَيْءٌ عُجَابٌ ﴿5﴾
کیا اس نے سب معبودوں کو ایک معبود بنا دیا؟ بے شک یہ تو بڑی عجیب بات ہے۔
وَٱنطَلَقَ ٱلْمَلَأُ مِنْهُمْ أَنِ ٱمْشُوا۟ وَٱصْبِرُوا۟ عَلَىٰٓ ءَالِهَتِكُمْۖ إِنَّ هَـٰذَا لَشَيْءٞ يُرَادُ ﴿6﴾
اور ان میں سے سردار نکلے (اور کہا): چلو اور اپنے معبودوں پر جمے رہو، بے شک یہ تو ایک ارادہ کی گئی بات ہے۔
مَا سَمِعْنَا بِهَـٰذَا فِي ٱلْمِلَّةِ ٱلْـَٔاخِرَةِ إِنْ هَـٰذَآ إِلَّا ٱخْتِلَـٰقٌ ﴿7﴾
ہم نے یہ بات پچھلی ملت (دین) میں نہیں سنی، یہ تو محض گھڑنت ہے۔
أَءُنزِلَ عَلَيْهِ ٱلذِّكْرُ مِنۢ بَيْنِنَاۚ بَلْ هُمۡ فِي شَكّٖ مِّن ذِكْرِيۖ بَلْ لَمَّا يَذُوقُوا۟ عَذَابِ ﴿8﴾
کیا ہمارے درمیان اسی پر نصیحت نازل کی گئی ہے؟ بلکہ وہ میری نصیحت سے شک میں ہیں، بلکہ انہوں نے ابھی میرا عذاب چکھا نہیں۔
أَمْ عِندَهُمْ خَزَآئِنُ رَحْمَةِ رَبِّكَ ٱلْعَزِيزِ ٱلْوَهَّابِ ﴿9﴾
کیا ان کے پاس تیرے رب، زبردست عطا کرنے والے کی رحمت کے خزانے ہیں؟
أَمْ لَهُم مُّلْكُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَاۖ فَلْيَرْتَقُوا۟ فِي ٱلْأَسْبَـٰبِ ﴿10﴾
یا ان کے پاس آسمانوں اور زمین اور جو کچھ ان کے درمیان ہے اس کی بادشاہی ہے؟ تو وہ اسباب (بلندیوں) پر چڑھ کر دکھائیں۔
جُندٞ مَّا هُنَالِكَ مَهْزُومٞ مِّنَ ٱلْأَحْزَابِ ﴿11﴾
یہ بھی ایک لشکر ہے جو یہاں شکست کھانے والا ہے، ان گروہوں میں سے۔
كَذَّبَتْ قَبْلَهُمْ قَوْمُ نُوحٖ وَعَادٞ وَفِرْعَوْنُ ذُو ٱلْأَوْتَادِ ﴿12﴾
ان سے پہلے نوح کی قوم، عاد اور میخوں والا فرعون جھٹلا چکے ہیں۔
وَثَمُودُ وَقَوْمُ لُوطٖ وَأَصْحَـٰبُ ٱلْأَيْكَةِۚ أُو۟لَـٰٓئِكَ ٱلْأَحْزَابُ ﴿13﴾
اور ثمود، لوط کی قوم اور ایکہ والے، یہ سب گروہ تھے۔
إِن كُلٌّ إِلَّا كَذَّبَ ٱلرُّسُلَ فَحَقَّ عِقَابِ ﴿14﴾
ان سب نے رسولوں کو جھٹلایا، تو میرا عذاب ان پر واجب ہو گیا۔
وَمَا يَنظُرُ هَـٰٓؤُلَآءِ إِلَّا صَيْحَةٗ وَٰحِدَةٗ مَّا لَهَا مِن فَوَاقٖ ﴿15﴾
یہ لوگ صرف ایک زوردار چیخ کا انتظار کر رہے ہیں، جس کے بعد کوئی مہلت نہ ہوگی۔
وَقَالُوا۟ رَبَّنَا عَجِّل لَّنَا قِطَّنَا قَبْلَ يَوْمِ ٱلْحِسَابِ ﴿16﴾
اور وہ کہتے ہیں: اے ہمارے رب! ہمارا حصہ ہمیں جلد دے دے، حساب کے دن سے پہلے۔
ٱصْبِرْ عَلَىٰ مَا يَقُولُونَ وَٱذْكُرْ عَبْدَنَا دَاوُۥدَ ذَا ٱلْأَيْدِۖ إِنَّهُۥٓ أَوَّابٌ ﴿17﴾
صبر کرو ان باتوں پر جو وہ کہتے ہیں، اور ہمارے بندے داؤد کو یاد کرو، جو قوت والا تھا، بے شک وہ بہت رجوع کرنے والا تھا۔
إِنَّا سَخَّرْنَا ٱلْجِبَالَ مَعَهُۥ يُسَبِّحْنَ بِٱلْعَشِيِّ وَٱلْإِشْرَاقِ ﴿18﴾
ہم نے پہاڑوں کو اس کے ساتھ مسخر کر دیا تھا کہ وہ شام اور صبح اس کے ساتھ تسبیح کرتے تھے۔
وَٱلطَّيْرَ مَحْشُورَةٗۖ كُلّٞ لَّهُۥٓ أَوَّابٞ ﴿19﴾
اور پرندے جمع ہو کر، سب اس کے تابع تھے، اور سب رجوع کرنے والے۔
وَشَدَدْنَا مُلْكَهُۥ وَءَاتَيْنَـٰهُ ٱلْحِكْمَةَ وَفَصْلَ ٱلْخِطَابِ ﴿20﴾
اور ہم نے اس کی بادشاہی کو مضبوط کیا، اور اسے حکمت اور فیصلہ کن بات عطا کی۔
وَهَلْ أَتَىٰكَ نَبَؤُا۟ ٱلْخَصْمِ إِذْ تَسَوَّرُوا۟ ٱلْمِحْرَابَ ﴿21﴾
اور کیا تمہیں جھگڑنے والوں کی خبر پہنچی جب وہ دیوار پھاند کر عبادت گاہ میں داخل ہوئے؟
إِذْ دَخَلُوا۟ عَلَىٰ دَاوُۥدَ فَفَزِعَ مِنْهُمْۖ قَالُوا۟ لَا تَخَفْۖ خَصْمَانِ بَغَىٰ بَعْضُنَا عَلَىٰ بَعْضٖ فَٱحْكُم بَيْنَنَا بِٱلْحَقِّ وَلَا تُشْطِطْ وَٱهْدِنَآ إِلَىٰ سَوَآءِ ٱلصِّرَٰطِ ﴿22﴾
جب وہ داؤد کے پاس داخل ہوئے تو وہ ان سے گھبرا گئے، انہوں نے کہا: نہ ڈریے، ہم دو فریق ہیں، جن میں ایک نے دوسرے پر زیادتی کی ہے، پس آپ ہمارے درمیان حق کے ساتھ فیصلہ کریں اور زیادتی نہ کریں، اور ہمیں سیدھے راستے کی طرف رہنمائی دیں۔
إِنَّ هَـٰذَآ أَخِي لَهُۥ تِسْعٞ وَتِسْعُونَ نَعْجَةٗ وَلِيَ نَعْجَةٞ وَٱحِدَةٞ فَقَالَ أَكْفِلْنِيهَا وَعَزَّنِي فِي ٱلْخِطَابِ ﴿23﴾
بے شک یہ میرا بھائی ہے، اس کے پاس ننانوے (99) دنبیاں ہیں اور میرے پاس ایک دنبی ہے، تو اس نے کہا: یہ بھی مجھے دے دے، اور اس نے گفتگو میں مجھے دبا لیا۔
قَالَ لَقَدْ ظَلَمَكَ بِسُؤَالِ نَعْجَتِكَ إِلَىٰ نِعَاجِهِۦۚ وَإِنَّ كَثِيرٗا مِّنَ ٱلْخُلَطَآءِ لَيَبْغِي بَعْضُهُمْ عَلَىٰ بَعْضٍ إِلَّا ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ وَقَلِيلٞ مَّا هُمْۗ وَظَنَّ دَاوُۥدُ أَنَّمَا فَتَنَّـٰهُ فَٱسْتَغْفَرَ رَبَّهُۥ وَخَرَّ رَاكِعٗا وَأَنَابَ ﴿24﴾
داؤد نے کہا: اس نے تجھ پر ظلم کیا کہ تیری دنبی کو اپنی دنبیوں میں شامل کرنے کا مطالبہ کیا، اور بے شک بہت سے شریک ایک دوسرے پر زیادتی کرتے ہیں، سوائے ان کے جو ایمان لائے اور نیک عمل کیے، اور وہ بہت کم ہیں۔ اور داؤد نے سمجھا کہ ہم نے اسے آزمائش میں ڈالا ہے، تو اس نے اپنے رب سے معافی مانگی، اور سجدے میں گر پڑا اور رجوع کیا۔
فَغَفَرْنَا لَهُۥ ذَٰلِكَۖ وَإِنَّ لَهُۥ عِندَنَا لَزُلْفَىٰ وَحُسْنَ مَـَٔابٖ ﴿25﴾
تو ہم نے اسے وہ (قصور) معاف کر دیا، اور بے شک اس کے لیے ہمارے پاس قرب اور اچھا ٹھکانہ ہے۔
يَـٰدَاوُۥدُ إِنَّا جَعَلْنَـٰكَ خَلِيفَةٗ فِي ٱلْأَرْضِ فَٱحْكُم بَيْنَ ٱلنَّاسِ بِٱلْحَقِّ وَلَا تَتَّبِعِ ٱلْهَوَىٰ فَيُضِلَّكَ عَن سَبِيلِ ٱللَّهِۚ إِنَّ ٱلَّذِينَ يَضِلُّونَ عَن سَبِيلِ ٱللَّهِ لَهُمْ عَذَابٞ شَدِيدُۢ بِمَا نَسُوا۟ يَوْمَ ٱلْحِسَابِ ﴿26﴾
اے داؤد! ہم نے تمہیں زمین میں خلیفہ بنایا ہے، پس لوگوں کے درمیان حق کے ساتھ فیصلہ کرو، اور خواہش کی پیروی نہ کرو کہ وہ تمہیں اللہ کے راستے سے گمراہ کر دے۔ بے شک جو لوگ اللہ کے راستے سے گمراہ ہوتے ہیں ان کے لیے سخت عذاب ہے، اس وجہ سے کہ انہوں نے حساب کے دن کو بھلا دیا۔
وَمَا خَلَقْنَا ٱلسَّمَآءَ وَٱلْأَرْضَ وَمَا بَيْنَهُمَا بَـٰطِلٗاۚ ذَٰلِكَ ظَنُّ ٱلَّذِينَ كَفَرُوا۟ۚ فَوَيْلٞ لِّلَّذِينَ كَفَرُوا۟ مِنَ ٱلنَّارِ ﴿27﴾
اور ہم نے آسمان اور زمین اور جو کچھ ان کے درمیان ہے اسے بے مقصد نہیں بنایا، یہ ان لوگوں کا گمان ہے جنہوں نے کفر کیا، پس کافروں کے لیے آگ کا افسوسناک انجام ہے۔
أَمْ نَجْعَلُ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ كَٱلْمُفْسِدِينَ فِي ٱلْأَرْضِ أَمْ نَجْعَلُ ٱلْمُتَّقِينَ كَٱلْفُجَّارِ ﴿28﴾
کیا ہم ایمان والوں اور نیک عمل کرنے والوں کو زمین میں فساد کرنے والوں کی طرح بنا دیں؟ یا ہم متقیوں کو فاجروں کی طرح بنا دیں؟
كِتَٰبٌ أَنزَلْنَٰهُ إِلَيْكَ مُبَٰرَكٞ لِّيَدَّبَّرُوا۟ ءَايَٰتِهِۦ وَلِيَتَذَكَّرَ أُو۟لُوا۟ ٱلْأَلْبَٰبِ ﴿29﴾
یہ ایک بابرکت کتاب ہے جو ہم نے آپ کی طرف نازل کی ہے تاکہ لوگ اس کی آیات میں غور کریں اور عقل والے نصیحت حاصل کریں۔
وَوَهَبْنَا لِدَاوُۥدَ سُلَيْمَٰنَۚ نِعْمَ ٱلْعَبْدُ إِنَّهُۥٓ أَوَّابٌ ﴿30﴾
اور ہم نے داؤد کو سلیمان عطا کیا، کیا ہی اچھا بندہ! بے شک وہ بہت رجوع کرنے والا تھا۔
إِذْ عُرِضَ عَلَيْهِ بِٱلْعَشِيِّ ٱلصَّٰفِنَٰتُ ٱلْجِيَادُ ﴿31﴾
جب شام کے وقت اس کے سامنے تیز رفتار، خوب ایستادہ گھوڑے پیش کیے گئے۔
فَقَالَ إِنِّيٓ أَحْبَبْتُ حُبَّ ٱلْخَيْرِ عَن ذِكْرِ رَبِّي حَتَّىٰ تَوَارَتْ بِٱلْحِجَابِ ﴿32﴾
تو اس نے کہا: میں نے اپنے رب کی یاد کے مقابلے میں مال (گھوڑوں) کی محبت کو پسند کیا، یہاں تک کہ وہ (سورج) پردے کے پیچھے چھپ گیا۔
رُدُّوهَا عَلَيَّۖ فَطَفِقَ مَسْحَۢا بِٱلسُّوقِ وَٱلْأَعْنَاقِ ﴿33﴾
اس نے کہا: انہیں میرے پاس واپس لاؤ، پھر وہ ان کی پنڈلیوں اور گردنوں پر ہاتھ پھیرنے لگا۔
وَلَقَدْ فَتَنَّا سُلَيْمَٰنَ وَأَلْقَيْنَا عَلَىٰ كُرْسِيِّهِۦ جَسَدٗا ثُمَّ أَنَابَ ﴿34﴾
اور بے شک ہم نے سلیمان کو آزمائش میں ڈالا، اور اس کے تخت پر ایک جسم ڈال دیا، پھر اس نے رجوع کیا۔
قَالَ رَبِّ ٱغْفِرْ لِي وَهَبْ لِي مُلْكٗا لَّا يَنبَغِي لِأَحَدٖ مِّنۢ بَعْدِيٓۖ إِنَّكَ أَنتَ ٱلْوَهَّابُ ﴿35﴾
اس نے کہا: اے میرے رب! مجھے بخش دے اور مجھے ایسی بادشاہی عطا کر جو میرے بعد کسی کو سزاوار نہ ہو، بے شک تو ہی عطا کرنے والا ہے۔
فَسَخَّرْنَا لَهُ ٱلرِّيحَ تَجْرِي بِأَمْرِهِۦ رُخَآءً حَيْثُ أَصَابَ ﴿36﴾
تو ہم نے اس کے لیے ہوا کو مسخر کر دیا، جو اس کے حکم سے نرمی سے چلتی تھی جہاں وہ چاہتا۔
وَٱلشَّيَـٰطِينَ كُلَّ بَنَّآءٖ وَغَوَّاصٖ ﴿37﴾
اور شیاطین کو بھی، ہر قسم کے معمار اور غوطہ خور۔
وَءَاخَرِينَ مُقَرَّنِينَ فِي ٱلْأَصْفَادِ ﴿38﴾
اور دوسرے جو زنجیروں میں جکڑے ہوئے تھے۔
هَـٰذَا عَطَآؤُنَا فَٱمْنُنْ أَوْ أَمْسِكْ بِغَيْرِ حِسَابٖ ﴿39﴾
یہ ہماری عطا ہے، پس تو جسے چاہے دے یا روک رکھ، بغیر حساب۔
وَإِنَّ لَهُۥ عِندَنَا لَزُلْفَىٰ وَحُسْنَ مَـَٔابٖ ﴿40﴾
اور بے شک اس کے لیے ہمارے پاس قرب اور اچھا ٹھکانہ ہے۔
وَٱذْكُرْ عَبْدَنَآ أَيُّوبَ إِذْ نَادَىٰ رَبَّهُۥٓ أَنِّي مَسَّنِيَ ٱلشَّيْطَـٰنُ بِنُصْبٖ وَعَذَابٍ ﴿41﴾
اور ہمارے بندے ایوب کو یاد کرو، جب اس نے اپنے رب کو پکارا: مجھے شیطان نے تکلیف اور درد پہنچایا ہے۔
ٱرْكُضْ بِرِجْلِكَۖ هَـٰذَا مُغْتَسَلُۢ بَارِدٞ وَشَرَابٞ ﴿42﴾
(ہم نے فرمایا:) اپنے پاؤں سے زمین پر مارو، یہ ہے ٹھنڈا غسل اور پینے کا پانی۔
وَوَهَبْنَا لَهُۥٓ أَهْلَهُۥ وَمِثْلَهُم مَّعَهُمْ رَحْمَةٗ مِّنَّا وَذِكْرَىٰ لِأُو۟لِي ٱلْأَلْبَـٰبِ ﴿43﴾
اور ہم نے اسے اس کے اہل و عیال واپس دیے اور ان کے ساتھ اتنے ہی اور بھی، اپنی طرف سے رحمت کے طور پر اور عقل والوں کے لیے نصیحت۔
وَخُذْ بِيَدِكَ ضِغْثٗا فَٱضْرِب بِّهِۦ وَلَا تَحْنَثْۗ إِنَّا وَجَدْنَـٰهُ صَابِرٗاۚ نِّعْمَ ٱلْعَبْدُ إِنَّهُۥٓ أَوَّابٞ ﴿44﴾
(ہم نے فرمایا:) اپنے ہاتھ میں ایک جھاڑ لو اور اس سے مارو، اور قسم نہ توڑو۔ بے شک ہم نے اسے صابر پایا، کیا ہی اچھا بندہ! بے شک وہ بہت رجوع کرنے والا تھا۔
وَٱذْكُرْ عِبَادَنَآ إِبْرَٰهِيمَ وَإِسْحَٰقَ وَيَعْقُوبَ أُو۟لِي ٱلْأَيْدِي وَٱلْأَبْصَٰرِ ﴿45﴾
اور ہمارے بندوں ابراہیم، اسحاق اور یعقوب کو یاد کرو، جو قوت والے اور بصیرت والے تھے۔
إِنَّآ أَخْلَصْنَـٰهُم بِخَالِصَةٖ ذِكْرَى ٱلدَّارِ ﴿46﴾
ہم نے انہیں ایک خاص صفت کے ساتھ چن لیا تھا — آخرت کی یاد۔
وَإِنَّهُمْ عِندَنَا لَمِنَ ٱلْمُصْطَفَيْنَ ٱلْأَخْيَارِ ﴿47﴾
اور بے شک وہ ہمارے نزدیک چنے ہوئے نیک لوگوں میں سے ہیں۔
وَٱذْكُرْ إِسْمَٰعِيلَ وَٱلْيَسَعَ وَذَا ٱلْكِفْلِۖ وَكُلّٞ مِّنَ ٱلْأَخْيَارِ ﴿48﴾
اور اسماعیل، الیسع اور ذوالکفل کو یاد کرو، یہ سب نیک لوگوں میں سے تھے۔
هَـٰذَا ذِكْرٞۚ وَإِنَّ لِلْمُتَّقِينَ لَحُسْنَ مَـَٔابٍ ﴿49﴾
یہ ایک یاد دہانی ہے، اور بے شک متقیوں کے لیے بہترین ٹھکانہ ہے۔
جَنَّـٰتِ عَدْنٖ مُّفَتَّحَةٗ لَّهُمُ ٱلْأَبْوَٰبُ ﴿50﴾
ہمیشہ رہنے والی جنتیں، جن کے دروازے ان کے لیے کھلے ہوں گے۔
مُتَّكِـِٔينَ فِيهَا يَدْعُونَ فِيهَا بِفَـٰكِهَةٖ كَثِيرَةٖ وَشَرَابٖ ﴿51﴾
وہ اس میں تکیہ لگائے ہوں گے، اور اس میں بہت سے پھل اور مشروب طلب کریں گے۔
وَعِندَهُمْ قَـٰصِرَٰتُ ٱلطَّرْفِ أَتْرَابٌ ﴿52﴾
اور ان کے پاس نظریں جھکائے ہوئے ہم عمر بیویاں ہوں گی۔
هَـٰذَا مَا تُوعَدُونَ لِيَوْمِ ٱلْحِسَابِ ﴿53﴾
یہ ہے وہ جس کا تم سے وعدہ کیا گیا ہے حساب کے دن کے لیے۔
إِنَّ هَـٰذَا لَرِزْقُنَا مَا لَهُۥ مِن نَّفَادٍ ﴿54﴾
بے شک یہ ہمارا رزق ہے، جو کبھی ختم نہیں ہوگا۔
هَـٰذَاۚ وَإِنَّ لِلطَّاغِينَ لَشَرَّ مَـَٔابٍ ﴿55﴾
یہ (متقیوں کے لیے ہے)، اور بے شک سرکشوں کے لیے برا ٹھکانہ ہے۔
جَهَنَّمَ يَصْلَوْنَهَا فَبِئْسَ ٱلْمِهَادُ ﴿56﴾
جہنم، جس میں وہ داخل ہوں گے، پس وہ بہت ہی برا ٹھکانہ ہے۔
هَـٰذَا فَلْيَذُوقُوهُ حَمِيمٞ وَغَسَّاقٞ ﴿57﴾
یہ ہے، پس وہ اسے چکھیں — کھولتا ہوا پانی اور پیپ۔
وَءَاخَرُ مِن شَكْلِهِۦٓ أَزْوَٰجٌ ﴿58﴾
اور اسی قسم کی اور چیزیں بھی (عذاب کی) اقسام ہوں گی۔
هَـٰذَا فَوْجٞ مُّقْتَحِمٞ مَّعَكُمۖ لَا مَرْحَبَۢا بِهِمْ إِنَّهُمْ صَالُوا۟ ٱلنَّارِ ﴿59﴾
(دوزخی کہیں گے:) یہ ایک گروہ ہے جو تمہارے ساتھ گھس آیا ہے، ان کے لیے کوئی خوش آمدید نہیں، بے شک وہ آگ میں داخل ہونے والے ہیں۔
قَالُوا۟ بَلْ أَنتُمْ لَا مَرْحَبَۢا بِكُمْ أَنتُمْ قَدَّمْتُمُوهُ لَنَاۖ فَبِئْسَ ٱلْقَرَارُ ﴿60﴾
وہ کہیں گے: بلکہ تمہارے لیے خوش آمدید نہیں، تم ہی نے ہمارے لیے یہ (عذاب) پیش کیا، پس بہت برا ٹھکانہ ہے۔
قَالُوا۟ رَبَّنَا مَن قَدَّمَ لَنَا هَـٰذَا فَزِدْهُ عَذَابٗا ضِعْفٗا فِي ٱلنَّارِ ﴿61﴾
وہ کہیں گے: اے ہمارے رب! جس نے ہمارے لیے یہ (عذاب) پیش کیا، تو اسے آگ میں دوگنا عذاب دے۔
وَقَالُوا۟ مَا لَنَا لَا نَرَىٰ رِجَالٗا كُنَّا نَعُدُّهُم مِّنَ ٱلْأَشْرَارِ ﴿62﴾
اور وہ کہیں گے: ہمیں کیا ہوا کہ ہم ان مردوں کو نہیں دیکھ رہے جنہیں ہم برے شمار کرتے تھے؟
أَتَّخَذْنَـٰهُمْ سِخْرِيًّا أَمْ زَاغَتْ عَنْهُمُ ٱلْأَبْصَـٰرُ ﴿63﴾
کیا ہم نے انہیں مذاق بنا لیا تھا، یا ہماری نگاہیں ان سے ہٹ گئی ہیں؟
إِنَّ ذَٰلِكَ لَحَقّٞۖ تَخَاصُمُ أَهْلِ ٱلنَّارِ ﴿64﴾
بے شک یہ سچ ہے — دوزخیوں کی باہمی جھگڑنے کی بات۔
قُلْ إِنَّمَآ أَنَا۠ مُنذِرٞۖ وَمَا مِنْ إِلَـٰهٍ إِلَّا ٱللَّهُ ٱلْوَٰحِدُ ٱلْقَهَّارُ ﴿65﴾
کہہ دو: میں تو صرف خبردار کرنے والا ہوں، اور کوئی معبود نہیں مگر اللہ، جو یکتا اور سب پر غالب ہے۔
رَبُّ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَا ٱلْعَزِيزُ ٱلْغَفَّارُ ﴿66﴾
آسمانوں اور زمین اور جو کچھ ان کے درمیان ہے، سب کا رب، زبردست، بہت بخشنے والا۔
قُلْ هُوَ نَبَؤٌا عَظِيمٌ ﴿67﴾
کہہ دو: یہ ایک بڑی خبر ہے۔
أَنتُمْ عَنْهُ مُعْرِضُونَ ﴿68﴾
جس سے تم منہ موڑ رہے ہو۔
مَا كَانَ لِيَ مِنْ عِلْمِۭ بِٱلْمَلَإِ ٱلْأَعْلَىٰٓ إِذْ يَخْتَصِمُونَ ﴿69﴾
مجھے بلند جماعت (فرشتوں) کی اس وقت کی گفتگو کا علم نہ تھا جب وہ جھگڑ رہے تھے۔
إِن يُوحَىٰٓ إِلَيَّ إِلَّآ أَنَّمَآ أَنَا۠ نَذِيرٞ مُّبِينٌ ﴿70﴾
مجھے تو صرف یہ وحی کی جاتی ہے کہ میں صاف صاف خبردار کرنے والا ہوں۔
إِذْ قَالَ رَبُّكَ لِلْمَلَـٰٓئِكَةِ إِنِّي خَـٰلِقُۢ بَشَرٗا مِّن طِينٖ ﴿71﴾
جب تیرے رب نے فرشتوں سے کہا: میں مٹی سے ایک بشر پیدا کرنے والا ہوں۔
فَإِذَا سَوَّيْتُهُۥ وَنَفَخْتُ فِيهِ مِن رُّوحِي فَقَعُوا۟ لَهُۥ سَـٰجِدِينَ ﴿72﴾
پھر جب میں اسے درست کر لوں اور اس میں اپنی روح پھونک دوں تو تم اس کے لیے سجدے میں گر جانا۔
فَسَجَدَ ٱلْمَلَـٰٓئِكَةُ كُلُّهُمْ أَجْمَعُونَ ﴿73﴾
تو تمام فرشتوں نے سجدہ کیا، سب کے سب۔
إِلَّآ إِبْلِيسَ ٱسْتَكْبَرَ وَكَانَ مِنَ ٱلْكَـٰفِرِينَ ﴿74﴾
سوائے ابلیس کے، اس نے تکبر کیا اور وہ کافروں میں سے ہو گیا۔
قَالَ يَـٰٓإِبْلِيسُ مَا مَنَعَكَ أَن تَسْجُدَ لِمَا خَلَقْتُ بِيَدَيَّۖ أَسْتَكْبَرْتَ أَمْ كُنتَ مِنَ ٱلْعَالِينَ ﴿75﴾
اللہ نے فرمایا: اے ابلیس! تجھے کس چیز نے روکا کہ تو اس کو سجدہ نہ کرے جسے میں نے اپنے دونوں ہاتھوں سے پیدا کیا؟ کیا تو تکبر کر رہا ہے یا تو بلند مرتبہ والوں میں سے ہے؟
قَالَ أَنَا۠ خَيْرٞ مِّنْهُۖ خَلَقْتَنِي مِن نَّارٖ وَخَلَقْتَهُۥ مِن طِينٖ ﴿76﴾
اس نے کہا: میں اس سے بہتر ہوں، تو نے مجھے آگ سے پیدا کیا اور اسے مٹی سے۔
قَالَ فَٱخْرُجْ مِنْهَا فَإِنَّكَ رَجِيمٞ ﴿77﴾
اللہ نے فرمایا: تو یہاں سے نکل جا، بے شک تو مردود ہے۔
وَإِنَّ عَلَيْكَ لَعْنَتِيٓ إِلَىٰ يَوْمِ ٱلدِّينِ ﴿78﴾
اور بے شک تجھ پر میری لعنت ہے قیامت کے دن تک۔
قَالَ رَبِّ فَأَنظِرْنِيٓ إِلَىٰ يَوْمِ يُبْعَثُونَ ﴿79﴾
اس نے کہا: اے میرے رب! مجھے اس دن تک مہلت دے جب لوگ دوبارہ اٹھائے جائیں گے۔
قَالَ فَإِنَّكَ مِنَ ٱلْمُنظَرِينَ ﴿80﴾
اللہ نے فرمایا: بے شک تو مہلت پانے والوں میں سے ہے۔
إِلَىٰ يَوْمِ ٱلْوَقْتِ ٱلْمَعْلُومِ ﴿81﴾
اس وقت تک جو مقرر ہے۔
قَالَ فَبِعِزَّتِكَ لَأُغْوِيَنَّهُمْ أَجْمَعِينَ ﴿82﴾
اس نے کہا: تیری عزت کی قسم! میں ان سب کو ضرور گمراہ کروں گا۔
إِلَّا عِبَادَكَ مِنْهُمُ ٱلْمُخْلَصِينَ ﴿83﴾
سوائے تیرے ان بندوں کے جو چنے ہوئے ہوں۔
قَالَ فَٱلْحَقُّ وَٱلْحَقَّ أَقُولُ ﴿84﴾
اللہ نے فرمایا: حق ہے، اور میں حق ہی کہتا ہوں۔
لَأَمْلَأَنَّ جَهَنَّمَ مِنكَ وَمِمَّن تَبِعَكَ مِنْهُمْ أَجْمَعِينَ ﴿85﴾
میں ضرور جہنم کو تجھ سے اور ان سب سے بھر دوں گا جو تیرے پیچھے چلیں گے۔
قُلْ مَآ أَسْـَٔلُكُمْ عَلَيْهِ مِنْ أَجْرٖ وَمَآ أَنَا۠ مِنَ ٱلْمُتَكَلِّفِينَ ﴿86﴾
کہہ دو: میں تم سے اس (دعوت) پر کوئی اجر نہیں مانگتا، اور میں بناوٹ کرنے والوں میں سے نہیں ہوں۔
إِنْ هُوَ إِلَّا ذِكْرٞ لِّلْعَـٰلَمِينَ ﴿87﴾
یہ تو صرف تمام جہان والوں کے لیے ایک نصیحت ہے۔
وَلَتَعْلَمُنَّ نَبَأَهُۥ بَعْدَ حِينٖ ﴿88﴾
اور تم اس کی خبر کو کچھ وقت بعد ضرور جان لو گے۔

بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
تَنزِيلُ ٱلْكِتَـٰبِ مِنَ ٱللَّهِ ٱلْعَزِيزِ ٱلْحَكِيمِ ﴿1﴾
اس کتاب کا نزول اللہ کی طرف سے ہے جو زبردست، حکمت والا ہے۔
إِنَّآ أَنزَلْنَآ إِلَيْكَ ٱلْكِتَـٰبَ بِٱلْحَقِّ فَٱعْبُدِ ٱللَّهَ مُخْلِصٗا لَّهُ ٱلدِّينَ ﴿2﴾
بے شک ہم نے آپ کی طرف یہ کتاب حق کے ساتھ نازل کی ہے، پس اللہ کی عبادت کرو دین کو اسی کے لیے خالص کرتے ہوئے۔
أَلَا لِلَّهِ ٱلدِّينُ ٱلْخَالِصُۚ وَٱلَّذِينَ ٱتَّخَذُوا۟ مِن دُونِهِۦٓ أَوْلِيَآءَ مَا نَعْبُدُهُمْ إِلَّا لِيُقَرِّبُونَآ إِلَى ٱللَّهِ زُلْفَىٰٓۚ إِنَّ ٱللَّهَ يَحْكُمُ بَيْنَهُمْ فِي مَا هُمْ فِيهِ يَخْتَلِفُونَۚ إِنَّ ٱللَّهَ لَا يَهْدِي مَنْ هُوَ كَـٰذِبٞ كَفَّارٞ ﴿3﴾
خبردار! خالص دین اللہ ہی کے لیے ہے، اور جنہوں نے اس کے سوا دوست بنا رکھے ہیں (کہتے ہیں:) ہم ان کی عبادت صرف اس لیے کرتے ہیں کہ وہ ہمیں اللہ کے قریب کر دیں۔ بے شک اللہ ان کے درمیان فیصلہ کرے گا اس بارے میں جس میں وہ اختلاف کرتے ہیں۔ بے شک اللہ ایسے شخص کو ہدایت نہیں دیتا جو جھوٹا، ناشکرا ہو۔
لَّوْ أَرَادَ ٱللَّهُ أَن يَتَّخِذَ وَلَدٗا لَّصْطَفَىٰ مِمَّا يَخْلُقُ مَا يَشَآءُۚ سُبْحَـٰنَهُۥۖ هُوَ ٱللَّهُ ٱلْوَٰحِدُ ٱلْقَهَّارُ ﴿4﴾
اگر اللہ چاہتا کہ کوئی بیٹا بنائے تو اپنی مخلوق میں سے جسے چاہتا چن لیتا، وہ پاک ہے! وہی اللہ ہے، اکیلا، سب پر غالب۔
خَلَقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ بِٱلْحَقِّۖ يُكَوِّرُ ٱلَّيْلَ عَلَى ٱلنَّهَارِ وَيُكَوِّرُ ٱلنَّهَارَ عَلَى ٱلَّيْلِۖ وَسَخَّرَ ٱلشَّمْسَ وَٱلْقَمَرَۖ كُلّٞ يَجْرِي لِأَجَلٖ مُّسَمًّىۚ أَلَا هُوَ ٱلْعَزِيزُ ٱلْغَفَّـٰرُ ﴿5﴾
اس نے آسمانوں اور زمین کو حق کے ساتھ پیدا کیا، وہ رات کو دن پر لپیٹتا ہے اور دن کو رات پر لپیٹتا ہے، اور سورج اور چاند کو مسخر کر دیا، ہر ایک مقررہ وقت تک چل رہا ہے۔ خبردار! وہی زبردست، بخشنے والا ہے۔
خَلَقَكُم مِّن نَّفْسٖ وَٰحِدَةٖ ثُمَّ جَعَلَ مِنْهَا زَوْجَهَا وَأَنزَلَ لَكُم مِّنَ ٱلْأَنْعَـٰمِ ثَمَـٰنِيَةَ أَزْوَٰجٖۚ يَخْلُقُكُمْ فِي بُطُونِ أُمَّهَـٰتِكُمْ خَلْقٗا مِّنۢ بَعْدِ خَلْقٖ فِي ظُلُمَـٰتٍ ثَلَـٰثٖۚ ذَٰلِكُمُ ٱللَّهُ رَبُّكُمْۖ لَهُ ٱلْمُلْكُۖ لَآ إِلَـٰهَ إِلَّا هُوَۖ فَأَنَّىٰ تُصْرَفُونَ ﴿6﴾
اس نے تمہیں ایک جان سے پیدا کیا، پھر اسی سے اس کا جوڑا بنایا، اور تمہارے لیے چوپایوں میں سے آٹھ جوڑے نازل کیے۔ وہ تمہیں تمہاری ماؤں کے پیٹوں میں تین تاریکیوں میں ایک کے بعد دوسری تخلیق کرتا ہے۔ یہی اللہ تمہارا رب ہے، اسی کے لیے بادشاہی ہے، اس کے سوا کوئی معبود نہیں، پھر تم کہاں بہکائے جا رہے ہو؟
إِن تَكْفُرُوا۟ فَإِنَّ ٱللَّهَ غَنِيٌّ عَنكُمْۖ وَلَا يَرْضَىٰ لِعِبَادِهِ ٱلْكُفْرَۖ وَإِن تَشْكُرُوا۟ يَرْضَهُ لَكُمْۗ وَلَا تَزِرُ وَازِرَةٞ وِزْرَ أُخْرَىٰۚ ثُمَّ إِلَىٰ رَبِّكُمْ مَّرْجِعُكُمْ فَيُنَبِّئُكُم بِمَا كُنتُمْ تَعْمَلُونَۚ إِنَّهُۥ عَلِيمُۢ بِذَاتِ ٱلصُّدُورِ ﴿7﴾
اگر تم کفر کرو تو بے شک اللہ تم سے بے نیاز ہے، اور وہ اپنے بندوں کے لیے کفر کو پسند نہیں کرتا، اور اگر تم شکر کرو تو وہ اسے تمہارے لیے پسند کرتا ہے۔ اور کوئی بوجھ اٹھانے والا کسی دوسرے کا بوجھ نہیں اٹھائے گا۔ پھر تم سب کو اپنے رب کی طرف لوٹنا ہے، پھر وہ تمہیں بتائے گا جو تم کرتے رہے، بے شک وہ سینوں کے بھید جاننے والا ہے۔
وَإِذَا مَسَّ ٱلْإِنسَـٰنَ ضُرّٞ دَعَا رَبَّهُۥ مُنِيبًا إِلَيْهِ ثُمَّ إِذَا خَوَّلَهُۥ نِعْمَةٗ مِّنْهُ نَسِيَ مَا كَانَ يَدْعُوٓا۟ إِلَيْهِ مِن قَبْلُ وَجَعَلَ لِلَّهِ أَندَادٗا لِّيُضِلَّ عَن سَبِيلِهِۦۚ قُلْ تَمَتَّعْ بِكُفْرِكَ قَلِيلًا إِنَّكَ مِنْ أَصْحَـٰبِ ٱلنَّارِ ﴿8﴾
اور جب انسان کو کوئی تکلیف پہنچتی ہے تو وہ اپنے رب کو پکارتا ہے اس کی طرف رجوع کرتے ہوئے، پھر جب وہ اسے اپنی طرف سے کوئی نعمت عطا کرتا ہے تو وہ بھول جاتا ہے جس کے لیے وہ پہلے پکار رہا تھا، اور اللہ کے لیے شریک ٹھہراتا ہے تاکہ اس کے راستے سے گمراہ کرے۔ کہہ دو: اپنے کفر سے تھوڑا فائدہ اٹھا لو، بے شک تم دوزخ والوں میں سے ہو۔
أَمَّنْ هُوَ قَـٰنِتٌ ءَانَآءَ ٱلَّيْلِ سَاجِدٗا وَقَآئِمٗا يَحْذَرُ ٱلْـَٔاخِرَةَ وَيَرْجُوا۟ رَحْمَةَ رَبِّهِۦۗ قُلْ هَلْ يَسْتَوِي ٱلَّذِينَ يَعْلَمُونَ وَٱلَّذِينَ لَا يَعْلَمُونَۗ إِنَّمَا يَتَذَكَّرُ أُو۟لُوا۟ ٱلْأَلْبَـٰبِ ﴿9﴾
بھلا وہ شخص جو رات کے اوقات میں سجدہ اور قیام کی حالت میں عبادت کرتا ہے، آخرت سے ڈرتا ہے اور اپنے رب کی رحمت کی امید رکھتا ہے — (کیا وہ برابر ہے اس کے جو ایسا نہیں کرتا؟) کہہ دو: کیا جاننے والے اور نہ جاننے والے برابر ہو سکتے ہیں؟ نصیحت تو صرف عقل والے ہی حاصل کرتے ہیں۔
قُلْ يَـٰعِبَادِ ٱلَّذِينَ ءَامَنُوا۟ ٱتَّقُوا۟ رَبَّكُمۚ لِلَّذِينَ أَحْسَنُوا۟ فِي هَـٰذِهِ ٱلدُّنْيَا حَسَنَةٞۗ وَأَرْضُ ٱللَّهِ وَٰسِعَةٌۗ إِنَّمَا يُوَفَّى ٱلصَّـٰبِرُونَ أَجْرَهُم بِغَيْرِ حِسَابٖ ﴿10﴾
کہہ دو: اے میرے بندو جو ایمان لائے ہو! اپنے رب سے ڈرو۔ جنہوں نے اس دنیا میں نیکی کی ان کے لیے بھلائی ہے، اور اللہ کی زمین وسیع ہے۔ بے شک صبر کرنے والوں کو ان کا اجر بے حساب دیا جائے گا۔
قُلْ إِنِّيٓ أُمِرْتُ أَنْ أَعْبُدَ ٱللَّهَ مُخْلِصٗا لَّهُ ٱلدِّينَ ﴿11﴾
کہہ دو: مجھے حکم دیا گیا ہے کہ میں اللہ کی عبادت کروں دین کو اسی کے لیے خالص کرتے ہوئے۔
وَأُمِرْتُ لِأَنْ أَكُونَ أَوَّلَ ٱلْمُسْلِمِينَ ﴿12﴾
اور مجھے حکم دیا گیا ہے کہ میں سب سے پہلا فرمانبردار بنوں۔
قُلْ إِنِّيٓ أَخَافُ إِنْ عَصَيْتُ رَبِّي عَذَابَ يَوْمٍ عَظِيمٍۢ ﴿13﴾
کہہ دو: اگر میں اپنے رب کی نافرمانی کروں تو مجھے ایک بڑے دن کے عذاب کا خوف ہے۔
قُلِ ٱللَّهَ أَعْبُدُ مُخْلِصٗا لَّهُۥ دِينِي ﴿14﴾
کہہ دو: میں اللہ ہی کی عبادت کرتا ہوں، دین کو اسی کے لیے خالص کرتے ہوئے۔
فَٱعْبُدُوا۟ مَا شِئْتُم مِّن دُونِهِۦۗ قُلْ إِنَّ ٱلْخَـٰسِرِينَ ٱلَّذِينَ خَسِرُوٓا۟ أَنفُسَهُمْ وَأَهْلِيهِمْ يَوْمَ ٱلْقِيَـٰمَةِۗ أَلَا ذَٰلِكَ هُوَ ٱلْخُسْرَانُ ٱلْمُبِينُ ﴿15﴾
پس تم اس کے سوا جس کی چاہو عبادت کرو۔ کہہ دو: بے شک خسارہ اٹھانے والے وہ ہیں جنہوں نے قیامت کے دن اپنے آپ کو اور اپنے گھر والوں کو نقصان میں ڈال دیا۔ خبردار! یہی کھلا خسارہ ہے۔
لَهُم مِّن فَوْقِهِمْ ظُلَلٞ مِّنَ ٱلنَّارِ وَمِن تَحْتِهِمْ ظُلَلٞۚ ذَٰلِكَ يُخَوِّفُ ٱللَّهُ بِهِۦ عِبَادَهُۥۚ يَـٰعِبَادِ فَٱتَّقُونِ ﴿16﴾
ان کے اوپر اور نیچے آگ کے سائبان ہوں گے۔ اللہ اپنے بندوں کو اسی سے ڈراتا ہے۔ پس اے میرے بندو! مجھ سے ڈرو۔
وَٱلَّذِينَ ٱجْتَنَبُوا۟ ٱلطَّـٰغُوتَ أَن يَعْبُدُوهَا وَأَنَابُوٓا۟ إِلَى ٱللَّهِ لَهُمُ ٱلْبُشْرَىٰۚ فَبَشِّرْ عِبَادِ ﴿17﴾
اور جنہوں نے طاغوت کی عبادت سے اجتناب کیا اور اللہ کی طرف رجوع کیا، ان کے لیے خوشخبری ہے۔ پس میرے بندوں کو خوشخبری دے دو۔
ٱلَّذِينَ يَسْتَمِعُونَ ٱلْقَوْلَ فَيَتَّبِعُونَ أَحْسَنَهُۥٓۚ أُو۟لَـٰٓئِكَ ٱلَّذِينَ هَدَىٰهُمُ ٱللَّهُۖ وَأُو۟لَـٰٓئِكَ هُمْ أُو۟لُوا۟ ٱلْأَلْبَـٰبِ ﴿18﴾
جو لوگ بات کو غور سے سنتے ہیں اور اس کے بہترین پہلو کی پیروی کرتے ہیں، یہی وہ لوگ ہیں جنہیں اللہ نے ہدایت دی، اور یہی عقل والے ہیں۔
أَفَمَنْ حَقَّ عَلَيْهِ كَلِمَةُ ٱلْعَذَابِ أَفَأَنتَ تُنقِذُ مَنْ فِي ٱلنَّارِ ﴿19﴾
بھلا جس پر عذاب کا فیصلہ ہو چکا ہو، کیا تم اسے آگ سے بچا سکتے ہو؟
لَـٰكِنِ ٱلَّذِينَ ٱتَّقَوْا۟ رَبَّهُمْ لَهُمْ غُرَفٞ مِّن فَوْقِهَا غُرَفٞ مَّبْنِيَّةٞ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُۖ وَعْدَ ٱللَّهِۖ لَا يُخْلِفُ ٱللَّهُ ٱلْمِيعَادَ ﴿20﴾
لیکن جو لوگ اپنے رب سے ڈرتے رہے، ان کے لیے اونچی منزلیں ہیں جن کے اوپر اور بھی منزلیں بنی ہوئی ہیں، جن کے نیچے نہریں بہتی ہیں۔ یہ اللہ کا وعدہ ہے، اور اللہ وعدہ خلافی نہیں کرتا۔
أَلَمْ تَرَ أَنَّ ٱللَّهَ أَنزَلَ مِنَ ٱلسَّمَآءِ مَآءٗ فَسَلَكَهُۥ يَنَـٰبِيعَ فِي ٱلْأَرْضِ ثُمَّ يُخْرِجُ بِهِۦ زَرْعٗا مُّخْتَلِفًا أَلْوَٰنُهُۥ ثُمَّ يَهِيجُ فَتَرَىٰهُ مُصْفَرّٗا ثُمَّ يَجْعَلُهُۥ حُطَـٰمًاۚ إِنَّ فِي ذَٰلِكَ لَذِكْرَىٰ لِأُو۟لِي ٱلْأَلْبَـٰبِ ﴿21﴾
کیا تم نے نہیں دیکھا کہ اللہ نے آسمان سے پانی نازل کیا، پھر اسے زمین میں چشموں کی صورت میں داخل کیا، پھر اس سے مختلف رنگوں کی کھیتی نکالتا ہے، پھر وہ سوکھ جاتی ہے، پھر تم اسے زرد دیکھتے ہو، پھر وہ اسے چورا چورا کر دیتا ہے۔ بے شک اس میں عقل والوں کے لیے نصیحت ہے۔
أَفَمَنْ شَرَحَ ٱللَّهُ صَدْرَهُۥ لِلْإِسْلَـٰمِ فَهُوَ عَلَىٰ نُورٖ مِّن رَّبِّهِۦۚ فَوَيْلٞ لِّلْقَـٰسِيَةِ قُلُوبُهُم مِّن ذِكْرِ ٱللَّهِۚ أُو۟لَـٰٓئِكَ فِي ضَلَـٰلٖ مُّبِينٍ ﴿22﴾
کیا وہ شخص جس کا سینہ اللہ نے اسلام کے لیے کھول دیا ہو اور وہ اپنے رب کی طرف سے نور پر ہو؟ تو ایسے لوگوں کے لیے ہلاکت ہے جن کے دل اللہ کی یاد سے سخت ہو گئے ہیں۔ یہی لوگ کھلی گمراہی میں ہیں۔
ٱللَّهُ نَزَّلَ أَحْسَنَ ٱلْحَدِيثِ كِتَـٰبٗا مُّتَشَـٰبِهٗا مَّثَانِيَ تَقْشَعِرُّ مِنْهُ جُلُودُ ٱلَّذِينَ يَخْشَوْنَ رَبَّهُمْ ثُمَّ تَلِينُ جُلُودُهُمْ وَقُلُوبُهُمْ إِلَىٰ ذِكْرِ ٱللَّهِۚ ذَٰلِكَ هُدَى ٱللَّهِ يَهْدِي بِهِۦ مَن يَشَآءُۚ وَمَن يُضْلِلِ ٱللَّهُ فَمَا لَهُۥ مِنْ هَادٍ ﴿23﴾
اللہ نے بہترین بات نازل کی ہے، ایک ایسی کتاب جو ہم رنگ اور بار بار دہرائی گئی ہے، جس سے ان لوگوں کے جسم کانپ اٹھتے ہیں جو اپنے رب سے ڈرتے ہیں، پھر ان کے جسم اور دل اللہ کی یاد کی طرف نرم ہو جاتے ہیں۔ یہ اللہ کی ہدایت ہے، وہ جسے چاہے اس سے ہدایت دیتا ہے، اور جسے اللہ گمراہ کر دے تو اس کے لیے کوئی ہدایت دینے والا نہیں۔
أَفَمَن يَتَّقِي بِوَجْهِهِۦ سُوٓءَ ٱلْعَذَابِ يَوْمَ ٱلْقِيَـٰمَةِۚ وَقِيلَ لِلظَّـٰلِمِينَ ذُوقُوا۟ مَا كُنتُمْ تَكْسِبُونَ ﴿24﴾
کیا وہ شخص جو قیامت کے دن اپنے چہرے سے برے عذاب کو روکے گا؟ اور ظالموں سے کہا جائے گا: چکھو وہ جو تم کماتے رہے۔
كَذَّبَ ٱلَّذِينَ مِن قَبْلِهِمْ فَأَتَىٰهُمُ ٱلْعَذَابُ مِنْ حَيْثُ لَا يَشْعُرُونَ ﴿25﴾
ان سے پہلے والوں نے بھی جھٹلایا تھا، تو ان پر عذاب ایسی جگہ سے آیا جہاں سے انہیں گمان بھی نہ تھا۔
فَأَذَاقَهُمُ ٱللَّهُ ٱلْخِزْيَ فِي ٱلْحَيَوٰةِ ٱلدُّنْيَاۖ وَلَعَذَابُ ٱلْآخِرَةِ أَكْبَرُۚ لَوْ كَانُوا۟ يَعْلَمُونَ ﴿26﴾
تو اللہ نے انہیں دنیا کی زندگی میں ذلت کا مزہ چکھایا، اور آخرت کا عذاب تو بہت بڑا ہے، کاش وہ جانتے۔
وَلَقَدْ ضَرَبْنَا لِلنَّاسِ فِي هَـٰذَا ٱلْقُرْءَانِ مِن كُلِّ مَثَلٖ لَّعَلَّهُمْ يَتَذَكَّرُونَ ﴿27﴾
اور بے شک ہم نے اس قرآن میں لوگوں کے لیے ہر قسم کی مثال بیان کی ہے تاکہ وہ نصیحت حاصل کریں۔
قُرْآنًا عَرَبِيّٗا غَيْرَ ذِي عِوَجٖ لَّعَلَّهُمْ يَتَّقُونَ ﴿28﴾
یہ قرآن عربی زبان میں ہے، جس میں کوئی کجی نہیں، تاکہ وہ تقویٰ اختیار کریں۔
ضَرَبَ ٱللَّهُ مَثَلٗا رَّجُلٗا فِيهِ شُرَكَآءُ مُتَشَـٰكِسُونَ وَرَجُلٗا سَلَمٗا لِّرَجُلٍۢ هَلْ يَسْتَوِيَانِ مَثَلًاۚ ٱلْحَمْدُ لِلَّهِۚ بَلْ أَكْثَرُهُمْ لَا يَعْلَمُونَ ﴿29﴾
اللہ نے ایک مثال بیان کی: ایک شخص جس کے کئی شریک ہیں جو آپس میں جھگڑنے والے ہیں، اور ایک شخص جو صرف ایک مالک کے تابع ہے — کیا یہ دونوں مثال میں برابر ہو سکتے ہیں؟ تمام تعریف اللہ کے لیے ہے، بلکہ ان میں سے اکثر نہیں جانتے۔
إِنَّكَ مَيِّتٞ وَإِنَّهُم مَّيِّتُونَ ﴿30﴾
بے شک آپ فوت ہونے والے ہیں اور وہ بھی فوت ہونے والے ہیں۔
ثُمَّ إِنَّكُمْ يَوْمَ ٱلْقِيَـٰمَةِ عِندَ رَبِّكُمْ تَخْتَصِمُونَ ﴿31﴾
پھر قیامت کے دن تم سب اپنے رب کے سامنے جھگڑو گے۔
