سورة الشعراء

بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
طٰسٓمٓ﴿1﴾
طسم
تِلْكَ آيَاتُ ٱلْكِتَٰبِ ٱلْمُبِينِ ﴿2﴾
یہ روشن کتاب کی آیات ہیں۔
لَعَلَّكَ بَـٰخِعٌۭ نَّفْسَكَ أَلَّا يَكُونُوا۟ مُؤْمِنِينَ ﴿3﴾
شاید آپ اپنی جان ہلاک کر ڈالیں اس غم میں کہ وہ ایمان نہیں لاتے۔
إِن نَّشَأْ نُنَزِّلْ عَلَيْهِم مِّنَ ٱلسَّمَآءِ ءَايَةٗ فَظَلَّتْ أَعْنَـٰقُهُمْ لَهَا خَـٰضِعِينَ ﴿4﴾
اگر ہم چاہیں تو ان پر آسمان سے کوئی نشانی نازل کر دیں، تو ان کی گردنیں اس کے سامنے جھک جائیں۔
وَمَا يَأْتِيهِم مِّن ذِكْرٖ مِّنَ ٱلرَّحْمَـٰنِ مُحْدَثٍ إِلَّا كَانُوا۟ عَنْهُ مُعْرِضِينَ ﴿5﴾
اور ان کے پاس رحمٰن کی طرف سے کوئی نئی نصیحت نہیں آتی مگر وہ اس سے منہ پھیر لیتے ہیں۔
فَقَدْ كَذَّبُوا۟ فَسَيَأْتِيهِمْ أَنۢبَـٰٓؤُا۟ مَا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ ﴿6﴾
پس انہوں نے جھٹلایا، تو جلد ہی ان کے پاس اس چیز کی خبریں آئیں گی جس کا وہ مذاق اڑاتے تھے۔
أَوَلَمْ يَرَوْا۟ إِلَى ٱلْأَرْضِ كَمْ أَنۢبَتْنَا فِيهَا مِن كُلِّ زَوْجٖ كَرِيمٍ ﴿7﴾
کیا انہوں نے زمین کی طرف نہیں دیکھا کہ ہم نے اس میں ہر قسم کی نفیس چیزیں اگائیں؟
إِنَّ فِي ذَٰلِكَ لَـَٔايَةٗۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿8﴾
بے شک اس میں ایک نشانی ہے، مگر ان میں سے اکثر ایمان نہیں لاتے۔
وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿9﴾
اور بے شک آپ کا رب زبردست، رحم والا ہے۔
وَإِذْ نَادَىٰ رَبُّكَ مُوسَىٰٓ أَنِ ٱئْتِ ٱلْقَوْمَ ٱلظَّـٰلِمِينَ ﴿10﴾
اور جب آپ کے رب نے موسیٰ کو پکارا کہ ظالم قوم کے پاس جاؤ۔
قَوْمَ فِرْعَوْنَ ۚ أَلَا يَتَّقُونَ ﴿11﴾
فرعون کی قوم کے پاس (جاؤ)، کیا وہ ڈرتے نہیں؟
قَالَ رَبِّ إِنِّيٓ أَخَافُ أَن يُكَذِّبُونِ ﴿12﴾
اس نے کہا: اے میرے رب! مجھے اندیشہ ہے کہ وہ مجھے جھٹلائیں گے۔
وَيَضِيقُ صَدْرِي وَلَا يَنطَلِقُ لِسَانِي فَأَرْسِلْ إِلَىٰ هَـٰرُونَ ﴿13﴾
اور میرا سینہ تنگ ہوتا ہے اور میری زبان نہیں چلتی، پس ہارون کو بھیج دے۔
وَلَهُمْ عَلَيَّ ذَنبٞ فَأَخَافُ أَن يَقْتُلُونِ ﴿14﴾
اور ان پر میرا ایک گناہ ہے، پس مجھے ڈر ہے کہ وہ مجھے قتل کر دیں گے۔
قَالَ كَلَّاۖ فَٱذْهَبَا بِـَٔايَـٰتِنَآۖ إِنَّا مَعَكُم مُّسْتَمِعُونَ ﴿15﴾
فرمایا: ہرگز نہیں، پس تم دونوں ہماری نشانیاں لے کر جاؤ، ہم تمہارے ساتھ سننے والے ہیں۔
فَأْتِيَا فِرْعَوْنَ فَقُولَآ إِنَّا رَسُولُ رَبِّ ٱلْعَـٰلَمِينَ ﴿16﴾
پس تم دونوں فرعون کے پاس جاؤ اور کہو: ہم رب العالمین کے بھیجے ہوئے ہیں۔
أَنْ أَرْسِلْ مَعَنَا بَنِيٓ إِسْرَٰٓءِيلَ ﴿17﴾
کہ تو بنی اسرائیل کو ہمارے ساتھ بھیج دے۔
قَالَ أَلَمْ نُرَبِّكَ فِينَا وَلِيدٗا وَلَبِثْتَ فِينَا مِنْ عُمُرِكَ سِنِينَ ﴿18﴾
فرعون نے کہا: کیا ہم نے تجھے بچپن میں اپنے ہاں نہیں پالا، اور تو نے اپنی عمر کے کئی سال ہمارے درمیان نہیں گزارے؟
وَفَعَلْتَ فَعْلَتَكَ ٱلَّتِي فَعَلْتَ وَأَنتَ مِنَ ٱلْكَـٰفِرِينَ ﴿19﴾
اور تو نے وہ کام کیا جو کیا، اور تو ناشکروں میں سے تھا۔
قَالَ فَعَلْتُهَآ إِذٗا وَأَنَا۠ مِنَ ٱلضَّآلِّينَ ﴿20﴾
موسیٰ نے کہا: میں نے وہ کام اس وقت کیا جب میں بھٹکا ہوا تھا۔
فَفَرَرْتُ مِنكُمْ لَمَّا خِفْتُكُمْ فَوَهَبَ لِي رَبِّي حُكْمٗا وَجَعَلَنِي مِنَ ٱلْمُرْسَلِينَ ﴿21﴾
پس میں تم سے بھاگ گیا جب مجھے تمہارا خوف ہوا، پھر میرے رب نے مجھے حکم (نبوت) عطا کیا اور مجھے رسولوں میں شامل کر دیا۔
وَتِلْكَ نِعْمَةٞ تَمُنُّهَا عَلَيَّ أَنْ عَبَّدتَ بَنِيٓ إِسْرَٰٓءِيلَ ﴿22﴾
اور کیا یہ وہ احسان ہے جو تو مجھ پر جتاتا ہے کہ تو نے بنی اسرائیل کو غلام بنا رکھا ہے؟
قَالَ فِرْعَوْنُ وَمَا رَبُّ ٱلْعَـٰلَمِينَ ﴿23﴾
فرعون نے کہا: رب العالمین کون ہے؟
قَالَ رَبُّ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَآۖ إِن كُنتُم مُّوقِنِينَ ﴿24﴾
موسیٰ نے کہا: آسمانوں اور زمین اور ان کے درمیان کی چیزوں کا رب، اگر تم یقین رکھتے ہو۔
قَالَ لِمَنْ حَوْلَهُۥٓ أَلَا تَسْتَمِعُونَ ﴿25﴾
فرعون نے اپنے اردگرد والوں سے کہا: کیا تم نہیں سن رہے؟
قَالَ رَبُّكُمْ وَرَبُّ ءَابَآئِكُمُ ٱلْأَوَّلِينَ ﴿26﴾
موسیٰ نے کہا: تمہارا رب اور تمہارے پہلے باپ دادا کا رب۔
قَالَ إِنَّ رَسُولَكُمُ ٱلَّذِيٓ أُرْسِلَ إِلَيْكُمْ لَمَجْنُونٞ ﴿27﴾
فرعون نے کہا: بے شک تمہارا رسول جو تمہاری طرف بھیجا گیا ہے، دیوانہ ہے۔
قَالَ رَبُّ ٱلْمَشْرِقِ وَٱلْمَغْرِبِ وَمَا بَيْنَهُمَآۖ إِن كُنتُمْ تَعْقِلُونَ ﴿28﴾
موسیٰ نے کہا: مشرق و مغرب اور ان کے درمیان کی چیزوں کا رب، اگر تم عقل رکھتے ہو۔
قَالَ لَئِنِ ٱتَّخَذْتَ إِلَـٰهًا غَيْرِي لَأَجْعَلَنَّكَ مِنَ ٱلْمَسْجُونِينَ ﴿29﴾
فرعون نے کہا: اگر تو نے میرے سوا کسی اور کو معبود بنایا تو میں تجھے قیدیوں میں شامل کر دوں گا۔
قَالَ أَوَلَوْ جِئْتُكَ بِشَيْءٖ مُّبِينٖ ﴿30﴾
موسیٰ نے کہا: اگر میں تیرے پاس کوئی واضح چیز لے آؤں (تو بھی مجھے قید کرے گا؟)
قَالَ فَأْتِ بِهِۦٓ إِن كُنتَ مِنَ ٱلصَّـٰدِقِينَ ﴿31﴾
فرعون نے کہا: اگر تو سچا ہے تو وہ (نشانی) لے آ۔
فَأَلْقَىٰ عَصَاهُ فَإِذَا هِيَ ثُعْبَانٞ مُّبِينٞ ﴿32﴾
پس موسیٰ نے اپنی لاٹھی پھینکی تو وہ ایک ظاہر اژدہا بن گئی۔
وَنَزَعَ يَدَهُۥ فَإِذَا هِيَ بَيْضَآءُ لِلنَّـٰظِرِينَ ﴿33﴾
اور اس نے اپنا ہاتھ نکالا تو وہ دیکھنے والوں کے لیے چمکتا ہوا سفید تھا۔
قَالَ لِلْمَلَإِ حَوْلَهُۥٓ إِنَّ هَـٰذَا لَسَـٰحِرٌ عَلِيمٞ ﴿34﴾
فرعون نے اپنے اردگرد کے سرداروں سے کہا: بے شک یہ تو بڑا ماہر جادوگر ہے۔
يُرِيدُ أَن يُخْرِجَكُم مِّنْ أَرْضِكُم بِسِحْرِهِۦ فَمَاذَا تَأْمُرُونَ ﴿35﴾
یہ چاہتا ہے کہ اپنے جادو کے ذریعے تمہیں تمہاری زمین سے نکال دے، تو تم کیا حکم دیتے ہو؟
قَالُوٓا۟ أَرْجِهْ وَأَخَاهُ وَٱبْعَثْ فِي ٱلْمَدَآئِنِ حَـٰشِرِينَ ﴿36﴾
انہوں نے کہا: اسے اور اس کے بھائی کو مہلت دے، اور شہروں میں جمع کرنے والے بھیج دے۔
يَأْتُوكَ بِكُلِّ سَحَّارٍ عَلِيمٍۢ ﴿37﴾
وہ تیرے پاس ہر بڑے ماہر جادوگر کو لے آئیں گے۔
فَجُمِعَ ٱلسَّحَرَةُ لِمِيقَـٰتِ يَوْمٖ مَّعْلُومٖ ﴿38﴾
پس جادوگر ایک مقرر دن کے وقت جمع کیے گئے۔
وَقِيلَ لِلنَّاسِ هَلْ أَنتُم مُّجْتَمِعُونَ ﴿39﴾
اور لوگوں سے کہا گیا: کیا تم سب جمع ہو رہے ہو؟
لَعَلَّنَا نَتَّبِعُ ٱلسَّحَرَةَ إِن كَانُوا۟ هُمُ ٱلْغَـٰلِبِينَ ﴿40﴾
شاید ہم جادوگروں کی پیروی کریں اگر وہ غالب آ جائیں۔
فَلَمَّا جَآءَ ٱلسَّحَرَةُ قَالُوا۟ لِفِرْعَوْنَ أَإِنَّ لَنَا لَأَجْرًا إِن كُنَّا نَحْنُ ٱلْغَـٰلِبِينَ ﴿41﴾
پھر جب جادوگر آئے تو انہوں نے فرعون سے کہا: اگر ہم غالب آ گئے تو کیا ہمیں کوئی انعام ملے گا؟
قَالَ نَعَمْ وَإِنَّكُمْ إِذٗا لَّمِنَ ٱلْمُقَرَّبِينَ ﴿42﴾
اس نے کہا: ہاں، اور تم اس وقت مقربین میں سے ہو گے۔
قَالَ لَهُمْ مُوسَىٰٓ أَلْقُوا۟ مَآ أَنتُم مُّلْقُونَ ﴿43﴾
موسیٰ نے ان سے کہا: جو کچھ تم ڈالنے والے ہو، ڈال دو۔
فَأَلْقَوْا۟ حِبَالَهُمْ وَعِصِيَّهُمْ وَقَالُوا۟ بِعِزَّةِ فِرْعَوْنَ إِنَّا لَنَحْنُ ٱلْغَـٰلِبُونَ ﴿44﴾
پس انہوں نے اپنی رسیاں اور لاٹھیاں ڈالیں اور کہا: فرعون کی عزت کی قسم! ہم ہی غالب آنے والے ہیں۔
فَأَلْقَىٰ مُوسَىٰ عَصَاهُ فَإِذَا هِيَ تَلْقَفُ مَا يَأْفِكُونَ ﴿45﴾
پھر موسیٰ نے اپنی لاٹھی پھینکی تو وہ ان کی جھوٹی چیزوں کو نگلنے لگی۔
فَأُلْقِيَ ٱلسَّحَرَةُ سَـٰجِدِينَ ﴿46﴾
پس جادوگر سجدے میں گر پڑے۔
قَالُوٓا۟ ءَامَنَّا بِرَبِّ ٱلْعَـٰلَمِينَ ﴿47﴾
انہوں نے کہا: ہم رب العالمین پر ایمان لے آئے۔
رَبِّ مُوسَىٰ وَهَـٰرُونَ ﴿48﴾
موسیٰ اور ہارون کے رب پر۔
قَالَ ءَامَنتُمْ لَهُۥ قَبْلَ أَنْ ءَاذَنَ لَكُمْۖ إِنَّهُۥ لَكَبِيرُكُمُ ٱلَّذِي عَلَّمَكُمُ ٱلسِّحْرَۖ فَلَسَوْفَ تَعْلَمُونَۖ لَأُقَطِّعَنَّ أَيْدِيَكُمْ وَأَرْجُلَكُم مِّنْ خِلَـٰفٖ وَلَأُصَلِّبَنَّكُمْ أَجْمَعِينَ ﴿49﴾
فرعون نے کہا: تم اس پر ایمان لے آئے قبل اس کے کہ میں تمہیں اجازت دوں؟ بے شک وہ تمہارا بڑا ہے جس نے تمہیں جادو سکھایا، پس تم ضرور جان لوگے، میں تمہارے ہاتھ اور پاؤں مخالف سمتوں سے کاٹوں گا اور تم سب کو سولی پر چڑھاؤں گا۔
قَالُوا۟ لَا ضَيْرَ إِنَّآ إِلَىٰ رَبِّنَا مُنقَلِبُونَ ﴿50﴾
انہوں نے کہا: کوئی حرج نہیں، ہم تو اپنے رب کی طرف لوٹنے والے ہیں۔
إِنَّا نَطْمَعُ أَن يَغْفِرَ لَنَا رَبُّنَا خَطَـٰيَـٰنَآ أَن كُنَّآ أَوَّلَ ٱلْمُؤْمِنِينَ ﴿51﴾
ہم امید رکھتے ہیں کہ ہمارا رب ہماری خطائیں بخش دے گا، کیونکہ ہم سب سے پہلے ایمان لانے والے ہیں۔
وَأَوْحَيْنَآ إِلَىٰ مُوسَىٰٓ أَنْ أَسْرِ بِعِبَادِيٓ إِنَّكُم مُّتَّبَعُونَ ﴿52﴾
اور ہم نے موسیٰ کو وحی کی کہ میرے بندوں کو لے کر نکل جاؤ، بے شک تمہارا پیچھا کیا جائے گا۔
فَأَرْسَلَ فِرْعَوْنُ فِي ٱلْمَدَآئِنِ حَـٰشِرِينَ ﴿53﴾
تو فرعون نے شہروں میں جمع کرنے والے بھیجے۔
إِنَّ هَـٰٓؤُلَآءِ لَشِرْذِمَةٞ قَلِيلُونَ ﴿54﴾
بے شک یہ تو تھوڑے سے لوگ ہیں۔
وَإِنَّهُمْ لَنَا لَغَآئِظُونَ ﴿55﴾
اور بے شک یہ ہمیں غصہ دلا رہے ہیں۔
وَإِنَّا لَجَمِيعٌ حَـٰذِرُونَ ﴿56﴾
اور ہم سب ہوشیار رہنے والے ہیں۔
فَأَخْرَجْنَـٰهُم مِّن جَنَّـٰتٖ وَعُيُونٖ ﴿57﴾
پس ہم نے انہیں باغوں اور چشموں سے نکال دیا۔
وَكُنُوزٖ وَمَقَامٖ كَرِيمٖ ﴿58﴾
اور خزانے اور عزت والے مقام سے۔
كَذَٰلِكَۖ وَأَوْرَثْنَـٰهَا بَنِيٓ إِسْرَٰٓءِيلَ ﴿59﴾
اسی طرح ہوا، اور ہم نے بنی اسرائیل کو ان کا وارث بنا دیا۔
فَأَتْبَعُوهُم مُّشْرِقِينَ ﴿60﴾
تو وہ ان کے پیچھے سورج نکلتے وقت نکلے۔
فَلَمَّا تَرَآءَا ٱلْجَمْعَانِ قَالَ أَصْحَـٰبُ مُوسَىٰٓ إِنَّا لَمُدْرَكُونَ ﴿61﴾
پھر جب دونوں گروہ آمنے سامنے ہوئے تو موسیٰ کے ساتھیوں نے کہا: ہم تو پکڑے گئے۔
قَالَ كَلَّآۖ إِنَّ مَعِيَ رَبِّي سَيَهْدِينِ ﴿62﴾
موسیٰ نے کہا: ہرگز نہیں، بے شک میرے ساتھ میرا رب ہے، وہ مجھے راستہ دکھائے گا۔
فَأَوْحَيْنَآ إِلَىٰ مُوسَىٰٓ أَنِ ٱضْرِب بِّعَصَاكَ ٱلْبَحْرَۖ فَٱنفَلَقَ فَكَانَ كُلُّ فِرْقٖ كَٱلطَّوْدِ ٱلْعَظِيمِ ﴿63﴾
تو ہم نے موسیٰ کو وحی کی کہ اپنی لاٹھی سمندر پر مار، تو وہ پھٹ گیا، اور ہر حصہ بڑے پہاڑ کی طرح ہو گیا۔
وَأَزْلَفْنَا ثَمَّ ٱلْـَٔاخَرِينَ ﴿64﴾
اور ہم نے وہاں دوسرے گروہ کو قریب کر دیا۔
وَأَنجَيْنَا مُوسَىٰ وَمَن مَّعَهُۥٓ أَجْمَعِينَ ﴿65﴾
اور ہم نے موسیٰ اور اس کے ساتھیوں کو سب کو نجات دی۔
ثُمَّ أَغْرَقْنَا ٱلْـَٔاخَرِينَ ﴿66﴾
پھر ہم نے دوسرے گروہ کو غرق کر دیا۔
إِنَّ فِي ذَٰلِكَ لَـَٔايَةٗۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿67﴾
بے شک اس میں ایک نشانی ہے، مگر ان میں سے اکثر ایمان نہیں لاتے۔
وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿68﴾
اور بے شک آپ کا رب زبردست، رحم والا ہے۔
وَٱتْلُ عَلَيْهِمْ نَبَأَ إِبْرَٰهِيمَ ﴿69﴾
اور انہیں ابراہیم کی خبر سناؤ۔
إِذْ قَالَ لِأَبِيهِ وَقَوْمِهِۦ مَا تَعْبُدُونَ ﴿70﴾
جب اس نے اپنے باپ اور قوم سے کہا: تم کس چیز کی عبادت کرتے ہو؟
قَالُوا۟ نَعْبُدُ أَصْنَامٗا فَنَظَلُّ لَهَا عَـٰكِفِينَ ﴿71﴾
انہوں نے کہا: ہم بتوں کی عبادت کرتے ہیں، اور ان پر جمے رہتے ہیں۔
قَالَ هَلْ يَسْمَعُونَكُمْ إِذْ تَدْعُونَ ﴿72﴾
اس نے کہا: کیا وہ تمہیں سنتے ہیں جب تم انہیں پکارتے ہو؟
أَوْ يَنفَعُونَكُمْ أَوْ يَضُرُّونَ ﴿73﴾
یا وہ تمہیں نفع دیتے ہیں یا نقصان پہنچاتے ہیں؟
قَالُوا۟ بَلْ وَجَدْنَآ ءَابَآءَنَا كَذَٰلِكَ يَفْعَلُونَ ﴿74﴾
انہوں نے کہا: بلکہ ہم نے اپنے باپ دادا کو اسی طرح کرتے پایا۔
قَالَ أَفَرَءَيْتُم مَّا كُنتُمْ تَعْبُدُونَ ﴿75﴾
اس نے کہا: تو کیا تم نے دیکھا ان چیزوں کو جن کی تم عبادت کرتے ہو؟
أَنتُمْ وَءَابَآؤُكُمُ ٱلْأَقْدَمُونَ ﴿76﴾
تم اور تمہارے پہلے باپ دادا۔
فَإِنَّهُمْ عَدُوّٞ لِّيٓ إِلَّا رَبَّ ٱلْعَـٰلَمِينَ ﴿77﴾
تو بے شک وہ میرے دشمن ہیں، سوائے رب العالمین کے۔
ٱلَّذِي خَلَقَنِي فَهُوَ يَهْدِينِ ﴿78﴾
جس نے مجھے پیدا کیا، پھر وہی مجھے ہدایت دیتا ہے۔
وَٱلَّذِي هُوَ يُطْعِمُنِي وَيَسْقِينِ ﴿79﴾
اور وہی مجھے کھلاتا ہے اور پلاتا ہے۔
وَإِذَا مَرِضْتُ فَهُوَ يَشْفِينِ ﴿80﴾
اور جب میں بیمار ہوتا ہوں تو وہی مجھے شفا دیتا ہے۔
وَٱلَّذِي يُمِيتُنِي ثُمَّ يُحْيِينِ ﴿81﴾
اور وہی مجھے موت دیتا ہے، پھر زندہ کرتا ہے۔
وَٱلَّذِيٓ أَطْمَعُ أَن يَغْفِرَ لِي خَطِيٓـَٔتِي يَوْمَ ٱلدِّينِ ﴿82﴾
اور وہی جس سے مجھے امید ہے کہ وہ قیامت کے دن میری خطا کو بخش دے گا۔
رَبِّ هَبْ لِي حُكْمٗا وَأَلْحِقْنِي بِٱلصَّـٰلِحِينَ ﴿83﴾
اے میرے رب! مجھے حکمت عطا فرما اور مجھے نیک لوگوں کے ساتھ ملا دے۔
وَٱجْعَل لِّي لِسَانَ صِدْقٖ فِي ٱلْـَٔاخِرِينَ ﴿84﴾
اور میرے لیے بعد والوں میں سچی شہرت رکھ۔
وَٱجْعَلْنِي مِن وَرَثَةِ جَنَّةِ ٱلنَّعِيمِ ﴿85﴾
اور مجھے نعمتوں والی جنت کا وارث بنا۔
وَٱغْفِرْ لِأَبِيٓ إِنَّهُۥ كَانَ مِنَ ٱلضَّآلِّينَ ﴿86﴾
اور میرے باپ کو بخش دے، بے شک وہ گمراہوں میں سے تھا۔
وَلَا تُخْزِنِي يَوْمَ يُبْعَثُونَ ﴿87﴾
اور مجھے اس دن رسوا نہ کرنا جب سب اٹھائے جائیں گے۔
يَوْمَ لَا يَنفَعُ مَالٞ وَلَا بَنُونَ ﴿88﴾
جس دن مال اور اولاد کچھ فائدہ نہ دیں گے۔
إِلَّا مَنْ أَتَى ٱللَّهَ بِقَلْبٖ سَلِيمٖ ﴿89﴾
سوائے اس کے جو اللہ کے پاس پاک دل لے کر آئے۔
وَأُزْلِفَتِ ٱلْجَنَّةُ لِلْمُتَّقِينَ ﴿90﴾
اور جنت پرہیزگاروں کے قریب کر دی جائے گی۔
وَبُرِّزَتِ ٱلْجَحِيمُ لِلْغَاوِينَ ﴿91﴾
اور جہنم گمراہوں کے سامنے ظاہر کر دی جائے گی۔
وَقِيلَ لَهُمْ أَيْنَ مَا كُنتُمْ تَعْبُدُونَ ﴿92﴾
اور ان سے کہا جائے گا: کہاں ہیں وہ جن کی تم عبادت کرتے تھے؟
مِن دُونِ ٱللَّهِ ۖ هَلْ يَنصُرُونَكُمْ أَوْ يَنتَصِرُونَ ﴿93﴾
اللہ کے سوا، کیا وہ تمہاری مدد کر سکتے ہیں یا خود بچ سکتے ہیں؟
فَكُبْكِبُوا۟ فِيهَا هُمْ وَٱلْغَاوُۥنَ ﴿94﴾
پس وہ سب اس میں اوندھے ڈال دیے جائیں گے، وہ اور گمراہ لوگ۔
وَجُنُودُ إِبْلِيسَ أَجْمَعُونَ ﴿95﴾
اور ابلیس کے تمام لشکر۔
قَالُوا۟ وَهُمْ فِيهَا يَخْتَصِمُونَ ﴿96﴾
وہ اس میں جھگڑتے ہوئے کہیں گے۔
تَٱللَّهِ إِن كُنَّا لَفِي ضَلَـٰلٖ مُّبِينٍ ﴿97﴾
اللہ کی قسم! ہم تو کھلی گمراہی میں تھے۔
إِذْ نُسَوِّيكُم بِرَبِّ ٱلْعَـٰلَمِينَ ﴿98﴾
جب ہم تمہیں رب العالمین کے برابر ٹھہراتے تھے۔
وَمَآ أَضَلَّنَآ إِلَّا ٱلْمُجْرِمُونَ ﴿99﴾
اور ہمیں تو صرف مجرموں نے گمراہ کیا۔
فَمَا لَنَا مِن شَـٰفِعِينَ ﴿100﴾
تو ہمارے لیے کوئی سفارش کرنے والا نہیں۔
وَلَا صَدِيقٍ حَمِيمٖ ﴿101﴾
اور نہ کوئی گرمجوش دوست۔
فَلَوْ أَنَّ لَنَا كَرَّةٗ فَنَكُونَ مِنَ ٱلْمُؤْمِنِينَ ﴿102﴾
کاش ہمیں ایک بار لوٹنے کا موقع ملے تاکہ ہم ایمان والوں میں شامل ہو جائیں۔
إِنَّ فِي ذَٰلِكَ لَـَٔايَةٗۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿103﴾
بے شک اس میں ایک نشانی ہے، مگر ان میں سے اکثر ایمان نہیں لاتے۔
وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿104﴾
اور بے شک آپ کا رب زبردست، رحم والا ہے۔
كَذَّبَتْ قَوْمُ نُوحٍ ٱلْمُرْسَلِينَ ﴿105﴾
نوح کی قوم نے رسولوں کو جھٹلایا۔
إِذْ قَالَ لَهُمْ أَخُوهُمْ نُوحٌ أَلَا تَتَّقُونَ ﴿106﴾
جب ان سے ان کے بھائی نوح نے کہا: کیا تم نہیں ڈرتے؟
إِنِّي لَكُمْ رَسُولٌ أَمِينٞ ﴿107﴾
بے شک میں تمہارے لیے ایک امانت دار رسول ہوں۔
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿108﴾
پس اللہ سے ڈرو اور میری اطاعت کرو۔
وَمَآ أَسْـَٔلُكُمْ عَلَيْهِ مِنْ أَجْرٍۖ إِنْ أَجْرِيَ إِلَّا عَلَىٰ رَبِّ ٱلْعَـٰلَمِينَ ﴿109﴾
اور میں تم سے اس پر کوئی اجرت نہیں مانگتا، میرا اجر تو صرف رب العالمین کے ذمے ہے۔
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿110﴾
پس اللہ سے ڈرو اور میری اطاعت کرو۔
قَالُوٓا۟ أَنُؤْمِنُ لَكَ وَٱتَّبَعَكَ ٱلْأَرْذَلُونَ ﴿111﴾
انہوں نے کہا: کیا ہم تم پر ایمان لے آئیں حالانکہ تمہاری پیروی کمینوں نے کی ہے؟
قَالَ وَمَا عِلْمِي بِمَا كَانُوا۟ يَعْمَلُونَ ﴿112﴾
اس نے کہا: ان کے اعمال کا علم مجھے نہیں۔
إِنْ حِسَابُهُمْ إِلَّا عَلَىٰ رَبِّيۖ لَوْ تَشْعُرُونَ ﴿113﴾
ان کا حساب تو میرے رب ہی کے ذمے ہے، اگر تم سمجھو۔
وَمَآ أَنَا۠ بِطَارِدِ ٱلْمُؤْمِنِينَ ﴿114﴾
اور میں ایمان والوں کو نکالنے والا نہیں ہوں۔
إِنْ أَنَا۠ إِلَّا نَذِيرٞ مُّبِينٞ ﴿115﴾
میں تو صرف واضح طور پر خبردار کرنے والا ہوں۔
قَالُوا۟ لَئِن لَّمْ تَنتَهِ يَـٰنُوحُ لَتَكُونَنَّ مِنَ ٱلْمَرْجُومِينَ ﴿116﴾
انہوں نے کہا: اے نوح! اگر تو باز نہ آیا تو ضرور سنگسار کیا جائے گا۔
قَالَ رَبِّ إِنَّ قَوْمِي كَذَّبُونِ ﴿117﴾
نوح نے کہا: اے میرے رب! میری قوم نے مجھے جھٹلایا۔
فَٱفْتَحْ بَيْنِي وَبَيْنَهُمۡ فَتْحٗا وَنَجِّنِي وَمَن مَّعِيَ مِنَ ٱلْمُؤْمِنِينَ ﴿118﴾
پس میرے اور ان کے درمیان فیصلہ کر دے، اور مجھے اور میرے ساتھ ایمان والوں کو نجات دے۔
فَأَنجَيْنَـٰهُ وَمَن مَّعَهُۥ فِي ٱلْفُلْكِ ٱلْمَشْحُونِ ﴿119﴾
تو ہم نے اسے اور اس کے ساتھ والوں کو بھری ہوئی کشتی میں نجات دی۔
ثُمَّ أَغْرَقْنَا بَعْدُ ٱلْبَاقِينَ ﴿120﴾
پھر ہم نے بعد میں باقی لوگوں کو غرق کر دیا۔
إِنَّ فِي ذَٰلِكَ لَـَٔايَةٗۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿121﴾
بے شک اس میں ایک نشانی ہے، مگر ان میں سے اکثر ایمان نہیں لاتے۔
وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿122﴾
اور بے شک آپ کا رب زبردست، رحم والا ہے۔
كَذَّبَتْ عَادٌ ٱلْمُرْسَلِينَ ﴿123﴾
عاد نے رسولوں کو جھٹلایا۔
إِذْ قَالَ لَهُمْ أَخُوهُمْ هُودٌ أَلَا تَتَّقُونَ ﴿124﴾
جب ان سے ان کے بھائی ہود نے کہا: کیا تم نہیں ڈرتے؟
إِنِّي لَكُمْ رَسُولٌ أَمِينٞ ﴿125﴾
بے شک میں تمہارے لیے ایک امانت دار رسول ہوں۔
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿126﴾
پس اللہ سے ڈرو اور میری اطاعت کرو۔
وَمَآ أَسْـَٔلُكُمْ عَلَيْهِ مِنْ أَجْرٍۖ إِنْ أَجْرِيَ إِلَّا عَلَىٰ رَبِّ ٱلْعَـٰلَمِينَ ﴿127﴾
اور میں تم سے اس پر کوئی اجرت نہیں مانگتا، میرا اجر تو صرف رب العالمین کے ذمے ہے۔
أَتَبْنُونَ بِكُلِّ رِيعٍ ءَايَةٗ تَعْبَثُونَ ﴿128﴾
کیا تم ہر اونچی جگہ پر ایک نشانی بناتے ہو، یوں ہی کھیل کے طور پر؟
وَتَتَّخِذُونَ مَصَانِعَ لَعَلَّكُمْ تَخْلُدُونَ ﴿129﴾
اور تم مضبوط قلعے بناتے ہو، گویا تم ہمیشہ رہو گے؟
وَإِذَا بَطَشْتُم بَطَشْتُمْ جَبَّارِينَ ﴿130﴾
اور جب تم پکڑتے ہو تو ظالمانہ طریقے سے پکڑتے ہو؟
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿131﴾
پس اللہ سے ڈرو اور میری اطاعت کرو۔
وَٱتَّقُوا۟ ٱلَّذِيٓ أَمَدَّكُم بِمَا تَعْلَمُونَ ﴿132﴾
اور اس سے ڈرو جس نے تمہیں وہ سب کچھ دیا جو تم جانتے ہو۔
أَمَدَّكُم بِأَنْعَـٰمٖ وَبَنِينَ ﴿133﴾
اس نے تمہیں مویشی اور بیٹے دیے۔
وَجَنَّـٰتٖ وَعُيُونٍ ﴿134﴾
اور باغات اور چشمے دیے۔
إِنِّيٓ أَخَافُ عَلَيْكُمْ عَذَابَ يَوْمٍ عَظِيمٖ ﴿135﴾
بے شک میں تم پر ایک بڑے دن کے عذاب سے ڈرتا ہوں۔
قَالُوا۟ سَوَآءٌ عَلَيْنَآ أَوَعَظْتَ أَمْ لَمْ تَكُن مِّنَ ٱلْوَٰعِظِينَ ﴿136﴾
انہوں نے کہا: ہمارے لیے برابر ہے، چاہے تو نصیحت کرے یا نہ کرے۔
إِنْ هَـٰذَآ إِلَّا خُلُقُ ٱلْأَوَّلِينَ ﴿137﴾
یہ تو صرف پہلوں کا طریقہ ہے۔
وَمَا نَحْنُ بِمُعَذَّبِينَ ﴿138﴾
اور ہم ہرگز عذاب دیے جانے والے نہیں۔
فَكَذَّبُوهُ فَأَهْلَكْنَـٰهُمْۚ إِنَّ فِي ذَٰلِكَ لَـَٔايَةٗۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿139﴾
پس انہوں نے اسے جھٹلایا، تو ہم نے انہیں ہلاک کر دیا۔ بے شک اس میں ایک نشانی ہے، مگر ان میں سے اکثر ایمان نہیں لاتے۔
وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿140﴾
اور بے شک آپ کا رب زبردست، رحم والا ہے۔
كَذَّبَتْ ثَمُودُ ٱلْمُرْسَلِينَ ﴿141﴾
ثمود نے رسولوں کو جھٹلایا۔
إِذْ قَالَ لَهُمْ أَخُوهُمْ صَـٰلِحٌ أَلَا تَتَّقُونَ ﴿142﴾
جب ان سے ان کے بھائی صالح نے کہا: کیا تم نہیں ڈرتے؟
إِنِّي لَكُمْ رَسُولٌ أَمِينٞ ﴿143﴾
بے شک میں تمہارے لیے ایک امانت دار رسول ہوں۔
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿144﴾
پس اللہ سے ڈرو اور میری اطاعت کرو۔
وَمَآ أَسْـَٔلُكُمْ عَلَيْهِ مِنْ أَجْرٍۖ إِنْ أَجْرِيَ إِلَّا عَلَىٰ رَبِّ ٱلْعَـٰلَمِينَ ﴿145﴾
اور میں تم سے اس پر کوئی اجرت نہیں مانگتا، میرا اجر تو صرف رب العالمین کے ذمے ہے۔
أَتُتْرَكُونَ فِي مَا هَـٰهُنَآ ءَامِنِينَ ﴿146﴾
کیا تمہیں یہاں امن و امان کے ساتھ چھوڑ دیا جائے گا؟
فِي جَنَّـٰتٖ وَعُيُونٖ ﴿147﴾
باغوں اور چشموں میں۔
وَزُرُوعٖ وَنَخْلٖ طَلْعُهَا هَضِيمٞ ﴿148﴾
اور کھیتوں اور کھجوروں میں جن کے خوشے نرم ہیں۔
وَتَنْحِتُونَ مِنَ ٱلْجِبَالِ بُيُوتٗا فَـٰرِهِينَ ﴿149﴾
اور تم پہاڑوں کو تراش کر خوشحال گھر بناتے ہو۔
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿150﴾
پس اللہ سے ڈرو اور میری اطاعت کرو۔
وَلَا تُطِيعُوٓا۟ أَمْرَ ٱلْمُسْرِفِينَ ﴿151﴾
اور حد سے بڑھنے والوں کا حکم نہ مانو۔
ٱلَّذِينَ يُفْسِدُونَ فِي ٱلْأَرْضِ وَلَا يُصْلِحُونَ ﴿152﴾
جو زمین میں فساد کرتے ہیں اور اصلاح نہیں کرتے۔
قَالُوا۟ إِنَّمَآ أَنتَ مِنَ ٱلْمُسَحَّرِينَ ﴿153﴾
انہوں نے کہا: تو تو بس جادو زدہ لوگوں میں سے ہے۔
مَآ أَنتَ إِلَّا بَشَرٞ مِّثْلُنَا فَأْتِ بِـَٔايَةٍ إِن كُنتَ مِنَ ٱلصَّـٰدِقِينَ ﴿154﴾
تو تو صرف ایک انسان ہے جیسے ہم، پس کوئی نشانی لے آ اگر تو سچا ہے۔
قَالَ هَـٰذِهِۦ نَاقَةٞ لَّهَا شِرْبٞ وَلَكُمْ شِرْبُ يَوْمٖ مَّعْلُومٖ ﴿155﴾
اس نے کہا: یہ اونٹنی ہے، اس کے لیے پانی پینے کا دن ہے اور تمہارے لیے مقرر دن۔
وَلَا تَمَسُّوهَا بِسُوٓءٖ فَيَأْخُذَكُمْ عَذَابُ يَوْمٍ عَظِيمٖ ﴿156﴾
اور اسے برائی سے ہاتھ نہ لگانا، ورنہ تمہیں بڑے دن کا عذاب آ لے گا۔
فَعَقَرُوهَا فَأَصْبَحُوا۟ نَـٰدِمِينَ ﴿157﴾
پھر انہوں نے اسے مار ڈالا، تو وہ صبح کو پچھتانے لگے۔
فَأَخَذَهُمُ ٱلْعَذَابُۚ إِنَّ فِي ذَٰلِكَ لَـَٔايَةٗۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿158﴾
تو انہیں عذاب نے آ لیا۔ بے شک اس میں ایک نشانی ہے، مگر ان میں سے اکثر ایمان نہیں لاتے۔
وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿159﴾
اور بے شک آپ کا رب زبردست، رحم والا ہے۔
كَذَّبَتْ قَوْمُ لُوطٍ ٱلْمُرْسَلِينَ ﴿160﴾
لوط کی قوم نے رسولوں کو جھٹلایا۔
إِذْ قَالَ لَهُمْ أَخُوهُمْ لُوطٌ أَلَا تَتَّقُونَ ﴿161﴾
جب ان سے ان کے بھائی لوط نے کہا: کیا تم نہیں ڈرتے؟
إِنِّي لَكُمْ رَسُولٌ أَمِينٞ ﴿162﴾
بے شک میں تمہارے لیے ایک امانت دار رسول ہوں۔
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿163﴾
پس اللہ سے ڈرو اور میری اطاعت کرو۔
وَمَآ أَسْـَٔلُكُمْ عَلَيْهِ مِنْ أَجْرٍۖ إِنْ أَجْرِيَ إِلَّا عَلَىٰ رَبِّ ٱلْعَـٰلَمِينَ ﴿164﴾
اور میں تم سے اس پر کوئی اجرت نہیں مانگتا، میرا اجر تو صرف رب العالمین کے ذمے ہے۔
أَتَأْتُونَ ٱلذُّكْرَانَ مِنَ ٱلْعَـٰلَمِينَ ﴿165﴾
کیا تم دنیا والوں میں سے مردوں کے پاس جاتے ہو؟
وَتَذَرُونَ مَا خَلَقَ لَكُمْ رَبُّكُم مِّنْ أَزْوَٰجِكُمۚ بَلْ أَنتُمْ قَوْمٌ عَادُونَ ﴿166﴾
اور چھوڑ دیتے ہو جو تمہارے رب نے تمہارے لیے تمہاری بیویوں میں پیدا کیا؟ بلکہ تم حد سے بڑھنے والے لوگ ہو۔
قَالُوا۟ لَئِن لَّمْ تَنتَهِ يَـٰلُوطُ لَتَكُونَنَّ مِنَ ٱلْمُخْرَجِينَ ﴿167﴾
انہوں نے کہا: اے لوط! اگر تو باز نہ آیا تو ضرور نکال دیا جائے گا۔
قَالَ إِنِّي لِعَمَلِكُم مِّنَ ٱلْقَالِينَ ﴿168﴾
اس نے کہا: بے شک میں تمہارے عمل سے سخت نفرت کرتا ہوں۔
رَبِّ نَجِّنِي وَأَهْلِي مِمَّا يَعْمَلُونَ ﴿169﴾
اے میرے رب! مجھے اور میرے گھر والوں کو ان کے عمل سے نجات دے۔
فَنَجَّيْنَـٰهُ وَأَهْلَهُۥٓ أَجْمَعِينَ ﴿170﴾
تو ہم نے اسے اور اس کے سب گھر والوں کو نجات دی۔
إِلَّا عَجُوزٗا فِي ٱلْغَٰبِرِينَ ﴿171﴾
سوائے ایک بوڑھی عورت کے جو پیچھے رہ جانے والوں میں تھی۔
ثُمَّ دَمَّرْنَا ٱلْـَٔاخَرِينَ ﴿172﴾
پھر ہم نے باقی لوگوں کو تباہ کر دیا۔
وَأَمْطَرْنَا عَلَيْهِم مَّطَرٗاۖ فَسَآءَ مَطَرُ ٱلْمُنذَرِينَ ﴿173﴾
اور ہم نے ان پر بارش برسائی، پس بہت ہی بری تھی وہ بارش ان لوگوں کے لیے جنہیں ڈرایا گیا تھا۔
إِنَّ فِي ذَٰلِكَ لَـَٔايَةٗۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿174﴾
بے شک اس میں ایک نشانی ہے، مگر ان میں سے اکثر ایمان نہیں لاتے۔
وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿175﴾
اور بے شک آپ کا رب زبردست، رحم والا ہے۔
كَذَّبَ أَصْحَـٰبُ ٱلْأَيْكَةِ ٱلْمُرْسَلِينَ ﴿176﴾
ایکہ والوں نے رسولوں کو جھٹلایا۔
إِذْ قَالَ لَهُمْ شُعَيْبٌ أَلَا تَتَّقُونَ ﴿177﴾
جب ان سے شعیب نے کہا: کیا تم نہیں ڈرتے؟
إِنِّي لَكُمْ رَسُولٌ أَمِينٞ ﴿178﴾
بے شک میں تمہارے لیے ایک امانت دار رسول ہوں۔
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِيعُونِ ﴿179﴾
پس اللہ سے ڈرو اور میری اطاعت کرو۔
وَمَآ أَسْـَٔلُكُمْ عَلَيْهِ مِنْ أَجْرٍۖ إِنْ أَجْرِيَ إِلَّا عَلَىٰ رَبِّ ٱلْعَـٰلَمِينَ ﴿180﴾
اور میں تم سے اس پر کوئی اجرت نہیں مانگتا، میرا اجر تو صرف رب العالمین کے ذمے ہے۔
أَوْفُوا۟ ٱلْكَيْلَ وَلَا تَكُونُوا۟ مِنَ ٱلْمُخْسِرِينَ ﴿181﴾
پیمانہ پورا کرو اور نقصان دینے والوں میں سے نہ بنو۔
وَزِنُوا۟ بِٱلْقِسْطَاسِ ٱلْمُسْتَقِيمِ ﴿182﴾
اور سیدھے ترازو سے تول کرو۔
وَلَا تَبْخَسُوا۟ ٱلنَّاسَ أَشْيَآءَهُمْ وَلَا تَعْثَوْا۟ فِي ٱلْأَرْضِ مُفْسِدِينَ ﴿183﴾
اور لوگوں کو ان کی چیزیں کم نہ دو، اور زمین میں فساد نہ پھیلاؤ۔
وَٱتَّقُوا۟ ٱلَّذِي خَلَقَكُمْ وَٱلْجِبِلَّةَ ٱلْأَوَّلِينَ ﴿184﴾
اور اس سے ڈرو جس نے تمہیں اور پہلے لوگوں کو پیدا کیا۔
قَالُوٓا۟ إِنَّمَآ أَنتَ مِنَ ٱلْمُسَحَّرِينَ ﴿185﴾
انہوں نے کہا: تو تو بس جادو زدہ لوگوں میں سے ہے۔
وَمَآ أَنتَ إِلَّا بَشَرٞ مِّثْلُنَا وَإِن نَّظُنُّكَ لَمِنَ ٱلْكَـٰذِبِينَ ﴿186﴾
اور تو تو صرف ایک انسان ہے جیسے ہم، اور ہم تو تجھے جھوٹوں میں سے سمجھتے ہیں۔
فَأَسْقِطْ عَلَيْنَا كِسَفٗا مِّنَ ٱلسَّمَآءِ إِن كُنتَ مِنَ ٱلصَّـٰدِقِينَ ﴿187﴾
پس ہم پر آسمان کا کوئی ٹکڑا گرا دے اگر تو سچوں میں سے ہے۔
قَالَ رَبِّيٓ أَعْلَمُ بِمَا تَعْمَلُونَ ﴿188﴾
اس نے کہا: میرا رب خوب جانتا ہے جو کچھ تم کرتے ہو۔
فَكَذَّبُوهُ فَأَخَذَهُمْ عَذَابُ يَوْمِ ٱلظُّلَّةِۚ إِنَّهُۥ كَانَ عَذَابَ يَوْمٍ عَظِيمٍ ﴿189﴾
پھر انہوں نے اسے جھٹلایا، تو انہیں سائبان والے دن کا عذاب نے آ لیا، بے شک وہ بڑے دن کا عذاب تھا۔
إِنَّ فِي ذَٰلِكَ لَـَٔايَةٗۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ﴿190﴾
بے شک اس میں ایک نشانی ہے، مگر ان میں سے اکثر ایمان نہیں لاتے۔
وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿191﴾
اور بے شک آپ کا رب زبردست، رحم والا ہے۔
وَإِنَّهُۥ لَتَنزِيلُ رَبِّ ٱلْعَـٰلَمِينَ ﴿192﴾
اور بے شک یہ رب العالمین کی طرف سے نازل کردہ ہے۔
نَزَلَ بِهِ ٱلرُّوحُ ٱلْأَمِينُ ﴿193﴾
اسے لے کر نازل ہوئے روح الامین۔
عَلَىٰ قَلْبِكَ لِتَكُونَ مِنَ ٱلْمُنذِرِينَ ﴿194﴾
آپ کے دل پر تاکہ آپ ڈرانے والوں میں سے ہوں۔
بِلِسَانٍ عَرَبِيّٖ مُّبِينٖ ﴿195﴾
صاف عربی زبان میں۔
وَإِنَّهُۥ لَفِي زُبُرِ ٱلْأَوَّلِينَ ﴿196﴾
اور بے شک یہ پہلے لوگوں کی کتابوں میں بھی ہے۔
أَوَلَمْ يَكُن لَّهُمْ ءَايَةً أَن يَعْلَمَهُۥ عُلَمَـٰٓؤُا۟ بَنِيٓ إِسْرَٰٓءِيلَ ﴿197﴾
کیا ان کے لیے یہ نشانی کافی نہیں کہ بنی اسرائیل کے علماء اسے جانتے ہیں؟
وَلَوْ نَزَّلْنَـٰهُ عَلَىٰ بَعْضِ ٱلْأَعْجَمِينَ ﴿198﴾
اور اگر ہم اسے کسی غیر عرب پر نازل کرتے۔
فَقَرَأَهُۥ عَلَيْهِم مَّا كَانُوا۟ بِهِۦ مُؤْمِنِينَ ﴿199﴾
پھر وہ اسے ان کو پڑھ کر سناتا، تب بھی وہ ایمان نہ لاتے۔
كَذَٰلِكَ سَلَكْنَـٰهُ فِي قُلُوبِ ٱلْمُجْرِمِينَ ﴿200﴾
اسی طرح ہم نے اسے مجرموں کے دلوں میں داخل کیا۔
لَا يُؤْمِنُونَ بِهِۦ حَتَّىٰ يَرَوُا۟ ٱلْعَذَابَ ٱلْأَلِيمَ ﴿201﴾
وہ اس پر ایمان نہیں لائیں گے یہاں تک کہ دردناک عذاب دیکھ لیں۔
فَيَأْتِيَهُم بَغْتَةٗ وَهُمْ لَا يَشْعُرُونَ ﴿202﴾
تو وہ ان پر اچانک آ جائے گا اور وہ شعور بھی نہ رکھتے ہوں گے۔
فَيَقُولُوا۟ هَلْ نَحْنُ مُنظَرُونَ ﴿203﴾
تو وہ کہیں گے: کیا ہمیں مہلت دی جائے گی؟
أَفَبِعَذَابِنَا يَسْتَعْجِلُونَ ﴿204﴾
کیا وہ ہمارے عذاب کو جلدی مانگتے ہیں؟
أَفَرَءَيْتَ إِن مَّتَّعْنَـٰهُمْ سِنِينَ ﴿205﴾
تو کیا آپ نے دیکھا اگر ہم انہیں کئی سال فائدہ پہنچائیں۔
ثُمَّ جَآءَهُم مَّا كَانُوا۟ يُوعَدُونَ ﴿206﴾
پھر ان کے پاس وہ چیز آ جائے جس کا انہیں وعدہ دیا گیا تھا۔
مَآ أَغْنَىٰ عَنْهُم مَّا كَانُوا۟ يُمَتَّعُونَ ﴿207﴾
تو وہ فائدہ جو انہیں دیا گیا تھا، ان کے کچھ کام نہ آئے گا۔
وَمَآ أَهْلَكْنَا مِن قَرْيَةٍ إِلَّا لَهَا مُنذِرُونَ ﴿208﴾
اور ہم نے کوئی بستی ہلاک نہیں کی مگر اس کے لیے ڈرانے والے تھے۔
ذِكْرَىٰ وَمَا كُنَّا ظَـٰلِمِينَ ﴿209﴾
نصیحت کے لیے، اور ہم ظلم کرنے والے نہ تھے۔
وَمَا تَنَزَّلَتْ بِهِ ٱلشَّيَـٰطِينُ ﴿210﴾
اور اسے شیطان نازل نہیں کرتے۔
وَمَا يَنۢبَغِي لَهُمْ وَمَا يَسْتَطِيعُونَ ﴿211﴾
اور نہ ان کے لائق ہے اور نہ وہ اس پر قادر ہیں۔
إِنَّهُمْ عَنِ ٱلسَّمْعِ لَمَعْزُولُونَ ﴿212﴾
بے شک وہ سننے سے دور رکھے گئے ہیں۔
فَلَا تَدْعُ مَعَ ٱللَّهِ إِلَـٰهًا ءَاخَرَ فَتَكُونَ مِنَ ٱلْمُعَذَّبِينَ ﴿213﴾
پس اللہ کے ساتھ کسی اور کو نہ پکارو، ورنہ تم سزا پانے والوں میں سے ہو جاؤ گے۔
وَأَنذِرْ عَشِيرَتَكَ ٱلْأَقْرَبِينَ ﴿214﴾
اور اپنے قریبی رشتہ داروں کو ڈراؤ۔
وَٱخْفِضْ جَنَاحَكَ لِمَنِ ٱتَّبَعَكَ مِنَ ٱلْمُؤْمِنِينَ ﴿215﴾
اور ایمان لانے والوں میں سے جو تمہاری پیروی کریں، ان کے لیے اپنے بازو جھکا دو۔
فَإِنْ عَصَوْكَ فَقُلْ إِنِّي بَرِيٓءٞ مِّمَّا تَعْمَلُونَ ﴿216﴾
پھر اگر وہ تمہاری نافرمانی کریں تو کہہ دو: میں تمہارے اعمال سے بری ہوں۔
وَتَوَكَّلْ عَلَى ٱلْعَزِيزِ ٱلرَّحِيمِ ﴿217﴾
اور زبردست، رحم والے پر بھروسہ کرو۔
ٱلَّذِي يَرَىٰكَ حِينَ تَقُومُ ﴿218﴾
جو تمہیں دیکھتا ہے جب تم کھڑے ہوتے ہو۔
وَتَقَلُّبَكَ فِي ٱلسَّاجِدِينَ ﴿219﴾
اور سجدہ کرنے والوں میں تمہاری گردش کو دیکھتا ہے۔
إِنَّهُۥ هُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ ﴿220﴾
بے شک وہی سننے والا، جاننے والا ہے۔
هَلْ أُنَبِّئُكُمْ عَلَىٰ مَن تَنَزَّلُ ٱلشَّيَـٰطِينُ ﴿221﴾
کیا میں تمہیں بتاؤں کہ شیطان کس پر نازل ہوتے ہیں؟
تَنَزَّلُ عَلَىٰ كُلِّ أَفَّاكٍ أَثِيمٖ ﴿222﴾
وہ ہر جھوٹے گناہگار پر نازل ہوتے ہیں۔
يُلْقُونَ ٱلسَّمْعَ وَأَكْثَرُهُمْ كَـٰذِبُونَ ﴿223﴾
وہ کان لگاتے ہیں، اور ان میں سے اکثر جھوٹے ہوتے ہیں۔
وَٱلشُّعَرَآءُ يَتَّبِعُهُمُ ٱلْغَاوُۥنَ ﴿224﴾
اور شاعروں کی پیروی گمراہ لوگ کرتے ہیں۔
أَلَمْ تَرَ أَنَّهُمْ فِي كُلِّ وَادٖ يَهِيمُونَ ﴿225﴾
کیا تم نے نہیں دیکھا کہ وہ ہر وادی میں بھٹکتے ہیں؟
وَأَنَّهُمْ يَقُولُونَ مَا لَا يَفْعَلُونَ ﴿226﴾
اور وہ کہتے ہیں جو کرتے نہیں۔
إِلَّا ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ وَذَكَرُوا۟ ٱللَّهَ كَثِيرٗا وَٱنتَصَرُوا۟ مِنۢ بَعْدِ مَا ظُلِمُوا۟ۗ وَسَيَعْلَمُ ٱلَّذِينَ ظَلَمُوٓا۟ أَيَّ مُنقَلَبٖ يَنقَلِبُونَ ﴿227﴾
سوائے ان کے جو ایمان لائے، نیک عمل کیے، اللہ کو بہت یاد کیا، اور ظلم کے بعد بدلہ لیا۔ اور ظالم جان لیں گے کہ وہ کس پلٹنے کی جگہ پلٹیں گے۔
