24 پارہ


بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
فَمَنْ أَظْلَمُ مِمَّنْ كَذَبَ عَلَى ٱللَّهِ وَكَذَّبَ بِٱلصِّدْقِ إِذْ جَآءَهُۥٓۚ أَلَيْسَ فِي جَهَنَّمَ مَثْوٗى لِّلْكَـٰفِرِينَ ﴿32﴾
تو اس شخص سے بڑھ کر ظالم کون ہے جو اللہ پر جھوٹ باندھے اور جب اس کے پاس سچ آئے تو اسے جھٹلا دے؟ کیا جہنم کافروں کا ٹھکانہ نہیں؟
وَٱلَّذِي جَآءَ بِٱلصِّدْقِ وَصَدَّقَ بِهِۦٓ أُو۟لَـٰٓئِكَ هُمُ ٱلْمُتَّقُونَ ﴿33﴾
اور جو سچ لے کر آیا اور جس نے اس کی تصدیق کی، وہی لوگ متقی ہیں۔
لَهُم مَّا يَشَآءُونَ عِندَ رَبِّهِمۚ ذَٰلِكَ جَزَآءُ ٱلْمُحْسِنِينَ ﴿34﴾
ان کے لیے ان کے رب کے پاس وہ سب کچھ ہے جو وہ چاہیں گے۔ یہی نیکوکاروں کا بدلہ ہے۔
لِيُكَفِّرَ ٱللَّهُ عَنْهُمْ أَسْوَأَ ٱلَّذِي عَمِلُوا۟ وَيَجْزِيَهُمْ أَجْرَهُم بِأَحْسَنِ ٱلَّذِي كَانُوا۟ يَعْمَلُونَ ﴿35﴾
تاکہ اللہ ان کے برے اعمال کو دور کر دے اور انہیں ان کے اچھے اعمال کا بہترین بدلہ دے۔
أَلَيْسَ ٱللَّهُ بِكَافٍ عَبْدَهُۥۖ وَيُخَوِّفُونَكَ بِٱلَّذِينَ مِن دُونِهِۦۚ وَمَن يُضْلِلِ ٱللَّهُ فَمَا لَهُۥ مِنْ هَادٍ ﴿36﴾
کیا اللہ اپنے بندے کے لیے کافی نہیں؟ اور یہ لوگ تمہیں اس کے سوا دوسروں سے ڈراتے ہیں۔ اور جسے اللہ گمراہ کر دے تو اس کے لیے کوئی ہدایت دینے والا نہیں۔
وَمَن يَهْدِ ٱللَّهُ فَمَا لَهُۥ مِن مُّضِلٍّۚ أَلَيْسَ ٱللَّهُ بِعَزِيزٖ ذِي ٱنتِقَامٖ ﴿37﴾
اور جسے اللہ ہدایت دے تو اس کے لیے کوئی گمراہ کرنے والا نہیں۔ کیا اللہ غالب، انتقام لینے والا نہیں؟
وَلَئِن سَأَلْتَهُم مَّنْ خَلَقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ لَيَقُولُنَّ ٱللَّهُۚ قُلْ أَفَرَءَيْتُم مَّا تَدْعُونَ مِن دُونِ ٱللَّهِ إِنْ أَرَادَنِيَ ٱللَّهُ بِضُرٍّ هَلْ هُنَّ كَـٰشِفَـٰتُ ضُرِّهِۦٓ أَوْ أَرَادَنِي بِرَحْمَةٍ هَلْ هُنَّ مُمْسِكَـٰتُ رَحْمَتِهِۦۚ قُلْ حَسْبِيَ ٱللَّهُۖ عَلَيْهِ يَتَوَكَّلُ ٱلْمُتَوَكِّلُونَ ﴿38﴾
اور اگر تم ان سے پوچھو کہ آسمانوں اور زمین کو کس نے پیدا کیا؟ تو وہ ضرور کہیں گے: اللہ نے۔ کہہ دو: تو کیا تم نے دیکھا کہ جنہیں تم اللہ کے سوا پکارتے ہو، اگر اللہ مجھے نقصان پہنچانا چاہے تو کیا وہ اس کے نقصان کو دور کر سکتے ہیں؟ یا اگر وہ مجھے رحمت دینا چاہے تو کیا وہ اس کی رحمت کو روک سکتے ہیں؟ کہہ دو: میرے لیے اللہ کافی ہے، اسی پر بھروسہ کرنے والے بھروسہ کرتے ہیں۔
قُلْ يَـٰقَوْمِ ٱعْمَلُوا۟ عَلَىٰ مَكَانَتِكُمْ إِنِّي عَامِلٞۖ فَسَوْفَ تَعْلَمُونَ ﴿39﴾
کہہ دو: اے میری قوم! اپنی جگہ پر عمل کرو، میں بھی عمل کر رہا ہوں، عنقریب تم جان لو گے۔
مَن يَأْتِيهِ عَذَابٞ يُخْزِيهِ وَيَحِلُّ عَلَيْهِ عَذَابٞ مُّقِيمٌ ﴿40﴾
کس پر وہ عذاب آتا ہے جو اسے رسوا کر دے، اور کس پر ہمیشہ رہنے والا عذاب نازل ہوتا ہے۔
إِنَّآ أَنزَلْنَا عَلَيْكَ ٱلْكِتَـٰبَ لِلنَّاسِ بِٱلْحَقِّۖ فَمَنِ ٱهْتَدَىٰ فَلِنَفْسِهِۦۖ وَمَن ضَلَّ فَإِنَّمَا يَضِلُّ عَلَيْهَاۖ وَمَآ أَنتَ عَلَيْهِم بِوَكِيلٖ ﴿41﴾
بے شک ہم نے آپ پر لوگوں کے لیے حق کے ساتھ کتاب نازل کی، پس جو ہدایت پائے وہ اپنے ہی فائدے کے لیے ہے، اور جو گمراہ ہو تو وہ اپنے ہی نقصان کے لیے گمراہ ہوتا ہے، اور آپ ان کے ذمہ دار نہیں ہیں۔
ٱللَّهُ يَتَوَفَّى ٱلْأَنفُسَ حِينَ مَوْتِهَا وَٱلَّتِي لَمْ تَمُتْ فِي مَنَامِهَاۖ فَيُمْسِكُ ٱلَّتِي قَضَىٰ عَلَيْهَا ٱلْمَوْتَ وَيُرْسِلُ ٱلْأُخْرَىٰٓ إِلَىٰٓ أَجَلٖ مُّسَمًّىۚ إِنَّ فِي ذَٰلِكَ لَءَايَـٰتٖ لِّقَوْمٖ يَتَفَكَّرُونَ ﴿42﴾
اللہ جانوں کو ان کی موت کے وقت قبض کر لیتا ہے، اور جو نہیں مریں انہیں نیند کے وقت (قبض کرتا ہے)، پھر جس پر موت کا فیصلہ ہو چکا ہو اسے روک لیتا ہے، اور باقی کو مقررہ وقت تک واپس بھیج دیتا ہے۔ بے شک اس میں غور و فکر کرنے والوں کے لیے نشانیاں ہیں۔
أَمِ ٱتَّخَذُوا۟ مِن دُونِ ٱللَّهِ شُفَعَآءَۚ قُلْ أَوَلَوْ كَانُوا۟ لَا يَمْلِكُونَ شَيْـٔٗا وَلَا يَعْقِلُونَ ﴿43﴾
کیا انہوں نے اللہ کے سوا سفارش کرنے والے بنا لیے ہیں؟ کہہ دو: کیا اگر وہ کچھ اختیار نہ رکھتے ہوں اور نہ ہی عقل رکھتے ہوں (تب بھی سفارش کریں گے)؟
قُل لِّلَّهِ ٱلشَّفَـٰعَةُ جَمِيعٗاۖ لَّهُۥ مُلْكُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۖ ثُمَّ إِلَيْهِ تُرْجَعُونَ ﴿44﴾
کہہ دو: سفارش سب کی سب اللہ ہی کے لیے ہے، اسی کے لیے آسمانوں اور زمین کی بادشاہی ہے، پھر تم اسی کی طرف لوٹائے جاؤ گے۔
وَإِذَا ذُكِرَ ٱللَّهُ وَحْدَهُ ٱشْمَأَزَّتْ قُلُوبُ ٱلَّذِينَ لَا يُؤْمِنُونَ بِٱلْـَٔاخِرَةِۖ وَإِذَا ذُكِرَ ٱلَّذِينَ مِن دُونِهِۦٓ إِذَا هُمْ يَسْتَبْشِرُونَ ﴿45﴾
اور جب اللہ کا ذکر اکیلا کیا جاتا ہے تو وہ لوگ جنہیں آخرت پر ایمان نہیں، ان کے دل بیزار ہو جاتے ہیں، اور جب اس کے سوا دوسروں کا ذکر کیا جاتا ہے تو وہ خوش ہو جاتے ہیں۔
قُلِ ٱللَّهُمَّ فَاطِرَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ عَـٰلِمَ ٱلْغَيْبِ وَٱلشَّهَـٰدَةِ أَنتَ تَحْكُمُ بَيْنَ عِبَادِكَ فِي مَا كَانُوا۟ فِيهِ يَخْتَلِفُونَ ﴿46﴾
کہہ دو: اے اللہ! آسمانوں اور زمین کے پیدا کرنے والے، پوشیدہ اور ظاہر کے جاننے والے، تو ہی اپنے بندوں کے درمیان ان باتوں میں فیصلہ کرے گا جن میں وہ اختلاف کرتے ہیں۔
وَلَوْ أَنَّ لِلَّذِينَ ظَلَمُوا۟ مَا فِي ٱلْأَرْضِ جَمِيعٗا وَمِثْلَهُۥ مَعَهُۥ لَٱفْتَدَوْا۟ بِهِۦ مِن سُوٓءِ ٱلْعَذَابِ يَوْمَ ٱلْقِيَـٰمَةِۚ وَبَدَا لَهُم مِّنَ ٱللَّهِ مَا لَمْ يَكُونُوا۟ يَحْتَسِبُونَ ﴿47﴾
اور اگر ظلم کرنے والوں کے پاس وہ سب کچھ ہو جو زمین میں ہے، اور اس کے ساتھ اتنا ہی اور بھی ہو، تو وہ قیامت کے دن کے برے عذاب سے بچنے کے لیے اسے فدیہ میں دے دیں گے۔ اور ان پر اللہ کی طرف سے وہ کچھ ظاہر ہوگا جس کا انہوں نے گمان بھی نہ کیا تھا۔
وَبَدَا لَهُمْ سَيِّـَٔاتُ مَا كَسَبُوا۟ وَحَاقَ بِهِم مَّا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ ﴿48﴾
اور ان پر ان کے اعمال کی برائیاں ظاہر ہو جائیں گی، اور جس چیز کا وہ مذاق اڑاتے تھے وہی انہیں گھیر لے گی۔
فَإِذَا مَسَّ ٱلْإِنسَـٰنَ ضُرّٞ دَعَانَا ثُمَّ إِذَا خَوَّلْنَـٰهُ نِعْمَةٗ مِّنَّا قَالَ إِنَّمَآ أُوتِيتُهُۥ عَلَىٰ عِلْمِۭۚ بَلْ هِيَ فِتْنَةٞ وَلَـٰكِنَّ أَكْثَرَهُمْ لَا يَعْلَمُونَ ﴿49﴾
جب انسان کو کوئی تکلیف پہنچتی ہے تو وہ ہمیں پکارتا ہے، پھر جب ہم اسے اپنی طرف سے کوئی نعمت عطا کرتے ہیں تو کہتا ہے: یہ تو مجھے علم کی بنا پر دیا گیا ہے۔ بلکہ یہ آزمائش ہے، لیکن ان میں سے اکثر نہیں جانتے۔
قَدْ قَالَهَا ٱلَّذِينَ مِن قَبْلِهِمْ فَمَآ أَغْنَىٰ عَنْهُم مَّا كَانُوا۟ يَكْسِبُونَ ﴿50﴾
یہی بات ان سے پہلے والوں نے بھی کہی تھی، تو ان کے کمائے ہوئے مال نے ان کے کچھ کام نہ آیا۔
فَأَصَابَهُمْ سَيِّـَٔاتُ مَا كَسَبُوا۟ۖ وَٱلَّذِينَ ظَلَمُوا۟ مِنْ هَـٰذَا سَيُصِيبُهُمْ سَيِّـَٔاتُ مَا كَسَبُوا۟ وَمَا هُم بِمُعْجِزِينَ ﴿51﴾
تو ان پر ان کے اعمال کی برائیاں آ پڑیں، اور ان میں سے جنہوں نے ظلم کیا ان پر بھی ان کے اعمال کی برائیاں آ پڑیں گی، اور وہ عاجز نہیں کر سکتے۔
أَوَلَمْ يَعْلَمُوٓا۟ أَنَّ ٱللَّهَ يَبْسُطُ ٱلرِّزْقَ لِمَن يَشَآءُ وَيَقْدِرُۚ إِنَّ فِي ذَٰلِكَ لَـَٔايَـٰتٖ لِّقَوْمٖ يُؤْمِنُونَ ﴿52﴾
کیا انہوں نے نہیں جانا کہ اللہ جس کے لیے چاہتا ہے رزق کشادہ کرتا ہے اور تنگ بھی کرتا ہے؟ بے شک اس میں ایمان والوں کے لیے نشانیاں ہیں۔
قُلْ يَـٰعِبَادِ ٱلَّذِينَ أَسْرَفُوا۟ عَلَىٰٓ أَنفُسِهِمْ لَا تَقْنَطُوا۟ مِن رَّحْمَةِ ٱللَّهِۚ إِنَّ ٱللَّهَ يَغْفِرُ ٱلذُّنُوبَ جَمِيعًاۚ إِنَّهُۥ هُوَ ٱلْغَفُورُ ٱلرَّحِيمُ ﴿53﴾
کہہ دو: اے میرے بندو جنہوں نے اپنی جانوں پر زیادتی کی ہے! اللہ کی رحمت سے نا امید نہ ہو، بے شک اللہ سب گناہ معاف کر دیتا ہے، بے شک وہی بخشنے والا، مہربان ہے۔
وَأَنِيبُوٓا۟ إِلَىٰ رَبِّكُمْ وَأَسْلِمُوا۟ لَهُۥ مِن قَبْلِ أَن يَأْتِيَكُمُ ٱلْعَذَابُ ثُمَّ لَا تُنصَرُونَ ﴿54﴾
اور اپنے رب کی طرف رجوع کرو اور اس کے فرمانبردار بن جاؤ، اس سے پہلے کہ تم پر عذاب آ جائے، پھر تم مدد نہ دیے جاؤ۔
وَٱتَّبِعُوٓا۟ أَحْسَنَ مَآ أُنزِلَ إِلَيْكُم مِّن رَّبِّكُم مِّن قَبْلِ أَن يَأْتِيَكُمُ ٱلْعَذَابُ بَغْتَةٗ وَأَنتُمْ لَا تَشْعُرُونَ ﴿55﴾
اور اس بہترین چیز کی پیروی کرو جو تمہارے رب کی طرف سے تم پر نازل کی گئی ہے، اس سے پہلے کہ تم پر اچانک عذاب آ جائے اور تمہیں خبر بھی نہ ہو۔
أَن تَقُولَ نَفْسٞ يَـٰحَسْرَتَىٰ عَلَىٰ مَا فَرَّطتُ فِي جَنبِ ٱللَّهِ وَإِن كُنتُ لَمِنَ ٱلسَّاخِرِينَ ﴿56﴾
کہ (کہیں ایسا نہ ہو کہ) کوئی جان کہے: ہائے افسوس! میں نے اللہ کے حق میں کوتاہی کی، اور میں تو مذاق اڑانے والوں میں سے تھا۔
أَوْ تَقُولَ لَوْ أَنَّ ٱللَّهَ هَدَىٰنِي لَكُنتُ مِنَ ٱلْمُتَّقِينَ ﴿57﴾
یا کہے: اگر اللہ نے مجھے ہدایت دی ہوتی تو میں بھی متقیوں میں سے ہوتا۔
أَوْ تَقُولَ حِينَ تَرَى ٱلْعَذَابَ لَوْ أَنَّ لِي كَرَّةٗ فَأَكُونَ مِنَ ٱلْمُحْسِنِينَ ﴿58﴾
یا جب وہ عذاب دیکھے تو کہے: کاش مجھے ایک اور موقع ملے تاکہ میں نیکوکاروں میں شامل ہو جاؤں۔
بَلَىٰ قَدْ جَآءَتْكَ ءَايَـٰتِي فَكَذَّبْتَ بِهَا وَٱسْتَكْبَرْتَ وَكُنتَ مِنَ ٱلْكَـٰفِرِينَ ﴿59﴾
(جواب دیا جائے گا:) کیوں نہیں! میری آیات تیرے پاس آئیں تو تُو نے انہیں جھٹلایا اور تکبر کیا، اور تُو کافروں میں سے ہو گیا۔
وَيَوْمَ ٱلْقِيَـٰمَةِ تَرَى ٱلَّذِينَ كَذَبُوا۟ عَلَى ٱللَّهِ وُجُوهُهُم مُّسْوَدَّةٌۚ أَلَيْسَ فِي جَهَنَّمَ مَثْوٗى لِّلْمُتَكَبِّرِينَ ﴿60﴾
اور قیامت کے دن تم دیکھو گے کہ جنہوں نے اللہ پر جھوٹ باندھا ان کے چہرے سیاہ ہوں گے۔ کیا جہنم متکبروں کا ٹھکانہ نہیں؟
وَيُنَجِّي ٱللَّهُ ٱلَّذِينَ ٱتَّقَوْا۟ بِمَفَازَتِهِمْ لَا يَمَسُّهُمُ ٱلسُّوٓءُ وَلَا هُمْ يَحْزَنُونَ ﴿61﴾
اور اللہ ان لوگوں کو جو تقویٰ اختیار کرتے ہیں، ان کی کامیابی کے سبب نجات دے گا، نہ انہیں کوئی برائی چھوئے گی اور نہ وہ غمگین ہوں گے۔
ٱللَّهُ خَـٰلِقُ كُلِّ شَيْءٖ وَهُوَ عَلَىٰ كُلِّ شَيْءٖ وَكِيلٞ ﴿62﴾
اللہ ہر چیز کا خالق ہے، اور وہی ہر چیز پر نگہبان ہے۔
لَهُۥ مَقَالِيدُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۗ وَٱلَّذِينَ كَفَرُوا۟ بِـَٔايَـٰتِ ٱللَّهِ أُو۟لَـٰٓئِكَ هُمُ ٱلْخَـٰسِرُونَ ﴿63﴾
اسی کے ہاتھ میں آسمانوں اور زمین کی کنجیاں ہیں، اور جنہوں نے اللہ کی آیات کا انکار کیا وہی نقصان اٹھانے والے ہیں۔
قُلْ أَفَغَيْرَ ٱللَّهِ تَأْمُرُوٓنِّيٓ أَعْبُدُ أَيُّهَا ٱلْجَـٰهِلُونَ ﴿64﴾
کہہ دو: کیا تم مجھے اللہ کے سوا کسی اور کی عبادت کا حکم دیتے ہو، اے جاہلو؟
وَلَقَدْ أُوحِيَ إِلَيْكَ وَإِلَى ٱلَّذِينَ مِن قَبْلِكَ لَئِنْ أَشْرَكْتَ لَيَحْبَطَنَّ عَمَلُكَ وَلَتَكُونَنَّ مِنَ ٱلْخَـٰسِرِينَ ﴿65﴾
اور بے شک تمہاری طرف اور تم سے پہلے والوں کی طرف وحی کی گئی کہ اگر تم نے شرک کیا تو تمہارا عمل ضائع ہو جائے گا اور تم ضرور خسارہ اٹھانے والوں میں سے ہو جاؤ گے۔
بَلِ ٱللَّهَ فَٱعْبُدْ وَكُن مِّنَ ٱلشَّـٰكِرِينَ ﴿66﴾
بلکہ اللہ ہی کی عبادت کرو اور شکر گزاروں میں سے ہو جاؤ۔
وَمَا قَدَرُوا۟ ٱللَّهَ حَقَّ قَدْرِهِۦ وَٱلْأَرْضُ جَمِيعٗا قَبْضَتُهُۥ يَوْمَ ٱلْقِيَـٰمَةِ وَٱلسَّمَـٰوَٰتُ مَطْوِيَّـٰتُۢ بِيَمِينِهِۦۚ سُبْحَـٰنَهُۥ وَتَعَـٰلَىٰ عَمَّا يُشْرِكُونَ ﴿67﴾
اور انہوں نے اللہ کی قدر نہ کی جیسا کہ اس کی قدر کرنے کا حق تھا، اور قیامت کے دن ساری زمین اس کی مٹھی میں ہوگی اور آسمان اس کے داہنے ہاتھ میں لپٹے ہوں گے۔ وہ پاک ہے اور بلند ہے اس سے جو وہ شریک ٹھہراتے ہیں۔
وَنُفِخَ فِي ٱلصُّورِ فَصَعِقَ مَن فِي ٱلسَّمَـٰوَٰتِ وَمَن فِي ٱلْأَرْضِ إِلَّا مَن شَآءَ ٱللَّهُۖ ثُمَّ نُفِخَ فِيهِ أُخْرَىٰ فَإِذَا هُمْ قِيَامٞ يَنظُرُونَ ﴿68﴾
اور صور میں پھونکا جائے گا تو جو آسمانوں اور زمین میں ہیں سب بے ہوش ہو جائیں گے، سوائے اس کے جسے اللہ چاہے، پھر دوبارہ صور پھونکا جائے گا تو وہ سب کھڑے ہو کر دیکھنے لگیں گے۔
وَأَشْرَقَتِ ٱلْأَرْضُ بِنُورِ رَبِّهَا وَوُضِعَ ٱلْكِتَـٰبُ وَجِيٓءَ بِٱلنَّبِيِّـۧنَ وَٱلشُّهَدَآءِ وَقُضِيَ بَيْنَهُم بِٱلْحَقِّ وَهُمْ لَا يُظْلَمُونَ ﴿69﴾
اور زمین اپنے رب کے نور سے روشن ہو جائے گی، اور کتاب رکھ دی جائے گی، اور نبیوں اور گواہوں کو لایا جائے گا، اور ان کے درمیان حق کے ساتھ فیصلہ کیا جائے گا، اور ان پر ظلم نہیں کیا جائے گا۔
وَوُفِّيَتْ كُلُّ نَفْسٖ مَّا عَمِلَتْ وَهُوَ أَعْلَمُ بِمَا يَفْعَلُونَ ﴿70﴾
اور ہر شخص کو اس کے اعمال کا پورا بدلہ دیا جائے گا، اور وہ خوب جانتا ہے جو کچھ وہ کرتے ہیں۔
وَسِيقَ ٱلَّذِينَ كَفَرُوٓا۟ إِلَىٰ جَهَنَّمَ زُمَرًاۖ حَتَّىٰٓ إِذَا جَآءُوهَا فُتِحَتْ أَبْوَٰبُهَا وَقَالَ لَهُمْ خَزَنَتُهَآ أَلَمْ يَأْتِكُمْ رُسُل...َايَـٰتِ رَبِّكُمْ وَيُنذِرُونَكُمْ لِقَآءَ يَوْمِكُمْ هَـٰذَاۚ قَالُوا۟ بَلَىٰ وَلَـٰكِنْ حَقَّتْ كَلِمَةُ ٱلْعَذَابِ عَلَى ٱلْكَـٰفِرِينَ ﴿71﴾
اور جنہوں نے کفر کیا انہیں گروہ در گروہ جہنم کی طرف ہانکا جائے گا، یہاں تک کہ جب وہ اس کے پاس آئیں گے اور اس کے دروازے کھول دیے جائیں گے، تو اس کے داروغہ ان سے کہیں گے: کیا تمہارے پاس تم میں سے رسول نہیں آئے تھے جو تم پر تمہارے رب کی آیات پڑھتے تھے اور تمہیں اس دن کی ملاقات سے ڈراتے تھے؟ وہ کہیں گے: کیوں نہیں، لیکن عذاب کا وعدہ کافروں پر سچ ہو چکا تھا۔
قِيلَ ٱدْخُلُوٓا۟ أَبْوَٰبَ جَهَنَّمَ خَـٰلِدِينَ فِيهَاۖ فَبِئْسَ مَثْوَى ٱلْمُتَكَبِّرِينَ ﴿72﴾
کہا جائے گا: جہنم کے دروازوں میں داخل ہو جاؤ، ہمیشہ اس میں رہنے کے لیے۔ تکبر کرنے والوں کا ٹھکانہ بہت برا ہے۔
وَسِيقَ ٱلَّذِينَ ٱتَّقَوْا۟ رَبَّهُمْ إِلَى ٱلْجَنَّةِ زُمَرًاۖ حَتَّىٰٓ إِذَا جَآءُوهَا وَفُتِحَتْ أَبْوَٰبُهَا وَقَالَ لَهُمْ خَزَنَتُهَا سَلَـٰمٌ عَلَيْكُمْ طِبْتُمْ فَٱدْخُلُوهَا خَـٰلِدِينَ ﴿73﴾
اور جو لوگ اپنے رب سے ڈرتے رہے انہیں گروہ در گروہ جنت کی طرف لے جایا جائے گا، یہاں تک کہ جب وہ اس کے پاس آئیں گے اور اس کے دروازے کھول دیے جائیں گے، تو اس کے داروغہ ان سے کہیں گے: تم پر سلام ہو، تم پاکیزہ ہو، پس اس میں داخل ہو جاؤ ہمیشہ کے لیے۔
وَقَالُوا۟ ٱلْحَمْدُ لِلَّهِ ٱلَّذِي صَدَقَنَا وَعْدَهُۥ وَأَوْرَثَنَا ٱلْأَرْضَ نَتَبَوَّأُ مِنَ ٱلْجَنَّةِ حَيْثُ نَشَآءُۖ فَنِعْمَ أَجْرُ ٱلْعَـٰمِلِينَ ﴿74﴾
اور وہ کہیں گے: سب تعریف اللہ کے لیے ہے جس نے ہمیں اپنا وعدہ سچ کر دکھایا، اور ہمیں زمین کا وارث بنایا کہ ہم جنت میں جہاں چاہیں رہیں۔ پس عمل کرنے والوں کا اجر کیا ہی خوب ہے۔
وَتَرَى ٱلْمَلَـٰٓئِكَةَ حَآفِّينَ مِنْ حَوْلِ ٱلْعَرْشِ يُسَبِّحُونَ بِحَمْدِ رَبِّهِمْۖ وَقُضِيَ بَيْنَهُم بِٱلْحَقِّ وَقِيلَ ٱلْحَمْدُ لِلَّهِ رَبِّ ٱلْعَـٰلَمِينَ ﴿75﴾
اور تم فرشتوں کو دیکھو گے کہ وہ عرش کے گرد حلقہ بنائے ہوئے ہوں گے، اپنے رب کی حمد کے ساتھ تسبیح کر رہے ہوں گے، اور ان کے درمیان حق کے ساتھ فیصلہ کر دیا جائے گا، اور کہا جائے گا: سب تعریف اللہ رب العالمین کے لیے ہے۔
سورة المؤمن
بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
حمٓ ﴿1﴾
حٰمٓ
تَنزِيلُ ٱلْكِتَـٰبِ مِنَ ٱللَّهِ ٱلْعَزِيزِ ٱلْعَلِيمِ ﴿2﴾
اس کتاب کا نزول اللہ کی طرف سے ہے جو زبردست، علم والا ہے۔
غَافِرِ ٱلذَّنۢبِ وَقَابِلِ ٱلتَّوْبِ شَدِيدِ ٱلْعِقَابِ ذِي ٱلطَّوْلِۖ لَآ إِلَـٰهَ إِلَّا هُوَۖ إِلَيْهِ ٱلْمَصِيرُ ﴿3﴾
گناہ بخشنے والا، توبہ قبول کرنے والا، سخت سزا دینے والا، فضل والا ہے۔ اس کے سوا کوئی معبود نہیں، اسی کی طرف لوٹنا ہے۔
مَا يُجَـٰدِلُ فِيٓ ءَايَـٰتِ ٱللَّهِ إِلَّا ٱلَّذِينَ كَفَرُوا۟ فَلَا يَغْرُرْكَ تَقَلُّبُهُمْ فِي ٱلْبِلَـٰدِ ﴿4﴾
اللہ کی آیات میں وہی لوگ جھگڑتے ہیں جو کافر ہیں، پس ان کا شہروں میں چلنا پھرنا آپ کو دھوکے میں نہ ڈالے۔
كَذَّبَتْ قَبْلَهُمْ قَوْمُ نُوحٖ وَٱلْأَحْزَابُ مِنۢ بَعْدِهِمْۖ وَهَمَّتْ كُلُّ أُمَّةِۭ بِرَسُولِهِمْ لِيَأْخُذُوهُۖ وَجَـٰدَلُوا۟ بِٱلْبَـٰطِلِ لِيُدْحِضُوا۟ بِهِ ٱلْحَقَّ فَأَخَذْتُهُمْۖ فَكَيْفَ كَانَ عِقَابِ ﴿5﴾
ان سے پہلے نوح کی قوم اور ان کے بعد کے گروہوں نے بھی جھٹلایا، اور ہر امت نے اپنے رسول کو پکڑنے کا ارادہ کیا، اور انہوں نے باطل کے ذریعے حق کو مٹانے کی کوشش کی، تو میں نے انہیں پکڑ لیا، پس کیسا تھا میرا عذاب؟
وَكَذَٰلِكَ حَقَّتْ كَلِمَتُ رَبِّكَ عَلَى ٱلَّذِينَ كَفَرُوٓا۟ أَنَّهُمْ أَصْحَـٰبُ ٱلنَّارِ ﴿6﴾
اور اسی طرح آپ کے رب کی بات ان کافروں پر ثابت ہو گئی کہ وہی دوزخ والے ہیں۔
ٱلَّذِينَ يَحْمِلُونَ ٱلْعَرْشَ وَمَنْ حَوْلَهُۥ يُسَبِّحُونَ بِحَمْدِ رَبِّهِمْ وَيُؤْمِنُونَ بِهِۦ وَيَسْتَغْفِرُونَ لِلَّذِينَ ءَامَنُوا۟ۖ رَبَّنَا وَسِعْتَ كُلَّ شَيْءٖ رَّحْمَةٗ وَعِلْمٗا فَٱغْفِرْ لِلَّذِينَ تَابُوا۟ وَٱتَّبَعُوا۟ سَبِيلَكَ وَقِهِمْ عَذَابَ ٱلْجَحِيمِ ﴿7﴾
جو لوگ عرش کو اٹھائے ہوئے ہیں اور جو اس کے گرد ہیں وہ اپنے رب کی حمد کے ساتھ تسبیح کرتے ہیں، اور اس پر ایمان رکھتے ہیں، اور ایمان والوں کے لیے بخشش مانگتے ہیں: اے ہمارے رب! تو ہر چیز کو اپنی رحمت اور علم سے گھیرے ہوئے ہے، پس انہیں بخش دے جنہوں نے توبہ کی اور تیرے راستے کی پیروی کی، اور انہیں دوزخ کے عذاب سے بچا۔
رَبَّنَا وَأَدْخِلْهُمْ جَنَّـٰتِ عَدْنٍ ٱلَّتِي وَعَدتَّهُمْ وَمَن صَلَحَ مِنْ ءَابَآئِهِمْ وَأَزْوَٰجِهِمْ وَذُرِّيَّـٰتِهِمْۚ إِنَّكَ أَنتَ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿8﴾
اے ہمارے رب! انہیں ہمیشہ رہنے والی جنتوں میں داخل کر جن کا تو نے ان سے وعدہ کیا ہے، اور ان کے نیک باپ دادا، بیویوں اور اولاد میں سے بھی، بے شک تو ہی زبردست، حکمت والا ہے۔
وَقِهِمُ ٱلسَّيِّـَٔاتِۚ وَمَن تَقِ ٱلسَّيِّـَٔاتِ يَوْمَئِذٖ فَقَدْ رَحِمْتَهُۥۚ وَذَٰلِكَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿9﴾
اور انہیں برائیوں سے بچا، اور جسے تو اس دن برائیوں سے بچا لے، تو یقیناً تو نے اس پر رحم کیا، اور یہی بڑی کامیابی ہے۔
إِنَّ ٱلَّذِينَ كَفَرُوا۟ يُنَادَوْنَ لَمَقْتُ ٱللَّهِ أَكْبَرُ مِن مَّقْتِكُمْ أَنفُسَكُمْ إِذْ تُدْعَوْنَ إِلَى ٱلْإِيمَـٰنِ فَتَكْفُرُونَ ﴿10﴾
بے شک جنہوں نے کفر کیا انہیں پکارا جائے گا: اللہ کی ناراضی تمہاری اپنی ناراضی سے زیادہ شدید ہے، جب تم ایمان کی طرف بلائے گئے اور تم نے انکار کیا۔
قَالُوا۟ رَبَّنَآ أَمَتَّنَا ٱثْنَتَيْنِ وَأَحْيَيْتَنَا ٱثْنَتَيْنِ فَٱعْتَرَفْنَا بِذُنُوبِنَا فَهَلْ إِلَىٰ خُرُوجٖ مِّن سَبِيلٍ ﴿11﴾
وہ کہیں گے: اے ہمارے رب! تو نے ہمیں دو بار موت دی اور دو بار زندگی دی، پس ہم نے اپنے گناہوں کا اعتراف کر لیا، تو کیا نکلنے کی کوئی راہ ہے؟
ذَٰلِكُم بِأَنَّهُۥٓ إِذَا دُعِيَ ٱللَّهُ وَحْدَهُۥ كَفَرْتُمْ وَإِن يُشْرَكْ بِهِۦ تُؤْمِنُوا۟ۚ فَٱلْحُكْمُ لِلَّهِ ٱلْعَلِيِّ ٱلْكَبِيرِ ﴿12﴾
یہ اس لیے ہے کہ جب اللہ اکیلا پکارا جاتا تھا تو تم انکار کرتے تھے، اور جب اس کے ساتھ شریک ٹھہرایا جاتا تھا تو تم ایمان لاتے تھے، پس فیصلہ اللہ ہی کا ہے جو بلند و بالا، عظمت والا ہے۔
هُوَ ٱلَّذِي يُرِيكُمْ ءَايَـٰتِهِۦ وَيُنَزِّلُ لَكُم مِّنَ ٱلسَّمَآءِ رِزْقٗاۚ وَمَا يَتَذَكَّرُ إِلَّا مَن يُنِيبُ ﴿13﴾
وہی ہے جو تمہیں اپنی نشانیاں دکھاتا ہے اور تمہارے لیے آسمان سے رزق نازل کرتا ہے، اور نصیحت وہی حاصل کرتا ہے جو رجوع کرتا ہے۔
فَٱدْعُوا۟ ٱللَّهَ مُخْلِصِينَ لَهُ ٱلدِّينَ وَلَوْ كَرِهَ ٱلْكَـٰفِرُونَ ﴿14﴾
پس اللہ کو پکارو دین کو اسی کے لیے خالص کرتے ہوئے، اگرچہ کافر اسے ناپسند کریں۔
رَفِيعُ ٱلدَّرَجَـٰتِ ذُو ٱلْعَرْشِ يُلْقِي ٱلرُّوحَ مِنْ أَمْرِهِۦ عَلَىٰ مَن يَشَآءُ مِنْ عِبَادِهِۦ لِيُنذِرَ يَوْمَ ٱلتَّلَاقِ ﴿15﴾
بلند درجات والا، عرش والا، اپنے حکم سے روح کو اپنے بندوں میں سے جس پر چاہے نازل کرتا ہے تاکہ وہ ملاقات کے دن سے ڈرائے۔
يَوْمَ هُم بَـٰرِزُونَ لَا يَخْفَىٰ عَلَى ٱللَّهِ مِنْهُمْ شَيْءٞۚ لِّمَنِ ٱلْمُلْكُ ٱلْيَوْمَۖ لِلَّهِ ٱلْوَٰحِدِ ٱلْقَهَّارِ ﴿16﴾
جس دن وہ سب ظاہر ہوں گے، اللہ سے ان میں کوئی چیز چھپی نہ رہے گی۔ (پوچھا جائے گا:) آج کس کی بادشاہی ہے؟ (جواب ہوگا:) اللہ واحد قہار کی۔
ٱلْيَوْمَ تُجْزَىٰ كُلُّ نَفْسِۭ بِمَا كَسَبَتْۚ لَا ظُلْمَ ٱلْيَوْمَۚ إِنَّ ٱللَّهَ سَرِيعُ ٱلْحِسَابِ ﴿17﴾
آج ہر شخص کو اس کے کیے کا بدلہ دیا جائے گا، آج کسی پر ظلم نہیں ہوگا، بے شک اللہ حساب لینے میں بہت تیز ہے۔
وَأَنذِرْهُمْ يَوْمَ ٱلْـَٔازِفَةِ إِذِ ٱلْقُلُوبُ لَدَى ٱلْحَنَاجِرِ كَـٰظِمِينَۚ مَا لِلظَّـٰلِمِينَ مِنْ حَمِيمٖ وَلَا شَفِيعٖ يُطَاعُ ﴿18﴾
اور انہیں قریب آنے والے دن سے ڈراؤ، جب دل گلوں تک آ جائیں گے اور وہ ضبط کیے ہوں گے۔ ظالموں کے لیے نہ کوئی دوست ہوگا اور نہ کوئی سفارش کرنے والا جس کی بات مانی جائے۔
يَعْلَمُ خَآئِنَةَ ٱلْأَعْيُنِ وَمَا تُخْفِي ٱلصُّدُورُ ﴿19﴾
وہ آنکھوں کی خیانت کو جانتا ہے اور جو سینے چھپاتے ہیں۔
وَٱللَّهُ يَقْضِي بِٱلْحَقِّۖ وَٱلَّذِينَ يَدْعُونَ مِن دُونِهِۦ لَا يَقْضُونَ بِشَيْءٍۚ إِنَّ ٱللَّهَ هُوَ ٱلسَّمِيعُ ٱلْبَصِيرُ ﴿20﴾
اور اللہ حق کے ساتھ فیصلہ کرتا ہے، اور جنہیں وہ اس کے سوا پکارتے ہیں وہ کسی چیز کا فیصلہ نہیں کرتے، بے شک اللہ ہی سننے والا، دیکھنے والا ہے۔
أَوَلَمْ يَسِيرُوا۟ فِي ٱلْأَرْضِ فَيَنظُرُوا۟ كَيْفَ كَانَ عَـٰقِبَةُ ٱلَّذِينَ كَانُوا۟ مِن قَبْلِهِمۖ كَانُوا۟ هُمْ أَشَدَّ مِنْهُمْ قُوَّةٗ وَءَاثَارٗا فِي ٱلْأَرْضِ فَأَخَذَهُمُ ٱللَّهُ بِذُنُوبِهِمۡۖ وَمَا كَانَ لَهُم مِّنَ ٱللَّهِ مِن وَاقٖ ﴿21﴾
کیا انہوں نے زمین میں سفر نہیں کیا تاکہ دیکھیں کہ ان سے پہلے لوگوں کا انجام کیا ہوا؟ وہ ان سے زیادہ طاقتور تھے اور زمین میں زیادہ آثار چھوڑ گئے، تو اللہ نے انہیں ان کے گناہوں کے سبب پکڑ لیا، اور اللہ سے بچانے والا کوئی نہ تھا۔
ذَٰلِكَ بِأَنَّهُمْ كَانَتْ تَأْتِيهِمْ رُسُلُهُم بِٱلْبَيِّنَـٰتِ فَكَفَرُوا۟ فَأَخَذَهُمُ ٱللَّهُۚ إِنَّهُۥ قَوِيّٞ شَدِيدُ ٱلْعِقَابِ ﴿22﴾
یہ اس لیے ہوا کہ ان کے پاس ان کے رسول واضح دلائل لے کر آتے تھے، مگر وہ انکار کرتے تھے، تو اللہ نے انہیں پکڑ لیا۔ بے شک وہ طاقتور ہے، سخت سزا دینے والا۔
وَلَقَدْ أَرْسَلْنَا مُوسَىٰ بِـَٔايَـٰتِنَا وَسُلْطَـٰنٖ مُّبِينٍ ﴿23﴾
اور بے شک ہم نے موسیٰ کو اپنی آیات اور واضح دلیل کے ساتھ بھیجا۔
إِلَىٰ فِرْعَوْنَ وَهَـٰمَـٰنَ وَقَـٰرُونَ فَقَالُوا۟ سَـٰحِرٞ كَذَّابٞ ﴿24﴾
فرعون، ہامان اور قارون کی طرف، تو انہوں نے کہا: یہ تو جادوگر ہے، جھوٹا۔
فَلَمَّا جَآءَهُم بِٱلْحَقِّ مِنْ عِندِنَا قَالُوا۟ ٱقْتُلُوا۟ أَبْنَآءَ ٱلَّذِينَ ءَامَنُوا۟ مَعَهُۥ وَٱسْتَحْيُوا۟ نِسَآءَهُمۡۚ وَمَا كَيْدُ ٱلْكَـٰفِرِينَ إِلَّا فِي ضَلَـٰلٖ ﴿25﴾
پھر جب وہ ان کے پاس ہماری طرف سے حق لے کر آیا تو انہوں نے کہا: ان لوگوں کے بیٹوں کو قتل کرو جو اس پر ایمان لائے ہیں اور ان کی عورتوں کو زندہ رکھو۔ اور کافروں کی چال تو صرف گمراہی میں ہے۔
وَقَالَ فِرْعَوْنُ ذَرُونِيٓ أَقْتُلْ مُوسَىٰ وَلْيَدْعُ رَبَّهُۥٓۖ إِنِّيٓ أَخَافُ أَن يُبَدِّلَ دِينَكُمْ أَوْ أَن يُظْهِرَ فِي ٱلْأَرْضِ ٱلْفَسَادَ ﴿26﴾
اور فرعون نے کہا: مجھے چھوڑو کہ میں موسیٰ کو قتل کر دوں، اور وہ اپنے رب کو پکارے۔ مجھے ڈر ہے کہ وہ تمہارا دین بدل دے گا یا زمین میں فساد پھیلائے گا۔
وَقَالَ مُوسَىٰٓ إِنِّي عُذْتُ بِرَبِّي وَرَبِّكُم مِّن كُلِّ مُتَكَبِّرٖ لَّا يُؤْمِنُ بِيَوْمِ ٱلْحِسَابِ ﴿27﴾
اور موسیٰ نے کہا: میں اپنے رب اور تمہارے رب کی پناہ لیتا ہوں ہر اس متکبر سے جو یومِ حساب پر ایمان نہیں رکھتا۔
وَقَالَ رَجُلٞ مُّؤْمِنٞ مِّنْ ءَالِ فِرْعَوْنَ يَكْتُمُ إِيمَـٰنَهُۥٓ أَتَقْتُلُونَ رَجُلًا أَن يَقُولَ رَبِّيَ ٱللَّهُ وَقَدْ جَآءَكُم بِٱلْبَيِّنَـٰتِ مِن رَّبِّكُمۖ وَإِن يَكُ كَـٰذِبٗا فَعَلَيْهِ كَذِبُهُۥۖ وَإِن يَكُ صَادِقٗا يُصِبْكُم بَعْضُ ٱلَّذِي يَعِدُكُمۖ إِنَّ ٱللَّهَ لَا يَهْدِي مَنْ هُوَ مُسْرِفٞ كَذَّابٞ ﴿28﴾
اور ایک مؤمن شخص نے کہا جو فرعون کے خاندان میں تھا اور اپنا ایمان چھپائے ہوئے تھا: کیا تم ایک شخص کو صرف اس بات پر قتل کرنا چاہتے ہو کہ وہ کہتا ہے میرا رب اللہ ہے، حالانکہ وہ تمہارے رب کی طرف سے واضح دلائل لے کر آیا ہے؟ اگر وہ جھوٹا ہے تو اس کا جھوٹ اسی پر ہے، اور اگر وہ سچا ہے تو تمہیں وہ کچھ پہنچ سکتا ہے جس کا وہ وعدہ کرتا ہے۔ بے شک اللہ حد سے بڑھنے والے جھوٹے کو ہدایت نہیں دیتا۔
يَـٰقَوْمِ لَكُمُ ٱلْمُلْكُ ٱلْيَوْمَ ظَـٰهِرِينَ فِي ٱلْأَرْضِ فَمَن يَنصُرُنَا مِنۢ بَأْسِ ٱللَّهِ إِن جَآءَنَاۚ قَالَ فِرْعَوْنُ مَآ أُرِيكُمْ إِلَّا مَآ أَرَىٰ وَمَآ أَهْدِيكُمْ إِلَّا سَبِيلَ ٱلرَّشَادِ ﴿29﴾
اے میری قوم! آج تمہیں زمین میں بادشاہی حاصل ہے، لیکن اگر اللہ کا عذاب ہم پر آ گیا تو کون ہماری مدد کرے گا؟ فرعون نے کہا: میں تمہیں وہی دکھاتا ہوں جو میں خود دیکھتا ہوں، اور میں تمہیں صرف بھلائی کا راستہ دکھاتا ہوں۔
وَقَالَ ٱلَّذِيٓ ءَامَنَ يَـٰقَوْمِ إِنِّيٓ أَخَافُ عَلَيْكُم مِّثْلَ يَوْمِ ٱلْأَحْزَابِ ﴿30﴾
اور ایمان لانے والے نے کہا: اے میری قوم! مجھے تم پر ان جماعتوں کے دن جیسا انجام آنے کا خوف ہے۔
مِثْلَ دَأْبِ قَوْمِ نُوحٖ وَعَادٖ وَثَمُودَ وَٱلَّذِينَ مِنۢ بَعْدِهِمۡۚ وَمَا ٱللَّهُ يُرِيدُ ظُلْمٗا لِّلْعِبَادِ ﴿31﴾
نوح، عاد، ثمود اور ان کے بعد کی قوموں کی طرح، اور اللہ بندوں پر ظلم نہیں چاہتا۔
وَيَـٰقَوْمِ إِنِّيٓ أَخَافُ عَلَيْكُمْ يَوْمَ ٱلتَّنَادِ ﴿32﴾
اے میری قوم! مجھے تم پر پکار کے دن (قیامت) کا خوف ہے۔
يَوْمَ تُوَلُّونَ مُدْبِرِينَ مَا لَكُم مِّنَ ٱللَّهِ مِنْ عَاصِمٖۚ وَمَن يُضْلِلِ ٱللَّهُ فَمَا لَهُۥ مِنْ هَادٖ ﴿33﴾
جس دن تم پیٹھ پھیر کر بھاگو گے، تمہیں اللہ سے بچانے والا کوئی نہ ہوگا، اور جسے اللہ گمراہ کر دے تو اس کے لیے کوئی ہدایت دینے والا نہیں۔
وَلَقَدْ جَآءَكُمْ يُوسُفُ مِن قَبْلُ بِٱلْبَيِّنَـٰتِ فَمَا زِلْتُمْ فِي شَكّٖ مِّمَّا جَآءَكُم بِهِۦۖ حَتَّىٰٓ إِذَا هَلَكَ قُلْتُمْ لَن يَبْعَثَ ٱللَّهُ مِنۢ بَعْدِهِۦ رَسُولٗاۚ كَذَٰلِكَ يُضِلُّ ٱللَّهُ مَنْ هُوَ مُسْرِفٞ مُّرْتَابٌ ﴿34﴾
اور بے شک تمہارے پاس اس سے پہلے یوسف واضح دلائل لے کر آیا، مگر تم ہمیشہ اس چیز کے بارے میں شک میں رہے جو وہ تمہارے پاس لایا، یہاں تک کہ جب وہ فوت ہو گیا تو تم نے کہا: اللہ اس کے بعد کوئی رسول نہیں بھیجے گا۔ اسی طرح اللہ گمراہ کرتا ہے ہر حد سے بڑھنے والے شک کرنے والے کو۔
ٱلَّذِينَ يُجَـٰدِلُونَ فِيٓ ءَايَـٰتِ ٱللَّهِ بِغَيْرِ سُلْطَـٰنٍ أَتَىٰهُمْۖ كَبُرَ مَقْتٗا عِندَ ٱللَّهِ وَعِندَ ٱلَّذِينَ ءَامَنُوا۟ۚ كَذَٰلِكَ يَطْبَعُ ٱللَّهُ عَلَىٰ كُلِّ قَلْبِ مُتَكَبِّرٖ جَبَّارٖ ﴿35﴾
جو لوگ اللہ کی آیات میں بغیر کسی دلیل کے جھگڑتے ہیں، ان کی بات اللہ اور ایمان والوں کے نزدیک سخت ناپسندیدہ ہے۔ اسی طرح اللہ ہر متکبر جابر کے دل پر مہر لگا دیتا ہے۔
وَقَالَ فِرْعَوْنُ يَـٰهَـٰمَـٰنُ ٱبْنِ لِي صَرْحٗا لَّعَلِّيٓ أَبْلُغُ ٱلْأَسْبَـٰبَ ﴿36﴾
اور فرعون نے کہا: اے ہامان! میرے لیے ایک بلند عمارت بنا تاکہ میں راستوں تک پہنچ سکوں۔
أَسْبَـٰبَ ٱلسَّمَـٰوَٰتِ فَأَطَّلِعَ إِلَىٰٓ إِلَـٰهِ مُوسَىٰ وَإِنِّي لَأَظُنُّهُۥ كَذِبٗاۚ وَكَذَٰلِكَ زُيِّنَ لِفِرْعَوْنَ سُوٓءُ عَمَلِهِۦ وَصُدَّ عَنِ ٱلسَّبِيلِۚ وَمَا كَيْدُ فِرْعَوْنَ إِلَّا فِي تَبَابٖ ﴿37﴾
آسمانوں کے راستے تاکہ میں موسیٰ کے معبود کو دیکھ سکوں، اور میں تو اسے جھوٹا ہی سمجھتا ہوں۔ اور اسی طرح فرعون کے لیے اس کا برا عمل خوشنما بنا دیا گیا اور وہ راہِ حق سے روک دیا گیا، اور فرعون کی چال تو صرف تباہی میں ہے۔
وَقَالَ ٱلَّذِيٓ ءَامَنَ يَـٰقَوْمِ ٱتَّبِعُونِ أَهْدِكُمْ سَبِيلَ ٱلرَّشَادِ ﴿38﴾
اور ایمان لانے والے نے کہا: اے میری قوم! میری پیروی کرو، میں تمہیں بھلائی کے راستے کی ہدایت دوں گا۔
يَـٰقَوْمِ إِنَّمَا هَـٰذِهِ ٱلْحَيَوٰةُ ٱلدُّنْيَا مَتَـٰعٞ وَإِنَّ ٱلْـَٔاخِرَةَ هِيَ دَارُ ٱلْقَرَارِ ﴿39﴾
اے میری قوم! یہ دنیا کی زندگی تو صرف ایک فائدہ ہے، اور آخرت ہی ہمیشہ رہنے والا گھر ہے۔
مَنْ عَمِلَ سَيِّئَةٗ فَلَا يُجْزَىٰٓ إِلَّا مِثْلَهَاۖ وَمَنْ عَمِلَ صَـٰلِحٗا مِّن ذَكَرٍ أَوْ أُنثَىٰ وَهُوَ مُؤْمِنٞ فَأُو۟لَـٰٓئِكَ يَدْخُلُونَ ٱلْجَنَّةَ يُرْزَقُونَ فِيهَا بِغَيْرِ حِسَابٖ ﴿40﴾
جو کوئی برائی کرے گا تو اسے اسی کے برابر بدلہ دیا جائے گا، اور جو نیک عمل کرے گا مرد ہو یا عورت اور وہ مؤمن ہو، تو وہ لوگ جنت میں داخل ہوں گے، اور وہاں بے حساب رزق دیے جائیں گے۔
وَيَـٰقَوْمِ مَا لِيٓ أَدْعُوكُمْ إِلَى ٱلنَّجَوٰةِ وَتَدْعُونَنِيٓ إِلَى ٱلنَّارِ ﴿41﴾
اے میری قوم! مجھے کیا ہے کہ میں تمہیں نجات کی طرف بلاتا ہوں اور تم مجھے آگ کی طرف بلاتے ہو؟
تَدْعُونَنِي لِأَكْفُرَ بِٱللَّهِ وَأُشْرِكَ بِهِۦ مَا لَيْسَ لِي بِهِۦ عِلْمٞ وَأَنَا۠ أَدْعُوكُمْ إِلَى ٱلْعَزِيزِ ٱلْغَفَّـٰرِ ﴿42﴾
تم مجھے بلاتے ہو کہ میں اللہ کا انکار کروں اور اس کے ساتھ شریک ٹھہراؤں جس کا مجھے کوئی علم نہیں، اور میں تمہیں زبردست بخشنے والے کی طرف بلاتا ہوں۔
لَا جَرَمَ أَنَّمَا تَدْعُونَنِيٓ إِلَيْهِ لَيْسَ لَهُۥ دَعْوَةٞ فِي ٱلدُّنْيَا وَلَا فِي ٱلْـَٔاخِرَةِ وَأَنَّ مَرَدَّنَآ إِلَى ٱللَّهِ وَأَنَّ ٱلْمُسْرِفِينَ هُمْ أَصْحَـٰبُ ٱلنَّارِ ﴿43﴾
یقیناً جس چیز کی طرف تم مجھے بلا رہے ہو، اس کی کوئی پکار نہ دنیا میں ہے اور نہ آخرت میں، اور ہمارا لوٹنا اللہ ہی کی طرف ہے، اور حد سے بڑھنے والے ہی دوزخ والے ہیں۔
فَسَتَذْكُرُونَ مَآ أَقُولُ لَكُمْۚ وَأُفَوِّضُ أَمْرِيٓ إِلَى ٱللَّهِۚ إِنَّ ٱللَّهَ بَصِيرُۢ بِٱلْعِبَادِ ﴿44﴾
تم جلد ہی یاد کرو گے جو میں تم سے کہہ رہا ہوں، اور میں اپنا معاملہ اللہ کے سپرد کرتا ہوں، بے شک اللہ بندوں کو خوب دیکھنے والا ہے۔
فَوَقَىٰهُ ٱللَّهُ سَيِّـَٔاتِ مَا مَكَرُوا۟ وَحَاقَ بِـَٔالِ فِرْعَوْنَ سُوٓءُ ٱلْعَذَابِ ﴿45﴾
تو اللہ نے اسے ان کی چالوں کی برائی سے بچا لیا، اور آلِ فرعون کو برے عذاب نے گھیر لیا۔
ٱلنَّارُ يُعْرَضُونَ عَلَيْهَا غُدُوّٗا وَعَشِيّٗاۖ وَيَوْمَ تَقُومُ ٱلسَّاعَةُ أَدْخِلُوٓا۟ ءَالَ فِرْعَوْنَ أَشَدَّ ٱلْعَذَابِ ﴿46﴾
آگ کے سامنے صبح و شام پیش کیے جاتے ہیں، اور جس دن قیامت قائم ہوگی (کہا جائے گا:) آلِ فرعون کو سخت ترین عذاب میں داخل کرو۔
وَإِذْ يَتَحَآجُّونَ فِي ٱلنَّارِ فَيَقُولُ ٱلضُّعَفَـٰٓؤُا۟ لِلَّذِينَ ٱسْتَكْبَرُوٓا۟ إِنَّا كُنَّا لَكُمْ تَبَعٗا فَهَلْ أَنتُم مُّغْنُونَ عَنَّا نَصِيبٗا مِّنَ ٱلنَّارِ ﴿47﴾
اور جب وہ آگ میں آپس میں جھگڑیں گے تو کمزور لوگ متکبروں سے کہیں گے: ہم تمہارے تابع تھے، تو کیا تم ہمارے لیے آگ کا کچھ حصہ ہٹا سکتے ہو؟
قَالَ ٱلَّذِينَ ٱسْتَكْبَرُوٓا۟ إِنَّا كُلّٞ فِيهَآ إِنَّ ٱللَّهَ قَدْ حَكَمَ بَيْنَ ٱلْعِبَادِ ﴿48﴾
متکبر لوگ کہیں گے: ہم سب اس میں ہیں، بے شک اللہ نے بندوں کے درمیان فیصلہ کر دیا ہے۔
وَقَالَ ٱلَّذِينَ فِي ٱلنَّارِ لِخَزَنَةِ جَهَنَّمَ ٱدْعُوا۟ رَبَّكُمْ يُخَفِّفْ عَنَّا يَوْمٗا مِّنَ ٱلْعَذَابِ ﴿49﴾
اور جو لوگ آگ میں ہوں گے وہ جہنم کے داروغوں سے کہیں گے: اپنے رب سے دعا کرو کہ وہ ایک دن کے لیے ہم سے عذاب ہلکا کر دے۔
قَالُوٓا۟ أَوَلَمْ تَكُ تَأْتِيكُمْ رُسُلُكُم بِٱلْبَيِّنَـٰتِۖ قَالُوا۟ بَلَىٰۚ قَالُوا۟ فَٱدْعُوا۟ۗ وَمَا دُعَـٰٓؤُا۟ ٱلْكَـٰفِرِينَ إِلَّا فِي ضَلَـٰلٖ ﴿50﴾
وہ کہیں گے: کیا تمہارے پاس تمہارے رسول واضح دلائل لے کر نہیں آئے تھے؟ وہ کہیں گے: کیوں نہیں! (داروغے) کہیں گے: تو دعا کرو، اور کافروں کی دعا تو صرف گمراہی میں ہے۔
إِنَّا لَنَنصُرُ رُسُلَنَا وَٱلَّذِينَ ءَامَنُوا۟ فِي ٱلْحَيَوٰةِ ٱلدُّنْيَا وَيَوْمَ يَقُومُ ٱلْأَشْهَادُ ﴿51﴾
بے شک ہم اپنے رسولوں اور ایمان والوں کی دنیا کی زندگی میں اور جس دن گواہ کھڑے ہوں گے، ضرور مدد کرتے ہیں۔
يَوْمَ لَا يَنفَعُ ٱلظَّـٰلِمِينَ مَعْذِرَتُهُمْۖ وَلَهُمُ ٱلْلَّعْنَةُ وَلَهُمْ سُوٓءُ ٱلدَّارِ ﴿52﴾
جس دن ظالموں کو ان کا عذر فائدہ نہ دے گا، اور ان کے لیے لعنت ہوگی اور ان کے لیے بدترین گھر ہوگا۔
وَلَقَدْ ءَاتَيْنَا مُوسَى ٱلْهُدَىٰ وَأَوْرَثْنَا بَنِيٓ إِسْرَٰٓءِيلَ ٱلْكِتَـٰبَ ﴿53﴾
اور بے شک ہم نے موسیٰ کو ہدایت دی اور بنی اسرائیل کو کتاب کا وارث بنایا۔
هُدٗى وَذِكْرَىٰ لِأُو۟لِي ٱلْأَلْبَـٰبِ ﴿54﴾
ہدایت اور نصیحت عقل والوں کے لیے۔
فَٱصْبِرْ إِنَّ وَعْدَ ٱللَّهِ حَقّٞۖ وَٱسْتَغْفِرْ لِذَنۢبِكَ وَسَبِّحْ بِحَمْدِ رَبِّكَ بِٱلْعَشِيِّ وَٱلْإِبْكَـٰرِ ﴿55﴾
پس صبر کرو، بے شک اللہ کا وعدہ سچا ہے، اور اپنے گناہ کی معافی مانگو، اور اپنے رب کی حمد کے ساتھ شام اور صبح تسبیح کرو۔
إِنَّ ٱلَّذِينَ يُجَـٰدِلُونَ فِيٓ ءَايَـٰتِ ٱللَّهِ بِغَيْرِ سُلْطَـٰنٍ أَتَىٰهُمْ إِن فِي صُدُورِهِمْ إِلَّا كِبْرٞ مَّا هُم بِبَـٰلِغِيهِۚ فَٱسْتَعِذْ بِٱللَّهِۖ إِنَّهُۥ هُوَ ٱلسَّمِيعُ ٱلْبَصِيرُ ﴿56﴾
بے شک جو لوگ اللہ کی آیات میں بغیر کسی دلیل کے جھگڑتے ہیں، ان کے سینوں میں صرف تکبر ہے جس تک وہ پہنچنے والے نہیں، پس اللہ سے پناہ مانگو، بے شک وہی سننے والا، دیکھنے والا ہے۔
لَخَلْقُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ أَكْبَرُ مِنْ خَلْقِ ٱلنَّاسِ وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَعْلَمُونَ ﴿57﴾
آسمانوں اور زمین کی پیدائش انسانوں کی پیدائش سے بڑی ہے، لیکن اکثر لوگ نہیں جانتے۔
وَمَا يَسْتَوِي ٱلْأَعْمَىٰ وَٱلْبَصِيرُ وَٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ وَلَا ٱلْمُسِيٓءُۚ قَلِيلٗا مَّا تَتَذَكَّرُونَ ﴿58﴾
اندھا اور دیکھنے والا برابر نہیں، اور نہ وہ جو ایمان لائے اور نیک عمل کیے اور نہ بدکار، تم بہت کم نصیحت حاصل کرتے ہو۔
إِنَّ ٱلسَّاعَةَ لَـَٔاتِيَةٞ لَّا رَيْبَ فِيهَا وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يُؤْمِنُونَ ﴿59﴾
بے شک قیامت آنے والی ہے، اس میں کوئی شک نہیں، لیکن اکثر لوگ ایمان نہیں لاتے۔
وَقَالَ رَبُّكُمُ ٱدْعُونِيٓ أَسْتَجِبْ لَكُمْۚ إِنَّ ٱلَّذِينَ يَسْتَكْبِرُونَ عَنْ عِبَادَتِي سَيَدْخُلُونَ جَهَنَّمَ دَاخِرِينَ ﴿60﴾
اور تمہارے رب نے فرمایا: مجھے پکارو، میں تمہاری دعا قبول کروں گا۔ بے شک جو لوگ میری عبادت سے تکبر کرتے ہیں وہ ذلیل ہو کر جہنم میں داخل ہوں گے۔
ٱللَّهُ ٱلَّذِي جَعَلَ لَكُمُ ٱلَّيْلَ لِتَسْكُنُوا۟ فِيهِ وَٱلنَّهَارَ مُبْصِرًاۚ إِنَّ ٱللَّهَ لَذُو فَضْلٍ عَلَى ٱلنَّاسِ وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَشْكُرُونَ ﴿61﴾
اللہ وہ ہے جس نے تمہارے لیے رات بنائی تاکہ تم اس میں آرام کرو، اور دن کو روشن بنایا۔ بے شک اللہ لوگوں پر فضل کرنے والا ہے، لیکن اکثر لوگ شکر نہیں کرتے۔
ذَٰلِكُمُ ٱللَّهُ رَبُّكُمْ خَـٰلِقُ كُلِّ شَيْءٖ لَّآ إِلَـٰهَ إِلَّا هُوَۖ فَأَنَّىٰ تُؤْفَكُونَ ﴿62﴾
یہی اللہ تمہارا رب ہے، ہر چیز کا پیدا کرنے والا، اس کے سوا کوئی معبود نہیں، پھر تم کہاں بہکائے جا رہے ہو؟
كَذَٰلِكَ يُؤْفَكُ ٱلَّذِينَ كَانُوا۟ بِـَٔايَـٰتِ ٱللَّهِ يَجْحَدُونَ ﴿63﴾
اسی طرح وہ لوگ بہکائے جاتے ہیں جو اللہ کی آیات کا انکار کرتے ہیں۔
ٱللَّهُ ٱلَّذِي جَعَلَ لَكُمُ ٱلْأَرْضَ قَرَارٗا وَٱلسَّمَآءَ بِنَآءٗ وَصَوَّرَكُمْ فَأَحْسَنَ صُوَرَكُمْ وَرَزَقَكُم مِّنَ ٱلطَّيِّبَـٰتِۚ ذَٰلِكُمُ ٱللَّهُ رَبُّكُمْۖ فَتَبَارَكَ ٱللَّهُ رَبُّ ٱلْعَـٰلَمِينَ ﴿64﴾
اللہ وہ ہے جس نے تمہارے لیے زمین کو ٹھہرنے کی جگہ بنایا، اور آسمان کو چھت بنایا، اور تمہیں صورتیں دیں اور تمہاری صورتوں کو خوبصورت بنایا، اور تمہیں پاکیزہ چیزوں سے رزق دیا۔ یہی اللہ تمہارا رب ہے، پس بابرکت ہے اللہ رب العالمین۔
هُوَ ٱلْحَيُّ لَآ إِلَـٰهَ إِلَّا هُوَ فَٱدْعُوهُ مُخْلِصِينَ لَهُ ٱلدِّينَۚ ٱلْحَمْدُ لِلَّهِ رَبِّ ٱلْعَـٰلَمِينَ ﴿65﴾
وہی زندہ ہے، اس کے سوا کوئی معبود نہیں، پس اسی کو پکارو دین کو اسی کے لیے خالص کرتے ہوئے۔ تمام تعریفیں اللہ رب العالمین کے لیے ہیں۔
قُلْ إِنِّي نُهِيتُ أَنْ أَعْبُدَ ٱلَّذِينَ تَدْعُونَ مِن دُونِ ٱللَّهِ لَمَّا جَآءَنِيَ ٱلْبَيِّنَـٰتُ مِن رَّبِّي وَأُمِرْتُ أَنْ أُسْلِمَ لِرَبِّ ٱلْعَـٰلَمِينَ ﴿66﴾
کہہ دو: مجھے منع کیا گیا ہے کہ میں ان کو پوجوں جنہیں تم اللہ کے سوا پکارتے ہو، جب میرے پاس میرے رب کی طرف سے واضح دلائل آ گئے، اور مجھے حکم دیا گیا ہے کہ میں رب العالمین کے سامنے سرِ تسلیم خم کر دوں۔
هُوَ ٱلَّذِي خَلَقَكُم مِّن تُرَابٖ ثُمَّ مِن نُّطْفَةٖ ثُمَّ مِنْ عَلَقَةٖ ثُمَّ يُخْرِجُكُمْ طِفْلٗا ثُمَّ لِتَبْلُغُوٓا۟ أَشُدَّكُمْ ثُمَّ لِتَكُونُوا۟ شُيُوخٗاۚ وَمِنكُم مَّن يُتَوَفَّىٰ مِن قَبْلُۖ وَلِتَبْلُغُوٓا۟ أَجَلٗا مُّسَمّٗى وَلَعَلَّكُمْ تَعْقِلُونَ ﴿67﴾
وہی ہے جس نے تمہیں مٹی سے پیدا کیا، پھر نطفہ سے، پھر خون کے لوتھڑے سے، پھر تمہیں بچہ بنا کر نکالا، پھر تاکہ تم اپنی جوانی کو پہنچو، پھر تاکہ تم بوڑھے ہو جاؤ، اور تم میں سے بعض اس سے پہلے ہی وفات پا جاتے ہیں، اور تاکہ تم مقررہ وقت کو پہنچو، اور شاید تم عقل سے کام لو۔
هُوَ ٱلَّذِي يُحْيِۦ وَيُمِيتُۖ فَإِذَا قَضَىٰٓ أَمْرٗا فَإِنَّمَا يَقُولُ لَهُۥ كُن فَيَكُونُ ﴿68﴾
وہی ہے جو زندہ کرتا ہے اور مارتا ہے، پھر جب وہ کسی کام کا فیصلہ کرتا ہے تو صرف کہتا ہے: "ہو جا"، تو وہ ہو جاتا ہے۔
أَلَمْ تَرَ إِلَى ٱلَّذِينَ يُجَـٰدِلُونَ فِيٓ ءَايَـٰتِ ٱللَّهِ أَنَّىٰ يُصْرَفُونَ ﴿69﴾
کیا تم نے ان لوگوں کو نہیں دیکھا جو اللہ کی آیات میں جھگڑتے ہیں؟ وہ کہاں بہکائے جا رہے ہیں؟
ٱلَّذِينَ كَذَّبُوا۟ بِٱلْكِتَـٰبِ وَبِمَآ أَرْسَلْنَا بِهِۦ رُسُلَنَا فَسَوْفَ يَعْلَمُونَ ﴿70﴾
جنہوں نے کتاب کو جھٹلایا اور اس چیز کو بھی جس کے ساتھ ہم نے اپنے رسولوں کو بھیجا، وہ جلد ہی جان لیں گے۔
إِذِ ٱلْأَغْلَـٰلُ فِيٓ أَعْنَاقِهِمْ وَٱلسَّلَـٰسِلُ يُسْحَبُونَ ﴿71﴾
جب ان کی گردنوں میں طوق ہوں گے اور زنجیریں، وہ گھسیٹے جائیں گے۔
فِي ٱلْحَمِيمِ ثُمَّ فِي ٱلنَّارِ يُسْجَرُونَ ﴿72﴾
وہ کھولتے پانی میں ہوں گے، پھر آگ میں جھونکے جائیں گے۔
ثُمَّ قِيلَ لَهُمْ أَيْنَ مَا كُنتُمْ تُشْرِكُونَ ﴿73﴾
پھر ان سے کہا جائے گا: کہاں ہیں وہ جنہیں تم شریک ٹھہراتے تھے؟
مِن دُونِ ٱللَّهِۖ قَالُوا۟ ضَلُّوا۟ عَنَّا بَل لَّمْ نَكُن نَّدْعُوا۟ مِن قَبْلُ شَيْـٔٗاۚ كَذَٰلِكَ يُضِلُّ ٱللَّهُ ٱلْكَـٰفِرِينَ ﴿74﴾
اللہ کے سوا؟ وہ کہیں گے: وہ ہم سے گم ہو گئے، بلکہ ہم پہلے کچھ بھی نہیں پکارتے تھے۔ اسی طرح اللہ کافروں کو گمراہ کرتا ہے۔
ذَٰلِكُم بِمَا كُنتُمْ تَفْرَحُونَ فِي ٱلْأَرْضِ بِغَيْرِ ٱلْحَقِّ وَبِمَا كُنتُمْ تَمْرَحُونَ ﴿75﴾
یہ اس وجہ سے ہے کہ تم زمین میں ناحق خوشیاں مناتے تھے اور اتراتے پھرتے تھے۔
ٱدْخُلُوٓا۟ أَبْوَٰبَ جَهَنَّمَ خَـٰلِدِينَ فِيهَاۖ فَبِئْسَ مَثْوَى ٱلْمُتَكَبِّرِينَ ﴿76﴾
جہنم کے دروازوں میں داخل ہو جاؤ، ہمیشہ اس میں رہنے کے لیے۔ تکبر کرنے والوں کا ٹھکانہ بہت برا ہے۔
فَٱصْبِرْ إِنَّ وَعْدَ ٱللَّهِ حَقّٞۖ فَإِمَّا نُرِيَنَّكَ بَعْضَ ٱلَّذِي نَعِدُهُمْ أَوْ نَتَوَفَّيَنَّكَ فَإِلَيْنَا يُرْجَعُونَ ﴿77﴾
پس صبر کرو، بے شک اللہ کا وعدہ سچا ہے۔ تو چاہے ہم تمہیں وہ کچھ دکھا دیں جس کا ہم ان سے وعدہ کر رہے ہیں یا تمہیں وفات دے دیں، بہرحال وہ ہماری ہی طرف لوٹائے جائیں گے۔
وَلَقَدْ أَرْسَلْنَا رُسُلٗا مِّن قَبْلِكَۖ مِنْهُم مَّن قَصَصْنَا عَلَيْكَ وَمِنْهُم مَّن لَّمْ نَقْصُصْ عَلَيْكَۚ وَمَا كَانَ لِرَسُولٍ أَن يَأْتِيَ بِـَٔايَةٍ إِلَّا بِإِذْنِ ٱللَّهِۚ فَإِذَا جَآءَ أَمْرُ ٱللَّهِ قُضِيَ بِٱلْحَقِّ وَخَسِرَ هُنَالِكَ ٱلْمُبْطِلُونَ ﴿78﴾
اور بے شک ہم نے تم سے پہلے کئی رسول بھیجے، ان میں سے بعض کے حالات ہم نے تمہیں سنائے اور بعض کے نہیں سنائے۔ اور کسی رسول کے لیے ممکن نہیں کہ وہ اللہ کے اذن کے بغیر کوئی نشانی لے آئے۔ پھر جب اللہ کا حکم آتا ہے تو حق کے ساتھ فیصلہ ہو جاتا ہے، اور وہاں باطل پرست خسارے میں پڑ جاتے ہیں۔
ٱللَّهُ ٱلَّذِي جَعَلَ لَكُمُ ٱلْأَنْعَـٰمَ لِتَرْكَبُوا۟ مِنْهَا وَمِنْهَا تَأْكُلُونَ ﴿79﴾
اللہ وہ ہے جس نے تمہارے لیے چوپائے بنائے تاکہ تم ان میں سے کچھ پر سوار ہو اور کچھ کو کھاؤ۔
وَلَكُمْ فِيهَا مَنَـٰفِعُ وَلِتَبْلُغُوا۟ عَلَيْهِا حَاجَةٗ فِي صُدُورِكُمْ وَعَلَيْهَا وَعَلَى ٱلْفُلْكِ تُحْمَلُونَ ﴿80﴾
اور ان میں تمہارے لیے فائدے بھی ہیں، اور تاکہ تم ان پر اپنی دل کی حاجت کو پہنچو، اور ان پر اور کشتیوں پر تم سوار کیے جاتے ہو۔
وَيُرِيكُمْ ءَايَـٰتِهِۦ فَأَيَّ ءَايَـٰتِ ٱللَّهِ تُنكِرُونَ ﴿81﴾
اور وہ تمہیں اپنی نشانیاں دکھاتا ہے، تو اللہ کی کون سی نشانیاں تم جھٹلاتے ہو؟
أَفَلَمْ يَسِيرُوا۟ فِي ٱلْأَرْضِ فَيَنظُرُوا۟ كَيْفَ كَانَ عَـٰقِبَةُ ٱلَّذِينَ مِن قَبْلِهِمۡۚ كَانُوا۟ أَكْثَرَ مِنْهُمْ وَأَشَدَّ قُوَّةٗ وَءَاثَارٗا فِي ٱلْأَرْضِ فَمَآ أَغْنَىٰ عَنْهُم مَّا كَانُوا۟ يَكْسِبُونَ ﴿82﴾
کیا انہوں نے زمین میں سفر نہیں کیا تاکہ دیکھیں کہ ان سے پہلے لوگوں کا انجام کیا ہوا؟ وہ ان سے زیادہ تھے اور زیادہ طاقتور تھے اور زمین میں زیادہ آثار چھوڑ گئے، مگر جو کچھ وہ کماتے تھے وہ ان کے کسی کام نہ آیا۔
فَلَمَّا جَآءَتْهُمْ رُسُلُهُم بِٱلْبَيِّنَـٰتِ فَرِحُوا۟ بِمَا عِندَهُم مِّنَ ٱلْعِلْمِ وَحَاقَ بِهِم مَّا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ ﴿83﴾
پھر جب ان کے پاس ان کے رسول واضح دلائل لے کر آئے تو وہ اپنے علم پر خوش ہو گئے، اور جس چیز کا وہ مذاق اڑاتے تھے وہی انہیں گھیر لیا۔
فَلَمَّا رَأَوْا۟ بَأْسَنَا قَالُوٓا۟ ءَامَنَّا بِٱللَّهِ وَحْدَهُۥ وَكَفَرْنَا بِمَا كُنَّا بِهِۦ مُشْرِكِينَ ﴿84﴾
پھر جب انہوں نے ہمارا عذاب دیکھا تو کہنے لگے: ہم اللہ پر ایمان لائے جو اکیلا ہے، اور ہم نے ان چیزوں کا انکار کیا جنہیں ہم شریک ٹھہراتے تھے۔
فَلَمْ يَكُ يَنفَعُهُمْ إِيمَـٰنُهُمْ لَمَّا رَأَوْا۟ بَأْسَنَاۖ سُنَّةَ ٱللَّهِ ٱلَّتِي قَدْ خَلَتْ فِي عِبَادِهِۦۖ وَخَسِرَ هُنَالِكَ ٱلْكَـٰفِرُونَ ﴿85﴾
پھر جب انہوں نے ہمارا عذاب دیکھا تو ان کا ایمان انہیں فائدہ نہ دے سکا۔ یہ اللہ کی سنت ہے جو اس کے بندوں میں پہلے گزر چکی ہے، اور وہاں کافر خسارے میں پڑ گئے۔

تَنزِيلٞ مِّنَ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ ﴿2﴾
یہ نازل کیا گیا ہے رحمٰن، رحم کرنے والے کی طرف سے۔
كِتَـٰبٞ فُصِّلَتْ ءَايَـٰتُهُۥ قُرْءَانًا عَرَبِيّٗا لِّقَوْمٖ يَعْلَمُونَ ﴿3﴾
یہ ایک کتاب ہے جس کی آیات واضح طور پر بیان کی گئی ہیں، عربی قرآن ہے ان لوگوں کے لیے جو جانتے ہیں۔
بَشِيرٗا وَنَذِيرٗا فَأَعْرَضَ أَكْثَرُهُمْ فَهُمْ لَا يَسْمَعُونَ ﴿4﴾
خوشخبری دینے والا اور ڈرانے والا، مگر ان میں سے اکثر نے منہ موڑ لیا، پس وہ سنتے نہیں۔
وَقَالُوا۟ قُلُوبُنَا فِيٓ أَكِنَّةٖ مِّمَّا تَدْعُونَآ إِلَيْهِ وَفِيٓ ءَاذَانِنَا وَقْرٞ وَمِنۢ بَيْنِنَا وَبَيْنِكَ حِجَابٞ فَٱعْمَلْ إِنَّنَا عَـٰمِلُونَ ﴿5﴾
اور وہ کہتے ہیں: ہمارے دل اس چیز سے پردے میں ہیں جس کی طرف آپ ہمیں بلاتے ہیں، اور ہمارے کانوں میں بوجھ ہے، اور ہمارے اور آپ کے درمیان پردہ ہے، پس آپ اپنا کام کریں، ہم بھی اپنا کام کر رہے ہیں۔
قُلْ إِنَّمَآ أَنَا۠ بَشَرٞ مِّثْلُكُمْ يُوحَىٰٓ إِلَيَّ أَنَّمَآ إِلَـٰهُكُمْ إِلَـٰهٞ وَٰحِدٞ فَٱسْتَقِيمُوٓا۟ إِلَيْهِ وَٱسْتَغْفِرُوهُۚ وَوَيْلٞ لِّلْمُشْرِكِينَ ﴿6﴾
کہہ دو: میں تو تم جیسا ہی ایک انسان ہوں، میری طرف وحی کی جاتی ہے کہ تمہارا معبود ایک ہی معبود ہے، پس اسی کی طرف سیدھے ہو جاؤ اور اس سے بخشش مانگو، اور مشرکوں کے لیے ہلاکت ہے۔
ٱلَّذِينَ لَا يُؤْتُونَ ٱلزَّكَوٰةَ وَهُم بِٱلْـَٔاخِرَةِ هُمْ كَـٰفِرُونَ ﴿7﴾
جو زکوٰۃ نہیں دیتے اور وہی آخرت کے منکر ہیں۔
إِنَّ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ لَهُمْ أَجْرٌ غَيْرُ مَمْنُونٖ ﴿8﴾
بے شک جو لوگ ایمان لائے اور نیک عمل کیے، ان کے لیے ایسا اجر ہے جس میں کمی نہیں۔
قُلْ أَئِنَّكُمْ لَتَكْفُرُونَ بِٱلَّذِي خَلَقَ ٱلْأَرْضَ فِي يَوْمَيْنِ وَتَجْعَلُونَ لَهُۥٓ أَندَادٗاۚ ذَٰلِكَ رَبُّ ٱلْعَـٰلَمِينَ ﴿9﴾
کہہ دو: کیا تم اس سے کفر کرتے ہو جس نے زمین کو دو دن میں پیدا کیا، اور اس کے لیے شریک بناتے ہو؟ وہی رب ہے تمام جہانوں کا۔
وَجَعَلَ فِيهَا رَوَٰسِيَ مِن فَوْقِهَا وَبَـٰرَكَ فِيهَا وَقَدَّرَ فِيهَآ أَقْوَاتَهَا فِيٓ أَرْبَعَةِ أَيَّامٖۚ سَوَآءٗ لِّلسَّآئِلِينَ ﴿10﴾
اور اس نے اس میں اوپر سے پہاڑ رکھے، اور اس میں برکت دی، اور اس میں اس کی خوراکیں مقرر کیں چار دن میں، سوال کرنے والوں کے لیے برابر۔
ثُمَّ ٱسْتَوَىٰٓ إِلَى ٱلسَّمَآءِ وَهِيَ دُخَانٞ فَقَالَ لَهَا وَلِلْأَرْضِ ٱئْتِيَا طَوْعًا أَوْ كَرْهٗا قَالَتَآ أَتَيْنَا طَآئِعِينَ ﴿11﴾
پھر وہ آسمان کی طرف متوجہ ہوا جبکہ وہ دھواں تھا، تو اس نے آسمان اور زمین سے کہا: خوشی سے یا زبردستی آ جاؤ، دونوں نے کہا: ہم خوشی سے حاضر ہیں۔
فَقَضَىٰهُنَّ سَبْعَ سَمَـٰوَٰتٖ فِي يَوْمَيْنِ وَأَوْحَىٰ فِي كُلِّ سَمَآءٍ أَمْرَهَاۚ وَزَيَّنَّا ٱلسَّمَآءَ ٱلدُّنْيَا بِمَصَـٰبِيحَ وَحِفْظٗاۚ ذَٰلِكَ تَقْدِيرُ ٱلْعَزِيزِ ٱلْعَلِيمِ ﴿12﴾
پھر اس نے انہیں دو دن میں سات آسمان بنا دیے، اور ہر آسمان میں اس کا حکم نازل کیا، اور ہم نے دنیا کے آسمان کو چراغوں سے آراستہ کیا اور حفاظت کے لیے، یہ زبردست، علم والے کا اندازہ ہے۔
فَإِنْ أَعْرَضُوا۟ فَقُلْ أَنذَرْتُكُمْ صَـٰعِقَةٗ مِّثْلَ صَـٰعِقَةِ عَادٖ وَثَمُودَ ﴿13﴾
پھر اگر وہ منہ موڑیں تو کہہ دو: میں تمہیں ایسی کڑک سے ڈراتا ہوں جیسی عاد اور ثمود پر آئی تھی۔
إِذْ جَآءَتْهُمُ ٱلرُّسُلُ مِنۢ بَيْنِ أَيْدِيهِمْ وَمِنْ خَلْفِهِمْ أَلَّا تَعْبُدُوٓا۟ إِلَّا ٱللَّهَۖ قَالُوا۟ لَوْ شَآءَ رَبُّنَا لَأَنزَلَ مَلَـٰٓئِكَةٗ فَإِنَّا بِمَآ أُرْسِلْتُم بِهِۦ كَـٰفِرُونَ ﴿14﴾
جب ان کے پاس رسول آئے ان کے آگے اور پیچھے سے (کہتے ہوئے): اللہ کے سوا کسی کی عبادت نہ کرو، تو انہوں نے کہا: اگر ہمارا رب چاہتا تو فرشتے نازل کرتا، پس ہم اس چیز کے منکر ہیں جس کے ساتھ تم بھیجے گئے ہو۔
فَأَمَّا عَادٞ فَٱسْتَكْبَرُوا۟ فِي ٱلْأَرْضِ بِغَيْرِ ٱلْحَقِّ وَقَالُوا۟ مَنْ أَشَدُّ مِنَّا قُوَّةًۚ أَوَلَمْ يَرَوْا۟ أَنَّ ٱللَّهَ ٱلَّذِي خَلَقَهُمْ هُوَ أَشَدُّ مِنْهُمْ قُوَّةٗۚ وَكَانُوا۟ بِـَٔايَـٰتِنَا يَجْحَدُونَ ﴿15﴾
پس عاد نے زمین میں ناحق تکبر کیا اور کہا: ہم سے زیادہ طاقتور کون ہے؟ کیا انہوں نے نہیں دیکھا کہ اللہ جس نے انہیں پیدا کیا وہ ان سے زیادہ طاقتور ہے؟ اور وہ ہماری آیات کا انکار کرتے رہے۔
فَأَرْسَلْنَا عَلَيْهِمْ رِيحٗا صَرْصَرٗا فِيٓ أَيَّامٖ نَّحِسَاتٖ لِّنُذِيقَهُمْ عَذَابَ ٱلْخِزْيِ فِي ٱلْحَيَوٰةِ ٱلدُّنْيَاۖ وَلَعَذَابُ ٱلْـَٔاخِرَةِ أَخْزَىٰۖ وَهُمْ لَا يُنصَرُونَ ﴿16﴾
تو ہم نے ان پر سخت آندھی بھیجی کئی منحوس دنوں میں تاکہ ہم انہیں دنیا کی زندگی میں ذلت کا عذاب چکھائیں، اور یقیناً آخرت کا عذاب زیادہ رسوا کرنے والا ہے، اور وہ مدد نہیں کیے جائیں گے۔
وَأَمَّا ثَمُودُ فَهَدَيْنَـٰهُمْ فَٱسْتَحَبُّوا۟ ٱلْعَمَىٰ عَلَى ٱلْهُدَىٰ فَأَخَذَتْهُمْ صَـٰعِقَةُ ٱلْعَذَابِ ٱلْهُونِ بِمَا كَانُوا۟ يَكْسِبُونَ ﴿17﴾
اور رہے ثمود، تو ہم نے انہیں ہدایت دی، مگر انہوں نے ہدایت پر اندھے پن کو ترجیح دی، تو انہیں ذلت کے عذاب کی کڑک نے پکڑ لیا ان کے اعمال کے سبب۔
وَنَجَّيْنَا ٱلَّذِينَ ءَامَنُوا۟ وَكَانُوا۟ يَتَّقُونَ ﴿18﴾
اور ہم نے ان لوگوں کو نجات دی جو ایمان لائے اور پرہیزگار تھے۔
وَيَوْمَ يُحْشَرُ أَعْدَآءُ ٱللَّهِ إِلَى ٱلنَّارِ فَهُمْ يُوزَعُونَ ﴿19﴾
اور جس دن اللہ کے دشمنوں کو جہنم کی طرف اکٹھا کیا جائے گا، تو وہ روکے جائیں گے۔
حَتَّىٰٓ إِذَا مَا جَآءُوهَا شَهِدَ عَلَيْهِمْ سَمْعُهُمْ وَأَبْصَـٰرُهُمْ وَجُلُودُهُم بِمَا كَانُوا۟ يَعْمَلُونَ ﴿20﴾
یہاں تک کہ جب وہ اس کے پاس پہنچیں گے تو ان کے کان، ان کی آنکھیں اور ان کی کھالیں ان کے خلاف گواہی دیں گی ان اعمال کی جو وہ کیا کرتے تھے۔
وَقَالُوا۟ لِجُلُودِهِمْ لِمَ شَهِدتُّمْ عَلَيْنَاۖ قَالُوٓا۟ أَنطَقَنَا ٱللَّهُ ٱلَّذِيٓ أَنطَقَ كُلَّ شَيْءٖۖ وَهُوَ خَلَقَكُمْ أَوَّلَ مَرَّةٖ وَإِلَيْهِ تُرْجَعُونَ ﴿21﴾
اور وہ اپنی کھالوں سے کہیں گے: تم نے ہمارے خلاف کیوں گواہی دی؟ وہ کہیں گی: ہمیں اللہ نے بولنے کی طاقت دی جس نے ہر چیز کو بولنے کی طاقت دی، اور اسی نے تمہیں پہلی بار پیدا کیا، اور اسی کی طرف تم لوٹائے جاؤ گے۔
وَمَا كُنتُمْ تَسْتَتِرُونَ أَن يَشْهَدَ عَلَيْكُمْ سَمْعُكُمْ وَلَآ أَبْصَـٰرُكُمْ وَلَا جُلُودُكُمْ وَلَـٰكِن ظَنَنتُمْ أَنَّ ٱللَّهَ لَا يَعْلَمُ كَثِيرٗا مِّمَّا تَعْمَلُونَ ﴿22﴾
اور تم اس سے نہیں چھپتے تھے کہ تمہارے خلاف تمہارے کان، تمہاری آنکھیں اور تمہاری کھالیں گواہی دیں، بلکہ تم نے یہ گمان کیا کہ اللہ تمہارے بہت سے اعمال کو نہیں جانتا۔
وَذَٰلِكُمْ ظَنُّكُمُ ٱلَّذِي ظَنَنتُم بِرَبِّكُمْ أَرْدَىٰكُمْ فَأَصْبَحْتُم مِّنَ ٱلْخَـٰسِرِينَ ﴿23﴾
اور یہی تمہارا گمان تھا جو تم نے اپنے رب کے بارے میں کیا، اس نے تمہیں ہلاکت میں ڈال دیا، اور تم خسارہ اٹھانے والوں میں سے ہو گئے۔
فَإِن يَصْبِرُوا۟ فَٱلنَّارُ مَثْوٗى لَّهُمْۖ وَإِن يَسْتَعْتِبُوا۟ فَمَا هُم مِّنَ ٱلْمُعْتَبِينَ ﴿24﴾
پس اگر وہ صبر کریں تو آگ ہی ان کا ٹھکانہ ہے، اور اگر وہ معافی مانگیں تو وہ معاف کیے جانے والوں میں سے نہیں ہوں گے۔
وَقَيَّضْنَا لَهُمْ قُرَنَآءَ فَزَيَّنُوا۟ لَهُم مَّا بَيْنَ أَيْدِيهِمْ وَمَا خَلْفَهُمْ وَحَقَّ عَلَيْهِمُ ٱلْقَوْلُ فِيٓ أُمَمٖ قَدْ خَلَتْ مِن قَبْلِهِم مِّنَ ٱلْجِنِّ وَٱلْإِنسِۚ إِنَّهُمْ كَانُوا۟ خَـٰسِرِينَ ﴿25﴾
اور ہم نے ان کے لیے ساتھی مقرر کر دیے، تو انہوں نے ان کے آگے اور پیچھے کی چیزیں ان کے لیے خوشنما بنا دیں، اور ان پر وہی بات ثابت ہو گئی جو ان سے پہلے گزرے ہوئے جنوں اور انسانوں کی جماعتوں پر ثابت ہوئی، بے شک وہ خسارہ اٹھانے والے تھے۔
وَقَالَ ٱلَّذِينَ كَفَرُوا۟ لَا تَسْمَعُوا۟ لِهَـٰذَا ٱلْقُرْءَانِ وَٱلْغَوْا۟ فِيهِ لَعَلَّكُمْ تَغْلِبُونَ ﴿26﴾
اور کافروں نے کہا: اس قرآن کو مت سنو اور اس میں شور مچاؤ تاکہ تم غالب آ جاؤ۔
فَلَنُذِيقَنَّ ٱلَّذِينَ كَفَرُوا۟ عَذَابٗا شَدِيدٗا وَلَنَجْزِيَنَّهُمْ أَسْوَأَ ٱلَّذِي كَانُوا۟ يَعْمَلُونَ ﴿27﴾
تو ہم کافروں کو سخت عذاب چکھائیں گے، اور ہم انہیں ان کے بدترین اعمال کا بدلہ دیں گے۔
ذَٰلِكَ جَزَآءُ أَعْدَآءِ ٱللَّهِ ٱلنَّارُۖ لَهُمْ فِيهَا دَارُ ٱلْخُلْدِۖ جَزَآءًۢ بِمَا كَانُوا۟ بِـَٔايَـٰتِنَا يَجْحَدُونَ ﴿28﴾
یہ اللہ کے دشمنوں کی جزا ہے — جہنم، اس میں ان کے لیے ہمیشہ رہنے کا گھر ہے، بدلہ اس چیز کا جو وہ ہماری آیات کا انکار کرتے تھے۔
وَقَالَ ٱلَّذِينَ كَفَرُوا۟ رَبَّنَآ أَرِنَا ٱلَّذَيْنِ أَضَلَّانَا مِنَ ٱلْجِنِّ وَٱلْإِنسِ نَجْعَلْهُمَا تَحْتَ أَقْدَامِنَا لِيَكُونَا مِنَ ٱلْأَسْفَلِينَ ﴿29﴾
اور کافر کہیں گے: اے ہمارے رب! ہمیں وہ دونوں دکھا دے جنہوں نے ہمیں گمراہ کیا — جنوں اور انسانوں میں سے — تاکہ ہم انہیں اپنے قدموں کے نیچے روند ڈالیں تاکہ وہ سب سے نیچے ہوں۔
إِنَّ ٱلَّذِينَ قَالُوا۟ رَبُّنَا ٱللَّهُ ثُمَّ ٱسْتَقَـٰمُوا۟ تَتَنَزَّلُ عَلَيْهِمُ ٱلْمَلَـٰٓئِكَةُ أَلَّا تَخَافُوا۟ وَلَا تَحْزَنُوا۟ وَأَبْشِرُوا۟ بِٱلْجَنَّةِ ٱلَّتِي كُنتُمْ تُوعَدُونَ ﴿30﴾
بے شک جنہوں نے کہا: ہمارا رب اللہ ہے، پھر اس پر قائم رہے، ان پر فرشتے نازل ہوتے ہیں (کہتے ہیں): نہ ڈرو اور نہ غم کرو، اور خوش ہو جاؤ اس جنت کی بشارت سے جس کا تم سے وعدہ کیا گیا تھا۔
نَحْنُ أَوْلِيَآؤُكُمْ فِي ٱلْحَيَوٰةِ ٱلدُّنْيَا وَفِي ٱلْـَٔاخِرَةِۖ وَلَكُمْ فِيهَا مَا تَشْتَهِيٓ أَنفُسُكُمْ وَلَكُمْ فِيهَا مَا تَدَّعُونَ ﴿31﴾
ہم تمہارے دوست ہیں دنیا کی زندگی میں بھی اور آخرت میں بھی، اور تمہارے لیے اس میں وہ سب کچھ ہے جو تمہارے دل چاہیں گے، اور تمہارے لیے اس میں وہ کچھ ہے جو تم مانگو گے۔
نُزُلٗا مِّنْ غَفُورٖ رَّحِيمٖ ﴿32﴾
یہ بخشنے والے مہربان کی طرف سے مہمانی ہے۔
وَمَنْ أَحْسَنُ قَوْلٗا مِّمَّن دَعَآ إِلَى ٱللَّهِ وَعَمِلَ صَـٰلِحٗا وَقَالَ إِنَّنِي مِنَ ٱلْمُسْلِمِينَ ﴿33﴾
اور اس شخص سے بہتر بات کس کی ہو سکتی ہے جو اللہ کی طرف بلائے اور نیک عمل کرے اور کہے: بے شک میں مسلمانوں میں سے ہوں۔
وَلَا تَسْتَوِي ٱلْحَسَنَةُ وَلَا ٱلسَّيِّئَةُۚ ٱدْفَعْ بِٱلَّتِي هِيَ أَحْسَنُ فَإِذَا ٱلَّذِي بَيْنَكَ وَبَيْنَهُۥ عَدَاوَةٞ كَأَنَّهُۥ وَلِيٌّ حَمِيمٞ ﴿34﴾
اور نیکی اور برائی برابر نہیں ہو سکتی، برائی کو اس طریقے سے دفع کرو جو بہتر ہو، تو وہ شخص جس کے ساتھ تمہاری دشمنی ہے گویا وہ تمہارا گرمجوش دوست بن جائے گا۔
وَمَا يُلَقَّىٰهَآ إِلَّا ٱلَّذِينَ صَبَرُوا۟ۖ وَمَا يُلَقَّىٰهَآ إِلَّا ذُو حَظٍّ عَظِيمٖ ﴿35﴾
اور یہ بات صرف وہی لوگ حاصل کرتے ہیں جو صبر کرتے ہیں، اور یہ صرف وہی حاصل کرتا ہے جو بڑے نصیب والا ہو۔
وَإِمَّا يَنزَغَنَّكَ مِنَ ٱلشَّيْطَـٰنِ نَزْغٞ فَٱسْتَعِذْ بِٱللَّهِۚ إِنَّهُۥ هُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ ﴿36﴾
اور اگر شیطان کی طرف سے کوئی وسوسہ آئے تو اللہ کی پناہ مانگو، بے شک وہی سننے والا، جاننے والا ہے۔
وَمِنْ ءَايَـٰتِهِۦ ٱلَّيْلُ وَٱلنَّهَارُ وَٱلشَّمْسُ وَٱلْقَمَرُۚ لَا تَسْجُدُوا۟ لِلشَّمْسِ وَلَا لِلْقَمَرِ وَٱسْجُدُوا۟ لِلَّهِ ٱلَّذِي خَلَقَهُنَّ إِن كُنتُمْ إِيَّاهُ تَعْبُدُونَ ﴿37﴾
اور اس کی نشانیوں میں سے رات اور دن، سورج اور چاند ہیں، سورج کو سجدہ نہ کرو اور نہ چاند کو، بلکہ اللہ کو سجدہ کرو جس نے انہیں پیدا کیا، اگر تم اسی کی عبادت کرتے ہو۔
فَإِنِ ٱسْتَكْبَرُوا۟ فَٱلَّذِينَ عِندَ رَبِّكَ يُسَبِّحُونَ لَهُۥ بِٱلَّيْلِ وَٱلنَّهَارِ وَهُمْ لَا يَسْـَٔمُونَ ﴿38﴾
پھر اگر وہ تکبر کریں تو جو لوگ آپ کے رب کے پاس ہیں وہ رات دن اس کی تسبیح کرتے ہیں اور وہ تھکتے نہیں۔
وَمِنْ ءَايَـٰتِهِۦٓ أَنَّكَ تَرَى ٱلْأَرْضَ خَـٰشِعَةٗ فَإِذَآ أَنزَلْنَا عَلَيْهَا ٱلْمَآءَ ٱهْتَزَّتْ وَرَبَتْۚ إِنَّ ٱلَّذِيٓ أَحْيَاهَا لَمُحْيِ ٱلْمَوْتَىٰٓۚ إِنَّهُۥ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٌ ﴿39﴾
اور اس کی نشانیوں میں سے یہ ہے کہ تم زمین کو جھکی ہوئی دیکھتے ہو، پھر جب ہم اس پر پانی برساتے ہیں تو وہ حرکت کرتی ہے اور پھولتی ہے، بے شک جس نے اسے زندہ کیا وہی مردوں کو زندہ کرے گا، بے شک وہ ہر چیز پر قادر ہے۔
إِنَّ ٱلَّذِينَ يُلْحِدُونَ فِيٓ ءَايَـٰتِنَا لَا يَخْفَوْنَ عَلَيْنَآۗ أَفَمَن يُلْقَىٰ فِي ٱلنَّارِ خَيْرٌ أَم مَّن يَأْتِيٓ ءَامِنٗا يَوْمَ ٱلْقِيَـٰمَةِۚ ٱعْمَلُوا۟ مَا شِئْتُمْ إِنَّهُۥ بِمَا تَعْمَلُونَ بَصِيرٌ ﴿40﴾
بے شک جو لوگ ہماری آیات میں کجی اختیار کرتے ہیں وہ ہم سے چھپے نہیں، کیا وہ شخص جو آگ میں ڈالا جائے بہتر ہے یا وہ جو قیامت کے دن امن کے ساتھ آئے؟ جو چاہو کرو، بے شک وہ تمہارے اعمال کو دیکھنے والا ہے۔
إِنَّ ٱلَّذِينَ كَفَرُوا۟ بِٱلذِّكْرِ لَمَّا جَآءَهُمْۚ وَإِنَّهُۥ لَكِتَـٰبٌ عَزِيزٞ ﴿41﴾
بے شک جنہوں نے ذکر (قرآن) کا انکار کیا جب وہ ان کے پاس آیا، اور بے شک وہ ایک زبردست کتاب ہے۔
لَّا يَأْتِيهِ ٱلْبَـٰطِلُ مِنۢ بَيْنِ يَدَيْهِ وَلَا مِنْ خَلْفِهِۦۖ تَنزِيلٞ مِّنۡ حَكِيمٍ حَمِيدٖ ﴿42﴾
اس کے آگے اور پیچھے سے باطل نہیں آ سکتا، یہ حکمت والے، قابلِ تعریف کی طرف سے نازل کردہ ہے۔
مَا يُقَالُ لَكَ إِلَّا مَا قَدْ قِيلَ لِلرُّسُلِ مِن قَبْلِكَۚ إِنَّ رَبَّكَ لَذُو مَغْفِرَةٖ وَذُو عِقَابٍ أَلِيمٖ ﴿43﴾
آپ سے وہی بات کہی جاتی ہے جو آپ سے پہلے رسولوں سے کہی گئی، بے شک آپ کا رب بخشنے والا بھی ہے اور دردناک سزا دینے والا بھی۔
وَلَوْ جَعَلْنَـٰهُ قُرْءَانًا أَعْجَمِيّٗا لَّقَالُوا۟ لَوْلَا فُصِّلَتْ ءَايَـٰتُهُۥٓۚ أَأَعْجَمِيّٞ وَعَرَبِيّٞۚ قُلْ هُوَ لِلَّذِينَ ءَامَنُوا۟ هُدٗى وَشِفَآءٞۖ وَٱلَّذِينَ لَا يُؤْمِنُونَ فِيٓ ءَاذَانِهِمْ وَقْرٞ وَهُوَ عَلَيْهِمْ عَمٗىۚ أُو۟لَـٰٓئِكَ يُنَادَوْنَ مِن مَّكَانِۭ بَعِيدٖ ﴿44﴾
اور اگر ہم اسے عجمی قرآن بناتے تو وہ کہتے: اس کی آیات واضح کیوں نہیں کی گئیں؟ کیا عجمی اور عربی؟ کہہ دو: یہ ایمان والوں کے لیے ہدایت اور شفا ہے، اور جو ایمان نہیں رکھتے ان کے کانوں میں بوجھ ہے اور وہ ان پر اندھا پن ہے، وہ ایسے ہیں جیسے دور سے پکارے جا رہے ہوں۔
وَلَقَدْ ءَاتَيْنَا مُوسَى ٱلْكِتَـٰبَ فَٱخْتُلِفَ فِيهِۚ وَلَوْلَا كَلِمَةٞ سَبَقَتْ مِن رَّبِّكَ لَقُضِيَ بَيْنَهُمْۚ وَإِنَّهُمْ لَفِي شَكّٖ مِّنْهُ مُرِيبٖ ﴿45﴾
اور ہم نے موسیٰ کو کتاب دی تو اس میں اختلاف کیا گیا، اور اگر آپ کے رب کی طرف سے ایک بات پہلے نہ ہو چکی ہوتی تو ان کے درمیان فیصلہ کر دیا جاتا، اور بے شک وہ اس کے بارے میں شک میں پڑے ہوئے ہیں۔
مَنۡ عَمِلَ صَـٰلِحٗا فَلِنَفْسِهِۦۖ وَمَنْ أَسَآءَ فَعَلَيْهَاۗ وَمَا رَبُّكَ بِظَلَّـٰمٖ لِّلْعَبِيدِ ﴿46﴾
جو نیک عمل کرے گا وہ اپنے ہی فائدے کے لیے کرے گا، اور جو برائی کرے گا وہ اسی پر پڑے گی، اور آپ کا رب بندوں پر ظلم کرنے والا نہیں ہے۔
