29 پارہ


سورة الملك
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
تَبَارَكَ ٱلَّذِى بِيَدِهِ ٱلْمُلْكُ وَهُوَ عَلَىٰ كُلِّ شَىْءٍۢ قَدِيرٌ ﴿1﴾
بہت بابرکت ہے وہ ذات جس کے ہاتھ میں بادشاہی ہے، اور وہ ہر چیز پر قدرت رکھتا ہے۔
ٱلَّذِى خَلَقَ ٱلْمَوْتَ وَٱلْحَيَوٰةَ لِيَبْلُوَكُمْ أَيُّكُمْ أَحْسَنُ عَمَلًۭا ۚ وَهُوَ ٱلْعَزِيزُ ٱلْغَفُورُ ﴿2﴾
جس نے موت اور زندگی کو پیدا کیا تاکہ تمہیں آزمائے کہ تم میں سے کون بہتر عمل کرتا ہے، اور وہی زبردست، بہت بخشنے والا ہے۔
ٱلَّذِى خَلَقَ سَبْعَ سَمَـٰوَٰتٍۢ طِبَاقًۭا ۖ مَّا تَرَىٰ فِى خَلْقِ ٱلرَّحْمَـٰنِ مِن تَفَـٰوُتٍۢ ۖ فَٱرْجِعِ ٱلْبَصَرَ هَلْ تَرَىٰ مِن فُطُورٍۢ ﴿3﴾
جس نے سات آسمان تہہ بہ تہہ بنائے، تم رحمان کی تخلیق میں کوئی بے ترتیبی نہیں دیکھو گے، دوبارہ نظر دوڑاؤ، کیا تمہیں کوئی شگاف نظر آتا ہے؟
ثُمَّ ٱرْجِعِ ٱلْبَصَرَ كَرَّتَيْنِ يَنقَلِبْ إِلَيْكَ ٱلْبَصَرُ خَاسِئًۭا وَهُوَ حَسِيرٌۭ ﴿4﴾
پھر دوبارہ نظر دوڑاؤ، نظر تمہاری تھک کر، نامراد لوٹے گی۔
وَلَقَدْ زَيَّنَّا ٱلسَّمَآءَ ٱلدُّنْيَا بِمَصَـٰبِيحَ وَجَعَلْنَـٰهَا رُجُومًۭا لِّلشَّيَـٰطِينِ ۖ وَأَعْتَدْنَا لَهُمْ عَذَابَ ٱلسَّعِيرِ ﴿5﴾
اور ہم نے قریب کے آسمان کو چراغوں سے مزین کیا، اور انہیں شیطانوں کے لیے مار بنانے والا بنایا، اور ہم نے ان کے لیے بھڑکتی آگ کا عذاب تیار کر رکھا ہے۔
وَلِلَّذِينَ كَفَرُوا۟ بِرَبِّهِمْ عَذَابُ جَهَنَّمَ ۖ وَبِئْسَ ٱلْمَصِيرُ ﴿6﴾
اور جنہوں نے اپنے رب سے کفر کیا ان کے لیے جہنم کا عذاب ہے، اور وہ بہت ہی برا ٹھکانہ ہے۔
إِذَآ أُلْقُوا۟ فِيهَا سَمِعُوا۟ لَهَا شَهِيقًۭا وَهِىَ تَفُورُ ﴿7﴾
جب وہ اس میں ڈالے جائیں گے تو وہ اس کا دھاڑنا سنیں گے، اور وہ جوش مار رہی ہوگی۔
تَكَادُ تَمَيَّزُ مِنَ ٱلْغَيْظِ ۖ كُلَّمَآ أُلْقِىَ فِيهَا فَوْجٌۭ سَأَلَهُمْ خَزَنَتُهَآ أَلَمْ يَأْتِكُمْ نَذِيرٌۭ ﴿8﴾
وہ غصے سے پھٹنے کو ہے، جب بھی اس میں کوئی گروہ ڈالا جائے گا، اس کے داروغہ ان سے پوچھیں گے: کیا تمہارے پاس کوئی ڈرانے والا نہیں آیا؟
قَالُوا۟ بَلَىٰ قَدْ جَآءَنَا نَذِيرٌۭ فَكَذَّبْنَا وَقُلْنَا مَا نَزَّلَ ٱللَّهُ مِن شَىْءٍ إِنْ أَنتُمْ إِلَّا فِى ضَلَـٰلٍۢ كَبِيرٍ ﴿9﴾
وہ کہیں گے: کیوں نہیں، ہمارے پاس ڈرانے والا آیا تھا، مگر ہم نے اسے جھٹلا دیا اور کہا: اللہ نے کچھ نازل نہیں کیا، تم تو بڑی گمراہی میں ہو۔
وَقَالُوا۟ لَوْ كُنَّا نَسْمَعُ أَوْ نَعْقِلُ مَا كُنَّا فِىٓ أَصْحَـٰبِ ٱلسَّعِيرِ ﴿10﴾
اور وہ کہیں گے: اگر ہم سنتے یا سمجھتے تو ہم بھڑکتی آگ والوں میں نہ ہوتے۔
فَٱعْتَرَفُوا۟ بِذَنۢبِهِمْ فَسُحْقًۭا لِّأَصْحَـٰبِ ٱلسَّعِيرِ ﴿11﴾
پس انہوں نے اپنے گناہ کا اعتراف کر لیا، سو دوری ہے بھڑکتی آگ والوں کے لیے۔
إِنَّ ٱلَّذِينَ يَخْشَوْنَ رَبَّهُم بِٱلْغَيْبِ لَهُم مَّغْفِرَةٌۭ وَأَجْرٌۭ كَبِيرٌ ﴿12﴾
بیشک جو لوگ اپنے رب سے غیب میں ڈرتے ہیں ان کے لیے بخشش اور بڑا اجر ہے۔
وَأَسِرُّوا۟ قَوْلَكُمْ أَوِ ٱجْهَرُوا۟ بِهِۦٓ ۖ إِنَّهُۥ عَلِيمٌۢ بِذَاتِ ٱلصُّدُورِ ﴿13﴾
تم اپنی بات چھپاؤ یا ظاہر کرو، وہ دلوں کے بھید خوب جانتا ہے۔
أَلَا يَعْلَمُ مَنْ خَلَقَ وَهُوَ ٱللَّطِيفُ ٱلْخَبِيرُ ﴿14﴾
کیا وہی نہ جانے جس نے پیدا کیا؟ اور وہی باریک بین، خوب خبردار ہے۔
هُوَ ٱلَّذِى جَعَلَ لَكُمُ ٱلْأَرْضَ ذَلُولًۭا فَٱمْشُوا۟ فِى مَنَاكِبِهَا وَكُلُوا۟ مِن رِّزْقِهِۦ ۖ وَإِلَيْهِ ٱلنُّشُورُ ﴿15﴾
وہی ہے جس نے تمہارے لیے زمین کو نرم بنایا، پس اس کے راستوں میں چلو اور اس کے رزق سے کھاؤ، اور اسی کی طرف تمہیں لوٹ کر جانا ہے۔
ءَأَمِنتُم مَّن فِى ٱلسَّمَآءِ أَن يَخْسِفَ بِكُمُ ٱلْأَرْضَ فَإِذَا هِىَ تَمُورُ ﴿16﴾
کیا تم اس سے بے خوف ہو گئے ہو جو آسمان میں ہے کہ وہ تمہیں زمین میں دھنسا دے، تو وہ اچانک ہلنے لگے؟
أَمْ أَمِنتُم مَّن فِى ٱلسَّمَآءِ أَن يُرْسِلَ عَلَيْكُمْ حَاصِبًۭا ۖ فَسَتَعْلَمُونَ كَيْفَ نَذِيرِ ﴿17﴾
یا تم اس سے بے خوف ہو گئے ہو جو آسمان میں ہے کہ وہ تم پر پتھروں کی بارش بھیج دے؟ پھر تم جان لو گے کہ میرا ڈرانا کیسا تھا۔
وَلَقَدْ كَذَّبَ ٱلَّذِينَ مِن قَبْلِهِمْ فَكَيْفَ كَانَ نَكِيرِ ﴿18﴾
اور یقیناً ان سے پہلے لوگوں نے بھی جھٹلایا، تو میرا عذاب کیسا تھا؟
أَوَلَمْ يَرَوْا۟ إِلَى ٱلطَّيْرِ فَوْقَهُمْ صَـٰٓفَّـٰتٍۢ وَيَقْبِضْنَ ۚ مَا يُمْسِكُهُنَّ إِلَّا ٱلرَّحْمَـٰنُ ۚ إِنَّهُۥ بِكُلِّ شَىْءٍۢ بَصِيرٌ ﴿19﴾
کیا انہوں نے اپنے اوپر پرندوں کو نہیں دیکھا جو پر پھیلائے ہوتے ہیں اور سمیٹتے ہیں؟ انہیں رحمان کے سوا کوئی تھامے نہیں رکھتا، بے شک وہ ہر چیز کو دیکھنے والا ہے۔
أَمَّنْ هَـٰذَا ٱلَّذِى هُوَ جُندٌۭ لَّكُمْ يَنصُرُكُم مِّن دُونِ ٱلرَّحْمَـٰنِ ۚ إِنِ ٱلْكَـٰفِرُونَ إِلَّا فِى غُرُورٍ ﴿20﴾
یا وہ کون ہے جو تمہارا لشکر ہو کر تمہاری مدد کرے رحمان کے سوا؟ کافر تو صرف دھوکے میں ہیں۔
أَمَّنْ هَـٰذَا ٱلَّذِى يَرْزُقُكُمْ إِنْ أَمْسَكَ رِزْقَهُۥ ۚ بَل لَّجُّوا۟ فِى عُتُوٍّۢ وَنُفُورٍ ﴿21﴾
یا وہ کون ہے جو تمہیں رزق دے اگر وہ اپنا رزق روک لے؟ بلکہ وہ سرکشی اور نفرت میں ہی پڑے ہوئے ہیں۔
أَفَمَن يَمْشِى مُكِبًّا عَلَىٰ وَجْهِهِۦٓ أَهْدَىٰٓ أَمَّن يَمْشِى سَوِيًّا عَلَىٰ صِرَٰطٍۢ مُّسْتَقِيمٍۢ ﴿22﴾
تو کیا وہ جو منہ کے بل چلتا ہے زیادہ ہدایت یافتہ ہے یا وہ جو سیدھا راستے پر چلتا ہے؟
قُلْ هُوَ ٱلَّذِىٓ أَنشَأَكُمْ وَجَعَلَ لَكُمُ ٱلسَّمْعَ وَٱلْأَبْصَـٰرَ وَٱلْأَفْـِٔدَةَ ۚ قَلِيلًۭا مَّا تَشْكُرُونَ ﴿23﴾
کہہ دو: وہی ہے جس نے تمہیں پیدا کیا اور تمہارے لیے کان، آنکھیں اور دل بنائے، تم بہت کم شکر کرتے ہو۔
قُلْ هُوَ ٱلَّذِى ذَرَأَكُمْ فِى ٱلْأَرْضِ وَإِلَيْهِ تُحْشَرُونَ ﴿24﴾
کہہ دو: وہی ہے جس نے تمہیں زمین میں بکھیر دیا، اور اسی کی طرف تم سب کو جمع کیا جائے گا۔
وَيَقُولُونَ مَتَىٰ هَـٰذَا ٱلْوَعْدُ إِن كُنتُمْ صَـٰدِقِينَ ﴿25﴾
اور وہ کہتے ہیں: یہ وعدہ کب ہوگا اگر تم سچے ہو؟
قُلْ إِنَّمَا ٱلْعِلْمُ عِندَ ٱللَّهِ وَإِنَّمَآ أَنَا۠ نَذِيرٌۭ مُّبِينٌۭ ﴿26﴾
کہہ دو: علم تو اللہ کے پاس ہے، اور میں تو بس ایک واضح ڈرانے والا ہوں۔
فَلَمَّا رَأَوْهُ زُلْفَةًۭ سِيٓـَٔتْ وُجُوهُ ٱلَّذِينَ كَفَرُوا۟ وَقِيلَ هَـٰذَا ٱلَّذِى كُنتُم بِهِۦ تَدَّعُونَ ﴿27﴾
پھر جب وہ (عذاب) قریب آ جائے گا تو کافروں کے چہرے بگڑ جائیں گے، اور کہا جائے گا: یہ وہی ہے جسے تم مانگتے تھے۔
قُلْ أَرَءَيْتُمْ إِنْ أَهْلَكَنِىَ ٱللَّهُ وَمَن مَّعِىَ أَوْ رَحِمَنَا فَمَن يُجِيرُ ٱلْكَـٰفِرِينَ مِنْ عَذَابٍ أَلِيمٍ ﴿28﴾
کہہ دو: بتاؤ اگر اللہ مجھے اور میرے ساتھیوں کو ہلاک کر دے یا ہم پر رحم کرے، تو کون ہے جو کافروں کو دردناک عذاب سے بچا سکے؟
قُلْ هُوَ ٱلرَّحْمَـٰنُ ءَامَنَّا بِهِۦ وَعَلَيْهِ تَوَكَّلْنَا ۖ فَسَتَعْلَمُونَ مَنْ هُوَ فِى ضَلَـٰلٍۢ مُّبِينٍ ﴿29﴾
کہہ دو: وہی رحمان ہے، ہم اس پر ایمان لائے اور اسی پر بھروسہ کیا، عنقریب تم جان لو گے کہ کون کھلی گمراہی میں ہے۔
قُلْ أَرَءَيْتُمْ إِنْ أَصْبَحَ مَآؤُكُمْ غَوْرًۭا فَمَن يَأْتِيكُم بِمَآءٍۢ مَّعِينٍۭ ﴿30﴾
کہہ دو: بتاؤ اگر تمہارا پانی زمین میں جذب ہو جائے تو کون ہے جو تمہیں بہتا ہوا پانی لا کر دے؟
سورة القلم
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
نٓ ۚ وَٱلْقَلَمِ وَمَا يَسْطُرُونَ ﴿1﴾
ن، قلم کی قسم اور جو کچھ وہ لکھتے ہیں اس کی قسم۔
مَآ أَنتَ بِنِعْمَةِ رَبِّكَ بِمَجْنُونٍۢ ﴿2﴾
تم اپنے رب کے فضل سے ہرگز مجنون نہیں ہو۔
وَإِنَّ لَكَ لَأَجْرًا غَيْرَ مَمْنُونٍۢ ﴿3﴾
اور یقیناً تمہارے لیے ایسا اجر ہے جو کبھی ختم نہ ہوگا۔
وَإِنَّكَ لَعَلَىٰ خُلُقٍ عَظِيمٍۢ ﴿4﴾
اور یقیناً تم عظیم اخلاق پر فائز ہو۔
فَسَتُبْصِرُ وَيُبْصِرُونَ ﴿5﴾
تو عنقریب تم دیکھو گے اور وہ بھی دیکھیں گے۔
بِأَيِّكُمُ ٱلْمَفْتُونُ ﴿6﴾
کہ تم میں سے کون دیوانہ ہے۔
إِنَّ رَبَّكَ هُوَ أَعْلَمُ بِمَن ضَلَّ عَن سَبِيلِهِۦ ۖ وَهُوَ أَعْلَمُ بِٱلْمُهْتَدِينَ ﴿7﴾
بیشک تمہارا رب خوب جانتا ہے کہ کون اس کی راہ سے بھٹکا ہوا ہے اور وہ ہدایت یافتہ لوگوں کو بھی خوب جانتا ہے۔
فَلَا تُطِعِ ٱلْمُكَذِّبِينَ ﴿8﴾
تو تم جھٹلانے والوں کی بات نہ مانو۔
وَدُّوا۟ لَوْ تُدْهِنُ فَيُدْهِنُونَ ﴿9﴾
وہ چاہتے ہیں کہ تم نرمی کرو تو وہ بھی نرم پڑ جائیں۔
وَلَا تُطِعْ كُلَّ حَلَّافٍۢ مَّهِينٍ ﴿10﴾
اور ہر ایسے شخص کی بات نہ مانو جو بہت زیادہ قسمیں کھانے والا، ذلیل ہے۔
هَمَّازٍۢ مَّشَّآءٍۢ بِنَمِيمٍۢ ﴿11﴾
عیب جو، چغل خور۔
مَّنَّاعٍۢ لِّلْخَيْرِ مُعْتَدٍ أَثِيمٍ ﴿12﴾
بھلائی سے روکنے والا، حد سے بڑھنے والا، گناہگار۔
عُتُلِّۭ بَعْدَ ذَٰلِكَ زَنِيمٍ ﴿13﴾
بدخلق، اس کے بعد نسب میں مشکوک۔
أَن كَانَ ذَا مَالٍۢ وَبَنِينَ ﴿14﴾
اس لیے کہ وہ مال اور بیٹوں والا ہے۔
إِذَا تُتْلَىٰ عَلَيْهِ ءَايَـٰتُنَا قَالَ أَسَـٰطِيرُ ٱلْأَوَّلِينَ ﴿15﴾
جب اس پر ہماری آیات پڑھی جاتی ہیں تو کہتا ہے: یہ پہلے لوگوں کی کہانیاں ہیں۔
سَنَسِمُهُۥ عَلَى ٱلْخُرْطُومِ ﴿16﴾
ہم عنقریب اس کی ناک پر نشان لگا دیں گے۔
إِنَّا بَلَوْنَـٰهُمْ كَمَا بَلَوْنَآ أَصْحَـٰبَ ٱلْجَنَّةِ إِذْ أَقْسَمُوا۟ لَيَصْرِمُنَّهَا مُصْبِحِينَ ﴿17﴾
ہم نے ان کو اسی طرح آزمایا جیسے ہم نے باغ والوں کو آزمایا، جب انہوں نے قسم کھائی کہ صبح ہوتے ہی اسے ضرور کاٹیں گے۔
وَلَا يَسْتَثْنُونَ ﴿18﴾
اور انہوں نے کوئی استثناء نہ کیا۔
فَطَافَ عَلَيْهَا طَآئِفٌۭ مِّن رَّبِّكَ وَهُمْ نَآئِمُونَ ﴿19﴾
تو تیرے رب کی طرف سے ایک آفت اس پر رات کو آ گئی جب وہ سو رہے تھے۔
فَأَصْبَحَتْ كَٱلصَّرِيمِ ﴿20﴾
تو وہ ایسی ہو گئی جیسے کٹی ہوئی ہو۔
فَتَنَادَوْا۟ مُصْبِحِينَ ﴿21﴾
پھر وہ صبح ہوتے ہی ایک دوسرے کو پکارنے لگے۔
أَنِ ٱغْدُوا۟ عَلَىٰ حَرْثِكُمْ إِن كُنتُمْ صَـٰرِمِينَ ﴿22﴾
کہ اپنے کھیت کی طرف صبح سویرے نکلو اگر تم کاٹنے والے ہو۔
فَٱنطَلَقُوا۟ وَهُمْ يَتَخَـٰفَتُونَ ﴿23﴾
پس وہ چپکے چپکے چل پڑے۔
أَن لَّا يَدْخُلَنَّهَا ٱلْيَوْمَ عَلَيْكُم مِّسْكِينٌۭ ﴿24﴾
کہ آج تمہارے باغ میں کوئی مسکین داخل نہ ہو۔
وَغَدَوْا۟ عَلَىٰ حَرْدٍۢ قَـٰدِرِينَ ﴿25﴾
اور وہ غضبناک ہو کر نکلے، گویا وہ قدرت رکھتے ہیں۔
فَلَمَّا رَأَوْهَا قَالُوٓا۟ إِنَّا لَضَآلُّونَ ﴿26﴾
پھر جب انہوں نے اسے دیکھا تو کہنے لگے: ہم تو راستہ بھول گئے ہیں۔
بَلْ نَحْنُ مَحْرُومُونَ ﴿27﴾
بلکہ ہم محروم کر دیے گئے ہیں۔
قَالَ أَوْسَطُهُمْ أَلَمْ أَقُل لَّكُمْ لَوْلَا تُسَبِّحُونَ ﴿28﴾
ان میں جو درمیانہ فہم والا تھا، اس نے کہا: کیا میں نے تم سے نہیں کہا تھا کہ تسبیح کیوں نہیں کرتے؟
قَالُوا۟ سُبْحَـٰنَ رَبِّنَآ إِنَّا كُنَّا ظَـٰلِمِينَ ﴿29﴾
انہوں نے کہا: پاک ہے ہمارا رب، بے شک ہم ہی ظالم تھے۔
فَأَقْبَلَ بَعْضُهُمْ عَلَىٰ بَعْضٍۢ يَتَلَـٰوَمُونَ ﴿30﴾
پھر وہ ایک دوسرے کی طرف ملامت کرتے ہوئے متوجہ ہوئے۔
قَالُوا۟ يَـٰوَيْلَنَآ إِنَّا كُنَّا طَـٰغِينَ ﴿31﴾
انہوں نے کہا: ہائے ہماری بدبختی! ہم حد سے بڑھ گئے تھے۔
عَسَىٰ رَبُّنَآ أَن يُبْدِلَنَا خَيْرًۭا مِّنْهَآ إِنَّآ إِلَىٰ رَبِّنَا رَٰغِبُونَ ﴿32﴾
امید ہے کہ ہمارا رب ہمیں اس سے بہتر عطا کرے گا، ہم اپنے رب کی طرف رغبت رکھتے ہیں۔
كَذَٰلِكَ ٱلْعَذَابُ ۖ وَلَعَذَابُ ٱلْـَٔاخِرَةِ أَكْبَرُ ۚ لَوْ كَانُوا۟ يَعْلَمُونَ ﴿33﴾
اسی طرح دنیا کا عذاب ہے، اور آخرت کا عذاب تو اس سے بھی بڑا ہے، کاش وہ جانتے۔
إِنَّ لِلْمُتَّقِينَ عِندَ رَبِّهِمْ جَنَّـٰتِ ٱلنَّعِيمِ ﴿34﴾
یقیناً پرہیزگاروں کے لیے ان کے رب کے پاس نعمتوں والی جنتیں ہیں۔
أَفَنَجْعَلُ ٱلْمُسْلِمِينَ كَٱلْمُجْرِمِينَ ﴿35﴾
تو کیا ہم فرمانبرداروں کو مجرموں جیسا بنا دیں؟
مَا لَكُمْ كَيْفَ تَحْكُمُونَ ﴿36﴾
تمہیں کیا ہو گیا ہے، کیسے فیصلے کرتے ہو؟
أَمْ لَكُمْ كِتَـٰبٌۭ فِيهِ تَدْرُسُونَ ﴿37﴾
یا تمہارے پاس کوئی کتاب ہے جس میں تم پڑھتے ہو؟
إِنَّ لَكُمْ فِيهِ لَمَا تَخَيَّرُونَ ﴿38﴾
کہ تمہارے لیے اس میں وہی کچھ ہے جو تم پسند کرتے ہو؟
أَمْ لَكُمْ أَيْمَـٰنٌ عَلَيْنَا بَـٰلِغَةٌ إِلَىٰ يَوْمِ ٱلْقِيَـٰمَةِ ۙ إِنَّ لَكُمْ لَمَا تَحْكُمُونَ ﴿39﴾
یا تمہارے پاس ہم پر کوئی پختہ قسمیں ہیں قیامت کے دن تک، کہ تمہارے لیے وہی کچھ ہے جو تم فیصلہ کرو؟
سَلْهُمْ أَيُّهُم بِذَٰلِكَ زَعِيمٌ ﴿40﴾
ان سے پوچھو، ان میں کون اس کا ذمہ دار ہے؟
أَمْ لَهُمْ شُرَكَآءُ فَلْيَأْتُوا۟ بِشُرَكَآئِهِمْ إِن كَانُوا۟ صَـٰدِقِينَ ﴿41﴾
یا ان کے کچھ شریک ہیں؟ تو اگر وہ سچے ہیں تو اپنے شریکوں کو لے آئیں۔
يَوْمَ يُكْشَفُ عَن سَاقٍۢ وَيُدْعَوْنَ إِلَى ٱلسُّجُودِ فَلَا يَسْتَطِيعُونَ ﴿42﴾
جس دن پنڈلی کھولی جائے گی اور انہیں سجدہ کرنے کو بلایا جائے گا، تو وہ سجدہ نہ کر سکیں گے۔
خَـٰشِعَةً أَبْصَـٰرُهُمْ تَرْهَقُهُمْ ذِلَّةٌۭ ۖ وَقَدْ كَانُوا۟ يُدْعَوْنَ إِلَى ٱلسُّجُودِ وَهُمْ سَـٰلِمُونَ ﴿43﴾
ان کی آنکھیں جھکی ہوں گی، ان پر ذلت چھا رہی ہوگی، حالانکہ وہ سلامت تھے اور انہیں سجدہ کرنے کو بلایا جاتا تھا۔
فَذَرْنِى وَمَن يُكَذِّبُ بِهَـٰذَا ٱلْحَدِيثِ ۖ سَنَسْتَدْرِجُهُم مِّنْ حَيْثُ لَا يَعْلَمُونَ ﴿44﴾
تو مجھے اور اس کو چھوڑ دو جو اس بات کو جھٹلاتا ہے، ہم انہیں آہستہ آہستہ پکڑیں گے جہاں سے انہیں خبر بھی نہ ہوگی۔
وَأُمْلِى لَهُمْ ۚ إِنَّ كَيْدِى مَتِينٌ ﴿45﴾
اور میں انہیں مہلت دیتا ہوں، بے شک میری چال مضبوط ہے۔
أَمْ تَسْـَٔلُهُمْ أَجْرًۭا فَهُم مِّن مَّغْرَمٍۢ مُّثْقَلُونَ ﴿46﴾
کیا تم ان سے کوئی معاوضہ مانگتے ہو کہ وہ تاوان کے بوجھ سے دبے جا رہے ہیں؟
أَمْ عِندَهُمُ ٱلْغَيْبُ فَهُمْ يَكْتُبُونَ ﴿47﴾
یا ان کے پاس غیب ہے کہ وہ اسے لکھ رہے ہیں؟
فَٱصْبِرْ لِحُكْمِ رَبِّكَ وَلَا تَكُن كَصَاحِبِ ٱلْحُوتِ إِذْ نَادَىٰ وَهُوَ مَكْظُومٌ ﴿48﴾
تو اپنے رب کے حکم کے لیے صبر کرو، اور مچھلی والے کی طرح نہ ہو جاؤ جب اس نے پکارا اور وہ غم سے بھرا ہوا تھا۔
لَّوْلَآ أَن تَدَٰرَكَهُۥ نِعْمَةٌۭ مِّن رَّبِّهِۦ لَنُبِذَ بِٱلْعَرَآءِ وَهُوَ مَذْمُومٌۭ ﴿49﴾
اگر اسے اس کے رب کی نعمت نہ پہنچتی تو وہ چٹیل میدان میں پھینک دیا جاتا اور مذمت کیا جاتا۔
فَٱجْتَبَـٰهُ رَبُّهُۥ فَجَعَلَهُۥ مِنَ ٱلصَّـٰلِحِينَ ﴿50﴾
پھر اس کے رب نے اسے چن لیا اور اسے صالحین میں شامل کر دیا۔
وَإِن يَكَادُ ٱلَّذِينَ كَفَرُوا۟ لَيُزْلِقُونَكَ بِأَبْصَـٰرِهِمْ لَمَّا سَمِعُوا۟ ٱلذِّكْرَ وَيَقُولُونَ إِنَّهُۥ لَمَجْنُونٌۭ ﴿51﴾
اور جب کافر ذکر سنتے ہیں تو وہ تمہیں اپنی نگاہوں سے گرا دینے کو ہوتے ہیں اور کہتے ہیں: یہ تو دیوانہ ہے۔
وَمَا هُوَ إِلَّا ذِكْرٌۭ لِّلْعَـٰلَمِينَ ﴿52﴾
حالانکہ وہ تو تمام جہانوں کے لیے نصیحت ہے۔
سورة الحاقة
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
ٱلْحَاقَّةُ ﴿1﴾
وہ واقع ہونے والی حقیقت۔
مَا ٱلْحَاقَّةُ ﴿2﴾
کیا ہے وہ واقع ہونے والی حقیقت؟
وَمَآ أَدْرَىٰكَ مَا ٱلْحَاقَّةُ ﴿3﴾
اور تمہیں کیا معلوم کہ وہ واقع ہونے والی حقیقت کیا ہے؟
كَذَّبَتْ ثَمُودُ وَعَادٌۢ بِٱلْقَارِعَةِ ﴿4﴾
ثمود اور عاد نے زبردست جھٹکے والی آفت کو جھٹلایا۔
فَأَمَّا ثَمُودُ فَأُهْلِكُوا۟ بِٱلطَّاغِيَةِ ﴿5﴾
پس ثمود حد سے بڑھی ہوئی آواز سے ہلاک کیے گئے۔
وَأَمَّا عَادٌۭ فَأُهْلِكُوا۟ بِرِيحٍۢ صَرْصَرٍ عَاتِيَةٍ ﴿6﴾
اور عاد کو نہایت تیز و تند آندھی سے ہلاک کیا گیا۔
سَخَّرَهَا عَلَيْهِمْ سَبْعَ لَيَالٍۢ وَثَمَـٰنِيَةَ أَيَّامٍ حُسُومًۭا فَتَرَى ٱلْقَوْمَ فِيهَا صَرْعَىٰ كَأَنَّهُمْ أَعْجَازُ نَخْلٍ خَاوِيَةٍۢ ﴿7﴾
اللہ نے اس آندھی کو ان پر مسلسل سات راتیں اور آٹھ دن مسلط رکھا، تو تم ان لوگوں کو اس میں گرے ہوئے دیکھتے جیسے وہ کھجور کے کھوکھلے تنے ہوں۔
فَهَلْ تَرَىٰ لَهُم مِّنۢ بَاقِيَةٍۢ ﴿8﴾
تو کیا تم ان میں سے کسی کو باقی دیکھتے ہو؟
وَجَآءَ فِرْعَوْنُ وَمَن قَبْلَهُۥ وَٱلْمُؤْتَفِكَـٰتُ بِٱلْخَاطِئَةِ ﴿9﴾
اور فرعون اور اس سے پہلے کے لوگ اور الٹی بستیوں والے سب خطا میں مبتلا ہو گئے۔
فَعَصَوْا۟ رَسُولَ رَبِّهِمْ فَأَخَذَهُمْ أَخْذَةًۭ رَّابِيَةً ﴿10﴾
تو انہوں نے اپنے رب کے رسول کی نافرمانی کی، پس اللہ نے انہیں سخت گرفت میں لیا۔
إِنَّا لَمَّا طَغَا ٱلْمَآءُ حَمَلْنَـٰكُمْ فِى ٱلْجَارِيَةِ ﴿11﴾
جب پانی حد سے بڑھ گیا تو ہم نے تمہیں کشتی میں سوار کیا۔
لِنَجْعَلَهَا لَكُمْ تَذْكِرَةًۭ وَتَعِيَهَآ أُذُنٌۭ وَٰعِيَةٌۭ ﴿12﴾
تاکہ ہم اسے تمہارے لیے نصیحت بنا دیں اور یاد رکھنے والے کان اسے یاد رکھیں۔
فَإِذَا نُفِخَ فِى ٱلصُّورِ نَفْخَةٌۭ وَٰحِدَةٌۭ ﴿13﴾
پھر جب صور میں ایک بار پھونکا جائے گا۔
وَحُمِلَتِ ٱلْأَرْضُ وَٱلْجِبَالُ فَدُكَّتَا دَكَّةًۭ وَٰحِدَةًۭ ﴿14﴾
اور زمین اور پہاڑ اٹھا لیے جائیں گے، پھر ایک ہی بار چورا چورا کر دیے جائیں گے۔
فَيَوْمَئِذٍۢ وَقَعَتِ ٱلْوَاقِعَةُ ﴿15﴾
پس اس دن وہ واقعہ واقع ہو جائے گا۔
وَٱنشَقَّتِ ٱلسَّمَآءُ فَهِىَ يَوْمَئِذٍۢ وَاهِيَةٌۭ ﴿16﴾
اور آسمان پھٹ جائے گا، پس وہ اس دن کمزور ہو جائے گا۔
وَٱلْمَلَكُ عَلَىٰٓ أَرْجَآئِهَا ۖ وَيَحْمِلُ عَرْشَ رَبِّكَ فَوْقَهُمْ يَوْمَئِذٍۢ ثَمَـٰنِيَةٌۭ ﴿17﴾
اور فرشتے اس کے کناروں پر ہوں گے، اور اس دن تمہارے رب کا عرش آٹھ فرشتے اٹھائے ہوں گے۔
يَوْمَئِذٍۢ تُعْرَضُونَ لَا تَخْفَىٰ مِنكُمْ خَافِيَةٌۭ ﴿18﴾
اس دن تم پیش کیے جاؤ گے، تمہاری کوئی بات چھپی نہ رہے گی۔
فَأَمَّا مَنْ أُوتِىَ كِتَـٰبَهُۥ بِيَمِينِهِۦ فَيَقُولُ هَآؤُمُ ٱقْرَءُوا۟ كِتَـٰبِيَهْ ﴿19﴾
پس جس کو اس کا نامہ اعمال دائیں ہاتھ میں دیا جائے گا، وہ کہے گا: لو پڑھو میرا نامہ اعمال۔
إِنِّى ظَنَنتُ أَنِّى مُلَـٰقٍ حِسَابِيَهْ ﴿20﴾
بیشک مجھے یقین تھا کہ مجھے اپنا حساب ملنے والا ہے۔
فَهُوَ فِى عِيشَةٍۢ رَّاضِيَةٍۢ ﴿21﴾
پس وہ خوشحال زندگی میں ہوگا۔
فِى جَنَّةٍ عَالِيَةٍۢ ﴿22﴾
بلند مقام والی جنت میں۔
قُطُوفُهَا دَانِيَةٌۭ ﴿23﴾
جس کے پھل جھکے ہوئے ہوں گے۔
كُلُوا۟ وَٱشْرَبُوا۟ هَنِيٓـًۭٔا بِمَآ أَسْلَفْتُمْ فِى ٱلْأَيَّامِ ٱلْخَالِيَةِ ﴿24﴾
کھاؤ اور پیو مزے سے، ان اعمال کے بدلے جو تم نے گزشتہ دنوں میں کیے۔
وَأَمَّا مَنْ أُوتِىَ كِتَـٰبَهُۥ بِشِمَالِهِۦ فَيَقُولُ يَـٰلَيْتَنِى لَمْ أُوتَ كِتَـٰبِيَهْ ﴿25﴾
اور جس کو اس کا نامہ اعمال بائیں ہاتھ میں دیا جائے گا، وہ کہے گا: کاش مجھے میرا نامہ اعمال نہ دیا جاتا۔
وَلَمْ أَدْرِ مَا حِسَابِيَهْ ﴿26﴾
اور میں نہ جانتا کہ میرا حساب کیا ہے۔
يَـٰلَيْتَهَا كَانَتِ ٱلْقَاضِيَةَ ﴿27﴾
کاش وہی (موت) فیصلہ کن ہوتی۔
مَآ أَغْنَىٰ عَنِّى مَالِيَهْ ﴿28﴾
میرا مال میرے کچھ کام نہ آیا۔
هَلَكَ عَنِّى سُلْطَـٰنِيَهْ ﴿29﴾
میری سلطنت بھی میرے ہاتھ سے جاتی رہی۔
خُذُوهُ فَغُلُّوهُ ﴿30﴾
پکڑو اسے اور اس کو زنجیروں میں جکڑ دو۔
ثُمَّ ٱلْجَحِيمَ صَلُّوهُ ﴿31﴾
پھر اسے جہنم میں جھونک دو۔
ثُمَّ فِى سِلْسِلَةٍۢ ذَرْعُهَا سَبْعُونَ ذِرَاعًۭا فَٱسْلُكُوهُ ﴿32﴾
پھر ایک زنجیر میں جکڑ دو جس کی لمبائی ستر ہاتھ ہے۔
إِنَّهُۥ كَانَ لَا يُؤْمِنُ بِٱللَّهِ ٱلْعَظِيمِ ﴿33﴾
بیشک وہ اللہ عظیم پر ایمان نہیں رکھتا تھا۔
وَلَا يَحُضُّ عَلَىٰ طَعَامِ ٱلْمِسْكِينِ ﴿34﴾
اور وہ مسکین کو کھانا کھلانے پر ترغیب نہیں دیتا تھا۔
فَلَيْسَ لَهُ ٱلْيَوْمَ هَـٰهُنَا حَمِيمٌۭ ﴿35﴾
پس آج یہاں اس کا کوئی دوست نہیں۔
وَلَا طَعَامٌ إِلَّا مِنْ غِسْلِينٍۢ ﴿36﴾
اور نہ کوئی کھانا، سوائے زخموں کے دھوون کے۔
لَّا يَأْكُلُهُۥٓ إِلَّا ٱلْخَـٰطِـُٔونَ ﴿37﴾
اسے صرف گناہگار ہی کھائیں گے۔
فَلَآ أُقْسِمُ بِمَا تُبْصِرُونَ ﴿38﴾
پس نہیں، میں قسم کھاتا ہوں ان چیزوں کی جنہیں تم دیکھتے ہو۔
وَمَا لَا تُبْصِرُونَ ﴿39﴾
اور ان کی بھی جنہیں تم نہیں دیکھتے۔
إِنَّهُۥ لَقَوْلُ رَسُولٍۢ كَرِيمٍۢ ﴿40﴾
بیشک یہ ایک معزز رسول کا قول ہے۔
وَمَا هُوَ بِقَوْلِ شَاعِرٍۢ ۚ قَلِيلًۭا مَّا تُؤْمِنُونَ ﴿41﴾
اور یہ کسی شاعر کا قول نہیں، تم بہت کم ایمان لاتے ہو۔
وَلَا بِقَوْلِ كَاهِنٍۢ ۚ قَلِيلًۭا مَّا تَذَكَّرُونَ ﴿42﴾
اور نہ ہی کسی کاہن کا قول ہے، تم بہت کم نصیحت حاصل کرتے ہو۔
تَنزِيلٌۭ مِّن رَّبِّ ٱلْعَـٰلَمِينَ ﴿43﴾
یہ رب العالمین کی طرف سے نازل کردہ ہے۔
وَلَوْ تَقَوَّلَ عَلَيْنَا بَعْضَ ٱلْأَقَاوِيلِ ﴿44﴾
اور اگر وہ ہماری طرف کچھ باتیں گھڑ لیتا۔
لَأَخَذْنَا مِنْهُ بِٱلْيَمِينِ ﴿45﴾
تو ہم اس کو دائیں ہاتھ سے پکڑ لیتے۔
ثُمَّ لَقَطَعْنَا مِنْهُ ٱلْوَتِينَ ﴿46﴾
پھر ہم اس کی گردن کی رگ کاٹ دیتے۔
فَمَا مِنكُم مِّنْ أَحَدٍ عَنْهُ حَـٰجِزِينَ ﴿47﴾
پھر تم میں سے کوئی اسے بچانے والا نہ ہوتا۔
وَإِنَّهُۥ لَتَذْكِرَةٌۭ لِّلْمُتَّقِينَ ﴿48﴾
اور بیشک یہ پرہیزگاروں کے لیے نصیحت ہے۔
وَإِنَّا لَنَعْلَمُ أَنَّ مِنكُم مُّكَذِّبِينَ ﴿49﴾
اور ہم خوب جانتے ہیں کہ تم میں سے بعض جھٹلانے والے ہیں۔
وَإِنَّهُۥ لَحَسْرَةٌ عَلَى ٱلْكَـٰفِرِينَ ﴿50﴾
اور بیشک یہ کافروں کے لیے حسرت کا باعث ہے۔
وَإِنَّهُۥ لَحَقُّ ٱلْيَقِينِ ﴿51﴾
اور بیشک یہ یقینی حق ہے۔
فَسَبِّحْ بِٱسْمِ رَبِّكَ ٱلْعَظِيمِ ﴿52﴾
پس اپنے عظیم رب کے نام کی تسبیح کرو۔
سورة المعارج
بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
سَأَلَ سَآئِلٌۢ بِعَذَابٍۢ وَاقِعٍۢ ﴿1﴾
ایک مانگنے والے نے واقع ہونے والے عذاب کے بارے میں سوال کیا۔
لِّلْكَـٰفِرِينَ لَيْسَ لَهُۥ دَافِعٌۭ ﴿2﴾
کافروں کے لیے ہے، اس کو کوئی ہٹانے والا نہیں۔
مِّنَ ٱللَّهِ ذِى ٱلْمَعَارِجِ ﴿3﴾
وہ اللہ کی طرف سے ہے جو بلند درجوں والا ہے۔
تَعْرُجُ ٱلْمَلَـٰٓئِكَةُ وَٱلرُّوحُ إِلَيْهِ فِى يَوْمٍۢ كَانَ مِقْدَارُهُۥ خَمْسِينَ أَلْفَ سَنَةٍۢ ﴿4﴾
فرشتے اور روح اس کی طرف چڑھتے ہیں ایک دن میں جس کی مقدار پچاس ہزار سال ہے۔
فَٱصْبِرْ صَبْرًۭا جَمِيلًا ﴿5﴾
تو خوبصورت صبر کرو۔
إِنَّهُمْ يَرَوْنَهُۥ بَعِيدًۭا ﴿6﴾
وہ اس کو دور سمجھتے ہیں۔
وَنَرَىٰهُ قَرِيبًۭا ﴿7﴾
اور ہم اسے قریب دیکھتے ہیں۔
يَوْمَ تَكُونُ ٱلسَّمَآءُ كَٱلْمُهْلِ ﴿8﴾
جس دن آسمان پگھلے ہوئے تانبے کی طرح ہو جائے گا۔
وَتَكُونُ ٱلْجِبَالُ كَٱلْعِهْنِ ﴿9﴾
اور پہاڑ رنگین اون کی طرح ہو جائیں گے۔
وَلَا يَسْـَٔلُ حَمِيمٌ حَمِيمًۭا ﴿10﴾
اور کوئی دوست کسی دوست کو نہیں پوچھے گا۔
يُبَصَّرُونَهُمْ ۚ يَوَدُّ ٱلْمُجْرِمُ لَوْ يَفْتَدِى مِنْ عَذَابِ يَوْمِئِذٍۢ بِبَنِيهِ ﴿11﴾
وہ ایک دوسرے کو دکھا دیے جائیں گے، مجرم چاہے گا کہ اس دن کے عذاب سے اپنے بیٹوں کے بدلے چھوٹ جائے۔
وَصَـٰحِبَتِهِۦ وَأَخِيهِ ﴿12﴾
اور اپنی بیوی اور اپنے بھائی کے۔
وَفَصِيلَتِهِ ٱلَّتِى تُـْٔوِيهِ ﴿13﴾
اور اپنے قبیلے کے جو اسے پناہ دیتے ہیں۔
وَمَن فِى ٱلْأَرْضِ جَمِيعًۭا ثُمَّ يُنجِيهِ ﴿14﴾
اور زمین کے سب لوگوں کو دے کر وہ نجات پا لے۔
كَلَّآ ۖ إِنَّهَا لَظَىٰ ﴿15﴾
ہرگز نہیں! وہ تو شعلہ زن آگ ہے۔
نَزَّاعَةًۭ لِّلشَّوَىٰ ﴿16﴾
جو کھال کو کھینچ لے گی۔
تَدْعُوا۟ مَنْ أَدْبَرَ وَتَوَلَّىٰ ﴿17﴾
وہ پکارے گی اسے جو پیٹھ پھیر گیا اور منہ موڑ لیا۔
وَجَمَعَ فَأَوْعَىٰ ﴿18﴾
اور مال جمع کیا اور اسے چھپا کر رکھا۔
إِنَّ ٱلْإِنسَـٰنَ خُلِقَ هَلُوعًا ﴿19﴾
بیشک انسان گھبراہٹ میں پیدا کیا گیا ہے۔
إِذَا مَسَّهُ ٱلشَّرُّ جَزُوعًا ﴿20﴾
جب اسے کوئی برائی پہنچتی ہے تو بے چین ہو جاتا ہے۔
وَإِذَا مَسَّهُ ٱلْخَيْرُ مَنُوعًا ﴿21﴾
اور جب اسے کوئی بھلائی پہنچتی ہے تو روک لیتا ہے۔
إِلَّا ٱلْمُصَلِّينَ ﴿22﴾
سوائے ان نمازیوں کے۔
ٱلَّذِينَ هُمْ عَلَىٰ صَلَاتِهِمْ دَآئِمُونَ ﴿23﴾
جو اپنی نماز پر ہمیشہ قائم رہتے ہیں۔
وَٱلَّذِينَ فِىٓ أَمْوَٰلِهِمْ حَقٌّۭ مَّعْلُومٌۭ ﴿24﴾
اور وہ جن کے مالوں میں ایک مقرر حصہ ہے۔
لِّلسَّآئِلِ وَٱلْمَحْرُومِ ﴿25﴾
مانگنے والے اور محروم کے لیے۔
وَٱلَّذِينَ يُصَدِّقُونَ بِيَوْمِ ٱلدِّينِ ﴿26﴾
اور وہ جو روزِ جزا کو سچ مانتے ہیں۔
وَٱلَّذِينَ هُم مِّنْ عَذَابِ رَبِّهِم مُّشْفِقُونَ ﴿27﴾
اور وہ جو اپنے رب کے عذاب سے ڈرتے رہتے ہیں۔
إِنَّ عَذَابَ رَبِّهِمْ غَيْرُ مَأْمُونٍۭ ﴿28﴾
بیشک ان کے رب کا عذاب بے خوفی کی چیز نہیں۔
وَٱلَّذِينَ هُمْ لِفُرُوجِهِمْ حَـٰفِظُونَ ﴿29﴾
اور وہ جو اپنی شرمگاہوں کی حفاظت کرتے ہیں۔
إِلَّا عَلَىٰٓ أَزْوَٰجِهِمْ أَوْ مَا مَلَكَتْ أَيْمَـٰنُهُمْ فَإِنَّهُمْ غَيْرُ مَلُومِينَ ﴿30﴾
مگر اپنی بیویوں یا مملوک عورتوں پر تو وہ ملامت کے لائق نہیں۔
فَمَنِ ٱبْتَغَىٰ وَرَآءَ ذَٰلِكَ فَأُو۟لَـٰٓئِكَ هُمُ ٱلْعَادُونَ ﴿31﴾
پس جو اس کے علاوہ چاہے تو وہی حد سے بڑھنے والے ہیں۔
وَٱلَّذِينَ هُمْ لِأَمَـٰنَـٰتِهِمْ وَعَهْدِهِمْ رَٰعُونَ ﴿32﴾
اور وہ جو اپنی امانتوں اور اپنے عہد کا لحاظ رکھتے ہیں۔
وَٱلَّذِينَ هُم بِشَهَـٰدَٰتِهِمْ قَآئِمُونَ ﴿33﴾
اور وہ جو اپنی گواہی پر قائم رہتے ہیں۔
وَٱلَّذِينَ هُمْ عَلَىٰ صَلَاتِهِمْ يُحَافِظُونَ ﴿34﴾
اور وہ جو اپنی نماز کی حفاظت کرتے ہیں۔
أُو۟لَـٰٓئِكَ فِى جَنَّـٰتٍۢ مُّكْرَمُونَ ﴿35﴾
یہی لوگ باغوں میں عزت والے ہوں گے۔
فَمَالِ ٱلَّذِينَ كَفَرُوا۟ قِبَلَكَ مُهْطِعِينَ ﴿36﴾
تو ان کافروں کو کیا ہو گیا ہے کہ تیرے سامنے دوڑتے آ رہے ہیں۔
عَنِ ٱلْيَمِينِ وَعَنِ ٱلشِّمَالِ عِزِينَ ﴿37﴾
دائیں اور بائیں گروہ در گروہ۔
أَيَطْمَعُ كُلُّ ٱمْرِئٍۭ مِّنْهُمْ أَن يُدْخَلَ جَنَّةَ نَعِيمٍۢ ﴿38﴾
کیا ان میں سے ہر کوئی یہ چاہتا ہے کہ نعمتوں کی جنت میں داخل کیا جائے؟
كَلَّآ ۖ إِنَّا خَلَقْنَـٰهُم مِّمَّا يَعْلَمُونَ ﴿39﴾
ہرگز نہیں! ہم نے انہیں اسی چیز سے پیدا کیا ہے جسے وہ جانتے ہیں۔
فَلَآ أُقْسِمُ بِرَبِّ ٱلْمَشَـٰرِقِ وَٱلْمَغَـٰرِبِ إِنَّا لَقَـٰدِرُونَ ﴿40﴾
سو مشرقوں اور مغربوں کے رب کی قسم! ہم ضرور قادر ہیں۔
عَلَىٰٓ أَن نُّبَدِّلَ خَيْرًۭا مِّنْهُمْ ۖ وَمَا نَحْنُ بِمَسْبُوقِينَ ﴿41﴾
کہ ہم ان کے بدلے ان سے بہتر لے آئیں، اور ہم عاجز نہیں۔
فَذَرْهُمْ يَخُوضُوا۟ وَيَلْعَبُوا۟ حَتَّىٰ يُلَـٰقُوا۟ يَوْمَهُمُ ٱلَّذِى يُوعَدُونَ ﴿42﴾
پس انہیں چھوڑ دے کہ وہ باتوں میں پڑے رہیں اور کھیلتے رہیں یہاں تک کہ اپنے اس دن سے جا ملیں جس کا انہیں وعدہ دیا جا رہا ہے۔
يَوْمَ يَخْرُجُونَ مِنَ ٱلْأَجْدَاثِ سِرَاعًۭا كَأَنَّهُمْ إِلَىٰ نُصُبٍۭ يُوفِضُونَ ﴿43﴾
جس دن وہ قبروں سے تیزی سے نکلیں گے گویا وہ کسی نشانے کی طرف دوڑ رہے ہیں۔
خَـٰشِعَةً أَبْصَـٰرُهُمْ تَرْهَقُهُمْ ذِلَّةٌۭ ۚ ذَٰلِكَ ٱلْيَوْمُ ٱلَّذِى كَانُوا۟ يُوعَدُونَ ﴿44﴾
ان کی آنکھیں جھکی ہوئی ہوں گی، ان پر ذلت چھا رہی ہوگی۔ یہی وہ دن ہے جس کا ان سے وعدہ کیا جاتا تھا۔
سورة نوح
بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
إِنَّآ أَرْسَلْنَا نُوحًا إِلَىٰ قَوْمِهِۦٓ أَنْ أَنْذِرْ قَوْمَكَ مِن قَبْلِ أَن يَأْتِيَهُمْ عَذَابٌ أَلِيمٌ ﴿1﴾
ہم نے نوح کو اس کی قوم کی طرف بھیجا کہ اپنی قوم کو ڈراؤ اس سے پہلے کہ ان پر دردناک عذاب آ جائے۔
قَالَ يَـٰقَوْمِ إِنِّى لَكُمْ نَذِيرٌۭ مُّبِينٌ ﴿2﴾
اس نے کہا: اے میری قوم! میں تمہارے لیے واضح ڈرانے والا ہوں۔
أَنِ ٱعْبُدُوا۟ ٱللَّهَ وَٱتَّقُوهُ وَأَطِيعُونِ ﴿3﴾
کہ اللہ کی عبادت کرو، اس سے ڈرو اور میری اطاعت کرو۔
يَغْفِرْ لَكُم مِّن ذُنُوبِكُمْ وَيُؤَخِّرْكُمْ إِلَىٰٓ أَجَلٍۢ مُّسَمًّى ۚ إِنَّ أَجَلَ ٱللَّهِ إِذَا جَآءَ لَا يُؤَخَّرُ ۖ لَوْ كُنتُمْ تَعْلَمُونَ ﴿4﴾
وہ تمہارے گناہ معاف کر دے گا اور تمہیں مقررہ وقت تک مہلت دے گا۔ بے شک اللہ کا وقت جب آتا ہے تو مؤخر نہیں کیا جاتا، کاش تم جانتے۔
قَالَ رَبِّ إِنِّى دَعَوْتُ قَوْمِى لَيْلًۭا وَنَهَارًۭا ﴿5﴾
نوح نے کہا: اے میرے رب! میں نے اپنی قوم کو رات دن بلایا۔
فَلَمْ يَزِدْهُمْ دُعَآءِىٓ إِلَّا فِرَارًۭا ﴿6﴾
تو میری دعوت نے ان کو بھاگنے کے سوا کچھ نہ بڑھایا۔
وَإِنِّى كُلَّمَا دَعَوْتُهُمْ لِتَغْفِرَ لَهُمْ جَعَلُوٓا۟ أَصَـٰبِعَهُمْ فِىٓ ءَاذَانِهِمْ وَٱسْتَغْشَوْا۟ ثِيَابَهُمْ وَأَصَرُّوا۟ وَٱسْتَكْبَرُوا۟ ٱسْتِكْبَارًۭا ﴿7﴾
اور جب بھی میں نے انہیں بلایا تاکہ تو انہیں بخش دے، انہوں نے اپنی انگلیاں کانوں میں ڈال لیں، کپڑے اوڑھ لیے اور اصرار کیا اور بڑا تکبر کیا۔
ثُمَّ إِنِّى دَعَوْتُهُمْ جِهَارًۭا ﴿8﴾
پھر میں نے انہیں کھلم کھلا بلایا۔
ثُمَّ إِنِّىٓ أَعْلَنتُ لَهُمْ وَأَسْرَرْتُ لَهُمْ إِسْرَارًۭا ﴿9﴾
پھر میں نے ان سے علانیہ بھی کہا اور چپکے چپکے بھی کہا۔
فَقُلْتُ ٱسْتَغْفِرُوا۟ رَبَّكُمْ إِنَّهُۥ كَانَ غَفَّارًۭا ﴿10﴾
میں نے کہا: اپنے رب سے بخشش مانگو، بے شک وہ بڑا بخشنے والا ہے۔
يُرْسِلِ ٱلسَّمَآءَ عَلَيْكُم مِّدْرَارًۭا ﴿11﴾
وہ تم پر آسمان سے خوب بارش برسائے گا۔
وَيُمْدِدْكُم بِأَمْوَٰلٍۢ وَبَنِينَ وَيَجْعَل لَّكُمْ جَنَّـٰتٍۢ وَيَجْعَل لَّكُمْ أَنْهَـٰرًۭا ﴿12﴾
اور وہ تمہیں مال و اولاد سے نوازے گا اور تمہارے لیے باغات اور نہریں بنائے گا۔
مَّا لَكُمْ لَا تَرْجُونَ لِلَّهِ وَقَارًۭا ﴿13﴾
تمہیں کیا ہوا ہے کہ تم اللہ کی عظمت کا خیال نہیں رکھتے؟
وَقَدْ خَلَقَكُمْ أَطْوَارًۭا ﴿14﴾
حالانکہ اس نے تمہیں مختلف حالتوں میں پیدا کیا۔
أَلَمْ تَرَوْا۟ كَيْفَ خَلَقَ ٱللَّهُ سَبْعَ سَمَـٰوَٰتٍۢ طِبَاقًۭا ﴿15﴾
کیا تم نے نہیں دیکھا کہ اللہ نے سات آسمان تہ بہ تہ بنائے؟
وَجَعَلَ ٱلْقَمَرَ فِيهِنَّ نُورًۭا وَجَعَلَ ٱلشَّمْسَ سِرَاجًۭا ﴿16﴾
اور اس نے ان میں چاند کو نور بنایا اور سورج کو چراغ بنایا۔
وَٱللَّهُ أَنۢبَتَكُم مِّنَ ٱلْأَرْضِ نَبَاتًۭا ﴿17﴾
اور اللہ نے تمہیں زمین سے اگایا جیسے پودے اگتے ہیں۔
ثُمَّ يُعِيدُكُمْ فِيهَا وَيُخْرِجُكُمْ إِخْرَاجًۭا ﴿18﴾
پھر وہ تمہیں اسی میں واپس لے جائے گا اور تمہیں دوبارہ نکالے گا۔
وَٱللَّهُ جَعَلَ لَكُمُ ٱلْأَرْضَ بِسَاطًۭا ﴿19﴾
اور اللہ نے تمہارے لیے زمین کو بچھونا بنایا۔
لِّتَسْلُكُوا۟ مِنْهَا سُبُلًۭا فِجَاجًۭا ﴿20﴾
تاکہ تم اس میں کشادہ راستوں پر چلو۔
قَالَ نُوحٌۭ رَّبِّ إِنَّهُمْ عَصَوْنِى وَٱتَّبَعُوا۟ مَن لَّمْ يَزِدْهُ مَالُهُۥ وَوَلَدُهُۥٓ إِلَّا خَسَارًۭا ﴿21﴾
نوح نے کہا: اے میرے رب! انہوں نے میری نافرمانی کی اور اس کی پیروی کی جس کا مال اور اولاد اسے خسارے کے سوا کچھ نہ بڑھائے۔
وَمَكَرُوا۟ مَكْرًۭا كُبَّارًۭا ﴿22﴾
اور انہوں نے بڑا سخت مکر کیا۔
وَقَالُوا۟ لَا تَذَرُنَّ ءَالِهَتَكُمْ وَلَا تَذَرُنَّ وَدًّۭا وَلَا سُوَاعًۭا وَلَا يَغُوثَ وَيَعُوقَ وَنَسْرًۭا ﴿23﴾
اور انہوں نے کہا: اپنے معبودوں کو مت چھوڑو، اور ودّ، سواع، یغوث، یعوق اور نسر کو ہرگز نہ چھوڑو۔
وَقَدْ أَضَلُّوا۟ كَثِيرًۭا ۖ وَلَا تَزِدِ ٱلظَّـٰلِمِينَ إِلَّا ضَلَـٰلًۭا ﴿24﴾
اور انہوں نے بہتوں کو گمراہ کیا، اور تو ظالموں کو اور زیادہ گمراہی ہی دے۔
مِّمَّا خَطِيٓـَٔـٰتِهِمْ أُغْرِقُوا۟ فَأُدْخِلُوا۟ نَارًۭا فَلَمْ يَجِدُوا۟ لَهُم مِّن دُونِ ٱللَّهِ أَنصَارًۭا ﴿25﴾
ان کے گناہوں کی وجہ سے وہ غرق کر دیے گئے، پھر آگ میں داخل کر دیے گئے اور اللہ کے سوا ان کے لیے کوئی مددگار نہ ملا۔
وَقَالَ نُوحٌۭ رَّبِّ لَا تَذَرْ عَلَى ٱلْأَرْضِ مِنَ ٱلْكَـٰفِرِينَ دَيَّارًۭا ﴿26﴾
نوح نے کہا: اے میرے رب! زمین پر کافروں میں سے کوئی رہنے والا نہ چھوڑ۔
إِنَّكَ إِن تَذَرْهُمْ يُضِلُّوا۟ عِبَادَكَ وَلَا يَلِدُوٓا۟ إِلَّا فَاجِرًۭا كَفَّارًۭا ﴿27﴾
بے شک اگر تو انہیں چھوڑ دے گا تو وہ تیرے بندوں کو گمراہ کریں گے اور بدکار کافر ہی پیدا کریں گے۔
رَّبِّ ٱغْفِرْ لِى وَلِوَٰلِدَىَّ وَلِمَن دَخَلَ بَيْتِىَ مُؤْمِنًۭا وَلِلْمُؤْمِنِينَ وَٱلْمُؤْمِنَـٰتِ ۖ وَلَا تَزِدِ ٱلظَّـٰلِمِينَ إِلَّا تَبَارًۭا ﴿28﴾
اے میرے رب! مجھے، میرے والدین کو، جو میرے گھر ایمان کے ساتھ داخل ہو اور سب ایمان والے مردوں اور عورتوں کو بخش دے، اور ظالموں کو ہلاکت کے سوا کچھ نہ بڑھا۔
سورة الجن
بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
قُلْ أُوحِىَ إِلَىَّ أَنَّهُ ٱسْتَمَعَ نَفَرٌۭ مِّنَ ٱلْجِنِّ فَقَالُوٓا۟ إِنَّا سَمِعْنَا قُرْءَانًا عَجَبًۭا ﴿1﴾
کہہ دو: میری طرف وحی کی گئی ہے کہ جنوں کی ایک جماعت نے (قرآن) سنا اور کہا: ہم نے عجیب قرآن سنا۔
يَهْدِىٓ إِلَى ٱلرُّشْدِ فَـَٔامَنَّا بِهِۦ ۖ وَلَن نُّشْرِكَ بِرَبِّنَآ أَحَدًۭا ﴿2﴾
جو سیدھی راہ کی طرف راہنمائی کرتا ہے، تو ہم اس پر ایمان لے آئے اور ہم ہرگز اپنے رب کے ساتھ کسی کو شریک نہ کریں گے۔
وَأَنَّهُۥ تَعَـٰلَىٰ جَدُّ رَبِّنَا مَا ٱتَّخَذَ صَـٰحِبَةًۭ وَلَا وَلَدًۭا ﴿3﴾
اور یہ کہ ہمارے رب کی شان بلند ہے، اس نے نہ بیوی بنائی اور نہ ہی اولاد۔
وَأَنَّهُۥ كَانَ يَقُولُ سَفِيهُنَا عَلَى ٱللَّهِ شَطَطًۭا ﴿4﴾
اور یہ کہ ہمارا احمق اللہ کے بارے میں بیہودہ باتیں کہا کرتا تھا۔
وَأَنَّا ظَنَنَّآ أَن لَّن تَقُولَ ٱلْإِنسُ وَٱلْجِنُّ عَلَى ٱللَّهِ كَذِبًۭا ﴿5﴾
اور یہ کہ ہم نے سمجھا تھا کہ انسان اور جن اللہ پر کبھی جھوٹ نہ بولیں گے۔
وَأَنَّهُۥ كَانَ رِجَالٌۭ مِّنَ ٱلْإِنسِ يَعُوذُونَ بِرِجَالٍۭ مِّنَ ٱلْجِنِّ فَزَادُوهُمْ رَهَقًۭا ﴿6﴾
اور یہ کہ انسانوں میں سے کچھ لوگ جنوں میں سے کچھ مردوں کی پناہ لیتے تھے، تو انہوں نے ان کا بوجھ بڑھا دیا۔
وَأَنَّهُمْ ظَنُّوا۟ كَمَا ظَنَنتُمْ أَن لَّن يَبْعَثَ ٱللَّهُ أَحَدًۭا ﴿7﴾
اور یہ کہ انہوں نے گمان کیا جیسے تم نے گمان کیا کہ اللہ کسی کو دوبارہ نہیں اٹھائے گا۔
وَأَنَّا لَمَسْنَا ٱلسَّمَآءَ فَوَجَدْنَـٰهَا مُلِئَتْ حَرَسًۭا شَدِيدًۭا وَشُهُبًۭا ﴿8﴾
اور یہ کہ ہم نے آسمان کو چھوا تو اسے سخت پہرے داروں اور شعلوں سے بھرا ہوا پایا۔
وَأَنَّا كُنَّا نَقْعُدُ مِنْهَا مَقَـٰعِدَ لِلسَّمْعِ ۖ فَمَن يَسْتَمِعِ ٱلْـَٔانَ يَجِدْ لَهُۥ شِهَابًۭا رَّصَدًۭا ﴿9﴾
اور یہ کہ ہم وہاں بیٹھا کرتے تھے سننے کے لیے، مگر اب جو کوئی سننا چاہے گا وہ اپنے لیے جلتا ہوا شعلہ پائے گا جو اس کی تاک میں ہے۔
وَأَنَّا لَا نَدْرِىٓ أَشَرٌّ أُرِيدَ بِمَن فِى ٱلْأَرْضِ أَمْ أَرَادَ بِهِمْ رَبُّهُمْ رَشَدًۭا ﴿10﴾
اور یہ کہ ہم نہیں جانتے کہ زمین والوں کے ساتھ برا ارادہ کیا گیا ہے یا ان کے رب نے ان کے ساتھ بھلائی کا ارادہ کیا ہے۔
وَأَنَّا مِنَّا ٱلصَّـٰلِحُونَ وَمِنَّا دُونَ ذَٰلِكَ ۖ كُنَّا طَرَآئِقَ قِدَدًۭا ﴿11﴾
اور یہ کہ ہم میں کچھ نیک ہیں اور ہم میں کچھ اس کے علاوہ بھی ہیں، ہم مختلف راستوں پر ہیں۔
وَأَنَّا ظَنَنَّآ أَن لَّن نُّعْجِزَ ٱللَّهَ فِى ٱلْأَرْضِ وَلَن نُّعْجِزَهُۥ هَرَبًۭا ﴿12﴾
اور یہ کہ ہم نے سمجھا تھا کہ ہم اللہ کو زمین میں عاجز نہیں کر سکتے اور نہ ہی بھاگ کر اس کو عاجز کر سکتے ہیں۔
وَأَنَّا لَّمَّا سَمِعْنَا ٱلْهُدَىٰٓ ءَامَنَّا بِهِۦ ۖ فَمَن يُؤْمِنۢ بِرَبِّهِۦ فَلَا يَخَافُ بَخْسًۭا وَلَا رَهَقًۭا ﴿13﴾
اور یہ کہ جب ہم نے ہدایت سنی تو ہم اس پر ایمان لے آئے، تو جو کوئی اپنے رب پر ایمان لائے گا اسے نہ نقصان کا ڈر ہوگا اور نہ زیادتی کا۔
وَأَنَّا مِنَّا ٱلْمُسْلِمُونَ وَمِنَّا ٱلْقَـٰسِطُونَ ۖ فَمَنْ أَسْلَمَ فَأُو۟لَـٰٓئِكَ تَحَرَّوْا۟ رَشَدًۭا ﴿14﴾
اور یہ کہ ہم میں کچھ فرمانبردار ہیں اور ہم میں کچھ نافرمان ہیں، تو جو فرمانبردار ہوئے انہوں نے راہِ راست تلاش کی۔
وَأَمَّا ٱلْقَـٰسِطُونَ فَكَانُوا۟ لِجَهَنَّمَ حَطَبًۭا ﴿15﴾
اور جو نافرمان ہوئے وہ جہنم کا ایندھن بنیں گے۔
وَأَلَّوِ ٱسْتَقَـٰمُوا۟ عَلَى ٱلطَّرِيقَةِ لَأَسْقَيْنَـٰهُم مَّآءً غَدَقًۭا ﴿16﴾
اور اگر وہ سیدھے راستے پر قائم رہتے تو ہم انہیں خوب پانی پلاتے۔
لِّنَفْتِنَهُمْ فِيهِ ۚ وَمَن يُعْرِضْ عَن ذِكْرِ رَبِّهِۦ يَسْلُكْهُ عَذَابًۭا صَعَدًۭا ﴿17﴾
تاکہ ہم اس میں ان کو آزمائیں، اور جو اپنے رب کی یاد سے منہ پھیرے گا اللہ اسے سخت عذاب میں ڈال دے گا۔
وَأَنَّ ٱلْمَسَـٰجِدَ لِلَّهِ فَلَا تَدْعُوا۟ مَعَ ٱللَّهِ أَحَدًۭا ﴿18﴾
اور یہ کہ مسجدیں اللہ ہی کے لیے ہیں، پس اللہ کے ساتھ کسی کو نہ پکارو۔
وَأَنَّهُۥ لَمَّا قَامَ عَبْدُ ٱللَّهِ يَدْعُوهُ كَادُوا۟ يَكُونُونَ عَلَيْهِ لِبَدًۭا ﴿19﴾
اور یہ کہ جب اللہ کا بندہ اسے پکارتا تو وہ اس پر ٹوٹ پڑنے کو ہوتے تھے۔
قُلْ إِنَّمَآ أَدْعُوا۟ رَبِّى وَلَآ أُشْرِكُ بِهِۦٓ أَحَدًۭا ﴿20﴾
کہہ دو: میں تو صرف اپنے رب کو پکارتا ہوں اور اس کے ساتھ کسی کو شریک نہیں کرتا۔
قُلْ إِنِّى لَآ أَمْلِكُ لَكُمْ ضَرًّۭا وَلَا رَشَدًۭا ﴿21﴾
کہہ دو: میں تمہارے لیے نہ نقصان کا مالک ہوں نہ بھلائی کا۔
قُلْ إِنِّى لَن يُجِيرَنِى مِنَ ٱللَّهِ أَحَدٌۭ وَلَنْ أَجِدَ مِن دُونِهِۦ مُلْتَحَدًا ﴿22﴾
کہہ دو: مجھے اللہ سے کوئی پناہ نہیں دے سکتا اور نہ ہی میں اس کے سوا کوئی جائے پناہ پاؤں گا۔
إِلَّا بَلَـٰغًۭا مِّنَ ٱللَّهِ وَرِسَـٰلَـٰتِهِۦ ۚ وَمَن يَعْصِ ٱللَّهَ وَرَسُولَهُۥ فَإِنَّ لَهُۥ نَارَ جَهَنَّمَ خَـٰلِدِينَ فِيهَآ أَبَدًۭا ﴿23﴾
میں تو صرف اللہ کی طرف سے پیغام اور اس کی رسالت پہنچاتا ہوں، اور جو اللہ اور اس کے رسول کی نافرمانی کرے اس کے لیے جہنم کی آگ ہے، جس میں ہمیشہ رہیں گے۔
حَتَّىٰٓ إِذَا رَأَوْا۟ مَا يُوعَدُونَ فَسَيَعْلَمُونَ مَنْ أَضْعَفُ نَاصِرًۭا وَأَقَلُّ عَدَدًۭا ﴿24﴾
یہاں تک کہ جب وہ دیکھیں گے جس کا انہیں وعدہ دیا جا رہا ہے تو انہیں معلوم ہو جائے گا کہ کس کا مددگار کمزور اور کس کی تعداد کم ہے۔
قُلْ إِنْ أَدْرِىٓ أَقَرِيبٌۭ مَّا تُوعَدُونَ أَمْ يَجْعَلُ لَهُۥ رَبِّىٓ أَمَدًۭا ﴿25﴾
کہہ دو: میں نہیں جانتا کہ جس کا تم سے وعدہ کیا جا رہا ہے وہ قریب ہے یا میرا رب اس کے لیے کوئی مدت مقرر کرے گا۔
عَـٰلِمُ ٱلْغَيْبِ فَلَا يُظْهِرُ عَلَىٰ غَيْبِهِۦٓ أَحَدًا ﴿26﴾
وہ غیب کا جاننے والا ہے، وہ اپنے غیب پر کسی کو مطلع نہیں کرتا۔
إِلَّا مَنِ ٱرْتَضَىٰ مِن رَّسُولٍۢ فَإِنَّهُۥ يَسْلُكُ مِنۢ بَيْنِ يَدَيْهِ وَمِنْ خَلْفِهِۦ رَصَدًۭا ﴿27﴾
مگر جس رسول کو وہ پسند کرے، تو وہ اس کے آگے اور پیچھے پہرے دار مقرر کر دیتا ہے۔
لِّيَعْلَمَ أَنْ قَدْ أَبْلَغُوا۟ رِسَـٰلَـٰتِ رَبِّهِمْ ۚ وَأَحَاطَ بِمَا لَدَيْهِمْ وَأَحْصَىٰ كُلَّ شَىْءٍ عَدَدًۭا ﴿28﴾
تاکہ وہ جان لے کہ انہوں نے اپنے رب کے پیغامات پہنچا دیے ہیں، اور اس نے ان کے پاس کی ہر چیز کو گھیر رکھا ہے اور ہر چیز کو گن رکھا ہے۔
سورة المزمل
بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
يَـٰٓأَيُّهَا ٱلْمُزَّمِّلُ ﴿1﴾
اے کپڑا اوڑھنے والے!
قُمِ ٱلَّيْلَ إِلَّا قَلِيلًۭا ﴿2﴾
رات کو (نماز کے لیے) کھڑا رہو مگر تھوڑا۔
نِّصْفَهُۥٓ أَوِ ٱنقُصْ مِنْهُ قَلِيلًا ﴿3﴾
آدھی رات یا اس سے کچھ کم کر دو۔
أَوْ زِدْ عَلَيْهِ وَرَتِّلِ ٱلْقُرْءَانَ تَرْتِيلًا ﴿4﴾
یا اس سے زیادہ کر دو اور قرآن ٹھہر ٹھہر کر پڑھو۔
إِنَّا سَنُلْقِى عَلَيْكَ قَوْلًۭا ثَقِيلًا ﴿5﴾
ہم تم پر بھاری کلام نازل کرنے والے ہیں۔
إِنَّ نَاشِئَةَ ٱلَّيْلِ هِىَ أَشَدُّ وَطْـًۭٔا وَأَقْوَمُ قِيلًا ﴿6﴾
رات کا اٹھنا دل کو مضبوط کرتا ہے اور بات سیدھی کرتا ہے۔
إِنَّ لَكَ فِى ٱلنَّهَارِ سَبْحًۭا طَوِيلًۭا ﴿7﴾
یقیناً دن کے وقت تمہارے لیے بہت مصروفیت ہے۔
وَٱذْكُرِ ٱسْمَ رَبِّكَ وَتَبَتَّلْ إِلَيْهِ تَبْتِيلًۭا ﴿8﴾
اور اپنے رب کا نام یاد کرو اور سب سے الگ ہو کر اسی کی طرف متوجہ رہو۔
رَّبُّ ٱلْمَشْرِقِ وَٱلْمَغْرِبِ لَآ إِلَـٰهَ إِلَّا هُوَ فَٱتَّخِذْهُ وَكِيلًۭا ﴿9﴾
وہی مشرق اور مغرب کا رب ہے، اس کے سوا کوئی معبود نہیں، پس اسی کو کارساز بناؤ۔
وَٱصْبِرْ عَلَىٰ مَا يَقُولُونَ وَٱهْجُرْهُمْ هَجْرًۭا جَمِيلًۭا ﴿10﴾
جو وہ کہتے ہیں اس پر صبر کرو اور ان سے خوبصورتی کے ساتھ کنارہ کشی اختیار کرو۔
وَذَرْنِى وَٱلْمُكَذِّبِينَ أُو۟لِى ٱلنَّعْمَةِ وَمَهِّلْهُمْ قَلِيلًا ﴿11﴾
مجھے اور ان جھٹلانے والوں کو جو آسائش میں ہیں چھوڑ دو اور انہیں تھوڑی مہلت دے دو۔
إِنَّ لَدَيْنَآ أَنكَالًۭا وَجَحِيمًۭا ﴿12﴾
یقیناً ہمارے پاس بیڑیاں اور جہنم ہے۔
وَطَعَامًۭا ذَا غُصَّةٍۢ وَعَذَابًا أَلِيمًۭا ﴿13﴾
اور گلے میں اٹکنے والا کھانا اور دردناک عذاب ہے۔
يَوْمَ تَرْجُفُ ٱلْأَرْضُ وَٱلْجِبَالُ وَكَانَتِ ٱلْجِبَالُ كَثِيبًۭا مَّهِيلًا ﴿14﴾
جس دن زمین اور پہاڑ کانپیں گے اور پہاڑ ریزہ ریزہ ریت ہو جائیں گے۔
إِنَّآ أَرْسَلْنَآ إِلَيْكُمْ رَسُولًۭا شَـٰهِدًۭا عَلَيْكُمْ كَمَآ أَرْسَلْنَآ إِلَىٰ فِرْعَوْنَ رَسُولًۭا ﴿15﴾
یقیناً ہم نے تمہاری طرف ایک رسول بھیجا جو تم پر گواہ ہے، جیسے ہم نے فرعون کی طرف رسول بھیجا۔
فَعَصَىٰ فِرْعَوْنُ ٱلرَّسُولَ فَأَخَذْنَـٰهُ أَخْذًۭا وَبِيلًۭا ﴿16﴾
پس فرعون نے رسول کی نافرمانی کی تو ہم نے اسے سخت پکڑ سے پکڑا۔
فَكَيْفَ تَتَّقُونَ إِن كَفَرْتُمْ يَوْمًۭا يَجْعَلُ ٱلْوِلْدَانَ شِيبًۭا ﴿17﴾
پھر اگر تم کفر کرو گے تو کس طرح اس دن سے بچو گے جو بچوں کو بوڑھا کر دے گا؟
ٱلسَّمَآءُ مُنفَطِرٌۢ بِهِۦ ۚ كَانَ وَعْدُهُۥ مَفْعُولًا ﴿18﴾
آسمان اس دن پھٹ جائے گا، اس کا وعدہ پورا ہو کر رہے گا۔
إِنَّ هَـٰذِهِۦ تَذْكِرَةٌۭ ۖ فَمَن شَآءَ ٱتَّخَذَ إِلَىٰ رَبِّهِۦ سَبِيلًۭا ﴿19﴾
یقیناً یہ نصیحت ہے، پس جو چاہے اپنے رب کی طرف راستہ اختیار کرے۔
إِنَّ رَبَّكَ يَعْلَمُ أَنَّكَ تَقُومُ أَدْنَىٰ مِن ثُلُثَىِ ٱلَّيْلِ وَنِصْفَهُۥ وَثُلُثَهُۥ وَطَآئِفَةٌۭ مِّنَ ٱلَّذِينَ مَعَكَ ۚ وَٱللَّهُ يُقَدِّرُ ٱلَّيْلَ وَٱلنَّهَارَ ۚ عَلِمَ أَن لَّن تُحْصُوهُ فَتَابَ عَلَيْكُمْ فَٱقْرَءُوا۟ مَا تَيَسَّرَ مِنَ ٱلْقُرْءَانِ ۚ عَلِمَ أَن سَيَكُونُ مِنكُم مَّرْضَىٰ وَءَاخَرُونَ يَضْرِبُونَ فِى ٱلْأَرْضِ يَبْتَغُونَ مِن فَضْلِ ٱللَّهِ وَءَاخَرُونَ يُقَـٰتِلُونَ فِى سَبِيلِ ٱللَّهِ فَٱقْرَءُوا۟ مَا تَيَسَّرَ مِنْهُ ۚ وَأَقِيمُوا۟ ٱلصَّلَوٰةَ وَءَاتُوا۟ ٱلزَّكَوٰةَ وَأَقْرِضُوا۟ ٱللَّهَ قَرْضًا حَسَنًۭا ۚ وَمَا تُقَدِّمُوا۟ لِأَنفُسِكُم مِّنْ خَيْرٍۢ تَجِدُوهُ عِندَ ٱللَّهِ هُوَ خَيْرًۭا وَأَعْظَمَ أَجْرًۭا ۚ وَٱسْتَغْفِرُوا۟ ٱللَّهَ ۖ إِنَّ ٱللَّهَ غَفُورٌۭ رَّحِيمٌۭ ﴿20﴾
یقیناً تمہارا رب جانتا ہے کہ تم کبھی دو تہائی رات اور کبھی آدھی اور کبھی تہائی کے قریب قیام کرتے ہو، اور تمہارے ساتھ کچھ لوگ بھی۔ اور اللہ رات اور دن کا حساب رکھتا ہے۔ وہ جانتا ہے کہ تم اس کو پوری طرح شمار نہیں کر سکو گے، پس اس نے تم پر رحمت کی۔ لہٰذا قرآن میں سے جو آسان ہو پڑھ لیا کرو۔ وہ جانتا ہے کہ تم میں کچھ بیمار ہوں گے، کچھ زمین میں سفر کریں گے اللہ کا فضل تلاش کرتے ہوئے، اور کچھ اللہ کی راہ میں لڑیں گے۔ پس قرآن میں سے جو میسر ہو پڑھ لیا کرو، اور نماز قائم کرو، زکوٰۃ دو اور اللہ کو اچھا قرض دو۔ اور جو بھلائی اپنے لیے آگے بھیجو گے، اللہ کے پاس اسے بہتر اور بڑے اجر کے ساتھ پاؤ گے۔ اور اللہ سے بخشش مانگو، یقیناً اللہ بخشنے والا مہربان ہے۔
سورة المدثر
بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
يَـٰٓأَيُّهَا ٱلْمُدَّثِّرُ ﴿1﴾
اے اوڑھنے والے!
قُمْ فَأَنذِرْ ﴿2﴾
کھڑے ہو جاؤ اور ڈراؤ۔
وَرَبَّكَ فَكَبِّرْ ﴿3﴾
اور اپنے رب کی بڑائی بیان کرو۔
وَثِيَابَكَ فَطَهِّرْ ﴿4﴾
اور اپنے کپڑوں کو پاک رکھو۔
وَٱلرُّجْزَ فَٱهْجُرْ ﴿5﴾
اور گندگی (بتوں) کو چھوڑ دو۔
وَلَا تَمْنُن تَسْتَكْثِرُ ﴿6﴾
اور احسان نہ کرو زیادہ پانے کے لیے۔
وَلِرَبِّكَ فَٱصْبِرْ ﴿7﴾
اور اپنے رب کی خاطر صبر کرو۔
فَإِذَا نُقِرَ فِى ٱلنَّاقُورِ ﴿8﴾
پھر جب صور میں پھونکا جائے گا۔
فَذَٰلِكَ يَوْمَئِذٍۭ يَوْمٌ عَسِيرٌ ﴿9﴾
تو وہ دن اس دن سخت ہوگا۔
عَلَى ٱلْكَـٰفِرِينَ غَيْرُ يَسِيرٍۭ ﴿10﴾
کافروں پر آسان نہ ہوگا۔
ذَرْنِى وَمَنْ خَلَقْتُ وَحِيدًۭا ﴿11﴾
مجھے اور اس کو چھوڑ دے جسے میں نے اکیلا پیدا کیا۔
وَجَعَلْتُ لَهُۥ مَالًۭا مَّمْدُودًۭا ﴿12﴾
اور میں نے اسے پھیلا ہوا مال دیا۔
وَبَنِينَ شُهُودًۭا ﴿13﴾
اور حاضر رہنے والے بیٹے دیے۔
وَمَهَّدتُّ لَهُۥ تَمْهِيدًۭا ﴿14﴾
اور اس کے لیے سہولتیں فراہم کیں۔
ثُمَّ يَطْمَعُ أَنْ أَزِيدَ ﴿15﴾
پھر وہ طمع کرتا ہے کہ میں اور زیادہ دوں۔
كَلَّآ ۖ إِنَّهُۥ كَانَ لِـَٔايَـٰتِنَا عَنِيدًۭا ﴿16﴾
ہرگز نہیں! وہ تو ہماری آیات کا دشمن تھا۔
سَأُرْهِقُهُۥ صَعُودًا ﴿17﴾
میں اسے سخت چڑھائی پر چڑھاؤں گا۔
إِنَّهُۥ فَكَّرَ وَقَدَّرَ ﴿18﴾
بیشک اس نے سوچا اور اندازہ لگایا۔
فَقُتِلَ كَيْفَ قَدَّرَ ﴿19﴾
پس ہلاک ہو وہ! کس طرح اندازہ لگایا۔
ثُمَّ قُتِلَ كَيْفَ قَدَّرَ ﴿20﴾
پھر ہلاک ہو! کس طرح اندازہ لگایا۔
ثُمَّ نَظَرَ ﴿21﴾
پھر اس نے غور کیا۔
ثُمَّ عَبَسَ وَبَسَرَ ﴿22﴾
پھر اس نے تیوری چڑھائی اور منہ بگاڑا۔
ثُمَّ أَدْبَرَ وَٱسْتَكْبَرَ ﴿23﴾
پھر پیٹھ موڑی اور تکبر کیا۔
فَقَالَ إِنْ هَـٰذَآ إِلَّا سِحْرٌۭ يُؤْثَرُ ﴿24﴾
پھر کہا یہ تو محض پرانا جادو ہے۔
إِنْ هَـٰذَآ إِلَّا قَوْلُ ٱلْبَشَرِ ﴿25﴾
یہ تو بس انسان کا کلام ہے۔
سَأُصْلِيهِ سَقَرَ ﴿26﴾
میں اسے دوزخ میں داخل کروں گا۔
وَمَآ أَدْرَىٰكَ مَا سَقَرُ ﴿27﴾
اور تمہیں کیا معلوم دوزخ کیا ہے؟
لَا تُبْقِى وَلَا تَذَرُ ﴿28﴾
نہ باقی چھوڑے گی اور نہ چھوڑے گی۔
لَوَّاحَةٌۭ لِّلْبَشَرِ ﴿29﴾
یہ انسان کو جلا ڈالنے والی ہے۔
عَلَيْهَا تِسْعَةَ عَشَرَ ﴿30﴾
اس پر انیس (فرشتے) مقرر ہیں۔
وَمَا جَعَلْنَا أَصْحَـٰبَ ٱلنَّارِ إِلَّا مَلَـٰٓئِكَةًۭ ۙ وَمَا جَعَلْنَا عِدَّتَهُمْ إِلَّا فِتْنَةًۭ لِّلَّذِينَ كَفَرُوا۟ لِيَسْتَيْقِنَ ٱلَّذِينَ أُوتُوا۟ ٱلْكِتَـٰبَ وَيَزْدَادَ ٱلَّذِينَ ءَامَنُوٓا۟ إِيمَـٰنًۭا وَلَا يَرْتَابَ ٱلَّذِينَ أُوتُوا۟ ٱلْكِتَـٰبَ وَٱلْمُؤْمِنُونَ وَلِيَقُولَ ٱلَّذِينَ فِى قُلُوبِهِم مَّرَضٌۭ وَٱلْكَـٰفِرُونَ مَاذَآ أَرَادَ ٱللَّهُ بِهَـٰذَا مَثَلًۭا ۚ كَذَٰلِكَ يُضِلُّ ٱللَّهُ مَن يَشَآءُ وَيَهْدِى مَن يَشَآءُ ۚ وَمَا يَعْلَمُ جُنُودَ رَبِّكَ إِلَّا هُوَ ۚ وَمَا هِىَ إِلَّا ذِكْرَىٰ لِلْبَشَرِ ﴿31﴾
اور ہم نے دوزخ کے نگہبان صرف فرشتے بنائے ہیں اور ہم نے ان کی تعداد کافروں کے لیے آزمائش بنائی ہے، تاکہ اہلِ کتاب یقین حاصل کریں اور ایمان والوں کا ایمان بڑھے، اور اہلِ کتاب اور مومنین شک نہ کریں، اور جن کے دلوں میں بیماری ہے اور کافر کہیں: اللہ نے اس مثال سے کیا چاہا ہے؟ اسی طرح اللہ جسے چاہتا ہے گمراہ کرتا ہے اور جسے چاہتا ہے ہدایت دیتا ہے، اور تیرے رب کے لشکروں کو اس کے سوا کوئی نہیں جانتا، اور یہ قرآن تو صرف انسانوں کے لیے نصیحت ہے۔
كَلَّا وَٱلْقَمَرِ ﴿32﴾
ہرگز نہیں! قسم ہے چاند کی۔
وَٱلَّيْلِ إِذْ أَدْبَرَ ﴿33﴾
اور رات کی جب وہ پیٹھ پھیر جائے۔
وَٱلصُّبْحِ إِذَآ أَسْفَرَ ﴿34﴾
اور صبح کی جب وہ روشن ہو۔
إِنَّهَا لَإِحْدَى ٱلْكُبَرِ ﴿35﴾
یقیناً یہ (دوزخ) بڑی چیزوں میں سے ایک ہے۔
نَذِيرًۭا لِّلْبَشَرِ ﴿36﴾
انسان کے لیے ڈرانے والی ہے۔
لِمَن شَآءَ مِنكُمْ أَن يَتَقَدَّمَ أَوْ يَتَأَخَّرَ ﴿37﴾
تم میں سے جو چاہے آگے بڑھے یا پیچھے ہٹے۔
كُلُّ نَفْسٍۭ بِمَا كَسَبَتْ رَهِينَةٌ ﴿38﴾
ہر شخص اپنے اعمال کے بدلے گروی ہے۔
إِلَّآ أَصْحَـٰبَ ٱلْيَمِينِ ﴿39﴾
مگر دائیں ہاتھ والے۔
فِى جَنَّـٰتٍۢ يَتَسَآءَلُونَ ﴿40﴾
وہ جنتوں میں ہوں گے، سوال کریں گے۔
عَنِ ٱلْمُجْرِمِينَ ﴿41﴾
مجرموں کے بارے میں۔
مَا سَلَكَكُمْ فِى سَقَرَ ﴿42﴾
کہ تمہیں دوزخ میں کس چیز نے ڈالا؟
قَالُوا۟ لَمْ نَكُ مِنَ ٱلْمُصَلِّينَ ﴿43﴾
وہ کہیں گے ہم نمازیوں میں سے نہ تھے۔
وَلَمْ نَكُ نُطْعِمُ ٱلْمِسْكِينَ ﴿44﴾
اور ہم مسکین کو کھانا نہیں کھلاتے تھے۔
وَكُنَّا نَخُوضُ مَعَ ٱلْخَآئِضِينَ ﴿45﴾
اور ہم بیہودہ باتوں میں مشغول رہتے تھے۔
وَكُنَّا نُكَذِّبُ بِيَوْمِ ٱلدِّينِ ﴿46﴾
اور ہم قیامت کے دن کو جھٹلاتے تھے۔
حَتَّىٰٓ أَتَىٰنَا ٱلْيَقِينُ ﴿47﴾
یہاں تک کہ ہمیں موت آ پہنچی۔
فَمَا تَنفَعُهُمْ شَفَـٰعَةُ ٱلشَّـٰفِعِينَ ﴿48﴾
پھر ان کے لیے سفارش کرنے والوں کی سفارش فائدہ نہ دے گی۔
فَمَا لَهُمْ عَنِ ٱلتَّذْكِرَةِ مُعْرِضِينَ ﴿49﴾
تو انہیں کیا ہے کہ وہ نصیحت سے منہ موڑتے ہیں؟
كَأَنَّهُمْ حُمُرٌۭ مُّسْتَنفِرَةٌ ﴿50﴾
گویا وہ بدکنے والے گدھے ہیں۔
فَرَّتْ مِن قَسْوَرَةٍۭ ﴿51﴾
جو شیر سے بھاگے ہوں۔
بَلْ يُرِيدُ كُلُّ ٱمْرِئٍۭ مِّنْهُمْ أَن يُؤْتَىٰ صُحُفًۭا مُّنَشَّرَةًۭ ﴿52﴾
بلکہ ان میں سے ہر ایک چاہتا ہے کہ اسے کھلی ہوئی کتابیں دی جائیں۔
كَلَّا ۖ بَل لَّا يَخَافُونَ ٱلْـَٔاخِرَةَ ﴿53﴾
ہرگز نہیں! بلکہ یہ آخرت سے نہیں ڈرتے۔
كَلَّآ إِنَّهُۥ تَذْكِرَةٌۭ ﴿54﴾
یقیناً یہ قرآن نصیحت ہے۔
فَمَن شَآءَ ذَكَرَهُۥ ﴿55﴾
تو جو چاہے اسے یاد کرے۔
وَمَا يَذْكُرُونَ إِلَّآ أَن يَشَآءَ ٱللَّهُ ۚ هُوَ أَهْلُ ٱلتَّقْوَىٰ وَأَهْلُ ٱلْمَغْفِرَةِ ﴿56﴾
اور وہ یاد نہیں کرتے مگر جب اللہ چاہے، وہی تقویٰ کا اہل اور بخشش کا اہل ہے۔
سورة القيامة
بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
لَآ أُقْسِمُ بِيَوْمِ ٱلْقِيَـٰمَةِ ﴿1﴾
میں قیامت کے دن کی قسم کھاتا ہوں۔
وَلَآ أُقْسِمُ بِٱلنَّفْسِ ٱللَّوَّامَةِ ﴿2﴾
اور ملامت کرنے والی جان کی قسم کھاتا ہوں۔
أَيَحْسَبُ ٱلْإِنسَـٰنُ أَلَّن نَّجْمَعَ عِظَامَهُۥ ﴿3﴾
کیا انسان سمجھتا ہے کہ ہم اس کی ہڈیاں جمع نہ کریں گے؟
بَلَىٰ قَـٰدِرِينَ عَلَىٰٓ أَن نُّسَوِّىَ بَنَانَهُۥ ﴿4﴾
کیوں نہیں! ہم تو اس کی انگلیوں کے پوروں کو بھی درست بنانے پر قادر ہیں۔
بَلْ يُرِيدُ ٱلْإِنسَـٰنُ لِيَفْجُرَ أَمَامَهُۥ ﴿5﴾
بلکہ انسان چاہتا ہے کہ آئندہ بھی بداعمالیاں کرتا رہے۔
يَسْـَٔلُ أَيَّانَ يَوْمُ ٱلْقِيَـٰمَةِ ﴿6﴾
وہ پوچھتا ہے قیامت کا دن کب آئے گا؟
فَإِذَا بَرِقَ ٱلْبَصَرُ ﴿7﴾
پھر جب آنکھ پتھرا جائے گی۔
وَخَسَفَ ٱلْقَمَرُ ﴿8﴾
اور چاند بے نور ہو جائے گا۔
وَجُمِعَ ٱلشَّمْسُ وَٱلْقَمَرُ ﴿9﴾
اور سورج اور چاند جمع کر دیے جائیں گے۔
يَقُولُ ٱلْإِنسَـٰنُ يَوْمَئِذٍ أَيْنَ ٱلْمَفَرُّ ﴿10﴾
اس دن انسان کہے گا کہاں بھاگ کر جائے؟
كَلَّا لَا وَزَرَ ﴿11﴾
ہرگز نہیں! کوئی پناہ نہیں۔
إِلَىٰ رَبِّكَ يَوْمَئِذٍ ٱلْمُسْتَقَرُّ ﴿12﴾
اس دن تیرے رب ہی کے پاس ٹھکانا ہوگا۔
يُنَبَّؤُا۟ ٱلْإِنسَـٰنُ يَوْمَئِذٍۭ بِمَا قَدَّمَ وَأَخَّرَ ﴿13﴾
اس دن انسان کو بتایا جائے گا جو کچھ اس نے آگے بھیجا اور جو پیچھے چھوڑا۔
بَلِ ٱلْإِنسَـٰنُ عَلَىٰ نَفْسِهِۦ بَصِيرَةٌۭ ﴿14﴾
بلکہ انسان خود اپنے اوپر گواہ ہے۔
وَلَوْ أَلْقَىٰ مَعَاذِيرَهُۥ ﴿15﴾
اگرچہ وہ اپنے عذر پیش کرے۔
لَا تُحَرِّكْ بِهِۦ لِسَانَكَ لِتَعْجَلَ بِهِۦ ﴿16﴾
(اے نبیﷺ) آپ اس کو جلدی لینے کے لیے اپنی زبان کو حرکت نہ دیں۔
إِنَّ عَلَيْنَا جَمْعَهُۥ وَقُرْءَانَهُۥ ﴿17﴾
اس کو جمع کرنا اور پڑھانا ہمارے ذمے ہے۔
فَإِذَا قَرَأْنَـٰهُ فَٱتَّبِعْ قُرْءَانَهُۥ ﴿18﴾
پھر جب ہم اسے پڑھیں تو آپ اس کی پیروی کریں۔
ثُمَّ إِنَّ عَلَيْنَا بَيَانَهُۥ ﴿19﴾
پھر اس کی وضاحت ہمارے ذمے ہے۔
كَلَّا بَلْ تُحِبُّونَ ٱلْعَاجِلَةَ ﴿20﴾
ہرگز نہیں! بلکہ تم جلدی ملنے والی دنیا سے محبت کرتے ہو۔
وَتَذَرُونَ ٱلْـَٔاخِرَةَ ﴿21﴾
اور آخرت کو چھوڑ دیتے ہو۔
وُجُوهٌۭ يَوْمَئِذٍۢ نَّاضِرَةٌ ﴿22﴾
اس دن کچھ چہرے تروتازہ ہوں گے۔
إِلَىٰ رَبِّهَا نَاظِرَةٌۭ ﴿23﴾
اپنے رب کی طرف دیکھتے ہوئے۔
وَوُجُوهٌۭ يَوْمَئِذٍۭ بَاسِرَةٌۭ ﴿24﴾
اور کچھ چہرے اس دن بگڑے ہوئے ہوں گے۔
تَظُنُّ أَن يُفْعَلَ بِهَا فَاقِرَةٌۭ ﴿25﴾
انہیں یقین ہوگا کہ ان کے ساتھ کمر توڑ دینے والا سلوک کیا جائے گا۔
كَلَّآ إِذَا بَلَغَتِ ٱلتَّرَاقِىَ ﴿26﴾
ہرگز نہیں! جب جان حلق تک پہنچ جائے۔
وَقِيلَ مَنْ ۜ رَاقٍۢ ﴿27﴾
اور کہا جائے گا کوئی جھاڑ پھونک کرنے والا ہے؟
وَظَنَّ أَنَّهُ ٱلْفِرَاقُ ﴿28﴾
اور اسے یقین ہو جائے گا کہ یہ جدائی کا وقت ہے۔
وَٱلْتَفَّتِ ٱلسَّاقُ بِٱلسَّاقِ ﴿29﴾
اور ٹانگ سے ٹانگ لپٹ جائے گی۔
إِلَىٰ رَبِّكَ يَوْمَئِذٍ ٱلْمَسَاقُ ﴿30﴾
اس دن تیرے رب ہی کی طرف لے جایا جائے گا۔
فَلَا صَدَّقَ وَلَا صَلَّىٰ ﴿31﴾
پس نہ اس نے تصدیق کی اور نہ نماز پڑھی۔
وَلَـٰكِن كَذَّبَ وَتَوَلَّىٰ ﴿32﴾
بلکہ اس نے جھٹلایا اور منہ موڑ لیا۔
ثُمَّ ذَهَبَ إِلَىٰٓ أَهْلِهِۦ يَتَمَطَّىٰٓ ﴿33﴾
پھر اپنے گھر والوں کے پاس اتراتا ہوا گیا۔
أَوْلَىٰ لَكَ فَأَوْلَىٰ ﴿34﴾
ہلاکت ہے تجھ پر، پھر ہلاکت ہے۔
ثُمَّ أَوْلَىٰ لَكَ فَأَوْلَىٰ ﴿35﴾
پھر ہلاکت ہے تجھ پر، پھر ہلاکت ہے۔
أَيَحْسَبُ ٱلْإِنسَـٰنُ أَن يُتْرَكَ سُدًۭى ﴿36﴾
کیا انسان سمجھتا ہے کہ اسے یوں ہی چھوڑ دیا جائے گا؟
أَلَمْ يَكُ نُطْفَةًۭ مِّن مَّنِىٍّۢ يُمْنَىٰ ﴿37﴾
کیا وہ ایک قطرہ پانی نہ تھا جو ٹپکایا جاتا ہے؟
ثُمَّ كَانَ عَلَقَةًۭ فَخَلَقَ فَسَوَّىٰ ﴿38﴾
پھر وہ جمے ہوئے خون کی صورت ہوا، پھر اللہ نے اسے بنایا اور درست کیا۔
فَجَعَلَ مِنْهُ ٱلزَّوْجَيْنِ ٱلذَّكَرَ وَٱلْأُنثَىٰٓ ﴿39﴾
پھر اس سے نر اور مادہ کے دو جوڑے بنائے۔
أَلَيْسَ ذَٰلِكَ بِقَـٰدِرٍ عَلَىٰٓ أَن يُحْـِۧىَ ٱلْمَوْتَىٰ ﴿40﴾
کیا وہ اس پر قادر نہیں کہ مردوں کو زندہ کرے؟
سورة الإنسان
بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
هَلْ أَتَىٰ عَلَى ٱلْإِنسَـٰنِ حِينٌۭ مِّنَ ٱلدَّهْرِ لَمْ يَكُن شَيْـًۭٔا مَّذْكُورًا ﴿1﴾
کیا انسان پر ایسا وقت نہیں گزرا جب وہ کوئی قابلِ ذکر چیز نہ تھا؟
إِنَّا خَلَقْنَا ٱلْإِنسَـٰنَ مِن نُّطْفَةٍ أَمْشَاجٍۢ نَّبْتَلِيهِ فَجَعَلْنَـٰهُ سَمِيعًۢا بَصِيرًا ﴿2﴾
بے شک ہم نے انسان کو ملی جلی بوند سے پیدا کیا تاکہ ہم اس کو آزمائیں، پس ہم نے اسے سننے والا اور دیکھنے والا بنایا۔
إِنَّا هَدَيْنَـٰهُ ٱلسَّبِيلَ إِمَّا شَاكِرًۭا وَإِمَّا كَفُورًا ﴿3﴾
ہم نے اسے راستہ دکھا دیا، خواہ وہ شکر گزار ہو یا ناشکرا۔
إِنَّآ أَعْتَدْنَا لِلْكَـٰفِرِينَ سَلَـٰسِلَا۟ وَأَغْلَـٰلًۭا وَسَعِيرًا ﴿4﴾
ہم نے کافروں کے لیے زنجیریں اور طوق اور بھڑکتی آگ تیار کر رکھی ہے۔
إِنَّ ٱلْأَبْرَارَ يَشْرَبُونَ مِن كَأْسٍۢ كَانَ مِزَاجُهَا كَافُورًا ﴿5﴾
یقیناً نیک لوگ ایک ایسے پیالے سے پئیں گے جس میں کافور ملا ہوا ہوگا۔
عَيْنًۭا يَشْرَبُ بِهَا عِبَادُ ٱللَّهِ يُفَجِّرُونَهَا تَفْجِيرًۭا ﴿6﴾
وہ ایک چشمہ ہوگا جس سے اللہ کے بندے پئیں گے، اور وہ اسے بہا کر نکالیں گے۔
يُوفُونَ بِٱلنَّذْرِ وَيَخَافُونَ يَوْمًۭا كَانَ شَرُّهُۥ مُسْتَطِيرًۭا ﴿7﴾
وہ نذر پوری کرتے ہیں اور اس دن سے ڈرتے ہیں جس کی سختی ہر طرف پھیل جائے گی۔
وَيُطْعِمُونَ ٱلطَّعَامَ عَلَىٰ حُبِّهِۦ مِسْكِينًۭا وَيَتِيمًۭا وَأَسِيرًا ﴿8﴾
اور وہ کھانا کھلاتے ہیں اس کی محبت کے باوجود مسکین کو اور یتیم کو اور قیدی کو۔
إِنَّمَا نُطْعِمُكُمْ لِوَجْهِ ٱللَّهِ لَا نُرِيدُ مِنكُمْ جَزَآءًۭ وَلَا شُكُورًا ﴿9﴾
ہم تو تمہیں صرف اللہ کی رضا کے لیے کھلاتے ہیں، نہ تم سے بدلہ چاہتے ہیں اور نہ شکر گزاری۔
إِنَّا نَخَافُ مِن رَّبِّنَا يَوْمًۭا عَبُوسًۭا قَمْطَرِيرًۭا ﴿10﴾
ہم اپنے رب سے اس دن سے ڈرتے ہیں جو سخت اور ہولناک ہوگا۔
فَوَقَىٰهُمُ ٱللَّهُ شَرَّ ذَٰلِكَ ٱلْيَوْمِ وَلَقَّىٰهُمْ نَضْرَةًۭ وَسُرُورًۭا ﴿11﴾
پس اللہ نے انہیں اس دن کی برائی سے بچا لیا اور انہیں تازگی اور خوشی عطا کی۔
وَجَزَىٰهُم بِمَا صَبَرُوا۟ جَنَّةًۭ وَحَرِيرًۭا ﴿12﴾
اور ان کے صبر کے بدلے انہیں جنت اور ریشم دیا۔
مُّتَّكِـِٔينَ فِيهَا عَلَى ٱلْأَرَآئِكِ لَا يَرَوْنَ فِيهَا شَمْسًۭا وَلَا زَمْهَرِيرًۭا ﴿13﴾
وہ اس میں تختوں پر تکیہ لگائے ہوں گے، وہاں نہ دھوپ دیکھیں گے اور نہ شدید سردی۔
وَدَانِيَةً عَلَيْهِمْ ظِلَـٰلُهَا وَذُلِّلَتْ قُطُوفُهَا تَذْلِيلًۭا ﴿14﴾
اور ان پر اس کے سائے قریب ہوں گے اور اس کے خوشے نیچے لٹکائے ہوئے ہوں گے۔
وَيُطَافُ عَلَيْهِم بِـَٔانِيَةٍۢ مِّن فِضَّةٍۢ وَأَكْوَابٍۢ كَانَتْ قَوَارِيرَا ﴿15﴾
ان پر چاندی کے برتن اور شیشے کے پیالے گردش کریں گے۔
قَوَارِيرَا مِن فِضَّةٍۢ قَدَّرُوهَا تَقْدِيرًۭا ﴿16﴾
وہ شیشے چاندی کے ہوں گے جنہیں انہوں نے خوب ناپ تول کر تیار کیا ہوگا۔
وَيُسْقَوْنَ فِيهَا كَأْسًۭا كَانَ مِزَاجُهَا زَنجَبِيلًۭا ﴿17﴾
اور وہاں انہیں ایک ایسا پیالہ پلایا جائے گا جس میں ادرک ملا ہوگا۔
عَيْنًۭا فِيهَا تُسَمَّىٰ سَلْسَبِيلًۭا ﴿18﴾
وہ جنت کا ایک چشمہ ہوگا جسے سلسبیل کہا جاتا ہے۔
وَيَطُوفُ عَلَيْهِمْ وِلْدَٰنٌۭ مُّخَلَّدُونَ إِذَا رَأَيْتَهُمْ حَسِبْتَهُمْ لُؤْلُؤًۭا مَّنثُورًۭا ﴿19﴾
اور ان پر ہمیشہ کے لیے رہنے والے نوجوان خدمت کریں گے، جب تم انہیں دیکھو گے تو انہیں بکھرے ہوئے موتی سمجھو گے۔
وَإِذَا رَأَيْتَ ثَمَّ رَأَيْتَ نَعِيمًۭا وَمُلْكًۭا كَبِيرًۭا ﴿20﴾
اور جب تم وہاں دیکھو گے تو نعمت اور بڑی بادشاہی دیکھو گے۔
عَـٰلِيَهُمْ ثِيَابُ سُندُسٍ خُضْرٌۭ وَإِسْتَبْرَقٌۭ وَحُلُّوٓا۟ أَسَاوِرَ مِن فِضَّةٍۢ وَسَقَىٰهُمْ رَبُّهُمْ شَرَابًۭا طَهُورًا ﴿21﴾
ان پر باریک اور دبیز ریشم کے سبز کپڑے ہوں گے اور انہیں چاندی کے کنگن پہنائے جائیں گے، اور ان کا رب انہیں پاکیزہ شراب پلائے گا۔
إِنَّ هَـٰذَا كَانَ لَكُمْ جَزَآءًۭ وَكَانَ سَعْيُكُم مَّشْكُورًا ﴿22﴾
یہ تمہارے لیے بدلہ ہے اور تمہاری کوشش قبول کی گئی۔
إِنَّا نَحْنُ نَزَّلْنَا عَلَيْكَ ٱلْقُرْءَانَ تَنزِيلًۭا ﴿23﴾
یقیناً ہم ہی نے تم پر قرآن نازل کیا ہے۔
فَٱصْبِرْ لِحُكْمِ رَبِّكَ وَلَا تُطِعْ مِنْهُمْ ءَاثِمًا أَوْ كَفُورًۭا ﴿24﴾
پس اپنے رب کے حکم پر صبر کرو اور ان میں سے کسی گناہ گار یا ناشکرے کا کہا نہ مانو۔
وَٱذْكُرِ ٱسْمَ رَبِّكَ بُكْرَةًۭ وَأَصِيلًۭا ﴿25﴾
اور اپنے رب کا نام صبح و شام یاد کرو۔
وَمِنَ ٱلَّيْلِ فَٱسْجُدْ لَهُۥ وَسَبِّحْهُ لَيْلًۭا طَوِيلًا ﴿26﴾
اور رات کو اس کے آگے سجدہ کرو اور رات کو لمبے عرصے تک اس کی تسبیح کرو۔
إِنَّ هَـٰٓؤُلَآءِ يُحِبُّونَ ٱلْعَاجِلَةَ وَيَذَرُونَ وَرَآءَهُمْ يَوْمًۭا ثَقِيلًۭا ﴿27﴾
یہ لوگ دنیا کو پسند کرتے ہیں اور اپنے پیچھے بھاری دن کو چھوڑ دیتے ہیں۔
نَّحْنُ خَلَقْنَـٰهُمْ وَشَدَدْنَآ أَسْرَهُمْ وَإِذَا شِئْنَا بَدَّلْنَآ أَمْثَـٰلَهُمْ تَبْدِيلًا ﴿28﴾
ہم ہی نے انہیں پیدا کیا اور ان کے جوڑ مضبوط کیے، اور جب چاہیں ہم ان کی جگہ دوسروں کو لے آئیں۔
إِنَّ هَـٰذِهِۦ تَذْكِرَةٌۭ فَمَن شَآءَ ٱتَّخَذَ إِلَىٰ رَبِّهِۦ سَبِيلًۭا ﴿29﴾
یقیناً یہ ایک نصیحت ہے، پس جو چاہے اپنے رب کی طرف راستہ اختیار کرے۔
وَمَا تَشَآءُونَ إِلَّآ أَن يَشَآءَ ٱللَّهُ ۚ إِنَّ ٱللَّهَ كَانَ عَلِيمًا حَكِيمًۭا ﴿30﴾
اور تم کچھ نہیں چاہ سکتے مگر جو اللہ چاہے، بے شک اللہ جاننے والا، حکمت والا ہے۔
يُدْخِلُ مَن يَشَآءُ فِى رَحْمَتِهِۦ ۚ وَٱلظَّـٰلِمِينَ أَعَدَّ لَهُمْ عَذَابًا أَلِيمًۭا ﴿31﴾
وہ جسے چاہے اپنی رحمت میں داخل کرتا ہے، اور ظالموں کے لیے اس نے دردناک عذاب تیار کر رکھا ہے۔
سورة المرسلات
بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
وَٱلْمُرْسَلَـٰتِ عُرْفًۭا ﴿1﴾
قسم ہے بھیجی جانے والی ہواؤں کی جو لگاتار بھیجی جاتی ہیں۔
فَٱلْعَـٰصِفَـٰتِ عَصْفًۭا ﴿2﴾
پھر تیز چلنے والی۔
وَٱلنَّـٰشِرَٰتِ نَشْرًۭا ﴿3﴾
اور پھیلانے والی۔
فَٱلْفَـٰرِقَـٰتِ فَرْقًۭا ﴿4﴾
پھر جدا جدا کرنے والی۔
فَٱلْمُلْقِيَـٰتِ ذِكْرًۭا ﴿5﴾
پھر ذکر (وحی) ڈالنے والی۔
عُذْرًۭا أَوْ نُذْرًۭا ﴿6﴾
عذر یا ڈرانے کے لیے۔
إِنَّمَا تُوعَدُونَ لَوَٰقِعٌۭ ﴿7﴾
یقیناً جس چیز کا وعدہ کیا گیا ہے وہ ضرور واقع ہوگی۔
فَإِذَا ٱلنُّجُومُ طُمِسَتْ ﴿8﴾
پھر جب ستارے بےنور کر دیے جائیں گے۔
وَإِذَا ٱلسَّمَآءُ فُرِجَتْ ﴿9﴾
اور جب آسمان پھاڑ دیا جائے گا۔
وَإِذَا ٱلْجِبَالُ نُسِفَتْ ﴿10﴾
اور جب پہاڑ اڑا دیے جائیں گے۔
وَإِذَا ٱلرُّسُلُ أُقِّتَتْ ﴿11﴾
اور جب رسولوں کو مقررہ وقت پر لایا جائے گا۔
لِأَىِّ يَوْمٍ أُجِّلَتْ ﴿12﴾
کس دن کے لیے مؤخر کیا گیا ہے؟
لِيَوْمِ ٱلْفَصْلِ ﴿13﴾
فیصلے کے دن کے لیے۔
وَمَآ أَدْرَىٰكَ مَا يَوْمُ ٱلْفَصْلِ ﴿14﴾
اور تمہیں کیا معلوم کہ فیصلہ کا دن کیا ہے؟
وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿15﴾
اس دن جھٹلانے والوں کے لیے ہلاکت ہے۔
أَلَمْ نُهْلِكِ ٱلْأَوَّلِينَ ﴿16﴾
کیا ہم نے اگلے لوگوں کو ہلاک نہیں کیا؟
ثُمَّ نُتْبِعُهُمُ ٱلْـَٔاخِرِينَ ﴿17﴾
پھر ہم پچھلوں کو ان کے پیچھے لائیں گے۔
كَذَٰلِكَ نَفْعَلُ بِٱلْمُجْرِمِينَ ﴿18﴾
اسی طرح ہم مجرموں کے ساتھ کرتے ہیں۔
وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿19﴾
اس دن جھٹلانے والوں کے لیے ہلاکت ہے۔
أَلَمْ نَخْلُقكُّم مِّن مَّآءٍۢ مَّهِينٍۢ ﴿20﴾
کیا ہم نے تمہیں ذلیل پانی سے پیدا نہیں کیا؟
فَجَعَلْنَـٰهُ فِى قَرَارٍۢ مَّكِينٍ ﴿21﴾
پھر ہم نے اسے ایک محفوظ جگہ ٹھہرایا۔
إِلَىٰ قَدَرٍۢ مَّعْلُومٍۢ ﴿22﴾
ایک مقررہ وقت تک۔
فَقَدَرْنَا فَنِعْمَ ٱلْقَـٰدِرُونَ ﴿23﴾
پھر ہم نے اندازہ مقرر کیا، پس ہم کیا ہی اچھے اندازہ کرنے والے ہیں۔
وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿24﴾
اس دن جھٹلانے والوں کے لیے ہلاکت ہے۔
أَلَمْ نَجْعَلِ ٱلۡأَرۡضَ كِفَاتٗا ﴿25﴾
کیا ہم نے زمین کو سمیٹنے والی نہیں بنایا؟
أَحۡيَآءٗ وَأَمۡوَٰتٗا ﴿26﴾
زندوں اور مردوں کے لیے۔
وَجَعَلۡنَا فِيهَا رَوَٰسِيَ شَٰمِخَٰتٖ وَأَسۡقَيۡنَٰكُم مَّآءٗ فُرَاتٗا ﴿27﴾
اور ہم نے اس میں بلند پہاڑ رکھے اور تمہیں میٹھا پانی پلایا۔
وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿28﴾
اس دن جھٹلانے والوں کے لیے ہلاکت ہے۔
ٱنطَلِقُوٓاْ إِلَىٰ مَا كُنتُم بِهِۦ تُكَذِّبُونَ ﴿29﴾
چلو اس چیز کی طرف جسے تم جھٹلایا کرتے تھے۔
ٱنطَلِقُوٓاْ إِلَىٰ ظِلّٖ ذِي ثَلَٰثِ شُعَبٖ ﴿30﴾
چلو تین شاخوں والے سائے کی طرف۔
لَّا ظَلِيلٖ وَلَا يُغۡنِي مِنَ ٱللَّهَبِ ﴿31﴾
جو نہ ٹھنڈک دے گا اور نہ شعلے سے بچائے گا۔
إِنَّهَا تَرۡمِي بِشَرَرٖ كَٱلۡقَصۡرِ ﴿32﴾
بے شک وہ آگ چنگاریاں پھینکے گی جو محل جیسے ہوں گی۔
كَأَنَّهُۥ جِمَٰلَتٞ صُفۡرٞ ﴿33﴾
گویا وہ زرد اونٹ ہیں۔
وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿34﴾
اس دن جھٹلانے والوں کے لیے ہلاکت ہے۔
هَٰذَا يَوْمُ لَا يَنطِقُونَ ﴿35﴾
یہ وہ دن ہے جس دن وہ بات نہیں کریں گے۔
وَلَا يُؤۡذَنُ لَهُمۡ فَيَعۡتَذِرُونَ ﴿36﴾
اور نہ انہیں اجازت دی جائے گی کہ عذر پیش کریں۔
وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿37﴾
اس دن جھٹلانے والوں کے لیے ہلاکت ہے۔
هَٰذَا يَوْمُ ٱلْفَصْلِ جَمَعْنَـٰكُمْ وَٱلۡأَوَّلِينَ ﴿38﴾
یہ فیصلے کا دن ہے، ہم نے تمہیں اور پہلے لوگوں کو جمع کیا۔
فَإِن كَانَ لَكُمْ كَيۡدٞ فَكِيدُونِ ﴿39﴾
اگر تمہارے پاس کوئی تدبیر ہے تو میرے خلاف کر دیکھو۔
وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿40﴾
اس دن جھٹلانے والوں کے لیے ہلاکت ہے۔
إِنَّ ٱلۡمُتَّقِينَ فِي ظِلَٰلٖ وَعُيُونٖ ﴿41﴾
یقیناً پرہیزگار لوگ سایوں اور چشموں میں ہوں گے۔
وَفَوَٰكِهَ مِمَّا يَشۡتَهُونَ ﴿42﴾
اور ہر طرح کے پھل جنہیں وہ چاہیں گے۔
كُلُواْ وَٱشۡرَبُواْ هَنِيَٓٔۢا بِمَا كُنتُمۡ تَعۡمَلُونَ ﴿43﴾
کھاؤ اور پیو مزے سے اس کے بدلے جو تم عمل کرتے تھے۔
إِنَّا كَذَٰلِكَ نَجۡزِي ٱلۡمُحۡسِنِينَ ﴿44﴾
یقیناً ہم نیکی کرنے والوں کو اسی طرح بدلہ دیتے ہیں۔
وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿45﴾
اس دن جھٹلانے والوں کے لیے ہلاکت ہے۔
كُلُواْ وَتَمَتَّعُواْ قَلِيلًا إِنَّكُم مُّجْرِمُونَ ﴿46﴾
کھالو اور مزے کرلو تھوڑا سا، بے شک تم مجرم ہو۔
وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿47﴾
اس دن جھٹلانے والوں کے لیے ہلاکت ہے۔
وَإِذَا قِيلَ لَهُمُ ٱرْكَعُواْ لَا يَرْكَعُونَ ﴿48﴾
اور جب ان سے کہا جاتا ہے جھک جاؤ تو وہ نہیں جھکتے۔
وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ ﴿49﴾
اس دن جھٹلانے والوں کے لیے ہلاکت ہے۔
فَبِأَيِّ حَدِيثِۭ بَعْدَهُۥ يُؤْمِنُونَ ﴿50﴾
پھر اس کے بعد وہ کس بات پر ایمان لائیں گے؟
