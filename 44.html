سورة الدخان

بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
حمٓ ﴿1﴾
حٰمٓ
وَٱلْكِتَـٰبِ ٱلْمُبِينِ ﴿2﴾
قسم ہے واضح کتاب کی۔
إِنَّآ أَنزَلْنَـٰهُ فِي لَيْلَةٖ مُّبَـٰرَكَةٍ إِنَّا كُنَّا مُنذِرِينَ ﴿3﴾
بے شک ہم نے اسے ایک بابرکت رات میں نازل کیا، بے شک ہم ہی ڈرانے والے ہیں۔
فِيهَا يُفْرَقُ كُلُّ أَمْرٍ حَكِيمٍ ﴿4﴾
اسی رات میں ہر حکمت والے کام کا فیصلہ کیا جاتا ہے۔
أَمْرٗا مِّنْ عِندِنَآۚ إِنَّا كُنَّا مُرْسِلِينَ ﴿5﴾
ہمارے حکم سے، بے شک ہم ہی بھیجنے والے ہیں۔
رَحْمَةٗ مِّن رَّبِّكَۚ إِنَّهُۥ هُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ ﴿6﴾
یہ آپ کے رب کی رحمت ہے، بے شک وہی سننے والا، جاننے والا ہے۔
رَبِّ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَآۖ إِن كُنتُم مُّوقِنِينَ ﴿7﴾
آسمانوں اور زمین اور جو کچھ ان کے درمیان ہے، سب کا رب ہے، اگر تم یقین رکھتے ہو۔
لَآ إِلَـٰهَ إِلَّا هُوَۖ يُحْيِۦ وَيُمِيتُۖ رَبُّكُمْ وَرَبُّ ءَابَآئِكُمُ ٱلْأَوَّلِينَ ﴿8﴾
اس کے سوا کوئی معبود نہیں، وہی زندہ کرتا ہے اور مارتا ہے، وہی تمہارا رب اور تمہارے پہلے باپ دادا کا رب ہے۔
بَلْ هُمْ فِي شَكّٖ يَلْعَبُونَ ﴿9﴾
بلکہ وہ شک میں پڑے کھیل رہے ہیں۔
فَٱرْتَقِبْ يَوْمَ تَأْتِي ٱلسَّمَآءُ بِدُخَانٖ مُّبِينٖ ﴿10﴾
پس انتظار کرو اس دن کا جب آسمان واضح دھوئیں لے کر آئے گا۔
يَغْشَى ٱلنَّاسَ ۖ هَـٰذَا عَذَابٌ أَلِيمٌ ﴿11﴾
وہ لوگوں کو ڈھانپ لے گا، یہ دردناک عذاب ہے۔
رَبَّنَا ٱكْشِفْ عَنَّا ٱلْعَذَابَ إِنَّا مُؤْمِنُونَ ﴿12﴾
اے ہمارے رب! ہم سے عذاب دور کر دے، بے شک ہم ایمان لاتے ہیں۔
أَنَّىٰ لَهُمُ ٱلذِّكْرَىٰ وَقَدْ جَآءَهُمْ رَسُولٞ مُّبِينٞ ﴿13﴾
انہیں نصیحت کہاں فائدہ دے گی جبکہ ان کے پاس واضح رسول آ چکا تھا؟
ثُمَّ تَوَلَّوْا۟ عَنْهُ وَقَالُوا۟ مُعَلَّمٞ مَّجْنُونٌ ﴿14﴾
پھر انہوں نے اس سے منہ موڑ لیا اور کہا: یہ تو سکھایا ہوا دیوانہ ہے۔
إِنَّا كَاشِفُوا۟ ٱلْعَذَابِ قَلِيلًا إِنَّكُمْ عَآئِدُونَ ﴿15﴾
ہم تھوڑے وقت کے لیے عذاب کو دور کر دیتے ہیں، تم ضرور پھر لوٹ آؤ گے۔
يَوْمَ نَبْطِشُ ٱلْبَطْشَةَ ٱلْكُبْرَىٰٓ إِنَّا مُنتَقِمُونَ ﴿16﴾
جس دن ہم بڑی پکڑ کریں گے، ہم ضرور انتقام لینے والے ہیں۔
وَلَقَدْ فَتَنَّا قَبْلَهُمْ قَوْمَ فِرْعَوْنَ وَجَآءَهُمْ رَسُولٞ كَرِيمٌ ﴿17﴾
اور ہم نے ان سے پہلے فرعون کی قوم کو آزمائش میں ڈالا، اور ان کے پاس ایک معزز رسول آیا۔
أَنْ أَدُّوٓا۟ إِلَيَّ عِبَادَ ٱللَّهِ ۖ إِنِّي لَكُمْ رَسُولٌ أَمِينٞ ﴿18﴾
کہ اللہ کے بندوں کو میرے حوالے کر دو، بے شک میں تمہارے لیے امانت دار رسول ہوں۔
وَأَلَّا تَعْلُوا۟ عَلَى ٱللَّهِ ۖ إِنِّيٓ ءَاتِيكُم بِسُلْطَـٰنٖ مُّبِينٖ ﴿19﴾
اور اللہ کے مقابلے میں سرکشی نہ کرو، بے شک میں تمہارے پاس واضح دلیل کے ساتھ آیا ہوں۔
وَإِنِّي عُذْتُ بِرَبِّي وَرَبِّكُمۡ أَن تَرْجُمُونِ ﴿20﴾
اور میں اپنے رب اور تمہارے رب کی پناہ لیتا ہوں کہ تم مجھے سنگسار کرو۔
فَإِن لَّمْ تُؤْمِنُوا۟ لِي فَٱعْتَزِلُونِ ﴿21﴾
پس اگر تم مجھ پر ایمان نہیں لاتے تو مجھ سے الگ ہو جاؤ۔
فَدَعَا رَبَّهُۥٓ أَنَّ هَـٰٓؤُلَآءِ قَوْمٞ مُّجْرِمُونَ ﴿22﴾
پھر اس نے اپنے رب سے دعا کی: یہ لوگ مجرم ہیں۔
فَأَسْرِ بِعِبَادِي لَيْلًا إِنَّكُم مُّتَّبَعُونَ ﴿23﴾
تو میرے بندوں کو راتوں رات لے چلو، بے شک تمہارا پیچھا کیا جائے گا۔
وَٱتْرُكِ ٱلْبَحْرَ رَهْوًاۖ إِنَّهُمْ جُندٞ مُّغْرَقُونَ ﴿24﴾
اور سمندر کو ساکن چھوڑ دو، بے شک وہ لشکر غرق ہونے والا ہے۔
كَمۡ تَرَكُوا۟ مِن جَنَّـٰتٖ وَعُيُونٖ ﴿25﴾
کتنے ہی باغات اور چشمے وہ چھوڑ گئے۔
وَزُرُوعٖ وَمَقَامٖ كَرِيمٖ ﴿26﴾
اور کھیتیاں اور عزت والی رہائشیں۔
وَنَعْمَةٖ كَانُوا۟ فِيهَا فَـٰكِهِينَ ﴿27﴾
اور نعمتیں جن میں وہ مزے کرتے تھے۔
كَذَٰلِكَۖ وَأَوْرَثْنَـٰهَا قَوْمًا ءَاخَرِينَ ﴿28﴾
اسی طرح ہوا، اور ہم نے ان چیزوں کو دوسری قوم کو وارث بنا دیا۔
فَمَا بَكَتْ عَلَيْهِمُ ٱلسَّمَآءُ وَٱلْأَرْضُ وَمَا كَانُوا۟ مُنظَرِينَ ﴿29﴾
تو آسمان اور زمین ان پر روئے نہیں، اور نہ ہی انہیں مہلت دی گئی۔
وَلَقَدْ نَجَّيْنَا بَنِيٓ إِسْرَٰٓءِيلَ مِنَ ٱلْعَذَابِ ٱلْمُهِينِ ﴿30﴾
اور ہم نے بنی اسرائیل کو ذلت والے عذاب سے نجات دی۔
مِّن فِرْعَوْنَۚ إِنَّهُۥ كَانَ عَالِيٗا مِّنَ ٱلْمُسْرِفِينَ ﴿31﴾
فرعون سے، بے شک وہ سرکش لوگوں میں سے تھا۔
وَلَقَدِ ٱخْتَرْنَـٰهُمْ عَلَىٰ عِلْمٍ عَلَى ٱلْعَـٰلَمِينَ ﴿32﴾
اور ہم نے انہیں علم کے ساتھ جہان والوں پر چن لیا تھا۔
وَءَاتَيْنَـٰهُم مِّنَ ٱلْءَايَـٰتِ مَا فِيهِ بَلَـٰٓؤٞاۢ مُّبِينٞ ﴿33﴾
اور ہم نے انہیں ایسی نشانیاں دیں جن میں واضح آزمائش تھی۔
إِنَّ هَـٰٓؤُلَآءِ لَيَقُولُونَ ﴿34﴾
بے شک یہ لوگ کہتے ہیں۔
إِنْ هِيَ إِلَّا مَوْتَتُنَا ٱلْأُولَىٰ وَمَا نَحْنُ بِمُنشَرِينَ ﴿35﴾
یہ تو بس ہماری پہلی موت ہے، اور ہم دوبارہ نہیں اٹھائے جائیں گے۔
فَأْتُوا۟ بِـَٔابَآئِنَآ إِن كُنتُمْ صَـٰدِقِينَ ﴿36﴾
تو ہمارے باپ دادا کو لے آؤ اگر تم سچے ہو۔
أَهُمۡ خَيْرٌ أَمْ قَوْمُ تُبَّعٖ وَٱلَّذِينَ مِن قَبْلِهِمۡۚ أَهْلَكْنَـٰهُمْ إِنَّهُمۡ كَانُوا۟ مُجْرِمِينَ ﴿37﴾
کیا یہ بہتر ہیں یا قومِ تُبّع اور ان سے پہلے والے؟ ہم نے انہیں ہلاک کر دیا، بے شک وہ مجرم تھے۔
وَمَا خَلَقْنَا ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ وَمَا بَيْنَهُمَا لَـٰعِبِينَ ﴿38﴾
اور ہم نے آسمانوں اور زمین کو اور جو کچھ ان کے درمیان ہے کھیل کے طور پر نہیں بنایا۔
مَا خَلَقْنَـٰهُمَآ إِلَّا بِٱلْحَقِّ وَلَـٰكِنَّ أَكْثَرَهُمْ لَا يَعْلَمُونَ ﴿39﴾
ہم نے انہیں حق کے ساتھ پیدا کیا ہے، لیکن ان میں سے اکثر نہیں جانتے۔
إِنَّ يَوْمَ ٱلْفَصْلِ مِيقَـٰتُهُمْ أَجْمَعِينَ ﴿40﴾
بے شک فیصلے کا دن ان سب کا مقرر وقت ہے۔
يَوْمَ لَا يُغْنِي مَوْلٗى عَن مَّوْلٗى شَيْـٔٗا وَلَا هُمْ يُنصَرُونَ ﴿41﴾
جس دن کوئی دوست کسی دوست کے کام نہ آئے گا، اور نہ ہی ان کی مدد کی جائے گی۔
إِلَّا مَن رَّحِمَ ٱللَّهُۚ إِنَّهُۥ هُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ﴿42﴾
سوائے اس کے جس پر اللہ رحم کرے، بے شک وہی زبردست، رحم کرنے والا ہے۔
إِنَّ شَجَرَتَ ٱلزَّقُّومِ ﴿43﴾
بے شک زقوم کا درخت۔
طَعَامُ ٱلْأَثِيمِ ﴿44﴾
گناہگار کا کھانا ہے۔
كَٱلْمُهْلِ يَغْلِي فِي ٱلْبُطُونِ ﴿45﴾
پگھلے ہوئے تانبے کی طرح پیٹوں میں کھولتا ہے۔
كَغَلْيِ ٱلْحَمِيمِ ﴿46﴾
جیسے کھولتا ہوا پانی کھولتا ہے۔
خُذُوهُ فَٱعْتِلُوهُ إِلَىٰ سَوَآءِ ٱلْجَحِيمِ ﴿47﴾
پکڑو اسے اور گھسیٹتے ہوئے جہنم کے وسط تک لے جاؤ۔
ثُمَّ صُبُّوا۟ فَوْقَ رَأْسِهِۦ مِنْ عَذَابِ ٱلْحَمِيمِ ﴿48﴾
پھر اس کے سر پر کھولتے ہوئے پانی کا عذاب انڈیل دو۔
ذُقْ إِنَّكَ أَنتَ ٱلْعَزِيزُ ٱلْكَرِيمُ ﴿49﴾
چکھ! بے شک تو ہی عزت والا، بزرگی والا تھا۔
إِنَّ هَـٰذَا مَا كُنتُم بِهِۦ تَمْتَرُونَ ﴿50﴾
بے شک یہ وہی چیز ہے جس میں تم شک کرتے تھے۔
إِنَّ ٱلْمُتَّقِينَ فِي مَقَامٍ أَمِينٖ ﴿51﴾
بے شک پرہیزگار لوگ امن والی جگہ میں ہوں گے۔
فِي جَنَّـٰتٖ وَعُيُونٖ ﴿52﴾
باغوں اور چشموں میں۔
يَلْبَسُونَ مِن سُندُسٖ وَإِسْتَبْرَقٖ مُّتَقَـٰبِلِينَ ﴿53﴾
وہ باریک اور موٹے ریشم کے لباس پہنیں گے، آمنے سامنے بیٹھے ہوں گے۔
كَذَٰلِكَ وَزَوَّجْنَـٰهُم بِحُورٍ عِينٖ ﴿54﴾
اسی طرح ہوگا، اور ہم انہیں بڑی آنکھوں والی حوروں سے جوڑ دیں گے۔
يَدْعُونَ فِيهَا بِكُلِّ فَـٰكِهَةٍ ءَامِنِينَ ﴿55﴾
وہ وہاں ہر قسم کے پھل طلب کریں گے، امن کے ساتھ۔
لَا يَذُوقُونَ فِيهَا ٱلْمَوْتَ إِلَّا ٱلْمَوْتَةَ ٱلْأُولَىٰۖ وَوَقَىٰهُمْ عَذَابَ ٱلْجَحِيمِ ﴿56﴾
وہ وہاں موت کا ذائقہ نہیں چکھیں گے، سوائے پہلی موت کے، اور اللہ انہیں جہنم کے عذاب سے بچا لے گا۔
فَضْلٗا مِّن رَّبِّكَۚ ذَٰلِكَ هُوَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿57﴾
یہ آپ کے رب کا فضل ہے، یہی بڑی کامیابی ہے۔
فَإِنَّمَا يَسَّرْنَـٰهُ بِلِسَانِكَ لَعَلَّهُمْ يَتَذَكَّرُونَ ﴿58﴾
پس ہم نے اسے آپ کی زبان میں آسان کر دیا تاکہ وہ نصیحت حاصل کریں۔
فَٱرْتَقِبْ إِنَّهُم مُّرْتَقِبُونَ ﴿59﴾
پس انتظار کرو، بے شک وہ بھی انتظار کر رہے ہیں۔
