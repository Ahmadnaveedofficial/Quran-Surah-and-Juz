سورة إبراهيم

بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
الٓرۚ كِتَـٰبٌ أَنزَلْنَـٰهُ إِلَيْكَ لِتُخْرِجَ ٱلنَّاسَ مِنَ ٱلظُّلُمَـٰتِ إِلَى ٱلنُّورِ بِإِذْنِ رَبِّهِمْ إِلَىٰ صِرَٰطِ ٱلْعَزِيزِ ٱلْحَمِيدِ ﴿1﴾
الٓر۔ یہ ایک کتاب ہے جسے ہم نے تمہاری طرف نازل کیا تاکہ تم لوگوں کو ان کے رب کے حکم سے اندھیروں سے نکال کر روشنی کی طرف لے آؤ، اس زبردست، قابلِ تعریف ذات کے راستے کی طرف۔
ٱللَّهِ ٱلَّذِي لَهُۥ مَا فِي ٱلسَّمَـٰوَٰتِ وَمَا فِي ٱلْأَرْضِۗ وَوَيْلٞ لِّلْكَـٰفِرِينَ مِنْ عَذَابٖ شَدِيدٍ ﴿2﴾
اللہ ہی کا ہے جو کچھ آسمانوں اور زمین میں ہے، اور کافروں کے لیے سخت عذاب ہے۔
ٱلَّذِينَ يَسْتَحِبُّونَ ٱلْحَيَوٰةَ ٱلدُّنْيَا عَلَى ٱلْـَٔاخِرَةِ وَيَصُدُّونَ عَن سَبِيلِ ٱللَّهِ وَيَبْغُونَهَا عِوَجًاۚ أُوْلَـٰٓئِكَ فِي ضَلَـٰلِۢ بَعِيدٖ ﴿3﴾
جو دنیا کی زندگی کو آخرت پر ترجیح دیتے ہیں، اور اللہ کے راستے سے روکتے ہیں اور اس میں کجی چاہتے ہیں، یہی لوگ دور کی گمراہی میں ہیں۔
وَمَآ أَرْسَلْنَا مِن رَّسُولٍ إِلَّا بِلِسَانِ قَوْمِهِۦ لِيُبَيِّنَ لَهُمْۖ فَيُضِلُّ ٱللَّهُ مَن يَشَآءُ وَيَهْدِي مَن يَشَآءُۚ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿4﴾
اور ہم نے کوئی رسول نہیں بھیجا مگر اپنی قوم کی زبان میں تاکہ وہ ان کے لیے واضح کرے، پھر اللہ جسے چاہتا ہے گمراہ کرتا ہے اور جسے چاہتا ہے ہدایت دیتا ہے، اور وہ زبردست، حکمت والا ہے۔
وَلَقَدْ أَرْسَلْنَا مُوسَىٰ بِـَٔايَـٰتِنَآ أَنْ أَخْرِجْ قَوْمَكَ مِنَ ٱلظُّلُمَـٰتِ إِلَى ٱلنُّورِ وَذَكِّرْهُم بِأَيَّامِ ٱللَّهِۚ إِنَّ فِي ذَٰلِكَ لَـَٔايَـٰتٖ لِّكُلِّ صَبَّارٖ شَكُورٖ ﴿5﴾
اور بے شک ہم نے موسیٰ کو اپنی نشانیاں دے کر بھیجا کہ اپنی قوم کو اندھیروں سے نکال کر روشنی کی طرف لے آؤ، اور انہیں اللہ کے دن یاد دلاؤ، بے شک اس میں ہر صبر کرنے والے، شکر گزار کے لیے نشانیاں ہیں۔
وَإِذْ قَالَ مُوسَىٰ لِقَوْمِهِ ٱذْكُرُوا۟ نِعْمَتَ ٱللَّهِ عَلَيْكُمْ إِذْ أَنجَىٰكُم مِّنْ ءَالِ فِرْعَوْنَ يَسُومُونَكُمْ سُوٓءَ ٱلْعَذَابِ وَيُذَبِّحُونَ أَبْنَآءَكُمْ وَيَسْتَحْيُونَ نِسَآءَكُمْۚ وَفِي ذَٰلِكُم بَلَآءٞ مِّن رَّبِّكُمْ عَظِيمٞ ﴿6﴾
اور جب موسیٰ نے اپنی قوم سے کہا: اللہ کی نعمت کو یاد کرو جو اس نے تم پر کی، جب اس نے تمہیں فرعون کی قوم سے نجات دی، وہ تمہیں سخت عذاب دیتے تھے، تمہارے بیٹوں کو ذبح کرتے تھے اور تمہاری عورتوں کو زندہ رکھتے تھے، اور اس میں تمہارے رب کی طرف سے بڑی آزمائش تھی۔
وَإِذْ تَأَذَّنَ رَبُّكُمْ لَئِن شَكَرْتُمْ لَأَزِيدَنَّكُمْ وَلَئِن كَفَرْتُمْ إِنَّ عَذَابِي لَشَدِيدٞ ﴿7﴾
اور جب تمہارے رب نے اعلان کیا: اگر تم شکر کرو گے تو میں تمہیں ضرور زیادہ دوں گا، اور اگر تم ناشکری کرو گے تو بے شک میرا عذاب سخت ہے۔
وَقَالَ مُوسَىٰٓ إِن تَكْفُرُوا۟ أَنتُمْ وَمَن فِي ٱلْأَرْضِ جَمِيعٗا فَإِنَّ ٱللَّهَ لَغَنِيٌّ حَمِيدٌ ﴿8﴾
اور موسیٰ نے کہا: اگر تم اور زمین میں جو بھی ہیں سب کے سب کفر کرو تو بے شک اللہ بے نیاز، قابلِ تعریف ہے۔
أَلَمْ يَأْتِكُمْ نَبَؤُا۟ ٱلَّذِينَ مِن قَبْلِكُمْ قَوْمِ نُوحٖ وَعَادٖ وَثَمُودَۚ وَٱلَّذِينَ مِنۢ بَعْدِهِمْۛ لَا يَعْلَمُهُمْ إِلَّا ٱللَّهُۚ جَآءَتْهُمْ رُسُلُهُم بِٱلْبَيِّنَـٰتِ فَرَدُّوٓا۟ أَيْدِيَهُمْ فِيٓ أَفْوَٰهِهِمْ وَقَالُوٓا۟ إِنَّا كَفَرْنَا بِمَآ أُرْسِلْتُم بِهِۦ وَإِنَّا لَفِي شَكّٖ مِّمَّا تَدْعُونَنَآ إِلَيْهِ مُرِيبٖ ﴿9﴾
کیا تمہارے پاس ان لوگوں کی خبر نہیں آئی جو تم سے پہلے تھے — نوح کی قوم، عاد، ثمود، اور ان کے بعد کے لوگ جنہیں اللہ کے سوا کوئی نہیں جانتا؟ ان کے پاس ان کے رسول واضح دلائل لے کر آئے، تو انہوں نے اپنے ہاتھ اپنے منہ پر رکھ لیے اور کہا: ہم اس چیز کے منکر ہیں جس کے ساتھ تم بھیجے گئے ہو، اور جس چیز کی طرف تم ہمیں بلا رہے ہو، ہم اس میں سخت شک میں ہیں۔
قَالَتْ رُسُلُهُمْ أَفِي ٱللَّهِ شَكّٞ فَاطِرِ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۖ يَدْعُوكُمْ لِيَغْفِرَ لَكُم مِّن ذُنُوبِكُمْ وَيُؤَخِّرَكُمْ إِلَىٰٓ أَجَلٖ مُّسَمًّىۚ قَالُوٓا۟ إِنْ أَنتُمْ إِلَّا بَشَرٞ مِّثْلُنَا تُرِيدُونَ أَن تَصُدُّونَا عَمَّا كَانَ يَعْبُدُ ءَابَآؤُنَا فَأْتُونَا بِسُلْطَـٰنٖ مُّبِينٖ ﴿10﴾
ان کے رسولوں نے کہا: کیا اللہ میں شک ہے؟ جو آسمانوں اور زمین کا پیدا کرنے والا ہے؟ وہ تمہیں بلاتا ہے تاکہ تمہارے گناہ بخش دے اور تمہیں ایک مقرر وقت تک مہلت دے۔ انہوں نے کہا: تم تو ہمارے جیسے انسان ہو، تم چاہتے ہو کہ ہمیں اس سے روک دو جس کی ہمارے باپ دادا عبادت کرتے تھے، پس ہمارے پاس کوئی واضح دلیل لے آؤ۔
قَالَتْ لَهُمْ رُسُلُهُمْ إِن نَّحْنُ إِلَّا بَشَرٞ مِّثْلُكُمْ وَلَـٰكِنَّ ٱللَّهَ يَمُنُّ عَلَىٰ مَن يَشَآءُ مِنْ عِبَادِهِۦۖ وَمَا كَانَ لَنَآ أَن نَّأْتِيَكُم بِسُلْطَـٰنٍ إِلَّا بِإِذْنِ ٱللَّهِۚ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُؤْمِنُونَ ﴿11﴾
ان کے رسولوں نے کہا: بے شک ہم تمہارے جیسے انسان ہی ہیں، لیکن اللہ اپنے بندوں میں سے جس پر چاہتا ہے فضل کرتا ہے، اور ہمارے لیے ممکن نہیں کہ ہم تمہارے پاس کوئی دلیل لائیں مگر اللہ کے حکم سے، اور ایمان والوں کو اللہ ہی پر بھروسہ کرنا چاہیے۔
وَمَا لَنَآ أَلَّا نَتَوَكَّلَ عَلَى ٱللَّهِ وَقَدْ هَدَىٰنَا سُبُلَنَاۚ وَلَنَصْبِرَنَّ عَلَىٰ مَآ ءَاذَيْتُمُونَاۚ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُتَوَكِّلُونَ ﴿12﴾
اور ہمیں کیا ہوا ہے کہ ہم اللہ پر بھروسہ نہ کریں جبکہ اس نے ہمیں ہمارے راستے دکھا دیے ہیں؟ اور ہم ضرور صبر کریں گے اس اذیت پر جو تم ہمیں دیتے ہو، اور بھروسہ کرنے والوں کو اللہ ہی پر بھروسہ کرنا چاہیے۔
وَقَالَ ٱلَّذِينَ كَفَرُوا۟ لِرُسُلِهِمْ لَنُخْرِجَنَّكُم مِّنْ أَرْضِنَآ أَوْ لَتَعُودُنَّ فِي مِلَّتِنَاۖ فَأَوْحَىٰٓ إِلَيْهِمْ رَبُّهُمْ لَنُهْلِكَنَّ ٱلظَّـٰلِمِينَ ﴿13﴾
اور کافروں نے اپنے رسولوں سے کہا: ہم تمہیں اپنی زمین سے ضرور نکال دیں گے یا تم ضرور ہمارے دین میں واپس آ جاؤ گے، تو ان کے رب نے ان کی طرف وحی کی: ہم ضرور ظالموں کو ہلاک کریں گے۔
وَلَنُسْكِنَنَّكُمُ ٱلْأَرْضَ مِنۢ بَعْدِهِمۡۚ ذَٰلِكَ لِمَنْ خَافَ مَقَامِي وَخَافَ وَعِيدِ ﴿14﴾
اور ہم تمہیں ان کے بعد زمین میں ضرور بسائیں گے، یہ اس کے لیے ہے جو میرے مقام سے ڈرے اور میری وعید سے ڈرے۔
وَٱسْتَفْتَحُوا۟ وَخَابَ كُلُّ جَبَّارٍ عَنِيدٍ ﴿15﴾
اور انہوں نے فیصلہ چاہا، تو ہر سرکش ضدی ناکام ہوا۔
مِّن وَرَآئِهِۦ جَهَنَّمُ وَيُسْقَىٰ مِن مَّآءٍ صَدِيدٖ ﴿16﴾
اس کے پیچھے جہنم ہے، اور اسے پیپ کا پانی پلایا جائے گا۔
يَتَجَرَّعُهُۥ وَلَا يَكَادُ يُسِيغُهُۥ وَيَأْتِيهِ ٱلْمَوْتُ مِن كُلِّ مَكَانٖ وَمَا هُوَ بِمَيِّتٖۖ وَمِن وَرَآئِهِۦ عَذَابٌ غَلِيظٞ ﴿17﴾
وہ اسے گھونٹ گھونٹ کر پئے گا اور اسے آسانی سے نگل نہ سکے گا، اور اسے ہر طرف سے موت آئے گی مگر وہ مرے گا نہیں، اور اس کے پیچھے سخت عذاب ہے۔
مَّثَلُ ٱلَّذِينَ كَفَرُوا۟ بِرَبِّهِمْ أَعْمَـٰلُهُمْ كَرَمَادٍ ٱشْتَدَّتْ بِهِ ٱلرِّيحُ فِي يَوْمٍ عَاصِفٖۖ لَّا يَقْدِرُونَ مِمَّا كَسَبُوا۟ عَلَىٰ شَيْءٖۚ ذَٰلِكَ هُوَ ٱلضَّلَـٰلُ ٱلْبَعِيدُ ﴿18﴾
جن لوگوں نے اپنے رب سے کفر کیا، ان کے اعمال کی مثال اس راکھ کی طرح ہے جسے آندھی والے دن تیز ہوا اڑا لے جائے، وہ اپنی کمائی میں سے کچھ بھی حاصل نہ کر سکیں گے، یہی دور کی گمراہی ہے۔
أَلَمْ تَرَ أَنَّ ٱللَّهَ خَلَقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ بِٱلْحَقِّۚ إِن يَشَأْ يُذْهِبْكُمْ وَيَأْتِ بِخَلْقٍ جَدِيدٖ ﴿19﴾
کیا تم نے نہیں دیکھا کہ اللہ نے آسمانوں اور زمین کو حق کے ساتھ پیدا کیا؟ اگر وہ چاہے تو تمہیں لے جائے اور نئی مخلوق لے آئے۔
وَمَا ذَٰلِكَ عَلَى ٱللَّهِ بِعَزِيزٖ ﴿20﴾
اور یہ اللہ کے لیے کچھ بھی مشکل نہیں۔
وَبَرَزُوا۟ لِلَّهِ جَمِيعٗا فَقَالَ ٱلضُّعَفَـٰٓؤُا۟ لِلَّذِينَ ٱسْتَكْبَرُوا۟ إِنَّا كُنَّا لَكُمْ تَبَعٗا فَهَلْ أَنتُم مُّغْنُونَ عَنَّا مِنْ عَذَابِ ٱللَّهِ مِن شَيْءٖۚ قَالُوا۟ لَوْ هَدَىٰنَا ٱللَّهُ لَهَدَيْنَـٰكُمۚ سَوَآءٌ عَلَيْنَآ أَجَزِعْنَآ أَمْ صَبَرْنَا مَا لَنَا مِن مَّحِيصٖ ﴿21﴾
اور سب اللہ کے سامنے ظاہر ہوں گے، تو کمزور لوگ ان سے کہیں گے جو تکبر کرتے تھے: ہم تمہارے تابع تھے، تو کیا تم اللہ کے عذاب سے کچھ بچا سکتے ہو؟ وہ کہیں گے: اگر اللہ ہمیں ہدایت دیتا تو ہم تمہیں بھی ہدایت دیتے، اب ہمارے لیے برابر ہے کہ ہم جزع کریں یا صبر کریں، ہمارے لیے کوئی بچاؤ نہیں۔
وَقَالَ ٱلشَّيْطَـٰنُ لَمَّا قُضِيَ ٱلْأَمْرُ إِنَّ ٱللَّهَ وَعَدَكُمْ وَعْدَ ٱلْحَقِّ وَوَعَدتُّكُمْ فَأَخْلَفْتُكُمْۖ وَمَا كَانَ لِيَ عَلَيْكُم مِّن سُلْطَـٰنٍ إِلَّآ أَن دَعَوْتُكُمْ فَٱسْتَجَبْتُمْ لِيۖ فَلَا تَلُومُونِي وَلُومُوٓا۟ أَنفُسَكُمۖ مَّآ أَنَا۠ بِمُصْرِخِكُمْ وَمَآ أَنتُم بِمُصْرِخِيَّۖ إِنِّي كَفَرْتُ بِمَآ أَشْرَكْتُمُونِ مِن قَبْلُۗ إِنَّ ٱلظَّـٰلِمِينَ لَهُمْ عَذَابٌ أَلِيمٞ ﴿22﴾
اور جب فیصلہ ہو چکے گا تو شیطان کہے گا: بے شک اللہ نے تم سے سچا وعدہ کیا تھا، اور میں نے بھی وعدہ کیا تھا، تو میں نے تم سے وعدہ خلافی کی، اور مجھے تم پر کوئی زور حاصل نہ تھا، سوائے اس کے کہ میں نے تمہیں بلایا اور تم نے میری بات مان لی، پس مجھے ملامت نہ کرو بلکہ اپنے آپ کو ملامت کرو، نہ میں تمہاری فریاد کو پہنچ سکتا ہوں اور نہ تم میری فریاد کو، میں اس سے بیزار ہوں جو تم نے پہلے مجھے شریک بنایا تھا، بے شک ظالموں کے لیے دردناک عذاب ہے۔
وَأُدْخِلَ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ جَنَّـٰتٖ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُ خَـٰلِدِينَ فِيهَا بِإِذْنِ رَبِّهِمۖ تَحِيَّتُهُمْ فِيهَا سَلَـٰمٌ ﴿23﴾
اور جو لوگ ایمان لائے اور نیک عمل کیے، وہ باغوں میں داخل کیے جائیں گے جن کے نیچے نہریں بہتی ہیں، وہ اس میں ہمیشہ رہیں گے اپنے رب کے حکم سے، ان کی ملاقات کا کلمہ اس میں "سلام" ہوگا۔
أَلَمْ تَرَ كَيْفَ ضَرَبَ ٱللَّهُ مَثَلٗا كَلِمَةٗ طَيِّبَةٗ كَشَجَرَةٖ طَيِّبَةٍ أَصْلُهَا ثَابِتٞ وَفَرْعُهَا فِي ٱلسَّمَآءِ ﴿24﴾
کیا تم نے نہیں دیکھا کہ اللہ نے پاکیزہ بات کی مثال ایک پاکیزہ درخت کی دی ہے، جس کی جڑ مضبوط ہے اور اس کی شاخ آسمان میں ہے۔
تُؤْتِيٓ أُكُلَهَا كُلَّ حِينِۢ بِإِذْنِ رَبِّهَاۗ وَيَضْرِبُ ٱللَّهُ ٱلْأَمْثَالَ لِلنَّاسِ لَعَلَّهُمْ يَتَذَكَّرُونَ ﴿25﴾
وہ اپنے رب کے حکم سے ہر وقت پھل دیتی ہے، اور اللہ لوگوں کے لیے مثالیں بیان کرتا ہے تاکہ وہ نصیحت حاصل کریں۔
وَمَثَلُ كَلِمَةٍ خَبِيثَةٖ كَشَجَرَةٍ خَبِيثَةٍ ٱجْتُثَّتْ مِن فَوْقِ ٱلْأَرْضِ مَا لَهَا مِن قَرَارٍ ﴿26﴾
اور ناپاک بات کی مثال ایک ناپاک درخت کی ہے جو زمین کے اوپر سے اکھاڑ دیا گیا ہو، اس کے لیے کوئی ٹھہراؤ نہیں۔
يُثَبِّتُ ٱللَّهُ ٱلَّذِينَ ءَامَنُوا۟ بِٱلْقَوْلِ ٱلثَّابِتِ فِي ٱلْحَيَوٰةِ ٱلدُّنْيَا وَفِي ٱلْـَٔاخِرَةِۖ وَيُضِلُّ ٱللَّهُ ٱلظَّـٰلِمِينَۚ وَيَفْعَلُ ٱللَّهُ مَا يَشَآءُ ﴿27﴾
اللہ ایمان والوں کو دنیا کی زندگی میں اور آخرت میں ثابت بات پر قائم رکھتا ہے، اور اللہ ظالموں کو گمراہ کرتا ہے، اور اللہ جو چاہتا ہے کرتا ہے۔
أَلَمْ تَرَ إِلَى ٱلَّذِينَ بَدَّلُوا۟ نِعْمَتَ ٱللَّهِ كُفْرٗا وَأَحَلُّوا۟ قَوْمَهُمْ دَارَ ٱلْبَوَارِ ﴿28﴾
کیا تم نے ان لوگوں کو نہیں دیکھا جنہوں نے اللہ کی نعمت کو ناشکری سے بدل دیا اور اپنی قوم کو ہلاکت کے گھر میں لا اتارا؟
جَهَنَّمَ يَصْلَوْنَهَاۖ وَبِئْسَ ٱلْقَرَارُ ﴿29﴾
جہنم، جس میں وہ داخل ہوں گے، اور وہ بہت ہی برا ٹھکانا ہے۔
وَجَعَلُوا۟ لِلَّهِ أَندَادٗا لِّيُضِلُّوا۟ عَن سَبِيلِهِۦۗ قُلْ تَمَتَّعُوا۟ فَإِنَّ مَصِيرَكُمْ إِلَى ٱلنَّارِ ﴿30﴾
اور انہوں نے اللہ کے لیے شریک بنائے تاکہ اس کے راستے سے گمراہ کریں، کہہ دو: فائدہ اٹھا لو، بے شک تمہارا انجام آگ کی طرف ہے۔
قُل لِّعِبَادِيَ ٱلَّذِينَ ءَامَنُوا۟ يُقِيمُوا۟ ٱلصَّلَوٰةَ وَيُنفِقُوا۟ مِمَّا رَزَقْنَـٰهُمْ سِرّٗا وَعَلَانِيَةٗ مِّن قَبْلِ أَن يَأْتِيَ يَوْمٞ لَّا بَيْعٞ فِيهِ وَلَا خِلَـٰلٌ ﴿31﴾
کہہ دو: میرے بندوں سے جو ایمان لائے ہیں کہ نماز قائم کریں اور ہم نے انہیں جو رزق دیا ہے اس میں سے چھپ کر اور ظاہر طور پر خرچ کریں، اس دن کے آنے سے پہلے جس میں نہ خرید و فروخت ہوگی اور نہ دوستی۔
ٱللَّهُ ٱلَّذِي خَلَقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ وَأَنزَلَ مِنَ ٱلسَّمَآءِ مَآءٗ فَأَخْرَجَ بِهِۦ مِنَ ٱلثَّمَرَٰتِ رِزْقٗا لَّكُمْۖ وَسَخَّرَ لَكُمُ ٱلْفُلْكَ لِتَجْرِيَ فِي ٱلْبَحْرِ بِأَمْرِهِۦۖ وَسَخَّرَ لَكُمُ ٱلْأَنْهَـٰرَ ﴿32﴾
اللہ وہ ہے جس نے آسمانوں اور زمین کو پیدا کیا، اور آسمان سے پانی نازل کیا، پھر اس سے تمہارے لیے پھلوں کو رزق بنایا، اور تمہارے لیے کشتی کو مسخر کیا تاکہ وہ اس کے حکم سے سمندر میں چلے، اور تمہارے لیے نہریں مسخر کیں۔
وَسَخَّرَ لَكُمُ ٱلشَّمْسَ وَٱلْقَمَرَ دَآئِبَيْنِۖ وَسَخَّرَ لَكُمُ ٱلَّيْلَ وَٱلنَّهَارَ ﴿33﴾
اور تمہارے لیے سورج اور چاند کو مسخر کیا جو مسلسل چل رہے ہیں، اور تمہارے لیے رات اور دن کو مسخر کیا۔
وَءَاتَىٰكُم مِّن كُلِّ مَا سَأَلْتُمُوهُۚ وَإِن تَعُدُّوا۟ نِعْمَتَ ٱللَّهِ لَا تُحْصُوهَآۗ إِنَّ ٱلْإِنسَـٰنَ لَظَلُومٞ كَفَّارٞ ﴿34﴾
اور اس نے تمہیں ہر وہ چیز دی جو تم نے مانگی، اور اگر تم اللہ کی نعمتوں کو شمار کرنا چاہو تو انہیں گن نہیں سکتے، بے شک انسان بہت ہی ظالم، ناشکرا ہے۔
وَإِذْ قَالَ إِبْرَٰهِيمُ رَبِّ ٱجْعَلْ هَـٰذَا ٱلْبَلَدَ ءَامِنٗا وَٱجْنُبْنِي وَبَنِيَّ أَن نَّعْبُدَ ٱلْأَصْنَامَ ﴿35﴾
اور جب ابراہیم نے کہا: اے میرے رب! اس شہر کو امن والا بنا دے، اور مجھے اور میری اولاد کو بتوں کی عبادت سے بچا۔
رَبِّ إِنَّهُنَّ أَضْلَلْنَ كَثِيرٗا مِّنَ ٱلنَّاسِۖ فَمَن تَبِعَنِي فَإِنَّهُۥ مِنِّيۖ وَمَنْ عَصَانِي فَإِنَّكَ غَفُورٞ رَّحِيمٞ ﴿36﴾
اے میرے رب! بے شک انہوں نے بہت سے لوگوں کو گمراہ کیا، پس جو میرے پیچھے چلے وہ مجھ سے ہے، اور جس نے میری نافرمانی کی تو بے شک تو بخشنے والا، رحم کرنے والا ہے۔
رَبَّنَآ إِنِّيٓ أَسْكَنْتُ مِن ذُرِّيَّتِي بِوَادٍ غَيْرِ ذِي زَرْعٍ عِندَ بَيْتِكَ ٱلْمُحَرَّمِ رَبَّنَا لِيُقِيمُوا۟ ٱلصَّلَوٰةَ فَٱجْعَلْ أَفْـِٔدَةٗ مِّنَ ٱلنَّاسِ تَهْوِيٓ إِلَيْهِمْ وَٱرْزُقْهُم مِّنَ ٱلثَّمَرَٰتِ لَعَلَّهُمْ يَشْكُرُونَ ﴿37﴾
اے ہمارے رب! میں نے اپنی اولاد کو تیرے محترم گھر کے پاس ایک بے زراعت وادی میں بسایا ہے، اے ہمارے رب! تاکہ وہ نماز قائم کریں، پس لوگوں کے دلوں کو ان کی طرف مائل کر دے، اور انہیں پھلوں سے رزق دے تاکہ وہ شکر گزار بنیں۔
رَبَّنَآ إِنَّكَ تَعْلَمُ مَا نُخْفِي وَمَا نُعْلِنُۖ وَمَا يَخْفَىٰ عَلَى ٱللَّهِ مِن شَيْءٖ فِي ٱلْأَرْضِ وَلَا فِي ٱلسَّمَآءِ ﴿38﴾
اے ہمارے رب! بے شک تو جانتا ہے جو ہم چھپاتے ہیں اور جو ظاہر کرتے ہیں، اور اللہ سے کوئی چیز چھپی نہیں ہے، نہ زمین میں اور نہ آسمان میں۔
ٱلْحَمْدُ لِلَّهِ ٱلَّذِي وَهَبَ لِي عَلَى ٱلْكِبَرِ إِسْمَـٰعِيلَ وَإِسْحَـٰقَۚ إِنَّ رَبِّي لَسَمِيعُ ٱلدُّعَآءِ ﴿39﴾
سب تعریف اللہ کے لیے ہے جس نے مجھے بڑھاپے میں اسماعیل اور اسحاق عطا کیے، بے شک میرا رب دعا سننے والا ہے۔
رَبِّ ٱجْعَلْنِي مُقِيمَ ٱلصَّلَوٰةِ وَمِن ذُرِّيَّتِيۚ رَبَّنَا وَتَقَبَّلْ دُعَآءِ ﴿40﴾
اے میرے رب! مجھے نماز قائم کرنے والا بنا اور میری اولاد میں سے بھی، اے ہمارے رب! اور میری دعا قبول فرما۔
رَبَّنَا ٱغْفِرْ لِي وَلِوَٰلِدَيَّ وَلِلْمُؤْمِنِينَ يَوْمَ يَقُومُ ٱلْحِسَابُ ﴿41﴾
اے ہمارے رب! مجھے، میرے والدین کو اور سب ایمان والوں کو بخش دے جس دن حساب قائم ہوگا۔
وَلَا تَحْسَبَنَّ ٱللَّهَ غَـٰفِلًا عَمَّا يَعْمَلُ ٱلظَّـٰلِمُونَۚ إِنَّمَا يُؤَخِّرُهُمْ لِيَوْمٖ تَشْخَصُ فِيهِ ٱلْأَبْصَـٰرُ ﴿42﴾
اور تم اللہ کو ان ظالموں کے اعمال سے غافل نہ سمجھو، وہ تو انہیں اس دن تک مہلت دے رہا ہے جس دن آنکھیں پھٹی کی پھٹی رہ جائیں گی۔
مُهْطِعِينَ مُقْنِعِي رُءُوسِهِمْ لَا يَرْتَدُّ إِلَيْهِمْ طَرْفُهُمْ وَأَفْـِٔدَتُهُمْ هَوَآءٞ ﴿43﴾
وہ سر اٹھائے دوڑتے ہوں گے، ان کی نگاہیں ان کی طرف لوٹ کر نہ آئیں گی، اور ان کے دل خالی ہوں گے۔
وَأَنذِرِ ٱلنَّاسَ يَوْمَ يَأْتِيهِمُ ٱلْعَذَابُ فَيَقُولُ ٱلَّذِينَ ظَلَمُوا۟ رَبَّنَآ أَخِّرْنَآ إِلَىٰٓ أَجَلٖ قَرِيبٖ نُّجِبْ دَعْوَتَكَ وَنَتَّبِعِ ٱلرُّسُلَۚ أَوَ لَمْ تَكُونُوا۟ أَقْسَمْتُم مِّن قَبْلُ مَا لَكُم مِّن زَوَالٖ ﴿44﴾
اور لوگوں کو اس دن سے ڈراؤ جب ان کے پاس عذاب آئے گا، تو ظالم کہیں گے: اے ہمارے رب! ہمیں تھوڑی مہلت دے دے، ہم تیری دعوت قبول کریں گے اور رسولوں کی پیروی کریں گے۔ کیا تم نے پہلے قسم نہیں کھائی تھی کہ تمہیں زوال نہیں ہوگا؟
وَسَكَنتُمْ فِي مَسَـٰكِنِ ٱلَّذِينَ ظَلَمُوٓا۟ أَنفُسَهُمْ وَتَبَيَّنَ لَكُمْ كَيْفَ فَعَلْنَا بِهِمْ وَضَرَبْنَا لَكُمُ ٱلْأَمْثَالَ ﴿45﴾
اور تم ان لوگوں کے گھروں میں بسے جنہوں نے اپنی جانوں پر ظلم کیا، اور تمہیں واضح ہو گیا کہ ہم نے ان کے ساتھ کیا کیا، اور ہم نے تمہارے لیے مثالیں بیان کیں۔
وَقَدْ مَكَرُوا۟ مَكْرَهُمْ وَعِندَ ٱللَّهِ مَكْرُهُمْۖ وَإِن كَانَ مَكْرُهُمْ لِتَزُولَ مِنْهُ ٱلْجِبَالُ ﴿46﴾
اور انہوں نے اپنی چالیں چلیں، اور اللہ کے پاس ان کی چالیں ہیں، اور اگر ان کی چالیں ایسی ہوں کہ پہاڑ بھی ان سے ہٹ جائیں۔
فَلَا تَحْسَبَنَّ ٱللَّهَ مُخْلِفَ وَعْدِهِۦ رُسُلَهُۥٓۚ إِنَّ ٱللَّهَ عَزِيزٞ ذُو ٱنتِقَامٖ ﴿47﴾
پس تم اللہ کو اس کے رسولوں سے کیے گئے وعدے کا خلاف کرنے والا نہ سمجھو، بے شک اللہ زبردست، انتقام لینے والا ہے۔
يَوْمَ تُبَدَّلُ ٱلْأَرْضُ غَيْرَ ٱلْأَرْضِ وَٱلسَّمَـٰوَٰتُ وَبَرَزُوا۟ لِلَّهِ ٱلْوَٰحِدِ ٱلْقَهَّارِ ﴿48﴾
جس دن زمین دوسری زمین سے بدل دی جائے گی اور آسمان بھی، اور سب اللہ کے سامنے حاضر ہوں گے، جو واحد، زبردست ہے۔
وَتَرَى ٱلْمُجْرِمِينَ يَوْمَئِذٖ مُّقَرَّنِينَ فِي ٱلْأَصْفَادِ ﴿49﴾
اور تم اس دن مجرموں کو زنجیروں میں جکڑے ہوئے دیکھو گے۔
سَرَابِيلُهُم مِّن قَطِرَانٖ وَتَغْشَىٰ وُجُوهَهُمُ ٱلنَّارُ ﴿50﴾
ان کے لباس تارکول کے ہوں گے، اور ان کے چہروں کو آگ ڈھانپ لے گی۔
لِيَجْزِيَ ٱللَّهُ كُلَّ نَفْسٖ مَّا كَسَبَتۚ إِنَّ ٱللَّهَ سَرِيعُ ٱلْحِسَابِ ﴿51﴾
تاکہ اللہ ہر جان کو اس کے کیے کا بدلہ دے، بے شک اللہ حساب لینے میں بہت تیز ہے۔
هَـٰذَا بَلَـٰغٞ لِّلنَّاسِ وَلِيُنذَرُوا۟ بِهِۦ وَلِيَعْلَمُوٓا۟ أَنَّمَا هُوَ إِلَـٰهٞ وَٰحِدٞ وَلِيَذَّكَّرَ أُوْلُوا۟ ٱلْأَلْبَـٰبِ ﴿52﴾
یہ لوگوں کے لیے ایک پیغام ہے تاکہ وہ اس کے ذریعے ڈرائے جائیں، اور جان لیں کہ وہی ایک معبود ہے، اور عقل والے نصیحت حاصل کریں۔
