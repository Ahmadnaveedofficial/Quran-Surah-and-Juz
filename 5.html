سورة المائدة

بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ أَوْفُوا۟ بِٱلْعُقُودِۚ أُحِلَّتْ لَكُم بَهِيمَةُ ٱلْأَنْعَـٰمِ إِلَّا مَا يُتْلَىٰ عَلَيْكُمْ غَيْرَ مُحِلِّي ٱلصَّيْدِ وَأَنتُمْ حُرُمٌۗ إِنَّ ٱللَّهَ يَحْكُمُ مَا يُرِيدُ ﴿1﴾
اے ایمان والو! وعدوں کو پورا کرو، تمہارے لیے چوپایوں کی حلال چیزیں جائز کی گئیں، سوائے ان کے جو تمہیں پڑھ کر سنائے جائیں، جب تم احرام میں ہو تو شکار کو حلال نہ سمجھو، بے شک اللہ جو چاہتا ہے حکم دیتا ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تُحِلُّوا۟ شَعَـٰٓئِرَ ٱللَّهِ وَلَا ٱلشَّهْرَ ٱلْحَرَامَ وَلَا ٱلْهَدْيَ وَلَا ٱلْقَلَـٰٓئِدَ وَلَآ ءَآمِّينَ ٱلْبَيْتَ ٱلْحَرَامَ يَبْتَغُونَ فَضْلٗا مِّن رَّبِّهِمْ وَرِضْوَٰنٗاۚ وَإِذَا حَلَلْتُمْ فَٱصْطَادُوا۟ۚ وَلَا يَجْرِمَنَّكُمْ شَنَـَٔانُ قَوْمٍ أَن صَدُّوكُمْ عَنِ ٱلْمَسْجِدِ ٱلْحَرَامِ أَن تَعْتَدُوا۟ۘ وَتَعَاوَنُوا۟ عَلَى ٱلْبِرِّ وَٱلتَّقْوَىٰۖ وَلَا تَعَاوَنُوا۟ عَلَى ٱلْإِثْمِ وَٱلْعُدْوَٰنِۚ وَٱتَّقُوا۟ ٱللَّهَۖ إِنَّ ٱللَّهَ شَدِيدُ ٱلْعِقَابِ ﴿2﴾
اے ایمان والو! اللہ کی نشانیوں کی بے حرمتی نہ کرو، اور نہ حرمت والے مہینے کی، اور نہ قربانی کے جانوروں کی، اور نہ ان جانوروں کی جن کے گلے میں پٹے ہوں، اور نہ ان لوگوں کی جو بیت الحرام کی طرف جا رہے ہوں اپنے رب کے فضل اور رضا کی تلاش میں، اور جب تم احرام سے نکل آؤ تو شکار کرو، اور کسی قوم کی دشمنی تمہیں اس بات پر آمادہ نہ کرے کہ تم زیادتی کرو، نیکی اور تقویٰ پر تعاون کرو، اور گناہ اور زیادتی پر تعاون نہ کرو، اور اللہ سے ڈرو، بے شک اللہ سخت سزا دینے والا ہے۔
حُرِّمَتْ عَلَيْكُمُ ٱلْمَيْتَةُ وَٱلدَّمُ وَلَحْمُ ٱلْخِنزِيرِ وَمَآ أُهِلَّ لِغَيْرِ ٱللَّهِ بِهِۦ وَٱلْمُنْخَنِقَةُ وَٱلْمَوْقُوذَةُ وَٱلْمُتَرَدِّيَةُ وَٱلنَّطِيحَةُ وَمَآ أَكَلَ ٱلسَّبُعُ إِلَّا مَا ذَكَّيْتُمْ وَمَا ذُبِحَ عَلَى ٱلنُّصُبِ وَأَن تَسْتَقْسِمُوا۟ بِٱلْأَزْلَـٰمِ ذَٰلِكُمْ فِسْقٌۗ ٱلْيَوْمَ يَئِسَ ٱلَّذِينَ كَفَرُوا۟ مِن دِينِكُمْ فَلَا تَخْشَوْهُمْ وَٱخْشَوْنِۚ ٱلْيَوْمَ أَكْمَلْتُ لَكُمْ دِينَكُمْ وَأَتْمَمْتُ عَلَيْكُمْ نِعْمَتِي وَرَضِيتُ لَكُمُ ٱلْإِسْلَـٰمَ دِينٗاۚ فَمَنِ ٱضْطُرَّ فِي مَخْمَصَةٍ غَيْرَ مُتَجَانِفٖ لِّإِثْمٖ فَإِنَّ ٱللَّهَ غَفُورٞ رَّحِيمٞ ﴿3﴾
تم پر مردار، خون، خنزیر کا گوشت، اور وہ چیز حرام کی گئی ہے جس پر غیر اللہ کا نام لیا گیا ہو، اور جو گلا گھٹ کر مر جائے، یا چوٹ سے مر جائے، یا بلندی سے گر کر مر جائے، یا سینگ لگنے سے مر جائے، اور جسے درندہ کھا جائے، مگر وہ جسے تم ذبح کر لو، اور جو بتوں کے نام پر ذبح کیا جائے، اور قسمت کے تیروں سے فال نکالنا، یہ سب گناہ ہیں۔ آج کافر تمہارے دین سے مایوس ہو گئے ہیں، پس ان سے نہ ڈرو بلکہ مجھ سے ڈرو۔ آج میں نے تمہارے لیے تمہارا دین مکمل کر دیا، اور تم پر اپنی نعمت پوری کر دی، اور اسلام کو تمہارے لیے دین کے طور پر پسند کیا۔ پھر جو شخص بھوک میں مجبور ہو جائے بغیر گناہ کی طرف مائل ہوئے، تو بے شک اللہ بخشنے والا، رحم کرنے والا ہے۔
يَسْـَٔلُونَكَ مَاذَآ أُحِلَّ لَهُمْۖ قُلْ أُحِلَّ لَكُمُ ٱلطَّيِّبَـٰتُۙ وَمَا عَلَّمْتُم مِّنَ ٱلْجَوَارِحِ مُكَلِّبِينَ تُعَلِّمُونَهُنَّ مِمَّا عَلَّمَكُمُ ٱللَّهُۖ فَكُلُوا۟ مِمَّآ أَمْسَكْنَ عَلَيْكُمْ وَٱذْكُرُوا۟ ٱسْمَ ٱللَّهِ عَلَيْهِۦۚ وَٱتَّقُوا۟ ٱللَّهَۚ إِنَّ ٱللَّهَ سَرِيعُ ٱلْحِسَابِ ﴿4﴾
وہ تم سے پوچھتے ہیں کہ ان کے لیے کیا چیزیں حلال کی گئی ہیں؟ کہہ دو: تمہارے لیے پاکیزہ چیزیں حلال کی گئی ہیں، اور وہ شکار جو تم شکاری جانوروں سے کراؤ، جنہیں تم نے سکھایا ہو اس علم سے جو اللہ نے تمہیں دیا ہے، پس جو وہ تمہارے لیے پکڑ رکھیں اسے کھا لیا کرو، اور اس پر اللہ کا نام لیا کرو، اور اللہ سے ڈرو، بے شک اللہ جلد حساب لینے والا ہے۔
ٱلْيَوْمَ أُحِلَّ لَكُمُ ٱلطَّيِّبَـٰتُۖ وَطَعَامُ ٱلَّذِينَ أُوتُوا۟ ٱلْكِتَـٰبَ حِلّٞ لَّكُمْ وَطَعَامُكُمْ حِلّٞ لَّهُمْۖ وَٱلْمُحْصَنَـٰتُ مِنَ ٱلْمُؤْمِنَـٰتِ وَٱلْمُحْصَنَـٰتُ مِنَ ٱلَّذِينَ أُوتُوا۟ ٱلْكِتَـٰبَ مِن قَبْلِكُمْ إِذَآ ءَاتَيْتُمُوهُنَّ أُجُورَهُنَّ مُحْصِنِينَ غَيْرَ مُسَـٰفِحِينَ وَلَا مُتَّخِذِيٓ أَخْدَانٖۗ وَمَن يَكْفُرْ بِٱلْإِيمَـٰنِ فَقَدْ حَبِطَ عَمَلُهُۥ وَهُوَ فِي ٱلْـَٔاخِرَةِ مِنَ ٱلْخَـٰسِرِينَ ﴿5﴾
آج تمہارے لیے پاکیزہ چیزیں حلال کر دی گئی ہیں، اور اہلِ کتاب کا کھانا تمہارے لیے حلال ہے، اور تمہارا کھانا ان کے لیے حلال ہے، اور مومن عورتیں جو پاک دامن ہوں اور وہ عورتیں بھی جو تم سے پہلے اہلِ کتاب میں سے پاک دامن ہوں، جب تم انہیں ان کے مہر دے دو پاکیزگی کے ساتھ، نہ کہ زناکاری کے لیے اور نہ ہی چوری چھپے تعلق کے لیے، اور جو ایمان سے انکار کرے تو اس کا عمل ضائع ہو گیا، اور وہ آخرت میں نقصان اٹھانے والوں میں سے ہوگا۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ إِذَا قُمْتُمْ إِلَى ٱلصَّلَوٰةِ فَٱغْسِلُوا۟ وُجُوهَكُمْ وَأَيْدِيَكُمْ إِلَى ٱلْمَرَافِقِ وَٱمْسَحُوا۟ بِرُءُوسِكُمْ وَأَرْجُلَكُمْ إِلَى ٱلْكَعْبَيْنِۚ وَإِن كُنتُمْ جُنُبٗا فَٱطَّهَّرُوا۟ۚ وَإِن كُنتُم مَّرْضَىٰٓ أَوْ عَلَىٰ سَفَرٍ أَوْ جَآءَ أَحَدٞ مِّنكُم مِّنَ ٱلْغَآئِطِ أَوْ لَـٰمَسْتُمُ ٱلنِّسَآءَ فَلَمْ تَجِدُوا۟ مَآءٗ فَتَيَمَّمُوا۟ صَعِيدٗا طَيِّبٗا فَٱمْسَحُوا۟ بِوُجُوهِكُمْ وَأَيْدِيكُم مِّنْهُۚ مَا يُرِيدُ ٱللَّهُ لِيَجْعَلَ عَلَيْكُم مِّنْ حَرَجٖ وَلَـٰكِن يُرِيدُ لِيُطَهِّرَكُمْ وَلِيُتِمَّ نِعْمَتَهُۥ عَلَيْكُمْ لَعَلَّكُمْ تَشْكُرُونَ ﴿6﴾
اے ایمان والو! جب تم نماز کے لیے اٹھو تو اپنے چہرے اور ہاتھ کہنیوں تک دھو لو، اور اپنے سروں کا مسح کرو اور پاؤں ٹخنوں تک دھو لو، اور اگر تم جنابت کی حالت میں ہو تو خوب پاکی حاصل کرو، اور اگر تم بیمار ہو یا سفر میں ہو یا تم میں سے کوئی قضائے حاجت سے آیا ہو یا تم نے عورتوں سے مباشرت کی ہو اور پانی نہ پاؤ تو پاک مٹی سے تیمم کرو، پس اپنے چہروں اور ہاتھوں پر اس سے مسح کرو، اللہ نہیں چاہتا کہ تم پر کوئی تنگی رکھے، بلکہ وہ چاہتا ہے کہ تمہیں پاک کرے اور اپنی نعمت تم پر پوری کرے تاکہ تم شکر گزار بنو۔
وَٱذْكُرُوا۟ نِعْمَتَ ٱللَّهِ عَلَيْكُمْ وَمِيثَـٰقَهُ ٱلَّذِي وَاثَقَكُم بِهِۦٓ إِذْ قُلْتُمْ سَمِعْنَا وَأَطَعْنَاۚ وَٱتَّقُوا۟ ٱللَّهَۚ إِنَّ ٱللَّهَ عَلِيمُۢ بِذَاتِ ٱلصُّدُورِ ﴿7﴾
اور اللہ کی نعمت کو یاد کرو جو اس نے تم پر کی، اور اس عہد کو بھی جو اس نے تم سے لیا، جب تم نے کہا: ہم نے سنا اور مانا، اور اللہ سے ڈرو، بے شک اللہ دلوں کے بھید جاننے والا ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ كُونُوا۟ قَوَّٰمِينَ لِلَّهِ شُهَدَآءَ بِٱلْقِسْطِۖ وَلَا يَجْرِمَنَّكُمْ شَنَـَٔانُ قَوْمٍ عَلَىٰٓ أَلَّا تَعْدِلُوا۟ۚ ٱعْدِلُوا۟ هُوَ أَقْرَبُ لِلتَّقْوَىٰۖ وَٱتَّقُوا۟ ٱللَّهَۚ إِنَّ ٱللَّهَ خَبِيرُۢ بِمَا تَعْمَلُونَ ﴿8﴾
اے ایمان والو! اللہ کے لیے انصاف پر قائم رہنے والے بنو، اور گواہی دو انصاف کے ساتھ، اور کسی قوم کی دشمنی تمہیں اس بات پر آمادہ نہ کرے کہ تم انصاف نہ کرو، انصاف کرو، یہی تقویٰ کے زیادہ قریب ہے، اور اللہ سے ڈرو، بے شک اللہ تمہارے اعمال سے خوب واقف ہے۔
وَعَدَ ٱللَّهُ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ لَهُم مَّغْفِرَةٞ وَأَجْرٌ عَظِيمٞ ﴿9﴾
اللہ نے وعدہ کیا ہے ان لوگوں سے جو ایمان لائے اور نیک عمل کیے کہ ان کے لیے بخشش اور بڑا اجر ہے۔
وَٱلَّذِينَ كَفَرُوا۟ وَكَذَّبُوا۟ بِـَٔايَـٰتِنَآ أُو۟لَـٰٓئِكَ أَصْحَـٰبُ ٱلْجَحِيمِ ﴿10﴾
اور جنہوں نے کفر کیا اور ہماری آیات کو جھٹلایا، وہی دوزخ والے ہیں۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ ٱذْكُرُوا۟ نِعْمَتَ ٱللَّهِ عَلَيْكُمْ إِذْ هَمَّ قَوْمٌ أَن يَبْسُطُوٓا۟ إِلَيْكُمْ أَيْدِيَهُمْ فَكَفَّ أَيْدِيَهُمْ عَنكُمْۖ وَٱتَّقُوا۟ ٱللَّهَۚ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُؤْمِنُونَ ﴿11﴾
اے ایمان والو! اللہ کی اس نعمت کو یاد کرو جو اس نے تم پر کی، جب ایک قوم نے ارادہ کیا کہ تم پر ہاتھ اٹھائیں تو اللہ نے ان کے ہاتھ تم سے روک دیے، اور اللہ سے ڈرو، اور ایمان والوں کو اللہ ہی پر بھروسہ کرنا چاہیے۔
وَلَقَدْ أَخَذَ ٱللَّهُ مِيثَـٰقَ بَنِيٓ إِسْرَٰٓءِيلَ وَبَعَثْنَا مِنْهُمُ ٱثْنَيْ عَشَرَ نَقِيبٗاۚ وَقَالَ ٱللَّهُ إِنِّي مَعَكُمْۖ لَئِنْ أَقَمْتُمُ ٱلصَّلَوٰةَ وَءَاتَيْتُمُ ٱلزَّكَوٰةَ وَءَامَنتُم بِرُسُلِي وَعَزَّرْتُمُوهُمْ وَأَقْرَضْتُمُ ٱللَّهَ قَرْضًا حَسَنٗا لَّأُكَفِّرَنَّ عَنكُمْ سَيِّـَٔاتِكُمْ وَلَأُدْخِلَنَّكُمْ جَنَّـٰتٖ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُۚ فَمَن كَفَرَ بَعْدَ ذَٰلِكَ مِنكُمْ فَقَدْ ضَلَّ سَوَآءَ ٱلسَّبِيلِ ﴿12﴾
اور اللہ نے بنی اسرائیل سے پختہ عہد لیا، اور ہم نے ان میں سے بارہ سردار مقرر کیے، اور اللہ نے فرمایا: میں تمہارے ساتھ ہوں، اگر تم نماز قائم کرو گے، زکوٰۃ دو گے، میرے رسولوں پر ایمان لاؤ گے، ان کی مدد کرو گے، اور اللہ کو اچھا قرض دو گے، تو میں تم سے تمہاری برائیاں دور کر دوں گا اور تمہیں ان جنتوں میں داخل کروں گا جن کے نیچے نہریں بہتی ہیں، پھر تم میں سے جو اس کے بعد کفر کرے گا تو وہ سیدھے راستے سے بھٹک گیا۔
فَبِمَا نَقْضِهِم مِّيثَـٰقَهُمْ لَعَنَّـٰهُمْ وَجَعَلْنَا قُلُوبَهُمْ قَاسِيَةٗ يُحَرِّفُونَ ٱلْكَلِمَ عَن مَّوَاضِعِهِۦ وَنَسُوا۟ حَظّٗا مِّمَّا ذُكِّرُوا۟ بِهِۦۚ وَلَا تَزَالُ تَطَّلِعُ عَلَىٰ خَآئِنَةٖ مِّنْهُمْ إِلَّا قَلِيلٗا مِّنْهُمْ فَٱعْفُ عَنْهُمْ وَٱصْفَحْۚ إِنَّ ٱللَّهَ يُحِبُّ ٱلْمُحْسِنِينَ ﴿13﴾
پھر ان کے عہد توڑنے کی وجہ سے ہم نے ان پر لعنت کی اور ان کے دل سخت کر دیے، وہ کلمات کو ان کی جگہوں سے بدل دیتے ہیں، اور اس نصیحت کا ایک حصہ بھول گئے جس کی انہیں یاد دہانی کرائی گئی تھی، اور تم ہمیشہ ان میں خیانت کی ایک نہ ایک بات پاؤ گے، سوائے چند لوگوں کے، پس انہیں معاف کر دو اور درگزر کرو، بے شک اللہ احسان کرنے والوں کو پسند کرتا ہے۔
وَمِنَ ٱلَّذِينَ قَالُوٓا۟ إِنَّا نَصَـٰرَىٰٓ أَخَذْنَا مِيثَـٰقَهُمْ فَنَسُوا۟ حَظّٗا مِّمَّا ذُكِّرُوا۟ بِهِۦ فَأَغْرَيْنَا بَيْنَهُمُ ٱلْعَدَاوَةَ وَٱلْبَغْضَآءَ إِلَىٰ يَوْمِ ٱلْقِيَـٰمَةِۚ وَسَوْفَ يُنَبِّئُهُمُ ٱللَّهُ بِمَا كَانُوا۟ يَصْنَعُونَ ﴿14﴾
اور جنہوں نے کہا کہ ہم نصاریٰ ہیں، ہم نے ان سے بھی عہد لیا، مگر انہوں نے اس نصیحت کا ایک حصہ بھلا دیا جو انہیں دی گئی تھی، تو ہم نے ان کے درمیان قیامت تک کے لیے دشمنی اور بغض ڈال دیا، اور اللہ انہیں بتا دے گا جو وہ کرتے رہے۔
يَـٰٓأَهْلَ ٱلْكِتَـٰبِ قَدْ جَآءَكُمْ رَسُولُنَا يُبَيِّنُ لَكُمْ كَثِيرٗا مِّمَّا كُنتُمْ تُخْفُونَ مِنَ ٱلْكِتَـٰبِ وَيَعْفُوا۟ عَن كَثِيرٖۚ قَدْ جَآءَكُم مِّنَ ٱللَّهِ نُورٞ وَكِتَـٰبٞ مُّبِينٞ ﴿15﴾
اے اہلِ کتاب! یقیناً تمہارے پاس ہمارا رسول آ چکا ہے، جو تمہارے لیے بہت سی وہ باتیں واضح کرتا ہے جو تم کتاب میں چھپاتے تھے، اور بہت سی باتوں سے درگزر کرتا ہے، یقیناً تمہارے پاس اللہ کی طرف سے ایک نور اور واضح کتاب آ چکی ہے۔
يَهْدِي بِهِ ٱللَّهُ مَنِ ٱتَّبَعَ رِضْوَٰنَهُۥ سُبُلَ ٱلسَّلَـٰمِ وَيُخْرِجُهُم مِّنَ ٱلظُّلُمَـٰتِ إِلَى ٱلنُّورِ بِإِذْنِهِۦ وَيَهْدِيهِمْ إِلَىٰ صِرَٰطٖ مُّسْتَقِيمٖ ﴿16﴾
اللہ اس کے ذریعے ان لوگوں کو سلامتی کے راستوں کی ہدایت دیتا ہے جو اس کی رضا کے پیچھے چلتے ہیں، اور انہیں اندھیروں سے نکال کر اپنے حکم سے روشنی کی طرف لے آتا ہے، اور انہیں سیدھے راستے کی طرف ہدایت دیتا ہے۔
لَّقَدْ كَفَرَ ٱلَّذِينَ قَالُوٓا۟ إِنَّ ٱللَّهَ هُوَ ٱلْمَسِيحُ ٱبْنُ مَرْيَمَۚ قُلْ فَمَن يَمْلِكُ مِنَ ٱللَّهِ شَيْـًٔا إِنْ أَرَادَ أَن يُهْلِكَ ٱلْمَسِيحَ ٱبْنَ مَرْيَمَ وَأُمَّهُۥ وَمَن فِي ٱلْأَرْضِ جَمِيعٗاۗ وَلِلَّهِ مُلْكُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَاۚ يَخْلُقُ مَا يَشَآءُۚ وَٱللَّهُ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٞ ﴿17﴾
یقیناً وہ لوگ کافر ہو گئے جنہوں نے کہا کہ اللہ ہی مسیح ابن مریم ہے، کہہ دو: اگر اللہ چاہے کہ مسیح ابن مریم اور اس کی ماں اور زمین میں جو کچھ ہے سب کو ہلاک کر دے تو کون ہے جو اللہ کے مقابلے میں کچھ اختیار رکھتا ہو؟ اور اللہ ہی کے لیے ہے آسمانوں اور زمین اور ان کے درمیان کی ہر چیز کی بادشاہی، وہ جو چاہتا ہے پیدا کرتا ہے، اور اللہ ہر چیز پر قادر ہے۔
وَقَالَتِ ٱلْيَهُودُ وَٱلنَّصَـٰرَىٰ نَحْنُ أَبْنَـٰٓؤُا۟ ٱللَّهِ وَأَحِبَّـٰٓؤُهُۚ قُلْ فَلِمَ يُعَذِّبُكُم بِذُنُوبِكُمۖ بَلْ أَنتُم بَشَرٞ مِّمَّنْ خَلَقَۚ يَغْفِرُ لِمَن يَشَآءُ وَيُعَذِّبُ مَن يَشَآءُۚ وَلِلَّهِ مُلْكُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَاۖ وَإِلَيْهِ ٱلْمَصِيرُ ﴿18﴾
اور یہود و نصاریٰ نے کہا: ہم اللہ کے بیٹے اور اس کے پیارے ہیں، کہہ دو: پھر وہ تمہیں تمہارے گناہوں پر سزا کیوں دیتا ہے؟ بلکہ تم بھی اس کی مخلوق میں سے ایک انسان ہو، وہ جسے چاہے بخش دے اور جسے چاہے سزا دے، اور اللہ ہی کے لیے ہے آسمانوں اور زمین اور ان کے درمیان کی ہر چیز کی بادشاہی، اور اسی کی طرف لوٹ کر جانا ہے۔
يَـٰٓأَهْلَ ٱلْكِتَـٰبِ قَدْ جَآءَكُمْ رَسُولُنَا يُبَيِّنُ لَكُمْ عَلَىٰ فَتْرَةٖ مِّنَ ٱلرُّسُلِ أَن تَقُولُوا۟ مَا جَآءَنَا مِنۢ بَشِيرٖ وَلَا نَذِيرٖۖ فَقَدْ جَآءَكُم بَشِيرٞ وَنَذِيرٞۗ وَٱللَّهُ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٞ ﴿19﴾
اے اہلِ کتاب! یقیناً تمہارے پاس ہمارے رسول آ چکے ہیں، رسولوں کے آنے کے وقفے کے بعد، تاکہ تم یہ نہ کہو کہ ہمارے پاس کوئی خوشخبری دینے والا یا ڈرانے والا نہیں آیا، تو لو! تمہارے پاس خوشخبری دینے والا اور ڈرانے والا آ چکا ہے، اور اللہ ہر چیز پر قادر ہے۔
وَإِذْ قَالَ مُوسَىٰ لِقَوْمِهِۦ يَـٰقَوْمِ ٱذْكُرُوا۟ نِعْمَتَ ٱللَّهِ عَلَيْكُمْ إِذْ جَعَلَ فِيكُمْ أَنبِيَآءَ وَجَعَلَكُم مُّلُوكٗا وَءَاتَىٰكُم مَّا لَمْ يُؤْتِ أَحَدٗا مِّنَ ٱلْعَـٰلَمِينَ ﴿20﴾
اور جب موسیٰ نے اپنی قوم سے کہا: اے میری قوم! اللہ کی اس نعمت کو یاد کرو جو اس نے تم پر کی، جب اس نے تم میں انبیاء بنائے، اور تمہیں بادشاہ بنایا، اور تمہیں وہ کچھ دیا جو جہان والوں میں کسی کو نہیں دیا۔
يَـٰقَوْمِ ٱدْخُلُوا ٱلْأَرْضَ ٱلْمُقَدَّسَةَ ٱلَّتِي كَتَبَ ٱللَّهُ لَكُمْ وَلَا تَرْتَدُّوا۟ عَلَىٰٓ أَدْبَـٰرِكُمْ فَتَنقَلِبُوا۟ خَـٰسِرِينَ ﴿21﴾
اے میری قوم! اس مقدس زمین میں داخل ہو جاؤ جو اللہ نے تمہارے لیے مقرر کی ہے، اور اپنی پیٹھ نہ پھیرو کہ نقصان اٹھا کر لوٹو۔
قَالُوا۟ يَـٰمُوسَىٰٓ إِنَّ فِيهَا قَوْمٗا جَبَّارِينَ وَإِنَّا لَن نَّدْخُلَهَا حَتَّىٰ يَخْرُجُوا۟ مِنْهَا فَإِن يَخْرُجُوا۟ مِنْهَا فَإِنَّا دَاخِلُونَ ﴿22﴾
انہوں نے کہا: اے موسیٰ! اس میں تو بڑے زبردست لوگ ہیں، اور ہم اس میں ہرگز داخل نہ ہوں گے جب تک وہ اس سے نکل نہ جائیں، پھر اگر وہ اس سے نکل جائیں تو ہم ضرور داخل ہوں گے۔
قَالَ رَجُلَانِ مِنَ ٱلَّذِينَ يَخَافُونَ أَنْعَمَ ٱللَّهُ عَلَيْهِمَا ٱدْخُلُوا۟ عَلَيْهِمُ ٱلْبَابَ فَإِذَا دَخَلْتُمُوهُ فَإِنَّكُمْ غَـٰلِبُونَۚ وَعَلَى ٱللَّهِ فَتَوَكَّلُوٓا۟ إِن كُنتُم مُّؤْمِنِينَ ﴿23﴾
دو آدمیوں نے، جنہیں اللہ نے نعمت دی تھی اور جو (اللہ سے) ڈرتے تھے، کہا: ان پر دروازے سے حملہ کرو، پھر جب تم اس میں داخل ہو جاؤ گے تو یقیناً غالب آ جاؤ گے، اور اللہ ہی پر بھروسہ کرو اگر تم ایمان والے ہو۔
قَالُوا۟ يَـٰمُوسَىٰٓ إِنَّا لَن نَّدْخُلَهَآ أَبَدٗا مَّا دَامُوا۟ فِيهَا فَٱذْهَبْ أَنتَ وَرَبُّكَ فَقَـٰتِلَآ إِنَّا هَـٰهُنَا قَـٰعِدُونَ ﴿24﴾
انہوں نے کہا: اے موسیٰ! ہم تو کبھی بھی اس میں داخل نہ ہوں گے جب تک وہ لوگ اس میں موجود ہیں، پس تم اور تمہارا رب جاؤ اور لڑو، ہم تو یہیں بیٹھے ہیں۔
قَالَ رَبِّ إِنِّي لَآ أَمْلِكُ إِلَّا نَفْسِي وَأَخِي فَٱفْرُقْ بَيْنَنَا وَبَيْنَ ٱلْقَوْمِ ٱلْفَـٰسِقِينَ ﴿25﴾
موسیٰ نے کہا: اے میرے رب! میں تو اپنے سوا اور اپنے بھائی کے سوا کسی پر اختیار نہیں رکھتا، پس ہمارے اور ان نافرمان لوگوں کے درمیان جدائی ڈال دے۔
قَالَ فَإِنَّهَا مُحَرَّمَةٌ عَلَيْهِمْ أَرْبَعِينَ سَنَةٗۚ يَهِيمُونَ فِي ٱلْأَرْضِۚ فَلَا تَأْسَ عَلَى ٱلْقَوْمِ ٱلْفَـٰسِقِينَ ﴿26﴾
اللہ نے فرمایا: پس وہ زمین ان پر چالیس سال کے لیے حرام کر دی گئی ہے، وہ زمین میں بھٹکتے پھریں گے، پس تم ان نافرمان لوگوں پر افسوس نہ کرو۔
وَٱتْلُ عَلَيْهِمْ نَبَأَ ٱبْنَيْٓ ءَادَمَ بِٱلْحَقِّ إِذْ قَرَّبَا قُرْبَانٗا فَتُقُبِّلَ مِنْ أَحَدِهِمَا وَلَمْ يُتَقَبَّلْ مِنَ ٱلْءَاخَرِۚ قَالَ لَأَقْتُلَنَّكَۖ قَالَ إِنَّمَا يَتَقَبَّلُ ٱللَّهُ مِنَ ٱلْمُتَّقِينَ ﴿27﴾
اور انہیں آدم کے دو بیٹوں کا واقعہ سچائی کے ساتھ سنا دو، جب دونوں نے قربانی پیش کی، تو ان میں سے ایک کی قربانی قبول کی گئی اور دوسرے کی قبول نہ کی گئی، اس نے کہا: میں تجھے ضرور قتل کروں گا، دوسرے نے کہا: اللہ تو صرف پرہیزگاروں سے قبول کرتا ہے۔
لَئِنۢ بَسَطْتَ إِلَيَّ يَدَكَ لِتَقْتُلَنِي مَآ أَنَا۠ بِبَاسِطٖ يَدِيَ إِلَيْكَ لِأَقْتُلَكَۖ إِنِّيٓ أَخَافُ ٱللَّهَ رَبَّ ٱلْعَـٰلَمِينَ ﴿28﴾
اگر تو مجھے قتل کرنے کے لیے اپنا ہاتھ بڑھائے گا، تو میں تجھے قتل کرنے کے لیے اپنا ہاتھ نہیں بڑھاؤں گا، بے شک میں اللہ رب العالمین سے ڈرتا ہوں۔
إِنِّيٓ أُرِيدُ أَن تَبُوٓأَ بِإِثْمِي وَإِثْمِكَ فَتَكُونَ مِنْ أَصْحَـٰبِ ٱلنَّارِۚ وَذَٰلِكَ جَزَٰٓؤُا۟ ٱلظَّـٰلِمِينَ ﴿29﴾
میں چاہتا ہوں کہ تو میرے اور اپنے گناہ کا بوجھ اٹھا لے، پھر تو دوزخ والوں میں سے ہو جائے، اور یہی ظالموں کی سزا ہے۔
فَطَوَّعَتْ لَهُۥ نَفْسُهُۥ قَتْلَ أَخِيهِ فَقَتَلَهُۥ فَأَصْبَحَ مِنَ ٱلْخَـٰسِرِينَ ﴿30﴾
پھر اس کے نفس نے اسے اپنے بھائی کو قتل کرنے پر آمادہ کر لیا، تو اس نے اسے قتل کر دیا، اور وہ نقصان اٹھانے والوں میں سے ہو گیا۔
فَبَعَثَ ٱللَّهُ غُرَابٗا يَبْحَثُ فِي ٱلْأَرْضِ لِيُرِيَهُۥ كَيْفَ يُوَٰرِي سَوْءَةَ أَخِيهِۚ قَالَ يَـٰوَيْلَتَىٰٓ أَعَجَزْتُ أَنْ أَكُونَ مِثْلَ هَـٰذَا ٱلْغُرَابِ فَأُوَٰرِيَ سَوْءَةَ أَخِيۖ فَأَصْبَحَ مِنَ ٱلنَّـٰدِمِينَ ﴿31﴾
پھر اللہ نے ایک کوّا بھیجا جو زمین کھودنے لگا تاکہ اسے دکھائے کہ وہ اپنے بھائی کی لاش کو کیسے چھپائے، وہ کہنے لگا: ہائے افسوس! کیا میں اس کوّے کی مانند بھی نہ ہو سکا کہ اپنے بھائی کی لاش چھپا لیتا؟ پس وہ نادم ہونے والوں میں سے ہو گیا۔
مِنْ أَجْلِ ذَٰلِكَ كَتَبْنَا عَلَىٰ بَنِيٓ إِسْرَٰٓءِيلَ أَنَّهُۥ مَن قَتَلَ نَفْسٗا بِغَيْرِ نَفْسٍ أَوْ فَسَادٖ فِي ٱلْأَرْضِ فَكَأَنَّمَا قَتَلَ ٱلنَّاسَ جَمِيعٗاۖ وَمَنْ أَحْيَاهَا فَكَأَنَّمَآ أَحْيَا ٱلنَّاسَ جَمِيعٗاۚ وَلَقَدْ جَآءَتْهُمْ رُسُلُنَا بِٱلْبَيِّنَـٰتِ ثُمَّ إِنَّ كَثِيرٗا مِّنْهُم بَعْدَ ذَٰلِكَ فِي ٱلْأَرْضِ لَمُسْرِفُونَ ﴿32﴾
اسی وجہ سے ہم نے بنی اسرائیل پر لکھ دیا کہ جس نے کسی جان کو بغیر جان کے بدلے یا زمین میں فساد کے بغیر قتل کیا، گویا اس نے تمام انسانوں کو قتل کیا، اور جس نے کسی جان کو زندہ کیا، گویا اس نے تمام انسانوں کو زندہ کیا، اور یقیناً ان کے پاس ہمارے رسول واضح دلائل لے کر آئے، پھر بھی ان میں سے بہت سے لوگ زمین میں حد سے بڑھنے والے ہیں۔
إِنَّمَا جَزَٰٓؤُا۟ ٱلَّذِينَ يُحَارِبُونَ ٱللَّهَ وَرَسُولَهُۥ وَيَسْعَوْنَ فِي ٱلْأَرْضِ فَسَادًا أَن يُقَتَّلُوٓا۟ أَوْ يُصَلَّبُوٓا۟ أَوْ تُقَطَّعَ أَيْدِيهِمْ وَأَرْجُلُهُم مِّنْ خِلَـٰفٍ أَوْ يُنفَوْا۟ مِنَ ٱلْأَرْضِۚ ذَٰلِكَ لَهُمْ خِزْيٞ فِي ٱلدُّنْيَاۖ وَلَهُمْ فِي ٱلْـَٔاخِرَةِ عَذَابٌ عَظِيمٌ ﴿33﴾
جو لوگ اللہ اور اس کے رسول سے جنگ کرتے ہیں اور زمین میں فساد پھیلانے کی کوشش کرتے ہیں، ان کی سزا یہی ہے کہ انہیں قتل کیا جائے یا سولی دی جائے یا ان کے ہاتھ اور پاؤں مخالف سمتوں سے کاٹے جائیں یا انہیں زمین سے نکال دیا جائے، یہ ان کے لیے دنیا میں رسوائی ہے، اور آخرت میں ان کے لیے بڑا عذاب ہے۔
إِلَّا ٱلَّذِينَ تَابُوا۟ مِن قَبْلِ أَن تَقْدِرُوا۟ عَلَيْهِمْ فَٱعْلَمُوٓا۟ أَنَّ ٱللَّهَ غَفُورٞ رَّحِيمٞ ﴿34﴾
مگر وہ لوگ جو اس سے پہلے توبہ کر لیں کہ تم ان پر قابو پاؤ، تو جان لو کہ اللہ بخشنے والا، رحم کرنے والا ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ ٱتَّقُوا۟ ٱللَّهَ وَٱبْتَغُوٓا۟ إِلَيْهِ ٱلْوَسِيلَةَ وَجَـٰهِدُوا۟ فِي سَبِيلِهِۦ لَعَلَّكُمْ تُفْلِحُونَ ﴿35﴾
اے ایمان والو! اللہ سے ڈرو اور اس کی طرف وسیلہ تلاش کرو، اور اس کے راستے میں جہاد کرو تاکہ تم کامیاب ہو جاؤ۔
إِنَّ ٱلَّذِينَ كَفَرُوا۟ لَوْ أَنَّ لَهُم مَّا فِي ٱلْأَرْضِ جَمِيعٗا وَمِثْلَهُۥ مَعَهُۥ لِيَفْتَدُوا۟ بِهِۦ مِنْ عَذَابِ يَوْمِ ٱلْقِيَـٰمَةِ مَا تُقُبِّلَ مِنْهُمْۖ وَلَهُمْ عَذَابٌ أَلِيمٞ ﴿36﴾
بے شک جنہوں نے کفر کیا، اگر ان کے پاس زمین کی ساری چیزیں ہوں اور اتنی ہی اور بھی ہو کہ وہ اسے قیامت کے دن کے عذاب سے بچنے کے لیے فدیہ میں دے دیں، تو بھی ان سے قبول نہ کیا جائے گا، اور ان کے لیے دردناک عذاب ہے۔
يُرِيدُونَ أَن يَخْرُجُوا۟ مِنَ ٱلنَّارِ وَمَا هُم بِخَـٰرِجِينَ مِنْهَا وَلَهُمْ عَذَابٞ مُّقِيمٞ ﴿37﴾
وہ آگ سے نکلنا چاہیں گے مگر وہ اس سے نکلنے والے نہیں ہوں گے، اور ان کے لیے ہمیشہ رہنے والا عذاب ہے۔
وَٱلسَّارِقُ وَٱلسَّارِقَةُ فَٱقْطَعُوٓا۟ أَيْدِيَهُمَا جَزَآءَۢ بِمَا كَسَبَا نَكَالٗا مِّنَ ٱللَّهِۗ وَٱللَّهُ عَزِيزٌ حَكِيمٞ ﴿38﴾
اور چور مرد اور چور عورت، ان دونوں کے ہاتھ کاٹ دو، ان کے کیے کا بدلہ، اللہ کی طرف سے عبرت، اور اللہ غالب ہے، حکمت والا ہے۔
فَمَن تَابَ مِنۢ بَعْدِ ظُلْمِهِۦ وَأَصْلَحَ فَإِنَّ ٱللَّهَ يَتُوبُ عَلَيْهِۦۚ إِنَّ ٱللَّهَ غَفُورٞ رَّحِيمٌ ﴿39﴾
پھر جو شخص اپنے ظلم کے بعد توبہ کرے اور اصلاح کرے تو بے شک اللہ اس کی توبہ قبول کرے گا، بے شک اللہ بخشنے والا، رحم کرنے والا ہے۔
أَلَمْ تَعْلَمْ أَنَّ ٱللَّهَ لَهُۥ مُلْكُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ يُعَذِّبُ مَن يَشَآءُ وَيَغْفِرُ لِمَن يَشَآءُۚ وَٱللَّهُ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٞ ﴿40﴾
کیا تم نہیں جانتے کہ اللہ ہی کے لیے آسمانوں اور زمین کی بادشاہی ہے؟ وہ جسے چاہے سزا دیتا ہے اور جسے چاہے بخش دیتا ہے، اور اللہ ہر چیز پر قدرت رکھتا ہے۔
يَـٰٓأَيُّهَا ٱلرَّسُولُ لَا يَحْزُنكَ ٱلَّذِينَ يُسَـٰرِعُونَ فِي ٱلْكُفْرِ مِنَ ٱلَّذِينَ قَالُوٓا۟ ءَامَنَّا بِأَفْوَٰهِهِمْ وَلَمْ تُؤْمِن قُلُوبُهُمْۛ وَمِنَ ٱلَّذِينَ هَادُوا۟ۛ سَمَّـٰعُونَ لِلْكَذِبِ سَمَّـٰعُونَ لِقَوْمٍ ءَاخَرِينَ لَمْ يَأْتُوكَۖ يُحَرِّفُونَ ٱلْكَلِمَ مِنۢ بَعْدِ مَوَاضِعِهِۦۖ يَقُولُونَ إِنْ أُوتِيتُمْ هَـٰذَا فَخُذُوهُ وَإِن لَّمْ تُؤْتَوْهُ فَٱحْذَرُوا۟ۚ وَمَن يُرِدِ ٱللَّهُ فِتْنَتَهُۥ فَلَن تَمْلِكَ لَهُۥ مِنَ ٱللَّهِ شَيْـٔٗاۚ أُو۟لَـٰٓئِكَ ٱلَّذِينَ لَمْ يُرِدِ ٱللَّهُ أَن يُطَهِّرَ قُلُوبَهُمۚ لَهُمْ فِي ٱلدُّنْيَا خِزْيٞ وَلَهُمْ فِي ٱلْـَٔاخِرَةِ عَذَابٌ عَظِيمٌ ﴿41﴾
اے رسول! وہ لوگ تمہیں غمگین نہ کریں جو کفر میں جلدی کرتے ہیں، ان میں سے جو کہتے ہیں کہ ہم ایمان لائے حالانکہ ان کے دل ایمان نہیں لائے، اور یہودیوں میں سے کچھ جھوٹ سننے والے ہیں، وہ ایسے لوگوں کی باتیں بھی سنتے ہیں جو تمہارے پاس نہیں آئے، وہ کلمات کو ان کی جگہوں کے بعد بدل دیتے ہیں، وہ کہتے ہیں: اگر تمہیں یہ (فیصلہ) دیا جائے تو لے لو، اور اگر یہ نہ دیا جائے تو بچو، اور جسے اللہ فتنے میں ڈالنا چاہے تو تم اللہ کے مقابلے میں اس کے لیے کچھ اختیار نہیں رکھتے، یہی وہ لوگ ہیں جن کے دلوں کو اللہ پاک کرنا نہیں چاہتا، ان کے لیے دنیا میں رسوائی ہے اور آخرت میں بڑا عذاب۔
سَمَّـٰعُونَ لِلْكَذِبِ أَكَّـٰلُونَ لِلسُّحْتِۚ فَإِن جَآءُوكَ فَٱحْكُم بَيْنَهُم أَوْ أَعْرِضْ عَنْهُمْۖ وَإِن تُعْرِضْ عَنْهُمْ فَلَن يَضُرُّوكَ شَيْـٔٗاۖ وَإِنْ حَكَمْتَ فَٱحْكُم بَيْنَهُم بِٱلْقِسْطِۚ إِنَّ ٱللَّهَ يُحِبُّ ٱلْمُقْسِطِينَ ﴿42﴾
یہ جھوٹ سننے والے اور حرام کھانے والے ہیں، پھر اگر وہ تمہارے پاس آئیں تو ان کے درمیان فیصلہ کرو یا ان سے منہ پھیر لو، اور اگر تم ان سے منہ پھیر لو تو وہ تمہیں کچھ نقصان نہیں پہنچا سکتے، اور اگر تم فیصلہ کرو تو انصاف سے فیصلہ کرو، بے شک اللہ انصاف کرنے والوں کو پسند کرتا ہے۔
وَكَيْفَ يُحَكِّمُونَكَ وَعِندَهُمُ ٱلتَّوْرَىٰةُ فِيهَا حُكْمُ ٱللَّهِ ثُمَّ يَتَوَلَّوْنَ مِنۢ بَعْدِ ذَٰلِكَۚ وَمَآ أُو۟لَـٰٓئِكَ بِٱلْمُؤْمِنِينَ ﴿43﴾
اور وہ تمہیں کیسے حکم بناتے ہیں جبکہ ان کے پاس تورات موجود ہے جس میں اللہ کا حکم ہے؟ پھر اس کے بعد بھی وہ منہ موڑ لیتے ہیں، اور وہ لوگ ایمان والے نہیں ہیں۔
إِنَّآ أَنزَلْنَا ٱلتَّوْرَىٰةَ فِيهَا هُدٗى وَنُورٞۚ يَحْكُمُ بِهَا ٱلنَّبِيُّونَ ٱلَّذِينَ أَسْلَمُوا۟ لِلَّذِينَ هَادُوا۟ وَٱلرَّبَّـٰنِيُّونَ وَٱلْأَحْبَارُ بِمَا ٱسْتُحْفِظُوا۟ مِن كِتَـٰبِ ٱللَّهِ وَكَانُوا۟ عَلَيْهِ شُهَدَآءَۚ فَلَا تَخْشَوُا۟ ٱلنَّاسَ وَٱخْشَوْنِ وَلَا تَشْتَرُوا۟ بِـَٔايَـٰتِي ثَمَنٗا قَلِيلٗاۚ وَمَن لَّمْ يَحْكُم بِمَآ أَنزَلَ ٱللَّهُ فَأُو۟لَـٰٓئِكَ هُمُ ٱلْكَـٰفِرُونَ ﴿44﴾
بے شک ہم نے تورات نازل کی جس میں ہدایت اور روشنی تھی، اس کے مطابق انبیاء جو اللہ کے فرمانبردار تھے یہودیوں کے لیے فیصلہ کرتے تھے، اور ربّانی اور علماء بھی، کیونکہ انہیں اللہ کی کتاب کا محافظ بنایا گیا تھا اور وہ اس پر گواہ تھے، پس لوگوں سے نہ ڈرو بلکہ مجھ سے ڈرو، اور میری آیات کو تھوڑی قیمت پر نہ بیچو، اور جو اللہ کے نازل کردہ حکم کے مطابق فیصلہ نہ کرے تو وہی لوگ کافر ہیں۔
وَكَتَبْنَا عَلَيْهِمْ فِيهَآ أَنَّ ٱلنَّفْسَ بِٱلنَّفْسِ وَٱلْعَيْنَ بِٱلْعَيْنِ وَٱلْأَنْفَ بِٱلْأَنْفِ وَٱلْأُذُنَ بِٱلْأُذُنِ وَٱلسِّنَّ بِٱلسِّنِّ وَٱلْجُرُوحَ قِصَاصٞۚ فَمَن تَصَدَّقَ بِهِۦ فَهُوَ كَفَّارَةٞ لَّهُۥۚ وَمَن لَّمْ يَحْكُم بِمَآ أَنزَلَ ٱللَّهُ فَأُو۟لَـٰٓئِكَ هُمُ ٱلظَّـٰلِمُونَ ﴿45﴾
اور ہم نے ان پر تورات میں لکھ دیا کہ جان کے بدلے جان، آنکھ کے بدلے آنکھ، ناک کے بدلے ناک، کان کے بدلے کان، دانت کے بدلے دانت، اور زخموں کا بھی بدلہ ہے، پھر جو اس کو معاف کر دے تو وہ اس کے لیے کفارہ ہے، اور جو اللہ کے نازل کردہ حکم کے مطابق فیصلہ نہ کرے تو وہی لوگ ظالم ہیں۔
وَقَفَّيْنَا عَلَىٰٓ ءَاثَـٰرِهِم بِعِيسَى ٱبْنِ مَرْيَمَ مُصَدِّقٗا لِّمَا بَيْنَ يَدَيْهِ مِنَ ٱلتَّوْرَىٰةِۖ وَءَاتَيْنَـٰهُ ٱلْإِنجِيلَ فِيهِ هُدٗى وَنُورٞ وَمُصَدِّقٗا لِّمَا بَيْنَ يَدَيْهِ مِنَ ٱلتَّوْرَىٰةِ وَهُدٗى وَمَوْعِظَةٗ لِّلْمُتَّقِينَ ﴿46﴾
اور ہم نے ان کے پیچھے عیسیٰ ابن مریم کو بھیجا، جو اپنے سے پہلے کی تورات کی تصدیق کرنے والے تھے، اور ہم نے انہیں انجیل عطا کی جس میں ہدایت اور نور تھا، اور وہ اپنے سے پہلے کی تورات کی تصدیق کرنے والی تھی، اور پرہیزگاروں کے لیے ہدایت اور نصیحت تھی۔
وَلْيَحْكُمْ أَهْلُ ٱلْإِنجِيلِ بِمَآ أَنزَلَ ٱللَّهُ فِيهِۚ وَمَن لَّمْ يَحْكُم بِمَآ أَنزَلَ ٱللَّهُ فَأُو۟لَـٰٓئِكَ هُمُ ٱلْفَـٰسِقُونَ ﴿47﴾
اور اہلِ انجیل کو چاہیے کہ اس کے مطابق فیصلہ کریں جو اللہ نے اس میں نازل کیا ہے، اور جو اللہ کے نازل کردہ حکم کے مطابق فیصلہ نہ کرے تو وہی لوگ فاسق ہیں۔
وَأَنزَلْنَآ إِلَيْكَ ٱلْكِتَـٰبَ بِٱلْحَقِّ مُصَدِّقٗا لِّمَا بَيْنَ يَدَيْهِ مِنَ ٱلْكِتَـٰبِ وَمُهَيْمِنًا عَلَيْهِۖ فَٱحْكُم بَيْنَهُم بِمَآ أَنزَلَ ٱللَّهُ وَلَا تَتَّبِعْ أَهْوَآءَهُمْ عَمَّا جَآءَكَ مِنَ ٱلْحَقِّۚ لِكُلّٖ جَعَلْنَا مِنكُمْ شِرْعَةٗ وَمِنْهَاجٗاۚ وَلَوْ شَآءَ ٱللَّهُ لَجَعَلَكُمْ أُمَّةٗ وَٰحِدَةٗ وَلَـٰكِن لِّيَبْلُوَكُمْ فِي مَآ ءَاتَىٰكُمۡۖ فَٱسْتَبِقُوا۟ ٱلْخَيْرَٰتِۚ إِلَى ٱللَّهِ مَرْجِعُكُمْ جَمِيعٗا فَيُنَبِّئُكُم بِمَا كُنتُمْ فِيهِ تَخْتَلِفُونَ ﴿48﴾
اور ہم نے آپ پر حق کے ساتھ کتاب نازل کی، جو اس کتاب کی تصدیق کرنے والی ہے جو اس سے پہلے تھی، اور اس پر نگہبان ہے، پس آپ ان کے درمیان اس کے مطابق فیصلہ کریں جو اللہ نے نازل کیا ہے، اور ان کی خواہشات کی پیروی نہ کریں، اس حق سے ہٹ کر جو آپ کے پاس آیا ہے، ہم نے تم میں سے ہر ایک کے لیے ایک شریعت اور طریقہ مقرر کیا ہے، اور اگر اللہ چاہتا تو تم سب کو ایک ہی امت بنا دیتا، لیکن وہ تمہیں آزماتا ہے اس میں جو اس نے تمہیں دیا، پس نیکیوں میں سبقت کرو، تم سب کو اللہ ہی کی طرف لوٹ کر جانا ہے، پھر وہ تمہیں ان باتوں سے آگاہ کرے گا جن میں تم اختلاف کرتے تھے۔
وَأَنِ ٱحْكُم بَيْنَهُم بِمَآ أَنزَلَ ٱللَّهُ وَلَا تَتَّبِعْ أَهْوَآءَهُمْ وَٱحْذَرْهُمْ أَن يَفْتِنُوكَ عَنۢ بَعْضِ مَآ أَنزَلَ ٱللَّهُ إِلَيْكَۖ فَإِن تَوَلَّوْا۟ فَٱعْلَمْ أَنَّمَا يُرِيدُ ٱللَّهُ أَن يُصِيبَهُم بِبَعْضِ ذُنُوبِهِمۚ وَإِنَّ كَثِيرٗا مِّنَ ٱلنَّاسِ لَفَـٰسِقُونَ ﴿49﴾
اور آپ ان کے درمیان اس کے مطابق فیصلہ کریں جو اللہ نے نازل کیا ہے، اور ان کی خواہشات کی پیروی نہ کریں، اور ان سے بچیں کہ کہیں وہ آپ کو اس چیز سے نہ بہکا دیں جو اللہ نے آپ کی طرف نازل کی ہے، پھر اگر وہ منہ موڑ لیں تو جان لو کہ اللہ چاہتا ہے کہ انہیں ان کے بعض گناہوں کی سزا دے، اور بے شک بہت سے لوگ نافرمان ہیں۔
أَفَحُكْمَ ٱلْجَـٰهِلِيَّةِ يَبْغُونَۚ وَمَنْ أَحْسَنُ مِنَ ٱللَّهِ حُكْمٗا لِّقَوْمٖ يُوقِنُونَ ﴿50﴾
کیا وہ جاہلیت کا فیصلہ چاہتے ہیں؟ اور اللہ سے بہتر کس کا فیصلہ ہو سکتا ہے ان لوگوں کے لیے جو یقین رکھتے ہیں؟
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تَتَّخِذُوا۟ ٱلْيَهُودَ وَٱلنَّصَـٰرَىٰٓ أَوْلِيَآءَۘ بَعْضُهُمْ أَوْلِيَآءُ بَعْضٖۚ وَمَن يَتَوَلَّهُم مِّنكُمْ فَإِنَّهُۥ مِنْهُمْۗ إِنَّ ٱللَّهَ لَا يَهْدِي ٱلْقَوْمَ ٱلظَّـٰلِمِينَ ﴿51﴾
اے ایمان والو! یہود و نصاریٰ کو دوست نہ بناؤ، وہ آپس میں ایک دوسرے کے دوست ہیں، اور تم میں سے جو ان سے دوستی کرے گا وہ انہی میں سے ہوگا، بے شک اللہ ظالم قوم کو ہدایت نہیں دیتا۔
فَتَرَى ٱلَّذِينَ فِي قُلُوبِهِم مَّرَضٞ يُسَـٰرِعُونَ فِيهِمْ يَقُولُونَ نَخْشَىٰٓ أَن تُصِيبَنَا دَآئِرَةٞۚ فَعَسَى ٱللَّهُ أَن يَأْتِيَ بِٱلْفَتْحِ أَوْ أَمْرٖ مِّنْ عِندِهِۦ فَيُصْبِحُوا۟ عَلَىٰ مَآ أَسَرُّوا۟ فِيٓ أَنفُسِهِمْ نَـٰدِمِينَ ﴿52﴾
تو تم ان لوگوں کو دیکھو گے جن کے دلوں میں بیماری ہے کہ وہ ان (کافروں) کی طرف دوڑتے ہیں، کہتے ہیں: ہمیں اندیشہ ہے کہ ہم پر کوئی گردش نہ آ جائے، پس قریب ہے کہ اللہ فتح یا اپنی طرف سے کوئی فیصلہ لے آئے، تو وہ اس پر جو اپنے دلوں میں چھپاتے تھے پچھتانے لگیں۔
وَيَقُولُ ٱلَّذِينَ ءَامَنُوٓا۟ أَهَـٰٓؤُلَآءِ ٱلَّذِينَ أَقْسَمُوا۟ بِٱللَّهِ جَهْدَ أَيْمَـٰنِهِمْ إِنَّهُمْ لَمَعَكُمْۚ حَبِطَتْ أَعْمَـٰلُهُمْ فَأَصْبَحُوا۟ خَـٰسِرِينَ ﴿53﴾
اور ایمان والے کہیں گے: کیا یہی وہ لوگ ہیں جنہوں نے اللہ کی سخت قسمیں کھائی تھیں کہ وہ تمہارے ساتھ ہیں؟ ان کے اعمال ضائع ہو گئے اور وہ نقصان اٹھانے والے ہو گئے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ مَن يَرْتَدَّ مِنكُمْ عَن دِينِهِۦ فَسَوْفَ يَأْتِي ٱللَّهُ بِقَوْمٖ يُحِبُّهُمْ وَيُحِبُّونَهُۥٓ أَذِلَّةٍ عَلَى ٱلْمُؤْمِنِينَ أَعِزَّةٍ عَلَى ٱلْكَـٰفِرِينَ يُجَـٰهِدُونَ فِي سَبِيلِ ٱللَّهِ وَلَا يَخَافُونَ لَوْمَةَ لَآئِمٖۚ ذَٰلِكَ فَضْلُ ٱللَّهِ يُؤْتِيهِ مَن يَشَآءُۚ وَٱللَّهُ وَٰسِعٌ عَلِيمٌ ﴿54﴾
اے ایمان والو! تم میں سے جو کوئی اپنے دین سے پھر جائے گا تو اللہ عنقریب ایسے لوگوں کو لائے گا جن سے وہ محبت کرے گا اور وہ اس سے محبت کریں گے، وہ مومنوں پر نرم اور کافروں پر سخت ہوں گے، اللہ کے راستے میں جہاد کریں گے اور کسی ملامت کرنے والے کی ملامت سے نہ ڈریں گے، یہ اللہ کا فضل ہے، وہ جسے چاہے عطا کرتا ہے، اور اللہ وسعت والا، علم والا ہے۔
إِنَّمَا وَلِيُّكُمُ ٱللَّهُ وَرَسُولُهُۥ وَٱلَّذِينَ ءَامَنُوا۟ ٱلَّذِينَ يُقِيمُونَ ٱلصَّلَوٰةَ وَيُؤْتُونَ ٱلزَّكَوٰةَ وَهُمْ رَاكِعُونَ ﴿55﴾
تمہارا ولی تو اللہ ہے، اور اس کا رسول، اور وہ ایمان والے جو نماز قائم کرتے ہیں، زکوٰۃ دیتے ہیں اور عاجزی کے ساتھ جھکتے ہیں۔
وَمَن يَتَوَلَّ ٱللَّهَ وَرَسُولَهُۥ وَٱلَّذِينَ ءَامَنُوا۟ فَإِنَّ حِزْبَ ٱللَّهِ هُمُ ٱلْغَـٰلِبُونَ ﴿56﴾
اور جو اللہ، اس کے رسول اور ایمان والوں کو اپنا دوست بنائے، تو بے شک اللہ کی جماعت ہی غالب آنے والی ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تَتَّخِذُوا۟ ٱلَّذِينَ ٱتَّخَذُوا۟ دِينَكُمْ هُزُوٗا وَلَعِبٗا مِّنَ ٱلَّذِينَ أُوتُوا۟ ٱلْكِتَـٰبَ مِن قَبْلِكُمْ وَٱلْكُفَّارَ أَوْلِيَآءَۚ وَٱتَّقُوا۟ ٱللَّهَ إِن كُنتُم مُّؤْمِنِينَ ﴿57﴾
اے ایمان والو! ان لوگوں کو دوست نہ بناؤ جنہوں نے تمہارے دین کو مذاق اور کھیل بنا رکھا ہے، ان میں سے جو تم سے پہلے اہلِ کتاب ہوئے اور کافر بھی، اور اللہ سے ڈرو اگر تم ایمان والے ہو۔
وَإِذَا نَادَيْتُمْ إِلَى ٱلصَّلَوٰةِ ٱتَّخَذُوهَا هُزُوٗا وَلَعِبٗاۚ ذَٰلِكَ بِأَنَّهُمْ قَوْمٞ لَّا يَعْقِلُونَ ﴿58﴾
اور جب تم نماز کے لیے اذان دیتے ہو تو وہ اسے مذاق اور کھیل بنا لیتے ہیں، یہ اس لیے کہ وہ ایسے لوگ ہیں جو عقل نہیں رکھتے۔
قُلْ يَـٰٓأَهْلَ ٱلْكِتَـٰبِ هَلْ تَنقِمُونَ مِنَّآ إِلَّآ أَنْ ءَامَنَّا بِٱللَّهِ وَمَآ أُنزِلَ إِلَيْنَا وَمَآ أُنزِلَ مِن قَبْلُ وَأَنَّ أَكْثَرَكُمْ فَـٰسِقُونَ ﴿59﴾
کہہ دو: اے اہلِ کتاب! کیا تم ہم سے صرف اس بات کا بدلہ لے رہے ہو کہ ہم اللہ پر ایمان لائے، اور اس پر جو ہماری طرف نازل کیا گیا، اور اس پر جو اس سے پہلے نازل کیا گیا، اور یہ کہ تم میں سے اکثر نافرمان ہیں؟
قُلْ هَلْ أُنَبِّئُكُم بِشَرّٖ مِّن ذَٰلِكَ مَثُوبَةً عِندَ ٱللَّهِۚ مَن لَّعَنَهُ ٱللَّهُ وَغَضِبَ عَلَيْهِ وَجَعَلَ مِنْهُمُ ٱلْقِرَدَةَ وَٱلْخَنَازِيرَ وَعَبَدَ ٱلطَّـٰغُوتَۚ أُو۟لَـٰٓئِكَ شَرّٞ مَّكَانٗا وَأَضَلُّ عَن سَوَآءِ ٱلسَّبِيلِ ﴿60﴾
کہہ دو: کیا میں تمہیں اس سے بھی بدتر انجام بتاؤں جو اللہ کے ہاں سزا کے لحاظ سے ہے؟ وہ لوگ جن پر اللہ نے لعنت کی، اور ان پر غضب کیا، اور ان میں سے بعض کو بندر اور خنزیر بنا دیا، اور جنہوں نے طاغوت کی عبادت کی، یہی لوگ بدترین مقام والے ہیں اور سیدھے راستے سے بہت دور جا پڑے ہیں۔
وَإِذَا جَآءُوكُمْ قَالُوٓا۟ ءَامَنَّا وَقَد دَّخَلُوا۟ بِٱلْكُفْرِ وَهُمْ قَدْ خَرَجُوا۟ بِهِۦۚ وَٱللَّهُ أَعْلَمُ بِمَا كَانُوا۟ يَكْتُمُونَ ﴿61﴾
اور جب وہ تمہارے پاس آتے ہیں تو کہتے ہیں: ہم ایمان لائے، حالانکہ وہ کفر کے ساتھ داخل ہوئے اور اسی کے ساتھ نکلے، اور اللہ خوب جانتا ہے جو کچھ وہ چھپاتے ہیں۔
وَتَرَىٰ كَثِيرٗا مِّنْهُمْ يُسَـٰرِعُونَ فِي ٱلْإِثْمِ وَٱلْعُدْوَٰنِ وَأَكْلِهِمُ ٱلسُّحْتَۚ لَبِئْسَ مَا كَانُوا۟ يَعْمَلُونَ ﴿62﴾
اور تم ان میں سے بہتوں کو دیکھو گے کہ وہ گناہ، زیادتی اور حرام کھانے میں جلدی کرتے ہیں، یقیناً وہ جو کچھ کرتے ہیں بہت برا ہے۔
لَوْلَا يَنْهَىٰهُمُ ٱلرَّبَّـٰنِيُّونَ وَٱلْأَحْبَارُ عَن قَوْلِهِمُ ٱلْإِثْمَ وَأَكْلِهِمُ ٱلسُّحْتَۚ لَبِئْسَ مَا كَانُوا۟ يَصْنَعُونَ ﴿63﴾
کیوں نہیں انہیں ان کے ربانی اور علماء گناہ کی بات کہنے اور حرام کھانے سے روکتے؟ یقیناً وہ جو کچھ کرتے ہیں بہت برا ہے۔
وَقَالَتِ ٱلْيَهُودُ يَدُ ٱللَّهِ مَغْلُولَةٌۚ غُلَّتْ أَيْدِيهِمْ وَلُعِنُوا۟ بِمَا قَالُوا۟ۘ بَلْ يَدَاهُ مَبْسُوطَتَانِ يُنفِقُ كَيْفَ يَشَآءُۚ وَلَيَزِيدَنَّ كَثِيرٗا مِّنْهُم مَّآ أُنزِلَ إِلَيْكَ مِن رَّبِّكَ طُغْيَـٰنٗا وَكُفْرٗاۚ وَأَلْقَيْنَا بَيْنَهُمُ ٱلْعَدَٰوَةَ وَٱلْبَغْضَآءَ إِلَىٰ يَوْمِ ٱلْقِيَـٰمَةِۚ كُلَّمَآ أَوْقَدُوا۟ نَارٗا لِّلْحَرْبِ أَطْفَـٰهَا ٱللَّهُۚ وَيَسْعَوْنَ فِي ٱلْأَرْضِ فَسَادٗاۚ وَٱللَّهُ لَا يُحِبُّ ٱلْمُفْسِدِينَ ﴿64﴾
اور یہود نے کہا: اللہ کا ہاتھ بندھا ہوا ہے، ان کے ہاتھ باندھ دیے گئے اور ان پر لعنت کی گئی اس بات کے سبب جو انہوں نے کہی، بلکہ اس کے دونوں ہاتھ کھلے ہوئے ہیں، وہ جس طرح چاہتا ہے خرچ کرتا ہے، اور جو کچھ آپ کے رب کی طرف سے آپ پر نازل ہوا ہے وہ ان میں سے بہتوں کو سرکشی اور کفر میں اور بڑھا دے گا، اور ہم نے ان کے درمیان قیامت تک کے لیے دشمنی اور بغض ڈال دیا ہے، جب بھی وہ جنگ کی آگ بھڑکاتے ہیں اللہ اسے بجھا دیتا ہے، اور وہ زمین میں فساد پھیلانے کی کوشش کرتے ہیں، اور اللہ فساد کرنے والوں کو پسند نہیں کرتا۔
وَلَوْ أَنَّ أَهْلَ ٱلْكِتَـٰبِ ءَامَنُوا۟ وَٱتَّقَوْا۟ لَكَفَّرْنَا عَنْهُمْ سَيِّـَٔاتِهِمْ وَلَأَدْخَلْنَـٰهُمْ جَنَّـٰتِ ٱلنَّعِيمِ ﴿65﴾
اور اگر اہلِ کتاب ایمان لے آتے اور تقویٰ اختیار کرتے تو ہم ان کی برائیاں ضرور دور کر دیتے اور انہیں نعمتوں والی جنتوں میں داخل کر دیتے۔
وَلَوْ أَنَّهُمْ أَقَامُوا۟ ٱلتَّوْرَىٰةَ وَٱلْإِنجِيلَ وَمَآ أُنزِلَ إِلَيْهِم مِّن رَّبِّهِمْ لَأَكَلُوا۟ مِن فَوْقِهِمْ وَمِن تَحْتِ أَرْجُلِهِمۚ مِّنْهُمْ أُمَّةٞ مُّقْتَصِدَةٞۖ وَكَثِيرٞ مِّنْهُمْ سَآءَ مَا يَعْمَلُونَ ﴿66﴾
اور اگر وہ تورات اور انجیل کو اور جو کچھ ان کے رب کی طرف سے ان پر نازل کیا گیا ہے قائم رکھتے تو وہ اپنے اوپر سے اور اپنے پاؤں کے نیچے سے کھاتے، ان میں سے ایک جماعت میانہ رو ہے، اور ان میں سے اکثر لوگ بہت برا عمل کرتے ہیں۔
يَـٰٓأَيُّهَا ٱلرَّسُولُ بَلِّغْ مَآ أُنزِلَ إِلَيْكَ مِن رَّبِّكَۖ وَإِن لَّمْ تَفْعَلْ فَمَا بَلَّغْتَ رِسَالَتَهُۥۚ وَٱللَّهُ يَعْصِمُكَ مِنَ ٱلنَّاسِۗ إِنَّ ٱللَّهَ لَا يَهْدِي ٱلْقَوْمَ ٱلْكَـٰفِرِينَ ﴿67﴾
اے رسول! پہنچا دو جو کچھ تمہارے رب کی طرف سے تم پر نازل کیا گیا ہے، اور اگر تم نے ایسا نہ کیا تو تم نے اس کا پیغام نہیں پہنچایا، اور اللہ تمہیں لوگوں سے بچائے گا، بے شک اللہ کافروں کو ہدایت نہیں دیتا۔
قُلْ يَـٰٓأَهْلَ ٱلْكِتَـٰبِ لَسْتُمْ عَلَىٰ شَيْءٍ حَتَّىٰ تُقِيمُوا۟ ٱلتَّوْرَىٰةَ وَٱلْإِنجِيلَ وَمَآ أُنزِلَ إِلَيْكُم مِّن رَّبِّكُمْۗ وَلَيَزِيدَنَّ كَثِيرٗا مِّنْهُم مَّآ أُنزِلَ إِلَيْكَ مِن رَّبِّكَ طُغْيَـٰنٗا وَكُفْرٗاۖ فَلَا تَأْسَ عَلَى ٱلْقَوْمِ ٱلْكَـٰفِرِينَ ﴿68﴾
کہہ دو: اے اہلِ کتاب! تم کسی چیز پر نہیں ہو جب تک تورات اور انجیل کو اور جو کچھ تمہارے رب کی طرف سے تم پر نازل کیا گیا ہے قائم نہ کرو، اور جو کچھ تمہارے رب کی طرف سے تم پر نازل کیا گیا ہے وہ ان میں سے اکثر کو سرکشی اور کفر میں اور بڑھا دے گا، پس تم کافر قوم پر افسوس نہ کرو۔
إِنَّ ٱلَّذِينَ ءَامَنُوا۟ وَٱلَّذِينَ هَادُوا۟ وَٱلصَّـٰبِـِٔينَ وَٱلنَّصَـٰرَىٰ مَنْ ءَامَنَ بِٱللَّهِ وَٱلْيَوْمِ ٱلْـَٔاخِرِ وَعَمِلَ صَـٰلِحٗا فَلَا خَوْفٌ عَلَيْهِمْ وَلَا هُمْ يَحْزَنُونَ ﴿69﴾
بے شک جو لوگ ایمان لائے، اور جو یہودی ہوئے، اور صابی، اور نصاریٰ، جو بھی اللہ اور آخرت کے دن پر ایمان لایا اور نیک عمل کیا، تو ان پر کوئی خوف نہیں ہوگا اور نہ وہ غمگین ہوں گے۔
لَقَدْ أَخَذْنَا مِيثَـٰقَ بَنِيٓ إِسْرَٰٓءِيلَ وَأَرْسَلْنَآ إِلَيْهِمْ رُسُلٗاۖ كُلَّمَا جَآءَهُمْ رَسُولُۢ بِمَا لَا تَهْوَىٰٓ أَنفُسُهُمْ فَرِيقٗا كَذَّبُوا۟ وَفَرِيقٗا يَقْتُلُونَ ﴿70﴾
یقیناً ہم نے بنی اسرائیل سے عہد لیا، اور ان کی طرف کئی رسول بھیجے، جب بھی ان کے پاس کوئی رسول ایسی بات لے کر آیا جو ان کے نفس کو پسند نہ تھی، تو ایک جماعت کو انہوں نے جھٹلایا اور ایک جماعت کو قتل کر دیا۔
وَحَسِبُوٓا۟ أَلَّا تَكُونَ فِتْنَةٞ فَعَمُوا۟ وَصَمُّوا۟ ثُمَّ تَابَ ٱللَّهُ عَلَيْهِمْ ثُمَّ عَمُوا۟ وَصَمُّوا۟ كَثِيرٗا مِّنْهُمْۚ وَٱللَّهُ بَصِيرُۢ بِمَا يَعْمَلُونَ ﴿71﴾
اور انہوں نے گمان کیا کہ کوئی آزمائش نہ ہوگی، تو وہ اندھے اور بہرے ہو گئے، پھر اللہ نے ان پر مہربانی فرمائی، پھر ان میں سے بہت سے لوگ پھر اندھے اور بہرے ہو گئے، اور اللہ ان کے اعمال کو خوب دیکھنے والا ہے۔
لَقَدْ كَفَرَ ٱلَّذِينَ قَالُوٓا۟ إِنَّ ٱللَّهَ هُوَ ٱلْمَسِيحُ ٱبْنُ مَرْيَمَۚ وَقَالَ ٱلْمَسِيحُ يَـٰبَنِيٓ إِسْرَٰٓءِيلَ ٱعْبُدُوا۟ ٱللَّهَ رَبِّي وَرَبَّكُمْۚ إِنَّهُۥ مَن يُشْرِكْ بِٱللَّهِ فَقَدْ حَرَّمَ ٱللَّهُ عَلَيْهِ ٱلْجَنَّةَ وَمَأْوَىٰهُ ٱلنَّارُۖ وَمَا لِلظَّـٰلِمِينَ مِنْ أَنصَارٖ ﴿72﴾
یقیناً وہ لوگ کافر ہو گئے جنہوں نے کہا کہ اللہ ہی مسیح ابن مریم ہے، حالانکہ مسیح نے کہا تھا: اے بنی اسرائیل! اللہ کی عبادت کرو جو میرا اور تمہارا رب ہے، بے شک جس نے اللہ کے ساتھ شرک کیا، اللہ نے اس پر جنت حرام کر دی ہے، اور اس کا ٹھکانہ جہنم ہے، اور ظالموں کا کوئی مددگار نہیں۔
لَّقَدْ كَفَرَ ٱلَّذِينَ قَالُوٓا۟ إِنَّ ٱللَّهَ ثَالِثُ ثَلَـٰثَةٖۘ وَمَا مِنْ إِلَـٰهٍ إِلَّآ إِلَـٰهٞ وَٰحِدٞۚ وَإِن لَّمْ يَنتَهُوا۟ عَمَّا يَقُولُونَ لَيَمَسَّنَّ ٱلَّذِينَ كَفَرُوا۟ مِنْهُمْ عَذَابٌ أَلِيمٌ ﴿73﴾
یقیناً وہ لوگ بھی کافر ہو گئے جنہوں نے کہا کہ اللہ تین میں سے ایک ہے، حالانکہ کوئی معبود نہیں مگر ایک ہی معبود، اور اگر وہ اس بات سے باز نہ آئیں گے جو وہ کہتے ہیں تو ان میں سے جو کافر ہیں انہیں ضرور دردناک عذاب پہنچے گا۔
أَفَلَا يَتُوبُونَ إِلَى ٱللَّهِ وَيَسْتَغْفِرُونَهُۥۚ وَٱللَّهُ غَفُورٞ رَّحِيمٌ ﴿74﴾
تو کیا وہ اللہ کی طرف توبہ نہیں کرتے اور اس سے بخشش نہیں مانگتے؟ حالانکہ اللہ بخشنے والا، رحم کرنے والا ہے۔
مَا ٱلْمَسِيحُ ٱبْنُ مَرْيَمَ إِلَّا رَسُولٞ قَدْ خَلَتْ مِن قَبْلِهِ ٱلرُّسُلُ وَأُمُّهُۥ صِدِّيقَةٞۖ كَانَا يَأْكُلَانِ ٱلطَّعَامَۗ ٱنظُرْ كَيْفَ نُبَيِّنُ لَهُمُ ٱلْـَٔايَـٰتِ ثُمَّ ٱنظُرْ أَنَّىٰ يُؤْفَكُونَ ﴿75﴾
مسیح ابن مریم تو صرف ایک رسول تھے، ان سے پہلے بھی بہت سے رسول گزر چکے ہیں، اور ان کی ماں صدیقہ تھی، دونوں کھانا کھایا کرتے تھے، دیکھو ہم ان کے لیے کس طرح نشانیاں واضح کرتے ہیں، پھر دیکھو وہ کیسے بہکائے جاتے ہیں۔
قُلْ أَتَعْبُدُونَ مِن دُونِ ٱللَّهِ مَا لَا يَمْلِكُ لَكُمْ ضَرّٗا وَلَا نَفْعٗاۚ وَٱللَّهُ هُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ ﴿76﴾
کہہ دو: کیا تم اللہ کے سوا اس کی عبادت کرتے ہو جو نہ تمہیں نقصان پہنچا سکتا ہے اور نہ نفع؟ اور اللہ ہی سننے والا، جاننے والا ہے۔
قُلْ يَـٰٓأَهْلَ ٱلْكِتَـٰبِ لَا تَغْلُوا۟ فِي دِينِكُمْ غَيْرَ ٱلْحَقِّ وَلَا تَتَّبِعُوٓا۟ أَهْوَآءَ قَوْمٖ قَدْ ضَلُّوا۟ مِن قَبْلُ وَأَضَلُّوا۟ كَثِيرٗا وَضَلُّوا۟ عَن سَوَآءِ ٱلسَّبِيلِ ﴿77﴾
کہہ دو: اے اہلِ کتاب! اپنے دین میں ناحق غلو نہ کرو، اور ان لوگوں کی خواہشات کی پیروی نہ کرو جو پہلے گمراہ ہو چکے، اور بہت سے لوگوں کو گمراہ کر چکے، اور سیدھے راستے سے بھٹک گئے۔
لُعِنَ ٱلَّذِينَ كَفَرُوا۟ مِنۢ بَنِيٓ إِسْرَٰٓءِيلَ عَلَىٰ لِسَانِ دَاوُۥدَ وَعِيسَى ٱبْنِ مَرْيَمَۚ ذَٰلِكَ بِمَا عَصَوا۟ وَّكَانُوا۟ يَعْتَدُونَ ﴿78﴾
بنی اسرائیل میں سے جنہوں نے کفر کیا ان پر داؤد اور عیسیٰ ابن مریم کی زبان سے لعنت کی گئی، یہ اس لیے کہ وہ نافرمانی کرتے تھے اور حد سے بڑھ جاتے تھے۔
كَانُوا۟ لَا يَتَنَاهَوْنَ عَن مُّنكَرٖ فَعَلُوهُۚ لَبِئْسَ مَا كَانُوا۟ يَفْعَلُونَ ﴿79﴾
وہ ایک دوسرے کو برے کام سے نہیں روکتے تھے جو وہ کرتے تھے، یقیناً وہ بہت برا کام کرتے تھے۔
تَرَىٰ كَثِيرٗا مِّنْهُمْ يَتَوَلَّوْنَ ٱلَّذِينَ كَفَرُوا۟ۚ لَبِئْسَ مَا قَدَّمَتْ لَهُمْ أَنفُسُهُمْ أَن سَخِطَ ٱللَّهُ عَلَيْهِمْ وَفِي ٱلْعَذَابِ هُمْ خَـٰلِدُونَ ﴿80﴾
تم ان میں سے بہتوں کو دیکھو گے کہ وہ کافروں سے دوستی رکھتے ہیں، یقیناً وہ جو کچھ اپنے لیے آگے بھیج رہے ہیں بہت برا ہے، کہ اللہ ان سے ناراض ہو گیا، اور وہ ہمیشہ کے لیے عذاب میں رہیں گے۔
وَلَوْ كَانُوا۟ يُؤْمِنُونَ بِٱللَّهِ وَٱلنَّبِيِّ وَمَآ أُنزِلَ إِلَيْهِ مَا ٱتَّخَذُوهُمْ أَوْلِيَآءَ وَلَـٰكِنَّ كَثِيرٗا مِّنْهُمْ فَـٰسِقُونَ ﴿81﴾
اور اگر وہ اللہ اور نبی پر اور جو کچھ اس پر نازل کیا گیا ہے ایمان رکھتے تو انہیں دوست نہ بناتے، لیکن ان میں سے اکثر نافرمان ہیں۔
لَتَجِدَنَّ أَشَدَّ ٱلنَّاسِ عَدَاوَةٗ لِّلَّذِينَ ءَامَنُوا۟ ٱلْيَهُودَ وَٱلَّذِينَ أَشْرَكُوا۟ۖ وَلَتَجِدَنَّ أَقْرَبَهُم مَّوَدَّةٗ لِّلَّذِينَ ءَامَنُوا۟ ٱلَّذِينَ قَالُوٓا۟ إِنَّا نَصَـٰرَىٰۚ ذَٰلِكَ بِأَنَّ مِنْهُمْ قِسِّيسِينَ وَرُهْبَانٗا وَأَنَّهُمْ لَا يَسْتَكْبِرُونَ ﴿82﴾
یقیناً تم ایمان والوں کے سب سے زیادہ دشمن یہود اور مشرکوں کو پاؤ گے، اور تم ان لوگوں کو ایمان والوں کے ساتھ محبت میں سب سے زیادہ قریب پاؤ گے جنہوں نے کہا: ہم نصاریٰ ہیں، یہ اس لیے کہ ان میں علماء اور عبادت گزار ہیں اور وہ تکبر نہیں کرتے۔
وَإِذَا سَمِعُوا۟ مَآ أُنزِلَ إِلَى ٱلرَّسُولِ تَرَىٰٓ أَعْيُنَهُمْ تَفِيضُ مِنَ ٱلدَّمْعِ مِمَّا عَرَفُوا۟ مِنَ ٱلْحَقِّۖ يَقُولُونَ رَبَّنَآ ءَامَنَّا فَٱكْتُبْنَا مَعَ ٱلشَّـٰهِدِينَ ﴿83﴾
اور جب وہ اس چیز کو سنتے ہیں جو رسول پر نازل کی گئی ہے تو تم ان کی آنکھوں کو آنسو بہاتے ہوئے دیکھتے ہو، اس حق کو پہچان لینے کی وجہ سے، وہ کہتے ہیں: اے ہمارے رب! ہم ایمان لائے، پس ہمیں گواہی دینے والوں کے ساتھ لکھ لے۔
وَمَا لَنَا لَا نُؤْمِنُ بِٱللَّهِ وَمَا جَآءَنَا مِنَ ٱلْحَقِّ وَنَطْمَعُ أَن يُدْخِلَنَا رَبُّنَا مَعَ ٱلْقَوْمِ ٱلصَّـٰلِحِينَ ﴿84﴾
اور ہمیں کیا ہے کہ ہم اللہ پر اور اس حق پر جو ہمارے پاس آیا ہے ایمان نہ لائیں، اور ہم امید رکھتے ہیں کہ ہمارا رب ہمیں نیک لوگوں کے ساتھ داخل کرے گا۔
فَأَثَـٰبَهُمُ ٱللَّهُ بِمَا قَالُوا۟ جَنَّـٰتٖ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُ خَـٰلِدِينَ فِيهَاۚ وَذَٰلِكَ جَزَآءُ ٱلْمُحْسِنِينَ ﴿85﴾
تو اللہ نے ان کے اس قول کے بدلے انہیں ایسی جنتیں عطا کیں جن کے نیچے نہریں بہتی ہیں، جن میں وہ ہمیشہ رہیں گے، اور یہی نیک لوگوں کا بدلہ ہے۔
وَٱلَّذِينَ كَفَرُوا۟ وَكَذَّبُوا۟ بِـَٔايَـٰتِنَآ أُو۟لَـٰٓئِكَ أَصْحَـٰبُ ٱلْجَحِيمِ ﴿86﴾
اور جنہوں نے کفر کیا اور ہماری آیات کو جھٹلایا، وہی جہنم والے ہیں۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تُحَرِّمُوا۟ طَيِّبَـٰتِ مَآ أَحَلَّ ٱللَّهُ لَكُمْ وَلَا تَعْتَدُوٓا۟ۚ إِنَّ ٱللَّهَ لَا يُحِبُّ ٱلْمُعْتَدِينَ ﴿87﴾
اے ایمان والو! ان پاکیزہ چیزوں کو حرام نہ کرو جو اللہ نے تمہارے لیے حلال کی ہیں، اور حد سے نہ بڑھو، بے شک اللہ حد سے بڑھنے والوں کو پسند نہیں کرتا۔
وَكُلُوا۟ مِمَّا رَزَقَكُمُ ٱللَّهُ حَلَـٰلٗا طَيِّبٗاۚ وَٱتَّقُوا۟ ٱللَّهَ ٱلَّذِيٓ أَنتُم بِهِۦ مُؤْمِنُونَ ﴿88﴾
اور جو کچھ اللہ نے تمہیں حلال اور پاکیزہ رزق دیا ہے اس میں سے کھاؤ، اور اللہ سے ڈرو جس پر تم ایمان رکھتے ہو۔
لَا يُؤَاخِذُكُمُ ٱللَّهُ بِٱللَّغْوِ فِيٓ أَيْمَـٰنِكُمْ وَلَـٰكِن يُؤَاخِذُكُم بِمَا عَقَّدتُّمُ ٱلْأَيْمَـٰنَۖ فَكَفَّـٰرَتُهُۥٓ إِطْعَامُ عَشَرَةِ مَسَـٰكِينَ مِنْ أَوْسَطِ مَا تُطْعِمُونَ أَهْلِيكُمْ أَوْ كِسْوَتُهُمْ أَوْ تَحْرِيرُ رَقَبَةٖۖ فَمَن لَّمْ يَجِدْ فَصِيَامُ ثَلَـٰثَةِ أَيَّامٖۚ ذَٰلِكَ كَفَّـٰرَةُ أَيْمَـٰنِكُمْ إِذَا حَلَفْتُمْۚ وَٱحْفَظُوٓا۟ أَيْمَـٰنَكُمْۚ كَذَٰلِكَ يُبَيِّنُ ٱللَّهُ لَكُمْ ءَايَـٰتِهِۦ لَعَلَّكُمْ تَشْكُرُونَ ﴿89﴾
اللہ تمہیں تمہاری لغو قسموں پر نہیں پکڑتا، لیکن تمہاری ان قسموں پر پکڑتا ہے جو تم پختگی سے کھاؤ، تو اس کا کفارہ دس مسکینوں کو کھانا کھلانا ہے، اس درمیانے درجے کا جو تم اپنے گھر والوں کو کھلاتے ہو، یا ان کو کپڑے دینا، یا ایک غلام آزاد کرنا، پھر جو نہ پائے تو تین دن کے روزے رکھے، یہ تمہاری قسموں کا کفارہ ہے جب تم قسم کھا لو، اور اپنی قسموں کی حفاظت کرو، اسی طرح اللہ تمہارے لیے اپنی آیات بیان کرتا ہے تاکہ تم شکر ادا کرو۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ إِنَّمَا ٱلْخَمْرُ وَٱلْمَيْسِرُ وَٱلْأَنصَابُ وَٱلْأَزْلَـٰمُ رِجْسٞ مِّنْ عَمَلِ ٱلشَّيْطَـٰنِ فَٱجْتَنِبُوهُ لَعَلَّكُمْ تُفْلِحُونَ ﴿90﴾
اے ایمان والو! بے شک شراب، جوا، بت، اور پانسے شیطانی کاموں میں سے گندگی ہیں، پس ان سے بچو تاکہ تم فلاح پاؤ۔
إِنَّمَا يُرِيدُ ٱلشَّيْطَـٰنُ أَن يُوقِعَ بَيْنَكُمُ ٱلْعَدَاوَةَ وَٱلْبَغْضَآءَ فِي ٱلْخَمْرِ وَٱلْمَيْسِرِ وَيَصُدَّكُمْ عَن ذِكْرِ ٱللَّهِ وَعَنِ ٱلصَّلَوٰةِۖ فَهَلْ أَنتُم مُّنتَهُونَ ﴿91﴾
شیطان تو یہی چاہتا ہے کہ شراب اور جوئے کے ذریعے تمہارے درمیان دشمنی اور بغض ڈال دے، اور تمہیں اللہ کے ذکر اور نماز سے روک دے، تو کیا تم باز آؤ گے؟
وَأَطِيعُوا۟ ٱللَّهَ وَأَطِيعُوا۟ ٱلرَّسُولَ وَٱحْذَرُوا۟ۚ فَإِن تَوَلَّيْتُمْ فَٱعْلَمُوٓا۟ أَنَّمَا عَلَىٰ رَسُولِنَا ٱلْبَلَـٰغُ ٱلْمُبِينُ ﴿92﴾
اور اللہ کی اطاعت کرو اور رسول کی اطاعت کرو اور بچتے رہو، پھر اگر تم منہ موڑ لو تو جان لو کہ ہمارے رسول پر صرف واضح طور پر پہنچا دینا ہے۔
لَيْسَ عَلَى ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ جُنَاحٞ فِيمَا طَعِمُوٓا۟ إِذَا مَا ٱتَّقَوا۟ وَءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ ثُمَّ ٱتَّقَوا۟ وَءَامَنُوا۟ ثُمَّ ٱتَّقَوا۟ وَأَحْسَنُوا۟ۚ وَٱللَّهُ يُحِبُّ ٱلْمُحْسِنِينَ ﴿93﴾
جو لوگ ایمان لائے اور نیک عمل کیے ان پر اس چیز میں کوئی گناہ نہیں جو انہوں نے کھائی، جب کہ وہ تقویٰ اختیار کریں اور ایمان لائیں اور نیک عمل کریں، پھر تقویٰ اختیار کریں اور ایمان لائیں، پھر تقویٰ اختیار کریں اور احسان کریں، اور اللہ احسان کرنے والوں کو پسند کرتا ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَيَبْلُوَنَّكُمُ ٱللَّهُ بِشَيْءٖ مِّنَ ٱلصَّيْدِ تَنَالُهُۥٓ أَيْدِيكُمْ وَرِمَاحُكُمْ لِيَعْلَمَ ٱللَّهُ مَن يَخَافُهُۥ بِٱلْغَيْبِۚ فَمَنِ ٱعْتَدَىٰ بَعْدَ ذَٰلِكَ فَلَهُۥ عَذَابٌ أَلِيمٞ ﴿94﴾
اے ایمان والو! اللہ تمہیں ضرور آزمائے گا کچھ شکار کے ذریعے جسے تمہارے ہاتھ اور نیزے حاصل کریں گے، تاکہ اللہ معلوم کرے کون اس سے غیب میں ڈرتا ہے، پھر جو اس کے بعد حد سے بڑھے گا تو اس کے لیے دردناک عذاب ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تَقْتُلُوا۟ ٱلصَّيْدَ وَأَنتُمْ حُرُمٞۚ وَمَن قَتَلَهُۥ مِنكُم مُّتَعَمِّدٗا فَجَزَآءٞ مِّثْلُ مَا قَتَلَ مِنَ ٱلنَّعَمِ يَحْكُمُ بِهِۦ ذَوَا عَدْلٖ مِّنكُمْ هَدْيٗاۭ بَـٰلِغَ ٱلْكَعْبَةِ أَوْ كَفَّـٰرَةٞ طَعَامُ مَسَـٰكِينَ أَوْ عَدْلُ ذَٰلِكَ صِيَامٗاۭ لِّيَذُوقَ وَبَالَ أَمْرِهِۦۗ عَفَا ٱللَّهُ عَمَّا سَلَفَۚ وَمَنْ عَادَ فَيَنتَقِمُ ٱللَّهُ مِنْهُۗ وَٱللَّهُ عَزِيزٞ ذُو ٱنتِقَامٖ ﴿95﴾
اے ایمان والو! احرام کی حالت میں شکار نہ مارو، اور تم میں سے جو جان بوجھ کر اسے مارے تو اس کا بدلہ اسی جیسا جانور ہے جو اس نے مارا، جس کا فیصلہ تم میں سے دو انصاف والے کریں، جو کعبہ تک قربانی کے طور پر پہنچایا جائے، یا کفارہ کے طور پر مسکینوں کو کھانا کھلانا، یا اس کے برابر روزے رکھنا، تاکہ وہ اپنے کیے کا مزہ چکھے، اللہ نے جو پہلے ہو چکا اسے معاف کر دیا، اور جو دوبارہ کرے گا تو اللہ اس سے انتقام لے گا، اور اللہ غالب ہے، انتقام لینے والا ہے۔
أُحِلَّ لَكُمْ صَيْدُ ٱلْبَحْرِ وَطَعَامُهُۥ مَتَـٰعٗا لَّكُمْ وَلِلسَّيَّارَةِۖ وَحُرِّمَ عَلَيْكُمْ صَيْدُ ٱلْبَرِّ مَا دُمْتُمْ حُرُمٗاۗ وَٱتَّقُوا۟ ٱللَّهَ ٱلَّذِيٓ إِلَيْهِ تُحْشَرُونَ ﴿96﴾
تمہارے لیے سمندر کا شکار اور اس کا کھانا تمہارے فائدے کے لیے اور قافلے والوں کے لیے حلال کیا گیا ہے، اور خشکی کا شکار تم پر حرام کیا گیا ہے جب تک تم احرام کی حالت میں ہو، اور اللہ سے ڈرو جس کی طرف تم جمع کیے جاؤ گے۔
جَعَلَ ٱللَّهُ ٱلْكَعْبَةَ ٱلْبَيْتَ ٱلْحَرَامَ قِيَـٰمٗا لِّلنَّاسِ وَٱلشَّهْرَ ٱلْحَرَامَ وَٱلْهَدْيَ وَٱلْقَلَـٰٓئِدَۚ ذَٰلِكَ لِتَعْلَمُوٓا۟ أَنَّ ٱللَّهَ يَعْلَمُ مَا فِي ٱلسَّمَـٰوَٰتِ وَمَا فِي ٱلْأَرْضِ وَأَنَّ ٱللَّهَ بِكُلِّ شَيْءٍ عَلِيمٌ ﴿97﴾
اللہ نے کعبہ، حرمت والے گھر کو لوگوں کے لیے قیام کا ذریعہ بنایا، اور حرمت والے مہینے کو، اور قربانی کے جانوروں کو، اور پٹے والے جانوروں کو بھی، یہ اس لیے ہے تاکہ تم جان لو کہ اللہ آسمانوں اور زمین کی ہر چیز کو جانتا ہے، اور بے شک اللہ ہر چیز کو خوب جاننے والا ہے۔
ٱعْلَمُوٓا۟ أَنَّ ٱللَّهَ شَدِيدُ ٱلْعِقَابِ وَأَنَّ ٱللَّهَ غَفُورٞ رَّحِيمٞ ﴿98﴾
جان لو کہ اللہ سخت سزا دینے والا ہے، اور بے شک اللہ بخشنے والا، رحم کرنے والا ہے۔
مَا عَلَى ٱلرَّسُولِ إِلَّا ٱلْبَلَـٰغُۚ وَٱللَّهُ يَعْلَمُ مَا تُبْدُونَ وَمَا تَكْتُمُونَ ﴿99﴾
رسول پر صرف پیغام پہنچا دینا ہے، اور اللہ جانتا ہے جو کچھ تم ظاہر کرتے ہو اور جو کچھ تم چھپاتے ہو۔
قُل لَّا يَسْتَوِي ٱلْخَبِيثُ وَٱلطَّيِّبُ وَلَوْ أَعْجَبَكَ كَثْرَةُ ٱلْخَبِيثِۚ فَٱتَّقُوا۟ ٱللَّهَ يَـٰٓأُو۟لِي ٱلْأَلْبَـٰبِ لَعَلَّكُمْ تُفْلِحُونَ ﴿100﴾
کہہ دو: ناپاک اور پاک برابر نہیں ہو سکتے، اگرچہ ناپاک کی کثرت تمہیں تعجب میں ڈالے، پس اے عقل والو! اللہ سے ڈرو تاکہ تم کامیاب ہو جاؤ۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تَسْـَٔلُوا۟ عَنْ أَشْيَآءَ إِن تُبْدَ لَكُمْ تَسُؤْكُمْۚ وَإِن تَسْـَٔلُوا۟ عَنْهَا حِينَ يُنَزَّلُ ٱلْقُرْءَانُ تُبْدَ لَكُمْۚ عَفَا ٱللَّهُ عَنْهَاۗ وَٱللَّهُ غَفُورٌ حَلِيمٌ ﴿101﴾
اے ایمان والو! ان چیزوں کے بارے میں سوال نہ کرو جو اگر تم پر ظاہر کر دی جائیں تو تمہیں ناگوار ہوں، اور اگر تم ان کے بارے میں اس وقت سوال کرو جب قرآن نازل ہو رہا ہو تو وہ تم پر ظاہر کر دی جائیں گی، اللہ نے ان سے درگزر فرمایا، اور اللہ بخشنے والا، بردبار ہے۔
قَدْ سَأَلَهَا قَوْمٞ مِّن قَبْلِكُمْ ثُمَّ أَصْبَحُوا۟ بِهَا كَـٰفِرِينَ ﴿102﴾
تم سے پہلے ایک قوم نے ان چیزوں کے بارے میں سوال کیا تھا، پھر وہ ان کے سبب کافر ہو گئے۔
مَا جَعَلَ ٱللَّهُ مِنۢ بَحِيرَةٖ وَلَا سَآئِبَةٖ وَلَا وَصِيلَةٖ وَلَا حَامٖ وَلَـٰكِنَّ ٱلَّذِينَ كَفَرُوا۟ يَفْتَرُونَ عَلَى ٱللَّهِ ٱلْكَذِبَۖ وَأَكْثَرُهُمْ لَا يَعْقِلُونَ ﴿103﴾
اللہ نے نہ کوئی بحیرہ مقرر کیا ہے، نہ سائبہ، نہ وصیلہ، نہ حام، بلکہ کافر لوگ اللہ پر جھوٹ باندھتے ہیں، اور ان میں سے اکثر عقل نہیں رکھتے۔
وَإِذَا قِيلَ لَهُمْ تَعَالَوْا۟ إِلَىٰ مَآ أَنزَلَ ٱللَّهُ وَإِلَى ٱلرَّسُولِ قَالُوا۟ حَسْبُنَا مَا وَجَدْنَا عَلَيْهِ ءَابَآءَنَاۚ أَوَلَوْ كَانَ ءَابَآؤُهُمْ لَا يَعْلَمُونَ شَيْـٔٗا وَلَا يَهْتَدُونَ ﴿104﴾
اور جب ان سے کہا جاتا ہے کہ آؤ اس چیز کی طرف جو اللہ نے نازل کی ہے اور رسول کی طرف، تو وہ کہتے ہیں: ہمیں وہی کافی ہے جس پر ہم نے اپنے باپ دادا کو پایا، کیا اگر ان کے باپ دادا کچھ نہ جانتے ہوں اور نہ ہدایت یافتہ ہوں؟
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ عَلَيْكُمْ أَنفُسَكُمْۖ لَا يَضُرُّكُم مَّن ضَلَّ إِذَا ٱهْتَدَيْتُمْۚ إِلَى ٱللَّهِ مَرْجِعُكُمْ جَمِيعٗا فَيُنَبِّئُكُم بِمَا كُنتُمْ تَعْمَلُونَ ﴿105﴾
اے ایمان والو! تم پر لازم ہے اپنی فکر کرنا، جب تم ہدایت پر ہو تو جو کوئی گمراہ ہو وہ تمہیں نقصان نہیں پہنچا سکتا، تم سب کو اللہ ہی کی طرف لوٹ کر جانا ہے، پھر وہ تمہیں بتا دے گا جو کچھ تم کیا کرتے تھے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ شَهَـٰدَةُ بَيْنِكُمْ إِذَا حَضَرَ أَحَدَكُمُ ٱلْمَوْتُ حِينَ ٱلْوَصِيَّةِ ٱثْنَانِ ذَوَا عَدْلٖ مِّنكُمْ أَوْ ءَاخَرَانِ مِنْ غَيْرِكُمْ إِنْ أَنتُمْ ضَرَبْتُمْ فِي ٱلْأَرْضِ فَأَصَـٰبَتْكُم مُّصِيبَةُ ٱلْمَوْتِۚ تَحْبِسُونَهُمَا مِنۢ بَعْدِ ٱلصَّلَوٰةِ فَيُقْسِمَانِ بِٱللَّهِ إِنِ ٱرْتَبْتُمْ لَا نَشْتَرِي بِهِۦ ثَمَنٗا وَلَوْ كَانَ ذَا قُرْبَىٰ وَلَا نَكْتُمُ شَهَـٰدَةَ ٱللَّهِ إِنَّآ إِذٗا لَّمِنَ ٱلْـَٔاثِمِينَ ﴿106﴾
اے ایمان والو! جب تم میں سے کسی کو موت آ جائے تو وصیت کے وقت تم میں سے دو انصاف والے گواہ ہوں، یا اگر تم سفر میں ہو اور تمہیں موت کی مصیبت آ پہنچے تو غیر قوم کے دو افراد، پھر اگر تمہیں شک ہو تو نماز کے بعد ان دونوں کو روک لو، پھر وہ اللہ کی قسم کھائیں کہ ہم اس کے بدلے کوئی قیمت نہیں لیں گے، اگرچہ وہ رشتہ دار ہی ہو، اور ہم اللہ کی گواہی کو نہیں چھپائیں گے، ورنہ ہم گناہگاروں میں سے ہوں گے۔
فَإِنْ عُثِرَ عَلَىٰٓ أَنَّهُمَا ٱسْتَحَقَّآ إِثْمٗا فَـَٔاخَرَانِ يَقُومَانِ مَقَامَهُمَا مِنَ ٱلَّذِينَ ٱسْتَحَقَّ عَلَيْهِمُ ٱلْأَوْلَيَـٰنِ فَيُقْسِمَانِ بِٱللَّهِ لَشَهَـٰدَتُنَآ أَحَقُّ مِن شَهَـٰدَتِهِمَا وَمَا ٱعْتَدَيْنَآ إِنَّآ إِذٗا لَّمِنَ ٱلظَّـٰلِمِينَ ﴿107﴾
پھر اگر معلوم ہو جائے کہ وہ دونوں گناہ کے مرتکب ہوئے ہیں، تو ان کی جگہ دوسرے دو افراد کھڑے ہوں جو ان لوگوں میں سے ہوں جن پر حق ہے، پھر وہ اللہ کی قسم کھائیں کہ ہماری گواہی ان دونوں کی گواہی سے زیادہ درست ہے، اور ہم نے زیادتی نہیں کی، ورنہ ہم ظالموں میں سے ہوں گے۔
ذَٰلِكَ أَدْنَىٰٓ أَن يَأْتُوا۟ بِٱلشَّهَـٰدَةِ عَلَىٰ وَجْهِهَآ أَوْ يَخَافُوٓا۟ أَن تُرَدَّ أَيْمَـٰنُۢ بَعْدَ أَيْمَـٰنِهِمْۗ وَٱتَّقُوا۟ ٱللَّهَ وَٱسْمَعُوا۟ۗ وَٱللَّهُ لَا يَهْدِي ٱلْقَوْمَ ٱلْفَـٰسِقِينَ ﴿108﴾
یہ اس بات کے زیادہ قریب ہے کہ وہ گواہی کو اس کی اصل صورت میں پیش کریں یا اس سے ڈریں کہ ان کی قسموں کے بعد دوسری قسمیں رد نہ کر دی جائیں، اور اللہ سے ڈرو اور سنو، اور اللہ فاسق قوم کو ہدایت نہیں دیتا۔
يَوْمَ يَجْمَعُ ٱللَّهُ ٱلرُّسُلَ فَيَقُولُ مَاذَآ أُجِبْتُمْۖ قَالُوا۟ لَا عِلْمَ لَنَآۖ إِنَّكَ أَنتَ عَلَّـٰمُ ٱلْغُيُوبِ ﴿109﴾
جس دن اللہ تمام رسولوں کو جمع کرے گا، پھر فرمائے گا: تمہیں کیا جواب دیا گیا؟ وہ کہیں گے: ہمیں کچھ علم نہیں، بے شک تو ہی غیبوں کا خوب جاننے والا ہے۔
إِذْ قَالَ ٱللَّهُ يَـٰعِيسَى ٱبْنَ مَرْيَمَ ٱذْكُرْ نِعْمَتِي عَلَيْكَ وَعَلَىٰ وَٰلِدَتِكَۖ إِذْ أَيَّدتُّكَ بِرُوحِ ٱلْقُدُسِ تُكَلِّمُ ٱلنَّاسَ فِي ٱلْمَهْدِ وَكَهْلٗاۖ وَإِذْ عَلَّمْتُكَ ٱلْكِتَـٰبَ وَٱلْحِكْمَةَ وَٱلتَّوْرَىٰةَ وَٱلْإِنجِيلَۖ وَإِذْ تَخْلُقُ مِنَ ٱلطِّينِ كَهَيْـَٔةِ ٱلطَّيْرِ بِإِذْنِي فَتَنفُخُ فِيهَا فَتَكُونُ طَيْرَۢا بِإِذْنِيۖ وَتُبْرِئُ ٱلْأَكْمَهَ وَٱلْأَبْرَصَ بِإِذْنِيۖ وَإِذْ تُخْرِجُ ٱلْمَوْتَىٰ بِإِذْنِيۖ وَإِذْ كَفَفْتُ بَنِيٓ إِسْرَٰٓءِيلَ عَنكَ إِذْ جِئْتَهُم بِٱلْبَيِّنَـٰتِ فَقَالَ ٱلَّذِينَ كَفَرُوا۟ مِنْهُمْ إِنْ هَـٰذَآ إِلَّا سِحْرٞ مُّبِينٞ ﴿110﴾
جب اللہ فرمائے گا: اے عیسیٰ ابن مریم! میری نعمت کو یاد کرو جو میں نے تم پر اور تمہاری والدہ پر کی، جب میں نے تمہیں روح القدس سے تقویت دی، تم لوگوں سے گہوارے میں اور بڑی عمر میں بات کرتے تھے، اور جب میں نے تمہیں کتاب، حکمت، تورات اور انجیل سکھائی، اور جب تم میرے حکم سے مٹی سے پرندے کی شکل بناتے، پھر اس میں پھونک مارتے تو وہ میرے حکم سے پرندہ بن جاتی، اور میرے حکم سے مادرزاد اندھے اور کوڑھی کو اچھا کرتے، اور میرے حکم سے مردوں کو نکالتے، اور جب میں نے بنی اسرائیل کو تم سے روکا، جب تم ان کے پاس نشانیاں لے کر آئے تو ان میں سے کافروں نے کہا: یہ تو کھلا جادو ہے۔
وَإِذْ أَوْحَيْتُ إِلَى ٱلْحَوَارِيِّـۧنَ أَنْ ءَامِنُوا۟ بِي وَبِرَسُولِي قَالُوٓا۟ ءَامَنَّا وَٱشْهَدْ بِأَنَّنَا مُسْلِمُونَ ﴿111﴾
اور جب میں نے حواریوں کی طرف وحی کی کہ مجھ پر اور میرے رسول پر ایمان لاؤ، تو انہوں نے کہا: ہم ایمان لائے، اور گواہ رہ کہ ہم فرمانبردار ہیں۔
إِذْ قَالَ ٱلْحَوَارِيُّونَ يَـٰعِيسَى ٱبْنَ مَرْيَمَ هَلْ يَسْتَطِيعُ رَبُّكَ أَن يُنَزِّلَ عَلَيْنَا مَآئِدَةٗ مِّنَ ٱلسَّمَآءِۖ قَالَ ٱتَّقُوا۟ ٱللَّهَ إِن كُنتُم مُّؤْمِنِينَ ﴿112﴾
جب حواریوں نے کہا: اے عیسیٰ ابن مریم! کیا آپ کا رب ہم پر آسمان سے ایک خوان نازل کر سکتا ہے؟ انہوں نے کہا: اللہ سے ڈرو اگر تم ایمان والے ہو۔
قَالُوا۟ نُرِيدُ أَن نَّأْكُلَ مِنْهَا وَتَطْمَئِنَّ قُلُوبُنَا وَنَعْلَمَ أَن قَدْ صَدَقْتَنَا وَنَكُونَ عَلَيْهَا مِنَ ٱلشَّـٰهِدِينَ ﴿113﴾
انہوں نے کہا: ہم چاہتے ہیں کہ اس میں سے کھائیں اور ہمارے دل مطمئن ہوں، اور ہم جان لیں کہ آپ نے ہم سے سچ کہا ہے، اور ہم اس پر گواہ بن جائیں۔
قَالَ عِيسَى ٱبْنُ مَرْيَمَ ٱللَّهُمَّ رَبَّنَآ أَنزِلْ عَلَيْنَا مَآئِدَةٗ مِّنَ ٱلسَّمَآءِ تَكُونُ لَنَا عِيدٗا لِّأَوَّلِنَا وَءَاخِرِنَا وَءَايَةٗ مِّنكَۖ وَٱرْزُقْنَا وَأَنتَ خَيْرُ ٱلرَّٰزِقِينَ ﴿114﴾
عیسیٰ ابن مریم نے کہا: اے اللہ! اے ہمارے رب! ہم پر آسمان سے ایک خوان نازل فرما جو ہمارے لیے ہمارے پہلے اور بعد والوں کے لیے عید ہو، اور تیری طرف سے ایک نشانی ہو، اور ہمیں رزق دے، اور تو سب سے بہتر رزق دینے والا ہے۔
قَالَ ٱللَّهُ إِنِّي مُنَزِّلُهَا عَلَيْكُمْۖ فَمَن يَكْفُرْ بَعْدُ مِنكُمْ فَإِنِّيٓ أُعَذِّبُهُۥ عَذَابٗا لَّآ أُعَذِّبُهُۥٓ أَحَدٗا مِّنَ ٱلْعَـٰلَمِينَ ﴿115﴾
اللہ نے فرمایا: بے شک میں وہ تم پر نازل کرنے والا ہوں، پھر جو تم میں سے اس کے بعد کفر کرے گا تو میں اسے ایسا عذاب دوں گا کہ دنیا والوں میں سے کسی کو ایسا عذاب نہ دوں گا۔
وَإِذْ قَالَ ٱللَّهُ يَـٰعِيسَى ٱبْنَ مَرْيَمَ أَأَنتَ قُلتَ لِلنَّاسِ ٱتَّخِذُونِي وَأُمِّيَ إِلَـٰهَيْنِ مِن دُونِ ٱللَّهِۖ قَالَ سُبْحَـٰنَكَ مَا يَكُونُ لِيٓ أَنْ أَقُولَ مَا لَيْسَ لِي بِحَقٍّۚ إِن كُنتُ قُلْتُهُۥ فَقَدْ عَلِمْتَهُۥۚ تَعْلَمُ مَا فِي نَفْسِي وَلَآ أَعْلَمُ مَا فِي نَفْسِكَۚ إِنَّكَ أَنتَ عَلَّـٰمُ ٱلْغُيُوبِ ﴿116﴾
اور جب اللہ فرمائے گا: اے عیسیٰ ابن مریم! کیا تم نے لوگوں سے کہا تھا کہ مجھے اور میری ماں کو اللہ کے سوا دو معبود بنا لو؟ تو وہ کہیں گے: تو پاک ہے، میرے لیے جائز نہیں کہ وہ بات کہوں جس کا مجھے حق نہیں، اگر میں نے کہا ہوتا تو یقیناً تو اسے جانتا، تو جانتا ہے جو میرے دل میں ہے، اور میں نہیں جانتا جو تیرے دل میں ہے، بے شک تو ہی غیبوں کا خوب جاننے والا ہے۔
مَا قُلْتُ لَهُمْ إِلَّا مَآ أَمَرْتَنِي بِهِۦٓ أَنِ ٱعْبُدُوا۟ ٱللَّهَ رَبِّي وَرَبَّكُمۖ وَكُنتُ عَلَيْهِمْ شَهِيدٗا مَّا دُمْتُ فِيهِمۖ فَلَمَّا تَوَفَّيْتَنِي كُنتَ أَنتَ ٱلرَّقِيبَ عَلَيْهِمۖ وَأَنتَ عَلَىٰ كُلِّ شَيْءٖ شَهِيدٌ ﴿117﴾
میں نے ان سے وہی بات کہی جو تو نے مجھے حکم دیا تھا، کہ اللہ کی عبادت کرو جو میرا اور تمہارا رب ہے، اور جب تک میں ان میں رہا ان پر گواہ رہا، پھر جب تو نے مجھے اٹھا لیا تو تو ہی ان پر نگہبان تھا، اور تو ہر چیز پر گواہ ہے۔
إِن تُعَذِّبْهُمْ فَإِنَّهُمْ عِبَادُكَۖ وَإِن تَغْفِرْ لَهُمْ فَإِنَّكَ أَنتَ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿118﴾
اگر تو انہیں عذاب دے تو وہ تیرے بندے ہیں، اور اگر تو انہیں بخش دے تو بے شک تو ہی غالب، حکمت والا ہے۔
قَالَ ٱللَّهُ هَـٰذَا يَوْمُ يَنفَعُ ٱلصَّـٰدِقِينَ صِدْقُهُمْۚ لَهُمْ جَنَّـٰتٞ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُ خَـٰلِدِينَ فِيهَآ أَبَدٗاۚ رَضِيَ ٱللَّهُ عَنْهُمْ وَرَضُوا۟ عَنْهُۚ ذَٰلِكَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿119﴾
اللہ فرمائے گا: یہ وہ دن ہے جس میں سچوں کو ان کا سچ فائدہ دے گا، ان کے لیے ایسی جنتیں ہیں جن کے نیچے نہریں بہتی ہیں، جن میں وہ ہمیشہ رہیں گے، اللہ ان سے راضی ہوا اور وہ اللہ سے راضی ہوئے، یہی بڑی کامیابی ہے۔
لِلَّهِ مُلْكُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَمَا فِيهِنَّۚ وَهُوَ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٌ ﴿120﴾
اللہ ہی کے لیے ہے آسمانوں اور زمین اور جو کچھ ان میں ہے سب کی بادشاہی، اور وہ ہر چیز پر قدرت رکھنے والا ہے۔
