26 پارہ


سورة الأحقاف
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
حمٓ ﴿1﴾
حٰمٓ
تَنزِيلُ ٱلْكِتَـٰبِ مِنَ ٱللَّهِ ٱلْعَزِيزِ ٱلْحَكِيمِ ﴿2﴾
اس کتاب کا نزول اللہ کی طرف سے ہے جو زبردست، حکمت والا ہے۔
مَا خَلَقْنَا ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ وَمَا بَيْنَهُمَآ إِلَّا بِٱلْحَقِّ وَأَجَلٖ مُّسَمّٗىۚ وَٱلَّذِينَ كَفَرُوا۟ عَمَّآ أُنذِرُوا۟ مُعْرِضُونَ ﴿3﴾
ہم نے آسمانوں اور زمین کو اور جو کچھ ان کے درمیان ہے حق کے ساتھ اور ایک مقررہ مدت کے لیے پیدا کیا ہے، اور جنہوں نے کفر کیا وہ اس چیز سے منہ موڑ رہے ہیں جس سے انہیں ڈرایا گیا ہے۔
قُلْ أَرَءَيْتُم مَّا تَدْعُونَ مِن دُونِ ٱللَّهِ أَرُونِي مَاذَا خَلَقُوا۟ مِنَ ٱلْأَرْضِ أَمْ لَهُمْ شِرْكٞ فِي ٱلسَّمَـٰوَٰتِ ۖ ٱئْتُونِي بِكِتَـٰبٖ مِّن قَبْلِ هَـٰذَآ أَوْ أَثَـٰرَةٖ مِّنْ عِلْمٍ إِن كُنتُمْ صَـٰدِقِينَ ﴿4﴾
کہہ دو: کیا تم نے ان چیزوں کو دیکھا جنہیں تم اللہ کے سوا پکارتے ہو؟ مجھے دکھاؤ کہ انہوں نے زمین میں کیا پیدا کیا ہے، یا آسمانوں میں ان کا کوئی حصہ ہے؟ میرے پاس اس سے پہلے کی کوئی کتاب یا علم کی کوئی باقیات لاؤ اگر تم سچے ہو۔
وَمَنْ أَضَلُّ مِمَّن يَدْعُوا۟ مِن دُونِ ٱللَّهِ مَن لَّا يَسْتَجِيبُ لَهُۥٓ إِلَىٰ يَوْمِ ٱلْقِيَـٰمَةِ وَهُمْ عَن دُعَآئِهِمْ غَـٰفِلُونَ ﴿5﴾
اور اس سے زیادہ گمراہ کون ہو سکتا ہے جو اللہ کے سوا ایسے کو پکارتا ہے جو قیامت تک اسے جواب نہیں دے سکتا، اور وہ ان کی دعا سے غافل ہیں؟
وَإِذَا حُشِرَ ٱلنَّاسُ كَانُوا۟ لَهُمْ أَعْدَآءٗ وَكَانُوا۟ بِعِبَادَتِهِمْ كَـٰفِرِينَ ﴿6﴾
اور جب لوگ جمع کیے جائیں گے تو وہ ان کے دشمن ہوں گے اور ان کی عبادت کا انکار کریں گے۔
وَإِذَا تُتْلَىٰ عَلَيْهِمْ ءَايَـٰتُنَا بَيِّنَـٰتٖ قَالَ ٱلَّذِينَ كَفَرُوا۟ لِلْحَقِّ لَمَّا جَآءَهُمْ هَـٰذَا سِحْرٞ مُّبِينٌ ﴿7﴾
اور جب ان پر ہماری واضح آیات پڑھی جاتی ہیں تو جو لوگ کفر کرتے ہیں وہ حق کے بارے میں جب وہ ان کے پاس آتا ہے کہتے ہیں: یہ تو کھلا جادو ہے۔
أَمْ يَقُولُونَ ٱفْتَرَىٰهُ ۚ قُلْ إِنِ ٱفْتَرَيْتُهُ فَلَا تَمْلِكُونَ لِي مِنَ ٱللَّهِ شَيْـٔٗا ۖ هُوَ أَعْلَمُ بِمَا تُفِيضُونَ فِيهِ ۚ كَفَىٰ بِهِۦ شَهِيدَۢا بَيْنِي وَبَيْنَكُمْ ۖ وَهُوَ ٱلْغَفُورُ ٱلرَّحِيمُ ﴿8﴾
کیا وہ کہتے ہیں: اس نے اسے خود گھڑ لیا ہے؟ کہہ دو: اگر میں نے اسے گھڑ لیا ہے تو تم میرے لیے اللہ سے کچھ بھی اختیار نہیں رکھتے۔ وہ خوب جانتا ہے جو باتیں تم اس میں کرتے ہو۔ میرے اور تمہارے درمیان وہی گواہ کافی ہے، اور وہی بخشنے والا، رحم کرنے والا ہے۔
قُلْ مَا كُنتُ بِدْعٗا مِّنَ ٱلرُّسُلِ وَمَآ أَدْرِي مَا يُفْعَلُ بِي وَلَا بِكُمْ ۖ إِنْ أَتَّبِعُ إِلَّا مَا يُوحَىٰٓ إِلَيَّ ۖ وَمَآ أَنَا۠ إِلَّا نَذِيرٞ مُّبِينٞ ﴿9﴾
کہہ دو: میں رسولوں میں کوئی نیا نہیں ہوں، اور مجھے نہیں معلوم کہ میرے ساتھ کیا کیا جائے گا اور تمہارے ساتھ کیا۔ میں تو صرف اس کی پیروی کرتا ہوں جو میری طرف وحی کی جاتی ہے، اور میں تو صرف واضح طور پر ڈرانے والا ہوں۔
قُلْ أَرَءَيْتُمْ إِن كَانَ مِنْ عِندِ ٱللَّهِ وَكَفَرْتُم بِهِۦ وَشَهِدَ شَاهِدٞ مِّنۢ بَنِيٓ إِسْرَٰٓءِيلَ عَلَىٰ مِثْلِهِۦ فَـَٔامَنَ وَٱسْتَكْبَرْتُمْ ۚ إِنَّ ٱللَّهَ لَا يَهْدِي ٱلْقَوْمَ ٱلظَّـٰلِمِينَ ﴿10﴾
کہہ دو: کیا تم نے دیکھا اگر یہ اللہ کی طرف سے ہو اور تم نے اس کا انکار کیا، اور بنی اسرائیل کے ایک گواہ نے اس جیسے (کلام) کی گواہی دی اور ایمان لے آیا، اور تم نے تکبر کیا؟ بے شک اللہ ظالم قوم کو ہدایت نہیں دیتا۔
وَقَالَ ٱلَّذِينَ كَفَرُوا۟ لِلَّذِينَ ءَامَنُوا۟ لَوْ كَانَ خَيْرٗا مَّا سَبَقُونَآ إِلَيْهِ ۚ وَإِذْ لَمْ يَهْتَدُوا۟ بِهِۦ فَسَيَقُولُونَ هَـٰذَآ إِفْكٞ قَدِيمٞ ﴿11﴾
اور کافروں نے ایمان والوں سے کہا: اگر یہ (کلام) اچھا ہوتا تو وہ اس میں ہم سے سبقت نہ لے جاتے، اور جب وہ اس سے ہدایت نہ پا سکے تو کہنے لگے: یہ تو پرانا جھوٹ ہے۔
وَمِن قَبْلِهِۦ كِتَـٰبُ مُوسَىٰٓ إِمَامٗا وَرَحْمَةٗۚ وَهَـٰذَا كِتَـٰبٞ مُّصَدِّقٞ لِّسَانًا عَرَبِيّٗا لِّيُنذِرَ ٱلَّذِينَ ظَلَمُوا۟ وَبُشْرَىٰ لِلْمُحْسِنِينَ ﴿12﴾
اور اس سے پہلے موسیٰ کی کتاب تھی، جو رہنما اور رحمت تھی، اور یہ کتاب ہے جو تصدیق کرنے والی ہے، عربی زبان میں، تاکہ وہ ظالموں کو ڈرائے اور نیکوکاروں کو خوشخبری دے۔
إِنَّ ٱلَّذِينَ قَالُوا۟ رَبُّنَا ٱللَّهُ ثُمَّ ٱسْتَقَـٰمُوا۟ فَلَا خَوْفٌ عَلَيْهِمْ وَلَا هُمْ يَحْزَنُونَ ﴿13﴾
بے شک جنہوں نے کہا: ہمارا رب اللہ ہے، پھر اس پر قائم رہے، ان پر نہ کوئی خوف ہوگا اور نہ وہ غمگین ہوں گے۔
أُو۟لَـٰٓئِكَ أَصْحَـٰبُ ٱلْجَنَّةِ خَـٰلِدِينَ فِيهَا جَزَآءًۢ بِمَا كَانُوا۟ يَعْمَلُونَ ﴿14﴾
یہی لوگ جنت والے ہیں، اس میں ہمیشہ رہیں گے، یہ ان کے اعمال کا بدلہ ہے جو وہ کرتے تھے۔
وَوَصَّيْنَا ٱلْإِنسَـٰنَ بِوَٰلِدَيْهِ إِحْسَـٰنٗا ۖ حَمَلَتْهُ أُمُّهُ كُرْهٗا وَوَضَعَتْهُ كُرْهٗا ۖ وَحَمْلُهُۥ وَفِصَـٰلُهُۥ ثَلَـٰثُونَ شَهْرًا ۚ حَتَّىٰٓ إِذَا بَلَغَ أَشُدَّهُۥ وَبَلَغَ أَرْبَعِينَ سَنَةٗ قَالَ رَبِّ أَوْزِعْنِيٓ أَنْ أَشْكُرَ نِعْمَتَكَ ٱلَّتِيٓ أَنْعَمْتَ عَلَيَّ وَعَلَىٰ وَٰلِدَيَّ وَأَنْ أَعْمَلَ صَـٰلِحٗا تَرْضَىٰهُ وَأَصْلِحْ لِي فِي ذُرِّيَّتِيٓ ۖ إِنِّي تُبْتُ إِلَيْكَ وَإِنِّي مِنَ ٱلْمُسْلِمِينَ ﴿15﴾
اور ہم نے انسان کو اس کے والدین کے ساتھ حسنِ سلوک کی وصیت کی، اس کی ماں نے اسے تکلیف سے اٹھایا اور تکلیف سے جنا، اور اس کا حمل اور دودھ چھڑانا تیس مہینے ہے۔ یہاں تک کہ جب وہ اپنی پختگی کو پہنچا اور چالیس سال کا ہو گیا، تو کہا: اے میرے رب! مجھے توفیق دے کہ میں تیری اس نعمت کا شکر ادا کروں جو تو نے مجھ پر اور میرے والدین پر کی، اور ایسا نیک عمل کروں جس سے تو راضی ہو، اور میری اولاد میں بھی اصلاح فرما، بے شک میں تیری طرف رجوع کرتا ہوں اور میں فرمانبرداروں میں سے ہوں۔
أُو۟لَـٰٓئِكَ ٱلَّذِينَ نَتَقَبَّلُ عَنْهُمْ أَحْسَنَ مَا عَمِلُوا۟ وَنَتَجَاوَزُ عَن سَيِّـَٔاتِهِمۡ فِيٓ أَصْحَـٰبِ ٱلْجَنَّةِ ۖ وَعْدَ ٱلصِّدْقِ ٱلَّذِي كَانُوا۟ يُوعَدُونَ ﴿16﴾
یہی وہ لوگ ہیں جن کے نیک اعمال ہم قبول کرتے ہیں اور ان کی برائیوں سے درگزر کرتے ہیں، وہ جنت والوں میں ہوں گے، یہ سچا وعدہ ہے جو ان سے کیا جاتا تھا۔
وَٱلَّذِي قَالَ لِوَٰلِدَيْهِ أُفّٖ لَّكُمَآ أَتَعِدَانِنِيٓ أَنْ أُخْرَجَ وَقَدْ خَلَتِ ٱلْقُرُونُ مِن قَبْلِي وَهُمَا يَسْتَغِيثَانِ ٱللَّهَ وَيْلَكَ ءَامِنْ إِنَّ وَعْدَ ٱللَّهِ حَقّٞ ۖ فَيَقُولُ مَا هَـٰذَآ إِلَّآ أَسَـٰطِيرُ ٱلْأَوَّلِينَ ﴿17﴾
اور وہ شخص جس نے اپنے والدین سے کہا: اُف تم دونوں پر! کیا تم مجھے وعدہ دیتے ہو کہ میں دوبارہ زندہ کیا جاؤں گا، حالانکہ مجھ سے پہلے کئی نسلیں گزر چکی ہیں؟ اور وہ دونوں اللہ سے فریاد کرتے ہیں: ہائے افسوس! ایمان لا، بے شک اللہ کا وعدہ سچا ہے۔ مگر وہ کہتا ہے: یہ تو صرف پہلے لوگوں کی کہانیاں ہیں۔
أُو۟لَـٰٓئِكَ ٱلَّذِينَ حَقَّ عَلَيْهِمُ ٱلْقَوْلُ فِيٓ أُمَمٖ قَدْ خَلَتۡ مِن قَبْلِهِم مِّنَ ٱلْجِنِّ وَٱلْإِنسِ ۖ إِنَّهُمْ كَانُوا۟ خَـٰسِرِينَ ﴿18﴾
یہی وہ لوگ ہیں جن پر اللہ کا فیصلہ ثابت ہو چکا ہے، ان امتوں میں جو ان سے پہلے جنوں اور انسانوں میں گزر چکی ہیں، بے شک وہی نقصان اٹھانے والے ہیں۔
وَلِكُلّٖ دَرَجَـٰتٞ مِّمَّا عَمِلُوا۟ ۖ وَلِيُوَفِّيَهُمْ أَعْمَـٰلَهُمْ وَهُمْ لَا يُظْلَمُونَ ﴿19﴾
اور ہر ایک کے لیے ان کے اعمال کے مطابق درجے ہیں، تاکہ اللہ انہیں ان کے اعمال کا پورا بدلہ دے، اور ان پر ظلم نہیں کیا جائے گا۔
وَيَوْمَ يُعْرَضُ ٱلَّذِينَ كَفَرُوا۟ عَلَى ٱلنَّارِ أَذْهَبْتُمْ طَيِّبَـٰتِكُمْ فِي حَيَاتِكُمُ ٱلدُّنْيَا وَٱسْتَمْتَعْتُم بِهَا فَٱلْيَوْمَ تُجْزَوْنَ عَذَابَ ٱلْهُونِ بِمَا كُنتُمْ تَسْتَكْبِرُونَ فِي ٱلْأَرْضِ بِغَيْرِ ٱلْحَقِّ وَبِمَا كُنتُمْ تَفْسُقُونَ ﴿20﴾
اور جس دن کافر آگ کے سامنے پیش کیے جائیں گے (کہا جائے گا): تم نے اپنی پاکیزہ چیزیں دنیا کی زندگی میں ہی لے لیں اور ان سے لطف اٹھا لیا، پس آج تمہیں ذلت کا عذاب دیا جائے گا، اس وجہ سے کہ تم زمین میں ناحق تکبر کرتے تھے اور نافرمانی کرتے تھے۔
وَٱذْكُرْ أَخَا عَادٍ إِذْ أَنذَرَ قَوْمَهُۥ بِٱلْأَحْقَافِ وَقَدْ خَلَتِ ٱلنُّذُرُ مِنۢ بَيْنِ يَدَيْهِ وَمِنْ خَلْفِهِۦٓ أَلَّا تَعْبُدُوٓا۟ إِلَّا ٱللَّهَ ۚ إِنِّيٓ أَخَافُ عَلَيْكُمْ عَذَابَ يَوْمٍ عَظِيمٖ ﴿21﴾
اور عاد کے بھائی کو یاد کرو جب اس نے اپنی قوم کو احقاف میں ڈرایا، اور اس سے پہلے اور بعد میں بھی ڈرانے والے گزر چکے تھے، کہ تم اللہ کے سوا کسی کی عبادت نہ کرو، بے شک میں تم پر ایک بڑے دن کے عذاب سے ڈرتا ہوں۔
قَالُوا۟ أَجِئْتَنَا لِتَأْفِكَنَا عَنْ ءَالِهَتِنَا فَأْتِنَا بِمَا تَعِدُنَآ إِن كُنتَ مِنَ ٱلصَّـٰدِقِينَ ﴿22﴾
انہوں نے کہا: کیا تم ہمارے پاس اس لیے آئے ہو کہ ہمیں ہمارے معبودوں سے پھیر دو؟ تو لے آؤ وہ (عذاب) جس کا تم ہم سے وعدہ کرتے ہو اگر تم سچے ہو۔
قَالَ إِنَّمَا ٱلْعِلْمُ عِندَ ٱللَّهِ وَأُبَلِّغُكُم مَّآ أُرْسِلْتُ بِهِۦ وَلَـٰكِنِّيٓ أَرَىٰكُمْ قَوْمٗا تَجْهَلُونَ ﴿23﴾
اس نے کہا: علم تو اللہ ہی کے پاس ہے، اور میں تو تمہیں وہ بات پہنچا رہا ہوں جس کے ساتھ مجھے بھیجا گیا ہے، لیکن میں دیکھ رہا ہوں کہ تم جاہل قوم ہو۔
فَلَمَّا رَأَوْهُ عَارِضٗا مُّسْتَقْبِلَ أَوْدِيَتِهِمْ قَالُوا۟ هَـٰذَا عَارِضٞ مُّمطِرُنَاۚ بَلْ هُوَ مَا ٱسْتَعْجَلْتُم بِهِۦۖ رِيحٞ فِيهَا عَذَابٌ أَلِيمٞ ﴿24﴾
پھر جب انہوں نے اسے بادل کی صورت میں اپنی وادیوں کی طرف آتے دیکھا تو کہنے لگے: یہ بادل ہے جو ہم پر بارش برسائے گا۔ (نہیں!) بلکہ یہ وہی چیز ہے جسے تم جلدی مانگ رہے تھے، ایک آندھی ہے جس میں دردناک عذاب ہے۔
تُدَمِّرُ كُلَّ شَيْءٍۭ بِأَمْرِ رَبِّهَا فَأَصْبَحُوا۟ لَا يُرَىٰٓ إِلَّا مَسَـٰكِنُهُمْۚ كَذَٰلِكَ نَجْزِي ٱلْقَوْمَ ٱلْمُجْرِمِينَ ﴿25﴾
وہ اپنے رب کے حکم سے ہر چیز کو تباہ کر رہی تھی، تو وہ ایسے ہو گئے کہ ان کے رہنے کی جگہوں کے سوا کچھ نظر نہ آتا تھا۔ ہم مجرم قوم کو اسی طرح سزا دیتے ہیں۔
وَلَقَدْ مَكَّنَّـٰهُمْ فِيمَآ إِن مَّكَّنَّـٰكُمْ فِيهِ وَجَعَلْنَا لَهُمْ سَمْعٗا وَأَبْصَـٰرٗا وَأَفْـِٔدَةٗ فَمَآ أَغْنَىٰ عَنْهُمْ سَمْعُهُمْ وَلَآ أَبْصَـٰرُهُمْ وَلَآ أَفْـِٔدَتُهُم مِّن شَيْءٍ إِذْ كَانُوا۟ يَجْحَدُونَ بِـَٔايَـٰتِ ٱللَّهِ وَحَاقَ بِهِم مَّا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ ﴿26﴾
اور ہم نے انہیں اس چیز میں قدرت دی تھی جس میں ہم نے تمہیں قدرت نہیں دی، اور ہم نے انہیں کان، آنکھیں اور دل دیے تھے، لیکن ان کے کان، آنکھیں اور دل ان کے کچھ کام نہ آئے جب وہ اللہ کی آیات کا انکار کرتے رہے، اور جس چیز کا وہ مذاق اڑاتے تھے وہی ان پر آ پڑی۔
وَلَقَدْ أَهْلَكْنَا مَا حَوْلَكُم مِّنَ ٱلْقُرَىٰ وَصَرَّفْنَا ٱلْءَايَـٰتِ لَعَلَّهُمْ يَرْجِعُونَ ﴿27﴾
اور ہم نے تمہارے اردگرد کی بستیاں ہلاک کر دیں، اور ہم نے بار بار نشانیاں بیان کیں تاکہ وہ رجوع کریں۔
فَلَوْلَا نَصَرَهُمُ ٱلَّذِينَ ٱتَّخَذُوا۟ مِن دُونِ ٱللَّهِ قُرْبَانًا ءَالِهَةٗۢ ۖ بَلْ ضَلُّوا۟ عَنْهُمْۚ وَذَٰلِكَ إِفْكُهُمْ وَمَا كَانُوا۟ يَفْتَرُونَ ﴿28﴾
تو جنہیں انہوں نے اللہ کے سوا قربت کے لیے معبود بنایا تھا، انہوں نے ان کی مدد کیوں نہ کی؟ بلکہ وہ ان سے گم ہو گئے، اور یہ ان کا جھوٹ تھا جو وہ گھڑتے تھے۔
وَإِذْ صَرَفْنَآ إِلَيْكَ نَفَرٗا مِّنَ ٱلْجِنِّ يَسْتَمِعُونَ ٱلْقُرْءَانَ فَلَمَّا حَضَرُوهُ قَالُوٓا۟ أَنصِتُوا۟ ۖ فَلَمَّا قُضِيَ وَلَّوْا۟ إِلَىٰ قَوْمِهِم مُّنذِرِينَ ﴿29﴾
اور جب ہم نے جنوں کے ایک گروہ کو آپ کی طرف متوجہ کیا جو قرآن سن رہے تھے، پھر جب وہ اس کے پاس پہنچے تو کہنے لگے: خاموش رہو۔ پھر جب وہ مکمل ہو گیا تو وہ اپنی قوم کی طرف واپس گئے، انہیں ڈرانے والے بن کر۔
قَالُوا۟ يَـٰقَوْمَنَآ إِنَّا سَمِعْنَا كِتَـٰبٗا أُنزِلَ مِنۢ بَعْدِ مُوسَىٰ مُصَدِّقٗا لِّمَا بَيْنَ يَدَيْهِ يَهْدِيٓ إِلَى ٱلْحَقِّ وَإِلَىٰ طَرِيقٖ مُّسْتَقِيمٖ ﴿30﴾
انہوں نے کہا: اے ہماری قوم! ہم نے ایک کتاب سنی ہے جو موسیٰ کے بعد نازل ہوئی ہے، جو اپنے سے پہلے کی تصدیق کرتی ہے، حق اور سیدھے راستے کی طرف رہنمائی کرتی ہے۔
يَـٰقَوْمَنَآ أَجِيبُوا۟ دَاعِيَ ٱللَّهِ وَءَامِنُوا۟ بِهِۦ يَغْفِرْ لَكُم مِّن ذُنُوبِكُمْ وَيُجِرْكُم مِّنْ عَذَابٍ أَلِيمٖ ﴿31﴾
اے ہماری قوم! اللہ کے بلانے والے کی بات مان لو اور اس پر ایمان لے آؤ، وہ تمہارے گناہوں کو بخش دے گا اور تمہیں دردناک عذاب سے بچا لے گا۔
وَمَن لَّا يُجِبْ دَاعِيَ ٱللَّهِ فَلَيْسَ بِمُعْجِزٖ فِي ٱلْأَرْضِ وَلَيْسَ لَهُۥ مِن دُونِهِۦٓ أَوْلِيَآءُۚ أُو۟لَـٰٓئِكَ فِي ضَلَـٰلٖ مُّبِينٍ ﴿32﴾
اور جو اللہ کے بلانے والے کی بات نہ مانے وہ زمین میں اللہ کو عاجز نہیں کر سکتا، اور اس کے لیے اللہ کے سوا کوئی مددگار نہیں، یہی لوگ کھلی گمراہی میں ہیں۔
أَوَلَمْ يَرَوْا۟ أَنَّ ٱللَّهَ ٱلَّذِي خَلَقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ وَلَمْ يَعْيَ بِخَلْقِهِنَّ بِقَـٰدِرٍ عَلَىٰٓ أَن يُحْـِۧيَ ٱلْمَوْتَىٰۚ بَلَىٰٓ إِنَّهُۥ عَلَىٰ كُلِّ شَيْءٖ قَدِيرٞ ﴿33﴾
کیا انہوں نے نہیں دیکھا کہ اللہ جس نے آسمانوں اور زمین کو پیدا کیا اور ان کی پیدائش سے تھکا نہیں، وہ مردوں کو زندہ کرنے پر قادر ہے؟ کیوں نہیں! بے شک وہ ہر چیز پر قدرت رکھتا ہے۔
وَيَوْمَ يُعْرَضُ ٱلَّذِينَ كَفَرُوا۟ عَلَى ٱلنَّارِ أَلَيْسَ هَـٰذَا بِٱلْحَقِّۚ قَالُوا۟ بَلَىٰ وَرَبِّنَاۚ قَالَ فَذُوقُوا۟ ٱلْعَذَابَ بِمَا كُنتُمْ تَكْفُرُونَ ﴿34﴾
اور جس دن کافر آگ کے سامنے پیش کیے جائیں گے (کہا جائے گا): کیا یہ حق نہیں ہے؟ وہ کہیں گے: کیوں نہیں! ہمارے رب کی قسم! (اللہ) کہے گا: پس اب اس عذاب کا مزہ چکھو جو تم کفر کرتے تھے۔
فَٱصْبِرْ كَمَا صَبَرَ أُو۟لُوا۟ ٱلْعَزْمِ مِنَ ٱلرُّسُلِ وَلَا تَسْتَعْجِلْ لَّهُمْۚ كَأَنَّهُمْ يَوْمَ يَرَوْنَ مَا يُوعَدُونَ لَمْ يَلْبَثُوٓا۟ إِلَّا سَاعَةٗ مِّن نَّهَارِۚ بَلَـٰغٞۚ فَهَلْ يُهْلَكُ إِلَّا ٱلْقَوْمُ ٱلْفَـٰسِقُونَ ﴿35﴾
پس صبر کرو جیسے اولوالعزم رسولوں نے صبر کیا، اور ان کے لیے جلدی نہ کرو۔ جس دن وہ اس چیز کو دیکھیں گے جس کا ان سے وعدہ کیا جا رہا ہے، تو انہیں ایسا لگے گا کہ وہ دن کی صرف ایک گھڑی ٹھہرے تھے۔ یہ ایک پیغام ہے، تو کیا نافرمان قوم کے سوا کوئی ہلاک ہوگا؟
سورة محمد
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
ٱلَّذِينَ كَفَرُوا۟ وَصَدُّوا۟ عَن سَبِيلِ ٱللَّهِ أَضَلَّ أَعْمَـٰلَهُمْ ﴿1﴾
جن لوگوں نے کفر کیا اور اللہ کے راستے سے روکا، اللہ نے ان کے اعمال ضائع کر دیے۔
وَٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ وَءَامَنُوا۟ بِمَا نُزِّلَ عَلَىٰ مُحَمَّدٖ وَهُوَ ٱلْحَقُّ مِن رَّبِّهِمْ كَفَّرَ عَنْهُمْ سَيِّـَٔاتِهِمْ وَأَصْلَحَ بَالَهُمْ ﴿2﴾
اور جنہوں نے ایمان لایا اور نیک عمل کیے اور اس چیز پر ایمان لائے جو محمد پر نازل کی گئی، اور وہ ان کے رب کی طرف سے حق ہے، اللہ نے ان کی برائیاں دور کر دیں اور ان کا حال درست کر دیا۔
ذَٰلِكَ بِأَنَّ ٱلَّذِينَ كَفَرُوا۟ ٱتَّبَعُوا۟ ٱلْبَاطِلَ وَأَنَّ ٱلَّذِينَ ءَامَنُوا۟ ٱتَّبَعُوا۟ ٱلْحَقَّ مِن رَّبِّهِمۖۚ كَذَٰلِكَ يَضْرِبُ ٱللَّهُ لِلنَّاسِ أَمْثَـٰلَهُمْ ﴿3﴾
یہ اس لیے کہ کافروں نے باطل کی پیروی کی، اور ایمان والوں نے اپنے رب کی طرف سے حق کی پیروی کی۔ اسی طرح اللہ لوگوں کے لیے ان کی مثالیں بیان کرتا ہے۔
فَإِذَا لَقِيتُمُ ٱلَّذِينَ كَفَرُوا۟ فَضَرْبَ ٱلرِّقَابِ حَتَّىٰٓ إِذَآ أَثْخَنتُمُوهُمْ فَشُدُّوا۟ ٱلْوَثَاقَ فَإِمَّا مَنّٗا بَعْدُ وَإِمَّا فِدَآءً حَتَّىٰ تَضَعَ ٱلْحَرْبُ أَوْزَارَهَاۚ ذَٰلِكَ وَلَوْ يَشَآءُ ٱللَّهُ لَٱنتَصَرَ مِنْهُمْ وَلَـٰكِن لِّيَبْلُوَ بَعْضَكُم بِبَعْضٖۗ وَٱلَّذِينَ قُتِلُوا۟ فِي سَبِيلِ ٱللَّهِ فَلَن يُضِلَّ أَعْمَـٰلَهُمْ ﴿4﴾
پس جب تم کافروں سے جنگ میں ملو تو گردنیں مارو، یہاں تک کہ جب تم انہیں خوب کچل لو تو قیدی مضبوط باندھو، پھر یا احسان کر کے چھوڑ دو یا فدیہ لے لو، یہاں تک کہ جنگ اپنے بوجھ اتار دے۔ یہ حکم ہے، اور اگر اللہ چاہتا تو خود ان سے بدلہ لے لیتا، لیکن تاکہ وہ تمہیں ایک دوسرے کے ذریعے آزمائے۔ اور جو لوگ اللہ کی راہ میں قتل کیے گئے، اللہ ان کے اعمال کو ہرگز ضائع نہیں کرے گا۔
سَيَهْدِيهِمْ وَيُصْلِحُ بَالَهُمْ ﴿5﴾
وہ انہیں ہدایت دے گا اور ان کا حال درست کرے گا۔
وَيُدْخِلُهُمُ ٱلْجَنَّةَ ٱلَّتِي عَرَّفَهَا لَهُمْ ﴿6﴾
اور انہیں اس جنت میں داخل کرے گا جس سے وہ پہلے ہی واقف کرا دیے گئے ہیں۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ إِن تَنصُرُوا۟ ٱللَّهَ يَنصُرْكُمْ وَيُثَبِّتْ أَقْدَامَكُمْ ﴿7﴾
اے ایمان والو! اگر تم اللہ کی مدد کرو گے تو وہ تمہاری مدد کرے گا اور تمہارے قدم جما دے گا۔
وَٱلَّذِينَ كَفَرُوا۟ فَتَعْسٗا لَّهُمْ وَأَضَلَّ أَعْمَـٰلَهُمْ ﴿8﴾
اور جنہوں نے کفر کیا، ان کے لیے تباہی ہے، اور اللہ نے ان کے اعمال ضائع کر دیے۔
ذَٰلِكَ بِأَنَّهُمْ كَرِهُوا۟ مَآ أَنزَلَ ٱللَّهُ فَأَحْبَطَ أَعْمَـٰلَهُمْ ﴿9﴾
یہ اس لیے کہ انہوں نے اس چیز کو ناپسند کیا جو اللہ نے نازل کی، تو اللہ نے ان کے اعمال ضائع کر دیے۔
أَفَلَمْ يَسِيرُوا۟ فِي ٱلْأَرْضِ فَيَنظُرُوا۟ كَيْفَ كَانَ عَـٰقِبَةُ ٱلَّذِينَ مِن قَبْلِهِمْ ۖ دَمَّرَ ٱللَّهُ عَلَيْهِمْ ۖ وَلِلْكَـٰفِرِينَ أَمْثَـٰلُهَا ﴿10﴾
کیا انہوں نے زمین میں سفر نہیں کیا تاکہ دیکھیں کہ ان سے پہلے لوگوں کا انجام کیسا ہوا؟ اللہ نے ان پر تباہی نازل کی، اور کافروں کے لیے اسی طرح کی مثالیں ہیں۔
ذَٰلِكَ بِأَنَّ ٱللَّهَ مَوْلَى ٱلَّذِينَ ءَامَنُوا۟ وَأَنَّ ٱلْكَـٰفِرِينَ لَا مَوْلَىٰ لَهُمْ ﴿11﴾
یہ اس لیے کہ اللہ ایمان والوں کا مددگار ہے، اور کافروں کا کوئی مددگار نہیں۔
إِنَّ ٱللَّهَ يُدْخِلُ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ جَنَّـٰتٖ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُ ۖ وَٱلَّذِينَ كَفَرُوا۟ يَتَمَتَّعُونَ وَيَأْكُلُونَ كَمَا تَأْكُلُ ٱلْأَنْعَـٰمُ وَٱلنَّارُ مَثْوٗى لَّهُمْ ﴿12﴾
بے شک اللہ ایمان والوں کو ان باغات میں داخل کرے گا جن کے نیچے نہریں بہتی ہیں، اور کافر فائدہ اٹھاتے ہیں اور کھاتے ہیں جیسے جانور کھاتے ہیں، اور ان کا ٹھکانہ آگ ہے۔
وَكَأَيِّن مِّن قَرْيَةٍ هِيَ أَشَدُّ قُوَّةٗ مِّن قَرْيَتِكَ ٱلَّتِيٓ أَخْرَجَتْكَ أَهْلَكْنَـٰهُمْ فَلَا نَـٰصِرَ لَهُمْ ﴿13﴾
اور کتنی ہی بستیاں تھیں جو تمہاری بستی سے زیادہ طاقتور تھیں جس نے تمہیں نکالا، ہم نے انہیں ہلاک کر دیا، اور ان کے لیے کوئی مددگار نہ تھا۔
أَفَمَن كَانَ عَلَىٰ بَيِّنَةٖ مِّن رَّبِّهِۦ كَمَن زُيِّنَ لَهُۥ سُوٓءُ عَمَلِهِۦ وَٱتَّبَعُوٓا۟ أَهْوَآءَهُمْ ﴿14﴾
کیا وہ شخص جو اپنے رب کی طرف سے واضح دلیل پر ہے، اس جیسا ہو سکتا ہے جس کے برے عمل اسے خوشنما دکھائے گئے ہوں اور وہ اپنی خواہشات کی پیروی کرتا ہو؟
مَّثَلُ ٱلْجَنَّةِ ٱلَّتِي وُعِدَ ٱلْمُتَّقُونَ ۖ فِيهَآ أَنْهَـٰرٞ مِّن مَّآءٍ غَيْرِ ءَاسِنٖ وَأَنْهَـٰرٞ مِّن لَّبَنٖ لَّمْ يَتَغَيَّرْ طَعْمُهُۥ وَأَنْهَـٰرٞ مِّنْ خَمْرٖ لَّذَّةٖ لِّلشَّـٰرِبِينَ وَأَنْهَـٰرٞ مِّنْ عَسَلٖ مُّصَفّٗى ۖ وَلَهُمْ فِيهَا مِن كُلِّ ٱلثَّمَرَٰتِ وَمَغْفِرَةٞ مِّن رَّبِّهِمْ ۖ كَمَنْ هُوَ خَـٰلِدٞ فِي ٱلنَّارِ وَسُقُوا۟ مَآءً حَمِيمٗا فَقَطَّعَ أَمْعَآءَهُمْ ﴿15﴾
اس جنت کی مثال جس کا وعدہ پرہیزگاروں سے کیا گیا ہے: اس میں پانی کی نہریں ہیں جو بدبو دار نہیں، دودھ کی نہریں ہیں جس کا ذائقہ نہیں بدلا، شراب کی نہریں ہیں جو پینے والوں کے لیے لذت بخش ہے، اور صاف شہد کی نہریں ہیں۔ اور ان کے لیے اس میں ہر قسم کے پھل ہوں گے اور ان کے رب کی طرف سے مغفرت۔ کیا یہ اس شخص کی طرح ہو سکتا ہے جو ہمیشہ آگ میں رہے گا اور جسے کھولتا ہوا پانی پلایا جائے گا جو اس کی آنتیں کاٹ دے گا؟
وَمِنْهُم مَّن يَسْتَمِعُ إِلَيْكَ حَتَّىٰٓ إِذَا خَرَجُوا۟ مِنْ عِندِكَ قَالُوا۟ لِلَّذِينَ أُوتُوا۟ ٱلْعِلْمَ مَاذَا قَالَ ءَانِفًا ۚ أُو۟لَـٰٓئِكَ ٱلَّذِينَ طَبَعَ ٱللَّهُ عَلَىٰ قُلُوبِهِمْ وَٱتَّبَعُوٓا۟ أَهْوَآءَهُمْ ﴿16﴾
اور ان میں کچھ ایسے ہیں جو آپ کی بات سنتے ہیں، یہاں تک کہ جب وہ آپ کے پاس سے نکلتے ہیں تو علم والوں سے کہتے ہیں: ابھی ابھی انہوں نے کیا کہا؟ یہی وہ لوگ ہیں جن کے دلوں پر اللہ نے مہر لگا دی ہے، اور وہ اپنی خواہشات کی پیروی کرتے ہیں۔
وَٱلَّذِينَ ٱهْتَدَوْا۟ زَادَهُمْ هُدٗى وَءَاتَىٰهُمْ تَقْوَىٰهُمْ ﴿17﴾
اور جو لوگ ہدایت یافتہ ہیں، اللہ نے انہیں مزید ہدایت دی اور انہیں ان کا تقویٰ عطا فرمایا۔
فَهَلْ يَنظُرُونَ إِلَّا ٱلسَّاعَةَ أَن تَأْتِيَهُم بَغْتَةٗۖ فَقَدْ جَآءَ أَشْرَاطُهَا ۚ فَأَنَّىٰ لَهُمْ إِذَا جَآءَتْهُمْ ذِكْرَىٰهُمْ ﴿18﴾
تو کیا وہ صرف قیامت کا انتظار کر رہے ہیں کہ وہ ان پر اچانک آ جائے؟ اس کی نشانیاں تو آ چکی ہیں، پھر جب وہ ان پر آ جائے گی تو ان کے لیے نصیحت کیسے فائدہ دے گی؟
فَٱعْلَمْ أَنَّهُۥ لَآ إِلَـٰهَ إِلَّا ٱللَّهُ وَٱسْتَغْفِرْ لِذَنبِكَ وَلِلْمُؤْمِنِينَ وَٱلْمُؤْمِنَـٰتِ ۚ وَٱللَّهُ يَعْلَمُ مُتَقَلَّبَكُمْ وَمَثْوَىٰكُمْ ﴿19﴾
پس جان لو کہ اللہ کے سوا کوئی معبود نہیں، اور اپنے گناہ کی اور مؤمن مردوں اور مؤمن عورتوں کی بخشش مانگو، اور اللہ تمہاری آمد و رفت اور ٹھہرنے کی جگہ کو جانتا ہے۔
وَيَقُولُ ٱلَّذِينَ ءَامَنُوا۟ لَوْلَا نُزِّلَتْ سُورَةٞ ۖ فَإِذَآ أُنزِلَتْ سُورَةٞ مُّحْكَمَةٞ وَذُكِرَ فِيهَا ٱلْقِتَالُ رَأَيْتَ ٱلَّذِينَ فِي قُلُوبِهِم مَّرَضٞ يَنظُرُونَ إِلَيْكَ نَظَرَ ٱلْمُغْشِيِّ عَلَيْهِ مِنَ ٱلْمَوْتِ ۖ فَأَوْلَىٰ لَهُمْ ﴿20﴾
اور جو لوگ ایمان لائے کہتے ہیں: کوئی سورت کیوں نہیں نازل کی گئی؟ پھر جب کوئی محکم سورت نازل کی جاتی ہے اور اس میں قتال کا ذکر ہوتا ہے، تو تم ان لوگوں کو دیکھتے ہو جن کے دلوں میں بیماری ہے کہ وہ تمہاری طرف اس طرح دیکھتے ہیں جیسے کسی پر موت کی بے ہوشی طاری ہو، تو ان کے لیے ہلاکت ہے۔
طَاعَةٞ وَقَوْلٞ مَّعْرُوفٞ ۖ فَإِذَا عَزَمَ ٱلْأَمْرُ فَلَوْ صَدَقُوا۟ ٱللَّهَ لَكَانَ خَيْرٗا لَّهُمْ ﴿21﴾
فرماں برداری اور اچھی بات (ہی بہتر ہے)، پھر جب قتال کا حکم ہو جائے تو اگر وہ اللہ سے سچ بولتے تو یہ ان کے لیے بہتر ہوتا۔
فَهَلْ عَسَيْتُمْ إِن تَوَلَّيْتُمْ أَن تُفْسِدُوا۟ فِي ٱلْأَرْضِ وَتُقَطِّعُوٓا۟ أَرْحَامَكُمْ ﴿22﴾
تو کیا تم سے یہ توقع کی جاتی ہے کہ اگر تم حکومت حاصل کرو تو زمین میں فساد کرو گے اور رشتہ داریاں توڑو گے؟
أُو۟لَـٰٓئِكَ ٱلَّذِينَ لَعَنَهُمُ ٱللَّهُ فَأَصَمَّهُمْ وَأَعْمَىٰٓ أَبْصَـٰرَهُمْ ﴿23﴾
یہی وہ لوگ ہیں جن پر اللہ نے لعنت کی، پھر انہیں بہرا اور ان کی آنکھوں کو اندھا کر دیا۔
أَفَلَا يَتَدَبَّرُونَ ٱلْقُرْءَانَ أَمْ عَلَىٰ قُلُوبٍ أَقْفَالُهَا ﴿24﴾
تو کیا وہ قرآن میں غور نہیں کرتے؟ یا ان کے دلوں پر تالے لگے ہوئے ہیں؟
إِنَّ ٱلَّذِينَ ٱرْتَدُّوا۟ عَلَىٰٓ أَدْبَـٰرِهِم مِّنۢ بَعْدِ مَا تَبَيَّنَ لَهُمُ ٱلْهُدَى ٱلشَّيْطَـٰنُ سَوَّلَ لَهُمْ وَأَمْلَىٰ لَهُمْ ﴿25﴾
بے شک وہ لوگ جو ہدایت ظاہر ہونے کے بعد پیٹھ پھیر گئے، شیطان نے انہیں بہکا دیا اور انہیں لمبی امیدیں دلائیں۔
ذَٰلِكَ بِأَنَّهُمْ قَالُوا۟ لِلَّذِينَ كَرِهُوا۟ مَا نَزَّلَ ٱللَّهُ سَنُطِيعُكُمْ فِي بَعْضِ ٱلْأَمْرِ ۖ وَٱللَّهُ يَعْلَمُ إِسْرَارَهُمْ ﴿26﴾
یہ اس لیے کہ انہوں نے ان لوگوں سے کہا جو اللہ کی نازل کردہ چیز کو ناپسند کرتے تھے: ہم بعض معاملات میں تمہاری بات مانیں گے، حالانکہ اللہ ان کے رازوں کو خوب جانتا ہے۔
فَكَيْفَ إِذَا تَوَفَّتْهُمُ ٱلْمَلَـٰٓئِكَةُ يَضْرِبُونَ وُجُوهَهُمْ وَأَدْبَـٰرَهُمْ ﴿27﴾
تو کیا حال ہوگا جب فرشتے ان کی جانیں قبض کریں گے، اور ان کے چہروں اور پیٹھوں پر ماریں گے؟
ذَٰلِكَ بِأَنَّهُمُ ٱتَّبَعُوا۟ مَآ أَسْخَطَ ٱللَّهَ وَكَرِهُوا۟ رِضْوَٰنَهُۥ فَأَحْبَطَ أَعْمَـٰلَهُمْ ﴿28﴾
یہ اس لیے کہ انہوں نے اس چیز کی پیروی کی جو اللہ کو ناراض کرتی ہے، اور اس کی رضا کو ناپسند کیا، تو اللہ نے ان کے اعمال ضائع کر دیے۔
أَمْ حَسِبَ ٱلَّذِينَ فِي قُلُوبِهِم مَّرَضٌ أَن لَّن يُخْرِجَ ٱللَّهُ أَضْغَـٰنَهُمْ ﴿29﴾
کیا وہ لوگ جن کے دلوں میں بیماری ہے یہ گمان کرتے ہیں کہ اللہ ان کے دلوں کی کدورتوں کو ظاہر نہیں کرے گا؟
وَلَوْ نَشَآءُ لَأَرَيْنَـٰكَهُمْ فَلَعَرَفْتَهُم بِسِيمَـٰهُمْ ۚ وَلَتَعْرِفَنَّهُمْ فِي لَحْنِ ٱلْقَوْلِ ۚ وَٱللَّهُ يَعْلَمُ أَعْمَـٰلَكُمْ ﴿30﴾
اور اگر ہم چاہیں تو انہیں آپ کو دکھا دیں، تو آپ انہیں ان کی پہچان سے پہچان لیں، اور آپ ضرور ان کو باتوں کے انداز سے پہچان لیں گے، اور اللہ تمہارے اعمال کو خوب جانتا ہے۔
وَلَنَبْلُوَنَّكُمْ حَتَّىٰ نَعْلَمَ ٱلْمُجَـٰهِدِينَ مِنكُمْ وَٱلصَّـٰبِرِينَ وَنَبْلُوٓا۟ أَخْبَارَكُمْ ﴿31﴾
اور ہم ضرور تمہیں آزمائیں گے یہاں تک کہ ہم جان لیں کہ تم میں سے کون جہاد کرنے والے اور صبر کرنے والے ہیں، اور ہم تمہاری خبریں جانچ لیں۔
إِنَّ ٱلَّذِينَ كَفَرُوا۟ وَصَدُّوا۟ عَن سَبِيلِ ٱللَّهِ وَشَآقُّوا۟ ٱلرَّسُولَ مِنۢ بَعْدِ مَا تَبَيَّنَ لَهُمُ ٱلْهُدَىٰ لَن يَضُرُّوا۟ ٱللَّهَ شَيْـٔٗا وَسَيُحْبِطُ أَعْمَـٰلَهُمْ ﴿32﴾
بے شک جنہوں نے کفر کیا اور اللہ کے راستے سے روکا اور رسول کی مخالفت کی، اس کے بعد کہ ان پر ہدایت ظاہر ہو گئی، وہ اللہ کا کچھ نقصان نہیں کر سکتے، اور اللہ ان کے اعمال ضائع کر دے گا۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ أَطِيعُوا۟ ٱللَّهَ وَأَطِيعُوا۟ ٱلرَّسُولَ وَلَا تُبْطِلُوٓا۟ أَعْمَـٰلَكُمْ ﴿33﴾
اے ایمان والو! اللہ کی اطاعت کرو اور رسول کی اطاعت کرو، اور اپنے اعمال ضائع نہ کرو۔
إِنَّ ٱلَّذِينَ كَفَرُوا۟ وَصَدُّوا۟ عَن سَبِيلِ ٱللَّهِ ثُمَّ مَاتُوا۟ وَهُمْ كُفَّارٞ فَلَن يَغْفِرَ ٱللَّهُ لَهُمْ ﴿34﴾
بے شک جنہوں نے کفر کیا اور اللہ کے راستے سے روکا، پھر وہ کافر ہی مر گئے، اللہ ہرگز انہیں نہیں بخشے گا۔
فَلَا تَهِنُوا۟ وَتَدْعُوٓا۟ إِلَى ٱلسَّلْمِ وَأَنتُمُ ٱلْأَعْلَوْنَ ۚ وَٱللَّهُ مَعَكُمْ وَلَن يَتِرَكُمْ أَعْمَـٰلَكُمْ ﴿35﴾
پس کمزور نہ پڑو اور صلح کی دعوت نہ دو جبکہ تم ہی غالب ہو، اور اللہ تمہارے ساتھ ہے، وہ تمہارے اعمال کو ہرگز ضائع نہیں کرے گا۔
إِنَّمَا ٱلْحَيَوٰةُ ٱلدُّنْيَا لَعِبٞ وَلَهٞ ۚ وَإِن تُؤْمِنُوا۟ وَتَتَّقُوا۟ يُؤْتِكُمْ أُجُورَكُمْ وَلَا يَسْـَٔلْكُمْ أَمْوَٰلَكُمْ ﴿36﴾
دنیا کی زندگی تو صرف کھیل اور تماشا ہے، اور اگر تم ایمان لاؤ اور تقویٰ اختیار کرو تو وہ تمہیں تمہارے اجر دے گا، اور تم سے تمہارے مال نہیں مانگے گا۔
إِن يَسْـَٔلْكُمُوهَا فَيُحْفِكُمْ تَبْخَلُوا۟ وَيُخْرِجْ أَضْغَـٰنَكُمْ ﴿37﴾
اگر وہ تم سے مال مانگے اور تم پر زور دے تو تم بخل کرو گے، اور وہ تمہارے دلوں کی کدورت ظاہر کر دے گا۔
هَـٰٓأَنتُمْ هَـٰٓؤُلَآءِ تُدْعَوْنَ لِتُنفِقُوا۟ فِي سَبِيلِ ٱللَّهِ فَمِنكُم مَّن يَبْخَلُ ۖ وَمَن يَبْخَلْ فَإِنَّمَا يَبْخَلُ عَن نَّفْسِهِۦ ۚ وَٱللَّهُ ٱلْغَنِيُّ وَأَنتُمُ ٱلْفُقَرَآءُ ۚ وَإِن تَتَوَلَّوْا۟ يَسْتَبْدِلْ قَوْمًا غَيْرَكُمْ ثُمَّ لَا يَكُونُوٓا۟ أَمْثَـٰلَكُم ﴿38﴾
دیکھو! تم وہ لوگ ہو جنہیں اللہ کی راہ میں خرچ کرنے کے لیے بلایا جاتا ہے، تو تم میں سے کچھ بخل کرتے ہیں، اور جو بخل کرتا ہے وہ اپنے ہی نقصان کے لیے بخل کرتا ہے۔ اللہ تو بے نیاز ہے اور تم محتاج ہو، اور اگر تم منہ موڑو گے تو وہ تمہاری جگہ کسی اور قوم کو لے آئے گا، پھر وہ تم جیسے نہ ہوں گے۔
سورة الفتح
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
إِنَّا فَتَحْنَا لَكَ فَتْحٗا مُّبِينٗا ﴿1﴾
بے شک ہم نے آپ کو کھلی فتح عطا کی۔
لِّيَغْفِرَ لَكَ ٱللَّهُ مَا تَقَدَّمَ مِن ذَنبِكَ وَمَا تَأَخَّرَ وَيُتِمَّ نِعْمَتَهُۥ عَلَيْكَ وَيَهْدِيَكَ صِرَٰطٗا مُّسْتَقِيمٗا ﴿2﴾
تاکہ اللہ آپ کے اگلے اور پچھلے گناہ معاف کر دے، اور آپ پر اپنی نعمت مکمل کرے، اور آپ کو سیدھا راستہ دکھائے۔
وَيَنصُرَكَ ٱللَّهُ نَصْرًا عَزِيزًا ﴿3﴾
اور اللہ آپ کو زبردست مدد عطا کرے۔
هُوَ ٱلَّذِيٓ أَنزَلَ ٱلسَّكِينَةَ فِي قُلُوبِ ٱلْمُؤْمِنِينَ لِيَزْدَادُوٓا۟ إِيمَـٰنٗا مَّعَ إِيمَـٰنِهِمْ ۗ وَلِلَّهِ جُنُودُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ ۚ وَكَانَ ٱللَّهُ عَلِيمًا حَكِيمٗا ﴿4﴾
وہی ہے جس نے مؤمنوں کے دلوں میں سکون نازل فرمایا تاکہ وہ اپنے ایمان کے ساتھ مزید ایمان حاصل کریں، اور اللہ ہی کے لیے ہیں آسمانوں اور زمین کے لشکر، اور اللہ خوب جاننے والا، حکمت والا ہے۔
لِّيُدْخِلَ ٱلْمُؤْمِنِينَ وَٱلْمُؤْمِنَـٰتِ جَنَّـٰتٖ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُ خَـٰلِدِينَ فِيهَا وَيُكَفِّرَ عَنْهُمْ سَيِّـَٔاتِهِمْ ۚ وَكَانَ ذَٰلِكَ عِندَ ٱللَّهِ فَوْزًا عَظِيمٗا ﴿5﴾
تاکہ اللہ مؤمن مردوں اور مؤمن عورتوں کو ان باغات میں داخل کرے جن کے نیچے نہریں بہتی ہیں، جن میں وہ ہمیشہ رہیں گے، اور ان کی برائیاں دور کر دے، اور یہ اللہ کے نزدیک بڑی کامیابی ہے۔
وَيُعَذِّبَ ٱلْمُنَـٰفِقِينَ وَٱلْمُنَـٰفِقَـٰتِ وَٱلْمُشْرِكِينَ وَٱلْمُشْرِكَـٰتِ ٱلظَّآنِّينَ بِٱللَّهِ ظَنَّ ٱلسَّوْءِ ۚ عَلَيْهِمْ دَآئِرَةُ ٱلسَّوْءِ ۖ وَغَضِبَ ٱللَّهُ عَلَيْهِمْ وَلَعَنَهُمْ وَأَعَدَّ لَهُمْ جَهَنَّمَ ۖ وَسَآءَتْ مَصِيرٗا ﴿6﴾
اور تاکہ اللہ منافق مردوں اور منافق عورتوں اور مشرک مردوں اور مشرک عورتوں کو عذاب دے، جو اللہ کے بارے میں برا گمان رکھتے ہیں۔ ان پر برا وقت آئے گا، اور اللہ ان پر غضبناک ہوا، اور انہیں لعنت کی، اور ان کے لیے جہنم تیار کی، اور وہ بہت ہی برا ٹھکانہ ہے۔
وَلِلَّهِ جُنُودُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ ۚ وَكَانَ ٱللَّهُ عَزِيزًا حَكِيمًا ﴿7﴾
اور اللہ ہی کے لیے ہیں آسمانوں اور زمین کے لشکر، اور اللہ زبردست، حکمت والا ہے۔
إِنَّآ أَرْسَلْنَـٰكَ شَـٰهِدٗا وَمُبَشِّرٗا وَنَذِيرٗا ﴿8﴾
بے شک ہم نے آپ کو گواہ، خوشخبری دینے والا اور ڈرانے والا بنا کر بھیجا ہے۔
لِّتُؤْمِنُوا۟ بِٱللَّهِ وَرَسُولِهِۦ وَتُعَزِّرُوهُ وَتُوَقِّرُوهُ وَتُسَبِّحُوهُ بُكْرَةٗ وَأَصِيلًا ﴿9﴾
تاکہ تم اللہ اور اس کے رسول پر ایمان لاؤ، اور اس کی مدد کرو، اور اس کی تعظیم کرو، اور صبح و شام اس کی تسبیح کرو۔
إِنَّ ٱلَّذِينَ يُبَايِعُونَكَ إِنَّمَا يُبَايِعُونَ ٱللَّهَ ۚ يَدُ ٱللَّهِ فَوْقَ أَيْدِيهِمْ ۚ فَمَن نَّكَثَ فَإِنَّمَا يَنكُثُ عَلَىٰ نَفْسِهِۦ ۖ وَمَنْ أَوْفَىٰ بِمَا عَـٰهَدَ عَلَيْهُ ٱللَّهَ فَسَيُؤْتِيهِ أَجْرًا عَظِيمٗا ﴿10﴾
بے شک جو لوگ آپ سے بیعت کرتے ہیں، وہ دراصل اللہ سے بیعت کرتے ہیں۔ اللہ کا ہاتھ ان کے ہاتھوں کے اوپر ہے۔ پس جو عہد توڑے گا وہ اپنے ہی نقصان کے لیے توڑے گا، اور جو اللہ سے کیے گئے عہد کو پورا کرے گا، تو اللہ اسے بڑا اجر عطا کرے گا۔
سَيَقُولُ لَكَ ٱلْمُخَلَّفُونَ مِنَ ٱلْأَعْرَابِ شَغَلَتْنَآ أَمْوَٰلُنَا وَأَهْلُونَا فَٱسْتَغْفِرْ لَنَا ۚ يَقُولُونَ بِأَلْسِنَتِهِم مَّا لَيْسَ فِي قُلُوبِهِمْ ۚ قُلْ فَمَن يَمْلِكُ لَكُم مِّنَ ٱللَّهِ شَيْـًٔا إِنْ أَرَادَ بِكُمْ ضَرًّا أَوْ أَرَادَ بِكُمْ نَفْعَۢا ۚ بَلْ كَانَ ٱللَّهُ بِمَا تَعْمَلُونَ خَبِيرٗا ﴿11﴾
پیچھے رہ جانے والے دیہاتی کہیں گے: ہمیں ہمارے مال اور اہل و عیال نے مشغول رکھا، پس ہمارے لیے استغفار کیجیے۔ وہ اپنی زبانوں سے وہ بات کہتے ہیں جو ان کے دلوں میں نہیں۔ کہہ دو: اگر اللہ تمہیں نقصان یا نفع دینا چاہے تو کون ہے جو تمہارے لیے کچھ اختیار رکھتا ہو؟ بلکہ اللہ تمہارے اعمال سے خوب باخبر ہے۔
بَلْ ظَنَنتُمْ أَن لَّن يَنقَلِبَ ٱلرَّسُولُ وَٱلْمُؤْمِنُونَ إِلَىٰٓ أَهْلِيهِمْ أَبَدٗا وَزُيِّنَ ذَٰلِكَ فِي قُلُوبِكُمْ وَظَنَنتُمْ ظَنَّ ٱلسَّوْءِ وَكُنتُمْ قَوْمَۢا بُورٗا ﴿12﴾
بلکہ تم نے گمان کیا کہ رسول اور مؤمن کبھی اپنے گھر والوں کی طرف لوٹ کر نہیں آئیں گے، اور یہ بات تمہارے دلوں میں خوب آراستہ کر دی گئی، اور تم نے برا گمان کیا، اور تم ہلاک ہونے والی قوم بن گئے۔
وَمَن لَّمْ يُؤْمِنۢ بِٱللَّهِ وَرَسُولِهِۦ فَإِنَّآ أَعْتَدْنَا لِلْكَـٰفِرِينَ سَعِيرٗا ﴿13﴾
اور جو اللہ اور اس کے رسول پر ایمان نہیں لاتا، تو ہم نے کافروں کے لیے بھڑکتی آگ تیار کر رکھی ہے۔
وَلِلَّهِ مُلْكُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ ۚ يَغْفِرُ لِمَن يَشَآءُ وَيُعَذِّبُ مَن يَشَآءُ ۚ وَكَانَ ٱللَّهُ غَفُورٗا رَّحِيمٗا ﴿14﴾
اور اللہ ہی کے لیے ہے آسمانوں اور زمین کی بادشاہی، وہ جسے چاہے بخش دے اور جسے چاہے عذاب دے، اور اللہ بخشنے والا، رحم کرنے والا ہے۔
سَيَقُولُ ٱلْمُخَلَّفُونَ إِذَا ٱنطَلَقْتُمْ إِلَىٰ مَغَانِمَ لِتَأْخُذُوهَا ذَرُونَا نَتَّبِعْكُمْ ۖ يُرِيدُونَ أَن يُبَدِّلُوا۟ كَلَـٰمَ ٱللَّهِ ۚ قُل لَّن تَتَّبِعُونَا كَذَٰلِكُمْ قَالَ ٱللَّهُ مِن قَبْلُ ۖ فَسَيَقُولُونَ بَلْ تَحْسُدُونَنَا ۚ بَلْ كَانُوا۟ لَا يَفْقَهُونَ إِلَّا قَلِيلٗا ﴿15﴾
جب تم غنیمتیں لینے کے لیے نکلو گے تو پیچھے رہ جانے والے کہیں گے: ہمیں بھی اپنے ساتھ چلنے دو۔ وہ چاہتے ہیں کہ اللہ کے کلام کو بدل دیں۔ کہہ دو: تم ہرگز ہمارے ساتھ نہیں چلو گے، اللہ نے پہلے ہی ایسا فرمایا ہے۔ پھر وہ کہیں گے: بلکہ تم ہم سے حسد کرتے ہو۔ (نہیں)، بلکہ وہ بہت کم سمجھ رکھتے ہیں۔
قُل لِّلْمُخَلَّفِينَ مِنَ ٱلْأَعْرَابِ سَتُدْعَوْنَ إِلَىٰ قَوْمٖ أُوْلِي بَأْسٖ شَدِيدٖ تُقَـٰتِلُونَهُمْ أَوْ يُسْلِمُونَ ۖ فَإِن تُطِيعُوا۟ يُؤْتِكُمُ ٱللَّهُ أَجْرًا حَسَنٗا ۖ وَإِن تَتَوَلَّوْا۟ كَمَا تَوَلَّيْتُم مِّن قَبْلُ يُعَذِّبْكُمْ عَذَابًا أَلِيمٗا ﴿16﴾
کہہ دو: جو دیہاتی پیچھے رہ گئے تھے، تمہیں ایک سخت جنگ والے قوم کی طرف بلایا جائے گا، تم ان سے لڑو گے یا وہ اطاعت کریں گے۔ اگر تم اطاعت کرو گے تو اللہ تمہیں اچھا اجر دے گا، اور اگر تم منہ موڑو گے جیسے پہلے منہ موڑا تھا، تو وہ تمہیں دردناک عذاب دے گا۔
لَّيْسَ عَلَى ٱلْأَعْمَىٰ حَرَجٞ وَلَا عَلَى ٱلْأَعْرَجِ حَرَجٞ وَلَا عَلَى ٱلْمَرِيضِ حَرَجٞ ۚ وَمَن يُطِعِ ٱللَّهَ وَرَسُولَهُۥ يُدْخِلْهُ جَنَّـٰتٖ تَجْرِي مِن تَحْتِهَا ٱلْأَنْهَـٰرُ ۖ وَمَن يَتَوَلَّ يُعَذِّبْهُ عَذَابًا أَلِيمٗا ﴿17﴾
نہ اندھے پر کوئی گناہ ہے، نہ لنگڑے پر کوئی گناہ ہے، اور نہ بیمار پر کوئی گناہ ہے۔ اور جو اللہ اور اس کے رسول کی اطاعت کرے گا، اللہ اسے ان باغات میں داخل کرے گا جن کے نیچے نہریں بہتی ہیں، اور جو منہ موڑے گا، اللہ اسے دردناک عذاب دے گا۔
لَّقَدْ رَضِيَ ٱللَّهُ عَنِ ٱلْمُؤْمِنِينَ إِذْ يُبَايِعُونَكَ تَحْتَ ٱلشَّجَرَةِ فَعَلِمَ مَا فِي قُلُوبِهِمْ فَأَنزَلَ ٱلسَّكِينَةَ عَلَيْهِمْ وَأَثَـٰبَهُمْ فَتْحٗا قَرِيبٗا ﴿18﴾
یقیناً اللہ مؤمنوں سے راضی ہو گیا جب وہ درخت کے نیچے آپ سے بیعت کر رہے تھے، تو اس نے ان کے دلوں کا حال جان لیا، اور ان پر سکون نازل فرمایا، اور انہیں قریب فتح کا انعام دیا۔
وَمَغَانِمَ كَثِيرَةٗ يَأْخُذُونَهَا ۗ وَكَانَ ٱللَّهُ عَزِيزًا حَكِيمٗا ﴿19﴾
اور بہت سی غنیمتیں جنہیں وہ حاصل کریں گے، اور اللہ زبردست، حکمت والا ہے۔
وَعَدَكُمُ ٱللَّهُ مَغَانِمَ كَثِيرَةٗ تَأْخُذُونَهَا فَعَجَّلَ لَكُمْ هَـٰذِهِۦ وَكَفَّ أَيْدِيَ ٱلنَّاسِ عَنكُمْ وَلِتَكُونَ ءَايَةٗ لِّلْمُؤْمِنِينَ وَيَهْدِيَكُمْ صِرَٰطٗا مُّسْتَقِيمٗا ﴿20﴾
اللہ نے تم سے بہت سی غنیمتوں کا وعدہ کیا جنہیں تم حاصل کرو گے، تو اس نے یہ جلدی تمہیں عطا کی، اور لوگوں کے ہاتھ تم سے روک دیے، تاکہ یہ مؤمنوں کے لیے نشانی بنے، اور وہ تمہیں سیدھا راستہ دکھائے۔
وَأُخْرَىٰ لَمْ تَقْدِرُوا۟ عَلَيْهَا قَدْ أَحَاطَ ٱللَّهُ بِهَا ۚ وَكَانَ ٱللَّهُ عَلَىٰ كُلِّ شَيْءٍ قَدِيرٗا ﴿21﴾
اور ایک اور (فتح) بھی ہے جس پر تم قابو نہیں پا سکے، اللہ نے اسے گھیر رکھا ہے، اور اللہ ہر چیز پر قدرت رکھتا ہے۔
وَلَوْ قَـٰتَلَكُمُ ٱلَّذِينَ كَفَرُوا۟ لَوَلَّوُا۟ ٱلْأَدْبَـٰرَ ثُمَّ لَا يَجِدُونَ وَلِيّٗا وَلَا نَصِيرٗا ﴿22﴾
اور اگر کافر تم سے لڑتے تو ضرور پیٹھ پھیر کر بھاگ جاتے، پھر وہ کوئی حمایتی اور مددگار نہ پاتے۔
سُنَّةَ ٱللَّهِ ٱلَّتِي قَدْ خَلَتْ مِن قَبْلُ ۖ وَلَن تَجِدَ لِسُنَّةِ ٱللَّهِ تَبْدِيلٗا ﴿23﴾
یہ اللہ کی سنت ہے جو پہلے گزر چکی ہے، اور تم اللہ کی سنت میں کوئی تبدیلی نہ پاؤ گے۔
وَهُوَ ٱلَّذِي كَفَّ أَيْدِيَهُمْ عَنكُمْ وَأَيْدِيَكُمْ عَنْهُم بِبَطْنِ مَكَّةَ مِنۢ بَعْدِ أَنْ أَظْفَرَكُمْ عَلَيْهِمْ ۚ وَكَانَ ٱللَّهُ بِمَا تَعْمَلُونَ بَصِيرٗا ﴿24﴾
اور وہی ہے جس نے مکہ کے میدان میں ان کے ہاتھ تم سے اور تمہارے ہاتھ ان سے روک دیے، اس کے بعد کہ اس نے تمہیں ان پر غلبہ دیا، اور اللہ تمہارے اعمال کو خوب دیکھنے والا ہے۔
هُمُ ٱلَّذِينَ كَفَرُوا۟ وَصَدُّوكُمْ عَنِ ٱلْمَسْجِدِ ٱلْحَرَامِ وَٱلْهَدْيَ مَعْكُوفًا أَن يَبْلُغَ مَحِلَّهُۥ ۚ وَلَوْلَا رِجَالٞ مُّؤْمِنُونَ وَنِسَآءٞ مُّؤْمِنَـٰتٞ لَّمْ تَعْلَمُوهُمْ أَن تَطَـُٔوهُمْ فَتُصِيبَكُم مِّنْهُم مَّعَرَّةُۢ بِغَيْرِ عِلْمٖ لِّيُدْخِلَ ٱللَّهُ فِي رَحْمَتِهِۦ مَن يَشَآءُ ۚ لَوْ تَزَيَّلُوا۟ لَعَذَّبْنَا ٱلَّذِينَ كَفَرُوا۟ مِنْهُمْ عَذَابًا أَلِيمًا ﴿25﴾
وہی لوگ ہیں جنہوں نے کفر کیا اور تمہیں مسجدِ حرام سے روکا، اور قربانی کے جانوروں کو بھی روکے رکھا کہ وہ اپنی جگہ تک نہ پہنچ سکیں۔ اور اگر کچھ مؤمن مرد اور مؤمن عورتیں نہ ہوتیں جنہیں تم نہیں جانتے تھے، (اور) کہیں تم انہیں روند ڈالتے اور تمہیں ان کی وجہ سے نادانستہ نقصان پہنچتا، (تو ہم تمہیں روکتے نہ) تاکہ اللہ جسے چاہے اپنی رحمت میں داخل کرے۔ اگر وہ الگ ہو جاتے تو ہم کافروں کو دردناک عذاب دیتے۔
إِذْ جَعَلَ ٱلَّذِينَ كَفَرُوا۟ فِي قُلُوبِهِمُ ٱلْحَمِيَّةَ حَمِيَّةَ ٱلْجَاهِلِيَّةِ فَأَنزَلَ ٱللَّهُ سَكِينَتَهُۥ عَلَىٰ رَسُولِهِۦ وَعَلَى ٱلْمُؤْمِنِينَ وَأَلْزَمَهُمْ كَلِمَةَ ٱلتَّقْوَىٰ وَكَانُوا۟ أَحَقَّ بِهَا وَأَهْلَهَا ۚ وَكَانَ ٱللَّهُ بِكُلِّ شَيْءٍ عَلِيمٗا ﴿26﴾
جب کافروں نے اپنے دلوں میں جاہلیت کی ضد بٹھا لی، تو اللہ نے اپنے رسول اور مؤمنوں پر سکون نازل فرمایا، اور انہیں تقویٰ کی بات پر قائم رکھا، اور وہ اس کے زیادہ حق دار اور اہل تھے، اور اللہ ہر چیز کو خوب جاننے والا ہے۔
لَّقَدْ صَدَقَ ٱللَّهُ رَسُولَهُ ٱلرُّءْيَا بِٱلْحَقِّ ۖ لَتَدْخُلُنَّ ٱلْمَسْجِدَ ٱلْحَرَامَ إِن شَآءَ ٱللَّهُ ءَامِنِينَ مُحَلِّقِينَ رُءُوسَكُمْ وَمُقَصِّرِينَ لَا تَخَافُونَ ۚ فَعَلِمَ مَا لَمْ تَعْلَمُوا۟ فَجَعَلَ مِن دُونِ ذَٰلِكَ فَتْحٗا قَرِيبٗا ﴿27﴾
یقیناً اللہ نے اپنے رسول کو سچائی کے ساتھ خواب دکھایا، کہ تم ضرور مسجدِ حرام میں داخل ہو گے، اگر اللہ نے چاہا، امن کے ساتھ، اپنے سروں کو منڈواتے اور بال چھوٹے کرتے ہوئے، بغیر خوف کے۔ پس اللہ نے وہ جانا جو تم نہ جانتے تھے، تو اس نے اس سے پہلے ایک قریب فتح مقرر کر دی۔
هُوَ ٱلَّذِيٓ أَرْسَلَ رَسُولَهُۥ بِٱلْهُدَىٰ وَدِينِ ٱلْحَقِّ لِيُظْهِرَهُۥ عَلَى ٱلدِّينِ كُلِّهِۦ ۚ وَكَفَىٰ بِٱللَّهِ شَهِيدٗا ﴿28﴾
وہی ہے جس نے اپنے رسول کو ہدایت اور دینِ حق کے ساتھ بھیجا تاکہ اسے تمام دینوں پر غالب کرے، اور اللہ گواہی کے لیے کافی ہے۔
مُّحَمَّدٌ رَّسُولُ ٱللَّهِ ۚ وَٱلَّذِينَ مَعَهُۥٓ أَشِدَّآءُ عَلَى ٱلْكُفَّارِ رُحَمَآءُ بَيْنَهُمْ ۖ تَرَىٰهُمْ رُكَّعٗا سُجَّدٗا يَبْتَغُونَ فَضْلٗا مِّنَ ٱللَّهِ وَرِضْوَٰنٗا ۖ سِيمَاهُمْ فِي وُجُوهِهِم مِّنْ أَثَرِ ٱلسُّجُودِ ۚ ذَٰلِكَ مَثَلُهُمْ فِي ٱلتَّوْرَىٰةِ ۚ وَمَثَلُهُمْ فِي ٱلْإِنجِيلِ كَزَرْعٍ أَخْرَجَ شَطْـَٔهُۥ فَـَٔازَرَهُۥ فَٱسْتَغْلَظَ فَٱسْتَوَىٰ عَلَىٰ سُوقِهِۦ يُعْجِبُ ٱلزُّرَّاعَ لِيَغِيظَ بِهِمُ ٱلْكُفَّارَ ۗ وَعَدَ ٱللَّهُ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ مِنْهُم مَّغْفِرَةٗ وَأَجْرًا عَظِيمٗا ﴿29﴾
محمد اللہ کے رسول ہیں، اور جو لوگ ان کے ساتھ ہیں وہ کافروں پر سخت اور آپس میں رحم دل ہیں۔ تم انہیں رکوع اور سجدے میں دیکھو گے، وہ اللہ کا فضل اور اس کی رضا چاہتے ہیں۔ ان کی علامت ان کے چہروں پر سجدے کے اثر سے ہے۔ یہی ان کی مثال تورات میں ہے، اور انجیل میں ان کی مثال اس کھیتی کی طرح ہے جس نے اپنی کونپل نکالی، پھر اسے مضبوط کیا، پھر وہ موٹی ہو گئی، پھر اپنے تنے پر سیدھی کھڑی ہو گئی، کسانوں کو وہ خوش کرتی ہے تاکہ اللہ ان کے ذریعے کافروں کو غصہ دلائے۔ اللہ نے ان لوگوں سے جو ایمان لائے اور نیک عمل کیے، مغفرت اور بڑا اجر وعدہ کیا ہے۔
سورة الحجرات
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تُقَدِّمُوا۟ بَيْنَ يَدَيِ ٱللَّهِ وَرَسُولِهِۦ ۖ وَٱتَّقُوا۟ ٱللَّهَ ۚ إِنَّ ٱللَّهَ سَمِيعٌ عَلِيمٌ ﴿1﴾
اے ایمان والو! اللہ اور اس کے رسول سے آگے نہ بڑھو، اور اللہ سے ڈرو، بے شک اللہ سب کچھ سننے والا، جاننے والا ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تَرْفَعُوٓا۟ أَصْوَاتَكُمْ فَوْقَ صَوْتِ ٱلنَّبِيِّ وَلَا تَجْهَرُوا۟ لَهُۥ بِٱلْقَوْلِ كَجَهْرِ بَعْضِكُمْ لِبَعْضٍ أَن تَحْبَطَ أَعْمَـٰلُكُمْ وَأَنتُمْ لَا تَشْعُرُونَ ﴿2﴾
اے ایمان والو! اپنی آوازیں نبی کی آواز سے بلند نہ کرو، اور نہ ہی اس سے بات کرو جیسے تم آپس میں ایک دوسرے سے کرتے ہو، کہیں ایسا نہ ہو کہ تمہارے اعمال ضائع ہو جائیں اور تمہیں خبر بھی نہ ہو۔
إِنَّ ٱلَّذِينَ يَغُضُّونَ أَصْوَاتَهُمْ عِندَ رَسُولِ ٱللَّهِ أُو۟لَـٰٓئِكَ ٱلَّذِينَ ٱمْتَحَنَ ٱللَّهُ قُلُوبَهُمْ لِلتَّقْوَىٰ ۚ لَهُم مَّغْفِرَةٞ وَأَجْرٌ عَظِيمٌ ﴿3﴾
بے شک جو لوگ رسول اللہ کے پاس اپنی آوازیں پست رکھتے ہیں، وہی ہیں جن کے دلوں کو اللہ نے تقویٰ کے لیے جانچ لیا ہے، ان کے لیے بخشش اور بڑا اجر ہے۔
إِنَّ ٱلَّذِينَ يُنَادُونَكَ مِن وَرَآءِ ٱلْحُجُرَٰتِ أَكْثَرُهُمْ لَا يَعْقِلُونَ ﴿4﴾
بے شک جو لوگ آپ کو حجرات کے باہر سے آواز دیتے ہیں، ان میں سے اکثر عقل نہیں رکھتے۔
وَلَوْ أَنَّهُمْ صَبَرُوا۟ حَتَّىٰ تَخْرُجَ إِلَيْهِمْ لَكَانَ خَيْرٗا لَّهُمْ ۚ وَٱللَّهُ غَفُورٞ رَّحِيمٞ ﴿5﴾
اور اگر وہ صبر کرتے یہاں تک کہ آپ ان کے پاس باہر آتے، تو یہ ان کے لیے بہتر ہوتا، اور اللہ بخشنے والا مہربان ہے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ إِن جَآءَكُمْ فَاسِقُۢ بِنَبَإٖ فَتَبَيَّنُوٓا۟ أَن تُصِيبُوا۟ قَوْمَۢا بِجَهَـٰلَةٖ فَتُصْبِحُوا۟ عَلَىٰ مَا فَعَلْتُمْ نَـٰدِمِينَ ﴿6﴾
اے ایمان والو! اگر کوئی فاسق تمہارے پاس کوئی خبر لائے تو تحقیق کر لو، کہیں ایسا نہ ہو کہ تم کسی قوم کو نادانی سے نقصان پہنچا بیٹھو، پھر اپنے کیے پر پچھتاؤ۔
وَٱعْلَمُوٓا۟ أَنَّ فِيكُمْ رَسُولَ ٱللَّهِ ۚ لَوْ يُطِيعُكُمْ فِي كَثِيرٖ مِّنَ ٱلْأَمْرِ لَعَنِتُّمْ وَلَـٰكِنَّ ٱللَّهَ حَبَّبَ إِلَيْكُمُ ٱلْإِيمَـٰنَ وَزَيَّنَهُۥ فِي قُلُوبِكُمْ وَكَرَّهَ إِلَيْكُمُ ٱلْكُفْرَ وَٱلْفُسُوقَ وَٱلْعِصْيَانَ ۚ أُو۟لَـٰٓئِكَ هُمُ ٱلرَّاشِدُونَ ﴿7﴾
اور جان لو کہ تم میں اللہ کا رسول موجود ہے، اگر وہ بہت سی باتوں میں تمہاری بات مان لے تو تم ضرور مشکل میں پڑ جاؤ، لیکن اللہ نے تمہارے دلوں میں ایمان کو محبوب بنا دیا اور اسے مزین کر دیا، اور کفر، فسق اور نافرمانی کو تمہارے لیے ناپسندیدہ بنا دیا، یہی لوگ ہدایت یافتہ ہیں۔
فَضْلٗا مِّنَ ٱللَّهِ وَنِعْمَةٗ ۚ وَٱللَّهُ عَلِيمٌ حَكِيمٞ ﴿8﴾
یہ اللہ کی طرف سے فضل اور نعمت ہے، اور اللہ سب کچھ جاننے والا، حکمت والا ہے۔
وَإِن طَآئِفَتَانِ مِنَ ٱلْمُؤْمِنِينَ ٱقْتَتَلُوا۟ فَأَصْلِحُوا۟ بَيْنَهُمَا ۖ فَإِنۢ بَغَتْ إِحْدَىٰهُمَا عَلَى ٱلْأُخْرَىٰ فَقَـٰتِلُوا۟ ٱلَّتِي تَبْغِي حَتَّىٰ تَفِيٓءَ إِلَىٰٓ أَمْرِ ٱللَّهِ ۚ فَإِن فَآءَتْ فَأَصْلِحُوا۟ بَيْنَهُمَا بِٱلْعَدْلِ وَأَقْسِطُوٓا۟ ۖ إِنَّ ٱللَّهَ يُحِبُّ ٱلْمُقْسِطِينَ ﴿9﴾
اور اگر مسلمانوں کے دو گروہ آپس میں لڑ پڑیں تو ان کے درمیان صلح کراؤ، پھر اگر ان میں سے ایک دوسرے پر زیادتی کرے تو اس سے لڑو جو زیادتی کر رہا ہے، یہاں تک کہ وہ اللہ کے حکم کی طرف لوٹ آئے، پھر اگر لوٹ آئے تو ان کے درمیان انصاف سے صلح کراؤ، اور انصاف سے کام لو، بے شک اللہ انصاف کرنے والوں کو پسند کرتا ہے۔
إِنَّمَا ٱلْمُؤْمِنُونَ إِخْوَةٞ فَأَصْلِحُوا۟ بَيْنَ أَخَوَيْكُمْ ۚ وَٱتَّقُوا۟ ٱللَّهَ لَعَلَّكُمْ تُرْحَمُونَ ﴿10﴾
بے شک سب مؤمن بھائی بھائی ہیں، پس اپنے دو بھائیوں کے درمیان صلح کراؤ، اور اللہ سے ڈرو تاکہ تم پر رحم کیا جائے۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا يَسْخَرْ قَوْمٞ مِّن قَوْمٍ عَسَىٰٓ أَن يَكُونُوا۟ خَيْرٗا مِّنْهُمْ وَلَا نِسَآءٞ مِّن نِّسَآءٍ عَسَىٰٓ أَن يَكُنَّ خَيْرٗا مِّنْهُنَّ ۖ وَلَا تَلْمِزُوٓا۟ أَنفُسَكُمْ وَلَا تَنَابَزُوا۟ بِٱلْأَلْقَـٰبِ ۖ بِئْسَ ٱلِٱسْمُ ٱلْفُسُوقُ بَعْدَ ٱلْإِيمَـٰنِ ۚ وَمَن لَّمْ يَتُبْ فَأُو۟لَـٰٓئِكَ هُمُ ٱلظَّـٰلِمُونَ ﴿11﴾
اے ایمان والو! کوئی قوم کسی قوم کا مذاق نہ اُڑائے، ممکن ہے وہ ان سے بہتر ہو، اور نہ عورتیں دوسری عورتوں کا مذاق اُڑائیں، ممکن ہے وہ ان سے بہتر ہوں۔ اور آپس میں طعن نہ کرو، اور نہ ایک دوسرے کو برے القاب سے پکارو۔ ایمان کے بعد فسق کا نام بہت برا ہے، اور جو توبہ نہ کریں وہی ظالم ہیں۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ ٱجْتَنِبُوا۟ كَثِيرٗا مِّنَ ٱلظَّنِّ إِنَّ بَعْضَ ٱلظَّنِّ إِثْمٞ ۖ وَلَا تَجَسَّسُوا۟ وَلَا يَغْتَب بَّعْضُكُم بَعْضًا ۚ أَيُحِبُّ أَحَدُكُمْ أَن يَأْكُلَ لَحْمَ أَخِيهِ مَيْتٗا فَكَرِهْتُمُوهُ ۚ وَٱتَّقُوا۟ ٱللَّهَ ۚ إِنَّ ٱللَّهَ تَوَّابٌ رَّحِيمٌ ﴿12﴾
اے ایمان والو! بہت سے گمانوں سے بچو، بے شک بعض گمان گناہ ہوتے ہیں۔ اور تجسس نہ کرو، اور نہ ہی تم میں سے کوئی دوسرے کی غیبت کرے۔ کیا تم میں سے کوئی پسند کرے گا کہ وہ اپنے مردہ بھائی کا گوشت کھائے؟ تو تم اسے ناپسند کرتے ہو۔ اور اللہ سے ڈرو، بے شک اللہ توبہ قبول کرنے والا، رحم کرنے والا ہے۔
يَـٰٓأَيُّهَا ٱلنَّاسُ إِنَّا خَلَقْنَـٰكُم مِّن ذَكَرٖ وَأُنثَىٰ وَجَعَلْنَـٰكُمْ شُعُوبٗا وَقَبَآئِلَ لِتَعَارَفُوٓا۟ ۚ إِنَّ أَكْرَمَكُمْ عِندَ ٱللَّهِ أَتْقَىٰكُمْ ۚ إِنَّ ٱللَّهَ عَلِيمٌ خَبِيرٞ ﴿13﴾
اے لوگو! ہم نے تمہیں ایک مرد اور ایک عورت سے پیدا کیا، اور تمہیں قوموں اور قبیلوں میں تقسیم کیا تاکہ تم ایک دوسرے کو پہچانو۔ بے شک اللہ کے نزدیک تم میں سب سے زیادہ عزت والا وہ ہے جو سب سے زیادہ پرہیزگار ہے۔ بے شک اللہ سب کچھ جاننے والا، خبردار ہے۔
قَالَتِ ٱلْأَعْرَابُ ءَامَنَّا ۖ قُل لَّمْ تُؤْمِنُوا۟ وَلَـٰكِن قُولُوٓا۟ أَسْلَمْنَا وَلَمَّا يَدْخُلِ ٱلْإِيمَـٰنُ فِي قُلُوبِكُمْ ۖ وَإِن تُطِيعُوا۟ ٱللَّهَ وَرَسُولَهُۥ لَا يَلِتْكُم مِّنْ أَعْمَـٰلِكُمْ شَيْـًٔا ۚ إِنَّ ٱللَّهَ غَفُورٞ رَّحِيمٌ ﴿14﴾
دیہاتیوں نے کہا: ہم ایمان لے آئے۔ کہہ دو: تم ایمان نہیں لائے، بلکہ یوں کہو کہ ہم نے اطاعت قبول کی، اور ابھی ایمان تمہارے دلوں میں داخل نہیں ہوا۔ اور اگر تم اللہ اور اس کے رسول کی اطاعت کرو گے تو وہ تمہارے اعمال میں سے کچھ بھی کم نہیں کرے گا۔ بے شک اللہ بخشنے والا، رحم کرنے والا ہے۔
إِنَّمَا ٱلْمُؤْمِنُونَ ٱلَّذِينَ ءَامَنُوا۟ بِٱللَّهِ وَرَسُولِهِۦ ثُمَّ لَمْ يَرْتَابُوا۟ وَجَـٰهَدُوا۟ بِأَمْوَٰلِهِمْ وَأَنفُسِهِمْ فِي سَبِيلِ ٱللَّهِ ۚ أُو۟لَـٰٓئِكَ هُمُ ٱلصَّـٰدِقُونَ ﴿15﴾
بے شک مؤمن وہی ہیں جو اللہ اور اس کے رسول پر ایمان لائے، پھر شک نہ کیا، اور اللہ کی راہ میں اپنے مالوں اور جانوں سے جہاد کیا، یہی لوگ سچے ہیں۔
قُلْ أَتُعَلِّمُونَ ٱللَّهَ بِدِينِكُمْ ۚ وَٱللَّهُ يَعْلَمُ مَا فِي ٱلسَّمَـٰوَٰتِ وَمَا فِي ٱلْأَرْضِ ۚ وَٱللَّهُ بِكُلِّ شَيْءٍ عَلِيمٞ ﴿16﴾
کہہ دو: کیا تم اللہ کو اپنے دین کی خبر دے رہے ہو؟ حالانکہ اللہ جانتا ہے جو کچھ آسمانوں اور زمین میں ہے، اور اللہ ہر چیز کو خوب جاننے والا ہے۔
يَمُنُّونَ عَلَيْكَ أَنْ أَسْلَمُوا۟ ۖ قُل لَّا تَمُنُّوا۟ عَلَيَّ إِسْلَـٰمَكُم ۖ بَلِ ٱللَّهُ يَمُنُّ عَلَيْكُمْ أَنْ هَدَىٰكُمْ لِلْإِيمَـٰنِ إِن كُنتُمْ صَـٰدِقِينَ ﴿17﴾
وہ تم پر احسان جتاتے ہیں کہ انہوں نے اسلام قبول کیا، کہہ دو: اپنے اسلام کا مجھ پر احسان نہ جتاؤ، بلکہ اللہ تم پر احسان کرتا ہے کہ اس نے تمہیں ایمان کی ہدایت دی، اگر تم سچے ہو۔
إِنَّ ٱللَّهَ يَعْلَمُ غَيْبَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ ۚ وَٱللَّهُ بَصِيرُۢ بِمَا تَعْمَلُونَ ﴿18﴾
بے شک اللہ آسمانوں اور زمین کی پوشیدہ چیزوں کو جانتا ہے، اور اللہ تمہارے اعمال کو خوب دیکھنے والا ہے۔
سورة قٓ
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
قٓ ۚ وَٱلْقُرْءَانِ ٱلْمَجِيدِ ﴿1﴾
قٓ۔ قسم ہے قرآنِ مجید کی۔
بَلْ عَجِبُوٓا۟ أَن جَآءَهُم مُّنذِرٞ مِّنْهُمْ فَقَالَ ٱلْكَـٰفِرُونَ هَـٰذَا شَيْءٌ عَجِيبٌ ﴿2﴾
بلکہ انہیں تعجب ہوا کہ ان کے پاس انہی میں سے ایک ڈرانے والا آیا، تو کافروں نے کہا: یہ تو عجیب بات ہے۔
أَءِذَا مِتْنَا وَكُنَّا تُرَابٗا ۖ ذَٰلِكَ رَجْعُۢ بَعِيدٞ ﴿3﴾
کیا جب ہم مر جائیں گے اور مٹی ہو جائیں گے، تو دوبارہ لوٹایا جانا؟ یہ تو بہت دور کی بات ہے۔
قَدْ عَلِمْنَا مَا تَنقُصُ ٱلْأَرْضُ مِنْهُمْ ۖ وَعِندَنَا كِتَٰبٌ حَفِيظٞ ﴿4﴾
ہم جانتے ہیں جو کچھ زمین ان میں سے کم کرتی ہے، اور ہمارے پاس ایک محفوظ کتاب ہے۔
بَلْ كَذَّبُوا۟ بِٱلْحَقِّ لَمَّا جَآءَهُمْ فَهُمْ فِيٓ أَمْرٖ مَّرِيجٖ ﴿5﴾
بلکہ انہوں نے حق کو جھٹلا دیا جب وہ ان کے پاس آیا، تو وہ ایک الجھے ہوئے معاملے میں پڑے ہوئے ہیں۔
أَفَلَمْ يَنظُرُوٓا۟ إِلَى ٱلسَّمَآءِ فَوْقَهُمْ كَيْفَ بَنَيْنَـٰهَا وَزَيَّنَّـٰهَا وَمَا لَهَا مِن فُرُوجٖ ﴿6﴾
کیا انہوں نے اپنے اوپر آسمان کی طرف نہیں دیکھا کہ ہم نے اسے کیسے بنایا اور آراستہ کیا، اور اس میں کوئی شگاف نہیں؟
وَٱلْأَرْضَ مَدَدْنَـٰهَا وَأَلْقَيْنَا فِيهَا رَوَٰسِيَ وَأَنۢبَتْنَا فِيهَا مِن كُلِّ زَوْجِۭ بَهِيجٖ ﴿7﴾
اور زمین کو ہم نے پھیلایا، اور اس میں پہاڑ رکھے، اور اس میں ہر قسم کی خوشنما چیزیں اگائیں۔
تَبْصِرَةٗ وَذِكْرَىٰ لِكُلِّ عَبْدٖ مُّنِيبٖ ﴿8﴾
یہ بصیرت اور نصیحت ہے ہر رجوع کرنے والے بندے کے لیے۔
وَنَزَّلْنَا مِنَ ٱلسَّمَآءِ مَآءۭٗ مُّبَٰرَكٗا فَأَنۢبَتْنَا بِهِۦ جَنَّـٰتٖ وَحَبَّ ٱلْحَصِيدِ ﴿9﴾
اور ہم نے آسمان سے بابرکت پانی نازل کیا، پھر اس سے باغات اور فصل کا اناج اگایا۔
وَٱلنَّخْلَ بَاسِقَـٰتٖ لَّهَا طَلْعٞ نَّضِيدٞ ﴿10﴾
اور بلند قامت کھجوریں جن کے خوشے تہ بہ تہ ہوتے ہیں۔
رِّزْقٗا لِّلْعِبَادِ ۖ وَأَحْيَيْنَا بِهِۦ بَلْدَةٗ مَّيْتٗا ۚ كَذَٰلِكَ ٱلْخُرُوجُ ﴿11﴾
(یہ سب کچھ) بندوں کے رزق کے لیے ہے، اور ہم نے اس سے مردہ زمین کو زندہ کیا، اسی طرح (تمہارا) نکلنا ہوگا۔
۞كَذَّبَتْ قَبْلَهُمْ قَوْمُ نُوحٖ وَأَصْحَـٰبُ ٱلرَّسِّ وَثَمُودُ ﴿12﴾
ان سے پہلے قومِ نوح، کنویں والے اور ثمود نے جھٹلایا۔
وَعَادٞ وَفِرْعَوْنُ وَإِخْوَٰنُ لُوطٖ ﴿13﴾
اور عاد، فرعون اور لوط کی قوم کے بھائیوں نے بھی۔
وَأَصْحَـٰبُ ٱلْأَيْكَةِ وَقَوْمُ تُبَّعٍۢ ۚ كُلّٞ كَذَّبَ ٱلرُّسُلَ فَحَقَّ وَعِيدِ ﴿14﴾
اور جنگل والوں اور قومِ تُبّع نے بھی۔ سب نے رسولوں کو جھٹلایا، تو میرا وعدہ (عذاب) سچ ہو گیا۔
أَفَعَيِينَا بِٱلْخَلْقِ ٱلْأَوَّلِ ۚ بَلْ هُمْ فِي لَبْسٖ مِّنْ خَلْقٖ جَدِيدٖ ﴿15﴾
کیا ہم پہلی بار کی پیدائش سے عاجز ہو گئے؟ بلکہ وہ نئی پیدائش کے بارے میں شک میں ہیں۔
وَلَقَدْ خَلَقْنَا ٱلْإِنسَـٰنَ وَنَعْلَمُ مَا تُوَسْوِسُ بِهِۦ نَفْسُهُۥ ۖ وَنَحْنُ أَقْرَبُ إِلَيْهِ مِنْ حَبْلِ ٱلْوَرِيدِ ﴿16﴾
اور یقیناً ہم نے انسان کو پیدا کیا اور ہم جانتے ہیں جو اس کا نفس اس کے دل میں وسوسہ ڈالتا ہے، اور ہم اس سے شہ رگ سے بھی زیادہ قریب ہیں۔
إِذْ يَتَلَقَّى ٱلْمُتَلَقِّيَانِ عَنِ ٱلْيَمِينِ وَعَنِ ٱلشِّمَالِ قَعِيدٞ ﴿17﴾
جب دو لکھنے والے فرشتے دائیں اور بائیں بیٹھے ہوئے ہوتے ہیں۔
مَا يَلْفِظُ مِن قَوْلٍ إِلَّا لَدَيْهِ رَقِيبٌ عَتِيدٞ ﴿18﴾
وہ کوئی بات زبان سے نہیں نکالتا مگر اس کے پاس ایک نگران تیار ہوتا ہے۔
وَجَآءَتْ سَكْرَةُ ٱلْمَوْتِ بِٱلْحَقِّ ۖ ذَٰلِكَ مَا كُنتَ مِنْهُ تَحِيدُ ﴿19﴾
اور موت کی بے ہوشی حق کے ساتھ آ گئی، یہی وہ چیز ہے جس سے تم بچتے تھے۔
وَنُفِخَ فِي ٱلصُّورِ ۚ ذَٰلِكَ يَوْمُ ٱلْوَعِيدِ ﴿20﴾
اور صور پھونکا جائے گا، یہی ہے وہ دن جس کا وعدہ تھا۔
وَجَآءَتۡ كُلُّ نَفۡسٖ مَّعَهَا سَآئِقٞ وَشَهِيدٞ ﴿21﴾
اور ہر شخص آئے گا، اس کے ساتھ ایک ہانکنے والا اور ایک گواہ ہوگا۔
لَّقَدۡ كُنتَ فِي غَفۡلَةٖ مِّنۡ هَـٰذَا فَكَشَفۡنَا عَنكَ غِطَآءَكَ فَبَصَرُكَ ٱلۡيَوۡمَ حَدِيدٞ ﴿22﴾
یقیناً تو اس سے غفلت میں تھا، تو ہم نے تجھ سے پردہ ہٹا دیا، آج تیری نگاہ بہت تیز ہے۔
وَقَالَ قَرِينُهُۥ هَـٰذَا مَا لَدَيَّ عَتِيدٌ ﴿23﴾
اور اس کے ساتھی (فرشتہ) نے کہا: یہ ہے جو میرے پاس تیار ہے۔
أَلۡقِيَا فِي جَهَنَّمَ كُلَّ كَفَّارٍ عَنِيدٖ ﴿24﴾
تم دونوں جہنم میں پھینک دو ہر سخت کافر، ضدی کو۔
مَّنَّاعٖ لِّلۡخَيۡرِ مُعۡتَدٖ مُّرِيبٍ ﴿25﴾
جو بھلائی سے روکنے والا، حد سے بڑھنے والا، شک کرنے والا تھا۔
ٱلَّذِي جَعَلَ مَعَ ٱللَّهِ إِلَـٰهًا ءَاخَرَ فَأَلۡقِيَاهُ فِي ٱلۡعَذَابِ ٱلشَّدِيدِ ﴿26﴾
جس نے اللہ کے ساتھ دوسرا معبود بنا رکھا تھا، تو اسے سخت عذاب میں پھینک دو۔
قَالَ قَرِينُهُۥ رَبَّنَا مَآ أَطۡغَيۡتُهُۥ وَلَـٰكِن كَانَ فِي ضَلَـٰلِۭ بَعِيدٖ ﴿27﴾
اس کے ساتھی نے کہا: اے ہمارے رب! میں نے اسے گمراہ نہیں کیا، بلکہ یہ خود دور کی گمراہی میں تھا۔
قَالَ لَا تَخۡتَصِمُوا۟ لَدَيَّ وَقَدۡ قَدَّمۡتُ إِلَيۡكُم بِٱلۡوَعِيدِ ﴿28﴾
اللہ فرمائے گا: میرے سامنے جھگڑا نہ کرو، میں نے تمہیں پہلے ہی ڈرا دیا تھا۔
مَا يُبَدَّلُ ٱلۡقَوۡلُ لَدَيَّ وَمَآ أَنَا۠ بِظَلَّـٰمٖ لِّلۡعَبِيدِ ﴿29﴾
میرے ہاں بات بدلی نہیں جاتی، اور میں بندوں پر ظلم کرنے والا نہیں ہوں۔
يَوۡمَ نَقُولُ لِجَهَنَّمَ هَلِ ٱمۡتَلَأۡتِ وَتَقُولُ هَلۡ مِن مَّزِيدٖ ﴿30﴾
جس دن ہم جہنم سے کہیں گے: کیا تو بھر گئی؟ اور وہ کہے گی: کیا کچھ اور ہے؟
وَأُزْلِفَتِ ٱلْجَنَّةُ لِلْمُتَّقِينَ غَيْرَ بَعِيدٍ ﴿31﴾
اور جنت پرہیزگاروں کے قریب کر دی جائے گی، کچھ دور نہ ہوگی۔
هَـٰذَا مَا تُوعَدُونَ لِكُلِّ أَوَّابٍ حَفِيظٍ ﴿32﴾
یہ وہ چیز ہے جس کا تم سے وعدہ کیا گیا تھا، ہر رجوع کرنے والے، حفاظت کرنے والے کے لیے۔
مَنْ خَشِيَ ٱلرَّحْمَـٰنَ بِٱلْغَيْبِ وَجَآءَ بِقَلْبٖ مُّنِيبٍ ﴿33﴾
جو رحمان سے غیب میں ڈرتا رہا اور رجوع کرنے والا دل لے کر آیا۔
ٱدْخُلُوهَا بِسَلَـٰمٖ ۖ ذَٰلِكَ يَوْمُ ٱلْخُلُودِ ﴿34﴾
اس میں سلامتی کے ساتھ داخل ہو جاؤ، یہی ہمیشہ رہنے کا دن ہے۔
لَهُم مَّا يَشَآءُونَ فِيهَا وَلَدَيْنَا مَزِيدٞ ﴿35﴾
ان کے لیے اس میں جو وہ چاہیں گے (موجود ہوگا)، اور ہمارے پاس اس سے بھی زیادہ ہے۔
وَكَمْ أَهْلَكْنَا قَبْلَهُم مِّن قَرْنٍ هُمْ أَشَدُّ مِنْهُم بَطْشٗا فَنَقَّبُوا۟ فِي ٱلْبِلَـٰدِ هَلْ مِن مَّحِيصٍ ﴿36﴾
اور ہم نے ان سے پہلے کتنی ہی قومیں ہلاک کیں جو ان سے زیادہ زور آور تھیں، پس انہوں نے ملکوں میں چھان مارا، کیا کوئی بچنے کی جگہ تھی؟
إِنَّ فِي ذَٰلِكَ لَذِكْرَىٰ لِمَن كَانَ لَهُۥ قَلْبٌ أَوْ أَلْقَى ٱلسَّمْعَ وَهُوَ شَهِيدٞ ﴿37﴾
یقیناً اس میں نصیحت ہے اس کے لیے جس کے پاس دل ہو یا جو توجہ سے سنے اور حاضر ہو۔
وَلَقَدْ خَلَقْنَا ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ وَمَا بَيْنَهُمَا فِي سِتَّةِ أَيَّامٖ وَمَا مَسَّنَا مِن لُّغُوبٖ ﴿38﴾
اور ہم نے آسمانوں اور زمین کو اور جو کچھ ان کے درمیان ہے چھ دنوں میں پیدا کیا، اور ہمیں کوئی تھکن نہیں ہوئی۔
فَٱصْبِرْ عَلَىٰ مَا يَقُولُونَ وَسَبِّحْ بِحَمْدِ رَبِّكَ قَبْلَ طُلُوعِ ٱلشَّمْسِ وَقَبْلَ ٱلْغُرُوبِ ﴿39﴾
پس جو وہ کہتے ہیں اس پر صبر کرو، اور اپنے رب کی حمد کے ساتھ تسبیح کرو سورج کے طلوع سے پہلے اور غروب سے پہلے۔
وَمِنَ ٱلَّيْلِ فَسَبِّحْهُ وَأَدْبَٰرَ ٱلسُّجُودِ ﴿40﴾
اور رات کے کچھ حصے میں بھی اس کی تسبیح کرو، اور سجدوں کے پیچھے بھی۔
وَٱسْتَمِعْ يَوْمَ يُنَادِ ٱلْمُنَادِ مِن مَّكَانٖ قَرِيبٍ ﴿41﴾
اور سنو! جس دن پکارنے والا قریب جگہ سے پکارے گا۔
يَوْمَ يَسْمَعُونَ ٱلصَّيْحَةَ بِٱلْحَقِّ ۚ ذَٰلِكَ يَوْمُ ٱلْخُرُوجِ ﴿42﴾
جس دن وہ چیخ کو حق کے ساتھ سنیں گے، وہی نکلنے کا دن ہوگا۔
إِنَّا نَحْنُ نُحْيِي وَنُمِيتُ وَإِلَيْنَا ٱلْمَصِيرُ ﴿43﴾
بے شک ہم ہی زندہ کرتے ہیں اور مارتے ہیں، اور ہماری ہی طرف لوٹنا ہے۔
يَوْمَ تَشَقَّقُ ٱلْأَرْضُ عَنْهُمْ سِرَاعٗا ۚ ذَٰلِكَ حَشْرٌ عَلَيْنَا يَسِيرٞ ﴿44﴾
جس دن زمین ان سے پھٹ جائے گی اور وہ تیزی سے نکلیں گے، یہ ہمارے لیے آسان جمع کرنا ہے۔
نَّحْنُ أَعْلَمُ بِمَا يَقُولُونَ ۖ وَمَآ أَنتَ عَلَيْهِم بِجَبَّارٖ ۖ فَذَكِّرْ بِٱلْقُرْءَانِ مَن يَخَافُ وَعِيدِ ﴿45﴾
ہم خوب جانتے ہیں جو وہ کہتے ہیں، اور تم ان پر جبر کرنے والے نہیں ہو، پس قرآن کے ذریعے نصیحت کرو اسے جو میرے وعید سے ڈرتا ہے۔
سورة الذاریات
بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
وَٱلذَّٰرِيَٰتِ ذَرْوٗا ﴿1﴾
قسم ہے ان ہواؤں کی جو بکھیرتی ہیں۔
فَٱلْحَـٰمِلَـٰتِ وِقْرٗا ﴿2﴾
پھر ان کی جو بوجھ اٹھاتی ہیں۔
فَٱلْجَـٰرِيَـٰتِ يُسْرٗا ﴿3﴾
پھر ان کی جو آسانی سے چلتی ہیں۔
فَٱلْمُقَسِّمَـٰتِ أَمْرًا ﴿4﴾
پھر ان کی جو حکم تقسیم کرتی ہیں۔
إِنَّمَا تُوعَدُونَ لَصَادِقٞ ﴿5﴾
یقیناً جس چیز کا تم سے وعدہ کیا جا رہا ہے وہ سچ ہے۔
وَإِنَّ ٱلدِّينَ لَوَاقِعٞ ﴿6﴾
اور یقیناً جزا کا دن واقع ہو کر رہے گا۔
وَٱلسَّمَآءِ ذَاتِ ٱلْحُبُكِ ﴿7﴾
قسم ہے آسمان کی جو (نظام میں) راستوں والی ہے۔
إِنَّكُمْ لَفِي قَوْلٖ مُّخْتَلِفٖ ﴿8﴾
یقیناً تم مختلف باتوں میں پڑے ہوئے ہو۔
يُؤْفَكُ عَنْهُ مَنْ أُفِكَ ﴿9﴾
اس سے وہی پھیر دیا جاتا ہے جو (پہلے ہی) پھیر دیا گیا ہو۔
قُتِلَ ٱلْخَرَّٰصُونَ ﴿10﴾
مارے گئے وہ جو اٹکل پچو باتیں کرتے ہیں۔
ٱلَّذِينَ هُمْ فِي غَمْرَةٍۢ سَاهُونَ ﴿11﴾
وہ لوگ جو غفلت میں ڈوبے ہوئے ہیں۔
يَسْـَٔلُونَ أَيَّانَ يَوْمُ ٱلدِّينِ ﴿12﴾
پوچھتے ہیں: جزا کا دن کب آئے گا؟
يَوْمَ هُمْ عَلَى ٱلنَّارِ يُفْتَنُونَ ﴿13﴾
جس دن وہ آگ پر آزمائے جائیں گے۔
ذُوقُوا۟ فِتْنَتَكُمْ ۗ هَـٰذَا ٱلَّذِى كُنتُم بِهِۦ تَسْتَعْجِلُونَ ﴿14﴾
چکھو اپنی آزمائش! یہی وہ چیز ہے جس کے لیے تم جلدی مچا رہے تھے۔
إِنَّ ٱلْمُتَّقِينَ فِي جَنَّـٰتٖ وَعُيُونٍ ﴿15﴾
یقیناً پرہیزگار لوگ باغوں اور چشموں میں ہوں گے۔
ءَاخِذِينَ مَآ ءَاتَىٰهُمْ رَبُّهُمْ ۚ إِنَّهُمْ كَانُوا۟ قَبْلَ ذَٰلِكَ مُحْسِنِينَ ﴿16﴾
اپنے رب کی عطا کردہ چیزیں لے رہے ہوں گے، بے شک وہ اس سے پہلے نیکوکار تھے۔
كَانُوا۟ قَلِيلٗا مِّنَ ٱلَّيْلِ مَا يَهْجَعُونَ ﴿17﴾
وہ رات کے تھوڑے حصے میں ہی سوتے تھے۔
وَبِٱلْأَسْحَارِ هُمْ يَسْتَغْفِرُونَ ﴿18﴾
اور سحر کے وقت وہ بخشش مانگا کرتے تھے۔
وَفِيٓ أَمْوَٰلِهِمْ حَقّٞ لِّلسَّآئِلِ وَٱلْمَحْرُومِ ﴿19﴾
اور ان کے مالوں میں سائل اور محروم کا حق تھا۔
وَفِي ٱلْأَرْضِ ءَايَـٰتٞ لِّلْمُوقِنِينَ ﴿20﴾
اور زمین میں یقین رکھنے والوں کے لیے نشانیاں ہیں۔
وَفِيٓ أَنفُسِكُمْ ۚ أَفَلَا تُبْصِرُونَ ﴿21﴾
اور تمہارے اپنے نفسوں میں بھی (نشانیاں ہیں)، کیا تم دیکھتے نہیں؟
وَفِي ٱلسَّمَآءِ رِزْقُكُمْ وَمَا تُوعَدُونَ ﴿22﴾
اور آسمان میں ہے تمہارا رزق اور وہ چیز جس کا تم سے وعدہ کیا جاتا ہے۔
فَوَرَبِّ ٱلسَّمَآءِ وَٱلْأَرْضِ إِنَّهُۥ لَحَقّٞ مِّثْلَ مَآ أَنَّكُمْ تَنطِقُونَ ﴿23﴾
پس آسمان اور زمین کے رب کی قسم! یقیناً یہ حق ہے جیسے کہ تم بولتے ہو۔
هَلْ أَتَىٰكَ حَدِيثُ ضَيْفِ إِبْرَٰهِيمَ ٱلْمُكْرَمِينَ ﴿24﴾
کیا تمہارے پاس ابراہیم کے معزز مہمانوں کی خبر آئی؟
إِذْ دَخَلُوا۟ عَلَيْهِ فَقَالُوا۟ سَلَـٰمٗا ۖ قَالَ سَلَـٰمٞ قَوْمٞ مُّنكَرُونَ ﴿25﴾
جب وہ اس کے پاس آئے اور کہا: سلام ہو! اس نے کہا: سلام ہو، یہ اجنبی لوگ ہیں۔
فَرَاغَ إِلَىٰٓ أَهْلِهِۦ فَجَآءَ بِعِجْلٖ سَمِينٖ ﴿26﴾
پھر وہ اپنے گھر والوں کی طرف چپکے سے گیا اور ایک موٹا بچڑا لے آیا۔
فَقَرَّبَهُۥٓ إِلَيْهِمْ قَالَ أَلَا تَأْكُلُونَ ﴿27﴾
پھر وہ اسے ان کے قریب لے آیا اور کہا: کیا تم کھاتے نہیں؟
فَأَوْجَسَ مِنْهُمْ خِيفَةٗ قَالُوا۟ لَا تَخَفْ ۖ وَبَشَّرُوهُ بِغُلَـٰمٍ عَلِيمٖ ﴿28﴾
تو اس نے ان سے خوف محسوس کیا، انہوں نے کہا: مت ڈرو، اور اسے ایک علم والے بیٹے کی خوشخبری دی۔
فَأَقْبَلَتِ ٱمْرَأَتُهُۥ فِي صَرَّةٖ فَصَكَّتْ وَجْهَهَا وَقَالَتْ عَجُوزٌ عَقِيمٞ ﴿29﴾
تو اس کی بیوی چیختی ہوئی آئی، اور اس نے اپنا منہ پیٹا اور کہا: میں تو بوڑھی بانجھ ہوں!
قَالُوا۟ كَذَٰلِكِ قَالَ رَبُّكِ ۖ إِنَّهُۥ هُوَ ٱلْحَكِيمُ ٱلْعَلِيمُ ﴿30﴾
انہوں نے کہا: ایسا ہی ہوگا، تمہارے رب نے فرمایا ہے۔ بے شک وہی حکمت والا، علم والا ہے۔
