سورة الحديد

بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
سَبَّحَ لِلَّهِ مَا فِى ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ ۖ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ ﴿1﴾
اللہ کی تسبیح کرتی ہیں وہ سب چیزیں جو آسمانوں اور زمین میں ہیں، اور وہی زبردست، حکمت والا ہے۔
لَهُۥ مُلْكُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ ۖ يُحْىِۦ وَيُمِيتُ ۖ وَهُوَ عَلَىٰ كُلِّ شَىْءٍۢ قَدِيرٌ ﴿2﴾
اسی کے لیے آسمانوں اور زمین کی بادشاہی ہے، وہی زندہ کرتا ہے اور مارتا ہے، اور وہ ہر چیز پر قدرت رکھتا ہے۔
هُوَ ٱلْأَوَّلُ وَٱلْـَٔاخِرُ وَٱلظَّـٰهِرُ وَٱلْبَاطِنُ ۖ وَهُوَ بِكُلِّ شَىْءٍ عَلِيمٌ ﴿3﴾
وہی اول ہے اور وہی آخر، وہی ظاہر ہے اور وہی باطن، اور وہ ہر چیز کو جاننے والا ہے۔
هُوَ ٱلَّذِى خَلَقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضَ فِى سِتَّةِ أَيَّامٍۢ ثُمَّ ٱسْتَوَىٰ عَلَى ٱلْعَرْشِ ۖ يَعْلَمُ مَا يَلِجُ فِى ٱلْأَرْضِ وَمَا يَخْرُجُ مِنْهَا وَمَا يَنزِلُ مِنَ ٱلسَّمَآءِ وَمَا يَعْرُجُ فِيهَا ۖ وَهُوَ مَعَكُمْ أَيْنَ مَا كُنتُمْ ۚ وَٱللَّهُ بِمَا تَعْمَلُونَ بَصِيرٌ ﴿4﴾
وہی ہے جس نے آسمانوں اور زمین کو چھ دنوں میں پیدا کیا، پھر عرش پر قائم ہوا۔ وہ جانتا ہے جو زمین میں داخل ہوتا ہے اور جو اس سے نکلتا ہے، اور جو آسمان سے نازل ہوتا ہے اور جو اس میں چڑھتا ہے، اور وہ تمہارے ساتھ ہے جہاں کہیں بھی تم ہو۔ اور اللہ تمہارے اعمال کو خوب دیکھنے والا ہے۔
لَهُۥ مُلْكُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ ۚ وَإِلَى ٱللَّهِ تُرْجَعُ ٱلْأُمُورُ ﴿5﴾
اسی کے لیے آسمانوں اور زمین کی بادشاہی ہے، اور تمام معاملات اللہ ہی کی طرف لوٹائے جاتے ہیں۔
يُولِجُ ٱلَّيْلَ فِى ٱلنَّهَارِ وَيُولِجُ ٱلنَّهَارَ فِى ٱلَّيْلِ ۚ وَهُوَ عَلِيمٌۢ بِذَاتِ ٱلصُّدُورِ ﴿6﴾
وہی رات کو دن میں داخل کرتا ہے اور دن کو رات میں داخل کرتا ہے، اور وہ دلوں کے بھید خوب جاننے والا ہے۔
ءَامِنُوا۟ بِٱللَّهِ وَرَسُولِهِۦ وَأَنفِقُوا۟ مِمَّا جَعَلَكُم مُّسْتَخْلَفِينَ فِيهِ ۖ فَٱلَّذِينَ ءَامَنُوا۟ مِنكُمْ وَأَنفَقُوا۟ لَهُمْ أَجْرٌۭ كَبِيرٌۭ ﴿7﴾
اللہ اور اس کے رسول پر ایمان لاؤ، اور اس مال میں سے خرچ کرو جس میں اللہ نے تمہیں نائب بنایا ہے۔ پس تم میں سے جو ایمان لائے اور خرچ کیا، ان کے لیے بڑا اجر ہے۔
وَمَا لَكُمْ لَا تُؤْمِنُونَ بِٱللَّهِ وَٱلرَّسُولُ يَدْعُوكُمْ لِتُؤْمِنُوا۟ بِرَبِّكُمْ وَقَدْ أَخَذَ مِيثَـٰقَكُمْ إِن كُنتُم مُّؤْمِنِينَ ﴿8﴾
تمہیں کیا ہوا کہ تم اللہ پر ایمان نہیں لاتے، حالانکہ رسول تمہیں تمہارے رب پر ایمان لانے کی دعوت دے رہا ہے، اور وہ تم سے عہد لے چکا ہے اگر تم ایمان والے ہو۔
هُوَ ٱلَّذِى يُنَزِّلُ عَلَىٰ عَبْدِهِۦٓ ءَايَـٰتٍۢ بَيِّنَـٰتٍۢ لِّيُخْرِجَكُم مِّنَ ٱلظُّلُمَـٰتِ إِلَى ٱلنُّورِ ۚ وَإِنَّ ٱللَّهَ بِكُمْ لَرَءُوفٌۭ رَّحِيمٌۭ ﴿9﴾
وہی ہے جو اپنے بندے پر واضح آیات نازل کرتا ہے تاکہ تمہیں اندھیروں سے نکال کر روشنی کی طرف لے آئے، اور بے شک اللہ تم پر بہت مہربان، رحم کرنے والا ہے۔
وَمَا لَكُمْ أَلَّا تُنفِقُوا۟ فِى سَبِيلِ ٱللَّهِ وَلِلَّهِ مِيرَٰثُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ ۚ لَا يَسْتَوِى مِنكُم مَّنْ أَنفَقَ مِن قَبْلِ ٱلْفَتْحِ وَقَـٰتَلَ ۚ أُو۟لَـٰٓئِكَ أَعْظَمُ دَرَجَةًۭ مِّنَ ٱلَّذِينَ أَنفَقُوا۟ مِنۢ بَعْدُ وَقَـٰتَلُوا۟ ۚ وَكُلًّۭا وَعَدَ ٱللَّهُ ٱلْحُسْنَىٰ ۚ وَٱللَّهُ بِمَا تَعْمَلُونَ خَبِيرٌۭ ﴿10﴾
تمہیں کیا ہوا کہ تم اللہ کی راہ میں خرچ نہیں کرتے، حالانکہ آسمانوں اور زمین کا وارث اللہ ہی ہے؟ تم میں سے جو لوگ فتح سے پہلے خرچ اور جہاد کرتے رہے، وہ درجے میں ان لوگوں سے بڑے ہیں جو بعد میں خرچ اور جہاد کرتے ہیں۔ اور اللہ نے دونوں سے اچھے وعدے کیے ہیں، اور اللہ تمہارے اعمال سے خوب باخبر ہے۔
مَّن ذَا ٱلَّذِى يُقْرِضُ ٱللَّهَ قَرْضًا حَسَنًۭا فَيُضَـٰعِفَهُۥ لَهُۥ وَلَهُۥٓ أَجْرٌۭ كَرِيمٌۭ ﴿11﴾
کون ہے جو اللہ کو اچھا قرض دے تاکہ وہ اسے کئی گنا بڑھا دے، اور اس کے لیے عزت والا اجر ہو؟
يَوْمَ تَرَى ٱلْمُؤْمِنِينَ وَٱلْمُؤْمِنَـٰتِ يَسْعَىٰ نُورُهُم بَيْنَ أَيْدِيهِمْ وَبِأَيْمَـٰنِهِم بُشْرَىٰكُمُ ٱلْيَوْمَ جَنَّـٰتٌۭ تَجْرِى مِن تَحْتِهَا ٱلْأَنْهَـٰرُ خَـٰلِدِينَ فِيهَا ۚ ذَٰلِكَ هُوَ ٱلْفَوْزُ ٱلْعَظِيمُ ﴿12﴾
جس دن تم مؤمن مردوں اور مؤمن عورتوں کو دیکھو گے کہ ان کا نور ان کے آگے اور ان کے دائیں جانب دوڑ رہا ہوگا، (کہا جائے گا:) آج تمہیں خوشخبری ہے ان جنتوں کی جن کے نیچے نہریں بہتی ہیں، جن میں تم ہمیشہ رہو گے۔ یہی بڑی کامیابی ہے۔
يَوْمَ يَقُولُ ٱلْمُنَـٰفِقُونَ وَٱلْمُنَـٰفِقَـٰتُ لِلَّذِينَ ءَامَنُوا۟ ٱنظُرُونَا نَقْتَبِسْ مِن نُّورِكُمْ قِيلَ ٱرْجِعُوا۟ وَرَآءَكُمْ فَٱلْتَمِسُوا۟ نُورًۭا ۖ فَضُرِبَ بَيْنَهُم بِسُورٍۢ لَّهُۥ بَابٌۭ ۖ بَاطِنُهُۥ فِيهِ ٱلرَّحْمَةُ وَظَـٰهِرُهُۥ مِن قِبَلِهِ ٱلْعَذَابُ ﴿13﴾
جس دن منافق مرد اور منافق عورتیں ایمان والوں سے کہیں گے: ہماری طرف دیکھو تاکہ ہم تمہارے نور سے کچھ لے لیں۔ کہا جائے گا: پیچھے لوٹ جاؤ اور وہاں سے نور تلاش کرو۔ پھر ان کے درمیان ایک دیوار کھڑی کر دی جائے گی جس میں ایک دروازہ ہوگا، اس کے اندر رحمت ہوگی اور باہر کی طرف عذاب۔
يُنَادُونَهُمْ أَلَمْ نَكُن مَّعَكُمْ ۖ قَالُوا۟ بَلَىٰ وَلَـٰكِنَّكُمْ فَتَنتُمْ أَنفُسَكُمْ وَتَرَبَّصْتُمْ وَٱرْتَبْتُمْ وَغَرَّتْكُمُ ٱلْأَمَانِىُّ حَتَّىٰ جَآءَ أَمْرُ ٱللَّهِ وَغَرَّكُم بِٱللَّهِ ٱلْغَرُورُ ﴿14﴾
وہ انہیں پکاریں گے: کیا ہم تمہارے ساتھ نہ تھے؟ وہ کہیں گے: ہاں، لیکن تم نے اپنے آپ کو فتنے میں ڈالا، انتظار کرتے رہے، شک میں پڑے رہے، اور تمہیں جھوٹی امیدوں نے دھوکہ دیا یہاں تک کہ اللہ کا حکم آ گیا، اور تمہیں اللہ کے بارے میں دھوکہ دینے والے نے دھوکہ دیا۔
فَٱلْيَوْمَ لَا يُؤْخَذُ مِنكُمْ فِدْيَةٌۭ وَلَا مِنَ ٱلَّذِينَ كَفَرُوا۟ ۚ مَأْوَىٰكُمُ ٱلنَّارُ ۖ هِىَ مَوْلَىٰكُمْ ۖ وَبِئْسَ ٱلْمَصِيرُ ﴿15﴾
پس آج تم سے کوئی فدیہ قبول نہیں کیا جائے گا اور نہ ہی ان لوگوں سے جنہوں نے کفر کیا۔ تمہارا ٹھکانہ جہنم ہے، وہی تمہاری رفیق ہے، اور وہ بہت ہی برا ٹھکانہ ہے۔
أَلَمْ يَأْنِ لِلَّذِينَ ءَامَنُوٓا۟ أَن تَخْشَعَ قُلُوبُهُمْ لِذِكْرِ ٱللَّهِ وَمَا نَزَلَ مِنَ ٱلْحَقِّ ۚ وَلَا يَكُونُوا۟ كَٱلَّذِينَ أُوتُوا۟ ٱلْكِتَـٰبَ مِن قَبْلُ فَطَالَ عَلَيْهِمُ ٱلْأَمَدُ فَقَسَتْ قُلُوبُهُمْ ۖ وَكَثِيرٌۭ مِّنْهُمْ فَـٰسِقُونَ ﴿16﴾
کیا ایمان والوں کے لیے ابھی وہ وقت نہیں آیا کہ ان کے دل اللہ کے ذکر اور اس کے نازل کردہ حق کے لیے جھک جائیں؟ اور وہ ان لوگوں کی طرح نہ ہو جائیں جنہیں اس سے پہلے کتاب دی گئی تھی، پھر ان پر مدت گزر گئی تو ان کے دل سخت ہو گئے، اور ان میں سے بہت سے لوگ فاسق ہیں۔
ٱعْلَمُوٓا۟ أَنَّ ٱللَّهَ يُحْىِ ٱلْأَرْضَ بَعْدَ مَوْتِهَا ۚ قَدْ بَيَّنَّا لَكُمُ ٱلْـَٔايَـٰتِ لَعَلَّكُمْ تَعْقِلُونَ ﴿17﴾
جان لو کہ اللہ زمین کو اس کی موت کے بعد زندہ کرتا ہے۔ ہم نے تمہارے لیے نشانیاں واضح کر دی ہیں تاکہ تم عقل سے کام لو۔
إِنَّ ٱلْمُصَّدِّقِينَ وَٱلْمُصَّدِّقَـٰتِ وَأَقْرَضُوا۟ ٱللَّهَ قَرْضًا حَسَنًۭا يُضَـٰعَفُ لَهُمْ وَلَهُمْ أَجْرٌۭ كَرِيمٌۭ ﴿18﴾
بے شک صدقہ دینے والے مرد اور صدقہ دینے والی عورتیں، اور جنہوں نے اللہ کو اچھا قرض دیا، ان کے لیے وہ کئی گنا بڑھا دیا جائے گا، اور ان کے لیے عزت والا اجر ہے۔
وَٱلَّذِينَ ءَامَنُوا۟ بِٱللَّهِ وَرُسُلِهِۦٓ أُو۟لَـٰٓئِكَ هُمُ ٱلصِّدِّيقُونَ ۖ وَٱلشُّهَدَآءُ عِندَ رَبِّهِمْ لَهُمْ أَجْرُهُمْ وَنُورُهُمْ ۖ وَٱلَّذِينَ كَفَرُوا۟ وَكَذَّبُوا۟ بِـَٔايَـٰتِنَآ أُو۟لَـٰٓئِكَ أَصْحَـٰبُ ٱلْجَحِيمِ ﴿19﴾
اور جو لوگ اللہ اور اس کے رسولوں پر ایمان لائے، وہی صدیق ہیں، اور شہداء اپنے رب کے حضور ہوں گے، ان کے لیے ان کا اجر اور ان کا نور ہے۔ اور جنہوں نے کفر کیا اور ہماری آیات کو جھٹلایا، وہی جہنم والے ہیں۔
ٱعْلَمُوٓا۟ أَنَّمَا ٱلْحَيَوٰةُ ٱلدُّنْيَا لَعِبٌۭ وَلَهْوٌۭ وَزِينَةٌۭ وَتَفَاخُرٌۢ بَيْنَكُمْ وَتَكَاثُرٌۭ فِى ٱلْأَمْوَٰلِ وَٱلْأَوْلَـٰدِ ۖ كَمَثَلِ غَيْثٍ أَعْجَبَ ٱلْكُفَّارَ نَبَاتُهُۥ ثُمَّ يَهِيجُ فَتَرَىٰهُ مُصْفَرًّۭا ثُمَّ يَكُونُ حُطَـٰمًۭا ۗ وَفِى ٱلْـَٔاخِرَةِ عَذَابٌۭ شَدِيدٌۭ وَمَغْفِرَةٌۭ مِّنَ ٱللَّهِ وَرِضْوَٰنٌۭ ۚ وَمَا ٱلْحَيَوٰةُ ٱلدُّنْيَآ إِلَّا مَتَـٰعُ ٱلْغُرُورِ ﴿20﴾
جان لو کہ دنیا کی زندگی کھیل، تماشا، زینت، آپس میں فخر، اور مال و اولاد میں زیادتی کی طلب ہے۔ اس کی مثال بارش کی سی ہے جس سے اگنے والی کھیتی کسانوں کو خوش کرتی ہے، پھر وہ خشک ہو جاتی ہے، پھر تم اسے زرد دیکھتے ہو، پھر وہ چورا ہو جاتی ہے۔ اور آخرت میں سخت عذاب بھی ہے اور اللہ کی طرف سے مغفرت اور رضا بھی۔ اور دنیا کی زندگی دھوکے کا سامان ہے۔
سَابِقُوٓا۟ إِلَىٰ مَغْفِرَةٍۢ مِّن رَّبِّكُمْ وَجَنَّةٍ عَرْضُهَا كَعَرْضِ ٱلسَّمَآءِ وَٱلْأَرْضِ ۖ أُعِدَّتْ لِلَّذِينَ ءَامَنُوا۟ بِٱللَّهِ وَرُسُلِهِۦ ۚ ذَٰلِكَ فَضْلُ ٱللَّهِ يُؤْتِيهِ مَن يَشَآءُ ۚ وَٱللَّهُ ذُو ٱلْفَضْلِ ٱلْعَظِيمِ ﴿21﴾
دوڑو اپنے رب کی مغفرت اور اس جنت کی طرف جس کی چوڑائی آسمان اور زمین کے برابر ہے، جو ان لوگوں کے لیے تیار کی گئی ہے جو اللہ اور اس کے رسولوں پر ایمان لائے۔ یہ اللہ کا فضل ہے، وہ جسے چاہے عطا کرتا ہے، اور اللہ بڑے فضل والا ہے۔
مَآ أَصَابَ مِن مُّصِيبَةٍۢ فِى ٱلْأَرْضِ وَلَا فِىٓ أَنفُسِكُمْ إِلَّا فِى كِتَـٰبٍۢ مِّن قَبْلِ أَن نَّبْرَأَهَآ ۚ إِنَّ ذَٰلِكَ عَلَى ٱللَّهِ يَسِيرٌۭ ﴿22﴾
زمین میں یا تمہارے نفسوں میں جو بھی مصیبت آتی ہے، وہ ایک کتاب میں لکھی ہوئی ہے اس سے پہلے کہ ہم اسے پیدا کریں۔ بے شک یہ اللہ کے لیے آسان ہے۔
لِكَيْلَا تَأْسَوْا۟ عَلَىٰ مَا فَاتَكُمْ وَلَا تَفْرَحُوا۟ بِمَآ ءَاتَىٰكُمْ ۗ وَٱللَّهُ لَا يُحِبُّ كُلَّ مُخْتَالٍۢ فَخُورٍ ﴿23﴾
تاکہ تم اس پر افسوس نہ کرو جو تم سے چھن گیا، اور اس پر خوش نہ ہو جاؤ جو تمہیں دیا گیا۔ اور اللہ کسی مغرور، فخر کرنے والے کو پسند نہیں کرتا۔
ٱلَّذِينَ يَبْخَلُونَ وَيَأْمُرُونَ ٱلنَّاسَ بِٱلْبُخْلِ ۗ وَمَن يَتَوَلَّ فَإِنَّ ٱللَّهَ هُوَ ٱلْغَنِىُّ ٱلْحَمِيدُ ﴿24﴾
وہ لوگ جو بخل کرتے ہیں اور دوسروں کو بخل کا حکم دیتے ہیں، اور جو منہ موڑ لے تو بے شک اللہ بے نیاز، تعریف کے لائق ہے۔
لَقَدْ أَرْسَلْنَا رُسُلَنَا بِٱلْبَيِّنَـٰتِ وَأَنزَلْنَا مَعَهُمُ ٱلْكِتَـٰبَ وَٱلْمِيزَانَ لِيَقُومَ ٱلنَّاسُ بِٱلْقِسْطِ ۖ وَأَنزَلْنَا ٱلْحَدِيدَ فِيهِ بَأْسٌۭ شَدِيدٌۭ وَمَنَـٰفِعُ لِلنَّاسِ وَلِيَعْلَمَ ٱللَّهُ مَن يَنصُرُهُۥ وَرُسُلَهُۥ بِٱلْغَيْبِ ۚ إِنَّ ٱللَّهَ قَوِىٌّ عَزِيزٌ ﴿25﴾
یقیناً ہم نے اپنے رسولوں کو واضح دلائل دے کر بھیجا، اور ان کے ساتھ کتاب اور میزان نازل کی تاکہ لوگ انصاف پر قائم رہیں۔ اور ہم نے لوہا نازل کیا جس میں سخت قوت ہے اور لوگوں کے لیے فائدے بھی ہیں، تاکہ اللہ جان لے کون اس کی اور اس کے رسولوں کی مدد کرتا ہے غیب میں۔ بے شک اللہ قوت والا، غالب ہے۔
وَلَقَدْ أَرْسَلْنَا نُوحًۭا وَإِبْرَٰهِيمَ وَجَعَلْنَا فِى ذُرِّيَّتِهِمَا ٱلنُّبُوَّةَ وَٱلْكِتَـٰبَ ۖ فَمِنْهُم مُّهْتَدٍۢ ۖ وَكَثِيرٌۭ مِّنْهُمْ فَـٰسِقُونَ ﴿26﴾
اور یقیناً ہم نے نوح اور ابراہیم کو بھیجا، اور ان دونوں کی اولاد میں نبوت اور کتاب رکھی، تو ان میں سے کچھ ہدایت یافتہ ہیں، اور ان میں سے اکثر فاسق ہیں۔
ثُمَّ قَفَّيْنَا عَلَىٰٓ ءَاثَـٰرِهِم بِرُسُلِنَا وَقَفَّيْنَا بِعِيسَى ٱبْنِ مَرْيَمَ وَءَاتَيْنَـٰهُ ٱلْإِنجِيلَ ۖ وَجَعَلْنَا فِى قُلُوبِ ٱلَّذِينَ ٱتَّبَعُوهُ رَأْفَةًۭ وَرَحْمَةًۭ ۚ وَرَهْبَانِيَّةً ٱبْتَدَعُوهَا مَا كَتَبْنَـٰهَا عَلَيْهِمْ إِلَّا ٱبْتِغَآءَ رِضْوَٰنِ ٱللَّهِ فَمَا رَعَوْهَا حَقَّ رِعَايَتِهَا ۖ فَـَٔاتَيْنَا ٱلَّذِينَ ءَامَنُوا۟ مِنْهُمْ أَجْرَهُمْ ۖ وَكَثِيرٌۭ مِّنْهُمْ فَـٰسِقُونَ ﴿27﴾
پھر ہم نے ان کے بعد اپنے رسولوں کو بھیجا، اور ہم نے عیسیٰ ابن مریم کو بھیجا، اور اسے انجیل عطا کی، اور جنہوں نے اس کی پیروی کی ان کے دلوں میں ہم نے نرمی اور رحمت رکھ دی۔ اور رہبانیت انہوں نے خود ایجاد کی، ہم نے وہ ان پر فرض نہیں کی تھی، مگر اللہ کی رضا حاصل کرنے کے لیے۔ پھر انہوں نے اس کی صحیح نگہداشت نہ کی۔ تو ہم نے ان میں سے جو ایمان لائے انہیں ان کا اجر دیا، اور ان میں سے اکثر فاسق ہیں۔
يَـٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ ٱتَّقُوا۟ ٱللَّهَ وَءَامِنُوا۟ بِرَسُولِهِۦ يُؤْتِكُمْ كِفْلَيْنِ مِن رَّحْمَتِهِۦ وَيَجْعَل لَكُمْ نُورًۭا تَمْشُونَ بِهِۦ وَيَغْفِرْ لَكُمْ ۚ وَٱللَّهُ غَفُورٌۭ رَّحِيمٌۭ ﴿28﴾
اے ایمان والو! اللہ سے ڈرو اور اس کے رسول پر ایمان لاؤ، وہ تمہیں اپنی رحمت سے دوہرا حصہ دے گا، اور تمہیں ایسا نور عطا کرے گا جس کے ساتھ تم چلو گے، اور تمہیں بخش دے گا۔ اور اللہ بخشنے والا، رحم کرنے والا ہے۔
لِئَلَّا يَعْلَمَ أَهْلُ ٱلْكِتَـٰبِ أَلَّا يَقْدِرُونَ عَلَىٰ شَىْءٍۢ مِّن فَضْلِ ٱللَّهِ وَأَنَّ ٱلْفَضْلَ بِيَدِ ٱللَّهِ يُؤْتِيهِ مَن يَشَآءُ ۗ وَٱللَّهُ ذُو ٱلْفَضْلِ ٱلْعَظِيمِ ﴿29﴾
تاکہ اہلِ کتاب جان لیں کہ وہ اللہ کے فضل پر کوئی اختیار نہیں رکھتے، اور فضل اللہ کے ہاتھ میں ہے، وہ جسے چاہے عطا کرتا ہے۔ اور اللہ بڑے فضل والا ہے۔
