سورة الكهف

بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ
ٱلْحَمْدُ لِلَّهِ ٱلَّذِيٓ أَنزَلَ عَلَىٰ عَبْدِهِ ٱلْكِتَـٰبَ وَلَمْ يَجْعَل لَّهُۥ عِوَجٗا ﴿1﴾
سب تعریف اللہ کے لیے ہے جس نے اپنے بندے پر کتاب نازل کی اور اس میں کوئی ٹیڑھ نہ رکھی۔
قَيِّمٗا لِّيُنذِرَ بَأْسٗا شَدِيدٗا مِّن لَّدُنْهُۥ وَيُبَشِّرَ ٱلْمُؤْمِنِينَ ٱلَّذِينَ يَعْمَلُونَ ٱلصَّـٰلِحَـٰتِ أَنَّ لَهُمْ أَجْرًا حَسَنٗا ﴿2﴾
سیدھی، تاکہ وہ اپنے پاس سے سخت عذاب سے ڈرائے، اور ایمان والوں کو خوشخبری دے جو نیک عمل کرتے ہیں کہ ان کے لیے اچھا اجر ہے۔
مَّـٰكِثِينَ فِيهِ أَبَدٗا ﴿3﴾
جس میں وہ ہمیشہ رہیں گے۔
وَيُنذِرَ ٱلَّذِينَ قَالُوا۟ ٱتَّخَذَ ٱللَّهُ وَلَدٗا ﴿4﴾
اور ان لوگوں کو ڈرائے جنہوں نے کہا کہ اللہ نے بیٹا بنا لیا ہے۔
مَا لَهُم بِهِۦ مِنْ عِلْمٖ وَلَا لِءَابَآئِهِمۡۚ كَبُرَتْ كَلِمَةٗ تَخْرُجُ مِنْ أَفْوَٰهِهِمۡۚ إِن يَقُولُونَ إِلَّا كَذِبٗا ﴿5﴾
نہ ان کے پاس اس کا کوئی علم ہے اور نہ ان کے باپ دادا کے پاس، یہ بات ان کے منہ سے بہت بڑی نکلتی ہے، وہ صرف جھوٹ بولتے ہیں۔
فَلَعَلَّكَ بَـٰخِعٌ نَّفْسَكَ عَلَىٰٓ ءَاثَـٰرِهِمْ إِن لَّمْ يُؤْمِنُوا۟ بِهَـٰذَا ٱلْحَدِيثِ أَسَفًا ﴿6﴾
تو شاید تم ان کے پیچھے افسوس میں اپنی جان ہلاک کر لو اگر وہ اس بات پر ایمان نہ لائیں۔
إِنَّا جَعَلْنَا مَا عَلَى ٱلْأَرْضِ زِينَةٗ لَّهَا لِنَبْلُوَهُمْ أَيُّهُمۡ أَحْسَنُ عَمَلٗا ﴿7﴾
ہم نے زمین کی جو چیزیں ہیں ان کو اس کے لیے زینت بنایا تاکہ ہم انہیں آزمائیں کہ کون بہتر عمل کرتا ہے۔
وَإِنَّا لَجَـٰعِلُونَ مَا عَلَيْهَا صَعِيدٗا جُرُزًا ﴿8﴾
اور ہم یقیناً اس پر موجود چیزوں کو چٹیل میدان بنا دیں گے۔
أَمْ حَسِبْتَ أَنَّ أَصْحَـٰبَ ٱلْكَهْفِ وَٱلرَّقِيمِ كَانُوا۟ مِنْ ءَايَـٰتِنَا عَجَبًا ﴿9﴾
کیا تم نے گمان کیا کہ غار اور رقیم والے ہماری نشانیوں میں سے عجیب تھے؟
إِذْ أَوَى ٱلْفِتْيَةُ إِلَى ٱلْكَهْفِ فَقَالُوا۟ رَبَّنَآ ءَاتِنَا مِن لَّدُنكَ رَحْمَةٗ وَهَيِّئْ لَنَا مِنْ أَمْرِنَا رَشَدٗا ﴿10﴾
جب وہ جوان غار کی طرف گئے اور کہا: اے ہمارے رب! ہمیں اپنی طرف سے رحمت عطا فرما اور ہمارے معاملے میں ہمارے لیے راہِ راست مہیا کر۔
فَضَرَبْنَا عَلَىٰٓ ءَاذَانِهِمْ فِي ٱلْكَهْفِ سِنِينَ عَدَدٗا ﴿11﴾
تو ہم نے ان کے کانوں پر غار میں کئی سالوں تک پردہ ڈال دیا۔
ثُمَّ بَعَثْنَـٰهُمْ لِنَعْلَمَ أَيُّ ٱلْحِزْبَيْنِ أَحْصَىٰ لِمَا لَبِثُوٓا۟ أَمَدٗا ﴿12﴾
پھر ہم نے انہیں اٹھایا تاکہ ہم جان لیں کہ دونوں گروہوں میں سے کس نے ان کے قیام کی مدت کو زیادہ درست شمار کیا۔
نَّحْنُ نَقُصُّ عَلَيْكَ نَبَأَهُم بِٱلْحَقِّۚ إِنَّهُمْ فِتْيَةٌ ءَامَنُوا۟ بِرَبِّهِمْ وَزِدْنَـٰهُمْ هُدٗى ﴿13﴾
ہم تمہیں ان کا واقعہ حق کے ساتھ سناتے ہیں، بے شک وہ چند نوجوان تھے جو اپنے رب پر ایمان لائے، اور ہم نے انہیں ہدایت میں بڑھا دیا۔
وَرَبَطْنَا عَلَىٰ قُلُوبِهِمْ إِذْ قَامُوا۟ فَقَالُوا۟ رَبُّنَا رَبُّ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ لَن نَّدْعُوَا۟ مِن دُونِهِۦٓ إِلَـٰهٗاۖ لَّقَدْ قُلْنَآ إِذٗا شَطَطًا ﴿14﴾
اور ہم نے ان کے دلوں کو مضبوط کر دیا جب وہ اٹھے اور کہا: ہمارا رب آسمانوں اور زمین کا رب ہے، ہم اس کے سوا کسی کو معبود نہیں پکاریں گے، اگر ہم ایسا کہیں تو یقیناً حد سے بڑھنے والی بات ہوگی۔
هَـٰٓؤُلَآءِ قَوْمُنَا ٱتَّخَذُوا۟ مِن دُونِهِۦٓ ءَالِهَةٗۖ لَّوْلَا يَأْتُونَ عَلَيْهِم بِسُلْطَـٰنِۭ بَيِّنٖۖ فَمَنْ أَظْلَمُ مِمَّنِ ٱفْتَرَىٰ عَلَى ٱللَّهِ كَذِبٗا ﴿15﴾
یہ ہماری قوم ہے جنہوں نے اللہ کے سوا دوسرے معبود بنا لیے ہیں، کیوں نہیں وہ ان پر کوئی واضح دلیل لاتے؟ اور اس سے بڑھ کر ظالم کون ہے جو اللہ پر جھوٹ باندھے؟
وَإِذِ ٱعْتَزَلْتُمُوهُمْ وَمَا يَعْبُدُونَ إِلَّا ٱللَّهَ فَأْوُا۟ إِلَى ٱلْكَهْفِ يَنشُرْ لَكُمْ رَبُّكُم مِّن رَّحْمَتِهِۦ وَيُهَيِّئْ لَكُم مِّنْ أَمْرِكُم مِّرْفَقٗا ﴿16﴾
اور جب تم ان سے اور ان کے معبودوں سے جنہیں وہ اللہ کے سوا پوجتے ہیں، الگ ہو گئے، تو غار میں جا بیٹھو، تمہارا رب تم پر اپنی رحمت پھیلائے گا اور تمہارے معاملے میں آسانی مہیا کرے گا۔
وَتَرَى ٱلشَّمْسَ إِذَا طَلَعَت تَّزَٰوَرُ عَن كَهْفِهِمْ ذَاتَ ٱلْيَمِينِ وَإِذَا غَرَبَت تَّقْرِضُهُمْ ذَاتَ ٱلشِّمَالِ وَهُمْ فِي فَجْوَةٖ مِّنْهُۚ ذَٰلِكَ مِنْ ءَايَـٰتِ ٱللَّهِۗ مَن يَهْدِ ٱللَّهُ فَهُوَ ٱلْمُهْتَدِيۖ وَمَن يُضْلِلْ فَلَن تَجِدَ لَهُۥ وَلِيّٗا مُّرْشِدٗا ﴿17﴾
اور تم سورج کو دیکھو گے کہ جب وہ طلوع ہوتا ہے تو ان کے غار کو دائیں طرف سے چھوڑ جاتا ہے، اور جب غروب ہوتا ہے تو بائیں طرف سے کتراتا ہے، اور وہ اس میں ایک کشادہ جگہ میں ہیں۔ یہ اللہ کی نشانیوں میں سے ہے، جسے اللہ ہدایت دے وہی ہدایت یافتہ ہے، اور جسے وہ گمراہ کرے تو تم اس کے لیے کوئی رہنما نہ پاؤ گے۔
وَتَحْسَبُهُمْ أَيْقَاظٗا وَهُمْ رُقُودٞۚ وَنُقَلِّبُهُمْ ذَاتَ ٱلْيَمِينِ وَذَاتَ ٱلشِّمَالِۖ وَكَلْبُهُمۡ بَـٰسِطٞ ذِرَاعَيْهِ بِٱلْوَصِيدِۚ لَوِ ٱطَّلَعْتَ عَلَيْهِمْ لَوَلَّيْتَ مِنْهُمْ فِرَارٗا وَلَمُلِئْتَ مِنْهُمْ رُعْبٗا ﴿18﴾
اور تم انہیں جاگتا ہوا سمجھو گے حالانکہ وہ سوئے ہوئے ہیں، اور ہم انہیں دائیں اور بائیں کروٹ دیتے رہتے تھے، اور ان کا کتا غار کے دہانے پر اپنے دونوں بازو پھیلائے بیٹھا تھا۔ اگر تم ان پر جھانکتے تو ضرور ان سے بھاگ کھڑے ہوتے اور ان سے ڈر سے بھر جاتے۔
وَكَذَٰلِكَ بَعَثْنَـٰهُمْ لِيَتَسَآءَلُوا۟ بَيْنَهُمْۚ قَالَ قَآئِلٞ مِّنْهُمْ كَمْ لَبِثْتُمْۖ قَالُوا۟ لَبِثْنَا يَوْمًا أَوْ بَعْضَ يَوْمٖۚ قَالُوا۟ رَبُّكُمْ أَعْلَمُ بِمَا لَبِثْتُمْ فَٱبْعَثُوٓا۟ أَحَدَكُم بِوَرِقِكُمْ هَـٰذِهِۦٓ إِلَى ٱلْمَدِينَةِ فَلْيَنظُرْ أَيُّهَآ أَزْكَىٰ طَعَامٗا فَلْيَأْتِكُم بِرِزْقٖ مِّنْهُ وَلْيَتَلَطَّفْ وَلَا يُشْعِرَنَّ بِكُمْ أَحَدًا ﴿19﴾
اور اسی طرح ہم نے انہیں اٹھایا تاکہ وہ آپس میں پوچھ گچھ کریں۔ ان میں سے ایک نے کہا: تم کتنی دیر ٹھہرے رہے؟ انہوں نے کہا: ایک دن یا دن کا کچھ حصہ۔ انہوں نے کہا: تمہارا رب بہتر جانتا ہے کہ تم کتنی دیر ٹھہرے۔ اب تم اپنے میں سے کسی کو یہ چاندی دے کر شہر بھیجو، وہ دیکھے کہ کون سا کھانا زیادہ پاکیزہ ہے، پھر تمہارے لیے اس میں سے کچھ رزق لائے، اور نرمی سے چلے، اور کسی کو تمہاری خبر نہ ہونے دے۔
إِنَّهُمْ إِن يَظْهَرُوا۟ عَلَيْكُمْ يَرْجُمُوكُمْ أَوْ يُعِيدُوكُمْ فِي مِلَّتِهِمْ وَلَن تُفْلِحُوٓا۟ إِذٗا أَبَدٗا ﴿20﴾
بے شک اگر وہ تم پر قابو پا گئے تو تمہیں سنگسار کر دیں گے یا تمہیں اپنے دین میں واپس لوٹا دیں گے، اور اس وقت تم کبھی فلاح نہ پاؤ گے۔
وَكَذَٰلِكَ أَعْثَرْنَا عَلَيْهِمْ لِيَعْلَمُوٓا۟ أَنَّ وَعْدَ ٱللَّهِ حَقّٞ وَأَنَّ ٱلسَّاعَةَ لَا رَيْبَ فِيهَآ إِذْ يَتَنَـٰزَعُونَ بَيْنَهُمْ أَمْرَهُمْ فَقَالُوا۟ ٱبْنُوا۟ عَلَيْهِم بُنْيَـٰنٗاۖ رَّبُّهُمْ أَعْلَمُ بِهِمْۚ قَالَ ٱلَّذِينَ غَلَبُوا۟ عَلَىٰٓ أَمْرِهِمْ لَنَتَّخِذَنَّ عَلَيْهِم مَّسْجِدٗا ﴿21﴾
اور اسی طرح ہم نے لوگوں کو ان پر مطلع کیا تاکہ وہ جان لیں کہ اللہ کا وعدہ سچا ہے اور قیامت میں کوئی شک نہیں، جب وہ آپس میں ان کے معاملے میں جھگڑنے لگے تو کہنے لگے: ان پر ایک عمارت بنا دو، ان کا رب ہی ان کو بہتر جانتا ہے۔ جو ان کے معاملے پر غالب تھے انہوں نے کہا: ہم ان پر ضرور ایک مسجد بنائیں گے۔
سَيَقُولُونَ ثَلَـٰثَةٞ رَّابِعُهُمْ كَلْبُهُمْ وَيَقُولُونَ خَمْسَةٞ سَادِسُهُمْ كَلْبُهُمْ رَجْمَۢا بِٱلْغَيْبِۖ وَيَقُولُونَ سَبْعَةٞ وَثَامِنُهُمْ كَلْبُهُمْۚ قُل رَّبِّيٓ أَعْلَمُ بِعِدَّتِهِم مَّا يَعْلَمُهُمْ إِلَّا قَلِيلٞۖ فَلَا تُمَارِ فِيهِمْ إِل...ِرَآءٗ ظَـٰهِرٗا وَلَا تَسْتَفْتِ فِيهِم مِّنْهُمْ أَحَدٗا ﴿22﴾
عنقریب وہ کہیں گے: وہ تین تھے، چوتھا ان کا کتا تھا، اور کہیں گے: پانچ تھے، چھٹا ان کا کتا تھا — یہ سب غیب پر اندازے ہیں — اور کہیں گے: سات تھے، آٹھواں ان کا کتا تھا۔ کہہ دو: میرا رب ان کی تعداد کو بہتر جانتا ہے، انہیں بہت کم لوگ جانتے ہیں، پس ان کے بارے میں ظاہری بحث کے سوا جھگڑا نہ کرو، اور ان کے بارے میں کسی سے فتویٰ نہ لو۔
وَلَا تَقُولَنَّ لِشَيْءٍ إِنِّي فَاعِلٞ ذَٰلِكَ غَدًا ﴿23﴾
اور کسی چیز کے بارے میں ہرگز نہ کہو کہ میں یہ کل ضرور کر دوں گا۔
إِلَّآ أَن يَشَآءَ ٱللَّهُۚ وَٱذْكُرْ رَّبَّكَ إِذَا نَسِيتَ وَقُلْ عَسَىٰٓ أَن يَهْدِيَنِ رَبِّي لِأَقْرَبَ مِنْ هَـٰذَا رَشَدٗا ﴿24﴾
مگر یہ کہ اللہ چاہے، اور جب تم بھول جاؤ تو اپنے رب کو یاد کرو، اور کہو: امید ہے کہ میرا رب مجھے اس سے بھی زیادہ ہدایت کی بات دکھا دے۔
وَلَبِثُوا۟ فِي كَهْفِهِمْ ثَلَـٰثَ مِاْئَةٖ سِنِينَ وَٱزْدَادُوا۟ تِسْعٗا ﴿25﴾
اور وہ اپنے غار میں تین سو سال ٹھہرے، اور نو سال مزید۔
قُلِ ٱللَّهُ أَعْلَمُ بِمَا لَبِثُوا۟ لَهُۥ غَيْبُ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِۖ أَبْصِرْ بِهِۦ وَأَسْمِعۡۚ مَا لَهُم مِّن دُونِهِۦ مِن وَلِيّٖ وَلَا يُشْرِكُ فِي حُكْمِهِۦٓ أَحَدٗا ﴿26﴾
کہہ دو: اللہ ہی بہتر جانتا ہے کہ وہ کتنی مدت ٹھہرے، آسمانوں اور زمین کا غیب اسی کے لیے ہے، کیا ہی خوب دیکھنے والا اور سننے والا ہے! اس کے سوا ان کا کوئی مددگار نہیں، اور وہ اپنے حکم میں کسی کو شریک نہیں کرتا۔
وَٱتْلُ مَآ أُوحِيَ إِلَيْكَ مِن كِتَـٰبِ رَبِّكَۖ لَا مُبَدِّلَ لِكَلِمَـٰتِهِۦۖ وَلَن تَجِدَ مِن دُونِهِۦ مُلْتَحَدٗا ﴿27﴾
اور اپنے رب کی کتاب میں سے جو تم پر وحی کی گئی ہے اسے پڑھو، اس کی باتوں کو کوئی بدلنے والا نہیں، اور تم اس کے سوا کوئی پناہ کی جگہ نہیں پاؤ گے۔
وَٱصْبِرْ نَفْسَكَ مَعَ ٱلَّذِينَ يَدْعُونَ رَبَّهُم بِٱلْغَدَوٰةِ وَٱلْعَشِيِّ يُرِيدُونَ وَجْهَهُۥۖ وَلَا تَعْدُ عَيْنَاكَ عَنْهُمْ تُرِيدُ زِينَةَ ٱلْحَيَوٰةِ ٱلدُّنْيَاۖ وَلَا تُطِعْ مَنْ أَغْفَلْنَا قَلْبَهُۥ عَن ذِكْرِنَا وَٱتَّبَعَ هَوَىٰهُ وَكَانَ أَمْرُهُۥ فُرُطٗا ﴿28﴾
اور اپنے آپ کو ان لوگوں کے ساتھ روکے رکھو جو صبح و شام اپنے رب کو پکارتے ہیں، اس کی رضا چاہتے ہیں، اور ان سے اپنی نگاہ نہ ہٹاؤ، دنیا کی زندگی کی زینت چاہتے ہوئے، اور اس کی بات نہ مانو جس کا دل ہم نے اپنی یاد سے غافل کر دیا ہے، اور وہ اپنی خواہش کے پیچھے لگا ہوا ہے، اور اس کا معاملہ حد سے بڑھا ہوا ہے۔
وَقُلِ ٱلْحَقُّ مِن رَّبِّكُمْۖ فَمَن شَآءَ فَلْيُؤْمِن وَمَن شَآءَ فَلْيَكْفُرْۚ إِنَّآ أَعْتَدْنَا لِلظَّـٰلِمِينَ نَارٗا أَحَاطَ بِهِمْ سُرَادِقُهَاۚ وَإِن يَسْتَغِيثُوا۟ يُغَاثُوا۟ بِمَآءٖ كَٱلْمُهْلِ يَشْوِي ٱلْوُجُوهَۚ بِئْسَ ٱلشَّرَابُ وَسَآءَتْ مُرْتَفَقًا ﴿29﴾
اور کہہ دو: حق تمہارے رب کی طرف سے ہے، پس جو چاہے ایمان لائے اور جو چاہے کفر کرے، بے شک ہم نے ظالموں کے لیے آگ تیار کر رکھی ہے، جس کے گرد ان کو گھیرنے والی دیواریں ہوں گی، اور اگر وہ فریاد کریں گے تو انہیں ایسا پانی دیا جائے گا جیسے پگھلا ہوا تانبہ ہو جو چہروں کو جھلسا دے گا، کیا ہی برا مشروب ہے اور کیا ہی بری آرام گاہ ہے۔
إِنَّ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ إِنَّا لَا نُضِيعُ أَجْرَ مَنْ أَحْسَنَ عَمَلًا ﴿30﴾
بے شک جو لوگ ایمان لائے اور نیک عمل کیے، ہم ان کے عمل کا اجر ضائع نہیں کرتے۔
أُو۟لَـٰٓئِكَ لَهُمْ جَنَّـٰتُ عَدْنٖ تَجْرِي مِن تَحْتِهِمُ ٱلْأَنْهَـٰرُ يُحَلَّوْنَ فِيهَا مِنْ أَسَاوِرَ مِن ذَهَبٖ وَيَلْبَسُونَ ثِيَابٗا خُضْرٗا مِّن سُندُسٖ وَإِسْتَبْرَقٖ مُّتَّكِـِٔينَ فِيهَا عَلَى ٱلْأَرَآئِكِۚ نِعْمَ ٱلثَّوَابُ وَحَسُنَتْ مُرْتَفَقًا ﴿31﴾
یہی لوگ ہیں جن کے لیے ہمیشہ رہنے والی جنتیں ہیں، جن کے نیچے نہریں بہتی ہیں، انہیں وہاں سونے کے کنگن پہنائے جائیں گے، اور وہ وہاں باریک اور موٹے ریشم کے سبز لباس پہنیں گے، اور تختوں پر تکیہ لگائے بیٹھے ہوں گے، کیا ہی اچھا بدلہ ہے اور کیا ہی خوبصورت آرام گاہ ہے۔
وَٱضْرِبْ لَهُم مَّثَلٗا رَجُلَيْنِ جَعَلْنَا لِأَحَدِهِمَا جَنَّتَيْنِ مِنْ أَعْنَابٖ وَحَفَفْنَـٰهُمَا بِنَخْلٖ وَجَعَلْنَا بَيْنَهُمَا زَرْعٗا ﴿32﴾
اور ان کے لیے دو آدمیوں کی مثال بیان کرو، ہم نے ان میں سے ایک کو انگور کے دو باغ دیے، اور ان کے گرد کھجور کے درخت لگائے، اور ان کے درمیان کھیتی رکھی۔
كِلْتَا ٱلْجَنَّتَيْنِ ءَاتَتْ أُكُلَهَا وَلَمْ تَظْلِم مِّنْهُ شَيْـٔٗاۖ وَفَجَّرْنَا خِلَـٰلَهُمَا نَهَرٗا ﴿33﴾
دونوں باغوں نے اپنا پھل دیا اور اس میں کچھ کمی نہ کی، اور ہم نے ان کے درمیان نہر جاری کر دی۔
وَكَانَ لَهُۥ ثَمَرٞ فَقَالَ لِصَاحِبِهِۦ وَهُوَ يُحَاوِرُهُۥٓ أَنَا۠ أَكْثَرُ مِنكَ مَالٗا وَأَعَزُّ نَفَرٗا ﴿34﴾
اور اس کے پاس پھل کی پیداوار تھی، تو اس نے اپنے ساتھی سے بات کرتے ہوئے کہا: میں تجھ سے زیادہ مالدار ہوں اور جماعت میں زیادہ طاقتور ہوں۔
وَدَخَلَ جَنَّتَهُۥ وَهُوَ ظَالِمٞ لِّنَفْسِهِۦ قَالَ مَآ أَظُنُّ أَن تَبِيدَ هَـٰذِهِۦٓ أَبَدٗا ﴿35﴾
اور وہ اپنی جنت میں داخل ہوا، اپنے نفس پر ظلم کرتے ہوئے، اور کہا: میں نہیں سمجھتا کہ یہ کبھی ختم ہو جائے گی۔
وَمَآ أَظُنُّ ٱلسَّاعَةَ قَآئِمَةٗۖ وَلَئِن رُّدِدتُّ إِلَىٰ رَبِّي لَأَجِدَنَّ خَيْرٗا مِّنْهَا مُنقَلَبٗا ﴿36﴾
اور میں نہیں سمجھتا کہ قیامت قائم ہوگی، اور اگر مجھے اپنے رب کی طرف لوٹایا بھی گیا تو میں ضرور اس سے بہتر جگہ پاؤں گا۔
قَالَ لَهُۥ صَاحِبُهُۥ وَهُوَ يُحَاوِرُهُۥٓ أَكَفَرْتَ بِٱلَّذِي خَلَقَكَ مِن تُرَابٖ ثُمَّ مِن نُّطْفَةٖ ثُمَّ سَوَّىٰكَ رَجُلٗا ﴿37﴾
اس کے ساتھی نے، اس سے بات کرتے ہوئے، کہا: کیا تو اس ذات کا انکار کرتا ہے جس نے تجھے مٹی سے پیدا کیا، پھر نطفے سے، پھر تجھے مکمل انسان بنایا؟
لَّـٰكِنَّآ هُوَ ٱللَّهُ رَبِّي وَلَآ أُشْرِكُ بِرَبِّيٓ أَحَدٗا ﴿38﴾
لیکن میں تو کہتا ہوں کہ اللہ ہی میرا رب ہے، اور میں اپنے رب کے ساتھ کسی کو شریک نہیں کرتا۔
وَلَوْلَآ إِذْ دَخَلْتَ جَنَّتَكَ قُلْتَ مَا شَآءَ ٱللَّهُ لَا قُوَّةَ إِلَّا بِٱللَّهِۚ إِن تَرَنِ أَنَا۠ أَقَلَّ مِنكَ مَالٗا وَوَلَدٗا ﴿39﴾
اور جب تو اپنی جنت میں داخل ہوا تو کیوں نہ کہا: ما شاء اللہ، اللہ کے سوا کوئی طاقت نہیں، اگر تو مجھے مال اور اولاد میں کم دیکھتا ہے۔
فَعَسَىٰ رَبِّيٓ أَن يُؤْتِيَنِ خَيْرٗا مِّن جَنَّتِكَ وَيُرْسِلَ عَلَيْهَا حُسْبَانٗا مِّنَ ٱلسَّمَآءِ فَتُصْبِحَ صَعِيدٗا زَلَقًا ﴿40﴾
تو ممکن ہے کہ میرا رب مجھے تیری جنت سے بہتر دے، اور اس پر آسمان سے کوئی آفت بھیجے، تو وہ چٹیل میدان ہو جائے۔
أَوْ يُصْبِحَ مَآؤُهَا غَوْرٗا فَلَن تَسْتَطِيعَ لَهُۥ طَلَبٗا ﴿41﴾
یا اس کا پانی زمین میں جذب ہو جائے، تو تو اسے کبھی نہ نکال سکے۔
وَأُحِيطَ بِثَمَرِهِۦ فَأَصْبَحَ يُقَلِّبُ كَفَّيْهِ عَلَىٰ مَآ أَنفَقَ فِيهَا وَهِيَ خَاوِيَةٌ عَلَىٰ عُرُوشِهَا وَيَقُولُ يَـٰلَيْتَنِي لَمْ أُشْرِكْ بِرَبِّيٓ أَحَدٗا ﴿42﴾
اور اس کے پھل کو گھیر لیا گیا، تو وہ اس پر خرچ کیے ہوئے مال پر ہاتھ ملتا رہ گیا، جبکہ وہ اپنی چھتوں پر گری پڑی تھی، اور کہنے لگا: کاش میں نے اپنے رب کے ساتھ کسی کو شریک نہ بنایا ہوتا۔
وَلَمْ تَكُن لَّهُۥ فِئَةٞ يَنصُرُونَهُۥ مِن دُونِ ٱللَّهِ وَمَا كَانَ مُنتَصِرًا ﴿43﴾
اور اس کے لیے اللہ کے سوا کوئی جماعت نہ تھی جو اس کی مدد کرتی، اور وہ خود بھی مدد حاصل کرنے والا نہ تھا۔
هُنَالِكَ ٱلْوَلَـٰيَةُ لِلَّهِ ٱلْحَقِّۚ هُوَ خَيْرٞ ثَوَابٗا وَخَيْرٌ عُقْبٗا ﴿44﴾
اسی جگہ اللہ ہی کے لیے اختیار ہے، جو حق ہے، وہی بہتر اجر دینے والا ہے اور انجام کے لحاظ سے بھی بہتر ہے۔
وَٱضْرِبْ لَهُم مَّثَلَ ٱلْحَيَوٰةِ ٱلدُّنْيَا كَمَآءٍ أَنزَلْنَـٰهُ مِنَ ٱلسَّمَآءِ فَٱخْتَلَطَ بِهِۦ نَبَاتُ ٱلْأَرْضِ فَأَصْبَحَ هَشِيمٗا تَذْرُوهُ ٱلرِّيَـٰحُۗ وَكَانَ ٱللَّهُ عَلَىٰ كُلِّ شَيْءٖ مُّقْتَدِرٗا ﴿45﴾
اور ان کے لیے دنیا کی زندگی کی مثال بیان کرو، جیسے پانی جسے ہم نے آسمان سے اتارا، تو زمین کی نباتات اس سے گھل مل گئیں، پھر وہ خشک ٹکڑے ہو گئے جنہیں ہوائیں اڑا لے جاتی ہیں، اور اللہ ہر چیز پر قدرت رکھنے والا ہے۔
ٱلْمَالُ وَٱلْبَنُونَ زِينَةُ ٱلْحَيَوٰةِ ٱلدُّنْيَاۖ وَٱلْبَاقِيَـٰتُ ٱلصَّـٰلِحَـٰتُ خَيْرٌ عِندَ رَبِّكَ ثَوَابٗا وَخَيْرٌ أَمَلٗا ﴿46﴾
مال اور اولاد دنیا کی زندگی کی زینت ہیں، اور باقی رہنے والے نیک اعمال تیرے رب کے نزدیک اجر میں بہتر اور امید کے لحاظ سے بھی بہتر ہیں۔
وَيَوْمَ نُسَيِّرُ ٱلْجِبَالَ وَتَرَى ٱلْأَرْضَ بَارِزَةٗ وَحَشَرْنَـٰهُمْ فَلَمْ نُغَادِرْ مِنْهُمْ أَحَدٗا ﴿47﴾
اور جس دن ہم پہاڑوں کو چلائیں گے اور تم زمین کو کھلا ہوا دیکھو گے، اور ہم انہیں جمع کریں گے، تو ان میں سے کسی کو نہ چھوڑیں گے۔
وَعُرِضُوا۟ عَلَىٰ رَبِّكَ صَفّٗاۖ لَّقَدْ جِئْتُمُونَا كَمَا خَلَقْنَـٰكُمْ أَوَّلَ مَرَّةٖۚ بَلْ زَعَمْتُمْ أَلَّن نَّجْعَلَ لَكُم مَّوْعِدٗا ﴿48﴾
اور وہ تمہارے رب کے سامنے صف باندھ کر پیش کیے جائیں گے، (اور کہا جائے گا:) تم ہمارے پاس آ گئے جیسے ہم نے تمہیں پہلی بار پیدا کیا تھا، بلکہ تم نے گمان کیا تھا کہ ہم تمہارے لیے کوئی وعدے کا وقت مقرر نہیں کریں گے۔
وَوُضِعَ ٱلْكِتَـٰبُ فَتَرَى ٱلْمُجْرِمِينَ مُشْفِقِينَ مِمَّا فِيهِ وَيَقُولُونَ يَـٰوَيْلَتَنَا مَالِ هَـٰذَا ٱلْكِتَـٰبِ لَا يُغَادِرُ صَغِيرَةٗ وَلَا كَبِيرَةً إِلَّآ أَحْصَىٰهَاۚ وَوَجَدُوا۟ مَا عَمِلُوا۟ حَاضِرٗاۗ وَلَا يَظْلِمُ رَبُّكَ أَحَدٗا ﴿49﴾
اور کتاب رکھی جائے گی، تو تم مجرموں کو دیکھو گے کہ وہ اس میں جو کچھ ہے اس سے ڈر رہے ہوں گے، اور کہیں گے: ہائے ہماری شامت! اس کتاب کو کیا ہے کہ نہ چھوٹی چیز چھوڑتی ہے نہ بڑی، مگر اسے شمار کر رکھا ہے، اور انہوں نے جو کچھ کیا وہ اسے حاضر پائیں گے، اور تیرا رب کسی پر ظلم نہیں کرتا۔
وَإِذْ قُلْنَا لِلْمَلَـٰٓئِكَةِ ٱسْجُدُوا۟ لِـَٔادَمَ فَسَجَدُوٓا۟ إِلَّآ إِبْلِيسَ كَانَ مِنَ ٱلْجِنِّ فَفَسَقَ عَنْ أَمْرِ رَبِّهِۦٓۗ أَفَتَتَّخِذُونَهُۥ وَذُرِّيَّتَهُۥٓ أَوْلِيَآءَ مِن دُونِي وَهُمْ لَكُمْ عَدُوّٞۚ بِئْسَ لِلظَّـٰلِمِينَ بَدَلٗا ﴿50﴾
اور جب ہم نے فرشتوں سے کہا: آدم کو سجدہ کرو، تو انہوں نے سجدہ کیا، سوائے ابلیس کے، وہ جنوں میں سے تھا، تو اس نے اپنے رب کے حکم سے سرتابی کی، کیا تم اسے اور اس کی اولاد کو میرے سوا دوست بناتے ہو، حالانکہ وہ تمہارے دشمن ہیں؟ ظالموں کے لیے یہ بہت برا بدل ہے۔
مَّآ أَشْهَدتُّهُمْ خَلْقَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ وَلَا خَلْقَ أَنفُسِهِمْۚ وَمَا كُنتُ مُتَّخِذَ ٱلْمُضِلِّينَ عَضُدٗا ﴿51﴾
میں نے انہیں آسمانوں اور زمین کی پیدائش کا مشاہدہ نہیں کروایا، اور نہ ہی ان کی اپنی پیدائش کا، اور میں گمراہ کرنے والوں کو اپنا مددگار بنانے والا نہیں ہوں۔
وَيَوْمَ يَقُولُ نَادُوا۟ شُرَكَآءِيَ ٱلَّذِينَ زَعَمْتُمْ فَدَعَوْهُمْ فَلَمْ يَسْتَجِيبُوا۟ لَهُمْ وَجَعَلْنَا بَيْنَهُم مَّوْبِقٗا ﴿52﴾
اور جس دن وہ کہے گا: میرے ان شریکوں کو پکارو جنہیں تم نے (میرا شریک) سمجھا تھا، تو وہ انہیں پکاریں گے مگر وہ انہیں جواب نہ دیں گے، اور ہم ان کے درمیان ہلاکت کی جگہ بنا دیں گے۔
وَرَءَا ٱلْمُجْرِمُونَ ٱلنَّارَ فَظَنُّوٓا۟ أَنَّهُم مُّوَاقِعُوهَا وَلَمْ يَجِدُوا۟ عَنْهَا مَصْرِفٗا ﴿53﴾
اور مجرم آگ کو دیکھیں گے، تو وہ سمجھیں گے کہ وہ اس میں گرنے والے ہیں، اور وہ اس سے بچنے کی کوئی راہ نہ پائیں گے۔
وَلَقَدْ صَرَّفْنَا فِي هَـٰذَا ٱلْقُرْءَانِ لِلنَّاسِ مِن كُلِّ مَثَلٖۚ وَكَانَ ٱلْإِنسَـٰنُ أَكْثَرَ شَيْءٖ جَدَلٗا ﴿54﴾
اور ہم نے اس قرآن میں لوگوں کے لیے ہر قسم کی مثال بیان کی ہے، اور انسان سب چیزوں سے زیادہ جھگڑالو ہے۔
وَمَا مَنَعَ ٱلنَّاسَ أَن يُؤْمِنُوٓا۟ إِذْ جَآءَهُمُ ٱلْهُدَىٰ وَيَسْتَغْفِرُوا۟ رَبَّهُمْ إِلَّآ أَن تَأْتِيَهُمْ سُنَّةُ ٱلْأَوَّلِينَ أَوْ يَأْتِيَهُمُ ٱلْعَذَابُ قُبُلٗا ﴿55﴾
اور لوگوں کو ایمان لانے اور اپنے رب سے بخشش مانگنے سے کوئی چیز مانع نہ ہوئی جب ان کے پاس ہدایت آ گئی، مگر یہ کہ ان پر پہلوں کا طریقہ آ جائے یا ان پر عذاب سامنے آ جائے۔
وَمَا نُرْسِلُ ٱلْمُرْسَلِينَ إِلَّا مُبَشِّرِينَ وَمُنذِرِينَۖ وَيُجَـٰدِلُ ٱلَّذِينَ كَفَرُوا۟ بِٱلْبَـٰطِلِ لِيُدْحِضُوا۟ بِهِ ٱلْحَقَّۚ وَٱتَّخَذُوٓا۟ ءَايَـٰتِي وَمَآ أُنذِرُوا۟ هُزُوٗا ﴿56﴾
اور ہم رسولوں کو نہیں بھیجتے مگر خوشخبری دینے والے اور ڈرانے والے بنا کر، اور کافر باطل کے ساتھ جھگڑا کرتے ہیں تاکہ اس سے حق کو مٹا دیں، اور انہوں نے میری آیات اور جس چیز سے انہیں ڈرایا گیا، مذاق بنا لیا۔
وَمَنْ أَظْلَمُ مِمَّن ذُكِّرَ بِـَٔايَـٰتِ رَبِّهِۦ فَأَعْرَضَ عَنْهَا وَنَسِيَ مَا قَدَّمَتْ يَدَاهُۚ إِنَّا جَعَلْنَا عَلَىٰ قُلُوبِهِمْ أَكِنَّةً أَن يَفْقَهُوهُ وَفِيٓ ءَاذَانِهِمْ وَقْرٗاۚ وَإِن تَدْعُهُمْ إِلَى ٱلْهُدَىٰ فَلَن يَهْتَدُوٓا۟ إِذًا أَبَدٗا ﴿57﴾
اور اس سے بڑھ کر ظالم کون ہے جسے اس کے رب کی آیات یاد دلائی گئیں تو اس نے منہ موڑ لیا اور جو اس کے ہاتھوں نے آگے بھیجا اسے بھول گیا؟ بے شک ہم نے ان کے دلوں پر پردے ڈال دیے ہیں کہ وہ اسے سمجھ نہ سکیں، اور ان کے کانوں میں بوجھ ہے، اور اگر تم انہیں ہدایت کی طرف بلاؤ تو وہ کبھی ہدایت نہ پائیں گے۔
وَرَبُّكَ ٱلْغَفُورُ ذُو ٱلرَّحْمَةِۖ لَوْ يُؤَاخِذُهُم بِمَا كَسَبُوا۟ لَعَجَّلَ لَهُمُ ٱلْعَذَابَۚ بَل لَّهُم مَّوْعِدٞ لَّن يَجِدُوا۟ مِن دُونِهِۦ مَوْئِلٗا ﴿58﴾
اور تیرا رب بخشنے والا، رحمت والا ہے، اگر وہ انہیں ان کے اعمال پر پکڑتا تو ان کے لیے فوراً عذاب بھیج دیتا، بلکہ ان کے لیے ایک وقت مقرر ہے جس سے وہ بچنے کی کوئی جگہ نہ پائیں گے۔
وَتِلْكَ ٱلْقُرَىٰٓ أَهْلَكْنَـٰهُمْ لَمَّا ظَلَمُوا۟ وَجَعَلْنَا لِمَهْلِكِهِم مَّوْعِدٗا ﴿59﴾
اور یہ بستیاں ہم نے تباہ کر دیں جب انہوں نے ظلم کیا، اور ہم نے ان کی ہلاکت کے لیے ایک وقت مقرر کر دیا۔
وَإِذْ قَالَ مُوسَىٰ لِفَتَىٰهُ لَآ أَبْرَحُ حَتَّىٰٓ أَبْلُغَ مَجْمَعَ ٱلْبَحْرَيْنِ أَوْ أَمْضِيَ حُقُبٗا ﴿60﴾
اور جب موسیٰ نے اپنے خادم سے کہا: میں باز نہ آؤں گا یہاں تک کہ دو دریاؤں کے سنگم تک پہنچ جاؤں، یا میں مدتوں چلتا رہوں۔
فَلَمَّا بَلَغَا مَجْمَعَ بَيْنِهِمَا نَسِيَا حُوتَهُمَا فَٱتَّخَذَ سَبِيلَهُۥ فِي ٱلْبَحْرِ سَرَبٗا ﴿61﴾
پھر جب وہ دونوں دریاؤں کے سنگم پر پہنچے تو اپنی مچھلی بھول گئے، اور وہ مچھلی سمندر میں اپنی راہ لے گئی، سرنگ کی طرح۔
فَلَمَّا جَاوَزَا قَالَ لِفَتَىٰهُ ءَاتِنَا غَدَآءَنَاۖ لَقَدْ لَقِينَا مِن سَفَرِنَا هَـٰذَا نَصَبٗا ﴿62﴾
پھر جب وہ آگے نکل گئے تو موسیٰ نے اپنے خادم سے کہا: ہمارا کھانا لاؤ، بے شک ہم نے اس سفر میں تھکن محسوس کی ہے۔
قَالَ أَرَءَيْتَ إِذْ أَوَيْنَآ إِلَى ٱلصَّخْرَةِ فَإِنِّي نَسِيتُ ٱلْحُوتَ وَمَآ أَنسَىٰنِيهُ إِلَّا ٱلشَّيْطَـٰنُ أَنْ أَذْكُرَهُۥۚ وَٱتَّخَذَ سَبِيلَهُۥ فِي ٱلْبَحْرِ عَجَبٗا ﴿63﴾
اس نے کہا: کیا آپ نے دیکھا جب ہم چٹان کے پاس ٹھہرے تھے؟ تو میں مچھلی بھول گیا، اور مجھے اسے یاد دلانے سے شیطان نے غافل کر دیا، اور وہ سمندر میں عجیب طریقے سے اپنی راہ لے گئی۔
قَالَ ذَٰلِكَ مَا كُنَّا نَبْغِۚ فَٱرْتَدَّا عَلَىٰٓ أَثَرِهِمَا قَصَصٗا ﴿64﴾
موسیٰ نے کہا: یہی تو وہ مقام ہے جسے ہم تلاش کر رہے تھے، پس وہ دونوں اپنے قدموں کے نشانوں پر واپس لوٹے۔
فَوَجَدَا عَبْدٗا مِّنْ عِبَادِنَآ ءَاتَيْنَـٰهُ رَحْمَةٗ مِّنْ عِندِنَا وَعَلَّمْنَـٰهُ مِن لَّدُنَّا عِلْمٗا ﴿65﴾
تو انہوں نے ہمارے بندوں میں سے ایک بندے کو پایا جسے ہم نے اپنی طرف سے رحمت دی تھی اور اسے اپنے پاس سے علم سکھایا تھا۔
قَالَ لَهُۥ مُوسَىٰ هَلْ أَتَّبِعُكَ عَلَىٰٓ أَن تُعَلِّمَنِ مِمَّا عُلِّمْتَ رُشْدٗا ﴿66﴾
موسیٰ نے اس سے کہا: کیا میں آپ کے ساتھ چلوں تاکہ آپ مجھے وہ ہدایت کی بات سکھائیں جو آپ کو سکھائی گئی ہے؟
قَالَ إِنَّكَ لَن تَسْتَطِيعَ مَعِيَ صَبْرٗا ﴿67﴾
اس نے کہا: بے شک تم میرے ساتھ صبر نہیں کر سکو گے۔
وَكَيْفَ تَصْبِرُ عَلَىٰ مَا لَمْ تُحِطْ بِهِۦ خُبْرٗا ﴿68﴾
اور تم کیسے صبر کرو گے اس چیز پر جس کا تمہیں علم نہیں؟
قَالَ سَتَجِدُنِيٓ إِن شَآءَ ٱللَّهُ صَابِرٗا وَلَآ أَعْصِي لَكَ أَمْرٗا ﴿69﴾
موسیٰ نے کہا: ان شاء اللہ آپ مجھے صبر کرنے والا پائیں گے، اور میں آپ کے کسی حکم کی نافرمانی نہ کروں گا۔
قَالَ فَإِنِ ٱتَّبَعْتَنِي فَلَا تَسْـَٔلْنِي عَن شَيْءٍ حَتَّىٰٓ أُحْدِثَ لَكَ مِنْهُ ذِكْرٗا ﴿70﴾
اس نے کہا: پھر اگر تم میرے ساتھ چلو تو کسی چیز کے بارے میں سوال نہ کرنا یہاں تک کہ میں خود اس کا ذکر تم سے کروں۔
فَٱنطَلَقَا حَتَّىٰٓ إِذَا رَكِبَا فِي ٱلسَّفِينَةِ خَرَقَهَاۖ قَالَ أَخَرَقْتَهَا لِتُغْرِقَ أَهْلَهَاۖ لَقَدْ جِئْتَ شَيْـٔٗا إِمْرٗا ﴿71﴾
پس وہ دونوں روانہ ہوئے، یہاں تک کہ جب وہ کشتی میں سوار ہوئے تو اس نے اسے پھاڑ دیا، موسیٰ نے کہا: کیا آپ نے اسے اس لیے پھاڑا کہ اس کے سواروں کو غرق کر دیں؟ یقیناً آپ نے بہت ہی خطرناک کام کیا۔
قَالَ أَلَمْ أَقُلْ إِنَّكَ لَن تَسْتَطِيعَ مَعِيَ صَبْرٗا ﴿72﴾
اس نے کہا: کیا میں نے نہیں کہا تھا کہ تم میرے ساتھ صبر نہیں کر سکو گے؟
قَالَ لَا تُؤَاخِذْنِي بِمَا نَسِيتُ وَلَا تُرْهِقْنِي مِنْ أَمْرِي عُسْرٗا ﴿73﴾
موسیٰ نے کہا: مجھے اس پر نہ پکڑیں جو میں بھول گیا، اور میرے معاملے میں سختی نہ کریں۔
فَٱنطَلَقَا حَتَّىٰٓ إِذَا لَقِيَا غُلَـٰمٗا فَقَتَلَهُۥۖ قَالَ أَقَتَلْتَ نَفْسٗا زَكِيَّةَۢ بِغَيْرِ نَفْسٖۖ لَّقَدْ جِئْتَ شَيْـٔٗا نُّكْرٗا ﴿74﴾
پھر وہ دونوں روانہ ہوئے، یہاں تک کہ جب ایک لڑکے سے ملاقات ہوئی تو اس نے اسے قتل کر دیا، موسیٰ نے کہا: کیا آپ نے ایک پاکیزہ جان کو بغیر کسی جان کے بدلے قتل کر دیا؟ یقیناً آپ نے بہت ہی ناپسندیدہ کام کیا۔
قَالَ أَلَمْ أَقُل لَّكَ إِنَّكَ لَن تَسْتَطِيعَ مَعِيَ صَبْرٗا ﴿75﴾
اس نے کہا: کیا میں نے تم سے نہیں کہا تھا کہ تم میرے ساتھ صبر نہیں کر سکو گے؟
قَالَ إِن سَأَلْتُكَ عَن شَيْءِۭ بَعْدَهَا فَلَا تُصَـٰحِبْنِيۖ قَدْ بَلَغْتَ مِن لَّدُنِّي عُذْرٗا ﴿76﴾
موسیٰ نے کہا: اگر میں اس کے بعد آپ سے کسی چیز کے بارے میں سوال کروں تو آپ مجھے ساتھ نہ رکھیں، یقیناً آپ میری طرف سے عذر کو پہنچ چکے ہیں۔
فَٱنطَلَقَا حَتَّىٰٓ إِذَا أَتَيَآ أَهْلَ قَرْيَةٍ ٱسْتَطْعَمَآ أَهْلَهَا فَأَبَوْا۟ أَن يُضَيِّفُوهُمَا فَوَجَدَا فِيهَا جِدَارٗا يُرِيدُ أَن يَنقَضَّ فَأَقَامَهُۥۖ قَالَ لَوْ شِئْتَ لَتَّخَذْتَ عَلَيْهِ أَجْرٗا ﴿77﴾
پھر وہ دونوں روانہ ہوئے، یہاں تک کہ ایک بستی والوں کے پاس پہنچے، تو ان سے کھانا مانگا، مگر انہوں نے ان کی مہمان نوازی سے انکار کر دیا، پھر انہوں نے وہاں ایک دیوار دیکھی جو گرنے والی تھی، تو اس نے اسے سیدھا کر دیا، موسیٰ نے کہا: اگر آپ چاہتے تو اس پر اجرت لے سکتے تھے۔
قَالَ هَـٰذَا فِرَاقُ بَيْنِي وَبَيْنِكَۚ سَأُنَبِّئُكَ بِتَأْوِيلِ مَا لَمْ تَسْتَطِعْ عَلَيْهِ صَبْرٗا ﴿78﴾
اس نے کہا: یہ میرے اور تمہارے درمیان جدائی ہے، اب میں تمہیں ان باتوں کی حقیقت بتاؤں گا جن پر تم صبر نہ کر سکے۔
أَمَّا ٱلسَّفِينَةُ فَكَانَتْ لِمَسَـٰكِينَ يَعْمَلُونَ فِي ٱلْبَحْرِ فَأَرَدتُّ أَنْ أَعِيبَهَاۖ وَكَانَ وَرَآءَهُم مَّلِكٞ يَأْخُذُ كُلَّ سَفِينَةٍ غَصْبٗا ﴿79﴾
رہی کشتی، تو وہ چند مسکینوں کی تھی جو دریا میں کام کرتے تھے، تو میں نے چاہا کہ اسے عیب دار کر دوں، کیونکہ ان کے پیچھے ایک بادشاہ تھا جو ہر کشتی کو زبردستی چھین لیتا تھا۔
وَأَمَّا ٱلْغُلَـٰمُ فَكَانَ أَبَوَاهُ مُؤْمِنَيْنِ فَخَشِينَآ أَن يُرْهِقَهُمَا طُغْيَـٰنٗا وَكُفْرٗا ﴿80﴾
اور رہا وہ لڑکا، تو اس کے والدین مؤمن تھے، تو ہم نے اندیشہ کیا کہ وہ انہیں سرکشی اور کفر میں مبتلا کر دے گا۔
فَأَرَدْنَآ أَن يُبْدِلَهُمَا رَبُّهُمَا خَيْرٗا مِّنْهُ زَكَوٰةٗ وَأَقْرَبَ رُحْمٗا ﴿81﴾
تو ہم نے چاہا کہ ان کا رب انہیں اس کے بدلے بہتر پاکیزگی والا اور زیادہ رحم دل بچہ عطا کرے۔
وَأَمَّا ٱلْجِدَارُ فَكَانَ لِغُلَـٰمَيْنِ يَتِيمَيْنِ فِي ٱلْمَدِينَةِ وَكَانَ تَحْتَهُۥ كَنزٞ لَّهُمَا وَكَانَ أَبُوهُمَا صَـٰلِحٗاۖ فَأَرَادَ رَبُّكَ أَن يَبْلُغَآ أَشُدَّهُمَا وَيَسْتَخْرِجَا كَنزَهُمَاۚ رَحْمَةٗ مِّن رَّبِّكَۚ وَمَا فَعَلْتُهُۥ عَنْ أَمْرِيۚ ذَٰلِكَ تَأْوِيلُ مَا لَمْ تَسْطِعْ عَلَيْهِ صَبْرٗا ﴿82﴾
اور رہا وہ دیوار، تو وہ شہر میں دو یتیم لڑکوں کی تھی، اور اس کے نیچے ان کا خزانہ تھا، اور ان کا باپ نیک آدمی تھا، تو تیرے رب نے چاہا کہ وہ اپنی جوانی کو پہنچیں اور اپنا خزانہ نکالیں، یہ تیرے رب کی طرف سے رحمت تھی، اور میں نے یہ سب کچھ اپنی مرضی سے نہیں کیا، یہ ان باتوں کی حقیقت ہے جن پر تم صبر نہ کر سکے۔
وَيَسْـَٔلُونَكَ عَن ذِي ٱلْقَرْنَيْنِۖ قُلْ سَأَتْلُوا۟ عَلَيْكُم مِّنْهُ ذِكْرًا ﴿83﴾
اور وہ آپ سے ذوالقرنین کے بارے میں پوچھتے ہیں، کہہ دیجیے: میں تمہیں اس کا کچھ ذکر سناتا ہوں۔
إِنَّا مَكَّنَّا لَهُۥ فِي ٱلْأَرْضِ وَءَاتَيْنَـٰهُ مِن كُلِّ شَيْءٖ سَبَبٗا ﴿84﴾
بے شک ہم نے اسے زمین میں اقتدار دیا تھا اور اسے ہر چیز کا ایک ذریعہ عطا کیا تھا۔
فَأَتْبَعَ سَبَبًا ﴿85﴾
تو اس نے ایک راستے کا پیچھا کیا۔
حَتَّىٰٓ إِذَا بَلَغَ مَغْرِبَ ٱلشَّمْسِ وَجَدَهَا تَغْرُبُ فِي عَيْنٍ حَمِئَةٖ وَوَجَدَ عِندَهَا قَوْمٗاۚ قُلْنَا يَـٰذَا ٱلْقَرْنَيْنِ إِمَّآ أَن تُعَذِّبَ وَإِمَّآ أَن تَتَّخِذَ فِيهِمْ حُسْنٗا ﴿86﴾
یہاں تک کہ جب وہ سورج کے غروب ہونے کی جگہ پہنچا، تو اسے ایک سیاہ چشمے میں غروب ہوتا پایا، اور وہاں ایک قوم کو پایا، ہم نے کہا: اے ذوالقرنین! یا تو انہیں سزا دو یا ان کے ساتھ نرمی کا معاملہ کرو۔
قَالَ أَمَّا مَن ظَلَمَ فَسَوْفَ نُعَذِّبُهُۥ ثُمَّ يُرَدُّ إِلَىٰ رَبِّهِۦ فَيُعَذِّبُهُۥ عَذَابٗا نُّكْرٗا ﴿87﴾
اس نے کہا: جو ظلم کرے گا ہم اسے ضرور سزا دیں گے، پھر وہ اپنے رب کی طرف لوٹایا جائے گا، تو وہ اسے سخت عذاب دے گا۔
وَأَمَّا مَنْ ءَامَنَ وَعَمِلَ صَـٰلِحٗا فَلَهُۥ جَزَآءً ٱلْحُسْنَىٰۖ وَسَنَقُولُ لَهُۥ مِنْ أَمْرِنَا يُسْرٗا ﴿88﴾
اور جو ایمان لایا اور نیک عمل کیا، تو اس کے لیے بہترین جزا ہے، اور ہم اس سے اپنے معاملے میں نرمی کی بات کہیں گے۔
ثُمَّ أَتْبَعَ سَبَبٗا ﴿89﴾
پھر اس نے ایک اور راستے کا پیچھا کیا۔
حَتَّىٰٓ إِذَا بَلَغَ مَطۡلِعَ ٱلشَّمۡسِ وَجَدَهَا تَطۡلُعُ عَلَىٰ قَوۡمٖ لَّمۡ نَجۡعَل لَّهُم مِّن دُونِهَا سِتۡرٗا ﴿90﴾
یہاں تک کہ جب وہ سورج کے نکلنے کی جگہ پہنچا، تو اس نے اسے ایک ایسی قوم پر طلوع ہوتے پایا جن کے لیے ہم نے سورج سے بچاؤ کی کوئی اوٹ نہ رکھی تھی۔
كَذَٰلِكَۖ وَقَدۡ أَحَطۡنَا بِمَا لَدَيۡهِ خُبۡرٗا ﴿91﴾
ایسا ہی تھا، اور ہم نے اس کے پاس جو کچھ تھا اس کا پورا احاطہ کر رکھا تھا۔
ثُمَّ أَتْبَعَ سَبَبٗا ﴿92﴾
پھر اس نے ایک اور راستے کا پیچھا کیا۔
حَتَّىٰٓ إِذَا بَلَغَ بَيْنَ ٱلسَّدَّيْنِ وَجَدَ مِن دُونِهِمَا قَوْمٗا لَّا يَكَادُونَ يَفْقَهُونَ قَوْلٗا ﴿93﴾
یہاں تک کہ جب وہ دو پہاڑوں کے درمیان پہنچا، تو ان کے آگے ایک قوم کو پایا جو بات کو سمجھنے کے قریب بھی نہ تھی۔
قَالُوا۟ يَـٰذَا ٱلْقَرْنَيْنِ إِنَّ يَأْجُوجَ وَمَأْجُوجَ مُفْسِدُونَ فِي ٱلْأَرْضِ فَهَلْ نَجْعَلُ لَكَ خَرْجًا عَلَىٰٓ أَن تَجْعَلَ بَيْنَنَا وَبَيْنَهُمْ سَدّٗا ﴿94﴾
انہوں نے کہا: اے ذوالقرنین! بے شک یاجوج اور ماجوج زمین میں فساد کرتے ہیں، تو کیا ہم آپ کے لیے کچھ خرچ مقرر کریں تاکہ آپ ہمارے اور ان کے درمیان ایک رکاوٹ بنا دیں؟
قَالَ مَا مَكَّنِّي فِيهِ رَبِّي خَيْرٞ فَأَعِينُونِي بِقُوَّةٍ أَجْعَلْ بَيْنَكُمْ وَبَيْنَهُمْ رَدْمٗا ﴿95﴾
اس نے کہا: جو کچھ میرے رب نے مجھے عطا کیا ہے وہ بہتر ہے، پس تم مجھے قوت کے ساتھ مدد دو، میں تمہارے اور ان کے درمیان ایک مضبوط دیوار بنا دوں گا۔
ءَاتُونِي زُبَرَ ٱلْحَدِيدِۖ حَتَّىٰٓ إِذَا سَاوَىٰ بَيْنَ ٱلصَّدَفَيْنِ قَالَ ٱنفُخُوا۟ۖ حَتَّىٰٓ إِذَا جَعَلَهُۥ نَارٗا قَالَ ءَاتُونِيٓ أُفْرِغْ عَلَيْهِ قِطْرٗا ﴿96﴾
مجھے لوہے کی چادریں لاؤ، یہاں تک کہ جب اس نے دونوں پہاڑوں کے درمیان خلا کو برابر کر دیا، تو کہا: پھونکو، یہاں تک کہ جب وہ اسے آگ بنا دیا، تو کہا: مجھے پگھلا ہوا تانبا لاؤ تاکہ میں اسے اس پر ڈال دوں۔
فَمَا ٱسْطَـٰعُوا۟ أَن يَظْهَرُوهُ وَمَا ٱسْتَطَـٰعُوا۟ لَهُۥ نَقْبٗا ﴿97﴾
تو وہ اس پر چڑھنے کے قابل نہ ہو سکے اور نہ ہی اس میں سوراخ کر سکے۔
قَالَ هَـٰذَا رَحْمَةٞ مِّن رَّبِّي فَإِذَا جَآءَ وَعْدُ رَبِّي جَعَلَهُۥ دَكَّآءَۖ وَكَانَ وَعْدُ رَبِّي حَقّٗا ﴿98﴾
اس نے کہا: یہ میرے رب کی طرف سے رحمت ہے، پھر جب میرے رب کا وعدہ آئے گا تو وہ اسے ریزہ ریزہ کر دے گا، اور میرے رب کا وعدہ سچا ہے۔
وَتَرَكْنَا بَعْضَهُمْ يَوْمَئِذٖ يَمُوجُ فِي بَعْضٖ وَنُفِخَ فِي ٱلصُّورِ فَجَمَعْنَـٰهُمْ جَمْعٗا ﴿99﴾
اور ہم نے اس دن انہیں چھوڑ دیا کہ وہ ایک دوسرے میں گتھم گتھا ہوں گے، اور صور پھونکا جائے گا، تو ہم انہیں سب کو جمع کریں گے۔
وَعَرَضْنَا جَهَنَّمَ يَوْمَئِذٖ لِّلْكَـٰفِرِينَ عَرْضًا ﴿100﴾
اور ہم نے اس دن جہنم کو کافروں کے سامنے ظاہر کر دیا۔
ٱلَّذِينَ كَانَتْ أَعْيُنُهُمْ فِي غِطَآءٍ عَن ذِكْرِي وَكَانُوا۟ لَا يَسْتَطِيعُونَ سَمْعًا ﴿101﴾
جن کی آنکھیں میری یاد سے پردے میں تھیں اور وہ سننے کی طاقت نہیں رکھتے تھے۔
أَفَحَسِبَ ٱلَّذِينَ كَفَرُوٓا۟ أَن يَتَّخِذُوا۟ عِبَادِي مِن دُونِيٓ أَوْلِيَآءَۚ إِنَّآ أَعْتَدْنَا جَهَنَّمَ لِلْكَـٰفِرِينَ نُزُلٗا ﴿102﴾
کیا کافروں نے یہ سمجھا کہ وہ میرے بندوں کو میرے سوا دوست بنا لیں گے؟ بے شک ہم نے جہنم کو کافروں کے لیے مہمان خانہ بنا رکھا ہے۔
قُلْ هَلْ نُنَبِّئُكُم بِٱلْأَخْسَرِينَ أَعْمَـٰلًا ﴿103﴾
کہہ دو: کیا ہم تمہیں بتائیں کہ اعمال کے لحاظ سے سب سے زیادہ خسارے میں کون ہیں؟
ٱلَّذِينَ ضَلَّ سَعْيُهُمْ فِي ٱلْحَيَوٰةِ ٱلدُّنْيَا وَهُمْ يَحْسَبُونَ أَنَّهُمْ يُحْسِنُونَ صُنْعًا ﴿104﴾
وہ لوگ جن کی کوشش دنیا کی زندگی میں گم ہو گئی، اور وہ سمجھتے رہے کہ وہ اچھا کام کر رہے ہیں۔
أُو۟لَـٰٓئِكَ ٱلَّذِينَ كَفَرُوا۟ بِـَٔايَـٰتِ رَبِّهِمْ وَلِقَآئِهِۦ فَحَبِطَتْ أَعْمَـٰلُهُمْ فَلَا نُقِيمُ لَهُمْ يَوْمَ ٱلْقِيَـٰمَةِ وَزْنًا ﴿105﴾
یہ وہ لوگ ہیں جنہوں نے اپنے رب کی آیات اور اس سے ملاقات کا انکار کیا، تو ان کے اعمال ضائع ہو گئے، اور ہم قیامت کے دن ان کے لیے کوئی وزن قائم نہیں کریں گے۔
ذَٰلِكَ جَزَآؤُهُمْ جَهَنَّمُ بِمَا كَفَرُوا۟ وَٱتَّخَذُوٓا۟ ءَايَـٰتِي وَرُسُلِي هُزُوٗا ﴿106﴾
یہ ان کا بدلہ ہے جہنم، اس وجہ سے کہ انہوں نے کفر کیا اور میری آیات اور میرے رسولوں کا مذاق بنایا۔
إِنَّ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ كَانَتْ لَهُمْ جَنَّـٰتُ ٱلْفِرْدَوْسِ نُزُلًا ﴿107﴾
بے شک جو لوگ ایمان لائے اور نیک عمل کیے، ان کے لیے فردوس کے باغ مہمان نوازی ہوں گے۔
خَـٰلِدِينَ فِيهَا لَا يَبْغُونَ عَنْهَا حِوَلٗا ﴿108﴾
وہ اس میں ہمیشہ رہیں گے، اور اس سے نکلنے کی خواہش نہ کریں گے۔
قُل لَّوْ كَانَ ٱلْبَحْرُ مِدَادٗا لِّكَلِمَـٰتِ رَبِّي لَنَفِدَ ٱلْبَحْرُ قَبْلَ أَن تَنفَدَ كَلِمَـٰتُ رَبِّي وَلَوْ جِئْنَا بِمِثْلِهِۦ مَدَدٗا ﴿109﴾
کہہ دو: اگر سمندر میرے رب کی باتوں کے لیے سیاہی بن جائے، تو سمندر ختم ہو جائے گا قبل اس کے کہ میرے رب کی باتیں ختم ہوں، اگرچہ ہم اس جیسا اور لے آئیں مدد کے لیے۔
قُلْ إِنَّمَآ أَنَا۠ بَشَرٞ مِّثْلُكُمْ يُوحَىٰٓ إِلَيَّ أَنَّمَآ إِلَـٰهُكُمْ إِلَـٰهٞ وَٰحِدٞۖ فَمَن كَانَ يَرْجُوا۟ لِقَآءَ رَبِّهِۦ فَلْيَعْمَلْ عَمَلٗا صَـٰلِحٗا وَلَا يُشْرِكْ بِعِبَادَةِ رَبِّهِۦٓ أَحَدَۢا ﴿110﴾
کہہ دو: میں تو تم جیسا ہی ایک انسان ہوں، میری طرف وحی کی جاتی ہے کہ تمہارا معبود ایک ہی معبود ہے، پس جو شخص اپنے رب سے ملاقات کی امید رکھتا ہے، اسے چاہیے کہ نیک عمل کرے اور اپنے رب کی عبادت میں کسی کو شریک نہ کرے۔
